📊 인덱스 전략​

B-Tree 인덱스 (기본)​

from sqlalchemy import Index, String
from sqlalchemy.orm import Mapped, mapped_column

class User(Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)
    email: Mapped[str] = mapped_column(String(100))
    username: Mapped[str] = mapped_column(String(50))
    created_at: Mapped[datetime]

    __table_args__ = (
        # 단일 컬럼 인덱스
        Index('idx_email', 'email'),

        # 복합 인덱스 (순서 중요!)
        Index('idx_username_created', 'username', 'created_at'),

        # 부분 인덱스
        Index('idx_active_users', 'email',
              postgresql_where=text('is_active = true')),

        # UNIQUE 인덱스
        Index('idx_email_unique', 'email', unique=True),
    )

인덱스 선택 기준:

# ✅ 인덱스가 도움이 되는 경우
stmt = select(User).where(User.email == 'alice@example.com')  # =
stmt = select(User).where(User.created_at > date(2024, 1, 1))  # >, <
stmt = select(User).order_by(User.username)  # ORDER BY

# ❌ 인덱스가 도움이 안 되는 경우
stmt = select(User).where(User.email.like('%@gmail.com'))  # LIKE %...
stmt = select(User).where(func.lower(User.email) == 'alice')  # 함수 적용

GIN 인덱스 (Full-Text, JSONB, Array)​

from sqlalchemy import Index, Text
from sqlalchemy.dialects.postgresql import TSVECTOR, JSONB, ARRAY

class Article(Base):
    __tablename__ = 'articles'

    id: Mapped[int] = mapped_column(primary_key=True)
    content: Mapped[str] = mapped_column(Text)
    search_vector: Mapped[str] = mapped_column(TSVECTOR)
    metadata: Mapped[dict] = mapped_column(JSONB)
    tags: Mapped[list[str]] = mapped_column(ARRAY(String(50)))

    __table_args__ = (
        # GIN 인덱스 - Full-Text Search
        Index('idx_search_vector', 'search_vector', postgresql_using='gin'),

        # GIN 인덱스 - JSONB
        Index('idx_metadata', 'metadata', postgresql_using='gin'),

        # GIN 인덱스 - Array
        Index('idx_tags', 'tags', postgresql_using='gin'),
    )

사용 예:

# Full-Text Search (GIN 인덱스 활용)
stmt = select(Article).where(Article.search_vector.match('python'))

# JSONB 쿼리 (GIN 인덱스 활용)
stmt = select(Article).where(Article.metadata['category'].astext == 'tech')

# Array 쿼리 (GIN 인덱스 활용)
stmt = select(Article).where(Article.tags.contains(['python']))

복합 인덱스 순서​

class Order(Base):
    __tablename__ = 'orders'

    id: Mapped[int] = mapped_column(primary_key=True)
    customer_id: Mapped[int]
    status: Mapped[str]
    created_at: Mapped[datetime]

    __table_args__ = (
        # ✅ 좋은 예: 카디널리티 높은 것 먼저
        Index('idx_customer_status_created', 'customer_id', 'status', 'created_at'),
    )

# 활용되는 쿼리
stmt = select(Order).where(
    Order.customer_id == 123,  # 인덱스 활용
    Order.status == 'pending',  # 인덱스 활용
    Order.created_at > date(2024, 1, 1)  # 인덱스 활용
)

# ⚠️ 앞 컬럼 없이 뒤 컬럼만 사용하면 인덱스 활용 안 됨
stmt = select(Order).where(Order.created_at > date(2024, 1, 1))  # 인덱스 X

🔍 EXPLAIN ANALYZE​

쿼리 실행 계획 분석​

from sqlalchemy import create_engine, select, text

# Echo 활성화
engine = create_engine(
    'postgresql://...',
    echo=True  # 모든 SQL 출력
)

with Session(engine) as session:
    stmt = select(User).join(Post).where(Post.created_at > '2024-01-01')

    # SQL 문자열 추출
    compiled = stmt.compile(
        dialect=session.bind.dialect,
        compile_kwargs={"literal_binds": True}
    )

    # EXPLAIN ANALYZE 실행
    explain_query = f"EXPLAIN (ANALYZE, BUFFERS, VERBOSE) {compiled}"
    result = session.execute(text(explain_query))

    print("\n=== EXPLAIN ANALYZE ===")
    for row in result:
        print(row[0])

EXPLAIN 출력 읽기​

Seq Scan on users  (cost=0.00..458.00 rows=10000 width=64) (actual time=0.010..5.234 rows=10000 loops=1)
  Filter: (is_active = true)
  Rows Removed by Filter: 5000
Planning Time: 0.123 ms
Execution Time: 6.789 ms

해석:

  • Seq Scan: 전체 테이블 스캔 (느림) → 인덱스 필요
  • cost=0.00..458.00: 예상 비용
  • rows=10000: 예상 행 수
  • actual time=0.010..5.234: 실제 시간
  • rows=10000: 실제 행 수

개선 후:

Index Scan using idx_is_active on users  (cost=0.29..350.00 rows=10000 width=64) (actual time=0.015..2.123 rows=10000 loops=1)
  Index Cond: (is_active = true)
Planning Time: 0.098 ms
Execution Time: 3.456 ms

⚡ 쿼리 최적화 패턴​

1. 필요한 컬럼만 조회​

# ❌ 나쁜 예: 모든 컬럼
stmt = select(User)
users = session.execute(stmt).scalars().all()

# ✅ 좋은 예: 필요한 컬럼만
stmt = select(User.id, User.username, User.email)
users = session.execute(stmt).all()

성능 차이:

# 전체 컬럼 (1MB)
SELECT * FROM users  -- 100ms

# 일부 컬럼 (100KB)
SELECT id, username, email FROM users  -- 10ms

2. exists() vs count()​

# ❌ 느림: 전체 개수 세기
stmt = select(func.count(User.id)).where(User.email == email)
exists = session.execute(stmt).scalar() > 0

# ✅ 빠름: 존재 여부만 확인
from sqlalchemy import exists

stmt = select(exists().where(User.email == email))
exists = session.execute(stmt).scalar()

3. IN vs JOIN​

# 작은 리스트 (<100): IN
user_ids = [1, 2, 3, 4, 5]
stmt = select(Post).where(Post.user_id.in_(user_ids))

# 큰 리스트 (>100): JOIN
stmt = select(Post).join(
    select(literal_column(str(id)).label('id') for id in user_ids).alias('ids'),
    Post.user_id == column('ids.id')
)

# ✅ 가장 좋음: 임시 테이블
if len(user_ids) > 1000:
    # 임시 테이블 생성
    temp_table = Table('temp_ids', metadata, Column('id', Integer), prefixes=['TEMPORARY'])
    temp_table.create(engine)

    session.execute(temp_table.insert(), [{'id': id} for id in user_ids])
    stmt = select(Post).join(temp_table, Post.user_id == temp_table.c.id)

4. LIMIT 활용​

# ❌ 나쁜 예: 전체 조회 후 Python에서 슬라이싱
users = session.execute(select(User)).scalars().all()
first_10 = users[:10]  # 메모리 낭비

# ✅ 좋은 예: DB에서 LIMIT
stmt = select(User).limit(10)
first_10 = session.execute(stmt).scalars().all()

🚀 Bulk 작업 최적화​

Bulk Insert​

from sqlalchemy.dialects.postgresql import insert

# ❌ 느림: 개별 INSERT (1000번 쿼리)
for i in range(1000):
    user = User(username=f'user{i}', email=f'user{i}@example.com')
    session.add(user)
session.commit()

# ✅ 중간: add_all (1번 쿼리, 하지만 객체 생성)
users = [User(username=f'user{i}', email=f'user{i}@example.com') for i in range(1000)]
session.add_all(users)
session.commit()

# ✅ 빠름: bulk_insert_mappings (1번 쿼리, 객체 생성 안 함)
session.bulk_insert_mappings(
    User,
    [{'username': f'user{i}', 'email': f'user{i}@example.com'} for i in range(1000)]
)
session.commit()

# ✅ 가장 빠름: PostgreSQL COPY (벌크 로드)
import io
import csv

data = io.StringIO()
writer = csv.writer(data)
for i in range(1000):
    writer.writerow([f'user{i}', f'user{i}@example.com'])

data.seek(0)
connection = session.connection()
cursor = connection.connection.cursor()
cursor.copy_from(data, 'users', columns=['username', 'email'], sep=',')
session.commit()

Bulk Update​

# ❌ 느림: 개별 UPDATE
users = session.execute(select(User)).scalars().all()
for user in users:
    user.is_active = True
session.commit()

# ✅ 빠름: bulk_update_mappings
session.bulk_update_mappings(
    User,
    [{'id': user.id, 'is_active': True} for user in users]
)
session.commit()

# ✅ 가장 빠름: UPDATE 문
from sqlalchemy import update

stmt = update(User).values(is_active=True)
session.execute(stmt)
session.commit()

Upsert (ON CONFLICT)​

from sqlalchemy.dialects.postgresql import insert

# INSERT ... ON CONFLICT UPDATE
stmt = insert(User).values(
    username='alice',
    email='alice@example.com'
).on_conflict_do_update(
    index_elements=['username'],  # UNIQUE 제약 컬럼
    set_={'email': 'alice@example.com', 'updated_at': func.now()}
)

session.execute(stmt)
session.commit()

🔌 연결 풀 튜닝​

연결 풀 설정​

from sqlalchemy import create_engine
from sqlalchemy.pool import QueuePool

engine = create_engine(
    'postgresql://...',

    # 연결 풀 설정
    poolclass=QueuePool,
    pool_size=20,              # 기본 연결 수
    max_overflow=10,           # 추가 가능한 연결 수
    pool_timeout=30,           # 연결 대기 시간 (초)
    pool_recycle=3600,         # 연결 재사용 주기 (1시간)
    pool_pre_ping=True,        # 연결 유효성 사전 체크

    # 쿼리 최적화
    echo_pool=False,           # 연결 풀 로그
    execution_options={
        "isolation_level": "READ COMMITTED"
    }
)

연결 풀 모니터링​

def monitor_pool(engine):
    """연결 풀 상태 확인"""
    pool = engine.pool

    print(f"Pool size: {pool.size()}")
    print(f"Checked out: {pool.checkedout()}")
    print(f"Overflow: {pool.overflow()}")
    print(f"Checked in: {pool.checkedin()}")

    # 연결 사용률
    total = pool.size() + pool.overflow()
    used = pool.checkedout()
    usage = (used / total * 100) if total > 0 else 0

    print(f"Usage: {usage:.1f}%")

    if usage > 80:
        print("⚠️  연결 풀 사용률이 높습니다!")

연결 풀 크기 계산​

최적 pool_size = (코어 수 * 2) + 효과적인 디스크 수

예시:

  • 4코어 CPU + 1 SSD = pool_size=9
  • 8코어 CPU + 2 SSD = pool_size=18
import os

cpu_count = os.cpu_count()
recommended_pool_size = (cpu_count * 2) + 1

engine = create_engine(
    'postgresql://...',
    pool_size=recommended_pool_size,
    max_overflow=recommended_pool_size // 2
)

💾 캐싱 전략​

Application-Level 캐싱​

from functools import lru_cache
from typing import Optional

# Python LRU 캐시
@lru_cache(maxsize=1000)
def get_user_by_id(user_id: int) -> Optional[dict]:
    with Session(engine) as session:
        user = session.get(User, user_id)
        if user:
            return {
                'id': user.id,
                'username': user.username,
                'email': user.email
            }
        return None

# 사용
user = get_user_by_id(1)  # DB 조회
user = get_user_by_id(1)  # 캐시에서 반환

Redis 캐싱​

import redis
import json
from typing import Optional

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def get_user_cached(user_id: int) -> Optional[User]:
    """Redis 캐시 활용"""
    cache_key = f"user:{user_id}"

    # 캐시 확인
    cached = redis_client.get(cache_key)
    if cached:
        data = json.loads(cached)
        return User(**data)

    # DB 조회
    with Session(engine) as session:
        user = session.get(User, user_id)
        if user:
            # 캐시 저장 (1시간)
            redis_client.setex(
                cache_key,
                3600,
                json.dumps({
                    'id': user.id,
                    'username': user.username,
                    'email': user.email
                })
            )
        return user

# 캐시 무효화
def invalidate_user_cache(user_id: int):
    redis_client.delete(f"user:{user_id}")

Query Result 캐싱​

from dogpile.cache import make_region

# Dogpile 캐시 설정
region = make_region().configure(
    'dogpile.cache.redis',
    arguments={
        'host': 'localhost',
        'port': 6379,
        'db': 0,
        'distributed_lock': True
    },
    expiration_time=3600
)

@region.cache_on_arguments()
def get_popular_posts(limit: int = 10):
    """인기 포스트 조회 (캐싱)"""
    with Session(engine) as session:
        stmt = select(Post)\
            .where(Post.views > 1000)\
            .order_by(Post.views.desc())\
            .limit(limit)

        return session.execute(stmt).scalars().all()

# 첫 호출: DB 조회
posts = get_popular_posts(10)

# 두 번째 호출: 캐시 반환
posts = get_popular_posts(10)  # 빠름!

# 캐시 무효화
get_popular_posts.invalidate(10)

📊 성능 측정​

쿼리 시간 측정​

import time
from sqlalchemy import event

# 쿼리 시간 로깅
@event.listens_for(engine, "before_cursor_execute")
def before_cursor_execute(conn, cursor, statement, parameters, context, executemany):
    conn.info.setdefault('query_start_time', []).append(time.time())

@event.listens_for(engine, "after_cursor_execute")
def after_cursor_execute(conn, cursor, statement, parameters, context, executemany):
    total_time = time.time() - conn.info['query_start_time'].pop()

    if total_time > 0.1:  # 100ms 이상 걸린 쿼리
        print(f"⚠️  Slow Query ({total_time:.3f}s): {statement[:100]}...")

프로파일링​

from sqlalchemy import event
import cProfile
import pstats

def profile_query(func):
    """쿼리 프로파일링 데코레이터"""
    def wrapper(*args, **kwargs):
        profiler = cProfile.Profile()
        profiler.enable()

        result = func(*args, **kwargs)

        profiler.disable()
        stats = pstats.Stats(profiler)
        stats.sort_stats('cumulative')
        stats.print_stats(10)  # 상위 10개

        return result

    return wrapper

@profile_query
def expensive_query():
    with Session(engine) as session:
        stmt = select(User).join(Post).join(Comment)
        return session.execute(stmt).scalars().all()
