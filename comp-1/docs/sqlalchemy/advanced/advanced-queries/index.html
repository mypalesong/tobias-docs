🪟 Window Functions​

ROW_NUMBER() - 행 번호​

from sqlalchemy import select, func, over

with Session(engine) as session:
    # 각 부서별 급여 순위
    stmt = select(
        Employee.name,
        Employee.department,
        Employee.salary,
        func.row_number().over(
            partition_by=Employee.department,
            order_by=Employee.salary.desc()
        ).label('rank_in_dept')
    )

    results = session.execute(stmt).all()
    for name, dept, salary, rank in results:
        print(f"{dept} - {rank}위: {name} (${salary})")

생성되는 SQL:

SELECT
    employees.name,
    employees.department,
    employees.salary,
    ROW_NUMBER() OVER (
        PARTITION BY employees.department
        ORDER BY employees.salary DESC
    ) AS rank_in_dept
FROM employees

RANK() vs DENSE_RANK()​

from sqlalchemy import select, func

with Session(engine) as session:
    # RANK(): 동점이면 다음 순위 건너뜀 (1, 2, 2, 4)
    stmt = select(
        Student.name,
        Student.score,
        func.rank().over(
            order_by=Student.score.desc()
        ).label('rank')
    )

    # DENSE_RANK(): 동점이어도 다음 순위 연속 (1, 2, 2, 3)
    stmt = select(
        Student.name,
        Student.score,
        func.dense_rank().over(
            order_by=Student.score.desc()
        ).label('dense_rank')
    )

    results = session.execute(stmt).all()

LAG() / LEAD() - 이전/다음 행 참조​

from sqlalchemy import select, func

with Session(engine) as session:
    # 이전 달 매출과 비교
    stmt = select(
        Sales.month,
        Sales.revenue,
        func.lag(Sales.revenue).over(
            order_by=Sales.month
        ).label('prev_month_revenue'),
        (Sales.revenue - func.lag(Sales.revenue).over(order_by=Sales.month))
            .label('growth')
    )

    results = session.execute(stmt).all()
    for month, revenue, prev_revenue, growth in results:
        if prev_revenue:
            print(f"{month}: ${revenue} (성장: ${growth})")

생성되는 SQL:

SELECT
    sales.month,
    sales.revenue,
    LAG(sales.revenue) OVER (ORDER BY sales.month) AS prev_month_revenue,
    sales.revenue - LAG(sales.revenue) OVER (ORDER BY sales.month) AS growth
FROM sales

SUM() / AVG() OVER - 누적/이동 평균​

from sqlalchemy import select, func

with Session(engine) as session:
    # 누적 합계
    stmt = select(
        Order.date,
        Order.amount,
        func.sum(Order.amount).over(
            order_by=Order.date,
            rows=(None, 0)  # ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
        ).label('cumulative_total')
    )

    # 7일 이동 평균
    stmt = select(
        Order.date,
        Order.amount,
        func.avg(Order.amount).over(
            order_by=Order.date,
            rows=(-6, 0)  # ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
        ).label('moving_avg_7d')
    )

    results = session.execute(stmt).all()

실전 예제: Top N per Group​

from sqlalchemy import select, func
from sqlalchemy.orm import aliased

# 각 카테고리별 상위 3개 제품
with Session(engine) as session:
    # 서브쿼리로 순위 계산
    ranked = select(
        Product.id,
        Product.name,
        Product.category,
        Product.sales,
        func.row_number().over(
            partition_by=Product.category,
            order_by=Product.sales.desc()
        ).label('rank')
    ).subquery()

    # 순위 3 이하만 선택
    stmt = select(ranked).where(ranked.c.rank <= 3)

    results = session.execute(stmt).all()
    for product in results:
        print(f"{product.category} - {product.rank}위: {product.name}")

📝 CTE (Common Table Expressions)​

기본 CTE​

from sqlalchemy import select, func

with Session(engine) as session:
    # CTE 정의: 활성 사용자
    active_users = select(User).where(User.is_active == True).cte('active_users')

    # CTE 사용
    stmt = select(
        active_users.c.username,
        func.count(Post.id).label('post_count')
    ).join(Post, active_users.c.id == Post.user_id)\
     .group_by(active_users.c.username)

    results = session.execute(stmt).all()

생성되는 SQL:

WITH active_users AS (
    SELECT * FROM users WHERE is_active = true
)
SELECT active_users.username, COUNT(posts.id) AS post_count
FROM active_users
JOIN posts ON active_users.id = posts.user_id
GROUP BY active_users.username

여러 CTE 사용​

from sqlalchemy import select, func

with Session(engine) as session:
    # CTE 1: 최근 30일 주문
    recent_orders = select(Order)\
        .where(Order.created_at >= func.current_date() - 30)\
        .cte('recent_orders')

    # CTE 2: 고객별 주문 수
    customer_order_counts = select(
        recent_orders.c.customer_id,
        func.count(recent_orders.c.id).label('order_count')
    ).group_by(recent_orders.c.customer_id)\
     .cte('customer_order_counts')

    # 최종 쿼리
    stmt = select(
        Customer.name,
        customer_order_counts.c.order_count
    ).join(customer_order_counts, Customer.id == customer_order_counts.c.customer_id)\
     .where(customer_order_counts.c.order_count >= 5)

    results = session.execute(stmt).all()

Recursive CTE - 계층 구조​

from sqlalchemy import select, union_all

# 예제: 조직도 (Employee - Manager 관계)
with Session(engine) as session:
    # 기본 케이스: 최상위 관리자
    base_case = select(
        Employee.id,
        Employee.name,
        Employee.manager_id,
        literal(0).label('level')
    ).where(Employee.manager_id.is_(None))

    # 재귀 케이스
    emp_cte = base_case.cte('employee_hierarchy', recursive=True)

    recursive_case = select(
        Employee.id,
        Employee.name,
        Employee.manager_id,
        (emp_cte.c.level + 1).label('level')
    ).join(emp_cte, Employee.manager_id == emp_cte.c.id)

    # 재귀 CTE 정의
    emp_hierarchy = union_all(base_case, recursive_case).cte(recursive=True)

    # 사용
    stmt = select(emp_hierarchy).order_by(emp_hierarchy.c.level, emp_hierarchy.c.name)

    results = session.execute(stmt).all()
    for emp in results:
        indent = "  " * emp.level
        print(f"{indent}{emp.name} (Level {emp.level})")

생성되는 SQL:

WITH RECURSIVE employee_hierarchy AS (
    -- Base case
    SELECT id, name, manager_id, 0 AS level
    FROM employees
    WHERE manager_id IS NULL

    UNION ALL

    -- Recursive case
    SELECT e.id, e.name, e.manager_id, eh.level + 1
    FROM employees e
    JOIN employee_hierarchy eh ON e.manager_id = eh.id
)
SELECT * FROM employee_hierarchy
ORDER BY level, name

🔍 Full-Text Search​

기본 전문 검색​

from sqlalchemy import select, func
from sqlalchemy.dialects.postgresql import TSVECTOR

with Session(engine) as session:
    # to_tsvector로 검색
    stmt = select(Article).where(
        func.to_tsvector('english', Article.content).match('python & postgresql')
    )

    articles = session.execute(stmt).scalars().all()

생성되는 SQL:

SELECT * FROM articles
WHERE to_tsvector('english', articles.content) @@ to_tsquery('python & postgresql')

TSVECTOR 컬럼 사용 (성능 최적화)​

from sqlalchemy import String, Index
from sqlalchemy.dialects.postgresql import TSVECTOR

class Article(Base):
    __tablename__ = 'articles'

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    content: Mapped[str] = mapped_column(Text)

    # TSVECTOR 컬럼
    search_vector: Mapped[str] = mapped_column(TSVECTOR)

    __table_args__ = (
        # GIN 인덱스 (전문 검색 최적화)
        Index('idx_article_search', 'search_vector', postgresql_using='gin'),
    )

# 트리거로 자동 업데이트
create_trigger = """
CREATE TRIGGER article_search_vector_update
BEFORE INSERT OR UPDATE ON articles
FOR EACH ROW EXECUTE FUNCTION
tsvector_update_trigger(
    search_vector, 'pg_catalog.english', title, content
);
"""

# 검색
with Session(engine) as session:
    stmt = select(Article).where(
        Article.search_vector.match('python & database')
    )
    articles = session.execute(stmt).scalars().all()

검색 순위와 하이라이트​

from sqlalchemy import select, func

with Session(engine) as session:
    # 검색 순위 포함
    search_query = 'python tutorial'

    stmt = select(
        Article.title,
        Article.content,
        func.ts_rank(
            func.to_tsvector('english', Article.content),
            func.to_tsquery('english', search_query)
        ).label('rank'),
        func.ts_headline(
            'english',
            Article.content,
            func.to_tsquery('english', search_query)
        ).label('headline')
    ).where(
        func.to_tsvector('english', Article.content).match(search_query)
    ).order_by(func.desc('rank'))

    results = session.execute(stmt).all()
    for article in results:
        print(f"순위: {article.rank:.3f}")
        print(f"제목: {article.title}")
        print(f"미리보기: {article.headline}")
        print()

PostgreSQL 특화 함수​

from sqlalchemy import select, func, cast
from sqlalchemy.dialects.postgresql import JSONB, ARRAY

with Session(engine) as session:
    # JSONB 연산
    stmt = select(Product).where(
        Product.specs['cpu'].astext.like('%Intel%')
    )

    # ARRAY 함수
    stmt = select(Post).where(
        func.array_length(Post.tags, 1) > 3  # 태그가 3개 이상
    )

    # ARRAY contains
    stmt = select(Post).where(
        Post.tags.contains(['python', 'tutorial'])
    )

    # ARRAY overlap
    stmt = select(Post).where(
        Post.tags.overlap(['python', 'javascript'])
    )

FILTER - 조건부 집계​

from sqlalchemy import select, func, case

with Session(engine) as session:
    # PostgreSQL FILTER 절
    stmt = select(
        User.department,
        func.count(User.id).label('total'),
        func.count(User.id).filter(User.is_active == True).label('active'),

    results = session.execute(stmt).all()

CASE - 조건부 로직​

from sqlalchemy import select, func, case

with Session(engine) as session:
    # CASE 표현식
    age_group = case(
        (User.age < 20, 'Teen'),
        (User.age < 30, '20s'),
        (User.age < 40, '30s'),
        else_='40+'
    ).label('age_group')

    stmt = select(
        age_group,
        func.count(User.id).label('count')
    ).group_by(age_group)

    results = session.execute(stmt).all()

PERCENTILE - 백분위수​

from sqlalchemy import select, func

with Session(engine) as session:

        func.percentile_cont(0.95).within_group(Order.amount).label('p95')

    result = session.execute(stmt).first()

    print(f"90 백분위: ${result.p90}")
    print(f"95 백분위: ${result.p95}")

🎯 실전 예제​

대시보드 쿼리 최적화​

from sqlalchemy import select, func, case
from datetime import datetime, timedelta

def get_dashboard_stats(session: Session):
    """단일 쿼리로 모든 통계 조회"""
    thirty_days_ago = datetime.now() - timedelta(days=30)

    # CTE: 최근 30일 주문
    recent_orders = select(Order)\
        .where(Order.created_at >= thirty_days_ago)\
        .cte('recent_orders')

    # 모든 통계를 한 번에
    stmt = select(
        # 전체 통계
        func.count(func.distinct(recent_orders.c.id)).label('total_orders'),
        func.sum(recent_orders.c.amount).label('total_revenue'),
        func.avg(recent_orders.c.amount).label('avg_order_value'),

        # 상태별 개수
        func.count(recent_orders.c.id).filter(
            recent_orders.c.status == 'completed'
        ).label('completed_orders'),
        func.count(recent_orders.c.id).filter(
            recent_orders.c.status == 'pending'
        ).label('pending_orders'),

        # 고객 통계
        func.count(func.distinct(recent_orders.c.customer_id)).label('unique_customers'),

        # 일별 평균
        (func.sum(recent_orders.c.amount) / 30).label('daily_avg_revenue')
    )

    result = session.execute(stmt).first()

    return {
        'total_orders': result.total_orders,
        'total_revenue': float(result.total_revenue or 0),
        'avg_order_value': float(result.avg_order_value or 0),
        'completed_orders': result.completed_orders,
        'pending_orders': result.pending_orders,
        'unique_customers': result.unique_customers,
        'daily_avg_revenue': float(result.daily_avg_revenue or 0)
    }

상품 추천 시스템​

from sqlalchemy import select, func, and_

def get_recommended_products(
    session: Session,
    user_id: int,
    limit: int = 10
):
    """협업 필터링 기반 상품 추천"""

    # CTE 1: 사용자의 구매 이력
    user_purchases = select(OrderItem.product_id)\
        .join(Order)\
        .where(Order.user_id == user_id)\
        .cte('user_purchases')

    # CTE 2: 비슷한 제품을 구매한 다른 사용자들
    similar_users = select(Order.user_id)\
        .join(OrderItem)\
        .where(
            OrderItem.product_id.in_(select(user_purchases)),
            Order.user_id != user_id
        )\
        .group_by(Order.user_id)\
        .having(func.count(func.distinct(OrderItem.product_id)) >= 2)\
        .cte('similar_users')

    # 추천 상품 (비슷한 사용자가 구매했지만 현재 사용자는 안 산 것)
    stmt = select(
        Product.id,
        Product.name,
        func.count(OrderItem.id).label('popularity')
    ).join(OrderItem)\
     .join(Order)\
     .where(
        and_(
            Order.user_id.in_(select(similar_users)),
            Product.id.notin_(select(user_purchases))
        )
    ).group_by(Product.id, Product.name)\
     .order_by(func.desc('popularity'))\
     .limit(limit)

    return session.execute(stmt).all()

💡 프로 팁​

1. Window Functions vs GROUP BY​

# ❌ GROUP BY: 집계된 결과만
stmt = select(
    User.department,
    func.avg(User.salary)
).group_by(User.department)
# 결과: 부서별 평균만

# ✅ Window Functions: 개별 행 + 집계
stmt = select(
    User.name,
    User.salary,
    func.avg(User.salary).over(
        partition_by=User.department
    ).label('dept_avg')
)
# 결과: 각 직원 + 부서 평균

2. CTE vs 서브쿼리​

# ✅ CTE: 가독성 좋음, 재사용 가능
cte = select(User).where(User.is_active == True).cte()
stmt1 = select(cte).where(cte.c.age > 25)
stmt2 = select(func.count(cte.c.id))

# ⚠️ 서브쿼리: 한 번만 사용할 때
subq = select(User.id).where(User.is_active == True).scalar_subquery()
stmt = select(Post).where(Post.user_id == subq)

3. Full-Text Search 최적화​

# ❌ 느림: to_tsvector 매번 계산
stmt = select(Article).where(
    func.to_tsvector('english', Article.content).match(query)
)

# ✅ 빠름: TSVECTOR 컬럼 + GIN 인덱스
stmt = select(Article).where(Article.search_vector.match(query))
