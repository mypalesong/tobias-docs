from sqlalchemy.exc import OperationalError
import time

def execute_with_retry(func, max_retries=3):
    """Deadlock ë°œìƒ ì‹œ ì¬ì‹œë„"""
    for attempt in range(max_retries):
        try:
            with Session(engine) as session:
                with session.begin():
                    return func(session)

        except OperationalError as e:
            if 'deadlock detected' in str(e) and attempt < max_retries - 1:
                print(f"Deadlock detected, retrying ({attempt + 1}/{max_retries})...")
                time.sleep(0.1 * (2 ** attempt))  # Exponential backoff
            else:
                raise

        session1.begin()
        session2.begin()

                conn.execute(text("ROLLBACK PREPARED 'txn1'"))

                conn.execute(text("ROLLBACK PREPARED 'txn2'"))
        except:
            pass

    finally:
        session1.close()
        session2.close()
