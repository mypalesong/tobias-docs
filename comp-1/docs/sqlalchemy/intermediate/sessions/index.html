📖 Session이란?​

Session의 역할​

Session은 ORM과 데이터베이스 사이의 중개자입니다:

  1. Identity Map: 같은 ID의 객체는 하나만 유지
  2. Unit of Work: 변경사항을 추적하고 일괄 처리
  3. Transaction: 여러 작업을 하나의 트랜잭션으로 묶음
from sqlalchemy.orm import Session

with Session(engine) as session:
    # Session 사용
    user = User(username='alice')
    session.add(user)
    session.commit()
# 자동으로 session.close() 호출

🔧 Session 생성 방법​

방법 1: 직접 생성 (간단한 스크립트)​

from sqlalchemy import create_engine
from sqlalchemy.orm import Session

engine = create_engine('postgresql://...')

# Session 생성
with Session(engine) as session:
    user = User(username='alice')
    session.add(user)
    session.commit()

방법 2: sessionmaker (애플리케이션)​

from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

engine = create_engine('postgresql://...')

# Session Factory 생성
SessionLocal = sessionmaker(
    autocommit=False,  # 자동 커밋 비활성화 (권장)
    autoflush=False,   # 자동 flush 비활성화
    bind=engine
)

# Session 사용
with SessionLocal() as session:
    user = User(username='alice')
    session.add(user)
    session.commit()

방법 3: scoped_session (멀티스레드)​

from sqlalchemy.orm import sessionmaker, scoped_session

engine = create_engine('postgresql://...')

# Thread-local Session
session_factory = sessionmaker(bind=engine)
Session = scoped_session(session_factory)

# 사용
session = Session()
try:
    user = User(username='alice')
    session.add(user)
    session.commit()
finally:
    Session.remove()  # 스레드 로컬 세션 제거

🔄 Session 생명주기​

Session 상태​

with Session(engine) as session:
    # 1. Transient (임시): DB와 무관
    user = User(username='alice')

    # 2. Pending (대기): Session에 추가됨, 아직 DB에는 없음
    session.add(user)

    # 3. Persistent (영속): DB에 저장됨
    session.commit()
    print(user.id)  # ID 할당됨

    # 4. Detached (분리): Session에서 제거됨
    session.expunge(user)

    # 5. Deleted (삭제 예정): 삭제 대기
    session.delete(user)
    session.commit()  # DB에서 삭제됨

상태 확인​

from sqlalchemy import inspect

with Session(engine) as session:
    user = User(username='alice')

    # Transient
    insp = inspect(user)
    print(insp.transient)  # True
    print(insp.pending)    # False

    session.add(user)

    # Pending
    print(insp.transient)  # False
    print(insp.pending)    # True

    session.commit()

    # Persistent
    print(insp.persistent)  # True

💾 트랜잭션 관리​

기본 패턴​

with Session(engine) as session:
    try:
        # 작업 수행
        user = User(username='alice')
        session.add(user)

        post = Post(title='First Post', author=user)
        session.add(post)

        # 모두 성공하면 커밋
        session.commit()
    except Exception as e:
        # 하나라도 실패하면 롤백
        session.rollback()
        raise

begin() 사용​

with Session(engine) as session:
    with session.begin():
        # begin() 블록 내에서는 자동 커밋/롤백
        user = User(username='alice')
        session.add(user)
        # 블록 종료 시 자동 커밋, 에러 시 자동 롤백

명시적 트랜잭션​

with Session(engine) as session:
    # 트랜잭션 시작
    session.begin()

    try:
        user1 = User(username='alice')
        user2 = User(username='bob')

        session.add_all([user1, user2])

        # 수동 커밋
        session.commit()
    except Exception:
        # 수동 롤백
        session.rollback()
        raise

🔁 flush() vs commit()​

flush() - DB 반영, 트랜잭션 유지​

with Session(engine) as session:
    user = User(username='alice')
    session.add(user)

    # DB에 INSERT 실행, 트랜잭션은 아직 열려있음
    session.flush()

    print(user.id)  # ID 사용 가능 (RETURNING으로 받음)

    # 다른 작업...
    post = Post(title='First Post', user_id=user.id)
    session.add(post)

    # 트랜잭션 완료
    session.commit()

생성되는 SQL:

-- flush()
INSERT INTO users (username) VALUES ('alice') RETURNING id;

-- commit()
INSERT INTO posts (title, user_id) VALUES ('First Post', 1);
COMMIT;

commit() - 트랜잭션 완료​

with Session(engine) as session:
    user = User(username='alice')
    session.add(user)

    # 트랜잭션 커밋 (모든 변경사항 저장)
    session.commit()

    # 새 트랜잭션 시작됨
    user.username = 'alice_updated'
    session.commit()

언제 flush()를 사용할까?​

with Session(engine) as session:
    # ✅ flush() 사용: ID가 필요할 때
    parent = Parent(name='Parent')
    session.add(parent)
    session.flush()  # parent.id 생성

    child = Child(name='Child', parent_id=parent.id)
    session.add(child)
    session.commit()

    # ❌ flush() 불필요: relationship 사용
    parent = Parent(name='Parent')
    child = Child(name='Child', parent=parent)  # relationship로 연결
    session.add(parent)
    session.commit()  # SQLAlchemy가 알아서 처리

🏗️ Session 패턴​

패턴 1: Context Manager (권장)​

# ✅ 가장 안전한 방법
def create_user(username: str):
    with Session(engine) as session:
        with session.begin():
            user = User(username=username)
            session.add(user)
            # 자동 커밋/롤백, 자동 close
        return user.id

패턴 2: Try-Finally​

def create_user(username: str):
    session = Session(engine)
    try:
        user = User(username=username)
        session.add(user)
        session.commit()
        return user.id
    except Exception:
        session.rollback()
        raise
    finally:
        session.close()

패턴 3: Dependency Injection (FastAPI)​

from sqlalchemy.orm import sessionmaker

SessionLocal = sessionmaker(bind=engine)

def get_db():
    """DB 세션 의존성"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# FastAPI 엔드포인트
from fastapi import Depends

@app.get("/users/")
def read_users(db: Session = Depends(get_db)):
    users = db.execute(select(User)).scalars().all()
    return users

패턴 4: Session per Request (Flask)​

from flask import Flask, g
from sqlalchemy.orm import sessionmaker

app = Flask(__name__)
SessionLocal = sessionmaker(bind=engine)

@app.before_request
def before_request():
    """요청 시작 시 세션 생성"""
    g.db = SessionLocal()

@app.teardown_request
def teardown_request(exception=None):
    """요청 종료 시 세션 닫기"""
    db = g.pop('db', None)
    if db is not None:
        if exception:
            db.rollback()
        db.close()

@app.route('/users')
def get_users():
    users = g.db.execute(select(User)).scalars().all()
    return {'users': [u.username for u in users]}

🔍 Identity Map​

같은 ID는 같은 객체​

with Session(engine) as session:
    # 첫 번째 조회
    user1 = session.get(User, 1)

    # 두 번째 조회 (쿼리 없음!)
    user2 = session.get(User, 1)

    # 같은 객체
    print(user1 is user2)  # True
    print(id(user1) == id(user2))  # True

Identity Map 초기화​

with Session(engine) as session:
    user = session.get(User, 1)

    # Identity Map에서 제거
    session.expunge(user)

    # 다시 조회 (쿼리 발생)
    user2 = session.get(User, 1)

    print(user is user2)  # False

전체 초기화​

with Session(engine) as session:
    user = session.get(User, 1)

    # 모든 객체 제거
    session.expunge_all()

    # 다시 조회
    user2 = session.get(User, 1)
    print(user is user2)  # False

🔄 refresh() - DB에서 재조회​

데이터 새로고침​

with Session(engine) as session:
    user = session.get(User, 1)
    print(user.username)  # 'alice'

    # 다른 세션에서 수정했다고 가정
    # UPDATE users SET username='alice_new' WHERE id=1

    # 세션의 user는 여전히 'alice'
    print(user.username)  # 'alice' (캐시된 값)

    # DB에서 재조회
    session.refresh(user)
    print(user.username)  # 'alice_new' (최신 값)

⚙️ Session 옵션​

autocommit (사용 비권장)​

# ❌ autocommit=True (권장하지 않음)
SessionLocal = sessionmaker(
    autocommit=True,  # 각 작업이 자동 커밋됨
    bind=engine
)

# ✅ autocommit=False (권장)
SessionLocal = sessionmaker(
    autocommit=False,  # 명시적 commit 필요
    bind=engine
)

autoflush​

# autoflush=True (기본값)
SessionLocal = sessionmaker(
    autoflush=True,  # 쿼리 전 자동 flush
    bind=engine
)

with SessionLocal() as session:
    user = User(username='alice')
    session.add(user)

    # 쿼리 실행 전 자동 flush
    count = session.execute(select(func.count(User.id))).scalar()
    # alice가 포함된 개수 반환

# autoflush=False
SessionLocal = sessionmaker(
    autoflush=False,  # 수동 flush 필요
    bind=engine
)

with SessionLocal() as session:
    user = User(username='alice')
    session.add(user)

    # flush 안 됨, alice 제외된 개수 반환
    count = session.execute(select(func.count(User.id))).scalar()

    session.flush()  # 수동 flush
    count = session.execute(select(func.count(User.id))).scalar()

expire_on_commit​

# expire_on_commit=True (기본값)
SessionLocal = sessionmaker(
    expire_on_commit=True,  # commit 후 객체 만료
    bind=engine
)

with SessionLocal() as session:
    user = User(username='alice')
    session.add(user)
    session.commit()

    # 접근 시 SELECT 쿼리 발생 (만료됨)
    print(user.username)  # SELECT 발생

# expire_on_commit=False
SessionLocal = sessionmaker(
    expire_on_commit=False,  # commit 후에도 유지
    bind=engine
)

with SessionLocal() as session:
    user = User(username='alice')
    session.add(user)
    session.commit()

    # 쿼리 없음 (캐시 사용)
    print(user.username)  # SELECT 없음

💡 실전 패턴​

Repository 패턴​

from typing import Optional, List
from sqlalchemy.orm import Session

class UserRepository:
    def __init__(self, session: Session):
        self.session = session

    def create(self, username: str, email: str) -> User:
        user = User(username=username, email=email)
        self.session.add(user)
        self.session.flush()
        return user

    def get_by_id(self, user_id: int) -> Optional[User]:
        return self.session.get(User, user_id)

    def get_by_username(self, username: str) -> Optional[User]:
        stmt = select(User).where(User.username == username)
        return self.session.execute(stmt).scalar_one_or_none()

    def get_all(self, skip: int = 0, limit: int = 100) -> List[User]:
        stmt = select(User).offset(skip).limit(limit)
        return self.session.execute(stmt).scalars().all()

    def update(self, user: User) -> User:
        self.session.flush()
        return user

    def delete(self, user: User):
        self.session.delete(user)
        self.session.flush()

# 사용
with Session(engine) as session:
    with session.begin():
        repo = UserRepository(session)

        # 생성
        user = repo.create(username='alice', email='alice@example.com')

        # 조회
        found = repo.get_by_username('alice')

        # 수정
        found.email = 'new@example.com'
        repo.update(found)

        # 삭제
        repo.delete(found)

Unit of Work 패턴​

from contextlib import contextmanager
from sqlalchemy.orm import Session

@contextmanager
def transaction_scope(session_factory):
    """트랜잭션 컨텍스트"""
    session = session_factory()
    try:
        yield session
        session.commit()
    except Exception:
        session.rollback()
        raise
    finally:
        session.close()

# 사용
SessionLocal = sessionmaker(bind=engine)

with transaction_scope(SessionLocal) as session:
    user = User(username='alice')
    session.add(user)
    # 자동 커밋/롤백

서비스 레이어 패턴​

from sqlalchemy.orm import Session

class UserService:
    def __init__(self, session: Session):
        self.session = session
        self.repo = UserRepository(session)

    def register_user(self, username: str, email: str, password: str) -> User:
        """사용자 등록 (비즈니스 로직)"""
        # 중복 체크
        if self.repo.get_by_username(username):
            raise ValueError(f"Username '{username}' already exists")

        if self.repo.get_by_email(email):
            raise ValueError(f"Email '{email}' already exists")

        # 비밀번호 해시화 (실제로는 bcrypt 등 사용)
        password_hash = hash_password(password)

        # 사용자 생성
        user = self.repo.create(
            username=username,
            email=email,
            password_hash=password_hash
        )

        # 환영 이메일 발송
        send_welcome_email(user.email)

        return user

# 사용
with Session(engine) as session:
    with session.begin():
        service = UserService(session)
        user = service.register_user('alice', 'alice@example.com', 'secret123')

session = Session(engine)

def create_user(username: str):

# ✅ 좋은 예: 각자 세션 사용
def create_user(session: Session, username: str):
    user = User(username=username)
    session.add(user)
    session.flush()

    post = Post(title=title)

with Session(engine) as session:

스레드 안전성​

# ❌ 스레드 간 세션 공유 금지
session = Session(engine)  # 전역 세션

def worker():
    user = session.get(User, 1)  # 위험!

# ✅ 스레드마다 세션 생성
from sqlalchemy.orm import scoped_session, sessionmaker

session_factory = sessionmaker(bind=engine)
Session = scoped_session(session_factory)

def worker():
    session = Session()  # 스레드별 세션
    try:
        user = session.get(User, 1)
        session.commit()
    finally:
        Session.remove()

💡 프로 팁​

1. 항상 Context Manager 사용​

# ✅ 권장
with Session(engine) as session:
    with session.begin():
        # 작업
        pass

# ❌ 비권장
session = Session(engine)
session.add(user)
session.commit()
session.close()  # 잊어버리기 쉬움

2. Session은 짧게 유지​

# ✅ 짧은 세션
def get_user(user_id: int):
    with Session(engine) as session:
        return session.get(User, user_id)

# ❌ 긴 세션 (메모리 누수)
session = Session(engine)  # 애플리케이션 시작 시
# ... 계속 사용 ...

3. Detached 객체 주의​

# 세션 밖에서 객체 사용 시
with Session(engine) as session:
    user = session.get(User, 1)
# session 닫힘, user는 detached

# ❌ detached 객체 수정
user.username = 'new_name'  # DB에 반영 안 됨

# ✅ merge로 재연결
with Session(engine) as session:
    user = session.merge(user)
    user.username = 'new_name'
    session.commit()
