쿼리 작성

SQLAlchemy의 강력한 쿼리 API로 복잡한 데이터 조회를 Python스럽게 작성하는 방법을 배웁니다.

🎯 목표​

  • ✅ select() 문 작성
  • ✅ where() 조건 필터링
  • ✅ order_by() 정렬
  • ✅ group_by() 그룹화
  • ✅ 집계 함수 (COUNT, SUM, AVG)
  • ✅ 서브쿼리와 CTE

📖 기본 쿼리 패턴​

SQLAlchemy 2.0 스타일​

from sqlalchemy import select
from sqlalchemy.orm import Session

# 패턴
stmt = select(Model).where(조건).order_by(정렬)
results = session.execute(stmt).scalars().all()

레거시 vs 2.0:

# ❌ 레거시 (1.4, deprecated)
users = session.query(User).filter(User.age > 25).all()

# ✅ 2.0 스타일 (권장)
stmt = select(User).where(User.age > 25)
users = session.execute(stmt).scalars().all()

🔍 select() - 기본 조회​

전체 조회​

from sqlalchemy import select

with Session(engine) as session:
    # 모든 사용자
    stmt = select(User)
    users = session.execute(stmt).scalars().all()

    for user in users:
        print(user.username)

생성되는 SQL:

SELECT users.id, users.username, users.email, users.age
FROM users

특정 컬럼만 조회​

with Session(engine) as session:
    # username과 email만
    stmt = select(User.username, User.email)
    results = session.execute(stmt).all()

    for username, email in results:
        print(f"{username}: {email}")

생성되는 SQL:

SELECT users.username, users.email
FROM users

단일 결과 조회​

with Session(engine) as session:
    # scalar_one() - 정확히 1개, 없거나 여러 개면 에러
    stmt = select(User).where(User.id == 1)
    user = session.execute(stmt).scalar_one()

    # scalar_one_or_none() - 0개 또는 1개
    user = session.execute(stmt).scalar_one_or_none()

    # first() - 첫 번째, 없으면 None
    user = session.execute(stmt).first()

⚡ where() - 조건 필터링​

기본 조건​

from sqlalchemy import select

with Session(engine) as session:
    # 등호 ==
    stmt = select(User).where(User.username == 'alice')

    # 부등호
    stmt = select(User).where(User.age > 25)
    stmt = select(User).where(User.age >= 25)
    stmt = select(User).where(User.age < 30)

    # 같지 않음 !=
    stmt = select(User).where(User.username != 'admin')

    # IS NULL
    stmt = select(User).where(User.email.is_(None))

    # IS NOT NULL
    stmt = select(User).where(User.email.isnot(None))

여러 조건 (AND)​

with Session(engine) as session:
    # 방법 1: 쉼표로 구분
    stmt = select(User).where(
        User.age > 25,
        User.email.like('%@gmail.com')
    )

    # 방법 2: and_() 함수
    from sqlalchemy import and_

    stmt = select(User).where(
        and_(
            User.age > 25,
            User.email.like('%@gmail.com')
        )
    )

    # 방법 3: 체이닝
    stmt = select(User)\
        .where(User.age > 25)\
        .where(User.email.like('%@gmail.com'))

생성되는 SQL:

SELECT * FROM users
WHERE users.age > 25 AND users.email LIKE '%@gmail.com'

OR 조건​

from sqlalchemy import or_

with Session(engine) as session:
    # OR 조건
    stmt = select(User).where(
        or_(
            User.username == 'alice',
            User.username == 'bob'
        )
    )

    # AND + OR
    stmt = select(User).where(
        User.age > 25,
        or_(
            User.email.like('%@gmail.com'),
            User.email.like('%@yahoo.com')
        )
    )

생성되는 SQL:

SELECT * FROM users
WHERE users.age > 25
  AND (users.email LIKE '%@gmail.com' OR users.email LIKE '%@yahoo.com')

NOT 조건​

from sqlalchemy import not_

with Session(engine) as session:
    # NOT
    stmt = select(User).where(
        not_(User.is_active == False)
    )

    # 간단하게: ~
    stmt = select(User).where(~(User.is_active == False))

문자열 패턴 매칭​

with Session(engine) as session:
    # LIKE
    stmt = select(User).where(User.username.like('ali%'))  # ali로 시작

    # ILIKE (대소문자 무시)
    stmt = select(User).where(User.username.ilike('ALI%'))

    # NOT LIKE
    stmt = select(User).where(User.username.notlike('%admin%'))

    # 여러 패턴
    stmt = select(User).where(
        or_(
            User.username.like('alice%'),
            User.username.like('bob%')
        )
    )

IN / NOT IN​

with Session(engine) as session:
    # IN
    stmt = select(User).where(User.id.in_([1, 2, 3]))

    # NOT IN
    stmt = select(User).where(User.id.notin_([1, 2, 3]))

    # 서브쿼리와 함께
    active_user_ids = select(User.id).where(User.is_active == True)
    stmt = select(Post).where(Post.user_id.in_(active_user_ids))

BETWEEN​

with Session(engine) as session:
    # BETWEEN
    stmt = select(User).where(User.age.between(20, 30))

    # 날짜 범위
    from datetime import date
    stmt = select(Post).where(
        Post.created_at.between(
            date(2024, 1, 1),
            date(2024, 12, 31)
        )
    )

📊 order_by() - 정렬​

기본 정렬​

from sqlalchemy import select

with Session(engine) as session:
    # 오름차순 (ASC, 기본값)
    stmt = select(User).order_by(User.username)

    # 내림차순 (DESC)
    stmt = select(User).order_by(User.age.desc())

    # 여러 컬럼
    stmt = select(User).order_by(
        User.age.desc(),
        User.username  # age 같으면 username 순
    )

생성되는 SQL:

SELECT * FROM users
ORDER BY users.age DESC, users.username ASC

NULL 처리​

with Session(engine) as session:
    # NULLS FIRST
    stmt = select(User).order_by(User.age.desc().nullsfirst())

    # NULLS LAST
    stmt = select(User).order_by(User.age.desc().nullslast())

🎯 limit() / offset() - 페이징​

기본 페이징​

with Session(engine) as session:
    # 처음 10개
    stmt = select(User).limit(10)

    # 11-20번째 (페이지 2)
    stmt = select(User).limit(10).offset(10)

    # 정렬 + 페이징
    stmt = select(User)\
        .order_by(User.created_at.desc())\
        .limit(10)\
        .offset(0)

페이징 함수​

from typing import List
from sqlalchemy import select

def get_page(
    session: Session,
    page: int = 1,
    per_page: int = 10
) -> List[User]:
    """페이지별 사용자 조회"""
    stmt = select(User)\
        .order_by(User.id)\
        .limit(per_page)\
        .offset((page - 1) * per_page)

    return session.execute(stmt).scalars().all()

# 사용
with Session(engine) as session:
    page1 = get_page(session, page=1, per_page=10)
    page2 = get_page(session, page=2, per_page=10)

🧮 집계 함수​

COUNT​

from sqlalchemy import select, func

with Session(engine) as session:
    # 전체 개수
    stmt = select(func.count()).select_from(User)
    total = session.execute(stmt).scalar()

    # 조건부 개수
    stmt = select(func.count()).select_from(User)\
        .where(User.is_active == True)
    active_count = session.execute(stmt).scalar()

    # COUNT(column)
    stmt = select(func.count(User.id))
    count = session.execute(stmt).scalar()

    # COUNT(DISTINCT)
    stmt = select(func.count(func.distinct(User.email)))
    unique_emails = session.execute(stmt).scalar()

SUM, AVG, MIN, MAX​

from sqlalchemy import select, func

with Session(engine) as session:
    # SUM
    stmt = select(func.sum(Order.total))
    total_sales = session.execute(stmt).scalar()

    # AVG
    stmt = select(func.avg(Product.price))
    avg_price = session.execute(stmt).scalar()

    # MIN, MAX
    stmt = select(
        func.min(Product.price).label('min_price'),
        func.max(Product.price).label('max_price')
    )
    result = session.execute(stmt).first()
    print(f"Price range: {result.min_price} - {result.max_price}")

📈 group_by() - 그룹화​

기본 그룹화​

from sqlalchemy import select, func

with Session(engine) as session:
    # 카테고리별 제품 수
    stmt = select(
        Product.category,
        func.count(Product.id).label('count')
    ).group_by(Product.category)

    results = session.execute(stmt).all()
    for category, count in results:
        print(f"{category}: {count}개")

생성되는 SQL:

SELECT products.category, COUNT(products.id) AS count
FROM products
GROUP BY products.category

HAVING​

with Session(engine) as session:
    # 포스트가 5개 이상인 사용자
    stmt = select(
        User.username,
        func.count(Post.id).label('post_count')
    ).join(Post)\
     .group_by(User.username)\
     .having(func.count(Post.id) >= 5)

    results = session.execute(stmt).all()

생성되는 SQL:

SELECT users.username, COUNT(posts.id) AS post_count
FROM users
JOIN posts ON users.id = posts.user_id
GROUP BY users.username
HAVING COUNT(posts.id) >= 5

복잡한 집계​

with Session(engine) as session:
    # 월별 주문 통계
    stmt = select(
        func.date_trunc('month', Order.created_at).label('month'),
        func.count(Order.id).label('order_count'),
        func.sum(Order.total).label('total_sales'),
        func.avg(Order.total).label('avg_order_value')
    ).group_by(func.date_trunc('month', Order.created_at))\
     .order_by(func.date_trunc('month', Order.created_at).desc())

    results = session.execute(stmt).all()
    for month, count, total, avg_val in results:
        print(f"{month}: {count}건, 총 ${total}, 평균 ${avg_val}")

🔗 JOIN​

기본 JOIN​

from sqlalchemy import select

with Session(engine) as session:
    # INNER JOIN
    stmt = select(User, Post).join(Post, User.id == Post.user_id)

    # 또는 relationship 사용
    stmt = select(User).join(User.posts)

    # LEFT JOIN
    stmt = select(User).outerjoin(User.posts)

생성되는 SQL:

-- INNER JOIN
SELECT * FROM users
JOIN posts ON users.id = posts.user_id

-- LEFT JOIN
SELECT * FROM users
LEFT OUTER JOIN posts ON users.id = posts.user_id

여러 테이블 JOIN​

with Session(engine) as session:
    # User → Post → Comment
    stmt = select(User, Post, Comment)\
        .join(Post, User.id == Post.user_id)\
        .join(Comment, Post.id == Comment.post_id)\
        .where(User.username == 'alice')

    results = session.execute(stmt).all()

📦 서브쿼리​

스칼라 서브쿼리​

from sqlalchemy import select, func

with Session(engine) as session:
    # 각 사용자의 포스트 수를 포함
    post_count_subq = select(func.count(Post.id))\
        .where(Post.user_id == User.id)\
        .scalar_subquery()

    stmt = select(
        User.username,
        post_count_subq.label('post_count')
    )

    results = session.execute(stmt).all()
    for username, count in results:
        print(f"{username}: {count}개 포스트")

IN 서브쿼리​

with Session(engine) as session:
    # 포스트가 있는 사용자만
    users_with_posts = select(Post.user_id).distinct()

    stmt = select(User).where(User.id.in_(users_with_posts))
    users = session.execute(stmt).scalars().all()

EXISTS 서브쿼리​

from sqlalchemy import exists

with Session(engine) as session:
    # 포스트가 있는 사용자
    has_posts = exists().where(Post.user_id == User.id)

    stmt = select(User).where(has_posts)
    users = session.execute(stmt).scalars().all()

    # 포스트가 없는 사용자
    stmt = select(User).where(~has_posts)
    users = session.execute(stmt).scalars().all()

📝 CTE (Common Table Expression)​

WITH절 사용​

from sqlalchemy import select, func

with Session(engine) as session:
    # CTE 정의
    popular_posts = select(
        Post.id,
        Post.title,
        func.count(Comment.id).label('comment_count')
    ).join(Comment)\
     .group_by(Post.id, Post.title)\
     .having(func.count(Comment.id) > 10)\
     .cte('popular_posts')

    # CTE 사용
    stmt = select(
        User.username,
        popular_posts.c.title,
        popular_posts.c.comment_count
    ).join(Post, User.id == Post.user_id)\
     .join(popular_posts, Post.id == popular_posts.c.id)

    results = session.execute(stmt).all()

생성되는 SQL:

WITH popular_posts AS (
    SELECT posts.id, posts.title, COUNT(comments.id) AS comment_count
    FROM posts
    JOIN comments ON posts.id = comments.post_id
    GROUP BY posts.id, posts.title
    HAVING COUNT(comments.id) > 10
)
SELECT users.username, popular_posts.title, popular_posts.comment_count
FROM users
JOIN posts ON users.id = posts.user_id
JOIN popular_posts ON posts.id = popular_posts.id

🎨 DISTINCT​

DISTINCT 사용​

from sqlalchemy import select, distinct

with Session(engine) as session:
    # 중복 제거
    stmt = select(User.email).distinct()
    emails = session.execute(stmt).scalars().all()

    # func.distinct()
    stmt = select(func.distinct(User.email))
    emails = session.execute(stmt).scalars().all()

    # COUNT(DISTINCT)
    stmt = select(func.count(distinct(User.email)))
    unique_count = session.execute(stmt).scalar()

💡 실전 쿼리 예제​

복잡한 검색​

from sqlalchemy import select, or_, and_, func

def search_posts(
    session: Session,
    keyword: str = None,
    author: str = None,
    tags: List[str] = None,
    min_likes: int = None,
    page: int = 1,
    per_page: int = 10
):
    """다양한 조건으로 포스트 검색"""
    stmt = select(Post)

    # 키워드 검색
    if keyword:
        stmt = stmt.where(
            or_(
                Post.title.ilike(f'%{keyword}%'),
                Post.content.ilike(f'%{keyword}%')
            )
        )

    # 작성자 필터
    if author:
        stmt = stmt.join(User).where(User.username == author)

    # 태그 필터
    if tags:
        stmt = stmt.join(Post.tags).where(Tag.name.in_(tags))

    # 좋아요 수 필터
    if min_likes:
        stmt = stmt.where(Post.like_count >= min_likes)

    # 정렬 + 페이징
    stmt = stmt\
        .order_by(Post.created_at.desc())\
        .limit(per_page)\
        .offset((page - 1) * per_page)

    return session.execute(stmt).scalars().unique().all()

# 사용
with Session(engine) as session:
    posts = search_posts(
        session,
        keyword='python',
        tags=['tutorial', 'beginner'],
        min_likes=10,
        page=1
    )

대시보드 통계​

from sqlalchemy import select, func
from datetime import datetime, timedelta

def get_dashboard_stats(session: Session):
    """대시보드 통계"""
    now = datetime.now()
    last_30_days = now - timedelta(days=30)

    # 총 사용자 수
    total_users = session.execute(
        select(func.count(User.id))
    ).scalar()

    # 활성 사용자 수
    active_users = session.execute(
        select(func.count(User.id)).where(User.is_active == True)
    ).scalar()

    # 최근 30일 신규 가입
    new_users = session.execute(
        select(func.count(User.id)).where(User.created_at >= last_30_days)
    ).scalar()

    # 총 포스트 수
    total_posts = session.execute(
        select(func.count(Post.id))
    ).scalar()

    # 최근 30일 포스트
    recent_posts = session.execute(
        select(func.count(Post.id)).where(Post.created_at >= last_30_days)
    ).scalar()

    # 인기 태그 TOP 10
    popular_tags = session.execute(
        select(
            Tag.name,
            func.count(post_tags.c.post_id).label('usage_count')
        ).join(post_tags)\
         .group_by(Tag.name)\
         .order_by(func.count(post_tags.c.post_id).desc())\
         .limit(10)
    ).all()

    return {
        'total_users': total_users,
        'active_users': active_users,
        'new_users_30d': new_users,
        'total_posts': total_posts,
        'recent_posts_30d': recent_posts,
        'popular_tags': popular_tags
    }

💡 프로 팁​

1. 조건 동적 구성​

def get_users(session, filters: dict):
    """동적 필터"""
    stmt = select(User)

    # 조건이 있을 때만 추가
    if 'min_age' in filters:
        stmt = stmt.where(User.age >= filters['min_age'])

    if 'email_domain' in filters:
        stmt = stmt.where(User.email.like(f"%@{filters['email_domain']}"))

    if 'is_active' in filters:
        stmt = stmt.where(User.is_active == filters['is_active'])

    return session.execute(stmt).scalars().all()

# 사용
users = get_users(session, {
    'min_age': 25,
    'email_domain': 'gmail.com'
})
