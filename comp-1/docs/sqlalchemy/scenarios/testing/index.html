conftest.py​

import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session
from sqlalchemy.pool import StaticPool

from models.base import Base
from core.database import get_db
from main import app

# 테스트 DB (in-memory SQLite)
SQLALCHEMY_TEST_DATABASE_URL = "sqlite:///:memory:"

@pytest.fixture(scope="function")
def test_engine():
    """테스트 엔진 (함수마다 새로 생성)"""
    engine = create_engine(
        SQLALCHEMY_TEST_DATABASE_URL,
        connect_args={"check_same_thread": False},
        poolclass=StaticPool,
    )
    Base.metadata.create_all(bind=engine)
    yield engine
    Base.metadata.drop_all(bind=engine)
    engine.dispose()

@pytest.fixture(scope="function")
def test_db(test_engine):
    """테스트 DB 세션"""
    TestingSessionLocal = sessionmaker(
        autocommit=False,
        autoflush=False,
        bind=test_engine
    )

    db = TestingSessionLocal()
    try:
        yield db
    finally:
        db.close()

@pytest.fixture(scope="function")
def client(test_db):
    """FastAPI 테스트 클라이언트"""
    def override_get_db():
        try:
            yield test_db
        finally:
            pass

    app.dependency_overrides[get_db] = override_get_db

    from fastapi.testclient import TestClient
    with TestClient(app) as c:
        yield c

    app.dependency_overrides.clear()

🏭 Factory Pattern​

factories.py​

from factory import Factory, Faker, SubFactory
from factory.alchemy import SQLAlchemyModelFactory
from faker import Faker as FakerInstance

from models.user import User
from models.product import Product
from models.order import Order, OrderItem

fake = FakerInstance()

class UserFactory(SQLAlchemyModelFactory):
    class Meta:
        model = User
        sqlalchemy_session_persistence = "commit"

    email = Faker('email')
    username = Faker('user_name')
    password_hash = Faker('sha256')
    first_name = Faker('first_name')
    last_name = Faker('last_name')
    is_active = True
    is_verified = False
    is_admin = False

    @classmethod
    def _create(cls, model_class, *args, **kwargs):
        """비밀번호 해싱"""
        if 'password' in kwargs:
            from core.security import get_password_hash
            kwargs['password_hash'] = get_password_hash(kwargs.pop('password'))
        return super()._create(model_class, *args, **kwargs)

class ProductFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Product
        sqlalchemy_session_persistence = "commit"

    name = Faker('word')
    slug = Faker('slug')
    description = Faker('text')
    price = Faker('pydecimal', left_digits=4, right_digits=2, positive=True)
    cost = Faker('pydecimal', left_digits=4, right_digits=2, positive=True)
    stock = Faker('random_int', min=0, max=1000)
    sku = Faker('ean13')
    is_active = True
    is_featured = False

class OrderFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Order
        sqlalchemy_session_persistence = "commit"

    user = SubFactory(UserFactory)
    order_number = Faker('uuid4')
    status = 'pending'
    subtotal = Faker('pydecimal', left_digits=4, right_digits=2, positive=True)
    shipping_cost = Faker('pydecimal', left_digits=2, right_digits=2, positive=True)
    tax = Faker('pydecimal', left_digits=3, right_digits=2, positive=True)
    total = Faker('pydecimal', left_digits=4, right_digits=2, positive=True)
    shipping_address = {}

class OrderItemFactory(SQLAlchemyModelFactory):
    class Meta:
        model = OrderItem
        sqlalchemy_session_persistence = "commit"

    order = SubFactory(OrderFactory)
    product = SubFactory(ProductFactory)
    product_name = Faker('word')
    price = Faker('pydecimal', left_digits=3, right_digits=2, positive=True)
    quantity = Faker('random_int', min=1, max=10)
    subtotal = Faker('pydecimal', left_digits=4, right_digits=2, positive=True)

import pytest
from models.user import User
from models.product import Product
from core.security import verify_password

def test_create_user(test_db):
    """사용자 생성 테스트"""
    user = User(
        email='test@example.com',
        username='testuser',
        password_hash='hashed_password'
    )

    test_db.add(user)
    test_db.commit()

    assert user.id is not None
    assert user.email == 'test@example.com'
    assert user.is_active is True

def test_create_product(test_db):
    """제품 생성 테스트"""
    product = Product(
        name='Test Product',
        slug='test-product',
        description='Description',
        price=99.99,
        cost=50.00,
        stock=100,
        sku='TEST-001'
    )

    test_db.add(product)
    test_db.commit()

    assert product.id is not None
    assert product.price == 99.99

def test_product_stock_constraint(test_db):
    """재고 제약조건 테스트"""
    from sqlalchemy.exc import IntegrityError

    product = Product(
        name='Test Product',
        slug='test-product',
        description='Description',
        price=99.99,
        cost=50.00,
        stock=-10,  # ❌ 음수 재고
        sku='TEST-001'
    )

    test_db.add(product)

    with pytest.raises(IntegrityError):
        test_db.commit()

tests/test_crud.py​

import pytest
from crud.user import user as user_crud
from schemas.user import UserCreate, UserUpdate

def test_create_user(test_db):
    """사용자 CRUD 생성"""
    user_in = UserCreate(
        email='test@example.com',
        username='testuser',
        password='testpassword123'
    )

    user = user_crud.create(test_db, user_in)
    test_db.commit()

    assert user.email == 'test@example.com'
    assert user.username == 'testuser'
    assert user.password_hash != 'testpassword123'  # 해싱됨

def test_get_user_by_email(test_db, factories):
    """이메일로 사용자 조회"""
    from factories import UserFactory

    # Factory로 사용자 생성
    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create()
    test_db.commit()

    # 조회
    found_user = user_crud.get_by_email(test_db, user.email)
    assert found_user is not None
    assert found_user.id == user.id

def test_update_user(test_db):
    """사용자 CRUD 업데이트"""
    from factories import UserFactory

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create()
    test_db.commit()

    # 업데이트
    user_in = UserUpdate(first_name='NewName')
    updated_user = user_crud.update(test_db, user, user_in)
    test_db.commit()

    assert updated_user.first_name == 'NewName'

def test_delete_user(test_db):
    """사용자 CRUD 삭제"""
    from factories import UserFactory

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create()
    test_db.commit()

    user_id = user.id

    # 삭제
    user_crud.delete(test_db, user_id)
    test_db.commit()

    # 확인
    deleted_user = user_crud.get(test_db, user_id)
    assert deleted_user is None

import pytest
from services.inventory import InventoryService
from services.order import OrderService
from factories import ProductFactory, UserFactory, AddressFactory

def test_reserve_stock(test_db):
    """재고 예약 테스트"""
    ProductFactory._meta.sqlalchemy_session = test_db
    product = ProductFactory.create(stock=100)
    test_db.commit()

    service = InventoryService(test_db)

    # 재고 예약
    result = service.reserve_stock(product.id, 10)
    test_db.commit()

    assert result is True
    assert product.stock == 90

def test_reserve_stock_insufficient(test_db):
    """재고 부족 테스트"""
    ProductFactory._meta.sqlalchemy_session = test_db
    product = ProductFactory.create(stock=5)
    test_db.commit()

    service = InventoryService(test_db)

    # 재고 부족
    result = service.reserve_stock(product.id, 10)

    assert result is False
    assert product.stock == 5  # 변경 안 됨

def test_create_order(test_db):
    """주문 생성 테스트"""
    # Setup
    UserFactory._meta.sqlalchemy_session = test_db
    ProductFactory._meta.sqlalchemy_session = test_db
    AddressFactory._meta.sqlalchemy_session = test_db

    user = UserFactory.create()
    address = AddressFactory.create(user=user)
    product = ProductFactory.create(stock=100, price=50.00)

    test_db.commit()

    # 장바구니 아이템 생성 (간단히)
    from models.cart import CartItem
    cart_item = CartItem(
        user_id=user.id,
        product_id=product.id,
        quantity=2
    )
    test_db.add(cart_item)
    test_db.commit()

    # 주문 생성
    service = OrderService(test_db)
    order = service.create_order(user.id, [cart_item], address.id)
    test_db.commit()

    # 검증
    assert order.id is not None
    assert order.user_id == user.id
    assert order.status == 'pending'
    assert len(order.items) == 1
    assert product.stock == 98  # 재고 차감

tests/test_api_users.py​

import pytest
from fastapi import status

def test_create_user(client):
    """사용자 등록 API"""
    response = client.post(
        "/api/v1/users/",
        json={
            "email": "test@example.com",
            "username": "testuser",
            "password": "testpassword123"
        }
    )

    assert response.status_code == status.HTTP_201_CREATED
    data = response.json()
    assert data["email"] == "test@example.com"
    assert data["username"] == "testuser"
    assert "password" not in data

def test_create_user_duplicate_email(client, test_db):
    """중복 이메일 등록 테스트"""
    from factories import UserFactory

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create(email='existing@example.com')
    test_db.commit()

    response = client.post(
        "/api/v1/users/",
        json={
            "email": "existing@example.com",
            "username": "newuser",
            "password": "password123"
        }
    )

    assert response.status_code == status.HTTP_400_BAD_REQUEST
    assert "already registered" in response.json()["detail"]

def test_login(client, test_db):
    """로그인 API"""
    from factories import UserFactory
    from core.security import get_password_hash

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create(
        email='test@example.com',
        password_hash=get_password_hash('testpass123')
    )
    test_db.commit()

    response = client.post(
        "/api/v1/auth/login",
        data={
            "username": "test@example.com",
            "password": "testpass123"
        }
    )

    assert response.status_code == status.HTTP_200_OK
    data = response.json()
    assert "access_token" in data
    assert data["token_type"] == "bearer"

def test_read_user_me(client, test_db):
    """현재 사용자 정보 API"""
    # 사용자 생성 및 로그인
    from factories import UserFactory
    from core.security import get_password_hash, create_access_token

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create()
    test_db.commit()

    # JWT 토큰 생성
    access_token = create_access_token(data={"sub": str(user.id)})

    # 인증된 요청
    response = client.get(
        "/api/v1/users/me",
        headers={"Authorization": f"Bearer {access_token}"}
    )

    assert response.status_code == status.HTTP_200_OK
    data = response.json()
    assert data["id"] == str(user.id)
    assert data["email"] == user.email

💡 테스트 Best Practices​

1. AAA 패턴​

def test_example(test_db):
    # Arrange (준비)
    user = UserFactory.create()
    test_db.commit()

    # Act (실행)
    result = user_crud.get(test_db, user.id)

    # Assert (검증)
    assert result is not None
    assert result.id == user.id

2. Fixture 재사용​

@pytest.fixture
def admin_user(test_db):
    """관리자 사용자 Fixture"""
    from factories import UserFactory

    UserFactory._meta.sqlalchemy_session = test_db
    user = UserFactory.create(is_admin=True)
    test_db.commit()
    return user

def test_admin_action(admin_user, test_db):
    # admin_user 사용
    assert admin_user.is_admin is True

3. Parametrize​

@pytest.mark.parametrize("stock,quantity,expected", [
    (100, 10, True),  # 재고 충분
    (5, 10, False),   # 재고 부족
    (10, 10, True),   # 재고 정확히 일치
])
def test_reserve_stock_scenarios(test_db, stock, quantity, expected):
    from factories import ProductFactory
    from services.inventory import InventoryService

    ProductFactory._meta.sqlalchemy_session = test_db
    product = ProductFactory.create(stock=stock)
    test_db.commit()

    service = InventoryService(test_db)
    result = service.reserve_stock(product.id, quantity)

    assert result is expected
