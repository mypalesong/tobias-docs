기본 Mixin​

import uuid
from datetime import datetime
from sqlalchemy import DateTime, func
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

class Base(DeclarativeBase):
    pass

class TimestampMixin:
    """생성/수정 시간 자동 관리"""
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now(),
        onupdate=func.now()
    )

class UUIDMixin:
    """UUID Primary Key"""
    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4
    )

from sqlalchemy import String, Boolean
from sqlalchemy.orm import Mapped, mapped_column, relationship

class User(UUIDMixin, TimestampMixin, Base):
    __tablename__ = 'users'

    # 기본 정보
    email: Mapped[str] = mapped_column(String(255), unique=True, index=True)
    username: Mapped[str] = mapped_column(String(50), unique=True, index=True)
    password_hash: Mapped[str] = mapped_column(String(255))

    # 프로필
    first_name: Mapped[str | None] = mapped_column(String(50))
    last_name: Mapped[str | None] = mapped_column(String(50))
    phone: Mapped[str | None] = mapped_column(String(20))

    # 상태
    is_active: Mapped[bool] = mapped_column(Boolean, default=True)
    is_verified: Mapped[bool] = mapped_column(Boolean, default=False)
    is_admin: Mapped[bool] = mapped_column(Boolean, default=False)

    # 관계
    addresses: Mapped[list["Address"]] = relationship(
        back_populates="user",
        cascade="all, delete-orphan"
    )
    orders: Mapped[list["Order"]] = relationship(
        back_populates="user",
        cascade="all, delete-orphan"
    )
    cart_items: Mapped[list["CartItem"]] = relationship(
        back_populates="user",
        cascade="all, delete-orphan"
    )
    reviews: Mapped[list["Review"]] = relationship(
        back_populates="user",
        cascade="all, delete-orphan"
    )

    __table_args__ = (
        Index('idx_user_email_active', 'email', 'is_active'),
    )

class Category(UUIDMixin, TimestampMixin, Base):
    __tablename__ = 'categories'

    name: Mapped[str] = mapped_column(String(100), unique=True)
    slug: Mapped[str] = mapped_column(String(100), unique=True, index=True)
    description: Mapped[str | None] = mapped_column(Text)

    # 계층 구조 (Self-referential)
    parent_id: Mapped[uuid.UUID | None] = mapped_column(ForeignKey('categories.id'))

    # 관계
    parent: Mapped["Category | None"] = relationship(
        back_populates="children",
        remote_side="Category.id"
    )
    children: Mapped[list["Category"]] = relationship(
        back_populates="parent",
        cascade="all, delete-orphan"
    )
    products: Mapped[list["Product"]] = relationship(
        secondary="product_categories",
        back_populates="categories"
    )

Product 모델​

from decimal import Decimal
from sqlalchemy import Numeric, Integer, Text, CheckConstraint
from sqlalchemy.dialects.postgresql import JSONB, ARRAY

class Product(UUIDMixin, TimestampMixin, Base):
    __tablename__ = 'products'

    # 기본 정보
    name: Mapped[str] = mapped_column(String(255), index=True)
    slug: Mapped[str] = mapped_column(String(255), unique=True, index=True)
    description: Mapped[str] = mapped_column(Text)
    short_description: Mapped[str | None] = mapped_column(String(500))

    # 가격 (정확한 금액)
    price: Mapped[Decimal] = mapped_column(Numeric(10, 2))
    compare_at_price: Mapped[Decimal | None] = mapped_column(Numeric(10, 2))
    cost: Mapped[Decimal] = mapped_column(Numeric(10, 2))

    # 재고
    stock: Mapped[int] = mapped_column(Integer, default=0)
    low_stock_threshold: Mapped[int] = mapped_column(Integer, default=10)

    # SKU
    sku: Mapped[str] = mapped_column(String(100), unique=True, index=True)

    # 상태
    is_active: Mapped[bool] = mapped_column(Boolean, default=True)
    is_featured: Mapped[bool] = mapped_column(Boolean, default=False)

    # PostgreSQL 특화 타입
    tags: Mapped[list[str]] = mapped_column(ARRAY(String(50)), default=list)
    specifications: Mapped[dict] = mapped_column(JSONB, default=dict)

    # 관계
    categories: Mapped[list["Category"]] = relationship(
        secondary="product_categories",
        back_populates="products"
    )
    images: Mapped[list["ProductImage"]] = relationship(
        back_populates="product",
        cascade="all, delete-orphan"
    )
    variants: Mapped[list["ProductVariant"]] = relationship(
        back_populates="product",
        cascade="all, delete-orphan"
    )
    reviews: Mapped[list["Review"]] = relationship(
        back_populates="product",
        cascade="all, delete-orphan"
    )

    __table_args__ = (
        CheckConstraint('price > 0', name='check_price_positive'),
        CheckConstraint('cost >= 0', name='check_cost_non_negative'),
        CheckConstraint('stock >= 0', name='check_stock_non_negative'),
        Index('idx_product_active_featured', 'is_active', 'is_featured'),
        Index('idx_product_price', 'price'),
        Index('idx_product_tags', 'tags', postgresql_using='gin'),
    )

class Order(UUIDMixin, TimestampMixin, Base):
    __tablename__ = 'orders'

    user_id: Mapped[uuid.UUID] = mapped_column(ForeignKey('users.id'))

    # 주문 번호
    order_number: Mapped[str] = mapped_column(String(50), unique=True, index=True)

    # 상태
    status: Mapped[str] = mapped_column(
        String(20),
        default='pending',
        index=True
    )  # pending, paid, processing, shipped, delivered, cancelled

    # 금액
    subtotal: Mapped[Decimal] = mapped_column(Numeric(10, 2))
    shipping_cost: Mapped[Decimal] = mapped_column(Numeric(10, 2))
    tax: Mapped[Decimal] = mapped_column(Numeric(10, 2))
    total: Mapped[Decimal] = mapped_column(Numeric(10, 2))

    # 배송 주소 (스냅샷)
    shipping_address: Mapped[dict] = mapped_column(JSONB)

    # 결제 정보
    payment_method: Mapped[str | None] = mapped_column(String(50))
    payment_id: Mapped[str | None] = mapped_column(String(255))
    paid_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))

    # 배송 정보
    shipped_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))
    delivered_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))
    tracking_number: Mapped[str | None] = mapped_column(String(100))

    # 관계
    user: Mapped["User"] = relationship(back_populates="orders")
    items: Mapped[list["OrderItem"]] = relationship(
        back_populates="order",
        cascade="all, delete-orphan"
    )

    __table_args__ = (
        CheckConstraint('total >= 0', name='check_total_non_negative'),
        Index('idx_order_user_status', 'user_id', 'status'),
        Index('idx_order_created', 'created_at'),
    )

재고 관리 Service​

from sqlalchemy import select
from sqlalchemy.orm import Session

class InventoryService:
    def __init__(self, session: Session):
        self.session = session

    def reserve_stock(self, product_id: uuid.UUID, quantity: int) -> bool:
        """재고 예약 (동시성 제어)"""
        stmt = select(Product)\
            .where(Product.id == product_id)\
            .with_for_update()

        product = self.session.execute(stmt).scalar_one()

        if product.stock >= quantity:
            product.stock -= quantity
            self.session.flush()
            return True

        return False

    def release_stock(self, product_id: uuid.UUID, quantity: int):
        """재고 해제"""
        stmt = select(Product)\
            .where(Product.id == product_id)\
            .with_for_update()

        product = self.session.execute(stmt).scalar_one()
        product.stock += quantity
        self.session.flush()

    def check_low_stock(self) -> list[Product]:
        """재고 부족 제품 조회"""
        stmt = select(Product).where(
            Product.stock <= Product.low_stock_threshold,
            Product.is_active == True
        )
        return self.session.execute(stmt).scalars().all()

주문 처리 Service​

import secrets
from decimal import Decimal

class OrderService:
    def __init__(self, session: Session):
        self.session = session
        self.inventory = InventoryService(session)

    def create_order(self, user_id: uuid.UUID, cart_items: list[CartItem], address_id: uuid.UUID) -> Order:
        """주문 생성"""
        # 주문 번호 생성
        order_number = self._generate_order_number()

        # 배송 주소 조회
        address = self.session.get(Address, address_id)
        if not address or address.user_id != user_id:
            raise ValueError("Invalid address")

        # 주문 생성
        order = Order(
            user_id=user_id,
            order_number=order_number,
            status='pending',
            shipping_address={
                'full_name': address.full_name,
                'phone': address.phone,
                'address_line1': address.address_line1,
                'address_line2': address.address_line2,
                'city': address.city,
                'state': address.state,
                'postal_code': address.postal_code,
                'country': address.country,
            }
        )

        subtotal = Decimal('0')

        # 주문 아이템 생성
        for cart_item in cart_items:
            # 재고 확인 및 예약
            product = cart_item.variant or cart_item.product

            if not self.inventory.reserve_stock(product.id, cart_item.quantity):
                raise ValueError(f"Insufficient stock for {product.name}")

            # 주문 아이템 생성
            order_item = OrderItem(
                order=order,
                product_id=cart_item.product_id,
                variant_id=cart_item.variant_id,
                product_name=cart_item.product.name,
                variant_name=cart_item.variant.name if cart_item.variant else None,
                price=product.price,
                quantity=cart_item.quantity,
                subtotal=product.price * cart_item.quantity
            )

            self.session.add(order_item)
            subtotal += order_item.subtotal

        # 금액 계산
        shipping_cost = self._calculate_shipping(subtotal)
        tax = self._calculate_tax(subtotal)
        total = subtotal + shipping_cost + tax

        order.subtotal = subtotal
        order.shipping_cost = shipping_cost
        order.tax = tax
        order.total = total

        self.session.add(order)
        self.session.flush()

        # 장바구니 비우기
        for cart_item in cart_items:
            self.session.delete(cart_item)

        return order

    def cancel_order(self, order_id: uuid.UUID) -> Order:
        """주문 취소"""
        order = self.session.get(Order, order_id)

        if order.status not in ['pending', 'paid']:
            raise ValueError("Cannot cancel order in current status")

        # 재고 복원
        for item in order.items:
            product = item.variant or item.product
            self.inventory.release_stock(product.id, item.quantity)

        order.status = 'cancelled'
        self.session.flush()

        return order

    def _generate_order_number(self) -> str:
        """주문 번호 생성"""
        timestamp = datetime.now().strftime('%Y%m%d')
        random_suffix = secrets.token_hex(4).upper()
        return f"ORD-{timestamp}-{random_suffix}"

    def _calculate_shipping(self, subtotal: Decimal) -> Decimal:
        """배송비 계산"""
        if subtotal >= Decimal('50'):
            return Decimal('0')  # 무료 배송
        return Decimal('5.99')

    def _calculate_tax(self, subtotal: Decimal) -> Decimal:
        """세금 계산 (10%)"""
        return (subtotal * Decimal('0.1')).quantize(Decimal('0.01'))

인기 상품​

def get_popular_products(session: Session, limit: int = 10):
    """판매량 기준 인기 상품"""
    stmt = select(
        Product.id,
        Product.name,
        func.sum(OrderItem.quantity).label('total_sold'),
        func.count(func.distinct(Order.id)).label('order_count')
    ).join(OrderItem)\
     .join(Order)\
     .where(Order.status.in_(['paid', 'processing', 'shipped', 'delivered']))\
     .group_by(Product.id, Product.name)\
     .order_by(func.desc('total_sold'))\
     .limit(limit)

    return session.execute(stmt).all()

사용자별 주문 통계​

def get_user_order_stats(session: Session, user_id: uuid.UUID):
    """사용자 주문 통계"""
    stmt = select(
        func.count(Order.id).label('total_orders'),
        func.sum(Order.total).label('total_spent'),
        func.avg(Order.total).label('avg_order_value'),
        func.max(Order.created_at).label('last_order_date')
    ).where(
        Order.user_id == user_id,
        Order.status != 'cancelled'
    )

    return session.execute(stmt).first()
