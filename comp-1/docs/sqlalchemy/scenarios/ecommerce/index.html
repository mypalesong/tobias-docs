from sqlalchemy import select
from sqlalchemy.orm import Session

class InventoryService:
    def __init__(self, session: Session):
        self.session = session

    def reserve_stock(self, product_id: uuid.UUID, quantity: int) -> bool:
        """ì¬ê³  ì˜ˆì•½ (ë™ì‹œì„± ì œì–´)"""
        stmt = select(Product)\
            .where(Product.id == product_id)\
            .with_for_update()

        product = self.session.execute(stmt).scalar_one()

        if product.stock >= quantity:
            product.stock -= quantity
            self.session.flush()
            return True

        return False

    def release_stock(self, product_id: uuid.UUID, quantity: int):
        """ì¬ê³  í•´ì œ"""
        stmt = select(Product)\
            .where(Product.id == product_id)\
            .with_for_update()

        product = self.session.execute(stmt).scalar_one()
        product.stock += quantity
        self.session.flush()

    def check_low_stock(self) -> list[Product]:
        """ì¬ê³  ë¶€ì¡± ì œí’ˆ ì¡°íšŒ"""
        stmt = select(Product).where(
            Product.stock <= Product.low_stock_threshold,
            Product.is_active == True
        )
        return self.session.execute(stmt).scalars().all()

import secrets
from decimal import Decimal

class OrderService:
    def __init__(self, session: Session):
        self.session = session
        self.inventory = InventoryService(session)

    def create_order(self, user_id: uuid.UUID, cart_items: list[CartItem], address_id: uuid.UUID) -> Order:
        """ì£¼ë¬¸ ìƒì„±"""
        # ì£¼ë¬¸ ë²ˆí˜¸ ìƒì„±
        order_number = self._generate_order_number()

        # ë°°ì†¡ ì£¼ì†Œ ì¡°íšŒ
        address = self.session.get(Address, address_id)
        if not address or address.user_id != user_id:
            raise ValueError("Invalid address")

        # ì£¼ë¬¸ ìƒì„±
        order = Order(
            user_id=user_id,
            order_number=order_number,
            status='pending',
            shipping_address={
                'full_name': address.full_name,
                'phone': address.phone,
                'address_line1': address.address_line1,
                'address_line2': address.address_line2,
                'city': address.city,
                'state': address.state,
                'postal_code': address.postal_code,
                'country': address.country,
            }
        )

        subtotal = Decimal('0')

        # ì£¼ë¬¸ ì•„ì´í…œ ìƒì„±
        for cart_item in cart_items:
            # ì¬ê³  í™•ì¸ ë° ì˜ˆì•½
            product = cart_item.variant or cart_item.product

            if not self.inventory.reserve_stock(product.id, cart_item.quantity):
                raise ValueError(f"Insufficient stock for {product.name}")

            # ì£¼ë¬¸ ì•„ì´í…œ ìƒì„±
            order_item = OrderItem(
                order=order,
                product_id=cart_item.product_id,
                variant_id=cart_item.variant_id,
                product_name=cart_item.product.name,
                variant_name=cart_item.variant.name if cart_item.variant else None,
                price=product.price,
                quantity=cart_item.quantity,
                subtotal=product.price * cart_item.quantity
            )

            self.session.add(order_item)
            subtotal += order_item.subtotal

        # ê¸ˆì•¡ ê³„ì‚°
        shipping_cost = self._calculate_shipping(subtotal)
        tax = self._calculate_tax(subtotal)
        total = subtotal + shipping_cost + tax

        order.subtotal = subtotal
        order.shipping_cost = shipping_cost
        order.tax = tax
        order.total = total

        self.session.add(order)
        self.session.flush()

        # ì¥ë°”êµ¬ë‹ˆ ë¹„ìš°ê¸°
        for cart_item in cart_items:
            self.session.delete(cart_item)

        return order

    def cancel_order(self, order_id: uuid.UUID) -> Order:
        """ì£¼ë¬¸ ì·¨ì†Œ"""
        order = self.session.get(Order, order_id)

        if order.status not in ['pending', 'paid']:
            raise ValueError("Cannot cancel order in current status")

        # ì¬ê³  ë³µì›
        for item in order.items:
            product = item.variant or item.product
            self.inventory.release_stock(product.id, item.quantity)

        order.status = 'cancelled'
        self.session.flush()

        return order

    def _generate_order_number(self) -> str:
        """ì£¼ë¬¸ ë²ˆí˜¸ ìƒì„±"""
        timestamp = datetime.now().strftime('%Y%m%d')
        random_suffix = secrets.token_hex(4).upper()
        return f"ORD-{timestamp}-{random_suffix}"

    def _calculate_shipping(self, subtotal: Decimal) -> Decimal:
        """ë°°ì†¡ë¹„ ê³„ì‚°"""
        if subtotal >= Decimal('50'):
            return Decimal('0')  # ë¬´ë£Œ ë°°ì†¡
        return Decimal('5.99')

    def _calculate_tax(self, subtotal: Decimal) -> Decimal:
        """ì„¸ê¸ˆ ê³„ì‚° (10%)"""
        return (subtotal * Decimal('0.1')).quantize(Decimal('0.01'))

def get_popular_products(session: Session, limit: int = 10):
    """íŒë§¤ëŸ‰ ê¸°ì¤€ ì¸ê¸° ìƒí’ˆ"""
    stmt = select(
        Product.id,
        Product.name,
        func.sum(OrderItem.quantity).label('total_sold'),
        func.count(func.distinct(Order.id)).label('order_count')
    ).join(OrderItem)\
     .join(Order)\
     .where(Order.status.in_(['paid', 'processing', 'shipped', 'delivered']))\
     .group_by(Product.id, Product.name)\
     .order_by(func.desc('total_sold'))\
     .limit(limit)

    return session.execute(stmt).all()

def get_user_order_stats(session: Session, user_id: uuid.UUID):
    """ì‚¬ìš©ì ì£¼ë¬¸ í†µê³„"""
    stmt = select(
        func.count(Order.id).label('total_orders'),
        func.sum(Order.total).label('total_spent'),
        func.avg(Order.total).label('avg_order_value'),
        func.max(Order.created_at).label('last_order_date')
    ).where(
        Order.user_id == user_id,
        Order.status != 'cancelled'
    )

    return session.execute(stmt).first()
