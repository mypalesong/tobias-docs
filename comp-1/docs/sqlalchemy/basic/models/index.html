ORM 모델 정의

🏗️ Declarative Base​

기본 설정​

SQLAlchemy 2.0 스타일 (권장):

from sqlalchemy import create_engine
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

# Declarative Base 클래스
class Base(DeclarativeBase):
    pass

# 사용 예
class User(Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str]

레거시 스타일 (1.4):

from sqlalchemy.orm import declarative_base
from sqlalchemy import Column, Integer, String

Base = declarative_base()

class User(Base):
    __tablename__ = 'users'

    id = Column(Integer, primary_key=True)
    name = Column(String(50))

이 가이드는 2.0 스타일을 기본으로 합니다.

Engine과 Base 연결​

from sqlalchemy import create_engine
from sqlalchemy.orm import DeclarativeBase

class Base(DeclarativeBase):
    pass

# Engine 생성
engine = create_engine('postgresql://postgres:password@localhost/myapp')

# 모든 테이블 생성
Base.metadata.create_all(engine)

# 특정 테이블만 생성
User.__table__.create(engine, checkfirst=True)

# 모든 테이블 삭제
Base.metadata.drop_all(engine)

📋 기본 모델 정의​

단순 모델​

from sqlalchemy import String
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

class Base(DeclarativeBase):
    pass

class User(Base):
    __tablename__ = 'users'

    # Primary Key
    id: Mapped[int] = mapped_column(primary_key=True)

    # Required 필드
    username: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100))

    # Optional 필드 (NULL 허용)
    full_name: Mapped[str | None] = mapped_column(String(100))

    # 기본값 설정
    is_active: Mapped[bool] = mapped_column(default=True)

    def __repr__(self):
        return f"<User(id={self.id}, username='{self.username}')>"

생성되는 SQL:

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    full_name VARCHAR(100),
    is_active BOOLEAN DEFAULT true
);

완전한 예제 모델​

from datetime import datetime
from sqlalchemy import String, Text, DateTime, func
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

class Base(DeclarativeBase):
    pass

class Article(Base):
    __tablename__ = 'articles'

    # Primary Key
    id: Mapped[int] = mapped_column(primary_key=True)

    # 필수 필드
    title: Mapped[str] = mapped_column(String(200))
    slug: Mapped[str] = mapped_column(String(200), unique=True, index=True)
    content: Mapped[str] = mapped_column(Text)

    # Optional 필드
    summary: Mapped[str | None] = mapped_column(String(500))

    # Boolean
    is_published: Mapped[bool] = mapped_column(default=False)

    # 타임스탬프 (서버 기본값)
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now(),
        onupdate=func.now()
    )

    # 숫자
    view_count: Mapped[int] = mapped_column(default=0)

    def __repr__(self):
        return f"<Article(id={self.id}, title='{self.title}')>"

🎨 PostgreSQL 데이터 타입​

기본 타입​

from sqlalchemy import Integer, BigInteger, SmallInteger
from sqlalchemy import String, Text
from sqlalchemy import Boolean
from sqlalchemy import Float, Numeric
from sqlalchemy import Date, Time, DateTime
from sqlalchemy.orm import Mapped, mapped_column

class Product(Base):
    __tablename__ = 'products'

    # 정수
    id: Mapped[int] = mapped_column(BigInteger, primary_key=True)
    stock: Mapped[int] = mapped_column(SmallInteger, default=0)

    # 문자열
    name: Mapped[str] = mapped_column(String(100))
    description: Mapped[str] = mapped_column(Text)
    sku: Mapped[str] = mapped_column(String(50), unique=True)

    # 불린
    is_available: Mapped[bool] = mapped_column(Boolean, default=True)

    # 숫자 (정밀도 중요한 금액)
    price: Mapped[float] = mapped_column(Numeric(10, 2))  # 12345678.99
    weight: Mapped[float] = mapped_column(Float)

    # 날짜/시간
    manufacturing_date: Mapped[datetime] = mapped_column(Date)
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True))

PostgreSQL 특화 타입​

from sqlalchemy import ARRAY, JSON
from sqlalchemy.dialects.postgresql import UUID, INET, JSONB, ENUM
import uuid

class AdvancedModel(Base):
    __tablename__ = 'advanced'

    # UUID (PostgreSQL UUID)
    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4
    )

    # JSON/JSONB
    metadata_json: Mapped[dict] = mapped_column(JSON)
    settings: Mapped[dict] = mapped_column(JSONB)  # JSONB 권장 (인덱싱 가능)

    # ARRAY
    tags: Mapped[list[str]] = mapped_column(ARRAY(String(50)))
    ratings: Mapped[list[int]] = mapped_column(ARRAY(Integer))

    # INET (IP 주소)
    ip_address: Mapped[str] = mapped_column(INET)

    # ENUM
    status: Mapped[str] = mapped_column(
        ENUM('draft', 'published', 'archived', name='status_enum')
    )

Unique Constraint​

from sqlalchemy import String, UniqueConstraint
from sqlalchemy.orm import Mapped, mapped_column

class User(Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)

    # 단일 컬럼 unique
    email: Mapped[str] = mapped_column(String(100), unique=True)
    username: Mapped[str] = mapped_column(String(50), unique=True)

    # 복합 unique (테이블 레벨)
    first_name: Mapped[str] = mapped_column(String(50))
    last_name: Mapped[str] = mapped_column(String(50))

    __table_args__ = (
        UniqueConstraint('first_name', 'last_name', name='uq_full_name'),
    )

Check Constraint​

from sqlalchemy import CheckConstraint, String
from sqlalchemy.orm import Mapped, mapped_column

class Product(Base):
    __tablename__ = 'products'

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(100))
    price: Mapped[float]
    stock: Mapped[int]
    rating: Mapped[float]

    __table_args__ = (
        CheckConstraint('price > 0', name='check_price_positive'),
        CheckConstraint('stock >= 0', name='check_stock_non_negative'),
        CheckConstraint('rating >= 0 AND rating <= 5', name='check_rating_range'),
    )

PostgreSQL 특화 인덱스​

from sqlalchemy import String, Index, Text
from sqlalchemy.dialects.postgresql import JSONB
from sqlalchemy.orm import Mapped, mapped_column

class Article(Base):
    __tablename__ = 'articles'

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    content: Mapped[str] = mapped_column(Text)
    metadata: Mapped[dict] = mapped_column(JSONB)

    __table_args__ = (
        # GIN 인덱스 (JSONB용)
        Index('idx_metadata_gin', 'metadata', postgresql_using='gin'),

        # 부분 인덱스 (조건부)
        Index('idx_published_recent', 'created_at',
              postgresql_where=Column('is_published') == True),

        # 전문 검색 인덱스
        Index('idx_content_fulltext', 'content', postgresql_using='gin',
              postgresql_ops={'content': 'gin_trgm_ops'}),
    )

🧬 모델 상속​

Single Table Inheritance​

from sqlalchemy import String
from sqlalchemy.orm import Mapped, mapped_column

class Employee(Base):
    __tablename__ = 'employees'

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    type: Mapped[str] = mapped_column(String(50))  # Discriminator

    __mapper_args__ = {
        'polymorphic_on': type,
        'polymorphic_identity': 'employee'
    }

class Engineer(Employee):
    # 같은 테이블 사용
    programming_language: Mapped[str | None] = mapped_column(String(50))

    __mapper_args__ = {
        'polymorphic_identity': 'engineer'
    }

class Manager(Employee):
    # 같은 테이블 사용
    department: Mapped[str | None] = mapped_column(String(50))

    __mapper_args__ = {
        'polymorphic_identity': 'manager'
    }

Joined Table Inheritance​

class Person(Base):
    __tablename__ = 'people'

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    type: Mapped[str] = mapped_column(String(50))

    __mapper_args__ = {
        'polymorphic_on': type,
        'polymorphic_identity': 'person'
    }

class Employee(Person):

🎭 Mixin 패턴​

타임스탬프 Mixin​

from datetime import datetime
from sqlalchemy import DateTime, func
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

class Base(DeclarativeBase):
    pass

class TimestampMixin:
    """타임스탬프 자동 추가"""
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now(),
        onupdate=func.now()
    )

class User(TimestampMixin, Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str]
    # created_at, updated_at 자동 포함

class Post(TimestampMixin, Base):
    __tablename__ = 'posts'

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str]
    # created_at, updated_at 자동 포함

ID와 타임스탬프 Mixin​

import uuid
from datetime import datetime
from sqlalchemy import DateTime, func
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.orm import Mapped, mapped_column

class IDMixin:
    """UUID Primary Key"""
    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4
    )

class TimestampMixin:
    """생성/수정 시간"""
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now(),
        onupdate=func.now()
    )

class SoftDeleteMixin:
    """소프트 삭제"""
    is_deleted: Mapped[bool] = mapped_column(default=False)
    deleted_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))

# 모든 Mixin 적용
class Article(IDMixin, TimestampMixin, SoftDeleteMixin, Base):
    __tablename__ = 'articles'

    title: Mapped[str]
    content: Mapped[str]
    # id, created_at, updated_at, is_deleted, deleted_at 자동 포함

✅ 유효성 검증​

Python 레벨 검증​

from sqlalchemy import String, event
from sqlalchemy.orm import Mapped, mapped_column, validates

class User(Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100))
    age: Mapped[int | None]

    @validates('email')
    def validate_email(self, key, email):
        """이메일 형식 검증"""
        if '@' not in email:
            raise ValueError("Invalid email format")
        return email.lower()

    @validates('age')
    def validate_age(self, key, age):
        """나이 범위 검증"""
        if age is not None and (age < 0 or age > 150):
            raise ValueError("Age must be between 0 and 150")
        return age

    @validates('username')
    def validate_username(self, key, username):
        """사용자명 검증"""
        if len(username) < 3:
            raise ValueError("Username must be at least 3 characters")
        return username

💡 실전 모델 예제​

블로그 시스템​

import uuid
from datetime import datetime
from sqlalchemy import String, Text, DateTime, func, CheckConstraint
from sqlalchemy.dialects.postgresql import UUID, ARRAY
from sqlalchemy.orm import Mapped, mapped_column, validates

class Base(DeclarativeBase):
    pass

class TimestampMixin:
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        server_default=func.now(),
        onupdate=func.now()
    )

class User(TimestampMixin, Base):
    __tablename__ = 'users'

    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4
    )
    username: Mapped[str] = mapped_column(String(50), unique=True, index=True)
    email: Mapped[str] = mapped_column(String(100), unique=True, index=True)
    password_hash: Mapped[str] = mapped_column(String(255))
    full_name: Mapped[str | None] = mapped_column(String(100))
    bio: Mapped[str | None] = mapped_column(Text)
    is_active: Mapped[bool] = mapped_column(default=True)
    is_admin: Mapped[bool] = mapped_column(default=False)

    @validates('email')
    def validate_email(self, key, email):
        if '@' not in email:
            raise ValueError("Invalid email")
        return email.lower()

class Post(TimestampMixin, Base):
    __tablename__ = 'posts'

    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4
    )
    title: Mapped[str] = mapped_column(String(200), index=True)
    slug: Mapped[str] = mapped_column(String(200), unique=True, index=True)
    content: Mapped[str] = mapped_column(Text)
    summary: Mapped[str | None] = mapped_column(String(500))

    # PostgreSQL ARRAY
    tags: Mapped[list[str]] = mapped_column(ARRAY(String(50)), default=list)

    # Status
    status: Mapped[str] = mapped_column(
        String(20),
        default='draft',
        index=True
    )

    # 통계
    view_count: Mapped[int] = mapped_column(default=0)
    like_count: Mapped[int] = mapped_column(default=0)

    # 게시 시간
    published_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))

    __table_args__ = (
        CheckConstraint("status IN ('draft', 'published', 'archived')",
                       name='check_status'),
        CheckConstraint('view_count >= 0', name='check_view_count'),
        CheckConstraint('like_count >= 0', name='check_like_count'),
    )

    @validates('status')
    def validate_status(self, key, status):
        allowed = ['draft', 'published', 'archived']
        if status not in allowed:
            raise ValueError(f"Status must be one of {allowed}")
        return status
