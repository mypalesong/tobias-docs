기본 CRUD 작업

SQLAlchemy ORM으로 PostgreSQL 데이터를 생성, 조회, 수정, 삭제하는 방법을 배웁니다.

🔧 기본 설정​

from sqlalchemy import create_engine, String, select
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, Session

# Base 클래스
class Base(DeclarativeBase):
    pass

# 모델 정의
class User(Base):
    __tablename__ = 'users'

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100))
    age: Mapped[int | None]

    def __repr__(self):
        return f"<User(id={self.id}, username='{self.username}')>"

# Engine 생성
engine = create_engine('postgresql://postgres:password@localhost/myapp')

# 테이블 생성
Base.metadata.create_all(engine)

➕ Create - 데이터 생성​

단일 레코드 생성​

from sqlalchemy.orm import Session

# Session 생성
with Session(engine) as session:
    # 1. User 객체 생성
    user = User(
        username='alice',
        email='alice@example.com',
        age=25
    )

    # 2. Session에 추가
    session.add(user)

    # 3. 커밋 (실제 DB에 저장)
    session.commit()

    # 4. ID 확인 (자동 생성됨)
    print(f"Created user ID: {user.id}")  # 1

생성되는 SQL:

INSERT INTO users (username, email, age)
VALUES ('alice', 'alice@example.com', 25)
RETURNING id;

여러 레코드 동시 생성​

with Session(engine) as session:
    # 여러 객체 생성
    users = [
        User(username='bob', email='bob@example.com', age=30),
        User(username='charlie', email='charlie@example.com', age=35),
        User(username='diana', email='diana@example.com', age=28),
    ]

    # 한 번에 추가
    session.add_all(users)
    session.commit()

    print(f"Created {len(users)} users")

생성 후 즉시 조회​

with Session(engine) as session:
    user = User(username='eve', email='eve@example.com')
    session.add(user)
    session.flush()  # DB에 반영 (커밋 전)

    # ID 사용 가능
    print(f"User ID before commit: {user.id}")

    session.commit()

🔍 Read - 데이터 조회​

전체 조회​

SQLAlchemy 2.0 스타일 (권장):

from sqlalchemy import select

with Session(engine) as session:
    # select() 사용
    stmt = select(User)
    users = session.execute(stmt).scalars().all()

    for user in users:
        print(user.username, user.email)

레거시 스타일 (1.4):

with Session(engine) as session:
    # query() 사용 (deprecated)
    users = session.query(User).all()

    for user in users:
        print(user.username)

Primary Key로 조회​

with Session(engine) as session:
    # get() - Primary Key로 단일 조회
    user = session.get(User, 1)

    if user:
        print(f"Found: {user.username}")
    else:
        print("User not found")

조건부 조회 (WHERE)​

from sqlalchemy import select

with Session(engine) as session:
    # 단일 조건
    stmt = select(User).where(User.username == 'alice')
    user = session.execute(stmt).scalar_one_or_none()

    # 여러 조건 (AND)
    stmt = select(User).where(
        User.age > 25,
        User.email.like('%@example.com')
    )
    users = session.execute(stmt).scalars().all()

    # OR 조건
    from sqlalchemy import or_

    stmt = select(User).where(
        or_(
            User.username == 'alice',
            User.username == 'bob'
        )
    )
    users = session.execute(stmt).scalars().all()

정렬 (ORDER BY)​

from sqlalchemy import select

with Session(engine) as session:
    # 오름차순
    stmt = select(User).order_by(User.age)
    users = session.execute(stmt).scalars().all()

    # 내림차순
    stmt = select(User).order_by(User.age.desc())
    users = session.execute(stmt).scalars().all()

    # 여러 컬럼 정렬
    stmt = select(User).order_by(User.age.desc(), User.username)
    users = session.execute(stmt).scalars().all()

페이징 (LIMIT/OFFSET)​

with Session(engine) as session:
    # 페이지 1 (처음 10개)
    stmt = select(User).limit(10).offset(0)
    users = session.execute(stmt).scalars().all()

    # 페이지 2 (11-20번째)
    stmt = select(User).limit(10).offset(10)
    users = session.execute(stmt).scalars().all()

    # 함수로 구현
    def get_users_page(page: int, per_page: int = 10):
        with Session(engine) as session:
            stmt = select(User)\
                .order_by(User.id)\
                .limit(per_page)\
                .offset((page - 1) * per_page)
            return session.execute(stmt).scalars().all()

    # 사용
    page1_users = get_users_page(1)
    page2_users = get_users_page(2)

개수 세기 (COUNT)​

from sqlalchemy import select, func

with Session(engine) as session:
    # 전체 개수
    stmt = select(func.count()).select_from(User)
    count = session.execute(stmt).scalar()
    print(f"Total users: {count}")

    # 조건부 개수
    stmt = select(func.count()).select_from(User).where(User.age > 30)
    count = session.execute(stmt).scalar()
    print(f"Users over 30: {count}")

특정 컬럼만 조회​

with Session(engine) as session:
    # 특정 컬럼만
    stmt = select(User.username, User.email)
    results = session.execute(stmt).all()

    for username, email in results:
        print(f"{username}: {email}")

    # 컬럼 하나만
    stmt = select(User.username).where(User.age > 25)
    usernames = session.execute(stmt).scalars().all()
    print(usernames)  # ['alice', 'bob', ...]

조회 결과 옵션​

with Session(engine) as session:
    stmt = select(User).where(User.username == 'alice')

    # scalar_one() - 정확히 1개, 없거나 여러 개면 에러
    user = session.execute(stmt).scalar_one()

    # scalar_one_or_none() - 0개 또는 1개, 여러 개면 에러
    user = session.execute(stmt).scalar_one_or_none()

    # first() - 첫 번째, 없으면 None
    user = session.execute(stmt).first()

    # all() - 모두 리스트로
    users = session.execute(stmt).scalars().all()

✏️ Update - 데이터 수정​

객체 수정 후 커밋​

with Session(engine) as session:
    # 1. 조회
    user = session.get(User, 1)

    # 2. 수정
    user.email = 'newemail@example.com'
    user.age = 26

    # 3. 커밋 (자동으로 UPDATE 실행)
    session.commit()

생성되는 SQL:

UPDATE users
SET email = 'newemail@example.com', age = 26
WHERE id = 1;

Bulk Update (대량 수정)​

from sqlalchemy import update

with Session(engine) as session:
    # update() 문 사용
    stmt = update(User).where(User.age < 20).values(age=20)
    result = session.execute(stmt)

    print(f"Updated {result.rowcount} rows")
    session.commit()

생성되는 SQL:

UPDATE users
SET age = 20
WHERE age < 20;

조건부 일괄 수정​

from sqlalchemy import update

with Session(engine) as session:
    # 모든 이메일을 소문자로
    stmt = update(User).values(email=func.lower(User.email))
    session.execute(stmt)
    session.commit()

    # 나이 10% 증가
    stmt = update(User).where(User.age.isnot(None)).values(age=User.age + 1)
    session.execute(stmt)
    session.commit()

SessionLocal 패턴 (FastAPI 등)​

from sqlalchemy.orm import sessionmaker

# Session Factory
SessionLocal = sessionmaker(
    autocommit=False,
    autoflush=False,
    bind=engine
)

# Dependency
def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# 사용
from fastapi import Depends

@app.get("/users/")
def read_users(db: Session = Depends(get_db)):
    users = db.execute(select(User)).scalars().all()
    return users

🔒 트랜잭션​

기본 트랜잭션​

with Session(engine) as session:
    try:
        # 여러 작업을 하나의 트랜잭션으로
        user1 = User(username='alice', email='alice@example.com')
        user2 = User(username='bob', email='bob@example.com')

        session.add(user1)
        session.add(user2)

        # 모두 성공하면 커밋
        session.commit()
    except Exception as e:
        # 하나라도 실패하면 롤백
        session.rollback()
        raise

명시적 트랜잭션​

with Session(engine) as session:
    with session.begin():
        # begin() 내에서는 자동 커밋/롤백
        user = User(username='alice', email='alice@example.com')
        session.add(user)
        # 블록 종료 시 자동 커밋

Savepoint (중첩 트랜잭션)​

with Session(engine) as session:
    session.begin()

    try:
        user1 = User(username='alice', email='alice@example.com')
        session.add(user1)

        # Savepoint 생성
        with session.begin_nested():
            user2 = User(username='invalid', email='invalid')
            session.add(user2)
            # 여기서 에러 발생 시 savepoint로 롤백

        session.commit()
    except Exception:
        session.rollback()

💡 실전 CRUD 패턴​

Repository 패턴​

from typing import Optional, List
from sqlalchemy import select
from sqlalchemy.orm import Session

class UserRepository:
    def __init__(self, session: Session):
        self.session = session

    def create(self, username: str, email: str, age: Optional[int] = None) -> User:
        """사용자 생성"""
        user = User(username=username, email=email, age=age)
        self.session.add(user)
        self.session.flush()
        return user

    def get_by_id(self, user_id: int) -> Optional[User]:
        """ID로 조회"""
        return self.session.get(User, user_id)

    def get_by_username(self, username: str) -> Optional[User]:
        """사용자명으로 조회"""
        stmt = select(User).where(User.username == username)
        return self.session.execute(stmt).scalar_one_or_none()

    def get_all(self, skip: int = 0, limit: int = 100) -> List[User]:
        """전체 조회 (페이징)"""
        stmt = select(User).offset(skip).limit(limit)
        return self.session.execute(stmt).scalars().all()

    def update(self, user: User, **kwargs) -> User:
        """사용자 수정"""
        for key, value in kwargs.items():
            setattr(user, key, value)
        self.session.flush()
        return user

    def delete(self, user: User) -> None:
        """사용자 삭제"""
        self.session.delete(user)
        self.session.flush()

# 사용 예
with Session(engine) as session:
    repo = UserRepository(session)

    # 생성
    user = repo.create(username='alice', email='alice@example.com', age=25)

    # 조회
    found_user = repo.get_by_username('alice')

    # 수정
    repo.update(found_user, age=26, email='newemail@example.com')

    # 삭제
    repo.delete(found_user)

    session.commit()

CRUD 헬퍼 함수​

from typing import Optional, List, Type, TypeVar
from sqlalchemy import select
from sqlalchemy.orm import Session

T = TypeVar('T')

def create_item(session: Session, item: T) -> T:
    """아이템 생성"""
    session.add(item)
    session.flush()
    return item

def get_item(session: Session, model: Type[T], item_id: int) -> Optional[T]:
    """ID로 조회"""
    return session.get(model, item_id)

def get_items(
    session: Session,
    model: Type[T],
    skip: int = 0,
    limit: int = 100
) -> List[T]:
    """전체 조회"""
    stmt = select(model).offset(skip).limit(limit)
    return session.execute(stmt).scalars().all()

def update_item(session: Session, item: T, **kwargs) -> T:
    """아이템 수정"""
    for key, value in kwargs.items():
        setattr(item, key, value)
    session.flush()
    return item

def delete_item(session: Session, item: T) -> None:
    """아이템 삭제"""
    session.delete(item)
    session.flush()

# 사용 예
with Session(engine) as session:
    # 생성
    user = User(username='alice', email='alice@example.com')
    user = create_item(session, user)

    # 조회
    user = get_item(session, User, user.id)

    # 수정
    user = update_item(session, user, age=26)

    # 삭제
    delete_item(session, user)

    session.commit()

🎯 실전 예제​

사용자 관리 시스템​

from typing import Optional, List
from sqlalchemy import select, func
from sqlalchemy.orm import Session

class UserService:
    """사용자 비즈니스 로직"""

    def __init__(self, session: Session):
        self.session = session

    def register_user(self, username: str, email: str, password: str) -> User:
        """신규 사용자 등록"""
        # 중복 체크
        if self.get_by_username(username):
            raise ValueError(f"Username '{username}' already exists")

        if self.get_by_email(email):
            raise ValueError(f"Email '{email}' already exists")

        # 사용자 생성
        user = User(
            username=username,
            email=email.lower(),
            password_hash=hash_password(password)  # 실제로는 bcrypt 등 사용
        )

        self.session.add(user)
        self.session.flush()
        return user

    def get_by_username(self, username: str) -> Optional[User]:
        """사용자명으로 조회"""
        stmt = select(User).where(User.username == username)
        return self.session.execute(stmt).scalar_one_or_none()

    def get_by_email(self, email: str) -> Optional[User]:
        """이메일로 조회"""
        stmt = select(User).where(User.email == email.lower())
        return self.session.execute(stmt).scalar_one_or_none()

    def search_users(self, query: str) -> List[User]:
        """사용자 검색"""
        pattern = f"%{query}%"
        stmt = select(User).where(
            User.username.like(pattern) | User.email.like(pattern)
        )
        return self.session.execute(stmt).scalars().all()

    def get_active_users_count(self) -> int:
        """활성 사용자 수"""
        stmt = select(func.count()).select_from(User).where(User.is_active == True)
        return self.session.execute(stmt).scalar()

    def deactivate_user(self, user_id: int) -> User:
        """사용자 비활성화"""
        user = self.session.get(User, user_id)
        if not user:
            raise ValueError(f"User {user_id} not found")

        user.is_active = False
        self.session.flush()
        return user

# 사용
with Session(engine) as session:
    service = UserService(session)

    try:
        # 등록
        user = service.register_user('alice', 'alice@example.com', 'secret123')
        print(f"Registered: {user.username}")

        # 검색
        results = service.search_users('ali')
        print(f"Found {len(results)} users")

        # 통계
        count = service.get_active_users_count()
        print(f"Active users: {count}")

        session.commit()
    except Exception as e:
        session.rollback()
        print(f"Error: {e}")

💡 프로 팁​

1. N+1 쿼리 문제 주의​

# ❌ 나쁜 예: N+1 쿼리
users = session.execute(select(User)).scalars().all()
for user in users:
    print(user.posts)  # 각 사용자마다 쿼리 실행!

# ✅ 좋은 예: Eager Loading (중급편에서 자세히)
from sqlalchemy.orm import selectinload

stmt = select(User).options(selectinload(User.posts))
users = session.execute(stmt).scalars().all()
for user in users:
    print(user.posts)  # 추가 쿼리 없음

2. flush() vs commit()​

with Session(engine) as session:
    user = User(username='alice', email='alice@example.com')
    session.add(user)

    # flush(): DB에 반영, 트랜잭션은 유지
    session.flush()
    print(user.id)  # ID 사용 가능

    # 다른 작업...

    # commit(): 트랜잭션 완료
    session.commit()

3. 자동 새로고침​

with Session(engine) as session:
    user = session.get(User, 1)
    user.age = 30
    session.commit()

    # 커밋 후 자동으로 DB에서 새로고침됨
    print(user.updated_at)  # 최신 값
