PostgreSQL 연결 설정

SQLAlchemy로 PostgreSQL 데이터베이스에 연결하는 방법을 배웁니다.

🎯 목표​

  • ✅ PostgreSQL 연결 문자열 작성

  • ✅ 연결 풀 관리
  • ✅ 환경별 설정 분리

🔌 PostgreSQL 연결 문자열​

기본 형식​

postgresql://[username]:[password]@[host]:[port]/[database]

예제​

# 로컬 개발 환경
DATABASE_URL = "postgresql://postgres:password@localhost:5432/myapp"

# 프로덕션 (AWS RDS)
DATABASE_URL = "postgresql://admin:secretpass@mydb.abc123.us-east-1.rds.amazonaws.com:5432/prod_db"

# Docker 컨테이너
DATABASE_URL = "postgresql://user:pass@postgres-container:5432/app_db"

특수 문자 처리​

from urllib.parse import quote_plus

password = "p@ssw0rd!#$"
encoded_password = quote_plus(password)

DATABASE_URL = f"postgresql://user:{encoded_password}@localhost/mydb"

🏗️ Engine 생성​

기본 Engine​

from sqlalchemy import create_engine

# 기본 연결
engine = create_engine('postgresql://postgres:password@localhost/myapp')

# 연결 테스트
with engine.connect() as conn:
    result = conn.execute(text("SELECT 1"))
    print(result.scalar())  # 1

Engine 옵션​

from sqlalchemy import create_engine
from sqlalchemy.pool import QueuePool

engine = create_engine(
    'postgresql://postgres:password@localhost/myapp',

    # 연결 풀 설정
    poolclass=QueuePool,
    pool_size=10,              # 기본 연결 수
    max_overflow=20,           # 추가 가능한 연결 수
    pool_timeout=30,           # 연결 대기 시간 (초)
    pool_recycle=3600,         # 연결 재사용 주기 (1시간)
    pool_pre_ping=True,        # 연결 유효성 사전 체크

    # 디버그 설정
    echo=False,                # SQL 로그 출력 (개발 시 True)
    echo_pool=False,           # 연결 풀 로그

    # 성능 설정
    isolation_level="READ COMMITTED",  # 트랜잭션 격리 수준
    connect_args={
        "connect_timeout": 10,          # 연결 타임아웃
        "options": "-c timezone=UTC"    # PostgreSQL 옵션
    }
)

비동기 Engine (asyncio)​

from sqlalchemy.ext.asyncio import create_async_engine

# asyncpg 드라이버 사용
async_engine = create_async_engine(
    'postgresql+asyncpg://postgres:password@localhost/myapp',
    echo=True,
    pool_size=20,
    max_overflow=0
)

💾 연결 풀 관리​

연결 풀이란?​

연결 풀은 데이터베이스 연결을 재사용하여 성능을 향상시킵니다.

from sqlalchemy import create_engine, text

engine = create_engine(
    'postgresql://postgres:password@localhost/myapp',
    pool_size=5,        # 항상 유지하는 연결 수
    max_overflow=10,    # 필요 시 추가 생성 가능한 수
)

# 총 최대 연결 수 = pool_size + max_overflow = 15

# 연결 사용
with engine.connect() as conn:
    result = conn.execute(text("SELECT * FROM users"))
    # 자동으로 연결이 풀로 반환됨

연결 풀 모니터링​

# 현재 연결 상태 확인
pool = engine.pool
print(f"Pool size: {pool.size()}")
print(f"Checked out: {pool.checkedout()}")
print(f"Overflow: {pool.overflow()}")

연결 풀 타입​

from sqlalchemy.pool import NullPool, StaticPool, QueuePool

# 1. QueuePool (기본값, 권장)
# - 가장 일반적, 멀티스레드 안전
engine = create_engine(
    'postgresql://...',
    poolclass=QueuePool,
    pool_size=10
)

# 2. NullPool (풀 사용 안 함)
# - 매번 새 연결 생성, 테스트/디버깅용
engine = create_engine(
    'postgresql://...',
    poolclass=NullPool
)

# 3. StaticPool (단일 연결)
# - SQLite 인메모리 DB용
engine = create_engine(
    'sqlite:///:memory:',
    poolclass=StaticPool
)

⚙️ 환경별 설정​

개발/프로덕션 분리​

config.py:

import os
from sqlalchemy import create_engine
from sqlalchemy.pool import NullPool, QueuePool

class Config:
    """기본 설정"""
    SQLALCHEMY_DATABASE_URI = os.getenv(
        'DATABASE_URL',
        'postgresql://postgres:password@localhost/myapp'
    )
    SQLALCHEMY_ECHO = False

class DevelopmentConfig(Config):
    """개발 환경"""
    SQLALCHEMY_ECHO = True
    SQLALCHEMY_POOL_SIZE = 5
    SQLALCHEMY_MAX_OVERFLOW = 10

class ProductionConfig(Config):
    """프로덕션 환경"""
    SQLALCHEMY_POOL_SIZE = 20
    SQLALCHEMY_MAX_OVERFLOW = 40
    SQLALCHEMY_POOL_RECYCLE = 3600
    SQLALCHEMY_POOL_PRE_PING = True

class TestConfig(Config):
    """테스트 환경"""
    SQLALCHEMY_DATABASE_URI = 'postgresql://postgres:password@localhost/test_db'
    SQLALCHEMY_POOLCLASS = NullPool

# 환경 선택
config = {
    'development': DevelopmentConfig,
    'production': ProductionConfig,
    'testing': TestConfig,
}

def get_config():
    env = os.getenv('ENV', 'development')
    return config[env]

database.py:

from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from config import get_config

config = get_config()

# Engine 생성
engine = create_engine(
    config.SQLALCHEMY_DATABASE_URI,
    echo=config.SQLALCHEMY_ECHO,
    pool_size=config.SQLALCHEMY_POOL_SIZE,
    max_overflow=config.SQLALCHEMY_MAX_OVERFLOW,
    pool_recycle=getattr(config, 'SQLALCHEMY_POOL_RECYCLE', 3600),
    pool_pre_ping=getattr(config, 'SQLALCHEMY_POOL_PRE_PING', False),
)

# Session Factory
SessionLocal = sessionmaker(
    autocommit=False,
    autoflush=False,
    bind=engine
)

# Dependency
def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

🔍 연결 테스트​

기본 연결 테스트​

from sqlalchemy import create_engine, text

def test_connection(database_url):
    """데이터베이스 연결 테스트"""
    try:
        engine = create_engine(database_url)
        with engine.connect() as conn:
            result = conn.execute(text("SELECT version()"))
            version = result.scalar()
            print(f"✓ 연결 성공!")
            print(f"PostgreSQL 버전: {version}")
            return True
    except Exception as e:
        print(f"✗ 연결 실패: {e}")
        return False

# 테스트 실행
test_connection('postgresql://postgres:password@localhost/myapp')

연결 풀 테스트​

import time
from concurrent.futures import ThreadPoolExecutor

def execute_query(engine, query_id):
    """쿼리 실행 테스트"""
    start = time.time()
    with engine.connect() as conn:
        result = conn.execute(text("SELECT pg_sleep(1)"))
        elapsed = time.time() - start
        print(f"Query {query_id}: {elapsed:.2f}s")

# 동시 연결 테스트
engine = create_engine(
    'postgresql://postgres:password@localhost/myapp',
    pool_size=5,
    max_overflow=5
)

with ThreadPoolExecutor(max_workers=10) as executor:
    futures = [executor.submit(execute_query, engine, i)
               for i in range(10)]
    for future in futures:
        future.result()

# pool_size=5, max_overflow=5이므로
# 처음 10개 연결 생성
# 이후 연결 재사용

🛠️ 실전 설정 예제​

FastAPI 통합​

from fastapi import FastAPI, Depends
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session

# Engine 생성
SQLALCHEMY_DATABASE_URL = "postgresql://postgres:password@localhost/myapp"
engine = create_engine(
    SQLALCHEMY_DATABASE_URL,
    pool_pre_ping=True,
    pool_size=10,
    max_overflow=20
)

# Session Factory
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

# FastAPI 앱
app = FastAPI()

# Dependency
def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# 엔드포인트
@app.get("/users/")
def read_users(db: Session = Depends(get_db)):
    users = db.query(User).all()
    return users

1. 연결 문자열 보안​

2. 연결 재시도​

from sqlalchemy import create_engine
from sqlalchemy.exc import OperationalError
import time

def create_engine_with_retry(url, max_retries=5):
    """연결 실패 시 재시도"""
    for attempt in range(max_retries):
        try:
            engine = create_engine(url, pool_pre_ping=True)
            # 연결 테스트
            with engine.connect() as conn:
                conn.execute(text("SELECT 1"))
            print(f"✓ 연결 성공 (시도 {attempt + 1})")
            return engine
        except OperationalError as e:
            if attempt < max_retries - 1:
                wait_time = 2 ** attempt
                print(f"✗ 연결 실패, {wait_time}초 후 재시도...")
                time.sleep(wait_time)
            else:
                raise

engine = create_engine_with_retry('postgresql://postgres:password@localhost/myapp')

3. 헬스 체크​

from sqlalchemy import text

def health_check(engine):
    """데이터베이스 헬스 체크"""
    try:
        with engine.connect() as conn:
            conn.execute(text("SELECT 1"))
        return {"status": "healthy", "database": "connected"}
    except Exception as e:
        return {"status": "unhealthy", "error": str(e)}

# FastAPI 엔드포인트
@app.get("/health")
def health():
    return health_check(engine)
