            return RedirectResponse(url, status_code=301)

# PKCE 구현
import hashlib
import base64

def generate_pkce():
    """PKCE Code Verifier + Challenge"""
    code_verifier = secrets.token_urlsafe(32)

    digest = hashlib.sha256(code_verifier.encode()).digest()
    code_challenge = base64.urlsafe_b64encode(digest).decode().rstrip('=')

    return code_verifier, code_challenge

# 사용
code_verifier, code_challenge = generate_pkce()
request.session['code_verifier'] = code_verifier

# Authorization Request
auth_url = f"...&code_challenge={code_challenge}&code_challenge_method=S256"

from passlib.context import CryptContext

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def store_refresh_token(user_id: str, refresh_token: str):
    """Refresh Token 해시 저장"""

    # Refresh Token 해시
    token_hash = pwd_context.hash(refresh_token)

    # DB 저장
    db.execute(
        "INSERT INTO refresh_tokens (user_id, token_hash, created_at) VALUES (?, ?, ?)",
        (user_id, token_hash, datetime.utcnow())

def verify_refresh_token(user_id: str, refresh_token: str) -> bool:
    """Refresh Token 검증"""

    # DB에서 해시 가져오기
    result = db.execute(
        "SELECT token_hash FROM refresh_tokens WHERE user_id = ?",
        (user_id,)
    ).fetchone()

    if not result:
        return False

    # 해시 비교
    return pwd_context.verify(refresh_token, result[0])

4. 토큰 폐기 (Revocation)​

# Redis로 토큰 블랙리스트 구현

import redis

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def revoke_token(token: str, expires_in: int):
    """토큰 폐기"""

    # Redis에 블랙리스트 저장 (TTL 설정)
    redis_client.setex(
        f"revoked:token:{token}",
        expires_in,  # TTL (남은 만료 시간)
        "1"
    )

def is_token_revoked(token: str) -> bool:
    """토큰 폐기 확인"""
    return redis_client.exists(f"revoked:token:{token}") > 0

# 의존성으로 사용
async def get_current_user(token: str = Depends(oauth2_scheme)):
    """블랙리스트 확인 포함"""

    # 폐기 확인
    if is_token_revoked(token):
        raise HTTPException(
            status_code=401,
            detail="Token has been revoked"
        )

    # 일반 검증
    payload = verify_token(token)
    return payload

# 로그아웃
@app.post('/logout')
async def logout(token: str = Depends(oauth2_scheme)):
    """로그아웃 (토큰 폐기)"""

    # Token 디코딩 (검증 없이)
    payload = jwt.decode(token, options={"verify_signature": False})
    exp = payload['exp']
    now = datetime.utcnow().timestamp()

    # 남은 시간
    expires_in = int(exp - now)

    if expires_in > 0:
        revoke_token(token, expires_in)

    return {"message": "Logged out successfully"}

from pydantic import BaseModel, validator, HttpUrl

class AuthorizationRequest(BaseModel):
    """Authorization Request 검증"""

    client_id: str
    redirect_uri: HttpUrl  # URL 형식 검증
    response_type: str
    scope: str
    state: str

    @validator('response_type')
    def validate_response_type(cls, v):
        """response_type 검증"""
        allowed = ['code', 'token', 'id_token']
        if v not in allowed:
            raise ValueError(f'Invalid response_type: {v}')
        return v

    @validator('scope')
    def validate_scope(cls, v):
        """scope 검증"""
        allowed_scopes = {'openid', 'email', 'profile', 'offline_access'}
        requested = set(v.split())

        if not requested.issubset(allowed_scopes):
            invalid = requested - allowed_scopes
            raise ValueError(f'Invalid scopes: {invalid}')

        return v

    @validator('state')
    def validate_state(cls, v):
        """state 최소 길이 검증"""
        if len(v) < 32:
            raise ValueError('State too short')
        return v

1. FastAPI Rate Limiting​

from fastapi import Request, HTTPException
from collections import defaultdict
from datetime import datetime, timedelta

# Rate Limit 저장소
rate_limit_store = defaultdict(list)

def rate_limit(max_requests: int, window_seconds: int):
    """Rate Limiting 데코레이터"""

    def decorator(func):
        async def wrapper(request: Request, *args, **kwargs):
            # 클라이언트 식별 (IP 또는 User ID)
            client_id = request.client.host

            # 현재 시간
            now = datetime.now()
            window_start = now - timedelta(seconds=window_seconds)

            # 윈도우 내 요청 필터링
            requests = rate_limit_store[client_id]
            requests = [r for r in requests if r > window_start]
            rate_limit_store[client_id] = requests

            # Rate Limit 확인
            if len(requests) >= max_requests:
                raise HTTPException(
                    status_code=429,
                    detail=f"Rate limit exceeded. Max {max_requests} requests per {window_seconds} seconds"
                )

            # 요청 기록
            requests.append(now)

            return await func(request, *args, **kwargs)

        return wrapper
    return decorator

2. Redis로 Rate Limiting​

import redis
from datetime import timedelta

redis_client = redis.Redis(host='localhost', port=6379, db=0)

async def check_rate_limit(
    client_id: str,
    max_requests: int,
    window_seconds: int
):
    """Redis 기반 Rate Limiting"""

    key = f"rate_limit:{client_id}"
    current = redis_client.get(key)

    if current is None:
        # 첫 요청
        redis_client.setex(key, window_seconds, 1)
        return True

    current = int(current)

    if current >= max_requests:
        # Rate limit 초과
        return False

    # 요청 증가
    redis_client.incr(key)
    return True

# 미들웨어로 적용
class RateLimitMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        client_id = request.client.host

        # Rate Limit 확인
        if not await check_rate_limit(client_id, max_requests=100, window_seconds=60):
            return JSONResponse(
                status_code=429,
                content={"detail": "Too many requests"}
            )

        return await call_next(request)

app.add_middleware(RateLimitMiddleware)

from freezegun import freeze_time

def test_token_expiration():
    """토큰 만료 테스트"""

    # Token 생성
    token = create_access_token(
        data={"sub": "user123"},
        expires_delta=timedelta(minutes=15)
    )

    # 즉시 검증 (성공)
    payload = verify_token(token)
    assert payload['sub'] == "user123"

    # 16분 후 (실패)
    with freeze_time(datetime.utcnow() + timedelta(minutes=16)):
        with pytest.raises(HTTPException) as exc_info:
            verify_token(token)
        assert exc_info.value.status_code == 401
