1. 토큰 발급​

from datetime import datetime, timedelta
from jose import jwt
import secrets

class TokenManager:
    """토큰 관리자"""

    def __init__(self, secret_key: str):
        self.secret_key = secret_key
        self.algorithm = "HS256"

    def create_token_pair(self, user_id: str) -> dict:
        """Access Token + Refresh Token 쌍 생성"""

        now = datetime.utcnow()

        # Access Token (짧은 수명)
        access_token_expires = now + timedelta(minutes=15)
        access_token_payload = {
            "sub": user_id,
            "type": "access",
            "exp": access_token_expires,
            "iat": now,
            "jti": secrets.token_urlsafe(16)  # 토큰 고유 ID
        }
        access_token = jwt.encode(
            access_token_payload,
            self.secret_key,
            algorithm=self.algorithm
        )

        # Refresh Token (긴 수명)
        refresh_token_expires = now + timedelta(days=30)
        refresh_token_payload = {
            "sub": user_id,
            "type": "refresh",
            "exp": refresh_token_expires,
            "iat": now,
            "jti": secrets.token_urlsafe(16)
        }
        refresh_token = jwt.encode(
            refresh_token_payload,
            self.secret_key,
            algorithm=self.algorithm
        )

        return {
            "access_token": access_token,
            "refresh_token": refresh_token,
            "token_type": "bearer",
            "expires_in": 900,  # 15분 (초)
            "refresh_expires_in": 2592000  # 30일 (초)
        }

2. 토큰 검증​

from jose import jwt, JWTError
from fastapi import HTTPException, status

class TokenManager:
    # ...

    def verify_token(self, token: str, token_type: str = "access") -> dict:
        """토큰 검증"""

        try:
            # JWT 디코딩 및 검증
            payload = jwt.decode(
                token,
                self.secret_key,
                algorithms=[self.algorithm]
            )

            # 토큰 타입 확인
            if payload.get("type") != token_type:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail=f"Invalid token type. Expected {token_type}"
                )

            # 사용자 ID 확인
            user_id = payload.get("sub")
            if not user_id:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="Invalid token: no subject"
                )

            return payload

        except JWTError as e:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail=f"Invalid token: {str(e)}"
            )

    def is_token_expired(self, token: str) -> bool:
        """토큰 만료 확인 (검증 없이)"""
        try:
            payload = jwt.decode(
                token,
                options={"verify_signature": False, "verify_exp": False}
            )
            exp = payload.get("exp")
            return datetime.fromtimestamp(exp) < datetime.utcnow()
        except:
            return True

4. 토큰 폐기​

import redis

redis_client = redis.Redis(host='localhost', port=6379, decode_responses=True)

class TokenManager:
    # ...

    def revoke_token(self, token: str):
        """토큰 폐기 (블랙리스트 추가)"""

        # 토큰 디코딩 (검증 없이)
        payload = jwt.decode(
            token,
            options={"verify_signature": False}
        )

        # 남은 시간 계산
        exp = payload.get("exp")
        now = datetime.utcnow().timestamp()
        ttl = int(exp - now)

        if ttl > 0:
            # Redis에 블랙리스트 저장 (TTL 설정)
            jti = payload.get("jti")
            redis_client.setex(
                f"blacklist:token:{jti}",
                ttl,
                "1"
            )

    def is_token_blacklisted(self, token: str) -> bool:
        """블랙리스트 확인"""

        try:
            payload = jwt.decode(
                token,
                options={"verify_signature": False}
            )
            jti = payload.get("jti")
            return redis_client.exists(f"blacklist:token:{jti}") > 0
        except:
            return True

    def revoke_all_user_tokens(self, user_id: str):
        """사용자의 모든 토큰 폐기"""

        # DB에서 사용자의 모든 refresh token 조회
        refresh_tokens = db.query(RefreshToken).filter_by(user_id=user_id).all()

        # 모두 폐기
        for rt in refresh_tokens:
            self.revoke_token(rt.token)

        # DB에서 삭제
        db.query(RefreshToken).filter_by(user_id=user_id).delete()
        db.commit()

💾 토큰 저장 전략​

1. 데이터베이스 저장​

from sqlalchemy import Column, Integer, String, DateTime, Boolean
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime
from passlib.context import CryptContext

Base = declarative_base()
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

class RefreshToken(Base):
    """Refresh Token 모델"""

    __tablename__ = 'refresh_tokens'

    id = Column(Integer, primary_key=True)
    user_id = Column(String(255), nullable=False, index=True)
    token_hash = Column(String(255), nullable=False, unique=True)  # 해시 저장
    jti = Column(String(255), nullable=False, unique=True)  # JWT ID
    expires_at = Column(DateTime, nullable=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    revoked = Column(Boolean, default=False)
    replaced_by_jti = Column(String(255), nullable=True)  # Rotation용

class TokenStore:
    """토큰 저장소"""

    def __init__(self, db_session):
        self.db = db_session

    def store_refresh_token(self, user_id: str, refresh_token: str, expires_at: datetime):
        """Refresh Token 저장 (해시)"""

        # JWT 디코딩 (JTI 추출)
        payload = jwt.decode(refresh_token, options={"verify_signature": False})
        jti = payload.get("jti")

        # 해시 저장
        token_hash = pwd_context.hash(refresh_token)

        rt = RefreshToken(
            user_id=user_id,
            token_hash=token_hash,
            jti=jti,
            expires_at=expires_at
        )

        self.db.add(rt)
        self.db.commit()

    def verify_refresh_token(self, refresh_token: str) -> RefreshToken:
        """Refresh Token 검증"""

        # JTI 추출
        payload = jwt.decode(refresh_token, options={"verify_signature": False})
        jti = payload.get("jti")

        # DB에서 조회
        rt = self.db.query(RefreshToken).filter_by(jti=jti).first()

        if not rt:
            raise ValueError("Token not found")

        if rt.revoked:
            raise ValueError("Token has been revoked")

        if rt.expires_at < datetime.utcnow():
            raise ValueError("Token expired")

        # 해시 검증
        if not pwd_context.verify(refresh_token, rt.token_hash):
            raise ValueError("Invalid token")

        return rt

    def revoke_refresh_token(self, jti: str):
        """Refresh Token 폐기"""
        rt = self.db.query(RefreshToken).filter_by(jti=jti).first()
        if rt:
            rt.revoked = True
            self.db.commit()

    def clean_expired_tokens(self):
        """만료된 토큰 정리"""
        self.db.query(RefreshToken).filter(
            RefreshToken.expires_at < datetime.utcnow()
        ).delete()
        self.db.commit()

2. Redis 캐시​

import redis
import json

redis_client = redis.Redis(host='localhost', port=6379, decode_responses=True)

class TokenCache:
    """Redis 토큰 캐시"""

    def __init__(self, redis_client):
        self.redis = redis_client

    def cache_access_token(self, user_id: str, access_token: str, expires_in: int):
        """Access Token 캐싱"""

        # 토큰 정보 저장
        key = f"access_token:{user_id}"
        value = json.dumps({
            "token": access_token,
            "created_at": datetime.utcnow().isoformat()
        })

        # TTL 설정
        self.redis.setex(key, expires_in, value)

    def get_cached_access_token(self, user_id: str) -> str | None:
        """캐시된 Access Token 가져오기"""

        key = f"access_token:{user_id}"
        value = self.redis.get(key)

        if value:
            data = json.loads(value)
            return data["token"]

        return None

    def invalidate_user_tokens(self, user_id: str):
        """사용자 토큰 캐시 무효화"""

        patterns = [
            f"access_token:{user_id}",
            f"refresh_token:{user_id}*"
        ]

        for pattern in patterns:
            keys = self.redis.keys(pattern)
            if keys:
                self.redis.delete(*keys)

3. HttpOnly Cookie (브라우저)​

from fastapi import Response, Request

def set_tokens_in_cookies(response: Response, tokens: dict):
    """토큰을 안전한 쿠키에 저장"""

    # Access Token
    response.set_cookie(
        key="access_token",
        value=tokens["access_token"],
        httponly=True,    # JavaScript 접근 불가
        secure=True,      # HTTPS만
        samesite='lax',   # CSRF 방지
        max_age=900       # 15분
    )

    # Refresh Token
    response.set_cookie(
        key="refresh_token",
        value=tokens["refresh_token"],
        httponly=True,
        secure=True,
        samesite='lax',
        max_age=2592000   # 30일
    )

def get_tokens_from_cookies(request: Request) -> dict:
    """쿠키에서 토큰 가져오기"""

    access_token = request.cookies.get("access_token")
    refresh_token = request.cookies.get("refresh_token")

    return {
        "access_token": access_token,
        "refresh_token": refresh_token
    }

def clear_token_cookies(response: Response):
    """토큰 쿠키 삭제"""

    response.delete_cookie("access_token")
    response.delete_cookie("refresh_token")

🔄 Refresh Token Rotation​

개념​

Refresh Token Rotation: 매번 새로운 Refresh Token 발급하여 재사용 방지

class TokenManager:
    # ...

    def refresh_with_rotation(
        self,
        refresh_token: str,
        token_store: TokenStore
    ) -> dict:
        """Refresh Token Rotation"""

        # 1. 현재 Refresh Token 검증
        rt_record = token_store.verify_refresh_token(refresh_token)

        user_id = rt_record.user_id

        # 2. 새 토큰 쌍 생성
        new_tokens = self.create_token_pair(user_id)

        # 3. 새 Refresh Token 저장
        new_rt_payload = jwt.decode(
            new_tokens["refresh_token"],
            options={"verify_signature": False}
        )
        new_jti = new_rt_payload.get("jti")

        token_store.store_refresh_token(
            user_id=user_id,
            refresh_token=new_tokens["refresh_token"],
            expires_at=datetime.fromtimestamp(new_rt_payload["exp"])
        )

        # 4. 이전 Refresh Token 폐기
        # - 즉시 폐기 OR
        # - Grace Period (재사용 탐지용)
        old_jti = rt_record.jti

        # Grace Period 방식
        self.db.query(RefreshToken).filter_by(jti=old_jti).update({
            "replaced_by_jti": new_jti
        })

        return new_tokens

    def detect_token_reuse(self, refresh_token: str, token_store: TokenStore):
        """Refresh Token 재사용 탐지"""

        payload = jwt.decode(refresh_token, options={"verify_signature": False})
        jti = payload.get("jti")

        # DB에서 조회
        rt = token_store.db.query(RefreshToken).filter_by(jti=jti).first()

        if rt and rt.replaced_by_jti:
            # 이미 교체된 토큰 → 재사용 탐지!
            # 보안 조치: 해당 사용자의 모든 토큰 폐기
            print(f"⚠️ Token reuse detected for user {rt.user_id}")

            # 토큰 체인 전체 폐기
            self.revoke_token_family(rt.user_id, rt.jti, token_store)

            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Token reuse detected. All tokens revoked."
            )

    def revoke_token_family(self, user_id: str, jti: str, token_store: TokenStore):
        """토큰 패밀리 전체 폐기"""

        # 해당 JTI와 연결된 모든 토큰 찾기
        token_chain = []
        current_jti = jti

        while current_jti:
            rt = token_store.db.query(RefreshToken).filter_by(jti=current_jti).first()
            if not rt:
                break

            token_chain.append(rt.jti)
            current_jti = rt.replaced_by_jti

        # 모두 폐기
        token_store.db.query(RefreshToken).filter(
            RefreshToken.jti.in_(token_chain)
        ).update({"revoked": True}, synchronize_session=False)

        token_store.db.commit()

from starlette.middleware.base import BaseHTTPMiddleware

class TokenRefreshMiddleware(BaseHTTPMiddleware):
    """자동 토큰 갱신 미들웨어"""

    async def dispatch(self, request, call_next):
        # Access Token 가져오기
        access_token = request.cookies.get("access_token")

        if access_token:
            # 만료 확인
            if token_manager.is_token_expired(access_token):
                # Refresh Token으로 갱신
                refresh_token = request.cookies.get("refresh_token")

                if refresh_token:
                    try:
                        new_tokens = token_manager.refresh_with_rotation(
                            refresh_token,
                            token_store
                        )

                        # 새 토큰을 쿠키에 설정
                        response = await call_next(request)
                        set_tokens_in_cookies(response, new_tokens)

                        return response

                    except:
                        # Refresh 실패 → 로그아웃
                        response = await call_next(request)
                        clear_token_cookies(response)
                        return response

        return await call_next(request)

2. 프론트엔드 (React)​

// React에서 자동 토큰 갱신

import { useEffect, useRef } from 'react';

function useTokenRefresh() {
    const refreshTimerRef = useRef(null);

    // Access Token 파싱
    function parseJWT(token) {
        try {
            const base64Url = token.split('.')[1];
            const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
            const jsonPayload = decodeURIComponent(
                atob(base64)
                    .split('')
                    .map(c => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2))
                    .join('')
            );
            return JSON.parse(jsonPayload);
        } catch {
            return null;
        }
    }

    // 토큰 갱신
    async function refreshToken() {
        try {
            const response = await fetch('/api/refresh', {
                method: 'POST',
                credentials: 'include'  // 쿠키 포함
            });

            if (response.ok) {
                const data = await response.json();
                scheduleRefresh(data.expires_in);
            } else {
                // Refresh 실패 → 로그아웃
                window.location.href = '/login';
            }
        } catch (error) {
            console.error('Token refresh failed:', error);
        }
    }

    // 갱신 스케줄링
    function scheduleRefresh(expiresIn) {
        // 만료 1분 전에 갱신
        const refreshTime = (expiresIn - 60) * 1000;

        clearTimeout(refreshTimerRef.current);
        refreshTimerRef.current = setTimeout(refreshToken, refreshTime);
    }

    // 초기화
    useEffect(() => {
        // 현재 토큰의 만료 시간 확인
        fetch('/api/me', { credentials: 'include' })
            .then(response => response.json())
            .then(data => {
                if (data.expires_in) {
                    scheduleRefresh(data.expires_in);
                }
            })
            .catch(console.error);

        // Cleanup
        return () => {
            if (refreshTimerRef.current) {
                clearTimeout(refreshTimerRef.current);
            }
        };
    }, []);

    return { refreshToken };
}

// 사용
function App() {
    useTokenRefresh();

    return <div>App Content</div>;
}

3. Axios Interceptor​

// Axios로 자동 갱신

import axios from 'axios';

let isRefreshing = false;
let failedQueue = [];

const processQueue = (error, token = null) => {
    failedQueue.forEach(prom => {
        if (error) {
            prom.reject(error);
        } else {
            prom.resolve(token);
        }
    });

    failedQueue = [];
};

// Response Interceptor
axios.interceptors.response.use(
    response => response,
    async error => {
        const originalRequest = error.config;

        // 401 에러 && 재시도 안 했으면
        if (error.response?.status === 401 && !originalRequest._retry) {
            if (isRefreshing) {
                // 이미 갱신 중이면 대기
                return new Promise((resolve, reject) => {
                    failedQueue.push({ resolve, reject });
                }).then(() => {
                    return axios(originalRequest);
                });
            }

            originalRequest._retry = true;
            isRefreshing = true;

            try {
                // Refresh Token으로 갱신
                await axios.post('/api/refresh', {}, { withCredentials: true });

                processQueue(null);
                isRefreshing = false;

                // 원래 요청 재시도
                return axios(originalRequest);
            } catch (refreshError) {
                processQueue(refreshError, null);
                isRefreshing = false;

                // Refresh 실패 → 로그아웃
                window.location.href = '/login';
                return Promise.reject(refreshError);
            }
        }

        return Promise.reject(error);
    }
);

from enum import Enum
from datetime import datetime

class TokenEvent(str, Enum):
    ISSUED = "issued"
    REFRESHED = "refreshed"
    REVOKED = "revoked"
    EXPIRED = "expired"
    REUSE_DETECTED = "reuse_detected"

class TokenAuditLog(Base):
    """토큰 감사 로그"""

    __tablename__ = 'token_audit_logs'

    id = Column(Integer, primary_key=True)
    user_id = Column(String(255), nullable=False, index=True)
    token_jti = Column(String(255), nullable=False)
    event = Column(String(50), nullable=False)
    ip_address = Column(String(45))  # IPv6 지원
    user_agent = Column(String(500))
    created_at = Column(DateTime, default=datetime.utcnow, index=True)

def log_token_event(
    user_id: str,
    token_jti: str,
    event: TokenEvent,
    request: Request
):
    """토큰 이벤트 로깅"""

    log = TokenAuditLog(
        user_id=user_id,
        token_jti=token_jti,
        event=event.value,
        ip_address=request.client.host,
        user_agent=request.headers.get("user-agent")
    )

    db.add(log)
    db.commit()
