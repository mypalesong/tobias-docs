PKCE 구현

PKCE (Proof Key for Code Exchange)를 이해하고 FastAPI에서 구현하는 방법을 배웁니다.

🎯 목표​

  • ✅ PKCE가 필요한 이유
  • ✅ PKCE 동작 원리
  • ✅ FastAPI에서 PKCE 구현
  • ✅ OAuth 2.1 표준 준수
  • ✅ 보안 강화

🤔 PKCE란?​

정의​

**PKCE (Proof Key for Code Exchange, "픽시"로 발음)**는 Authorization Code Flow에 보안 레이어를 추가하여 Authorization Code 탈취 공격을 방지하는 메커니즘입니다.

pkce = {
    "정의": "Authorization Code 보호 메커니즘",
    "표준": "RFC 7636",
    "목적": "Authorization Code Interception Attack 방지",
    "필수": "Public Clients (SPA, 모바일)",
    "권장": "모든 Clients (OAuth 2.1)"
}

# OAuth 2.1에서는 모든 클라이언트에 PKCE 필수!

⚠️ 문제: Authorization Code Interception​

공격 시나리오​

# PKCE 없는 플로우의 취약점

# 1. 정상 사용자가 로그인 시작
auth_url = (
    "https://auth.server.com/authorize"
    "?client_id=malicious-app"  # 공격자의 앱
    "&redirect_uri=malicious://callback"
    "&response_type=code"
)

# 2. 사용자가 승인
# → Authorization Code 발급: "AUTH_CODE_123"

# 3. 리다이렉트
# malicious://callback?code=AUTH_CODE_123

# 4. 문제: 악성 앱이 Code를 가로챔
# → 공격자가 Code로 Token 교환 가능 (Public Client는 Secret 없음)

# 5. 공격자가 Token 획득
token_request = {
    "code": "AUTH_CODE_123",  # 탈취한 Code
    "client_id": "malicious-app",
    "redirect_uri": "malicious://callback",
    # Client Secret 없음 (Public Client)
}

# → 공격자가 사용자의 Access Token 획득! 🚨

PKCE의 해결 방법​

# PKCE로 보호된 플로우

# 1. 정상 앱이 Code Verifier 생성 (로컬에 안전하게 보관)
code_verifier = "random-secret-string"

# 2. Code Challenge 생성
code_challenge = SHA256(code_verifier)

# 3. Authorization Request (Code Challenge 포함)
auth_url = (
    "https://auth.server.com/authorize"
    "?client_id=my-app"
    "&redirect_uri=myapp://callback"
    "&code_challenge=CHALLENGE_HASH"  # ✅ Challenge 전송
    "&code_challenge_method=S256"
)

# 4. Code 발급 (Challenge와 함께 저장됨)

# 5. 공격자가 Code 탈취해도...

# 6. Token Request (Code Verifier 필요)
token_request = {
    "code": "AUTH_CODE_123",
    "client_id": "my-app",
    "code_verifier": "???"  # ❌ 공격자는 모름!
}

# 7. Authorization Server 검증
# SHA256(code_verifier) == saved_code_challenge ?
# → 일치하지 않으면 거부! ✅

🔧 PKCE 구성 요소​

1. Code Verifier​

랜덤 문자열 (43-128자)

import secrets

def generate_code_verifier() -> str:
    """Code Verifier 생성"""
    # 43-128자 사이의 랜덤 문자열
    # A-Z, a-z, 0-9, -, ., _, ~ 사용 가능

    code_verifier = secrets.token_urlsafe(32)  # 43자
    return code_verifier

# 예시
code_verifier = "dBjftJeZ4CVP-mB92K27uhbUJU1p1r_wW1gFWFOEjXk"

# ⚠️ 중요: 클라이언트가 안전하게 보관!
# - 메모리, 세션 스토리지, 쿠키 등
# - 절대 URL에 포함하지 말 것

2. Code Challenge​

Code Verifier의 SHA-256 해시

import hashlib
import base64

def generate_code_challenge(code_verifier: str) -> str:
    """Code Challenge 생성 (SHA-256)"""

    # 1. SHA-256 해시
    digest = hashlib.sha256(code_verifier.encode()).digest()

    # 2. Base64 URL 인코딩
    code_challenge = base64.urlsafe_b64encode(digest).decode().rstrip('=')

    return code_challenge

# 예시
code_verifier = "dBjftJeZ4CVP-mB92K27uhbUJU1p1r_wW1gFWFOEjXk"
code_challenge = "E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM"

# SHA256(code_verifier) → Base64URL → code_challenge

3. Code Challenge Method​

해시 알고리즘 지정

code_challenge_methods = {
    "S256": "SHA-256 (권장)",  # ✅ 보안
    "plain": "평문 (비권장)"    # ❌ 보안 취약
}

# ✅ 항상 S256 사용
code_challenge_method = "S256"

# ❌ plain은 사용하지 마세요
# plain은 code_challenge = code_verifier (해싱 없음)
# → 중간에서 가로채면 verifier 노출!

🛠️ FastAPI에서 PKCE 구현​

완전한 구현​

from fastapi import FastAPI, Request, Response, HTTPException
from fastapi.responses import RedirectResponse
from starlette.middleware.sessions import SessionMiddleware
import secrets
import hashlib
import base64
import httpx

app = FastAPI()

# 세션 미들웨어
app.add_middleware(
    SessionMiddleware,
    secret_key=secrets.token_urlsafe(32)
)

# OAuth 설정
CLIENT_ID = "your-client-id"
CLIENT_SECRET = "your-client-secret"  # Public Client는 없음
AUTHORIZATION_ENDPOINT = "https://accounts.google.com/o/oauth2/v2/auth"
TOKEN_ENDPOINT = "https://oauth2.googleapis.com/token"
REDIRECT_URI = "http://localhost:8000/callback"

# ==================== PKCE 유틸리티 ====================

def generate_code_verifier() -> str:
    """Code Verifier 생성 (43-128자)"""
    return secrets.token_urlsafe(32)  # 43자

def generate_code_challenge(code_verifier: str) -> str:
    """Code Challenge 생성 (SHA-256)"""
    digest = hashlib.sha256(code_verifier.encode()).digest()
    code_challenge = base64.urlsafe_b64encode(digest).decode().rstrip('=')
    return code_challenge

# ==================== 라우트 ====================

@app.get('/login')
async def login(request: Request):
    """로그인 시작 (PKCE 포함)"""

    # 1. Code Verifier 생성
    code_verifier = generate_code_verifier()

    # 2. Code Challenge 생성
    code_challenge = generate_code_challenge(code_verifier)

    # 3. Code Verifier를 세션에 저장
    request.session['code_verifier'] = code_verifier

    # 4. State (CSRF 방지)
    state = secrets.token_urlsafe(32)
    request.session['oauth_state'] = state

    # 5. Authorization URL 생성
    params = {
        'client_id': CLIENT_ID,
        'redirect_uri': REDIRECT_URI,
        'response_type': 'code',
        'scope': 'openid email profile',
        'state': state,
        'code_challenge': code_challenge,        # ✅ PKCE
        'code_challenge_method': 'S256'          # ✅ SHA-256
    }

    from urllib.parse import urlencode
    auth_url = f"{AUTHORIZATION_ENDPOINT}?{urlencode(params)}"

    return RedirectResponse(auth_url)

@app.get('/callback')
async def callback(request: Request):
    """OAuth 콜백 (PKCE 검증)"""

    # 1. State 검증 (CSRF)
    state = request.query_params.get('state')
    saved_state = request.session.pop('oauth_state', None)

    if not state or state != saved_state:
        raise HTTPException(status_code=400, detail="Invalid state")

    # 2. Authorization Code 받기
    code = request.query_params.get('code')
    if not code:
        raise HTTPException(status_code=400, detail="No code")

    # 3. Code Verifier 가져오기
    code_verifier = request.session.pop('code_verifier', None)
    if not code_verifier:
        raise HTTPException(status_code=400, detail="Missing code verifier")

    # 4. Token 교환 (Code Verifier 포함)
    async with httpx.AsyncClient() as client:
        response = await client.post(
            TOKEN_ENDPOINT,
            data={
                'grant_type': 'authorization_code',
                'code': code,
                'redirect_uri': REDIRECT_URI,
                'client_id': CLIENT_ID,
                # 'client_secret': CLIENT_SECRET,  # Public Client는 생략
                'code_verifier': code_verifier  # ✅ PKCE Verifier
            }
        )

        if response.status_code != 200:
            raise HTTPException(
                status_code=response.status_code,
                detail=f"Token exchange failed: {response.text}"
            )

        token_data = response.json()

    # 5. 토큰 저장
    request.session['access_token'] = token_data['access_token']
    request.session['id_token'] = token_data.get('id_token')

    return {
        "message": "Login successful",
        "access_token": token_data['access_token'][:20] + "..."
    }

Authorization Server의 검증 로직​

# Authorization Server가 하는 일

# 1. Authorization Request 처리
@app.get('/authorize')
async def authorize(
    client_id: str,
    redirect_uri: str,
    code_challenge: str,
    code_challenge_method: str,
    state: str,
    scope: str
):
    """Authorization Endpoint"""

    # Code Challenge 저장
    authorization_code = generate_auth_code()

    save_to_db({
        'code': authorization_code,
        'code_challenge': code_challenge,  # ✅ 저장
        'code_challenge_method': code_challenge_method,
        'client_id': client_id,
        'redirect_uri': redirect_uri
    })

    # Redirect
    return RedirectResponse(f"{redirect_uri}?code={authorization_code}&state={state}")

# 2. Token Request 처리
@app.post('/token')
async def token(
    code: str,
    code_verifier: str,  # ✅ Verifier 받음
    client_id: str,
    redirect_uri: str
):
    """Token Endpoint"""

    # Code 조회
    auth_data = get_from_db(code)

    # PKCE 검증
    if auth_data['code_challenge']:
        # Code Challenge 재생성
        if auth_data['code_challenge_method'] == 'S256':
            computed_challenge = hashlib.sha256(
                code_verifier.encode()
            ).digest()
            computed_challenge = base64.urlsafe_b64encode(
                computed_challenge
            ).decode().rstrip('=')
        else:  # plain
            computed_challenge = code_verifier

        # 비교
        if computed_challenge != auth_data['code_challenge']:
            raise HTTPException(status_code=400, detail="Invalid code verifier")

    # Token 발급
    access_token = generate_access_token(auth_data)

    return {
        'access_token': access_token,
        'token_type': 'bearer',
        'expires_in': 3600
    }

🎨 Authlib와 함께 사용​

Authlib PKCE 자동 처리​

from authlib.integrations.starlette_client import OAuth
from starlette.config import Config

config = Config('.env')
oauth = OAuth(config)

# OAuth 등록
oauth.register(
    name='google',
    client_id=config('GOOGLE_CLIENT_ID'),
    client_secret=config('GOOGLE_CLIENT_SECRET'),
    server_metadata_url='https://accounts.google.com/.well-known/openid-configuration',
    client_kwargs={
        'scope': 'openid email profile',
        'code_challenge_method': 'S256'  # ✅ PKCE 활성화
    }
)

@app.get('/login')
async def login(request: Request):
    """Authlib가 PKCE 자동 처리"""
    redirect_uri = request.url_for('auth')
    return await oauth.google.authorize_redirect(request, redirect_uri)

@app.get('/auth')
async def auth(request: Request):
    """Authlib가 code_verifier 자동 전송"""
    token = await oauth.google.authorize_access_token(request)
    user = token.get('userinfo')
    return user

# Authlib는 내부적으로:
# 1. Code Verifier 생성 및 세션 저장
# 2. Code Challenge 계산 및 Authorization Request에 포함
# 3. Token Request 시 Code Verifier 자동 전송

수동 PKCE (Authlib 없이)​

from urllib.parse import urlencode

@app.get('/login/manual')
async def login_manual(request: Request):
    """수동 PKCE 구현"""

    # PKCE
    code_verifier = secrets.token_urlsafe(32)
    request.session['code_verifier'] = code_verifier

    digest = hashlib.sha256(code_verifier.encode()).digest()
    code_challenge = base64.urlsafe_b64encode(digest).decode().rstrip('=')

    # State
    state = secrets.token_urlsafe(32)
    request.session['state'] = state

    # Authorization URL
    params = {
        'client_id': CLIENT_ID,
        'redirect_uri': REDIRECT_URI,
        'response_type': 'code',
        'scope': 'openid email',
        'state': state,
        'code_challenge': code_challenge,
        'code_challenge_method': 'S256'
    }

    auth_url = f"{AUTHORIZATION_ENDPOINT}?{urlencode(params)}"

    return RedirectResponse(auth_url)

@app.get('/callback/manual')
async def callback_manual(request: Request):
    """수동 Token 교환"""

    # State 검증
    state = request.query_params.get('state')
    if state != request.session.pop('state', None):
        raise HTTPException(status_code=400, detail="Invalid state")

    # Code 받기
    code = request.query_params.get('code')

    # Verifier 가져오기
    code_verifier = request.session.pop('code_verifier')

    # Token 교환
    async with httpx.AsyncClient() as client:
        response = await client.post(
            TOKEN_ENDPOINT,
            data={
                'grant_type': 'authorization_code',
                'code': code,
                'redirect_uri': REDIRECT_URI,
                'client_id': CLIENT_ID,
                'code_verifier': code_verifier
            },
            headers={'Content-Type': 'application/x-www-form-urlencoded'}
        )

        return response.json()

📱 클라이언트별 PKCE 구현​

SPA (React)​

// React에서 PKCE
import { useEffect, useState } from 'react';

function useOAuthPKCE() {
    const [codeVerifier, setCodeVerifier] = useState(null);

    // Code Verifier 생성
    function generateCodeVerifier() {
        const array = new Uint8Array(32);
        crypto.getRandomValues(array);
        return base64URLEncode(array);
    }

    // Code Challenge 생성
    async function generateCodeChallenge(verifier) {
        const encoder = new TextEncoder();
        const data = encoder.encode(verifier);
        const digest = await crypto.subtle.digest('SHA-256', data);
        return base64URLEncode(new Uint8Array(digest));
    }

    // Base64 URL 인코딩
    function base64URLEncode(buffer) {
        return btoa(String.fromCharCode(...buffer))
            .replace(/\+/g, '-')
            .replace(/\//g, '_')
            .replace(/=+$/, '');
    }

    // 로그인
    async function login() {
        // 1. Code Verifier 생성 및 저장
        const verifier = generateCodeVerifier();
        sessionStorage.setItem('code_verifier', verifier);
        setCodeVerifier(verifier);

        // 2. Code Challenge 생성
        const challenge = await generateCodeChallenge(verifier);

        // 3. State 생성
        const state = generateCodeVerifier();
        sessionStorage.setItem('oauth_state', state);

        // 4. Authorization URL
        const params = new URLSearchParams({
            client_id: 'your-client-id',
            redirect_uri: window.location.origin + '/callback',
            response_type: 'code',
            scope: 'openid email profile',
            state: state,
            code_challenge: challenge,
            code_challenge_method: 'S256'
        });

        window.location.href = `https://auth.server.com/authorize?${params}`;
    }

    // 콜백 처리
    async function handleCallback() {
        const params = new URLSearchParams(window.location.search);
        const code = params.get('code');
        const state = params.get('state');

        // State 검증
        const savedState = sessionStorage.getItem('oauth_state');
        if (state !== savedState) {
            throw new Error('Invalid state');
        }

        // Code Verifier 가져오기
        const verifier = sessionStorage.getItem('code_verifier');

        // Token 교환
        const response = await fetch('https://auth.server.com/token', {
            method: 'POST',
            headers: {'Content-Type': 'application/x-www-form-urlencoded'},
            body: new URLSearchParams({
                grant_type: 'authorization_code',
                code: code,
                redirect_uri: window.location.origin + '/callback',
                client_id: 'your-client-id',
                code_verifier: verifier
            })
        });

        const tokens = await response.json();

        // 정리
        sessionStorage.removeItem('code_verifier');
        sessionStorage.removeItem('oauth_state');

        return tokens;
    }

    return { login, handleCallback };
}

모바일 (React Native)​

// React Native에서 PKCE
import * as Crypto from 'expo-crypto';
import * as WebBrowser from 'expo-web-browser';
import * as AuthSession from 'expo-auth-session';

async function loginWithPKCE() {
    // 1. Code Verifier 생성
    const codeVerifier = base64URLEncode(
        await Crypto.getRandomBytesAsync(32)
    );

    // 2. Code Challenge 생성
    const digest = await Crypto.digestStringAsync(
        Crypto.CryptoDigestAlgorithm.SHA256,
        codeVerifier
    );
    const codeChallenge = base64URLEncode(digest);

    // 3. Authorization Request
    const result = await AuthSession.startAsync({
        authUrl: `https://auth.server.com/authorize?${new URLSearchParams({
            client_id: 'your-client-id',
            redirect_uri: AuthSession.makeRedirectUri(),
            response_type: 'code',
            scope: 'openid email profile',
            code_challenge: codeChallenge,
            code_challenge_method: 'S256'
        })}`
    });

    if (result.type === 'success') {
        // 4. Token 교환
        const tokenResponse = await fetch('https://auth.server.com/token', {
            method: 'POST',
            headers: {'Content-Type': 'application/x-www-form-urlencoded'},
            body: new URLSearchParams({
                grant_type: 'authorization_code',
                code: result.params.code,
                redirect_uri: AuthSession.makeRedirectUri(),
                client_id: 'your-client-id',
                code_verifier: codeVerifier
            })
        });

        return await tokenResponse.json();
    }
}

🛡️ 보안 Best Practices​

1. 항상 S256 사용​

# ✅ S256 (SHA-256)
code_challenge_method = "S256"

# ❌ plain (평문)
code_challenge_method = "plain"  # 절대 사용 금지!

# plain은 code_challenge = code_verifier
# → 중간에서 가로채면 verifier 노출!

2. Code Verifier 안전하게 보관​

# ✅ 안전한 보관
# - 메모리 (변수)
# - 세션 (HttpOnly Cookie)
# - SessionStorage (브라우저)

request.session['code_verifier'] = code_verifier

# ❌ 위험한 보관
# - URL 파라미터
# - LocalStorage (XSS 위험)
# - 로그 파일

3. Code Verifier 충분한 엔트로피​

# ✅ 충분한 길이 (최소 43자)
code_verifier = secrets.token_urlsafe(32)  # 43자

# ❌ 짧은 길이
code_verifier = "short"  # 브루트 포스 공격 가능

4. OAuth 2.1 준수​

# OAuth 2.1에서는 모든 클라이언트에 PKCE 필수!

oauth_2_1_requirements = {
    "Public Clients": "PKCE 필수",
    "Confidential Clients": "PKCE 권장",
    "Implicit Flow": "금지 (Code + PKCE 사용)",
    "Password Flow": "금지"
}
