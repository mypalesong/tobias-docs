Multi-Provider 지원

여러 OAuth Provider (Google, GitHub, Microsoft 등)를 하나의 애플리케이션에 통합하는 방법을 배웁니다.

🎯 목표​

  • ✅ 다중 Provider 구조 설계
  • ✅ Provider별 설정 관리
  • ✅ 계정 연동 (Account Linking)
  • ✅ 사용자 정보 정규화
  • ✅ Provider 전환 처리

# ==================== Provider 추상화 ====================

class OAuthProvider(str, Enum):
    GOOGLE = "google"
    GITHUB = "github"
    MICROSOFT = "microsoft"

class ProviderAdapter:
    """Provider별 데이터 어댑터"""

    @staticmethod
    async def get_user_info(provider: OAuthProvider, token: dict) -> dict:
        """Provider별 사용자 정보 추출"""

        if provider == OAuthProvider.GOOGLE:
            return ProviderAdapter._google_user_info(token)
        elif provider == OAuthProvider.GITHUB:
            return await ProviderAdapter._github_user_info(token)
        elif provider == OAuthProvider.MICROSOFT:
            return ProviderAdapter._microsoft_user_info(token)
        else:
            raise ValueError(f"Unsupported provider: {provider}")

    @staticmethod
    def _google_user_info(token: dict) -> dict:
        """Google 사용자 정보 (OIDC)"""
        userinfo = token.get('userinfo')
        return {
            'provider': 'google',
            'provider_user_id': userinfo['sub'],
            'email': userinfo['email'],
            'email_verified': userinfo.get('email_verified', False),
            'name': userinfo.get('name'),
            'picture': userinfo.get('picture'),
            'given_name': userinfo.get('given_name'),
            'family_name': userinfo.get('family_name')
        }

    @staticmethod
    async def _github_user_info(token: dict) -> dict:
        """GitHub 사용자 정보 (별도 API 호출)"""
        import httpx

        access_token = token['access_token']

        # 기본 사용자 정보
        async with httpx.AsyncClient() as client:
            user_resp = await client.get(
                'https://api.github.com/user',
                headers={
                    'Authorization': f'Bearer {access_token}',
                    'Accept': 'application/vnd.github.v3+json'
                }
            )
            user_data = user_resp.json()

            # 이메일 정보
            emails_resp = await client.get(
                'https://api.github.com/user/emails',
                headers={
                    'Authorization': f'Bearer {access_token}',
                    'Accept': 'application/vnd.github.v3+json'
                }
            )
            emails = emails_resp.json()

        # Primary 이메일 찾기
        primary_email = next(
            (e['email'] for e in emails if e['primary']),
            user_data.get('email')
        )

        return {
            'provider': 'github',
            'provider_user_id': str(user_data['id']),
            'email': primary_email,
            'email_verified': next(
                (e['verified'] for e in emails if e['primary']),
                False
            ),
            'name': user_data.get('name', user_data['login']),
            'picture': user_data['avatar_url'],
            'username': user_data['login']
        }

    @staticmethod
    def _microsoft_user_info(token: dict) -> dict:
        """Microsoft 사용자 정보 (OIDC)"""
        userinfo = token.get('userinfo')
        return {
            'provider': 'microsoft',
            'provider_user_id': userinfo['sub'],
            'email': userinfo.get('email') or userinfo.get('preferred_username'),
            'email_verified': userinfo.get('email_verified', False),
            'name': userinfo.get('name'),
            'picture': None,  # Microsoft는 profile photo URL 제공 안 함
            'given_name': userinfo.get('given_name'),
            'family_name': userinfo.get('family_name')
        }

@app.get('/login/{provider}')
async def login(provider: OAuthProvider, request: Request):
    """Provider 선택 로그인"""

    # Provider 객체 가져오기
    provider_client = getattr(oauth, provider.value)

    # Redirect URI 생성
    redirect_uri = request.url_for('auth', provider=provider.value)

    # Authorization 시작
    return await provider_client.authorize_redirect(request, redirect_uri)

@app.get('/auth/{provider}')
async def auth(provider: OAuthProvider, request: Request):
    """OAuth 콜백 (모든 Provider 공통)"""

    try:
        # Provider 객체
        provider_client = getattr(oauth, provider.value)

        # Access Token 받기
        token = await provider_client.authorize_access_token(request)

        # 사용자 정보 가져오기 (Provider별 어댑터)
        user_info = await ProviderAdapter.get_user_info(provider, token)

        # DB에서 사용자 찾기 또는 생성
        user = await find_or_create_user(user_info)

        # 세션에 저장
        request.session['user_id'] = user.id
        request.session['provider'] = provider.value

        return RedirectResponse(url='/')

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

async def find_or_create_user(user_info: dict):
    """사용자 찾기 또는 생성"""

    provider = user_info['provider']
    provider_user_id = user_info['provider_user_id']

    # Provider 계정으로 찾기
    provider_account = db.query(ProviderAccount).filter_by(
        provider=provider,
        provider_user_id=provider_user_id
    ).first()

    if provider_account:
        # 기존 사용자
        return provider_account.user

    # 이메일로 찾기 (계정 연동)
    email = user_info.get('email')
    user = None

    if email:
        user = db.query(User).filter_by(email=email).first()

    if not user:
        # 새 사용자 생성
        user = User(
            email=email,
            name=user_info.get('name'),
            picture=user_info.get('picture'),
            email_verified=user_info.get('email_verified', False)
        )
        db.add(user)
        db.flush()

    # Provider 계정 연결
    provider_account = ProviderAccount(
        user_id=user.id,
        provider=provider,
        provider_user_id=provider_user_id,
        provider_data=user_info
    )
    db.add(provider_account)
    db.commit()

    return user

SQLAlchemy 모델​

from sqlalchemy import Column, Integer, String, Boolean, DateTime, JSON, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class User(Base):
    """사용자 모델"""

    __tablename__ = 'users'

    id = Column(Integer, primary_key=True)
    email = Column(String(255), unique=True, nullable=False, index=True)
    name = Column(String(255))
    picture = Column(String(500))
    email_verified = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    provider_accounts = relationship('ProviderAccount', back_populates='user')

class ProviderAccount(Base):
    """Provider 계정 연결"""

    __tablename__ = 'provider_accounts'

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'), nullable=False)
    provider = Column(String(50), nullable=False)  # google, github, etc.
    provider_user_id = Column(String(255), nullable=False)  # Provider의 사용자 ID
    provider_data = Column(JSON)  # Provider별 추가 정보
    access_token = Column(String(500))  # 암호화 권장
    refresh_token = Column(String(500))  # 암호화 권장
    token_expires_at = Column(DateTime)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    user = relationship('User', back_populates='provider_accounts')

    # Unique constraint
    __table_args__ = (
        UniqueConstraint('provider', 'provider_user_id', name='uq_provider_account'),
    )

async def find_or_create_user_with_linking(user_info: dict):
    """이메일 기반 자동 계정 연동"""

    provider = user_info['provider']
    provider_user_id = user_info['provider_user_id']
    email = user_info.get('email')

    # 1. Provider 계정으로 찾기
    provider_account = db.query(ProviderAccount).filter_by(
        provider=provider,
        provider_user_id=provider_user_id
    ).first()

    if provider_account:
        return provider_account.user

    # 2. 이메일로 기존 사용자 찾기
    user = None
    if email:
        user = db.query(User).filter_by(email=email).first()

        if user:
            # 이메일이 검증된 경우에만 자동 연동
            if user_info.get('email_verified'):
                print(f"🔗 Auto-linking {provider} account to existing user {user.id}")
            else:
                # 이메일 미검증 → 사용자 확인 필요
                raise HTTPException(
                    status_code=400,
                    detail="Email not verified. Please verify your email first."
                )

    if not user:
        # 3. 새 사용자 생성
        user = User(
            email=email,
            name=user_info.get('name'),
            picture=user_info.get('picture'),
            email_verified=user_info.get('email_verified', False)
        )
        db.add(user)
        db.flush()

    # 4. Provider 계정 연결
    provider_account = ProviderAccount(
        user_id=user.id,
        provider=provider,
        provider_user_id=provider_user_id,
        provider_data=user_info
    )
    db.add(provider_account)
    db.commit()

    return user

수동 계정 연동​

@app.post('/link-account/{provider}')
async def link_account(
    provider: OAuthProvider,
    request: Request,
    current_user: User = Depends(get_current_user)
):
    """기존 계정에 Provider 추가"""

    # 이미 연결되어 있는지 확인
    existing = db.query(ProviderAccount).filter_by(
        user_id=current_user.id,
        provider=provider.value
    ).first()

    if existing:
        raise HTTPException(
            status_code=400,
            detail=f"{provider.value} already linked"
        )

    # OAuth 플로우 시작
    provider_client = getattr(oauth, provider.value)
    redirect_uri = request.url_for('link_callback', provider=provider.value)

    # 세션에 연동 모드 표시
    request.session['linking_mode'] = True

    return await provider_client.authorize_redirect(request, redirect_uri)

@app.get('/link-callback/{provider}')
async def link_callback(
    provider: OAuthProvider,
    request: Request,
    current_user: User = Depends(get_current_user)
):
    """계정 연동 콜백"""

    # 연동 모드 확인
    if not request.session.pop('linking_mode', False):
        raise HTTPException(status_code=400, detail="Invalid request")

    try:
        # Token 받기
        provider_client = getattr(oauth, provider.value)
        token = await provider_client.authorize_access_token(request)

        # 사용자 정보
        user_info = await ProviderAdapter.get_user_info(provider, token)

        # 이미 다른 사용자에게 연결되어 있는지 확인
        existing_account = db.query(ProviderAccount).filter_by(
            provider=provider.value,
            provider_user_id=user_info['provider_user_id']
        ).first()

        if existing_account and existing_account.user_id != current_user.id:
            raise HTTPException(
                status_code=400,
                detail="This account is already linked to another user"
            )

        # Provider 계정 연결
        provider_account = ProviderAccount(
            user_id=current_user.id,
            provider=provider.value,
            provider_user_id=user_info['provider_user_id'],
            provider_data=user_info
        )
        db.add(provider_account)
        db.commit()

        return {
            "message": f"{provider.value} account linked successfully",
            "user_id": current_user.id
        }

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.delete('/unlink-account/{provider}')
async def unlink_account(
    provider: OAuthProvider,
    current_user: User = Depends(get_current_user)
):
    """Provider 계정 연결 해제"""

    # 최소 1개 계정 유지
    linked_accounts = db.query(ProviderAccount).filter_by(
        user_id=current_user.id
    ).count()

    if linked_accounts <= 1:
        raise HTTPException(
            status_code=400,
            detail="Cannot unlink last account"
        )

    # 연결 해제
    db.query(ProviderAccount).filter_by(
        user_id=current_user.id,
        provider=provider.value
    ).delete()
    db.commit()

    return {"message": f"{provider.value} account unlinked"}

3. 토큰 암호화​

from cryptography.fernet import Fernet

# 암호화 키
ENCRYPTION_KEY = Fernet.generate_key()
cipher = Fernet(ENCRYPTION_KEY)

def encrypt_token(token: str) -> str:
    """토큰 암호화"""
    return cipher.encrypt(token.encode()).decode()

def decrypt_token(encrypted_token: str) -> str:
    """토큰 복호화"""
    return cipher.decrypt(encrypted_token.encode()).decode()

# Provider 계정 저장 시
provider_account.access_token = encrypt_token(token['access_token'])
provider_account.refresh_token = encrypt_token(token.get('refresh_token', ''))
