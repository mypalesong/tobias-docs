class OAuthProvider(str, Enum):
    GOOGLE = "google"
    GITHUB = "github"
    MICROSOFT = "microsoft"

class ProviderAdapter:
    """Providerë³„ ë°ì´í„° ì–´ëŒ‘í„°"""

    @staticmethod
    async def get_user_info(provider: OAuthProvider, token: dict) -> dict:
        """Providerë³„ ì‚¬ìš©ì ì •ë³´ ì¶”ì¶œ"""

        if provider == OAuthProvider.GOOGLE:
            return ProviderAdapter._google_user_info(token)
        elif provider == OAuthProvider.GITHUB:
            return await ProviderAdapter._github_user_info(token)
        elif provider == OAuthProvider.MICROSOFT:
            return ProviderAdapter._microsoft_user_info(token)
        else:
            raise ValueError(f"Unsupported provider: {provider}")

    @staticmethod
    def _google_user_info(token: dict) -> dict:
        """Google ì‚¬ìš©ì ì •ë³´ (OIDC)"""
        userinfo = token.get('userinfo')
        return {
            'provider': 'google',
            'provider_user_id': userinfo['sub'],
            'email': userinfo['email'],
            'email_verified': userinfo.get('email_verified', False),
            'name': userinfo.get('name'),
            'picture': userinfo.get('picture'),
            'given_name': userinfo.get('given_name'),
            'family_name': userinfo.get('family_name')
        }

    @staticmethod
    async def _github_user_info(token: dict) -> dict:
        """GitHub ì‚¬ìš©ì ì •ë³´ (ë³„ë„ API í˜¸ì¶œ)"""
        import httpx

        access_token = token['access_token']

        # ê¸°ë³¸ ì‚¬ìš©ì ì •ë³´
        async with httpx.AsyncClient() as client:
            user_resp = await client.get(
                'https://api.github.com/user',
                headers={
                    'Authorization': f'Bearer {access_token}',
                    'Accept': 'application/vnd.github.v3+json'
                }
            )
            user_data = user_resp.json()

            # ì´ë©”ì¼ ì •ë³´
            emails_resp = await client.get(
                'https://api.github.com/user/emails',
                headers={
                    'Authorization': f'Bearer {access_token}',
                    'Accept': 'application/vnd.github.v3+json'
                }
            )
            emails = emails_resp.json()

        # Primary ì´ë©”ì¼ ì°¾ê¸°
        primary_email = next(
            (e['email'] for e in emails if e['primary']),
            user_data.get('email')
        )

        return {
            'provider': 'github',
            'provider_user_id': str(user_data['id']),
            'email': primary_email,
            'email_verified': next(
                (e['verified'] for e in emails if e['primary']),
                False
            ),
            'name': user_data.get('name', user_data['login']),
            'picture': user_data['avatar_url'],
            'username': user_data['login']
        }

    @staticmethod
    def _microsoft_user_info(token: dict) -> dict:
        """Microsoft ì‚¬ìš©ì ì •ë³´ (OIDC)"""
        userinfo = token.get('userinfo')
        return {
            'provider': 'microsoft',
            'provider_user_id': userinfo['sub'],
            'email': userinfo.get('email') or userinfo.get('preferred_username'),
            'email_verified': userinfo.get('email_verified', False),
            'name': userinfo.get('name'),
            'picture': None,  # MicrosoftëŠ” profile photo URL ì œê³µ ì•ˆ í•¨
            'given_name': userinfo.get('given_name'),
            'family_name': userinfo.get('family_name')
        }

        user_id=current_user.id,

        raise HTTPException(

            detail=f"{provider.value} already linked"
        )

    # OAuth í”Œë¡œìš° ì‹œì‘
    provider_client = getattr(oauth, provider.value)
    redirect_uri = request.url_for('link_callback', provider=provider.value)

    # ì„¸ì…˜ì— ì—°ë™ ëª¨ë“œ í‘œì‹œ
    request.session['linking_mode'] = True

    return await provider_client.authorize_redirect(request, redirect_uri)

@app.get('/link-callback/{provider}')
async def link_callback(
    provider: OAuthProvider,
    request: Request,
    current_user: User = Depends(get_current_user)
):
    """ê³„ì • ì—°ë™ ì½œë°±"""

    # ì—°ë™ ëª¨ë“œ í™•ì¸
    if not request.session.pop('linking_mode', False):
        raise HTTPException(status_code=400, detail="Invalid request")

    try:
        # Token ë°›ê¸°
        provider_client = getattr(oauth, provider.value)
        token = await provider_client.authorize_access_token(request)

        # ì‚¬ìš©ì ì •ë³´
        user_info = await ProviderAdapter.get_user_info(provider, token)

        # ì´ë¯¸ ë‹¤ë¥¸ ì‚¬ìš©ìì—ê²Œ ì—°ê²°ë˜ì–´ ìˆëŠ”ì§€ í™•ì¸
        existing_account = db.query(ProviderAccount).filter_by(
            provider=provider.value,
            provider_user_id=user_info['provider_user_id']
        ).first()

        if existing_account and existing_account.user_id != current_user.id:
            raise HTTPException(
                status_code=400,
                detail="This account is already linked to another user"
            )

        # Provider ê³„ì • ì—°ê²°
        provider_account = ProviderAccount(
            user_id=current_user.id,
            provider=provider.value,
            provider_user_id=user_info['provider_user_id'],
            provider_data=user_info
        )
        db.add(provider_account)
        db.commit()

        return {
            "message": f"{provider.value} account linked successfully",
            "user_id": current_user.id
        }

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.delete('/unlink-account/{provider}')

    provider: OAuthProvider,
    current_user: User = Depends(get_current_user)

    """Provider ê³„ì • ì—°ê²° í•´ì œ"""
