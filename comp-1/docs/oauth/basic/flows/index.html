Authorization Flows

OAuth 2.0의 다양한 인가 플로우를 이해하고 적절한 상황에서 선택하는 방법을 배웁니다.

🎯 목표​

  • ✅ 각 플로우의 용도 이해
  • ✅ 플로우별 보안 특성 파악
  • ✅ 상황에 맞는 플로우 선택
  • ✅ FastAPI로 구현
  • ✅ 보안 Best Practices

🔄 플로우 개요​

OAuth 2.0는 다양한 클라이언트 유형과 사용 사례를 지원하기 위해 여러 플로우를 제공합니다.

1️⃣ Authorization Code Flow​

정의​

가장 안전하고 일반적인 플로우입니다. 서버 측 애플리케이션에 최적화되어 있습니다.

사용 사례​

  • ✅ 전통적인 웹 애플리케이션
  • ✅ 백엔드 서버가 있는 앱
  • ✅ Client Secret을 안전하게 보관 가능

FastAPI 구현​

from fastapi import FastAPI, Request, HTTPException
from authlib.integrations.starlette_client import OAuth
from starlette.config import Config
from starlette.middleware.sessions import SessionMiddleware
import secrets

app = FastAPI()

# 세션 미들웨어 (state 저장용)
app.add_middleware(SessionMiddleware, secret_key=secrets.token_urlsafe(32))

# OAuth 설정
config = Config('.env')
oauth = OAuth(config)

oauth.register(
    name='google',
    client_id=config('GOOGLE_CLIENT_ID'),
    client_secret=config('GOOGLE_CLIENT_SECRET'),
    server_metadata_url='https://accounts.google.com/.well-known/openid-configuration',
    client_kwargs={'scope': 'openid email profile'}
)

@app.get('/login')
async def login(request: Request):
    """Step 1 & 2: Authorization Request"""
    # State parameter for CSRF protection
    redirect_uri = request.url_for('callback')

    # Redirect to authorization server
    return await oauth.google.authorize_redirect(request, redirect_uri)

@app.get('/callback')
async def callback(request: Request):
    """Step 4, 5, 6: Handle callback and exchange code for token"""
    try:
        # Exchange authorization code for access token
        token = await oauth.google.authorize_access_token(request)

        # Get user info
        user = token.get('userinfo')

        # Store in session
        request.session['user'] = {
            'sub': user['sub'],
            'email': user['email'],
            'name': user['name']
        }

        return {"message": "Login successful", "user": user}

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.get('/api/protected')
async def protected_route(request: Request):
    """Step 7: Use access token to access protected resource"""
    user = request.session.get('user')

    if not user:
        raise HTTPException(status_code=401, detail="Not authenticated")

    return {"message": "Protected data", "user": user}

2️⃣ Authorization Code Flow with PKCE​

정의​

**PKCE (Proof Key for Code Exchange)**를 추가하여 보안을 강화한 플로우입니다.

PKCE란?​

# PKCE는 Authorization Code 탈취 공격을 방지합니다

# 문제: Authorization Code Interception Attack
# 공격자가 Authorization Code를 가로채면?
# → Client Secret 없어도 Token 교환 가능 (Public Client의 경우)

# 해결: PKCE
# → Code Verifier로 Code를 보호
# → 공격자가 Code를 가로채도 Token 교환 불가

사용 사례​

  • ✅ 모든 클라이언트 (OAuth 2.1 표준)
  • ✅ SPA (React, Vue, Angular)
  • ✅ 모바일 앱 (iOS, Android)
  • ✅ Public Clients (Client Secret 없는 앱)

PKCE 플로우​

import hashlib
import base64
import secrets

# Step 1: Code Verifier 생성 (클라이언트)
code_verifier = secrets.token_urlsafe(32)
# 예: "dBjftJeZ4CVP-mB92K27uhbUJU1p1r_wW1gFWFOEjXk"

# Step 2: Code Challenge 생성
code_challenge = base64.urlsafe_b64encode(
    hashlib.sha256(code_verifier.encode()).digest()
).decode().rstrip('=')
# 예: "E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM"

# Step 3: Authorization Request (code_challenge 포함)
auth_url = (
    "https://accounts.google.com/o/oauth2/v2/auth"
    f"?client_id={CLIENT_ID}"
    f"&redirect_uri={REDIRECT_URI}"
    "&response_type=code"
    "&scope=openid email profile"
    f"&code_challenge={code_challenge}"
    "&code_challenge_method=S256"  # SHA-256
)

# Step 4: Token Request (code_verifier 포함)
token_data = {
    "code": authorization_code,
    "client_id": CLIENT_ID,
    "redirect_uri": REDIRECT_URI,
    "grant_type": "authorization_code",
    "code_verifier": code_verifier  # 원본 verifier
}

# Auth Server는 code_challenge와 code_verifier를 검증
# SHA256(code_verifier) == code_challenge ?

FastAPI 구현​

from fastapi import FastAPI, Request, Response
import httpx
import hashlib
import base64
import secrets

app = FastAPI()

@app.get('/login')
async def login(response: Response):
    """PKCE with Authorization Code Flow"""

    # 1. Code Verifier 생성
    code_verifier = secrets.token_urlsafe(32)

    # 2. Code Challenge 생성
    code_challenge = base64.urlsafe_b64encode(
        hashlib.sha256(code_verifier.encode()).digest()
    ).decode().rstrip('=')

    # 3. Code Verifier를 쿠키에 저장 (나중에 사용)
    response.set_cookie(
        key="code_verifier",
        value=code_verifier,
        httponly=True,
        secure=True,
        samesite='lax'
    )

    # 4. Authorization URL 생성 (code_challenge 포함)
    auth_url = (
        "https://accounts.google.com/o/oauth2/v2/auth"
        f"?client_id={CLIENT_ID}"
        f"&redirect_uri={REDIRECT_URI}"
        "&response_type=code"
        "&scope=openid email profile"
        f"&code_challenge={code_challenge}"
        "&code_challenge_method=S256"
    )

    return {"auth_url": auth_url}

@app.get('/callback')
async def callback(code: str, request: Request):
    """Token exchange with PKCE"""

    # 1. Code Verifier 가져오기
    code_verifier = request.cookies.get('code_verifier')

    if not code_verifier:
        raise HTTPException(status_code=400, detail="Missing code verifier")

    # 2. Token 요청 (code_verifier 포함)
    async with httpx.AsyncClient() as client:
        response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'code': code,
                'client_id': CLIENT_ID,
                'redirect_uri': REDIRECT_URI,
                'grant_type': 'authorization_code',
                'code_verifier': code_verifier  # PKCE
            }
        )

    token = response.json()

    return {"access_token": token['access_token']}

from functools import lru_cache
from datetime import datetime, timedelta

class TokenManager:
    def __init__(self):
        self.token = None
        self.expires_at = None

    async def get_token(self):
        """캐싱된 토큰 반환 (만료 시 갱신)"""

        if self.token and self.expires_at > datetime.now():
            return self.token

        # Token 갱신
        async with httpx.AsyncClient() as client:
            response = await client.post(
                'https://oauth2.googleapis.com/token',
                data={
                    'grant_type': 'client_credentials',
                    'client_id': CLIENT_ID,
                    'client_secret': CLIENT_SECRET
                }
            )

        data = response.json()
        self.token = data['access_token']
        self.expires_at = datetime.now() + timedelta(seconds=data['expires_in'])

        return self.token

token_manager = TokenManager()

@app.get('/api/cached-service')
async def cached_service():
    """캐싱된 토큰 사용"""
    token = await token_manager.get_token()

    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://api.example.com/data',
            headers={'Authorization': f'Bearer {token}'}
        )

    return response.json()

    async with httpx.AsyncClient() as client:
        response = await client.post(
            'https://oauth2.googleapis.com/device/code',
            data={
                'client_id': CLIENT_ID,
                'scope': 'email profile'
            }
        )

    data = response.json()

    # {
    #   "device_code": "4/4-GMMhmHCXhWEzkobqIHGG_EnNYYsAkukHspeYUk9E8",
    #   "user_code": "GQVQ-JKEC",
    #   "verification_url": "https://www.google.com/device",
    #   "expires_in": 1800,
    #   "interval": 5
    # }

    # 2. 사용자에게 안내
    print(f"방문: {data['verification_url']}")
    print(f"코드 입력: {data['user_code']}")

    # 3. Token 폴링

        device_code=data['device_code'],

    return {"access_token": token}

async def poll_for_token(device_code: str, interval: int):
    """주기적으로 Token 요청 (폴링)"""

    while True:
        await asyncio.sleep(interval)

        async with httpx.AsyncClient() as client:
            response = await client.post(
                'https://oauth2.googleapis.com/token',
                data={
                    'client_id': CLIENT_ID,
                    'client_secret': CLIENT_SECRET,
                    'device_code': device_code,
                    'grant_type': 'urn:ietf:params:oauth:grant-type:device_code'
                }
            )

        data = response.json()

        # 에러 처리
        if 'error' in data:
            if data['error'] == 'authorization_pending':
                # 아직 승인 안 됨, 계속 폴링
                continue
            elif data['error'] == 'slow_down':
                # 폴링 간격 증가
                interval += 5
                continue
            elif data['error'] == 'expired_token':
                # Device Code 만료
                raise Exception("Device code expired")
            else:
                raise Exception(f"Error: {data['error']}")

        # Token 받음!
        return data['access_token']

5️⃣ Refresh Token Flow​

정의​

만료된 Access Token을 갱신하는 플로우입니다.

사용 사례​

  • ✅ Access Token 만료 시
  • ✅ 사용자 재로그인 없이 계속 사용
  • ✅ 장기 세션 유지

FastAPI 구현​

from fastapi import FastAPI, HTTPException, Depends
from datetime import datetime, timedelta
from typing import Optional

app = FastAPI()

class TokenStore:
    """토큰 저장소 (실제로는 DB 사용)"""

    def __init__(self):
        self.tokens = {}

    def save_token(self, user_id: str, access_token: str,
                   refresh_token: str, expires_in: int):
        self.tokens[user_id] = {
            'access_token': access_token,
            'refresh_token': refresh_token,
            'expires_at': datetime.now() + timedelta(seconds=expires_in)
        }

    def get_token(self, user_id: str) -> Optional[dict]:
        return self.tokens.get(user_id)

    def is_expired(self, user_id: str) -> bool:
        token_data = self.tokens.get(user_id)
        if not token_data:
            return True
        return datetime.now() >= token_data['expires_at']

token_store = TokenStore()

async def refresh_access_token(refresh_token: str) -> dict:
    """Refresh Token으로 새 Access Token 발급"""

    async with httpx.AsyncClient() as client:
        response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'client_id': CLIENT_ID,
                'client_secret': CLIENT_SECRET,
                'refresh_token': refresh_token,
                'grant_type': 'refresh_token'
            }
        )

    if response.status_code != 200:
        raise HTTPException(status_code=401, detail="Refresh token expired")

    return response.json()

async def get_valid_token(user_id: str) -> str:
    """유효한 Access Token 반환 (필요 시 자동 갱신)"""

    token_data = token_store.get_token(user_id)

    if not token_data:
        raise HTTPException(status_code=401, detail="Not authenticated")

    # Token이 만료되었는지 확인
    if token_store.is_expired(user_id):
        # Refresh Token으로 갱신
        new_token = await refresh_access_token(token_data['refresh_token'])

        # 새 토큰 저장
        token_store.save_token(
            user_id=user_id,
            access_token=new_token['access_token'],
            refresh_token=token_data['refresh_token'],  # 기존 Refresh Token 유지
            expires_in=new_token['expires_in']
        )

        return new_token['access_token']

    return token_data['access_token']

@app.get('/api/auto-refresh')
async def auto_refresh_example(user_id: str):
    """자동 토큰 갱신 예제"""

    # 유효한 Token 가져오기 (만료 시 자동 갱신)
    access_token = await get_valid_token(user_id)

    # API 호출
    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://www.googleapis.com/oauth2/v2/userinfo',
            headers={'Authorization': f'Bearer {access_token}'}
        )

    return response.json()

# Middleware로 자동 갱신
from starlette.middleware.base import BaseHTTPMiddleware

class TokenRefreshMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        user_id = request.session.get('user_id')

        if user_id:
            try:
                # Token 갱신 확인
                await get_valid_token(user_id)
            except HTTPException:
                # Refresh Token도 만료됨 → 재로그인 필요
                request.session.clear()

        response = await call_next(request)
        return response

app.add_middleware(TokenRefreshMiddleware)

async def password_flow(username: str, password: str):
    """사용자 비밀번호로 직접 로그인"""
