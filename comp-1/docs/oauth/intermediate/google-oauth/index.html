🔧 기본 구현​

전체 코드 (main.py)​

from fastapi import FastAPI, Request, HTTPException
from fastapi.responses import RedirectResponse, HTMLResponse
from starlette.middleware.sessions import SessionMiddleware
from authlib.integrations.starlette_client import OAuth
from starlette.config import Config
import httpx

# 환경변수 로드
config = Config('.env')

# FastAPI 앱
app = FastAPI(title="Google OAuth Example")

# 세션 미들웨어
app.add_middleware(
    SessionMiddleware,
    secret_key=config('SECRET_KEY', default='super-secret-key')
)

# OAuth 설정
oauth = OAuth(config)

oauth.register(
    name='google',
    client_id=config('GOOGLE_CLIENT_ID'),
    client_secret=config('GOOGLE_CLIENT_SECRET'),
    server_metadata_url='https://accounts.google.com/.well-known/openid-configuration',
    client_kwargs={
        'scope': 'openid email profile'
    }
)

# ==================== 라우트 ====================

@app.get('/', response_class=HTMLResponse)
async def home(request: Request):
    """홈페이지"""
    user = request.session.get('user')

    if user:
        return f"""
        <html>
            <body>
                <h1>환영합니다, {user['name']}님!</h1>
                <img src="{user['picture']}" alt="프로필" width="100">
                <p>이메일: {user['email']}</p>
                <a href="/logout">로그아웃</a>
                <br><br>
                <a href="/profile">내 프로필</a>
                <a href="/api/calendar">캘린더 이벤트</a>
            </body>
        </html>
        """
    else:
        return """
        <html>
            <body>
                <h1>Google OAuth 예제</h1>
                <a href="/login">Google로 로그인</a>
            </body>
        </html>
        """

@app.get('/login')
async def login(request: Request):
    """Google 로그인 시작"""
    redirect_uri = request.url_for('auth')
    return await oauth.google.authorize_redirect(request, redirect_uri)

@app.get('/auth')
async def auth(request: Request):
    """Google OAuth 콜백"""
    try:
        # Access Token 받기
        token = await oauth.google.authorize_access_token(request)

        # 사용자 정보 가져오기
        user = token.get('userinfo')

        if not user:
            raise HTTPException(status_code=400, detail="Failed to get user info")

        # 세션에 저장
        request.session['user'] = {
            'sub': user['sub'],
            'email': user['email'],
            'name': user['name'],
            'picture': user.get('picture'),
            'email_verified': user.get('email_verified')
        }

        # 토큰 저장
        request.session['tokens'] = {
            'access_token': token['access_token'],
            'refresh_token': token.get('refresh_token'),
            'id_token': token.get('id_token'),
            'expires_in': token.get('expires_in')
        }

        return RedirectResponse(url='/')

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.get('/logout')
async def logout(request: Request):
    """로그아웃"""
    request.session.clear()
    return RedirectResponse(url='/')

@app.get('/profile')
async def profile(request: Request):
    """사용자 프로필"""
    user = request.session.get('user')

    if not user:
        raise HTTPException(status_code=401, detail="Not authenticated")

    return user

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

1. Discovery Document​

# Discovery Document URL
discovery_url = "https://accounts.google.com/.well-known/openid-configuration"

# Authlib는 이 URL로 모든 엔드포인트를 자동 설정
oauth.register(
    name='google',
    server_metadata_url=discovery_url,  # 자동 설정!
    # authorization_endpoint, token_endpoint 등 자동으로 가져옴
)

# 수동 설정 (권장 안 함)

🎨 Google API 활용​

Google Calendar API​

@app.get('/api/calendar')
async def get_calendar_events(request: Request):
    """Google Calendar 이벤트 가져오기"""

    # 인증 확인
    user = request.session.get('user')
    if not user:
        raise HTTPException(status_code=401, detail="Not authenticated")

    # Access Token 가져오기
    tokens = request.session.get('tokens')
    if not tokens:
        raise HTTPException(status_code=401, detail="No tokens")

    access_token = tokens['access_token']

    # Calendar API 호출
    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://www.googleapis.com/calendar/v3/calendars/primary/events',
            headers={'Authorization': f'Bearer {access_token}'},
            params={
                'maxResults': 10,
                'orderBy': 'startTime',
                'singleEvents': True,
                'timeMin': datetime.now().isoformat() + 'Z'
            }
        )

        if response.status_code != 200:
            raise HTTPException(
                status_code=response.status_code,
                detail="Failed to fetch calendar events"
            )

        return response.json()

자동 토큰 갱신​

from datetime import datetime, timedelta

async def refresh_access_token(request: Request):
    """Access Token 갱신"""

    tokens = request.session.get('tokens')
    if not tokens or 'refresh_token' not in tokens:
        raise HTTPException(status_code=401, detail="No refresh token")

    async with httpx.AsyncClient() as client:
        response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'grant_type': 'refresh_token',
                'refresh_token': tokens['refresh_token'],
                'client_id': GOOGLE_CLIENT_ID,
                'client_secret': GOOGLE_CLIENT_SECRET
            }
        )

        if response.status_code != 200:
            raise HTTPException(status_code=401, detail="Failed to refresh token")

        new_token = response.json()

        # 세션 업데이트
        tokens['access_token'] = new_token['access_token']
        tokens['expires_in'] = new_token['expires_in']
        request.session['tokens'] = tokens

        return new_token['access_token']

🛡️ 보안 강화​

1. PKCE 적용​

import hashlib
import base64
import secrets

@app.get('/login')
async def login(request: Request):
    """PKCE와 함께 로그인"""

    # Code Verifier 생성
    code_verifier = secrets.token_urlsafe(32)
    request.session['code_verifier'] = code_verifier

    # Code Challenge 생성
    code_challenge = base64.urlsafe_b64encode(
        hashlib.sha256(code_verifier.encode()).digest()
    ).decode().rstrip('=')

    # Redirect URI
    redirect_uri = request.url_for('auth')

    # PKCE 파라미터 포함
    return await oauth.google.authorize_redirect(
        request,
        redirect_uri,
        code_challenge=code_challenge,
        code_challenge_method='S256'
    )

@app.get('/auth')
async def auth(request: Request):
    """PKCE 검증과 함께 콜백"""

    code_verifier = request.session.pop('code_verifier', None)
    if not code_verifier:
        raise HTTPException(status_code=400, detail="Missing code verifier")

    # Token 요청 (code_verifier 포함)
    token = await oauth.google.authorize_access_token(
        request,
        code_verifier=code_verifier
    )

    # ... 나머지 처리

3. ID Token 검증​

from jose import jwt, JWTError

async def verify_id_token(id_token: str, client_id: str):
    """ID Token 검증"""

    try:
        # Google 공개 키 가져오기
        async with httpx.AsyncClient() as client:
            response = await client.get(
                'https://www.googleapis.com/oauth2/v3/certs'
            )
            jwks = response.json()

        # JWT 검증
        payload = jwt.decode(
            id_token,
            jwks,
            algorithms=['RS256'],
            audience=client_id,
            issuer='https://accounts.google.com'
        )

        # 추가 검증
        assert payload['aud'] == client_id
        assert payload['iss'] == 'https://accounts.google.com'
        assert payload['exp'] > datetime.now().timestamp()

        return payload

    except JWTError as e:
        raise HTTPException(status_code=401, detail=f"Invalid ID token: {e}")

# Authlib는 자동으로 ID Token 검증 수행

📊 사용자 DB 저장​

SQLAlchemy 연동​

from sqlalchemy import create_engine, Column, Integer, String, DateTime, Boolean
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from datetime import datetime

Base = declarative_base()

class User(Base):
    __tablename__ = 'users'

    id = Column(Integer, primary_key=True)
    google_sub = Column(String(255), unique=True, nullable=False)
    email = Column(String(255), nullable=False)
    name = Column(String(255))
    picture = Column(String(500))
    email_verified = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    last_login = Column(DateTime, default=datetime.utcnow)

# DB 연결
engine = create_engine('sqlite:///./users.db')
Base.metadata.create_all(engine)
SessionLocal = sessionmaker(bind=engine)

@app.get('/auth')
async def auth(request: Request):
    """OAuth 콜백 + DB 저장"""

    token = await oauth.google.authorize_access_token(request)
    userinfo = token.get('userinfo')

    # DB 세션
    db = SessionLocal()

    try:
        # 사용자 찾기 또는 생성
        user = db.query(User).filter_by(google_sub=userinfo['sub']).first()

        if user:
            # 기존 사용자 업데이트
            user.email = userinfo['email']
            user.name = userinfo['name']
            user.picture = userinfo.get('picture')
            user.last_login = datetime.utcnow()
        else:
            # 새 사용자 생성
            user = User(
                google_sub=userinfo['sub'],
                email=userinfo['email'],
                name=userinfo['name'],
                picture=userinfo.get('picture'),
                email_verified=userinfo.get('email_verified', False)
            )
            db.add(user)

        db.commit()

        # 세션에 저장
        request.session['user_id'] = user.id

        return RedirectResponse(url='/')

    finally:
        db.close()
