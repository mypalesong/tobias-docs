import hmac
import hashlib
import base64

def create_signature(header_b64: str, payload_b64: str, secret: str) -> str:
    """HMAC SHA-256 ì„œëª… ìƒì„±"""

    message = f"{header_b64}.{payload_b64}".encode()

    signature = hmac.new(
        secret.encode(),
        message,
        hashlib.sha256
    ).digest()

    signature_b64 = base64.urlsafe_b64encode(signature).decode().rstrip('=')

    return signature_b64

async def verify_id_token(id_token: str, client_id: str):
    """Google ID Token ê²€ì¦"""

    try:
        # 1. ê³µê°œ í‚¤ ê°€ì ¸ì˜¤ê¸°
        jwks = await get_google_public_keys()

        # 2. Headerì—ì„œ kid ì¶”ì¶œ

            (k for k in jwks['keys'] if k['kid'] == kid),
            None

        if not key:
            raise ValueError("Public key not found")

        # 4. JWT ê²€ì¦ ë° ë””ì½”ë”©

            id_token,

        assert payload['iss'] == 'https://accounts.google.com', "Invalid issuer"
        assert payload['exp'] > datetime.now().timestamp(), "Token expired"

        return payload

    except Exception as e:
        raise HTTPException(
            status_code=401,
            detail=f"Invalid ID token: {str(e)}"

class TokenManager {
    constructor() {
        this.accessToken = null;
        this.refreshToken = null;
        this.refreshTimer = null;
    }

    async login(username, password) {
        const response = await fetch('/login', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({username, password})
        });

        const data = await response.json();
        this.accessToken = data.access_token;
        this.refreshToken = data.refresh_token;

        // ìë™ ê°±ì‹  ìŠ¤ì¼€ì¤„ë§ (14ë¶„ í›„)
        this.scheduleRefresh(14 * 60 * 1000);
    }

    scheduleRefresh(delay) {
        clearTimeout(this.refreshTimer);
        this.refreshTimer = setTimeout(() => this.refresh(), delay);
    }

    async refresh() {
        const response = await fetch('/refresh', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({refresh_token: this.refreshToken})
        });

        const data = await response.json();
        this.accessToken = data.access_token;

        }

        return response;
    }
