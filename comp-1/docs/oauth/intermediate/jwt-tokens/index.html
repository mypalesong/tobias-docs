🔍 JWT란?​

정의​

**JWT (JSON Web Token)**은 당사자 간에 정보를 안전하게 전송하기 위한 컴팩트하고 자체 포함된 방식입니다.

jwt_definition = {
    "형식": "Base64URL 인코딩된 JSON",
    "구성": "Header.Payload.Signature",
    "특징": "자체 검증 가능 (self-contained)",
    "용도": "인증, 정보 교환",
    "표준": "RFC 7519"
}

# JWT 예시
jwt_example = (
    "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9."  # Header
    "eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIiwiaWF0IjoxNTE2MjM5MDIyfQ."  # Payload
    "SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c"  # Signature
)

📋 JWT 구조​

1. Header​

알고리즘과 토큰 타입을 포함합니다.

# Header (JSON)
header = {
    "alg": "HS256",  # 알고리즘 (HMAC SHA-256)
    "typ": "JWT"     # 토큰 타입
}

# Base64URL 인코딩
# eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9

# 지원 알고리즘
algorithms = {
    "HS256": "HMAC SHA-256 (대칭키)",
    "HS384": "HMAC SHA-384 (대칭키)",
    "HS512": "HMAC SHA-512 (대칭키)",
    "RS256": "RSA SHA-256 (비대칭키) - 권장",
    "RS384": "RSA SHA-384 (비대칭키)",
    "RS512": "RSA SHA-512 (비대칭키)",
    "ES256": "ECDSA SHA-256 (비대칭키)",
    "ES384": "ECDSA SHA-384 (비대칭키)",
    "ES512": "ECDSA SHA-512 (비대칭키)"
}

2. Payload (Claims)​

사용자 정보와 메타데이터를 포함합니다.

# Payload (JSON)
payload = {
    # Registered Claims (표준)
    "iss": "https://auth.example.com",  # Issuer (발급자)
    "sub": "1234567890",                 # Subject (주체, 사용자 ID)
    "aud": "https://api.example.com",    # Audience (대상)
    "exp": 1701234567,                   # Expiration Time (만료 시간)
    "nbf": 1701230967,                   # Not Before (유효 시작 시간)
    "iat": 1701230967,                   # Issued At (발급 시간)
    "jti": "unique-token-id",            # JWT ID (토큰 고유 ID)

    # Public Claims (공개)
    "name": "Alice Smith",
    "email": "alice@example.com",

    # Private Claims (비공개)
    "user_id": 42,
    "role": "admin",
    "permissions": ["read", "write"]
}

# Base64URL 인코딩
# eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIiwiaWF0IjoxNTE2MjM5MDIyfQ

3. Signature​

토큰의 무결성을 보장합니다.

# Signature 생성
import hmac
import hashlib
import base64

def create_signature(header_b64: str, payload_b64: str, secret: str) -> str:
    """HMAC SHA-256 서명 생성"""

    message = f"{header_b64}.{payload_b64}".encode()

    signature = hmac.new(
        secret.encode(),
        message,
        hashlib.sha256
    ).digest()

    signature_b64 = base64.urlsafe_b64encode(signature).decode().rstrip('=')

    return signature_b64

# 검증: 동일한 secret으로 서명 재생성 → 비교

JWT 생성 함수​

from jose import jwt, JWTError
from datetime import datetime, timedelta
from typing import Optional

# 설정
SECRET_KEY = "your-secret-key-keep-it-secret"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30

def create_access_token(
    data: dict,
    expires_delta: Optional[timedelta] = None
) -> str:
    """Access Token 생성"""

    to_encode = data.copy()

    # 만료 시간 설정
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=15)

    to_encode.update({"exp": expire})

    # JWT 생성
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

    return encoded_jwt

# 사용 예
user_data = {
    "sub": "user123",
    "email": "alice@example.com",
    "role": "admin"
}

access_token = create_access_token(
    data=user_data,
    expires_delta=timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
)

print(access_token)

JWT 검증 함수​

from jose import jwt, JWTError
from fastapi import HTTPException, status

def verify_token(token: str) -> dict:
    """JWT 검증 및 디코딩"""

    try:
        # JWT 디코딩 및 검증
        payload = jwt.decode(
            token,
            SECRET_KEY,
            algorithms=[ALGORITHM]
        )

        # sub claim 확인 (사용자 ID)
        user_id: str = payload.get("sub")
        if user_id is None:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid token"
            )

        return payload

    except JWTError as e:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail=f"Invalid token: {str(e)}"
        )

# 사용 예
try:
    payload = verify_token(access_token)
    print(payload)
    # {'sub': 'user123', 'email': 'alice@example.com', 'exp': 1701234567}
except HTTPException as e:
    print(f"Error: {e.detail}")

완전한 인증 시스템​

from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
from pydantic import BaseModel
from jose import jwt, JWTError
from passlib.context import CryptContext
from datetime import datetime, timedelta
from typing import Optional

# OAuth2 스킴

# 설정
SECRET_KEY = "your-secret-key"
ALGORITHM = "HS256"

class User(BaseModel):
    username: str
    email: str
    full_name: str | None = None
    disabled: bool = False

class UserInDB(User):
    hashed_password: str

class Token(BaseModel):
    access_token: str
    token_type: str

class TokenData(BaseModel):
    username: str | None = None

# ==================== 가짜 DB ====================

fake_users_db = {
    "alice": {
        "username": "alice",
        "full_name": "Alice Smith",
        "email": "alice@example.com",
        "hashed_password": "$2b$12$EixZaYVK1fsbw1ZfbX3OXePaWxn96p36WQoeG6Lruj3vjPGga31lW",  # "secret"
        "disabled": False,
    }
}

# ==================== 유틸리티 ====================

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """비밀번호 검증"""
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    """비밀번호 해싱"""
    return pwd_context.hash(password)

def get_user(db, username: str):
    """사용자 조회"""
    if username in db:
        user_dict = db[username]
        return UserInDB(**user_dict)

def authenticate_user(db, username: str, password: str):
    """사용자 인증"""
    user = get_user(db, username)
    if not user:
        return False
    if not verify_password(password, user.hashed_password):
        return False
    return user

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None):
    """Access Token 생성"""
    to_encode = data.copy()

    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=15)

    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

    return encoded_jwt

async def get_current_user(token: str = Depends(oauth2_scheme)):
    """현재 사용자 가져오기 (의존성)"""
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )

    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
        token_data = TokenData(username=username)
    except JWTError:
        raise credentials_exception

    user = get_user(fake_users_db, username=token_data.username)
    if user is None:
        raise credentials_exception

    return user

async def get_current_active_user(
    current_user: User = Depends(get_current_user)
):
    """활성 사용자만 (의존성)"""
    if current_user.disabled:
        raise HTTPException(status_code=400, detail="Inactive user")
    return current_user

# ==================== 라우트 ====================

@app.post("/token", response_model=Token)
async def login(form_data: OAuth2PasswordRequestForm = Depends()):
    """로그인 (JWT 발급)"""
    user = authenticate_user(
        fake_users_db,
        form_data.username,
        form_data.password
    )

    if not user:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect username or password",
            headers={"WWW-Authenticate": "Bearer"},
        )

    access_token_expires = timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    access_token = create_access_token(
        data={"sub": user.username},
        expires_delta=access_token_expires
    )

    return {"access_token": access_token, "token_type": "bearer"}

@app.get("/users/me", response_model=User)
async def read_users_me(current_user: User = Depends(get_current_active_user)):
    """현재 사용자 정보"""
    return current_user

@app.get("/users/me/items")
async def read_own_items(current_user: User = Depends(get_current_active_user)):
    """현재 사용자의 아이템"""
    return [{"item_id": "Foo", "owner": current_user.username}]

🆔 ID Token 처리 (OIDC)​

ID Token 검증​

from jose import jwt, jwk
from jose.utils import base64url_decode
import httpx

async def get_google_public_keys():
    """Google 공개 키 가져오기"""
    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://www.googleapis.com/oauth2/v3/certs'
        )
        return response.json()

async def verify_id_token(id_token: str, client_id: str):
    """Google ID Token 검증"""

    try:
        # 1. 공개 키 가져오기
        jwks = await get_google_public_keys()

        # 2. Header에서 kid 추출
        unverified_header = jwt.get_unverified_header(id_token)
        kid = unverified_header['kid']

        # 3. kid로 공개 키 찾기
        key = next(
            (k for k in jwks['keys'] if k['kid'] == kid),
            None
        )

        if not key:
            raise ValueError("Public key not found")

        # 4. JWT 검증 및 디코딩
        payload = jwt.decode(
            id_token,
            key,
            algorithms=['RS256'],
            audience=client_id,
            issuer='https://accounts.google.com'
        )

        # 5. Claims 검증
        assert payload['aud'] == client_id, "Invalid audience"
        assert payload['iss'] == 'https://accounts.google.com', "Invalid issuer"
        assert payload['exp'] > datetime.now().timestamp(), "Token expired"

        return payload

    except Exception as e:
        raise HTTPException(
            status_code=401,
            detail=f"Invalid ID token: {str(e)}"
        )

🔄 Refresh Token 패턴​

Refresh Token 생성​

def create_tokens(user_id: str):
    """Access Token + Refresh Token 생성"""

    # Access Token (짧은 수명)
    access_token_expires = timedelta(minutes=15)
    access_token = create_access_token(
        data={"sub": user_id, "type": "access"},
        expires_delta=access_token_expires
    )

    # Refresh Token (긴 수명)
    refresh_token_expires = timedelta(days=30)
    refresh_token = create_access_token(
        data={"sub": user_id, "type": "refresh"},
        expires_delta=refresh_token_expires
    )

    return {
        "access_token": access_token,
        "refresh_token": refresh_token,
        "token_type": "bearer"
    }

@app.post("/login")
async def login(username: str, password: str):
    """로그인 (Access + Refresh Token)"""

    user = authenticate_user(fake_users_db, username, password)
    if not user:
        raise HTTPException(status_code=401, detail="Invalid credentials")

    tokens = create_tokens(user.username)

    return tokens

Refresh Token으로 갱신​

@app.post("/refresh")
async def refresh_token(refresh_token: str):
    """Refresh Token으로 새 Access Token 발급"""

    try:
        # Refresh Token 검증
        payload = jwt.decode(refresh_token, SECRET_KEY, algorithms=[ALGORITHM])

        # 타입 확인
        if payload.get("type") != "refresh":
            raise HTTPException(status_code=401, detail="Invalid token type")

        user_id = payload.get("sub")
        if not user_id:
            raise HTTPException(status_code=401, detail="Invalid token")

        # 새 Access Token 생성
        access_token = create_access_token(
            data={"sub": user_id, "type": "access"},
            expires_delta=timedelta(minutes=15)
        )

        return {
            "access_token": access_token,
            "token_type": "bearer"
        }

    except JWTError:
        raise HTTPException(status_code=401, detail="Invalid refresh token")

1. 역할 기반 접근 제어 (RBAC)​

from enum import Enum

class Role(str, Enum):
    ADMIN = "admin"
    USER = "user"
    GUEST = "guest"

def create_access_token_with_role(user_id: str, role: Role):
    """역할 포함 토큰"""
    return create_access_token(
        data={
            "sub": user_id,
            "role": role.value,
            "permissions": get_permissions(role)
        }
    )

def get_permissions(role: Role) -> list[str]:
    """역할별 권한"""
    permissions_map = {
        Role.ADMIN: ["read", "write", "delete", "admin"],
        Role.USER: ["read", "write"],
        Role.GUEST: ["read"]
    }
    return permissions_map.get(role, [])

# 역할 확인 의존성
def require_role(required_role: Role):
    """특정 역할 필요"""
    async def role_checker(token: str = Depends(oauth2_scheme)):
        payload = verify_token(token)
        user_role = payload.get("role")

        if user_role != required_role.value:
            raise HTTPException(
                status_code=403,
                detail=f"Required role: {required_role.value}"
            )

        return payload

    return role_checker

# 사용 예
@app.delete("/users/{user_id}")
async def delete_user(
    user_id: int,
    current_user: dict = Depends(require_role(Role.ADMIN))
):
    """관리자 전용: 사용자 삭제"""
    return {"message": f"User {user_id} deleted by {current_user['sub']}"}

2. 토큰 블랙리스트 (로그아웃)​

from datetime import datetime

# 블랙리스트 (실제로는 Redis 사용 권장)
token_blacklist = set()

def revoke_token(token: str):
    """토큰 폐기"""
    token_blacklist.add(token)

def is_token_revoked(token: str) -> bool:
    """토큰 폐기 확인"""
    return token in token_blacklist

async def get_current_user_with_blacklist(token: str = Depends(oauth2_scheme)):
    """블랙리스트 확인 포함"""

    # 블랙리스트 확인
    if is_token_revoked(token):
        raise HTTPException(status_code=401, detail="Token has been revoked")

    # 일반 검증
    payload = verify_token(token)
    return payload

@app.post("/logout")
async def logout(token: str = Depends(oauth2_scheme)):
    """로그아웃 (토큰 폐기)"""
    revoke_token(token)
    return {"message": "Successfully logged out"}

3. 자동 토큰 갱신 (Frontend)​

// Frontend (JavaScript)
class TokenManager {
    constructor() {
        this.accessToken = null;
        this.refreshToken = null;
        this.refreshTimer = null;
    }

    async login(username, password) {
        const response = await fetch('/login', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({username, password})
        });

        const data = await response.json();
        this.accessToken = data.access_token;
        this.refreshToken = data.refresh_token;

        // 자동 갱신 스케줄링 (14분 후)
        this.scheduleRefresh(14 * 60 * 1000);
    }

    scheduleRefresh(delay) {
        clearTimeout(this.refreshTimer);
        this.refreshTimer = setTimeout(() => this.refresh(), delay);
    }

    async refresh() {
        const response = await fetch('/refresh', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({refresh_token: this.refreshToken})
        });

        const data = await response.json();
        this.accessToken = data.access_token;

        // 다음 갱신 스케줄링
        this.scheduleRefresh(14 * 60 * 1000);
    }

    async apiCall(url, options = {}) {
        options.headers = {
            ...options.headers,
            'Authorization': `Bearer ${this.accessToken}`
        };

        let response = await fetch(url, options);

        // 401 에러 시 토큰 갱신 후 재시도
        if (response.status === 401) {
            await this.refresh();
            options.headers['Authorization'] = `Bearer ${this.accessToken}`;
            response = await fetch(url, options);
        }

        return response;
    }
}

1. Secret Key 관리​

# ❌ 나쁜 예

import secrets

2. 알고리즘 선택​

# ✅ 권장: RS256 (비대칭키)
# - 공개 키로 검증 가능
# - Private 키 노출 위험 감소

ALGORITHM = "RS256"

# - Secret 키 공유 필요
# - 작은 프로젝트용

3. 짧은 만료 시간​

# ✅ Access Token: 15-30분
ACCESS_TOKEN_EXPIRE_MINUTES = 15

# ✅ Refresh Token: 7-30일
REFRESH_TOKEN_EXPIRE_DAYS = 7

# ❌ 너무 긴 만료 시간 (보안 위험)
ACCESS_TOKEN_EXPIRE_MINUTES = 1440  # 24시간

4. HTTPS 필수​

# 프로덕션에서는 HTTPS 필수!
# JWT는 암호화되지 않음 (서명만)

# payload는 Base64 디코딩으로 누구나 읽을 수 있음
import base64
import json

payload_b64 = "eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIn0"
payload_json = base64.urlsafe_b64decode(payload_b64 + "==")
payload = json.loads(payload_json)
# {'sub': '1234567890', 'name': 'John Doe'}

# → HTTPS 없이는 중간에서 탈취 가능!
