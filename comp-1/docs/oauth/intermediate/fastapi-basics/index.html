config.py​

from pydantic_settings import BaseSettings
from functools import lru_cache

class Settings(BaseSettings):
    """환경변수 설정"""

    # 앱 설정
    app_name: str = "OAuth FastAPI App"
    secret_key: str
    base_url: str = "http://localhost:8000"

    # Google OAuth
    google_client_id: str
    google_client_secret: str

    # GitHub OAuth
    github_client_id: str | None = None
    github_client_secret: str | None = None

    class Config:
        env_file = ".env"
        case_sensitive = False

@lru_cache()
def get_settings() -> Settings:
    """싱글톤 설정 인스턴스"""
    return Settings()

auth/oauth.py​

from authlib.integrations.starlette_client import OAuth
from starlette.config import Config

# Config 로딩
config = Config('.env')

# OAuth 인스턴스 생성
oauth = OAuth(config)

# Google OAuth 등록
oauth.register(
    name='google',
    client_id=config('GOOGLE_CLIENT_ID'),
    client_secret=config('GOOGLE_CLIENT_SECRET'),
    server_metadata_url='https://accounts.google.com/.well-known/openid-configuration',
    client_kwargs={
        'scope': 'openid email profile'
    }
)

# GitHub OAuth 등록
oauth.register(
    name='github',
    client_id=config('GITHUB_CLIENT_ID', default=None),
    client_secret=config('GITHUB_CLIENT_SECRET', default=None),
    authorize_url='https://github.com/login/oauth/authorize',
    access_token_url='https://github.com/login/oauth/access_token',
    api_base_url='https://api.github.com/',
    client_kwargs={'scope': 'user:email'}
)

🔐 인증 라우트​

auth/routes.py​

from fastapi import APIRouter, Request, HTTPException
from fastapi.responses import RedirectResponse
from auth.oauth import oauth

router = APIRouter()

# ==================== Google OAuth ====================

@router.get('/login/google')
async def login_google(request: Request):
    """Google 로그인 시작"""
    redirect_uri = request.url_for('auth_google')
    return await oauth.google.authorize_redirect(request, redirect_uri)

@router.get('/google')
async def auth_google(request: Request):
    """Google OAuth 콜백"""
    try:
        # Access Token 받기
        token = await oauth.google.authorize_access_token(request)

        # 사용자 정보 가져오기
        user = token.get('userinfo')

        if not user:
            raise HTTPException(status_code=400, detail="Failed to get user info")

        # 세션에 저장
        request.session['user'] = {
            'provider': 'google',
            'sub': user['sub'],
            'email': user['email'],
            'name': user['name'],
            'picture': user.get('picture')
        }

        # 토큰도 저장 (선택)
        request.session['tokens'] = {
            'access_token': token['access_token'],
            'refresh_token': token.get('refresh_token'),
            'id_token': token.get('id_token')
        }

        return RedirectResponse(url='/')

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

# ==================== GitHub OAuth ====================

@router.get('/login/github')
async def login_github(request: Request):
    """GitHub 로그인 시작"""
    redirect_uri = request.url_for('auth_github')
    return await oauth.github.authorize_redirect(request, redirect_uri)

@router.get('/github')
async def auth_github(request: Request):
    """GitHub OAuth 콜백"""
    try:
        # Access Token 받기
        token = await oauth.github.authorize_access_token(request)

        # 사용자 정보 가져오기
        resp = await oauth.github.get('user', token=token)
        user = resp.json()

        # 이메일 가져오기 (별도 요청)
        emails_resp = await oauth.github.get('user/emails', token=token)
        emails = emails_resp.json()
        primary_email = next(
            (email['email'] for email in emails if email['primary']),
            None
        )

        # 세션에 저장
        request.session['user'] = {
            'provider': 'github',
            'id': user['id'],
            'login': user['login'],
            'email': primary_email,
            'name': user.get('name', user['login']),
            'avatar': user['avatar_url']
        }

        request.session['tokens'] = {
            'access_token': token['access_token']
        }

        return RedirectResponse(url='/')

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

# ==================== 로그아웃 ====================

@router.post('/logout')
async def logout(request: Request):
    """로그아웃"""
    request.session.clear()
    return {"message": "Logged out successfully"}

@router.get('/logout')
async def logout_get(request: Request):
    """로그아웃 (GET)"""
    request.session.clear()
    return RedirectResponse(url='/')

🛡️ 의존성 주입으로 라우트 보호​

auth/dependencies.py​

from fastapi import Depends, HTTPException, Request, status
from typing import Annotated

def get_current_user(request: Request):
    """현재 로그인 사용자 가져오기"""
    user = request.session.get('user')

    if not user:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Not authenticated",
            headers={"WWW-Authenticate": "Bearer"},
        )

    return user

def get_current_user_optional(request: Request):
    """현재 로그인 사용자 가져오기 (선택적)"""
    return request.session.get('user')

def get_access_token(request: Request) -> str:
    """Access Token 가져오기"""
    tokens = request.session.get('tokens')

    if not tokens or 'access_token' not in tokens:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="No access token"
        )

    return tokens['access_token']

# 타입 별칭 (편의용)
CurrentUser = Annotated[dict, Depends(get_current_user)]
CurrentUserOptional = Annotated[dict | None, Depends(get_current_user_optional)]
AccessToken = Annotated[str, Depends(get_access_token)]

api/routes.py​

from fastapi import APIRouter, HTTPException
from auth.dependencies import CurrentUser, CurrentUserOptional, AccessToken
import httpx

router = APIRouter()

@router.get('/me')
async def get_me(user: CurrentUser):
    """현재 사용자 정보"""
    return {
        "message": "Authenticated user",
        "user": user
    }

@router.get('/profile')
async def get_profile(user: CurrentUserOptional):
    """프로필 (선택적 인증)"""
    if user:
        return {"message": "Logged in", "user": user}
    else:
        return {"message": "Not logged in", "user": None}

@router.get('/google/photos')
async def get_google_photos(user: CurrentUser, access_token: AccessToken):
    """Google Photos 가져오기 (예시)"""

    # Provider 확인
    if user.get('provider') != 'google':
        raise HTTPException(status_code=400, detail="Google 계정만 가능")

    # Google Photos API 호출
    async with httpx.AsyncClient() as client:
        try:
            response = await client.get(
                'https://photoslibrary.googleapis.com/v1/mediaItems',
                headers={'Authorization': f'Bearer {access_token}'},
                params={'pageSize': 10}
            )

            if response.status_code != 200:
                raise HTTPException(
                    status_code=response.status_code,
                    detail="Failed to fetch photos"
                )

            return response.json()

        except httpx.HTTPError as e:
            raise HTTPException(status_code=500, detail=str(e))

@router.get('/github/repos')
async def get_github_repos(user: CurrentUser, access_token: AccessToken):
    """GitHub 저장소 가져오기"""

    if user.get('provider') != 'github':
        raise HTTPException(status_code=400, detail="GitHub 계정만 가능")

    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://api.github.com/user/repos',
            headers={
                'Authorization': f'Bearer {access_token}',
                'Accept': 'application/vnd.github.v3+json'
            },
            params={'per_page': 10, 'sort': 'updated'}
        )

        if response.status_code != 200:
            raise HTTPException(
                status_code=response.status_code,
                detail="Failed to fetch repos"
            )

        return response.json()

# 관리자 전용 엔드포인트
def get_admin_user(user: CurrentUser):
    """관리자 권한 확인"""
    # 실제로는 DB에서 역할 확인
    if user.get('email') not in ['admin@example.com']:
        raise HTTPException(
            status_code=403,
            detail="Forbidden: Admin access required"
        )
    return user

@router.get('/admin/users')
async def admin_users(admin: dict = Depends(get_admin_user)):
    """관리자 전용: 사용자 목록"""
    return {
        "message": "Admin access granted",
        "admin": admin,
        "users": []  # DB에서 조회
    }

1. 사용자 정보 정규화​

# auth/models.py
from pydantic import BaseModel
from typing import Literal

class User(BaseModel):
    """정규화된 사용자 모델"""
    provider: Literal['google', 'github']
    id: str  # sub 또는 provider ID
    email: str
    name: str
    picture: str | None = None

    @classmethod
    def from_google(cls, userinfo: dict):
        """Google 사용자 정보 변환"""
        return cls(
            provider='google',
            id=userinfo['sub'],
            email=userinfo['email'],
            name=userinfo['name'],
            picture=userinfo.get('picture')
        )

    @classmethod
    def from_github(cls, user: dict, email: str):
        """GitHub 사용자 정보 변환"""
        return cls(
            provider='github',
            id=str(user['id']),
            email=email,
            name=user.get('name', user['login']),
            picture=user['avatar_url']
        )

# 사용
user_model = User.from_google(userinfo)

2. 토큰 갱신 미들웨어​

# auth/middleware.py
from starlette.middleware.base import BaseHTTPMiddleware
from datetime import datetime, timedelta
import httpx

class TokenRefreshMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        """Access Token 자동 갱신"""

        tokens = request.session.get('tokens')
        if tokens and 'refresh_token' in tokens:
            # 만료 시간 확인
            expires_at = tokens.get('expires_at')
            if expires_at:
                expires_dt = datetime.fromisoformat(expires_at)

                # 5분 이내 만료 예정이면 갱신
                if datetime.now() + timedelta(minutes=5) >= expires_dt:
                    await self.refresh_token(request, tokens['refresh_token'])

        response = await call_next(request)
        return response

    async def refresh_token(self, request, refresh_token: str):
        """Refresh Token으로 갱신"""
        async with httpx.AsyncClient() as client:
            response = await client.post(
                'https://oauth2.googleapis.com/token',
                data={
                    'grant_type': 'refresh_token',
                    'refresh_token': refresh_token,
                    'client_id': GOOGLE_CLIENT_ID,
                    'client_secret': GOOGLE_CLIENT_SECRET
                }
            )

            if response.status_code == 200:
                data = response.json()
                tokens = request.session['tokens']
                tokens['access_token'] = data['access_token']
                tokens['expires_at'] = (
                    datetime.now() + timedelta(seconds=data['expires_in'])
                ).isoformat()
                request.session['tokens'] = tokens

3. DB 연동 (SQLAlchemy)​

# db/models.py
from sqlalchemy import Column, Integer, String, DateTime
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class User(Base):
    __tablename__ = 'users'

    id = Column(Integer, primary_key=True)
    provider = Column(String(50), nullable=False)
    provider_id = Column(String(255), nullable=False, unique=True)
    email = Column(String(255), nullable=False)
    name = Column(String(255))
    picture = Column(String(500))
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

class OAuthToken(Base):
    __tablename__ = 'oauth_tokens'

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'))
    access_token = Column(String(500))
    refresh_token = Column(String(500))
    expires_at = Column(DateTime)

# auth/routes.py에서 사용
from db.models import User, OAuthToken
from db.session import get_db

@router.get('/google')
async def auth_google(request: Request, db: Session = Depends(get_db)):
    token = await oauth.google.authorize_access_token(request)
    userinfo = token.get('userinfo')

    # DB에서 사용자 찾기 또는 생성
    user = db.query(User).filter_by(
        provider='google',
        provider_id=userinfo['sub']
    ).first()

    if not user:
        user = User(
            provider='google',
            provider_id=userinfo['sub'],
            email=userinfo['email'],
            name=userinfo['name'],
            picture=userinfo.get('picture')
        )
        db.add(user)
        db.commit()

    # 토큰 저장
    oauth_token = OAuthToken(
        user_id=user.id,
        access_token=token['access_token'],
        refresh_token=token.get('refresh_token'),
        expires_at=datetime.now() + timedelta(seconds=token['expires_in'])
    )
    db.add(oauth_token)
    db.commit()

    # 세션에 저장
    request.session['user_id'] = user.id

    return RedirectResponse(url='/')
