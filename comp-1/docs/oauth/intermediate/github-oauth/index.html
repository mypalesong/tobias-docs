🔧 기본 구현​

전체 코드 (main.py)​

from fastapi import FastAPI, Request, HTTPException
from fastapi.responses import RedirectResponse, HTMLResponse
from starlette.middleware.sessions import SessionMiddleware
from authlib.integrations.starlette_client import OAuth
from starlette.config import Config
import httpx

# 환경변수 로드
config = Config('.env')

app = FastAPI(title="GitHub OAuth Example")

# 세션 미들웨어
app.add_middleware(
    SessionMiddleware,
    secret_key=config('SECRET_KEY', default='super-secret-key')
)

# OAuth 설정
oauth = OAuth(config)

oauth.register(
    name='github',
    client_id=config('GITHUB_CLIENT_ID'),
    client_secret=config('GITHUB_CLIENT_SECRET'),
    authorize_url='https://github.com/login/oauth/authorize',
    access_token_url='https://github.com/login/oauth/access_token',
    api_base_url='https://api.github.com/',
    client_kwargs={'scope': 'user:email'}
)

# ==================== 라우트 ====================

@app.get('/', response_class=HTMLResponse)
async def home(request: Request):
    """홈페이지"""
    user = request.session.get('user')

    if user:
        return f"""
        <html>
            <body>
                <h1>환영합니다, {user['name']}님!</h1>
                <img src="{user['avatar']}" alt="아바타" width="100">
                <p>사용자명: @{user['login']}</p>
                <p>이메일: {user.get('email', 'N/A')}</p>
                <a href="/logout">로그아웃</a>
                <br><br>
                <a href="/profile">내 프로필</a>
                <a href="/api/repos">내 저장소</a>
            </body>
        </html>
        """
    else:
        return """
        <html>
            <body>
                <h1>GitHub OAuth 예제</h1>
                <a href="/login">GitHub로 로그인</a>
            </body>
        </html>
        """

@app.get('/login')
async def login(request: Request):
    """GitHub 로그인 시작"""
    redirect_uri = request.url_for('auth')
    return await oauth.github.authorize_redirect(request, redirect_uri)

@app.get('/auth')
async def auth(request: Request):
    """GitHub OAuth 콜백"""
    try:
        # Access Token 받기
        token = await oauth.github.authorize_access_token(request)

        # ⚠️ GitHub는 ID Token 없음! API로 사용자 정보 요청
        # 1. 기본 사용자 정보
        resp = await oauth.github.get('user', token=token)
        user_data = resp.json()

        # 2. 이메일 정보 (별도 요청)
        emails_resp = await oauth.github.get('user/emails', token=token)
        emails = emails_resp.json()

        # Primary 이메일 찾기
        primary_email = next(
            (email['email'] for email in emails if email['primary']),
            user_data.get('email')  # fallback
        )

        # 세션에 저장
        request.session['user'] = {
            'id': user_data['id'],
            'login': user_data['login'],
            'name': user_data.get('name', user_data['login']),
            'email': primary_email,
            'avatar': user_data['avatar_url'],
            'bio': user_data.get('bio'),
            'company': user_data.get('company'),
            'location': user_data.get('location'),
            'blog': user_data.get('blog')
        }

        # 토큰 저장
        request.session['tokens'] = {
            'access_token': token['access_token'],
            'scope': token.get('scope', '')
        }

        return RedirectResponse(url='/')

    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.get('/logout')
async def logout(request: Request):
    """로그아웃"""
    request.session.clear()
    return RedirectResponse(url='/')

@app.get('/profile')
async def profile(request: Request):
    """사용자 프로필"""
    user = request.session.get('user')

    if not user:
        raise HTTPException(status_code=401, detail="Not authenticated")

    return user

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

# GitHub는 OIDC가 아니므로 API 호출 필요

async def get_github_user(access_token: str):
    """GitHub 사용자 정보 가져오기"""

    async with httpx.AsyncClient() as client:
        # 1. 기본 사용자 정보
        user_response = await client.get(
            'https://api.github.com/user',
            headers={
                'Authorization': f'Bearer {access_token}',
                'Accept': 'application/vnd.github.v3+json'
            }
        )
        user = user_response.json()

        # {
        #   "login": "octocat",
        #   "id": 1,
        #   "avatar_url": "https://github.com/images/error/octocat_happy.gif",
        #   "name": "The Octocat",
        #   "email": "octocat@github.com",  # public email만
        #   "bio": "There once was...",
        #   "company": "@github",
        #   "location": "San Francisco",
        #   "blog": "https://github.blog"
        # }

        # 2. 이메일 목록 (user:email scope 필요)
        emails_response = await client.get(
            'https://api.github.com/user/emails',
            headers={
                'Authorization': f'Bearer {access_token}',
                'Accept': 'application/vnd.github.v3+json'
            }
        )
        emails = emails_response.json()

        # [
        #   {
        #     "email": "octocat@github.com",
        #     "primary": true,
        #     "verified": true,
        #     "visibility": "public"
        #   }
        # ]

        # Primary 이메일 추가
        primary_email = next(
            (e['email'] for e in emails if e['primary']),
            user.get('email')
        )
        user['primary_email'] = primary_email

        return user

저장소 목록​

@app.get('/api/repos')
async def get_repos(request: Request):
    """사용자 저장소 목록"""

    tokens = request.session.get('tokens')
    if not tokens:
        raise HTTPException(status_code=401)

    async with httpx.AsyncClient() as client:
        response = await client.get(
            'https://api.github.com/user/repos',
            headers={
                'Authorization': f'Bearer {tokens["access_token"]}',
                'Accept': 'application/vnd.github.v3+json'
            },
            params={
                'sort': 'updated',  # created, updated, pushed, full_name
                'direction': 'desc',
                'per_page': 10
            }
        )

        if response.status_code != 200:
            raise HTTPException(
                status_code=response.status_code,
                detail="Failed to fetch repos"
            )

        repos = response.json()

        return {
            "count": len(repos),
            "repos": [
                {
                    "name": repo['name'],
                    "full_name": repo['full_name'],
                    "description": repo['description'],
                    "html_url": repo['html_url'],
                    "private": repo['private'],
                    "language": repo['language'],
                    "stargazers_count": repo['stargazers_count'],
                    "forks_count": repo['forks_count'],
                    "updated_at": repo['updated_at']
                }
                for repo in repos
            ]
        }

커밋 생성​

from pydantic import BaseModel

class CommitCreate(BaseModel):
    repo: str  # "owner/repo"
    path: str
    message: str
    content: str
    branch: str = "main"

@app.post('/api/commit')
async def create_commit(
    commit_data: CommitCreate,
    request: Request
):
    """파일 커밋"""

    tokens = request.session.get('tokens')
    if not tokens:
        raise HTTPException(status_code=401)

    # 필요 scope 확인
    if 'repo' not in tokens.get('scope', ''):
        raise HTTPException(
            status_code=403,
            detail="repo scope required"
        )

    import base64

    async with httpx.AsyncClient() as client:
        # 파일 생성/수정
        response = await client.put(
            f'https://api.github.com/repos/{commit_data.repo}/contents/{commit_data.path}',
            headers={
                'Authorization': f'Bearer {tokens["access_token"]}',
                'Accept': 'application/vnd.github.v3+json'
            },
            json={
                'message': commit_data.message,
                'content': base64.b64encode(
                    commit_data.content.encode()
                ).decode(),
                'branch': commit_data.branch
            }
        )

        if response.status_code not in [200, 201]:
            raise HTTPException(
                status_code=response.status_code,
                detail=response.json()
            )

        return response.json()

    repo: str  # "owner/repo"
    title: str
    body: str
    labels: list[str] = []

@app.post('/api/issues')
async def create_issue(
    issue: IssueCreate,
    request: Request
):
    """Issue 생성"""

    if not tokens:
        raise HTTPException(status_code=401)

    async with httpx.AsyncClient() as client:
        response = await client.post(
            f'https://api.github.com/repos/{issue.repo}/issues',
            headers={

                'Accept': 'application/vnd.github.v3+json'
            },
            json={
                'title': issue.title,
                'body': issue.body,
                'labels': issue.labels
            }
        )

        if response.status_code != 201:
            raise HTTPException(
                status_code=response.status_code,
                detail=response.json()
            )

        return response.json()

# Token 폐기
async def revoke_github_token(access_token: str):
    """GitHub Access Token 폐기"""

    async with httpx.AsyncClient() as client:
        response = await client.delete(
            f'https://api.github.com/applications/{CLIENT_ID}/token',
            auth=(CLIENT_ID, CLIENT_SECRET),
            headers={'Accept': 'application/vnd.github.v3+json'},
            json={'access_token': access_token}
        )

        return response.status_code == 204

# 로그아웃 시 폐기
@app.post('/logout')
async def logout(request: Request):
    tokens = request.session.get('tokens')

    if tokens:
        await revoke_github_token(tokens['access_token'])

    request.session.clear()
    return {"message": "Logged out"}

# Middleware로 Rate Limit 모니터링
from starlette.middleware.base import BaseHTTPMiddleware

class RateLimitMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        response = await call_next(request)

        # X-RateLimit 헤더 확인
        rate_limit = response.headers.get('X-RateLimit-Limit')
        rate_remaining = response.headers.get('X-RateLimit-Remaining')

        if rate_remaining and int(rate_remaining) < 100:
            # 알림 또는 로깅
            print(f"Warning: Only {rate_remaining} requests remaining")

        return response

Webhook Verification​

import hmac
import hashlib

def verify_github_webhook(payload: bytes, signature: str, secret: str) -> bool:
    """GitHub Webhook 서명 검증"""

    # signature: "sha256=hash"
    expected = hmac.new(
        secret.encode(),
        payload,
        hashlib.sha256
    ).hexdigest()

    return hmac.compare_digest(
        f"sha256={expected}",
        signature
    )

@app.post('/webhooks/github')
async def github_webhook(request: Request):
    """GitHub Webhook 핸들러"""

    # 서명 확인
    signature = request.headers.get('X-Hub-Signature-256')
    if not signature:
        raise HTTPException(status_code=400, detail="Missing signature")

    payload = await request.body()

    if not verify_github_webhook(payload, signature, WEBHOOK_SECRET):
        raise HTTPException(status_code=401, detail="Invalid signature")

        pass
    elif event_type == 'pull_request':
        # PR 이벤트 처리
        pass

    return {"status": "ok"}

프로토콜OAuth 2.0 + OIDCOAuth 2.0 only
