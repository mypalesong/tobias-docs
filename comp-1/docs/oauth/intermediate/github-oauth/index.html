    """GitHub OAuth ì½œë°±"""

            user_data.get('email')  # fallback

from pydantic import BaseModel

class CommitCreate(BaseModel):
    repo: str  # "owner/repo"
    path: str
    message: str
    content: str
    branch: str = "main"

@app.post('/api/commit')
async def create_commit(
    commit_data: CommitCreate,
    request: Request
):
    """íŒŒì¼ ì»¤ë°‹"""

    tokens = request.session.get('tokens')
    if not tokens:
        raise HTTPException(status_code=401)

    # í•„ìš” scope í™•ì¸
    if 'repo' not in tokens.get('scope', ''):
        raise HTTPException(
            status_code=403,
            detail="repo scope required"
        )

    import base64

    async with httpx.AsyncClient() as client:
        # íŒŒì¼ ìƒì„±/ìˆ˜ì •
        response = await client.put(
            f'https://api.github.com/repos/{commit_data.repo}/contents/{commit_data.path}',
            headers={
                'Authorization': f'Bearer {tokens["access_token"]}',
                'Accept': 'application/vnd.github.v3+json'
            },
            json={
                'message': commit_data.message,
                'content': base64.b64encode(
                    commit_data.content.encode()
                ).decode(),
                'branch': commit_data.branch
            }
        )

        if response.status_code not in [200, 201]:
            raise HTTPException(
                status_code=response.status_code,
                detail=response.json()
            )

        return response.json()

def verify_github_webhook(payload: bytes, signature: str, secret: str) -> bool:
    """GitHub Webhook ì„œëª… ê²€ì¦"""

    # signature: "sha256=hash"
    expected = hmac.new(
        secret.encode(),
        payload,
        hashlib.sha256
    ).hexdigest()

    return hmac.compare_digest(
        f"sha256={expected}",
        signature
    )
