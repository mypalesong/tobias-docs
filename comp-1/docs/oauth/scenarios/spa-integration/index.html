// src/utils/pkce.ts
export class PKCEHelper {
    /**
     * Code Verifier ìƒì„±
     */
    static generateCodeVerifier(): string {
        const array = new Uint8Array(32);
        crypto.getRandomValues(array);
        return this.base64URLEncode(array);
    }

    /**
     * Code Challenge ìƒì„±
     */
    static async generateCodeChallenge(verifier: string): Promise<string> {
        const encoder = new TextEncoder();
        const data = encoder.encode(verifier);
        const digest = await crypto.subtle.digest('SHA-256', data);
        return this.base64URLEncode(new Uint8Array(digest));
    }

    /**
     * Base64 URL ì¸ì½”ë”©
     */
    private static base64URLEncode(buffer: Uint8Array): string {
        const base64 = btoa(String.fromCharCode(...buffer));
        return base64
            .replace(/\+/g, '-')
            .replace(/\//g, '_')
            .replace(/=/g, '');
    }

    /**
     * State ìƒì„± (CSRF ë°©ì§€)
     */
    static generateState(): string {
        const array = new Uint8Array(16);
        crypto.getRandomValues(array);
        return this.base64URLEncode(array);
    }
}

import { useState, useEffect, useCallback } from 'react';
import { PKCEHelper } from '../utils/pkce';
import { oauthConfig } from '../config/oauth';

interface OAuthState {
    isAuthenticated: boolean;
    isLoading: boolean;
    user: User | null;
    error: string | null;
}

interface User {
    id: string;
    email: string;
    name: string;
    picture: string;
}

export const useOAuth = () => {
    const [state, setState] = useState<OAuthState>({
        isAuthenticated: false,
        isLoading: true,
        user: null,
        error: null
    });

    /**

    const login = useCallback(async () => {

            // 1. Code Verifier ìƒì„±
            const codeVerifier = PKCEHelper.generateCodeVerifier();
            sessionStorage.setItem('code_verifier', codeVerifier);

            // 2. Code Challenge ìƒì„±
            const codeChallenge = await PKCEHelper.generateCodeChallenge(codeVerifier);

            // 3. State ìƒì„±
            const state = PKCEHelper.generateState();
            sessionStorage.setItem('oauth_state', state);

            // 4. Authorization URL ìƒì„±
            const params = new URLSearchParams({
                client_id: oauthConfig.clientId,
                redirect_uri: oauthConfig.redirectUri,
                response_type: 'code',
                scope: oauthConfig.scopes.join(' '),
                state: state,
                code_challenge: codeChallenge,
                code_challenge_method: 'S256'

            const authUrl = `${oauthConfig.authorizationEndpoint}?${params}`;

            // 5. Authorization Serverë¡œ ë¦¬ë‹¤ì´ë ‰íŠ¸
            window.location.href = authUrl;
        } catch (error) {
            setState(prev => ({
                ...prev,
                error: 'Failed to start login'
            }));
        }
    }, []);

    /**
     * ì½œë°± ì²˜ë¦¬
     */
    const handleCallback = useCallback(async (code: string, state: string) => {
        try {
            // 1. State ê²€ì¦
            const savedState = sessionStorage.getItem('oauth_state');
            if (state !== savedState) {
                throw new Error('Invalid state');
            }

            // 2. Code Verifier ê°€ì ¸ì˜¤ê¸°
            const codeVerifier = sessionStorage.getItem('code_verifier');
            if (!codeVerifier) {
                throw new Error('Missing code verifier');
            }

            // 3. Token êµí™˜ (Backend Proxyë¥¼ í†µí•´)
            const response = await fetch(`${API_BASE_URL}/auth/token`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                credentials: 'include', // Cookie í¬í•¨
                body: JSON.stringify({
                    code,
                    code_verifier: codeVerifier,
                    redirect_uri: oauthConfig.redirectUri
                })
            });

            if (!response.ok) {
                throw new Error('Token exchange failed');
            }

            // 4. ì„¸ì…˜ ì •ë¦¬
            sessionStorage.removeItem('code_verifier');
            sessionStorage.removeItem('oauth_state');

            // 5. ì‚¬ìš©ì ì •ë³´ ê°€ì ¸ì˜¤ê¸°
            await fetchUser();

        } catch (error) {
            setState(prev => ({
                ...prev,
                error: error instanceof Error ? error.message : 'Login failed',
                isLoading: false
            }));
        }
    }, []);

    /**
     * ì‚¬ìš©ì ì •ë³´ ê°€ì ¸ì˜¤ê¸°
     */
    const fetchUser = useCallback(async () => {
        try {
            const response = await fetch(`${API_BASE_URL}/api/me`, {
                credentials: 'include' // Cookie í¬í•¨
            });

            if (response.ok) {
                const user = await response.json();
                setState({
                    isAuthenticated: true,
                    isLoading: false,
                    user,
                    error: null
                });
            } else {
                setState(prev => ({
                    ...prev,
                    isAuthenticated: false,
                    isLoading: false
                }));
            }
        } catch (error) {
            setState(prev => ({
                ...prev,
                isAuthenticated: false,
                isLoading: false
            }));
        }
    }, []);

    /**
     * ë¡œê·¸ì•„ì›ƒ

            method: 'POST',
            credentials: 'include'

        setState({
            isAuthenticated: false,
            isLoading: false,
            user: null,
            error: null
        });
    }, []);

    /**
     * ì´ˆê¸°í™”
     */
    useEffect(() => {
        fetchUser();
    }, [fetchUser]);

    return {
        ...state,
        login,
        logout,
        handleCallback
    };
};

@app.post("/auth/token")
async def token_proxy(request: Request, response: Response):
    """Token Exchange Proxy"""

    body = await request.json()
    code = body.get('code')
    code_verifier = body.get('code_verifier')
    redirect_uri = body.get('redirect_uri')

    # Google Token Endpointë¡œ ìš”ì²­
    async with httpx.AsyncClient() as client:

            'https://oauth2.googleapis.com/token',
            data={
                'grant_type': 'authorization_code',
                'code': code,
                'redirect_uri': redirect_uri,
                'client_id': GOOGLE_CLIENT_ID,
                'client_secret': GOOGLE_CLIENT_SECRET,  # Backendì—ì„œë§Œ!
                'code_verifier': code_verifier

        if token_response.status_code != 200:
            raise HTTPException(status_code=400, detail="Token exchange failed")

        tokens = token_response.json()

    # HttpOnly Cookieì— ì €ì¥

        secure=True,  # HTTPS
        samesite='lax',
        max_age=tokens.get('expires_in', 3600)
    )

    if 'refresh_token' in tokens:
        response.set_cookie(
            key="refresh_token",
            value=tokens['refresh_token'],
            httponly=True,
            secure=True,
            samesite='lax',
            max_age=2592000  # 30ì¼
        )

    return {"status": "ok"}

@app.post("/auth/refresh")
async def refresh_proxy(request: Request, response: Response):
    """Refresh Token Proxy"""

    refresh_token = request.cookies.get('refresh_token')
    if not refresh_token:
        raise HTTPException(status_code=401)

    async with httpx.AsyncClient() as client:
        token_response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'grant_type': 'refresh_token',
                'refresh_token': refresh_token,
                'client_id': GOOGLE_CLIENT_ID,
                'client_secret': GOOGLE_CLIENT_SECRET
            }
        )

        if token_response.status_code != 200:
            raise HTTPException(status_code=401)

        tokens = token_response.json()

    # ìƒˆ Access Token ì €ì¥
    response.set_cookie(
        key="access_token",
        value=tokens['access_token'],
        httponly=True,
        secure=True,
        samesite='lax',
        max_age=tokens.get('expires_in', 3600)
