3. PKCE 유틸리티​

// src/utils/pkce.ts
export class PKCEHelper {
    /**
     * Code Verifier 생성
     */
    static generateCodeVerifier(): string {
        const array = new Uint8Array(32);
        crypto.getRandomValues(array);
        return this.base64URLEncode(array);
    }

    /**
     * Code Challenge 생성
     */
    static async generateCodeChallenge(verifier: string): Promise<string> {
        const encoder = new TextEncoder();
        const data = encoder.encode(verifier);
        const digest = await crypto.subtle.digest('SHA-256', data);
        return this.base64URLEncode(new Uint8Array(digest));
    }

    /**
     * Base64 URL 인코딩
     */
    private static base64URLEncode(buffer: Uint8Array): string {
        const base64 = btoa(String.fromCharCode(...buffer));
        return base64
            .replace(/\+/g, '-')
            .replace(/\//g, '_')
            .replace(/=/g, '');
    }

    /**
     * State 생성 (CSRF 방지)
     */
    static generateState(): string {
        const array = new Uint8Array(16);
        crypto.getRandomValues(array);
        return this.base64URLEncode(array);
    }
}

4. OAuth 훅​

// src/hooks/useOAuth.ts
import { useState, useEffect, useCallback } from 'react';
import { PKCEHelper } from '../utils/pkce';
import { oauthConfig } from '../config/oauth';

interface OAuthState {
    isAuthenticated: boolean;
    isLoading: boolean;
    user: User | null;
    error: string | null;
}

interface User {
    id: string;
    email: string;
    name: string;
    picture: string;
}

export const useOAuth = () => {
    const [state, setState] = useState<OAuthState>({
        isAuthenticated: false,
        isLoading: true,
        user: null,
        error: null
    });

    /**
     * 로그인 시작
     */
    const login = useCallback(async () => {
        try {
            // 1. Code Verifier 생성
            const codeVerifier = PKCEHelper.generateCodeVerifier();
            sessionStorage.setItem('code_verifier', codeVerifier);

            // 2. Code Challenge 생성
            const codeChallenge = await PKCEHelper.generateCodeChallenge(codeVerifier);

            // 3. State 생성
            const state = PKCEHelper.generateState();
            sessionStorage.setItem('oauth_state', state);

            // 4. Authorization URL 생성
            const params = new URLSearchParams({
                client_id: oauthConfig.clientId,
                redirect_uri: oauthConfig.redirectUri,
                response_type: 'code',
                scope: oauthConfig.scopes.join(' '),
                state: state,
                code_challenge: codeChallenge,
                code_challenge_method: 'S256'
            });

            const authUrl = `${oauthConfig.authorizationEndpoint}?${params}`;

            // 5. Authorization Server로 리다이렉트
            window.location.href = authUrl;
        } catch (error) {
            setState(prev => ({
                ...prev,
                error: 'Failed to start login'
            }));
        }
    }, []);

    /**
     * 콜백 처리
     */
    const handleCallback = useCallback(async (code: string, state: string) => {
        try {
            // 1. State 검증
            const savedState = sessionStorage.getItem('oauth_state');
            if (state !== savedState) {
                throw new Error('Invalid state');
            }

            // 2. Code Verifier 가져오기
            const codeVerifier = sessionStorage.getItem('code_verifier');
            if (!codeVerifier) {
                throw new Error('Missing code verifier');
            }

            // 3. Token 교환 (Backend Proxy를 통해)
            const response = await fetch(`${API_BASE_URL}/auth/token`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                credentials: 'include', // Cookie 포함
                body: JSON.stringify({
                    code,
                    code_verifier: codeVerifier,
                    redirect_uri: oauthConfig.redirectUri
                })
            });

            if (!response.ok) {
                throw new Error('Token exchange failed');
            }

            // 4. 세션 정리
            sessionStorage.removeItem('code_verifier');
            sessionStorage.removeItem('oauth_state');

            // 5. 사용자 정보 가져오기
            await fetchUser();

        } catch (error) {
            setState(prev => ({
                ...prev,
                error: error instanceof Error ? error.message : 'Login failed',
                isLoading: false
            }));
        }
    }, []);

    /**
     * 사용자 정보 가져오기
     */
    const fetchUser = useCallback(async () => {
        try {
            const response = await fetch(`${API_BASE_URL}/api/me`, {
                credentials: 'include' // Cookie 포함
            });

            if (response.ok) {
                const user = await response.json();
                setState({
                    isAuthenticated: true,
                    isLoading: false,
                    user,
                    error: null
                });
            } else {
                setState(prev => ({
                    ...prev,
                    isAuthenticated: false,
                    isLoading: false
                }));
            }
        } catch (error) {
            setState(prev => ({
                ...prev,
                isAuthenticated: false,
                isLoading: false
            }));
        }
    }, []);

    /**
     * 로그아웃
     */
    const logout = useCallback(async () => {
        await fetch(`${API_BASE_URL}/auth/logout`, {
            method: 'POST',
            credentials: 'include'
        });

        setState({
            isAuthenticated: false,
            isLoading: false,
            user: null,
            error: null
        });
    }, []);

    /**
     * 초기화
     */
    useEffect(() => {
        fetchUser();
    }, [fetchUser]);

    return {
        ...state,
        login,
        logout,
        handleCallback
    };
};

5. Axios 인터셉터​

// src/api/axios.ts
import axios from 'axios';
import { API_BASE_URL } from '../config/oauth';

// Axios 인스턴스 생성
export const api = axios.create({
    baseURL: API_BASE_URL,
    withCredentials: true, // Cookie 포함
    headers: {
        'Content-Type': 'application/json'
    }
});

// Response Interceptor (자동 토큰 갱신)
api.interceptors.response.use(
    response => response,
    async error => {
        const originalRequest = error.config;

        // 401 에러 && 재시도 안 했으면
        if (error.response?.status === 401 && !originalRequest._retry) {
            originalRequest._retry = true;

            try {
                // Refresh Token으로 갱신
                await axios.post(
                    `${API_BASE_URL}/auth/refresh`,
                    {},
                    { withCredentials: true }
                );

                // 원래 요청 재시도
                return api(originalRequest);
            } catch (refreshError) {
                // Refresh 실패 → 로그아웃
                window.location.href = '/login';
                return Promise.reject(refreshError);
            }
        }

        return Promise.reject(error);
    }
);

6. 콜백 페이지​

// src/pages/AuthCallback.tsx
import { useEffect } from 'react';
import { useNavigate, useSearchParams } from 'react-router-dom';
import { useOAuth } from '../hooks/useOAuth';

export const AuthCallback = () => {
    const navigate = useNavigate();
    const [searchParams] = useSearchParams();
    const { handleCallback } = useOAuth();

    useEffect(() => {
        const code = searchParams.get('code');
        const state = searchParams.get('state');
        const error = searchParams.get('error');

        if (error) {
            console.error('OAuth error:', error);
            navigate('/login?error=' + error);
            return;
        }

        if (code && state) {
            handleCallback(code, state).then(() => {
                navigate('/');
            });
        } else {
            navigate('/login');
        }
    }, [searchParams, handleCallback, navigate]);

    return (
        <div style={{ textAlign: 'center', marginTop: '100px' }}>
            <h2>로그인 처리 중...</h2>
            <p>잠시만 기다려주세요.</p>
        </div>
    );
};

7. 메인 앱​

// src/App.tsx
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom';
import { useOAuth } from './hooks/useOAuth';
import { AuthCallback } from './pages/AuthCallback';

const LoginPage = () => {
    const { login } = useOAuth();

    return (
        <div style={{ textAlign: 'center', marginTop: '100px' }}>
            <h1>OAuth SPA Demo</h1>
            <button
                onClick={login}
                style={{
                    padding: '12px 24px',
                    fontSize: '16px',
                    backgroundColor: '#4285f4',
                    color: 'white',
                    border: 'none',
                    borderRadius: '4px',
                    cursor: 'pointer'
                }}
            >
                Google로 로그인
            </button>
        </div>
    );
};

const HomePage = () => {
    const { user, logout } = useOAuth();

    return (
        <div style={{ padding: '40px' }}>
            <h1>환영합니다!</h1>
            {user && (
                <div>
                    <img src={user.picture} alt={user.name} width="80" style={{ borderRadius: '50%' }} />
                    <h2>{user.name}</h2>
                    <p>{user.email}</p>
                </div>
            )}
            <button onClick={logout}>로그아웃</button>
        </div>
    );
};

const ProtectedRoute = ({ children }: { children: React.ReactNode }) => {
    const { isAuthenticated, isLoading } = useOAuth();

    if (isLoading) {
        return <div>Loading...</div>;
    }

    return isAuthenticated ? <>{children}</> : <Navigate to="/login" />;
};

export const App = () => {
    return (
        <BrowserRouter>
            <Routes>
                <Route path="/login" element={<LoginPage />} />
                <Route path="/auth/callback" element={<AuthCallback />} />
                <Route
                    path="/"
                    element={
                        <ProtectedRoute>
                            <HomePage />
                        </ProtectedRoute>
                    }
                />
            </Routes>
        </BrowserRouter>
    );
};

🔐 백엔드 (Token Proxy)​

FastAPI 백엔드​

from fastapi import FastAPI, Request, Response, HTTPException
from fastapi.middleware.cors import CORSMiddleware
import httpx

app = FastAPI()

# CORS 설정
app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000"],  # Frontend URL
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/auth/token")
async def token_proxy(request: Request, response: Response):
    """Token Exchange Proxy"""

    body = await request.json()
    code = body.get('code')
    code_verifier = body.get('code_verifier')
    redirect_uri = body.get('redirect_uri')

    # Google Token Endpoint로 요청
    async with httpx.AsyncClient() as client:
        token_response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'grant_type': 'authorization_code',
                'code': code,
                'redirect_uri': redirect_uri,
                'client_id': GOOGLE_CLIENT_ID,
                'client_secret': GOOGLE_CLIENT_SECRET,  # Backend에서만!
                'code_verifier': code_verifier
            }
        )

        if token_response.status_code != 200:
            raise HTTPException(status_code=400, detail="Token exchange failed")

        tokens = token_response.json()

    # HttpOnly Cookie에 저장
    response.set_cookie(
        key="access_token",
        value=tokens['access_token'],
        httponly=True,
        secure=True,  # HTTPS
        samesite='lax',
        max_age=tokens.get('expires_in', 3600)
    )

    if 'refresh_token' in tokens:
        response.set_cookie(
            key="refresh_token",
            value=tokens['refresh_token'],
            httponly=True,
            secure=True,
            samesite='lax',
            max_age=2592000  # 30일
        )

    return {"status": "ok"}

@app.post("/auth/refresh")
async def refresh_proxy(request: Request, response: Response):
    """Refresh Token Proxy"""

    refresh_token = request.cookies.get('refresh_token')
    if not refresh_token:
        raise HTTPException(status_code=401)

    async with httpx.AsyncClient() as client:
        token_response = await client.post(
            'https://oauth2.googleapis.com/token',
            data={
                'grant_type': 'refresh_token',
                'refresh_token': refresh_token,
                'client_id': GOOGLE_CLIENT_ID,
                'client_secret': GOOGLE_CLIENT_SECRET
            }
        )

        if token_response.status_code != 200:
            raise HTTPException(status_code=401)

        tokens = token_response.json()

    # 새 Access Token 저장
    response.set_cookie(
        key="access_token",
        value=tokens['access_token'],
        httponly=True,
        secure=True,
        samesite='lax',
        max_age=tokens.get('expires_in', 3600)
    )

    return {"status": "ok"}

@app.get("/api/me")
async def get_me(request: Request):
    """현재 사용자 정보"""

    access_token = request.cookies.get('access_token')
    if not access_token:
        raise HTTPException(status_code=401)

    # UserInfo Endpoint 호출
    async with httpx.AsyncClient() as client:
        user_response = await client.get(
            'https://www.googleapis.com/oauth2/v2/userinfo',
            headers={'Authorization': f'Bearer {access_token}'}
        )

        if user_response.status_code != 200:
            raise HTTPException(status_code=401)

        return user_response.json()

@app.post("/auth/logout")
async def logout(response: Response):
    """로그아웃"""

    response.delete_cookie("access_token")
    response.delete_cookie("refresh_token")

    return {"status": "ok"}

🎨 BFF Pattern (선택)​

Frontend (SPA)
     ↓
Backend For Frontend (BFF)
     ↓
OAuth Provider / Resource Server

장점:

  • Client Secret 안전하게 보관
  • 토큰을 HttpOnly Cookie에 저장 (XSS 방지)
  • CORS 이슈 없음
  • 추가 보안 레이어

단점:

  • 추가 서버 필요
  • 복잡도 증가
