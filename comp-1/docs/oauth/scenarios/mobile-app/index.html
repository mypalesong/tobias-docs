// src/hooks/useAuth.ts
import { useState, useEffect, useCallback } from 'react';
import { authorize, refresh, revoke } from 'react-native-app-auth';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { oauthConfig } from '../config/oauth';

interface AuthState {
    accessToken: string | null;
    refreshToken: string | null;
    idToken: string | null;
    isAuthenticated: boolean;
    isLoading: boolean;
}

interface User {
    id: string;
    email: string;
    name: string;
    picture: string;
}

export const useAuth = () => {
    const [authState, setAuthState] = useState<AuthState>({
        accessToken: null,
        refreshToken: null,
        idToken: null,
        isAuthenticated: false,
        isLoading: true
    });

    const [user, setUser] = useState<User | null>(null);

    /**
     * ë¡œê·¸ì¸
     */
    const login = useCallback(async (provider: 'google' | 'github') => {
        try {
            // react-native-app-authê°€ PKCE ìë™ ì²˜ë¦¬
            const result = await authorize(oauthConfig[provider]);

            // í† í° ì €ì¥
            await AsyncStorage.setItem('access_token', result.accessToken);
            await AsyncStorage.setItem('refresh_token', result.refreshToken);
            if (result.idToken) {
                await AsyncStorage.setItem('id_token', result.idToken);
            }

            setAuthState({
                accessToken: result.accessToken,
                refreshToken: result.refreshToken,
                idToken: result.idToken || null,
                isAuthenticated: true,
                isLoading: false
            });

            // ì‚¬ìš©ì ì •ë³´ ê°€ì ¸ì˜¤ê¸°
            await fetchUser(result.accessToken, provider);

        } catch (error) {
            console.error('Login failed:', error);
            setAuthState(prev => ({
                ...prev,
                isLoading: false
            }));
        }
    }, []);

    /**
     * í† í° ê°±ì‹ 
     */
    const refreshAccessToken = useCallback(async (provider: 'google' | 'github') => {
        const refreshToken = authState.refreshToken;
        if (!refreshToken) return;

        try {
            const result = await refresh(oauthConfig[provider], {
                refreshToken
            });

            await AsyncStorage.setItem('access_token', result.accessToken);

            setAuthState(prev => ({
                ...prev,
                accessToken: result.accessToken
            }));

            return result.accessToken;

        } catch (error) {
            console.error('Token refresh failed:', error);
            await logout();
        }
    }, [authState.refreshToken]);

    /**
     * ë¡œê·¸ì•„ì›ƒ
     */
    const logout = useCallback(async () => {
        try {
            // Token revocation (ì„ íƒ)
            if (authState.accessToken) {
                await revoke(oauthConfig.google, {
                    tokenToRevoke: authState.accessToken,
                    sendClientId: true
                });
            }

            // ì €ì¥ì†Œ ì •ë¦¬
            await AsyncStorage.multiRemove([
                'access_token',
                'refresh_token',
                'id_token'
            ]);

            setAuthState({
                accessToken: null,
                refreshToken: null,
                idToken: null,
                isAuthenticated: false,
                isLoading: false
            });

            setUser(null);

        } catch (error) {
            console.error('Logout failed:', error);
        }
    }, [authState.accessToken]);

    /**
     * ì‚¬ìš©ì ì •ë³´ ê°€ì ¸ì˜¤ê¸°
     */
    const fetchUser = useCallback(async (
        accessToken: string,
        provider: 'google' | 'github'
    ) => {
        try {
            const endpoint = provider === 'google'
                ? 'https://www.googleapis.com/oauth2/v2/userinfo'
                : 'https://api.github.com/user';

            const response = await fetch(endpoint, {
                headers: {
                    'Authorization': `Bearer ${accessToken}`
                }
            });

            const data = await response.json();

            if (provider === 'google') {
                setUser({
                    id: data.id,
                    email: data.email,
                    name: data.name,
                    picture: data.picture
                });
            } else if (provider === 'github') {
                setUser({
                    id: data.id.toString(),
                    email: data.email,
                    name: data.name || data.login,
                    picture: data.avatar_url
                });
            }

        } catch (error) {
            console.error('Failed to fetch user:', error);
        }
    }, []);

    /**
     * ì´ˆê¸°í™” (ì €ì¥ëœ í† í° ë¡œë“œ)
     */
    useEffect(() => {
        const loadTokens = async () => {
            try {
                const [accessToken, refreshToken, idToken] = await AsyncStorage.multiGet([
                    'access_token',
                    'refresh_token',
                    'id_token'
                ]);

                const access = accessToken[1];
                const refresh = refreshToken[1];
                const id = idToken[1];

                if (access) {
                    setAuthState({
                        accessToken: access,
                        refreshToken: refresh,
                        idToken: id,
                        isAuthenticated: true,
                        isLoading: false
                    });

                    // ì‚¬ìš©ì ì •ë³´ ê°€ì ¸ì˜¤ê¸°
                    await fetchUser(access, 'google');
                } else {
                    setAuthState(prev => ({
                        ...prev,
                        isLoading: false
                    }));
                }
            } catch (error) {
                console.error('Failed to load tokens:', error);
                setAuthState(prev => ({
                    ...prev,
                    isLoading: false
                }));
            }
        };

        loadTokens();
    }, [fetchUser]);

    return {
        ...authState,
        user,
        login,
        logout,
        refreshAccessToken
    };
};

// OAuthManager.swift
import Foundation
import AuthenticationServices

class OAuthManager: NSObject, ObservableObject {
    @Published var isAuthenticated = false
    @Published var user: User?

    private var authSession: ASWebAuthenticationSession?

    // OAuth Config
    let clientId = "YOUR_CLIENT_ID"
    let redirectURI = "com.myapp:/oauth/callback"
    let authEndpoint = "https://accounts.google.com/o/oauth2/v2/auth"
    let tokenEndpoint = "https://oauth2.googleapis.com/token"

    func login() {
        // 1. PKCE Code Verifier
        let codeVerifier = generateCodeVerifier()
        let codeChallenge = generateCodeChallenge(verifier: codeVerifier)

        // 2. State
        let state = UUID().uuidString

        // 3. Authorization URL
        var components = URLComponents(string: authEndpoint)!
        components.queryItems = [
            URLQueryItem(name: "client_id", value: clientId),
            URLQueryItem(name: "redirect_uri", value: redirectURI),
            URLQueryItem(name: "response_type", value: "code"),
            URLQueryItem(name: "scope", value: "openid email profile"),
            URLQueryItem(name: "state", value: state),
            URLQueryItem(name: "code_challenge", value: codeChallenge),
            URLQueryItem(name: "code_challenge_method", value: "S256")
        ]

        // 4. ASWebAuthenticationSession
        authSession = ASWebAuthenticationSession(
            url: components.url!,
            callbackURLScheme: "com.myapp"
        ) { [weak self] callbackURL, error in
            guard error == nil,
                  let url = callbackURL,
                  let code = URLComponents(url: url, resolvingAgainstBaseURL: false)?
                    .queryItems?.first(where: { $0.name == "code" })?.value
            else {
                return
            }

            // Token êµí™˜
            self?.exchangeToken(code: code, codeVerifier: codeVerifier)
        }

        authSession?.presentationContextProvider = self
        authSession?.start()
    }

    private func exchangeToken(code: String, codeVerifier: String) {
        var request = URLRequest(url: URL(string: tokenEndpoint)!)
        request.httpMethod = "POST"
        request.setValue("application/x-www-form-urlencoded", forHTTPHeaderField: "Content-Type")

        let body = [
            "grant_type": "authorization_code",
            "code": code,
            "redirect_uri": redirectURI,
            "client_id": clientId,
            "code_verifier": codeVerifier
        ]

        request.httpBody = body.percentEncoded()

        URLSession.shared.dataTask(with: request) { data, response, error in
            guard let data = data else { return }

            do {
                let tokenResponse = try JSONDecoder().decode(TokenResponse.self, from: data)

                // Keychainì— ì €ì¥
                self.saveToKeychain(accessToken: tokenResponse.accessToken)

                DispatchQueue.main.async {
                    self.isAuthenticated = true
                }
            } catch {
                print("Token exchange failed: \\(error)")
            }
        }.resume()
    }

    private func generateCodeVerifier() -> String {
        var buffer = [UInt8](repeating: 0, count: 32)
        _ = SecRandomCopyBytes(kSecRandomDefault, buffer.count, &buffer)
        return Data(buffer).base64EncodedString()
            .replacingOccurrences(of: "+", with: "-")
            .replacingOccurrences(of: "/", with: "_")
            .replacingOccurrences(of: "=", with: "")
    }

    private func generateCodeChallenge(verifier: String) -> String {
        guard let data = verifier.data(using: .utf8) else { return "" }
        var buffer = [UInt8](repeating: 0, count: Int(CC_SHA256_DIGEST_LENGTH))
        data.withUnsafeBytes {
            _ = CC_SHA256($0.baseAddress, CC_LONG(data.count), &buffer)
        }
        return Data(buffer).base64EncodedString()
            .replacingOccurrences(of: "+", with: "-")
            .replacingOccurrences(of: "/", with: "_")
            .replacingOccurrences(of: "=", with: "")
    }
}

extension OAuthManager: ASWebAuthenticationPresentationContextProviding {

        return ASPresentationAnchor()
    }
}

// OAuthManager.kt
import android.content.Intent
import android.net.Uri
import androidx.browser.customtabs.CustomTabsIntent
import net.openid.appauth.*
import java.security.MessageDigest
import java.util.*

class OAuthManager(private val activity: AppCompatActivity) {

    private val clientId = "YOUR_CLIENT_ID"
    private val redirectUri = Uri.parse("com.myapp:/oauth/callback")

    private val serviceConfig = AuthorizationServiceConfiguration(
        Uri.parse("https://accounts.google.com/o/oauth2/v2/auth"), // Auth endpoint
        Uri.parse("https://oauth2.googleapis.com/token") // Token endpoint
    )

    fun login() {
        // PKCE
        val codeVerifier = CodeVerifierUtil.generateRandomCodeVerifier()
        val codeChallenge = CodeVerifierUtil.deriveCodeVerifierChallenge(codeVerifier)

        // Authorization Request
        val authRequest = AuthorizationRequest.Builder(
            serviceConfig,
            clientId,
            ResponseTypeValues.CODE,
            redirectUri
        )
            .setScope("openid email profile")
            .setCodeVerifier(codeVerifier, codeChallenge, "S256")
            .build()

        // Custom Tabsë¡œ ì‹œì‘
        val authService = AuthorizationService(activity)
        val authIntent = authService.getAuthorizationRequestIntent(authRequest)

        activity.startActivityForResult(authIntent, RC_AUTH)
    }

    fun handleAuthResponse(intent: Intent) {
        val response = AuthorizationResponse.fromIntent(intent)
        val error = AuthorizationException.fromIntent(intent)

        if (response != null) {
            // Token êµí™˜
            val authService = AuthorizationService(activity)
            authService.performTokenRequest(
                response.createTokenExchangeRequest()
            ) { tokenResponse, exception ->
                if (tokenResponse != null) {
                    // ì„±ê³µ
                    val accessToken = tokenResponse.accessToken
                    saveToken(accessToken)
                } else {
                    // ì‹¤íŒ¨
                    Log.e("OAuth", "Token exchange failed", exception)
                }
            }
        }
    }

    private fun saveToken(token: String) {
        // SharedPreferences (ì•”í˜¸í™” ê¶Œì¥)
        activity.getSharedPreferences("oauth", Context.MODE_PRIVATE)
            .edit()
            .putString("access_token", token)
            .apply()
    }

    companion object {
        const val RC_AUTH = 100
    }
}
