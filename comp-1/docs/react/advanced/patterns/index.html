디자인 패턴

React에서 자주 사용되는 디자인 패턴을 학습합니다.

Compound Components​

관련된 컴포넌트들을 함께 사용하는 패턴입니다.

function Select({ children, value, onChange }) {
  return (
    <div className="select">
      {React.Children.map(children, child =>
        React.cloneElement(child, { value, onChange })
      )}
    </div>
  );
}

function Option({ value, onChange, children, ...props }) {
  return (
    <div onClick={() => onChange(value)} {...props}>
      {children}
    </div>
  );
}

// 사용
function App() {
  const [selected, setSelected] = useState('apple');

  return (
    <Select value={selected} onChange={setSelected}>
      <Option value="apple">사과</Option>
      <Option value="banana">바나나</Option>
      <Option value="orange">오렌지</Option>
    </Select>
  );
}

Render Props​

컴포넌트 간 코드를 공유하기 위해 함수 props를 사용하는 패턴입니다.

function Mouse({ render }) {
  const [position, setPosition] = useState({ x: 0, y: 0 });

  useEffect(() => {
    const handleMouseMove = (e) => {
      setPosition({ x: e.clientX, y: e.clientY });
    };

    window.addEventListener('mousemove', handleMouseMove);
    return () => window.removeEventListener('mousemove', handleMouseMove);
  }, []);

  return render(position);
}

// 사용
function App() {
  return (
    <Mouse
      render={({ x, y }) => (
        <h1>마우스 위치: ({x}, {y})</h1>
      )}
    />
  );
}

Higher-Order Components (HOC)​

컴포넌트를 받아서 새 컴포넌트를 반환하는 함수입니다.

function withAuth(Component) {
  return function AuthenticatedComponent(props) {
    const { isAuthenticated } = useAuth();

    if (!isAuthenticated) {
      return <Redirect to="/login" />;
    }

    return <Component {...props} />;
  };
}

// 사용
const ProtectedDashboard = withAuth(Dashboard);

Container/Presentational Pattern​

로직과 UI를 분리하는 패턴입니다.

// Container: 로직 담당
function UserListContainer() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/users')
      .then(res => res.json())
      .then(data => {
        setUsers(data);
        setLoading(false);
      });
  }, []);

  return <UserListPresentation users={users} loading={loading} />;
}

// Presentation: UI 담당
function UserListPresentation({ users, loading }) {
  if (loading) return <div>로딩 중...</div>;

  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  );
}

Custom Hooks Pattern​

로직을 Custom Hook으로 추출하는 현대적인 패턴입니다.

function useUsers() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch('/api/users')
      .then(res => res.json())
      .then(data => {
        setUsers(data);
        setLoading(false);
      })
      .catch(err => {
        setError(err);
        setLoading(false);
      });
  }, []);

  return { users, loading, error };
}

// 사용
function UserList() {
  const { users, loading, error } = useUsers();

  if (loading) return <div>로딩 중...</div>;
  if (error) return <div>오류: {error.message}</div>;

  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  );
}
