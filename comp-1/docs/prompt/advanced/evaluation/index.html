평가 및 테스트

프롬프트의 품질을 측정하고 검증하는 방법

좋은 프롬프트를 만들었다면, 그것이 정말 효과적인지 평가해야 합니다. 이 가이드에서는 프롬프트 성능을 측정하고 개선점을 찾는 체계적인 방법을 다룹니다.

메트릭 측정 방법​

1. 정확도 (Accuracy)​

# 테스트 케이스
test_cases = [
    {
        "input": "2 + 2 =",
        "expected": "4",
        "actual": "4",
        "correct": True
    },
    {
        "input": "3 * 5 =",
        "expected": "15",
        "actual": "15",
        "correct": True
    },
    {
        "input": "10 / 2 =",
        "expected": "5",
        "actual": "5.0",  # 형식 다름
        "correct": False  # 엄격한 평가
    }
]

accuracy = sum(tc["correct"] for tc in test_cases) / len(test_cases)
# = 2/3 = 66.7%

2. 완전성 (Completeness)​

requirements = [
    "함수 코드",
    "타입 힌트",
    "Docstring",
    "에러 처리",
    "사용 예제"
]

fulfilled = [
    "함수 코드",  # ✓
    "타입 힌트",  # ✓
    "Docstring",  # ✓
    "에러 처리",  # ✓
    # 사용 예제 누락
]

completeness = len(fulfilled) / len(requirements)
# = 4/5 = 80%

3. 일관성 (Consistency)​

# 같은 프롬프트 5번 실행
results = [
    "FastAPI는 빠른 웹 프레임워크입니다.",
    "FastAPI는 빠른 웹 프레임워크입니다.",
    "FastAPI는 고성능 웹 프레임워크입니다.",  # 다름
    "FastAPI는 빠른 웹 프레임워크입니다.",
    "FastAPI는 빠른 웹 프레임워크입니다."
]

consistent_results = 4
consistency = consistent_results / len(results)
# = 4/5 = 80%

🤖 자동화된 평가​

스크립트 예제​

def evaluate_prompt(prompt, test_cases):
    """프롬프트를 테스트 케이스로 평가"""

    results = {
        "passed": 0,
        "failed": 0,
        "errors": []
    }

    for tc in test_cases:
        try:
            # AI에게 프롬프트 전송
            response = ai.complete(prompt + tc["input"])

            # 기대값과 비교
            if matches(response, tc["expected"]):
                results["passed"] += 1
            else:
                results["failed"] += 1
                results["errors"].append({
                    "input": tc["input"],
                    "expected": tc["expected"],
                    "actual": response
                })

        except Exception as e:
            results["failed"] += 1
            results["errors"].append({
                "input": tc["input"],
                "error": str(e)
            })

    results["accuracy"] = results["passed"] / len(test_cases)

    return results

# 사용
test_cases = [
    {"input": "2+2", "expected": "4"},
    {"input": "3*5", "expected": "15"},
    {"input": "10/2", "expected": "5"}
]

results = evaluate_prompt(my_prompt, test_cases)
print(f"정확도: {results['accuracy'] * 100}%")

A/B 테스트​

"""
## Prompt A: 기본형
'다음 텍스트를 요약하세요: {text}'

## Prompt B: 구체화
'다음 텍스트를 3-5문장으로 요약하세요.
핵심 주장과 주요 근거만 포함하세요.
텍스트: {text}'

## 테스트 데이터
texts = [긴 기사 10개]

## 평가
results_a = [evaluate(prompt_a.format(text=t)) for t in texts]
results_b = [evaluate(prompt_b.format(text=t)) for t in texts]

## 비교
metric_comparison = {
    "정확도": {
        "A": mean([r["accuracy"] for r in results_a]),
        "B": mean([r["accuracy"] for r in results_b])
    },
    "완전성": {
        "A": mean([r["completeness"] for r in results_a]),
        "B": mean([r["completeness"] for r in results_b])
    },
    "토큰": {
        "A": mean([r["tokens"] for r in results_a]),
        "B": mean([r["tokens"] for r in results_b])
    }
}

# 결과:
# 정확도: A=75%, B=90% → B 승
# 완전성: A=60%, B=95% → B 승
# 토큰: A=100, B=150 → A 승 (효율)

# 결론: 정확도가 중요하므로 B 선택
"""
