PostgreSQL 데이터 타입

올바른 데이터 타입 선택으로 성능과 무결성 보장

🔢 숫자 타입​

-- 정수
CREATE TABLE examples (
    -- 자동 증가 ID
    id BIGSERIAL PRIMARY KEY,     -- 8바이트, 추천!
    small_id SERIAL,               -- 4바이트 (21억 제한)

    -- 정수
    age SMALLINT,                  -- 2바이트 (-32K ~ 32K)
    count INTEGER,                 -- 4바이트 (-21억 ~ 21억)
    big_number BIGINT,             -- 8바이트

    -- 실수 (정밀도 중요 시)
    price NUMERIC(10, 2),          -- 10자리, 소수점 2자리 (금액!)
    percentage NUMERIC(5, 2),      -- 999.99%

    -- 실수 (정밀도 덜 중요 시)
    temperature REAL,              -- 4바이트 부동소수점
    distance DOUBLE PRECISION      -- 8바이트 부동소수점
);

-- ✅ 금액은 항상 NUMERIC
price NUMERIC(10, 2) NOT NULL  -- 99,999,999.99까지

-- ❌ 금액에 REAL/FLOAT 사용 금지 (반올림 오류)
price REAL  -- 절대 안됨!

📝 문자 타입​

CREATE TABLE text_examples (
    -- 고정 길이 (패딩 추가, 거의 안씀)
    country_code CHAR(2),          -- 'KR', 'US' (항상 2자)

    -- 가변 길이 (제한 있음)
    email VARCHAR(255),            -- 최대 255자
    username VARCHAR(50),          -- 최대 50자
    status VARCHAR(20),            -- 'active', 'inactive' 등

    -- 가변 길이 (제한 없음)
    content TEXT,                  -- 긴 내용
    description TEXT,              -- 설명

    -- ✅ 권장 패턴
    title VARCHAR(200),            -- 제한 필요 시
    bio TEXT                       -- 제한 불필요 시
);

📅 날짜/시간 타입​

CREATE TABLE datetime_examples (
    -- 날짜만
    birth_date DATE,               -- '2024-01-01'

    -- 시간만
    open_time TIME,                -- '09:00:00'

    -- 날짜 + 시간 (타임존 없음)
    created_at TIMESTAMP,          -- '2024-01-01 09:00:00'

    -- 날짜 + 시간 (타임존 포함) ✅ 추천!
    updated_at TIMESTAMPTZ,        -- '2024-01-01 09:00:00+00'

    -- 간격
    session_duration INTERVAL      -- '2 hours 30 minutes'
);

-- ✅ 추천: TIMESTAMPTZ (타임존 고려)
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP

-- 현재 시간
CURRENT_TIMESTAMP  -- 타임존 포함
NOW()              -- CURRENT_TIMESTAMP와 동일
CURRENT_DATE       -- 오늘 날짜만

CREATE TABLE boolean_examples (
    is_active BOOLEAN DEFAULT true,
    is_verified BOOLEAN DEFAULT false,
    has_accepted_terms BOOLEAN NOT NULL
);

-- 쿼리
WHERE is_active = true

🎨 특수 타입​

JSON​

CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200),

    -- JSON (텍스트로 저장)
    metadata JSON,

    -- JSONB (바이너리, 인덱싱 가능) ✅ 추천!
    attributes JSONB
);

-- JSONB 사용 예
INSERT INTO products (name, attributes) VALUES
('Laptop', '{"brand": "Apple", "ram": "16GB", "ssd": "512GB"}');

-- JSONB 쿼리
SELECT * FROM products
WHERE attributes->>'brand' = 'Apple';

-- JSONB 인덱스
CREATE INDEX idx_product_attributes ON products USING GIN (attributes);

ARRAY​

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200),
    tags TEXT[]  -- 배열
);

-- 삽입
INSERT INTO posts (title, tags) VALUES
('PostgreSQL Guide', ARRAY['database', 'postgresql', 'tutorial']);

-- 또는
INSERT INTO posts (title, tags) VALUES
('PostgreSQL Guide', '{"database", "postgresql", "tutorial"}');

-- 쿼리
SELECT * FROM posts WHERE 'postgresql' = ANY(tags);
SELECT * FROM posts WHERE tags && ARRAY['database', 'sql'];

UUID​

CREATE TABLE sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id BIGINT NOT NULL,
    token VARCHAR(255),
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 장점: 고유성 보장, 예측 불가
-- 단점: 16바이트, 가독성 낮음

💡 선택 가이드​

-- ID
id BIGSERIAL PRIMARY KEY  -- 일반적
id UUID PRIMARY KEY DEFAULT gen_random_uuid()  -- 분산 시스템

-- 이메일/사용자명
email VARCHAR(255) UNIQUE NOT NULL
username VARCHAR(50) UNIQUE NOT NULL

-- 금액 (정밀도 중요!)
price NUMERIC(10, 2) NOT NULL
discount_rate NUMERIC(5, 2)  -- 백분율

-- 상태/타입
status VARCHAR(20) DEFAULT 'pending'  -- 'pending', 'approved', 'rejected'

-- 긴 텍스트
content TEXT
description TEXT

-- 날짜/시간
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP

-- 불린
is_active BOOLEAN DEFAULT true
is_deleted BOOLEAN DEFAULT false

-- 유연한 데이터
metadata JSONB
tags TEXT[]
