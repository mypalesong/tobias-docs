기본 개념

데이터베이스 설계의 핵심 개념 이해하기

데이터베이스 설계의 기본을 제대로 이해하면, 복잡한 시스템도 자신있게 설계할 수 있습니다. 이 가이드에서는 테이블, 컬럼, 키, 관계 등 핵심 개념을 다룹니다.

🎯 학습 목표​

  • 테이블과 컬럼의 개념과 설계 원칙
  • 기본 키와 외래 키의 역할
  • 데이터베이스 관계의 종류
  • NULL과 기본값 처리

📊 테이블 (Table)​

개념​

테이블은 **행(Row)**과 **열(Column)**로 구성된 데이터 저장 단위입니다.

-- users 테이블
┌────┬───────────────────┬──────────┬──────────────────────┐
│ id │ email             │ username │ created_at           │
├────┼───────────────────┼──────────┼──────────────────────┤
│  1 │ alice@example.com │ alice    │ 2024-01-01 10:00:00 │
│  2 │ bob@example.com   │ bob      │ 2024-01-02 11:00:00 │
└────┴───────────────────┴──────────┴──────────────────────┘

테이블 생성​

CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,           -- 자동 증가 ID
    email VARCHAR(255) UNIQUE NOT NULL,  -- 고유한 이메일
    username VARCHAR(50) NOT NULL,       -- 사용자명
    password_hash VARCHAR(255) NOT NULL, -- 해시된 비밀번호
    is_active BOOLEAN DEFAULT true,      -- 활성 상태
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

네이밍 규칙​

table_naming = {
    "✅ 권장": {
        "복수형": "users, posts, orders (테이블은 여러 행 포함)",
        "소문자": "user_profiles (대소문자 구분 이슈 방지)",
        "언더스코어": "order_items (단어 구분)"
    },

    "❌ 비권장": {
        "단수형": "user (혼란 야기)",
        "대문자": "Users, USERS (일관성 부족)",
        "카멜케이스": "orderItems (SQL 컨벤션 위반)"
    }
}

📝 컬럼 (Column)​

컬럼 정의​

CREATE TABLE products (
    -- 필수 정보
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,

    -- 선택 정보
    description TEXT,

    -- 숫자 정보
    price NUMERIC(10, 2) NOT NULL,  -- 정밀한 금액
    stock_quantity INTEGER DEFAULT 0,

    -- 불린
    is_available BOOLEAN DEFAULT true,

    -- 날짜/시간
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

NOT NULL vs NULL​

-- ❌ 나쁜 예: 필수 정보에 NULL 허용
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255),  -- NULL 가능 → 이메일 없는 사용자?
    username VARCHAR(50)  -- NULL 가능 → 사용자명 없는 사용자?
);

-- ✅ 좋은 예: 필수 정보는 NOT NULL
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL,      -- 필수
    username VARCHAR(50) NOT NULL,    -- 필수
    bio TEXT,                          -- 선택 (NULL 허용)
    website VARCHAR(255)               -- 선택 (NULL 허용)
);

기본값 (DEFAULT)​

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,

    -- 상태 기본값
    status VARCHAR(20) DEFAULT 'draft',  -- 'draft', 'published', 'archived'

    -- 숫자 기본값
    view_count INTEGER DEFAULT 0,
    like_count INTEGER DEFAULT 0,

    -- 불린 기본값
    is_featured BOOLEAN DEFAULT false,

    -- 타임스탬프 기본값
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    published_at TIMESTAMP  -- NULL 허용 (아직 발행 안됨)
);

🔑 기본 키 (Primary Key)​

개념​

기본 키는 테이블의 각 행을 고유하게 식별하는 컬럼(들)입니다.

-- 방법 1: 인라인 정의
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL
);

-- 방법 2: 테이블 제약조건으로 정의
CREATE TABLE users (
    id BIGSERIAL,
    email VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
);

-- 방법 3: 복합 기본 키
CREATE TABLE order_items (
    order_id BIGINT,
    product_id BIGINT,
    quantity INTEGER NOT NULL,
    PRIMARY KEY (order_id, product_id)
);

기본 키 선택 전략​

-- ✅ 권장: BIGSERIAL (자동 증가)
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,  -- 8바이트, 최대 9천경
    ...
);

-- ⚠️ 주의: SERIAL (4바이트)
CREATE TABLE users (
    id SERIAL PRIMARY KEY,  -- 최대 21억, 부족할 수 있음
    ...
);

-- ✅ 대안: UUID
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    ...
);
-- 장점: 분산 시스템에서 충돌 없음
-- 단점: 16바이트 크기, 인덱스 성능 약간 저하

-- ❌ 비권장: 자연 키 (Natural Key)
CREATE TABLE users (
    email VARCHAR(255) PRIMARY KEY,  -- 변경 가능성, 외래 키 복잡도
    ...
);

🔗 외래 키 (Foreign Key)​

개념​

외래 키는 다른 테이블의 기본 키를 참조하는 컬럼입니다.

-- 부모 테이블
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL
);

-- 자식 테이블
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    title VARCHAR(200) NOT NULL,

    -- 외래 키 제약조건
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- 간단한 방법
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    title VARCHAR(200) NOT NULL
);

참조 무결성 옵션​

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    title VARCHAR(200) NOT NULL,

    -- ON DELETE 옵션
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE      -- 사용자 삭제 시 게시물도 삭제
        ON UPDATE CASCADE      -- 사용자 ID 변경 시 게시물도 변경
);

ON DELETE 옵션 비교​

-- 1. CASCADE: 부모 삭제 시 자식도 삭제
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE;
-- 사용자 삭제 → 해당 사용자의 모든 게시물 삭제

-- 2. SET NULL: 부모 삭제 시 자식의 FK를 NULL로
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL;
-- 사용자 삭제 → 게시물의 user_id가 NULL
-- (게시물은 남지만 작성자 정보 없음)

-- 3. RESTRICT (기본값): 자식이 있으면 부모 삭제 불가
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE RESTRICT;
-- 게시물이 있는 사용자는 삭제 불가

-- 4. NO ACTION: RESTRICT와 유사하지만 체크 시점 다름
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE NO ACTION;

-- 5. SET DEFAULT: 부모 삭제 시 기본값으로
FOREIGN KEY (user_id) REFERENCES users(id)
    ON DELETE SET DEFAULT
    DEFAULT 0;  -- 삭제된 사용자는 ID 0으로

실전 예제​

-- 전자상거래 예제
CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    -- 사용자는 유지하고 주문만 orphan 처리
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE RESTRICT  -- 주문이 있는 사용자 삭제 불가
);

CREATE TABLE order_items (
    id BIGSERIAL PRIMARY KEY,
    order_id BIGINT NOT NULL,
    product_id BIGINT NOT NULL,
    quantity INTEGER NOT NULL,
    price NUMERIC(10, 2) NOT NULL,

    -- 주문 삭제 시 주문 항목도 삭제
    FOREIGN KEY (order_id) REFERENCES orders(id)
        ON DELETE CASCADE,

    -- 상품은 유지 (주문 이력 보존)
    FOREIGN KEY (product_id) REFERENCES products(id)
        ON DELETE RESTRICT
);

🔄 데이터베이스 관계​

1:1 관계 (One-to-One)​

한 행이 다른 테이블의 정확히 한 행과 연결됩니다.

-- 사용자 기본 정보
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) NOT NULL
);

-- 사용자 상세 정보 (1:1)
CREATE TABLE user_profiles (
    user_id BIGINT PRIMARY KEY,  -- FK이자 PK
    bio TEXT,
    avatar_url VARCHAR(500),
    date_of_birth DATE,

    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
);

-- 또는
CREATE TABLE user_profiles (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT UNIQUE NOT NULL,  -- UNIQUE 보장
    bio TEXT,
    ...
    FOREIGN KEY (user_id) REFERENCES users(id)
);

1:N 관계 (One-to-Many)​

한 행이 다른 테이블의 여러 행과 연결됩니다.

-- 한 사용자, 여러 게시물
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL
);

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,  -- FK (중복 가능)
    title VARCHAR(200) NOT NULL,

    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- 사용자 1 → 게시물 N개
-- user_id = 1인 게시물이 여러 개 존재 가능

N:M 관계 (Many-to-Many)​

여러 행이 다른 테이블의 여러 행과 연결됩니다.

-- 학생 ↔ 강좌 (N:M)
CREATE TABLE students (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

CREATE TABLE courses (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL
);

-- 중간 테이블 (Junction Table)
CREATE TABLE student_courses (
    student_id BIGINT,
    course_id BIGINT,
    enrolled_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    grade VARCHAR(2),

    PRIMARY KEY (student_id, course_id),
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE
);

-- 학생 1 → 여러 강좌
-- 강좌 1 → 여러 학생

💡 실전 팁​

1. ID 컬럼 네이밍​

-- ✅ 권장: 일관된 네이밍
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,  -- 모든 테이블에서 'id'
    ...
);

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,  -- 'user' + '_id'
    ...
);

-- ⚠️ 비권장: 혼재
CREATE TABLE users (
    user_id BIGSERIAL PRIMARY KEY,  -- 'user_id'
    ...
);

CREATE TABLE posts (
    post_id BIGSERIAL PRIMARY KEY,  -- 'post_id'
    author_id BIGINT NOT NULL,      -- user_id가 아닌 author_id?
    ...
);

2. 타임스탬프는 필수​

-- ✅ 모든 테이블에 추가
CREATE TABLE any_table (
    id BIGSERIAL PRIMARY KEY,
    ...
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 트리거로 updated_at 자동 갱신
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_user_updated_at
    BEFORE UPDATE ON users
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

3. Soft Delete 고려​

CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL,
    username VARCHAR(50) NOT NULL,

    -- Soft Delete
    deleted_at TIMESTAMP,  -- NULL = 활성, NOT NULL = 삭제됨

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 활성 사용자만 조회
SELECT * FROM users WHERE deleted_at IS NULL;

-- Soft Delete 실행
UPDATE users SET deleted_at = CURRENT_TIMESTAMP WHERE id = 1;

✅ 체크리스트​

테이블 설계​

  • 테이블명은 복수형, 소문자, 언더스코어
  • 모든 테이블에 기본 키 존재
  • created_at, updated_at 컬럼 포함

컬럼 설계​

  • 필수 컬럼은 NOT NULL
  • 적절한 기본값 설정
  • NULL의 의미가 명확

관계 설계​

  • 외래 키 제약조건 정의
  • ON DELETE 동작 명시
  • N:M 관계는 중간 테이블 사용
