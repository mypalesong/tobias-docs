정규화

데이터 중복 제거와 무결성 보장

정규화는 데이터 중복을 최소화하고 무결성을 보장하기 위해 테이블을 구조화하는 과정입니다.

🎯 정규화의 목적​

normalization_goals = {
    "중복 제거": "같은 데이터를 여러 곳에 저장하지 않음",
    "갱신 이상 방지": "데이터 수정 시 일관성 유지",
    "삽입 이상 방지": "불필요한 데이터 없이 삽입 가능",
    "삭제 이상 방지": "필요한 데이터가 함께 삭제되지 않음"
}

📊 정규화 단계​

비정규화 (0NF)​

-- ❌ 중복과 문제가 많은 테이블
CREATE TABLE orders_denormalized (
    order_id INTEGER,
    customer_name VARCHAR(100),
    customer_email VARCHAR(255),
    customer_phone VARCHAR(20),
    products TEXT,  -- "Apple,Banana,Orange" (반정규화!)
    total_price NUMERIC(10, 2)
);

-- 문제점:
-- 1. 고객 정보 중복 (같은 고객의 여러 주문)
-- 2. products 컬럼에 여러 값 (쿼리 어려움)
-- 3. 고객 정보 변경 시 모든 주문 수정 필요

제1정규형 (1NF)​

규칙: 모든 컬럼은 원자값(Atomic Value)을 가져야 함

-- ✅ 1NF: 원자값 보장
CREATE TABLE orders_1nf (
    order_id INTEGER,
    customer_name VARCHAR(100),
    customer_email VARCHAR(255),
    customer_phone VARCHAR(20),
    product_name VARCHAR(100),  -- 한 제품씩
    product_price NUMERIC(10, 2),
    quantity INTEGER
);

-- 여전히 문제:
-- 고객 정보가 여러 행에 중복

제2정규형 (2NF)​

규칙: 1NF + 부분 함수 종속 제거

-- ✅ 2NF: 고객 정보 분리
CREATE TABLE customers (
    customer_id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20)
);

CREATE TABLE orders (
    order_id BIGSERIAL PRIMARY KEY,
    customer_id BIGINT REFERENCES customers(id),
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE order_items (
    order_id BIGINT REFERENCES orders(id),
    product_name VARCHAR(100),
    product_price NUMERIC(10, 2),
    quantity INTEGER,
    PRIMARY KEY (order_id, product_name)
);

-- 여전히 문제:
-- 제품 정보 중복

제3정규형 (3NF)​

규칙: 2NF + 이행적 함수 종속 제거

-- ✅ 3NF: 제품 정보 분리
CREATE TABLE customers (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20)
);

CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price NUMERIC(10, 2) NOT NULL,
    description TEXT
);

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    customer_id BIGINT NOT NULL REFERENCES customers(id),
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'pending'
);

CREATE TABLE order_items (
    order_id BIGINT REFERENCES orders(id),
    product_id BIGINT REFERENCES products(id),
    quantity INTEGER NOT NULL,
    price_at_time NUMERIC(10, 2) NOT NULL,  -- 주문 당시 가격 보존
    PRIMARY KEY (order_id, product_id)
);

🔄 역정규화 (Denormalization)​

성능을 위해 의도적으로 중복을 허용하는 경우

-- 정규화된 구조
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(id),
    title VARCHAR(200),
    content TEXT
);

CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT REFERENCES posts(id)
);

-- ❌ 댓글 수를 세려면 JOIN 필요 (느림)
SELECT p.*, COUNT(c.id) as comment_count
FROM posts p
LEFT JOIN comments c ON p.id = c.post_id
GROUP BY p.id;

-- ✅ 역정규화: comment_count 컬럼 추가
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(id),
    title VARCHAR(200),
    content TEXT,
    comment_count INTEGER DEFAULT 0  -- 역정규화!
);

-- 트리거로 자동 갱신
CREATE OR REPLACE FUNCTION update_post_comment_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE posts SET comment_count = comment_count + 1
        WHERE id = NEW.post_id;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE posts SET comment_count = comment_count - 1
        WHERE id = OLD.post_id;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_comment_count
AFTER INSERT OR DELETE ON comments
FOR EACH ROW EXECUTE FUNCTION update_post_comment_count();
