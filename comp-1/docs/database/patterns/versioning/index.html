버전 관리

데이터 변경 이력과 시점 복원을 위한 버전 관리 시스템

버전 관리(Versioning)는 데이터의 모든 변경을 추적하고 특정 시점의 상태로 되돌릴 수 있게 하는 패턴입니다.

🎯 버전 관리가 필요한 이유​

🔧 기본 패턴​

패턴 1: 별도 버전 테이블​

가장 일반적인 패턴입니다.

-- 현재 버전 (메인 테이블)
CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    author_id BIGINT NOT NULL REFERENCES users(id),

    -- 버전 관리
    version INTEGER NOT NULL DEFAULT 1,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 버전 이력
CREATE TABLE document_versions (
    id BIGSERIAL PRIMARY KEY,
    document_id BIGINT NOT NULL REFERENCES documents(id) ON DELETE CASCADE,

    -- 버전 정보
    version INTEGER NOT NULL,
    title VARCHAR(200) NOT NULL,
    content TEXT,

    -- 메타데이터
    author_id BIGINT NOT NULL REFERENCES users(id),
    change_summary TEXT,  -- 변경 요약
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (document_id, version)
);

CREATE INDEX idx_document_versions_document ON document_versions(document_id);
CREATE INDEX idx_document_versions_version ON document_versions(document_id, version);

-- 트리거: 변경 시 자동으로 버전 저장
CREATE OR REPLACE FUNCTION save_document_version()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'UPDATE' THEN
        -- 이전 버전 저장
        INSERT INTO document_versions (
            document_id, version, title, content, author_id
        )
        VALUES (
            OLD.id, OLD.version, OLD.title, OLD.content, OLD.author_id
        );

        -- 버전 번호 증가
        NEW.version := OLD.version + 1;
        NEW.updated_at := CURRENT_TIMESTAMP;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER document_versioning
BEFORE UPDATE ON documents
FOR EACH ROW
EXECUTE FUNCTION save_document_version();

-- 사용 예
INSERT INTO documents (title, content, author_id) VALUES
('My Document', 'Initial content', 1);

UPDATE documents SET content = 'Updated content' WHERE id = 1;
-- document_versions에 v1 저장됨, documents는 v2

UPDATE documents SET title = 'New Title', content = 'New content' WHERE id = 1;
-- document_versions에 v2 저장됨, documents는 v3

-- 특정 버전 조회
SELECT * FROM document_versions
WHERE document_id = 1 AND version = 2;

-- 모든 버전 조회
SELECT version, title, created_at
FROM document_versions
WHERE document_id = 1
ORDER BY version DESC;

-- 특정 시점의 버전 조회
SELECT * FROM document_versions
WHERE document_id = 1
  AND created_at <= '2024-01-15 10:00:00'
ORDER BY created_at DESC
LIMIT 1;

패턴 2: 단일 테이블 버전 관리​

현재와 이전 버전을 모두 한 테이블에 저장합니다.

CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    document_id BIGINT NOT NULL,  -- 논리적 문서 ID
    version INTEGER NOT NULL,

    -- 데이터
    title VARCHAR(200) NOT NULL,
    content TEXT,
    author_id BIGINT NOT NULL REFERENCES users(id),

    -- 버전 상태
    is_current BOOLEAN DEFAULT false,  -- 현재 버전 여부
    valid_from TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    valid_to TIMESTAMPTZ,  -- NULL = 현재 버전

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (document_id, version)
);

-- 인덱스
CREATE INDEX idx_documents_current ON documents(document_id) WHERE is_current = true;
CREATE INDEX idx_documents_valid ON documents(document_id, valid_from, valid_to);

-- 시퀀스 (논리적 document_id 생성)
CREATE SEQUENCE document_id_seq;

-- 새 문서 생성
INSERT INTO documents (document_id, version, title, content, author_id, is_current)
VALUES (nextval('document_id_seq'), 1, 'My Document', 'Content', 1, true);

-- 새 버전 생성 함수
CREATE OR REPLACE FUNCTION create_new_document_version(
    p_document_id BIGINT,
    p_title VARCHAR(200),
    p_content TEXT,
    p_author_id BIGINT
) RETURNS BIGINT AS $$
DECLARE
    v_new_version INTEGER;
    v_new_id BIGINT;
BEGIN
    -- 현재 버전 번호 조회
    SELECT MAX(version) + 1 INTO v_new_version
    FROM documents
    WHERE document_id = p_document_id;

    -- 기존 현재 버전 비활성화
    UPDATE documents
    SET
        is_current = false,
        valid_to = CURRENT_TIMESTAMP
    WHERE document_id = p_document_id
      AND is_current = true;

    -- 새 버전 생성
    INSERT INTO documents (
        document_id, version, title, content, author_id, is_current
    )
    VALUES (
        p_document_id, v_new_version, p_title, p_content, p_author_id, true
    )
    RETURNING id INTO v_new_id;

    RETURN v_new_id;
END;
$$ LANGUAGE plpgsql;

-- 사용
SELECT create_new_document_version(1, 'Updated Title', 'Updated Content', 1);

-- 현재 버전 조회
SELECT * FROM documents
WHERE document_id = 1 AND is_current = true;

-- 모든 버전 조회
SELECT version, title, valid_from, valid_to
FROM documents
WHERE document_id = 1
ORDER BY version DESC;

-- 특정 시점의 버전 조회
SELECT * FROM documents
WHERE document_id = 1
  AND valid_from <= '2024-01-15 10:00:00'
  AND (valid_to IS NULL OR valid_to > '2024-01-15 10:00:00')
LIMIT 1;

🎨 고급 패턴​

차분(Delta) 저장​

전체 데이터 대신 변경된 부분만 저장합니다.

CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    version INTEGER NOT NULL DEFAULT 1,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE document_deltas (
    id BIGSERIAL PRIMARY KEY,
    document_id BIGINT NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
    version INTEGER NOT NULL,

    -- 변경 사항만 저장
    changes JSONB NOT NULL,  -- {"title": {"old": "...", "new": "..."}}

    author_id BIGINT NOT NULL REFERENCES users(id),
    change_summary TEXT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 변경 사항 기록 트리거
CREATE OR REPLACE FUNCTION save_document_delta()
RETURNS TRIGGER AS $$
DECLARE
    delta JSONB := '{}'::jsonb;
BEGIN
    -- 변경된 필드만 기록
    IF OLD.title IS DISTINCT FROM NEW.title THEN
        delta := delta || jsonb_build_object(
            'title',
            jsonb_build_object('old', OLD.title, 'new', NEW.title)
        );
    END IF;

    IF OLD.content IS DISTINCT FROM NEW.content THEN
        delta := delta || jsonb_build_object(
            'content',
            jsonb_build_object('old', OLD.content, 'new', NEW.content)
        );
    END IF;

    -- 변경사항이 있으면 저장
    IF delta != '{}'::jsonb THEN
        INSERT INTO document_deltas (document_id, version, changes, author_id)
        VALUES (NEW.id, OLD.version, delta,
                current_setting('app.current_user_id', true)::BIGINT);

        NEW.version := OLD.version + 1;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER document_delta_tracking
BEFORE UPDATE ON documents
FOR EACH ROW
EXECUTE FUNCTION save_document_delta();

-- 특정 버전으로 복원
CREATE OR REPLACE FUNCTION restore_document_version(
    p_document_id BIGINT,
    p_target_version INTEGER
) RETURNS void AS $$
DECLARE
    delta_record RECORD;
    current_doc RECORD;
    restored_title VARCHAR(200);
    restored_content TEXT;
BEGIN
    -- 현재 문서 조회
    SELECT * INTO current_doc FROM documents WHERE id = p_document_id;

    restored_title := current_doc.title;
    restored_content := current_doc.content;

    -- 목표 버전부터 현재까지 역순으로 되돌리기
    FOR delta_record IN
        SELECT changes
        FROM document_deltas
        WHERE document_id = p_document_id
          AND version > p_target_version
        ORDER BY version DESC
    LOOP
        -- 변경사항 되돌리기
        IF delta_record.changes ? 'title' THEN
            restored_title := delta_record.changes->'title'->>'old';
        END IF;

        IF delta_record.changes ? 'content' THEN
            restored_content := delta_record.changes->'content'->>'old';
        END IF;
    END LOOP;

    -- 문서 업데이트
    UPDATE documents
    SET title = restored_title,
        content = restored_content,
        version = p_target_version
    WHERE id = p_document_id;
END;
$$ LANGUAGE plpgsql;

분기(Branching) 지원​

Git처럼 분기를 지원합니다.

CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    document_id BIGINT NOT NULL,  -- 논리적 ID
    branch VARCHAR(50) NOT NULL DEFAULT 'main',
    version INTEGER NOT NULL,

    -- 부모 버전 (분기 추적)
    parent_id BIGINT REFERENCES documents(id),

    title VARCHAR(200) NOT NULL,
    content TEXT,
    author_id BIGINT NOT NULL REFERENCES users(id),

    is_current BOOLEAN DEFAULT false,  -- 브랜치의 현재 버전
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (document_id, branch, version)
);

CREATE INDEX idx_documents_branch ON documents(document_id, branch);

-- 새 브랜치 생성
CREATE OR REPLACE FUNCTION create_branch(
    p_document_id BIGINT,
    p_from_version_id BIGINT,
    p_new_branch VARCHAR(50),
    p_author_id BIGINT
) RETURNS BIGINT AS $$
DECLARE
    v_source RECORD;
    v_new_id BIGINT;
BEGIN
    -- 원본 버전 조회
    SELECT * INTO v_source FROM documents WHERE id = p_from_version_id;

    -- 새 브랜치 첫 버전 생성
    INSERT INTO documents (
        document_id, branch, version, parent_id,
        title, content, author_id, is_current
    )
    VALUES (
        p_document_id, p_new_branch, 1, p_from_version_id,
        v_source.title, v_source.content, p_author_id, true
    )
    RETURNING id INTO v_new_id;

    RETURN v_new_id;
END;
$$ LANGUAGE plpgsql;

-- 브랜치 병합
CREATE OR REPLACE FUNCTION merge_branch(
    p_document_id BIGINT,
    p_source_branch VARCHAR(50),
    p_target_branch VARCHAR(50),
    p_author_id BIGINT
) RETURNS BIGINT AS $$
DECLARE
    v_source RECORD;
    v_target RECORD;
    v_new_version INTEGER;
    v_new_id BIGINT;
BEGIN
    -- 소스 브랜치의 현재 버전
    SELECT * INTO v_source FROM documents
    WHERE document_id = p_document_id
      AND branch = p_source_branch
      AND is_current = true;

    -- 타겟 브랜치의 현재 버전
    SELECT * INTO v_target FROM documents
    WHERE document_id = p_document_id
      AND branch = p_target_branch
      AND is_current = true;

    v_new_version := v_target.version + 1;

    -- 타겟 브랜치 비활성화
    UPDATE documents
    SET is_current = false
    WHERE id = v_target.id;

    -- 병합된 새 버전 생성
    INSERT INTO documents (
        document_id, branch, version, parent_id,
        title, content, author_id, is_current
    )
    VALUES (
        p_document_id, p_target_branch, v_new_version, v_source.id,
        v_source.title, v_source.content, p_author_id, true
    )
    RETURNING id INTO v_new_id;

    RETURN v_new_id;
END;
$$ LANGUAGE plpgsql;

-- 사용 예
-- 1. main 브랜치에 문서 생성
INSERT INTO documents (document_id, branch, version, title, content, author_id, is_current)
VALUES (1, 'main', 1, 'My Document', 'Content', 1, true);

-- 2. feature 브랜치 생성
SELECT create_branch(1, 1, 'feature', 1);

-- 3. feature 브랜치에서 작업
-- ...

-- 4. main으로 병합
SELECT merge_branch(1, 'feature', 'main', 1);

동시성 제어​

-- 낙관적 잠금 (Optimistic Locking)
CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    version INTEGER NOT NULL DEFAULT 1,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 업데이트 시 버전 확인
CREATE OR REPLACE FUNCTION update_with_version_check(
    p_id BIGINT,
    p_title VARCHAR(200),
    p_content TEXT,
    p_expected_version INTEGER
) RETURNS BOOLEAN AS $$
DECLARE
    v_updated INTEGER;
BEGIN
    UPDATE documents
    SET
        title = p_title,
        content = p_content,
        version = version + 1,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_id
      AND version = p_expected_version;  -- 버전 확인

    GET DIAGNOSTICS v_updated = ROW_COUNT;

    IF v_updated = 0 THEN
        RAISE EXCEPTION 'Conflict: Document has been modified by another user';
    END IF;

    RETURN true;
END;
$$ LANGUAGE plpgsql;

-- 사용 (애플리케이션에서)
"""
# 문서 조회
doc = await conn.fetchrow("SELECT * FROM documents WHERE id = $1", doc_id)

        doc_id, new_title, new_content, current_version
    )

📊 쿼리 패턴​

버전 비교​

-- 두 버전 간 차이
WITH v1 AS (
    SELECT * FROM document_versions WHERE document_id = 1 AND version = 1
),
v2 AS (
    SELECT * FROM document_versions WHERE document_id = 1 AND version = 2
)
SELECT
    CASE
        WHEN v1.title != v2.title THEN 'title changed'
        WHEN v1.content != v2.content THEN 'content changed'
    END as change_type,
    v1.title as old_title,
    v2.title as new_title,
    v1.content as old_content,
    v2.content as new_content
FROM v1, v2;

-- 델타로 변경 이력
SELECT
    version,
    changes->>'field' as field,
    changes->'old' as old_value,
    changes->'new' as new_value,
    created_at
FROM document_deltas
WHERE document_id = 1
ORDER BY version;

버전 통계​

-- 문서별 버전 수
SELECT
    d.id,
    d.title,
    COUNT(dv.id) as version_count,
    MAX(dv.created_at) as last_modified
FROM documents d
LEFT JOIN document_versions dv ON d.id = dv.document_id
GROUP BY d.id, d.title
ORDER BY version_count DESC;

-- 가장 활발한 문서
SELECT
    document_id,
    COUNT(*) as edit_count,
    COUNT(DISTINCT author_id) as contributor_count,
    MAX(created_at) - MIN(created_at) as lifetime
FROM document_versions
GROUP BY document_id
ORDER BY edit_count DESC
LIMIT 10;

-- 사용자별 편집 통계
SELECT
    u.username,
    COUNT(*) as edit_count,
    COUNT(DISTINCT dv.document_id) as documents_edited
FROM document_versions dv
JOIN users u ON dv.author_id = u.id
GROUP BY u.id, u.username
ORDER BY edit_count DESC;

🗄️ 성능 최적화​

파티셔닝​

-- 오래된 버전은 별도 파티션
CREATE TABLE document_versions (
    id BIGSERIAL,
    document_id BIGINT NOT NULL,
    version INTEGER NOT NULL,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
) PARTITION BY RANGE (created_at);

-- 현재 버전 (자주 접근)
CREATE TABLE document_versions_current PARTITION OF document_versions
FOR VALUES FROM ('2024-01-01') TO ('2025-01-01');

-- 과거 버전 (드물게 접근)
CREATE TABLE document_versions_2023 PARTITION OF document_versions
FOR VALUES FROM ('2023-01-01') TO ('2024-01-01');

압축​

-- 오래된 버전 압축 (TOAST)
ALTER TABLE document_versions
ALTER COLUMN content SET STORAGE EXTERNAL;  -- 압축 활성화

-- 또는 애플리케이션 레벨에서 압축
CREATE TABLE document_versions (
    id BIGSERIAL PRIMARY KEY,
    document_id BIGINT NOT NULL,
    version INTEGER NOT NULL,
    title VARCHAR(200) NOT NULL,
    content_compressed BYTEA,  -- gzip 압축된 내용
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

✅ 버전 관리 체크리스트​

설계​

  • 버전 관리 전략 선택 (별도 테이블 vs 단일 테이블)
  • 전체 저장 vs 차분 저장 결정
  • 보존 기간 정책
  • 동시성 제어 방식

기능​

  • 버전 생성 자동화 (트리거)
  • 버전 조회 쿼리
  • 버전 비교 기능
  • 특정 버전 복원 기능

성능​

  • 인덱스 최적화
  • 파티셔닝 (대용량)
  • 오래된 버전 아카이브
  • 압축 고려

보안​

  • 버전 접근 권한
  • 변경 추적 (누가)
  • 무결성 보장
