감사 로그

데이터 변경 이력을 추적하는 감사 추적 시스템

감사 로그(Audit Log)는 누가, 언제, 무엇을, 왜 변경했는지 기록하는 시스템입니다. 보안, 규정 준수, 디버깅에 필수적입니다.

🎯 감사 로그가 필요한 이유​

주요 목적​

audit_log_purposes = {
    "보안": "무단 접근 및 변경 탐지",
    "규정 준수": "SOX, GDPR, HIPAA 등 법적 요구사항",
    "디버깅": "데이터 손상 원인 추적",
    "복구": "이전 상태로 롤백",
    "분석": "사용 패턴 및 트렌드 분석",
    "책임 추적": "누가 변경했는지 명확히"
}

무엇을 기록할까?​

audit_log_contents = {
    "Who (누가)": "사용자 ID, IP 주소, 세션 정보",
    "When (언제)": "타임스탬프 (UTC)",
    "What (무엇을)": "변경된 테이블, 레코드 ID, 필드",
    "How (어떻게)": "INSERT, UPDATE, DELETE",
    "Before/After (전/후)": "변경 전 값, 변경 후 값",
    "Why (왜)": "변경 이유, 요청 ID"
}

🔧 기본 구현​

단순 감사 로그​

-- 감사 대상 테이블
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    role VARCHAR(20) DEFAULT 'user',
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 감사 로그 테이블
CREATE TABLE audit_logs (
    id BIGSERIAL PRIMARY KEY,

    -- 대상
    table_name VARCHAR(100) NOT NULL,
    record_id BIGINT NOT NULL,

    -- 작업
    operation VARCHAR(10) NOT NULL,  -- 'INSERT', 'UPDATE', 'DELETE'

    -- 변경 내용
    old_data JSONB,  -- 변경 전 (UPDATE, DELETE)
    new_data JSONB,  -- 변경 후 (INSERT, UPDATE)

    -- 누가
    user_id BIGINT,  -- 변경한 사용자
    ip_address INET,

    -- 언제
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    -- 인덱스
    INDEX idx_audit_logs_table_record (table_name, record_id),
    INDEX idx_audit_logs_user (user_id),
    INDEX idx_audit_logs_created (created_at)
);

-- 트리거 함수
CREATE OR REPLACE FUNCTION audit_log_trigger()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO audit_logs (table_name, record_id, operation, new_data)
        VALUES (TG_TABLE_NAME, NEW.id, 'INSERT', row_to_json(NEW)::jsonb);
        RETURN NEW;

    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO audit_logs (table_name, record_id, operation, old_data, new_data)
        VALUES (TG_TABLE_NAME, NEW.id, 'UPDATE',
                row_to_json(OLD)::jsonb, row_to_json(NEW)::jsonb);
        RETURN NEW;

    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO audit_logs (table_name, record_id, operation, old_data)
        VALUES (TG_TABLE_NAME, OLD.id, 'DELETE', row_to_json(OLD)::jsonb);
        RETURN OLD;
    END IF;
END;
$$ LANGUAGE plpgsql;

-- 트리거 설정
CREATE TRIGGER users_audit
AFTER INSERT OR UPDATE OR DELETE ON users
FOR EACH ROW EXECUTE FUNCTION audit_log_trigger();

-- 사용 예
INSERT INTO users (email, username) VALUES
('alice@example.com', 'alice');
-- audit_logs에 INSERT 기록

UPDATE users SET role = 'admin' WHERE id = 1;
-- audit_logs에 UPDATE 기록 (변경 전/후)

DELETE FROM users WHERE id = 1;
-- audit_logs에 DELETE 기록

조회 예시​

-- 특정 레코드의 변경 이력
SELECT
    operation,
    old_data,
    new_data,
    created_at
FROM audit_logs
WHERE table_name = 'users'
  AND record_id = 1
ORDER BY created_at DESC;

-- 특정 사용자가 수행한 변경

    record_id,
    operation,
    created_at
FROM audit_logs
WHERE user_id = 100
ORDER BY created_at DESC;

-- 최근 24시간 변경 내역

WHERE created_at > CURRENT_TIMESTAMP - INTERVAL '24 hours'
ORDER BY created_at DESC;

🎨 고급 패턴​

필드별 변경 추적​

-- 변경된 필드만 기록
CREATE TABLE audit_logs (
    id BIGSERIAL PRIMARY KEY,
    table_name VARCHAR(100) NOT NULL,
    record_id BIGINT NOT NULL,
    field_name VARCHAR(100) NOT NULL,  -- 변경된 필드명
    old_value TEXT,
    new_value TEXT,
    user_id BIGINT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE OR REPLACE FUNCTION audit_field_changes()
RETURNS TRIGGER AS $$
DECLARE
    old_rec jsonb;
    new_rec jsonb;
    key text;
BEGIN
    IF TG_OP = 'UPDATE' THEN
        old_rec := row_to_json(OLD)::jsonb;
        new_rec := row_to_json(NEW)::jsonb;

        -- 각 필드 비교
        FOR key IN SELECT jsonb_object_keys(new_rec)
        LOOP
            IF old_rec->key IS DISTINCT FROM new_rec->key THEN
                INSERT INTO audit_logs (
                    table_name, record_id, field_name,
                    old_value, new_value, user_id
                )
                VALUES (
                    TG_TABLE_NAME,
                    NEW.id,
                    key,
                    old_rec->>key,
                    new_rec->>key,
                    current_setting('app.current_user_id', true)::BIGINT
                );
            END IF;
        END LOOP;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 조회: 특정 필드의 변경 이력
SELECT
    old_value,
    new_value,
    user_id,
    created_at
FROM audit_logs
WHERE table_name = 'users'
  AND record_id = 1
  AND field_name = 'role'
ORDER BY created_at DESC;

);

-- 세션 변수로 컨텍스트 전달
CREATE OR REPLACE FUNCTION audit_with_context()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO audit_logs (
        table_name,
        record_id,
        operation,
        old_data,
        new_data,
        user_id,
        username,
        ip_address,
        request_id,
        reason
    )
    VALUES (
        TG_TABLE_NAME,
        COALESCE(NEW.id, OLD.id),
        TG_OP,
        CASE WHEN TG_OP IN ('UPDATE', 'DELETE') THEN row_to_json(OLD)::jsonb END,
        CASE WHEN TG_OP IN ('INSERT', 'UPDATE') THEN row_to_json(NEW)::jsonb END,
        current_setting('app.current_user_id', true)::BIGINT,
        current_setting('app.current_username', true),
        current_setting('app.client_ip', true)::INET,
        current_setting('app.request_id', true)::UUID,
        current_setting('app.change_reason', true)
    );

    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

민감 정보 마스킹​

-- 민감한 필드는 마스킹하여 기록
CREATE OR REPLACE FUNCTION audit_with_masking()
RETURNS TRIGGER AS $$
DECLARE
    old_data jsonb;
    new_data jsonb;
    sensitive_fields text[] := ARRAY['password_hash', 'ssn', 'credit_card'];
    field text;
BEGIN
    old_data := row_to_json(OLD)::jsonb;
    new_data := row_to_json(NEW)::jsonb;

    -- 민감한 필드 마스킹
    FOREACH field IN ARRAY sensitive_fields
    LOOP
        IF old_data ? field THEN
            old_data := old_data || jsonb_build_object(field, '***REDACTED***');
        END IF;
        IF new_data ? field THEN
            new_data := new_data || jsonb_build_object(field, '***REDACTED***');
        END IF;
    END LOOP;

    INSERT INTO audit_logs (table_name, record_id, operation, old_data, new_data)
    VALUES (TG_TABLE_NAME, COALESCE(NEW.id, OLD.id), TG_OP, old_data, new_data);

    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

테이블별 감사 로그​

-- users 전용 감사 로그
CREATE TABLE users_audit (
    audit_id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    operation VARCHAR(10) NOT NULL,

    -- 변경된 필드만 저장
    email_old VARCHAR(255),
    email_new VARCHAR(255),
    username_old VARCHAR(50),
    username_new VARCHAR(50),
    role_old VARCHAR(20),
    role_new VARCHAR(20),

    changed_by BIGINT REFERENCES users(id),
    changed_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 트리거
CREATE OR REPLACE FUNCTION users_audit_trigger()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'UPDATE' THEN
        INSERT INTO users_audit (
            user_id, operation,
            email_old, email_new,
            username_old, username_new,
            role_old, role_new,
            changed_by
        )
        VALUES (
            NEW.id, 'UPDATE',
            NULLIF(OLD.email, NEW.email),
            NULLIF(NEW.email, OLD.email),
            NULLIF(OLD.username, NEW.username),
            NULLIF(NEW.username, OLD.username),
            NULLIF(OLD.role, NEW.role),
            NULLIF(NEW.role, OLD.role),
            current_setting('app.current_user_id', true)::BIGINT
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 장점: 타입 안전, 쿼리 간단, 인덱스 효율
-- 단점: 테이블마다 별도 감사 테이블 필요

선택적 감사​

-- 특정 필드만 감사
CREATE OR REPLACE FUNCTION audit_selective()
RETURNS TRIGGER AS $$
DECLARE
    audited_fields text[] := ARRAY['email', 'role', 'status'];  -- 감사할 필드
BEGIN
    -- 감사 대상 필드가 변경된 경우만 기록
    IF TG_OP = 'UPDATE' AND (
        OLD.email IS DISTINCT FROM NEW.email OR
        OLD.role IS DISTINCT FROM NEW.role OR
        OLD.status IS DISTINCT FROM NEW.status
    ) THEN
        INSERT INTO audit_logs (table_name, record_id, operation, old_data, new_data)
        VALUES (TG_TABLE_NAME, NEW.id, 'UPDATE',
                row_to_json(OLD)::jsonb, row_to_json(NEW)::jsonb);
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

🔍 분석 및 리포팅​

변경 이력 조회​

-- 특정 레코드의 전체 이력
WITH changes AS (
    SELECT
        operation,
        old_data,
        new_data,
        user_id,
        created_at,
        LAG(created_at) OVER (ORDER BY created_at) as prev_time
    FROM audit_logs
    WHERE table_name = 'users' AND record_id = 1
    ORDER BY created_at
)
SELECT
    operation,
    old_data,
    new_data,
    user_id,
    created_at,
    created_at - prev_time as time_since_last_change
FROM changes;

-- 필드별 변경 이력
SELECT
    (jsonb_each(new_data)).key as field,
    (jsonb_each(old_data)).value as old_value,
    (jsonb_each(new_data)).value as new_value,
    created_at
FROM audit_logs
WHERE table_name = 'users' AND record_id = 1
  AND operation = 'UPDATE'
ORDER BY created_at DESC;

활동 통계​

-- 사용자별 변경 횟수
SELECT
    u.username,
    COUNT(*) as change_count,
    COUNT(*) FILTER (WHERE al.operation = 'INSERT') as inserts,
    COUNT(*) FILTER (WHERE al.operation = 'UPDATE') as updates,
    COUNT(*) FILTER (WHERE al.operation = 'DELETE') as deletes
FROM audit_logs al
JOIN users u ON al.user_id = u.id
WHERE al.created_at > CURRENT_TIMESTAMP - INTERVAL '7 days'
GROUP BY u.id, u.username
ORDER BY change_count DESC;

-- 테이블별 변경 통계
SELECT
    table_name,
    operation,
    COUNT(*) as count,
    MIN(created_at) as first_change,
    MAX(created_at) as last_change
FROM audit_logs
WHERE created_at > CURRENT_TIMESTAMP - INTERVAL '30 days'
GROUP BY table_name, operation
ORDER BY table_name, operation;

-- 시간대별 활동
SELECT
    DATE_TRUNC('hour', created_at) as hour,
    COUNT(*) as changes
FROM audit_logs
WHERE created_at > CURRENT_TIMESTAMP - INTERVAL '24 hours'
GROUP BY hour
ORDER BY hour;

복구 쿼리 생성​

-- 특정 레코드를 이전 상태로 복구하는 SQL 생성
CREATE OR REPLACE FUNCTION generate_rollback_sql(
    p_table_name VARCHAR,
    p_record_id BIGINT,
    p_target_time TIMESTAMPTZ
) RETURNS TEXT AS $$
DECLARE
    old_state JSONB;
    sql_text TEXT;
BEGIN
    -- 지정 시간 이전의 마지막 상태 조회
    SELECT new_data INTO old_state
    FROM audit_logs
    WHERE table_name = p_table_name
      AND record_id = p_record_id
      AND created_at <= p_target_time
    ORDER BY created_at DESC
    LIMIT 1;

    -- UPDATE SQL 생성
    sql_text := format(
        'UPDATE %I SET %s WHERE id = %s',
        p_table_name,
        (SELECT string_agg(format('%I = %L', key, value), ', ')
         FROM jsonb_each_text(old_state)
         WHERE key != 'id'),
        p_record_id
    );

    RETURN sql_text;
END;
$$ LANGUAGE plpgsql;

-- 사용 예
SELECT generate_rollback_sql('users', 1, '2024-01-15 10:00:00'::TIMESTAMPTZ);
-- 결과: UPDATE users SET email = 'old@example.com', role = 'user' WHERE id = 1

🔒 보안 및 규정 준수​

감사 로그 보호​

-- 감사 로그는 수정/삭제 불가
REVOKE UPDATE, DELETE, TRUNCATE ON audit_logs FROM PUBLIC;
GRANT INSERT, SELECT ON audit_logs TO app_user;

-- 관리자만 삭제 가능 (보존 기간 이후)
GRANT DELETE ON audit_logs TO admin_user;

-- Row-Level Security
ALTER TABLE audit_logs ENABLE ROW LEVEL SECURITY;

-- 사용자는 자신의 감사 로그만 조회
CREATE POLICY user_audit_read ON audit_logs
FOR SELECT
TO app_user
USING (user_id = current_setting('app.current_user_id')::BIGINT);

-- 관리자는 모두 조회
CREATE POLICY admin_audit_read ON audit_logs
FOR SELECT
TO admin_user
USING (true);

보존 기간 관리​

-- 감사 로그 보존 정책
CREATE OR REPLACE FUNCTION purge_old_audit_logs()
RETURNS void AS $$
BEGIN
    -- 7년 이상 된 로그 아카이브
    INSERT INTO audit_logs_archive
    SELECT * FROM audit_logs
    WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '7 years';

    -- 원본에서 삭제
    DELETE FROM audit_logs
    WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '7 years';
END;
$$ LANGUAGE plpgsql;

-- 월별 실행
SELECT cron.schedule('purge-audit-logs', '0 0 1 * *',
    'SELECT purge_old_audit_logs()');

✅ 감사 로그 체크리스트​

설계​

  • 무엇을 감사할지 결정 (모든 테이블 vs 중요 테이블)
  • 무엇을 기록할지 정의 (전체 vs 변경된 필드만)
  • 민감 정보 마스킹 전략
  • 보존 기간 정책

구현​

  • 트리거로 자동 기록
  • 컨텍스트 정보 (user_id, IP, 이유 등)
  • JSONB로 유연한 데이터 저장
  • 파티셔닝으로 성능 최적화

보안​

  • 감사 로그 수정/삭제 방지
  • 접근 권한 제한
  • 암호화 (필요시)

운영​

  • 정기적인 아카이브
  • 모니터링 및 알림
  • 복구 프로세스 문서화
