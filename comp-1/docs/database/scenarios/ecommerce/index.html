전자상거래 시스템

완전한 전자상거래 데이터베이스 설계 예시

전자상거래는 가장 복잡한 데이터베이스 설계 중 하나입니다. 사용자, 상품, 주문, 결제, 재고, 리뷰 등 다양한 도메인을 다룹니다.

🎯 요구사항 분석​

핵심 기능​

ecommerce_features = {
    "사용자 관리": "회원가입, 로그인, 프로필, 주소록",
    "상품 관리": "카테고리, 상품, 재고, 가격, 이미지",
    "장바구니": "상품 담기, 수량 조절, 쿠폰 적용",
    "주문/결제": "주문 생성, 결제, 배송 추적",
    "리뷰/평점": "상품 리뷰, 평점, 이미지 리뷰",
    "검색": "상품 검색, 필터링, 정렬",
    "프로모션": "쿠폰, 할인, 특가"
}

🔧 스키마 설계​

1. 사용자 및 인증​

-- 사용자
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,

    -- 프로필
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    phone VARCHAR(20),
    date_of_birth DATE,

    -- 상태
    email_verified BOOLEAN DEFAULT false,
    is_active BOOLEAN DEFAULT true,
    is_seller BOOLEAN DEFAULT false,  -- 판매자 여부

    -- 타임스탬프
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    last_login_at TIMESTAMPTZ
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_username ON users(username);

-- 주소록
CREATE TABLE user_addresses (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,

    -- 주소 정보
    address_type VARCHAR(20) NOT NULL,  -- 'shipping', 'billing'
    recipient_name VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    postal_code VARCHAR(20) NOT NULL,
    address1 VARCHAR(255) NOT NULL,  -- 기본 주소
    address2 VARCHAR(255),  -- 상세 주소
    city VARCHAR(100),
    state VARCHAR(100),
    country VARCHAR(2) DEFAULT 'KR',  -- ISO 3166-1

    -- 기본 주소 여부
    is_default BOOLEAN DEFAULT false,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_user_addresses_user ON user_addresses(user_id);

-- 사용자당 하나의 기본 배송 주소
CREATE UNIQUE INDEX uq_user_default_shipping
ON user_addresses(user_id)
WHERE is_default = true AND address_type = 'shipping';

2. 상품 및 카테고리​

-- 카테고리 (계층 구조)
CREATE TABLE categories (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    slug VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    parent_id BIGINT REFERENCES categories(id),

    -- 순서
    sort_order INTEGER DEFAULT 0,

    -- 활성화 여부
    is_active BOOLEAN DEFAULT true,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_categories_parent ON categories(parent_id);
CREATE INDEX idx_categories_slug ON categories(slug);

-- 상품
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    seller_id BIGINT NOT NULL REFERENCES users(id),
    category_id BIGINT REFERENCES categories(id),

    -- 기본 정보
    name VARCHAR(200) NOT NULL,
    slug VARCHAR(200) UNIQUE NOT NULL,
    description TEXT,
    short_description VARCHAR(500),

    -- 가격
    price NUMERIC(10, 2) NOT NULL CHECK (price >= 0),
    compare_at_price NUMERIC(10, 2) CHECK (compare_at_price >= price),  -- 정가
    cost NUMERIC(10, 2),  -- 원가 (관리용)

    -- SKU (Stock Keeping Unit)
    sku VARCHAR(100) UNIQUE NOT NULL,

    -- 재고
    stock_quantity INTEGER NOT NULL DEFAULT 0 CHECK (stock_quantity >= 0),
    low_stock_threshold INTEGER DEFAULT 10,

    -- 속성
    weight NUMERIC(10, 2),  -- kg
    dimensions VARCHAR(50),  -- "10x20x30 cm"

    -- 상태
    status VARCHAR(20) DEFAULT 'draft',  -- 'draft', 'active', 'archived'
    is_featured BOOLEAN DEFAULT false,
    is_available BOOLEAN DEFAULT true,

    -- 통계 (역정규화)
    view_count BIGINT DEFAULT 0,
    order_count INTEGER DEFAULT 0,
    rating_average NUMERIC(3, 2) DEFAULT 0.00 CHECK (rating_average BETWEEN 0 AND 5),
    review_count INTEGER DEFAULT 0,

    -- 메타데이터
    meta_title VARCHAR(200),
    meta_description VARCHAR(500),
    meta_keywords TEXT,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    published_at TIMESTAMPTZ
);

CREATE INDEX idx_products_seller ON products(seller_id);
CREATE INDEX idx_products_category ON products(category_id);
CREATE INDEX idx_products_sku ON products(sku);
CREATE INDEX idx_products_slug ON products(slug);
CREATE INDEX idx_products_status ON products(status);

-- 전문 검색 인덱스
CREATE INDEX idx_products_search
ON products USING GIN (
    to_tsvector('english', name || ' ' || COALESCE(description, ''))
);

-- 상품 이미지
CREATE TABLE product_images (
    id BIGSERIAL PRIMARY KEY,
    product_id BIGINT NOT NULL REFERENCES products(id) ON DELETE CASCADE,

    url VARCHAR(500) NOT NULL,
    alt_text VARCHAR(255),
    sort_order INTEGER DEFAULT 0,
    is_primary BOOLEAN DEFAULT false,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_product_images_product ON product_images(product_id);

-- 상품당 하나의 대표 이미지
CREATE UNIQUE INDEX uq_product_primary_image
ON product_images(product_id)
WHERE is_primary = true;

-- 상품 변형 (옵션)
CREATE TABLE product_variants (
    id BIGSERIAL PRIMARY KEY,
    product_id BIGINT NOT NULL REFERENCES products(id) ON DELETE CASCADE,

    -- 변형 속성
    variant_name VARCHAR(100),  -- "Red / Large"
    options JSONB,  -- {"color": "Red", "size": "Large"}

    sku VARCHAR(100) UNIQUE NOT NULL,
    price NUMERIC(10, 2) NOT NULL CHECK (price >= 0),
    stock_quantity INTEGER NOT NULL DEFAULT 0,

    is_available BOOLEAN DEFAULT true,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_product_variants_product ON product_variants(product_id);

3. 장바구니​

CREATE TABLE cart_items (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT REFERENCES users(id) ON DELETE CASCADE,
    session_id VARCHAR(255),  -- 비로그인 사용자용

    product_id BIGINT NOT NULL REFERENCES products(id) ON DELETE CASCADE,
    variant_id BIGINT REFERENCES product_variants(id) ON DELETE CASCADE,

    quantity INTEGER NOT NULL CHECK (quantity > 0),

    -- 가격 스냅샷 (장바구니 담을 때 가격)
    price_snapshot NUMERIC(10, 2) NOT NULL,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 사용자당 같은 상품 중복 방지
CREATE UNIQUE INDEX uq_cart_user_product
ON cart_items(user_id, product_id, COALESCE(variant_id, 0))
WHERE user_id IS NOT NULL;

-- 세션당 같은 상품 중복 방지
CREATE UNIQUE INDEX uq_cart_session_product
ON cart_items(session_id, product_id, COALESCE(variant_id, 0))
WHERE session_id IS NOT NULL;

CREATE INDEX idx_cart_items_user ON cart_items(user_id);
CREATE INDEX idx_cart_items_session ON cart_items(session_id);

4. 주문 및 결제​

-- 주문 상태 ENUM
CREATE TYPE order_status AS ENUM (
    'pending',       -- 결제 대기
    'confirmed',     -- 결제 완료
    'processing',    -- 상품 준비중
    'shipped',       -- 배송중
    'delivered',     -- 배송 완료
    'cancelled',     -- 취소
    'refunded'       -- 환불
);

-- 주문
CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),

    -- 주문 번호 (외부 노출용)
    order_number VARCHAR(50) UNIQUE NOT NULL,

    -- 금액
    subtotal NUMERIC(10, 2) NOT NULL CHECK (subtotal >= 0),  -- 상품 합계
    shipping_fee NUMERIC(10, 2) NOT NULL DEFAULT 0,
    tax NUMERIC(10, 2) NOT NULL DEFAULT 0,
    discount NUMERIC(10, 2) NOT NULL DEFAULT 0,
    total NUMERIC(10, 2) NOT NULL CHECK (total >= 0),  -- 최종 금액

    -- 쿠폰
    coupon_code VARCHAR(50),
    coupon_discount NUMERIC(10, 2) DEFAULT 0,

    -- 배송 정보
    shipping_name VARCHAR(100) NOT NULL,
    shipping_phone VARCHAR(20) NOT NULL,
    shipping_address VARCHAR(255) NOT NULL,
    shipping_postal_code VARCHAR(20) NOT NULL,
    shipping_city VARCHAR(100),
    shipping_state VARCHAR(100),
    shipping_country VARCHAR(2) DEFAULT 'KR',

    -- 메모
    customer_notes TEXT,
    admin_notes TEXT,

    -- 상태
    status order_status DEFAULT 'pending',

    -- 상태별 타임스탬프
    confirmed_at TIMESTAMPTZ,
    processing_at TIMESTAMPTZ,
    shipped_at TIMESTAMPTZ,
    delivered_at TIMESTAMPTZ,
    cancelled_at TIMESTAMPTZ,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_orders_user ON orders(user_id);
CREATE INDEX idx_orders_number ON orders(order_number);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_orders_created ON orders(created_at DESC);

-- 주문 항목
CREATE TABLE order_items (
    id BIGSERIAL PRIMARY KEY,
    order_id BIGINT NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    product_id BIGINT NOT NULL REFERENCES products(id),
    variant_id BIGINT REFERENCES product_variants(id),

    -- 상품 정보 스냅샷 (주문 당시)
    product_name VARCHAR(200) NOT NULL,
    variant_name VARCHAR(100),
    sku VARCHAR(100) NOT NULL,

    -- 가격 및 수량
    quantity INTEGER NOT NULL CHECK (quantity > 0),
    unit_price NUMERIC(10, 2) NOT NULL,
    subtotal NUMERIC(10, 2) NOT NULL,  -- unit_price * quantity

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_order_items_order ON order_items(order_id);
CREATE INDEX idx_order_items_product ON order_items(product_id);

-- 결제
CREATE TABLE payments (
    id BIGSERIAL PRIMARY KEY,
    order_id BIGINT UNIQUE NOT NULL REFERENCES orders(id) ON DELETE RESTRICT,

    -- 결제 수단
    payment_method VARCHAR(50) NOT NULL,  -- 'credit_card', 'paypal', 'bank_transfer'

    -- 금액
    amount NUMERIC(10, 2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'KRW',

    -- 결제 게이트웨이 정보
    gateway VARCHAR(50),  -- 'stripe', 'toss', 'iamport'
    transaction_id VARCHAR(255),  -- 외부 결제 ID

    -- 상태
    status VARCHAR(20) DEFAULT 'pending',  -- 'pending', 'completed', 'failed', 'refunded'

    -- 실패 정보
    failure_reason TEXT,

    -- 타임스탬프
    paid_at TIMESTAMPTZ,
    refunded_at TIMESTAMPTZ,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_payments_order ON payments(order_id);
CREATE INDEX idx_payments_status ON payments(status);
CREATE INDEX idx_payments_transaction ON payments(transaction_id);

5. 리뷰 및 평점​

CREATE TABLE reviews (
    id BIGSERIAL PRIMARY KEY,
    product_id BIGINT NOT NULL REFERENCES products(id) ON DELETE CASCADE,
    user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    order_item_id BIGINT REFERENCES order_items(id),  -- 구매 확인

    -- 평점
    rating INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5),

    -- 리뷰 내용
    title VARCHAR(200),
    content TEXT NOT NULL,

    -- 이미지
    images JSONB,  -- ["url1", "url2", ...]

    -- 검증
    is_verified_purchase BOOLEAN DEFAULT false,

    -- 통계
    helpful_count INTEGER DEFAULT 0,
    unhelpful_count INTEGER DEFAULT 0,

    -- 판매자 응답
    seller_response TEXT,
    seller_response_at TIMESTAMPTZ,

    -- 상태
    status VARCHAR(20) DEFAULT 'pending',  -- 'pending', 'approved', 'rejected'

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 사용자는 상품당 하나의 리뷰만
CREATE UNIQUE INDEX uq_reviews_user_product
ON reviews(user_id, product_id);

CREATE INDEX idx_reviews_product ON reviews(product_id);
CREATE INDEX idx_reviews_user ON reviews(user_id);
CREATE INDEX idx_reviews_rating ON reviews(rating);

-- 리뷰 도움 투표
CREATE TABLE review_votes (
    review_id BIGINT NOT NULL REFERENCES reviews(id) ON DELETE CASCADE,
    user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,

    is_helpful BOOLEAN NOT NULL,  -- true = 도움됨, false = 도움 안됨

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (review_id, user_id)
);

6. 쿠폰 및 프로모션​

CREATE TABLE coupons (
    id BIGSERIAL PRIMARY KEY,

    -- 코드
    code VARCHAR(50) UNIQUE NOT NULL,
    description TEXT,

    -- 할인 타입
    discount_type VARCHAR(20) NOT NULL,  -- 'percentage', 'fixed_amount'

🔧 트리거 및 함수​

주문 생성 함수​

CREATE OR REPLACE FUNCTION create_order_from_cart(
    p_user_id BIGINT,
    p_shipping_address_id BIGINT,
    p_coupon_code VARCHAR(50) DEFAULT NULL
) RETURNS BIGINT AS $$
DECLARE
    v_order_id BIGINT;
    v_order_number VARCHAR(50);
    v_subtotal NUMERIC(10, 2) := 0;
    v_shipping_fee NUMERIC(10, 2) := 3000;  -- 고정 배송비
    v_tax NUMERIC(10, 2) := 0;
    v_discount NUMERIC(10, 2) := 0;
    v_total NUMERIC(10, 2);
    v_address RECORD;
BEGIN
    -- 주문 번호 생성
    v_order_number := 'ORD-' || TO_CHAR(CURRENT_TIMESTAMP, 'YYYYMMDD-') ||
                      LPAD(nextval('order_number_seq')::TEXT, 6, '0');

    -- 배송 주소 조회
    SELECT * INTO v_address FROM user_addresses
    WHERE id = p_shipping_address_id AND user_id = p_user_id;

    -- 장바구니 합계 계산
    SELECT SUM(quantity * price_snapshot) INTO v_subtotal
    FROM cart_items
    WHERE user_id = p_user_id;

    -- 쿠폰 할인 적용
    IF p_coupon_code IS NOT NULL THEN
        -- 쿠폰 검증 및 할인 계산 (생략)
        v_discount := 0;  -- 실제 계산 로직
    END IF;

    -- 총액 계산
    v_total := v_subtotal + v_shipping_fee + v_tax - v_discount;

    -- 주문 생성
    INSERT INTO orders (
        user_id, order_number,
        subtotal, shipping_fee, tax, discount, total,
        shipping_name, shipping_phone, shipping_address,
        shipping_postal_code, shipping_city, shipping_state,
        coupon_code
    )
    VALUES (
        p_user_id, v_order_number,
        v_subtotal, v_shipping_fee, v_tax, v_discount, v_total,
        v_address.recipient_name, v_address.phone, v_address.address1 || ' ' || COALESCE(v_address.address2, ''),
        v_address.postal_code, v_address.city, v_address.state,
        p_coupon_code
    )
    RETURNING id INTO v_order_id;

    -- 주문 항목 생성 (장바구니에서)
    INSERT INTO order_items (
        order_id, product_id, variant_id,
        product_name, sku, quantity, unit_price, subtotal
    )
    SELECT
        v_order_id,
        ci.product_id,
        ci.variant_id,
        p.name,
        COALESCE(pv.sku, p.sku),
        ci.quantity,
        ci.price_snapshot,
        ci.quantity * ci.price_snapshot
    FROM cart_items ci
    JOIN products p ON ci.product_id = p.id
    LEFT JOIN product_variants pv ON ci.variant_id = pv.id
    WHERE ci.user_id = p_user_id;

    -- 장바구니 비우기
    DELETE FROM cart_items WHERE user_id = p_user_id;

    RETURN v_order_id;
END;
$$ LANGUAGE plpgsql;

상품 평점 자동 갱신​

CREATE OR REPLACE FUNCTION update_product_rating()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE products
    SET
        rating_average = (
            SELECT AVG(rating)::NUMERIC(3,2)
            FROM reviews
            WHERE product_id = NEW.product_id
              AND status = 'approved'
        ),
        review_count = (
            SELECT COUNT(*)
            FROM reviews
            WHERE product_id = NEW.product_id
              AND status = 'approved'
        )
    WHERE id = NEW.product_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_rating_after_review
AFTER INSERT OR UPDATE ON reviews
FOR EACH ROW
EXECUTE FUNCTION update_product_rating();

📊 주요 쿼리​

상품 목록 조회​

-- 카테고리별 상품 목록 (필터링, 정렬, 페이지네이션)
SELECT
    p.id,
    p.name,
    p.slug,
    p.price,
    p.compare_at_price,
    p.rating_average,
    p.review_count,
    pi.url as image_url,
    p.stock_quantity > 0 as in_stock
FROM products p
LEFT JOIN product_images pi ON p.id = pi.product_id AND pi.is_primary = true
WHERE p.category_id = 1
  AND p.status = 'active'
  AND p.is_available = true
  AND p.price BETWEEN 10000 AND 100000  -- 가격 필터
  AND p.stock_quantity > 0  -- 재고 있음
ORDER BY p.order_count DESC  -- 인기순
LIMIT 20 OFFSET 0;

주문 상세 조회​

SELECT
    o.order_number,
    o.status,
    o.total,
    o.created_at,
    json_agg(
        json_build_object(
            'product_name', oi.product_name,
            'quantity', oi.quantity,
            'unit_price', oi.unit_price,
            'subtotal', oi.subtotal
        )
    ) as items
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
WHERE o.id = 1
GROUP BY o.id, o.order_number, o.status, o.total, o.created_at;

ON orders(user_id, created_at DESC);

파티셔닝​

-- 주문 테이블 월별 파티셔닝
CREATE TABLE orders_partitioned (
    LIKE orders INCLUDING ALL
) PARTITION BY RANGE (created_at);

CREATE TABLE orders_2024_01 PARTITION OF orders_partitioned
FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

이 설계는 실제 전자상거래 시스템의 기본 구조를 보여줍니다. 규모와 요구사항에 따라 추가 기능(위시리스트, 포인트, 멤버십 등)을 확장할 수 있습니다.
