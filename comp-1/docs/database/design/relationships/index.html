관계 설계

효과적인 테이블 간 관계 모델링 전략

데이터베이스의 핵심은 테이블 간의 관계입니다. 올바른 관계 설계는 데이터 무결성과 쿼리 효율성을 보장합니다.

🔗 관계 타입​

1:1 관계 (One-to-One)​

한 엔티티가 다른 엔티티와 정확히 1:1로 연결됩니다.

패턴 1: 외래 키를 기본 키로 사용​

-- 사용자 기본 정보
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 사용자 프로필 (1:1)
CREATE TABLE user_profiles (
    user_id BIGINT PRIMARY KEY,  -- FK이자 PK
    bio TEXT,
    avatar_url VARCHAR(500),
    website VARCHAR(255),
    date_of_birth DATE,
    location VARCHAR(100),

    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE  -- 사용자 삭제 시 프로필도 삭제
);

-- 사용 예
INSERT INTO users (email, username) VALUES
('alice@example.com', 'alice');

INSERT INTO user_profiles (user_id, bio, location) VALUES
(1, 'Software Engineer', 'Seoul, Korea');

-- 조회
SELECT u.username, p.bio, p.location
FROM users u
LEFT JOIN user_profiles p ON u.id = p.user_id;

패턴 2: UNIQUE 제약조건 사용​

CREATE TABLE user_settings (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT UNIQUE NOT NULL,  -- UNIQUE로 1:1 보장
    theme VARCHAR(20) DEFAULT 'light',
    language VARCHAR(10) DEFAULT 'ko',
    email_notifications BOOLEAN DEFAULT true,
    push_notifications BOOLEAN DEFAULT false,

    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
);

언제 1:1 관계를 사용하나?​

one_to_one_use_cases = {
    "성능 최적화": "자주 조회하지 않는 큰 데이터 분리",

    "선택적 정보": "모든 행에 필요하지 않은 정보",
    "테이블 크기 제한": "컬럼 수가 너무 많을 때 분할"
}
-- 예: 민감한 정보 분리
CREATE TABLE user_auth (
    user_id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    password_hash VARCHAR(255) NOT NULL,
    two_factor_secret VARCHAR(255),
    last_password_change TIMESTAMPTZ,
    failed_login_attempts INTEGER DEFAULT 0,
    locked_until TIMESTAMPTZ
);

1:N 관계 (One-to-Many)​

가장 흔한 관계. 한 엔티티가 여러 다른 엔티티와 연결됩니다.

기본 패턴​

-- 부모 테이블 (One)
CREATE TABLE blogs (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 자식 테이블 (Many)
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    blog_id BIGINT NOT NULL,  -- FK (중복 가능)
    title VARCHAR(200) NOT NULL,
    content TEXT,
    published_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    -- 외래 키 제약조건
    FOREIGN KEY (blog_id) REFERENCES blogs(id)
        ON DELETE CASCADE  -- 블로그 삭제 시 게시물도 삭제
);

-- 인덱스 (FK 컬럼은 거의 항상 인덱스 필요)
CREATE INDEX idx_posts_blog_id ON posts(blog_id);

-- 사용 예
INSERT INTO blogs (name, description) VALUES
('Tech Blog', 'Technology and programming');

INSERT INTO posts (blog_id, title, content) VALUES
(1, 'First Post', 'Hello World!'),
(1, 'Second Post', 'PostgreSQL is great'),
(1, 'Third Post', 'Database design patterns');

-- 조회: 블로그의 모든 게시물
SELECT b.name, p.title, p.created_at
FROM blogs b
JOIN posts p ON b.id = p.blog_id
WHERE b.id = 1
ORDER BY p.created_at DESC;

-- 조회: 게시물 수 집계
SELECT b.name, COUNT(p.id) as post_count
FROM blogs b
LEFT JOIN posts p ON b.id = p.blog_id
GROUP BY b.id, b.name;

계층 구조 (Self-Referencing)​

-- 댓글과 대댓글
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    parent_comment_id BIGINT,  -- NULL이면 최상위 댓글
    user_id BIGINT NOT NULL REFERENCES users(id),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    -- 자기 참조
    FOREIGN KEY (parent_comment_id) REFERENCES comments(id)
        ON DELETE CASCADE
);

CREATE INDEX idx_comments_post ON comments(post_id);
CREATE INDEX idx_comments_parent ON comments(parent_comment_id);

-- 사용 예
-- 최상위 댓글
INSERT INTO comments (post_id, user_id, content) VALUES
(1, 1, 'Great post!');

-- 대댓글
INSERT INTO comments (post_id, parent_comment_id, user_id, content) VALUES
(1, 1, 2, 'Thanks!');

-- 재귀 CTE로 전체 트리 조회
WITH RECURSIVE comment_tree AS (
    -- 최상위 댓글
    SELECT id, parent_comment_id, content, 0 as depth
    FROM comments
    WHERE post_id = 1 AND parent_comment_id IS NULL

    UNION ALL

    -- 자식 댓글
    SELECT c.id, c.parent_comment_id, c.content, ct.depth + 1
    FROM comments c
    JOIN comment_tree ct ON c.parent_comment_id = ct.id
)
SELECT * FROM comment_tree ORDER BY depth, id;

다중 부모 관계​

-- 게시물은 하나의 사용자에게 속하고, 하나의 카테고리에 속함
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    category_id BIGINT REFERENCES categories(id),
    title VARCHAR(200) NOT NULL,
    content TEXT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_posts_user ON posts(user_id);
CREATE INDEX idx_posts_category ON posts(category_id);

-- 조회: 사용자 + 카테고리 정보 함께
SELECT
    p.title,
    u.username,
    c.name as category_name
FROM posts p
JOIN users u ON p.user_id = u.id
LEFT JOIN categories c ON p.category_id = c.id;

N:M 관계 (Many-to-Many)​

양방향으로 여러 엔티티가 연결됩니다. 중간 테이블(Junction Table)이 필요합니다.

기본 패턴​

-- 엔티티 1
CREATE TABLE students (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 엔티티 2
CREATE TABLE courses (
    id BIGSERIAL PRIMARY KEY,
    code VARCHAR(20) UNIQUE NOT NULL,
    title VARCHAR(200) NOT NULL,
    credits INTEGER NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 중간 테이블 (Junction Table)
CREATE TABLE student_courses (
    student_id BIGINT NOT NULL,
    course_id BIGINT NOT NULL,

    -- 추가 속성
    enrolled_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    grade VARCHAR(2),  -- 'A+', 'A', 'B+' 등
    status VARCHAR(20) DEFAULT 'enrolled',  -- 'enrolled', 'completed', 'dropped'

    -- 복합 기본 키
    PRIMARY KEY (student_id, course_id),

    -- 외래 키
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE
);

-- 인덱스
CREATE INDEX idx_student_courses_student ON student_courses(student_id);
CREATE INDEX idx_student_courses_course ON student_courses(course_id);

-- 사용 예
INSERT INTO students (name, email) VALUES
('Alice', 'alice@university.edu'),
('Bob', 'bob@university.edu');

INSERT INTO courses (code, title, credits) VALUES
('CS101', 'Introduction to Programming', 3),
('CS201', 'Data Structures', 3),
('CS301', 'Algorithms', 4);

-- 수강 신청
INSERT INTO student_courses (student_id, course_id) VALUES
(1, 1),  -- Alice → CS101
(1, 2),  -- Alice → CS201
(2, 1),  -- Bob → CS101
(2, 3);  -- Bob → CS301

-- 조회: 학생의 수강 과목
SELECT s.name, c.code, c.title
FROM students s
JOIN student_courses sc ON s.id = sc.student_id
JOIN courses c ON sc.course_id = c.id
WHERE s.id = 1;

-- 조회: 과목의 수강 학생
SELECT c.code, c.title, s.name
FROM courses c
JOIN student_courses sc ON c.id = sc.course_id
JOIN students s ON sc.student_id = s.id
WHERE c.id = 1;

자기 참조 N:M 관계​

-- 소셜 네트워크: 사용자 간 팔로우 관계
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE user_follows (
    follower_id BIGINT NOT NULL,  -- 팔로우하는 사람
    following_id BIGINT NOT NULL,  -- 팔로우받는 사람
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (follower_id, following_id),

    FOREIGN KEY (follower_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (following_id) REFERENCES users(id) ON DELETE CASCADE,

    -- 자기 자신을 팔로우 못하게
    CHECK (follower_id != following_id)
);

CREATE INDEX idx_follows_follower ON user_follows(follower_id);
CREATE INDEX idx_follows_following ON user_follows(following_id);

-- 사용 예
INSERT INTO users (username) VALUES ('alice'), ('bob'), ('charlie');

-- Alice가 Bob과 Charlie를 팔로우
INSERT INTO user_follows (follower_id, following_id) VALUES
(1, 2),  -- Alice → Bob
(1, 3);  -- Alice → Charlie

-- Bob이 Alice를 팔로우
INSERT INTO user_follows (follower_id, following_id) VALUES
(2, 1);  -- Bob → Alice

-- 조회: Alice가 팔로우하는 사람들
SELECT u.username
FROM user_follows uf
JOIN users u ON uf.following_id = u.id
WHERE uf.follower_id = 1;

-- 조회: Alice를 팔로우하는 사람들
SELECT u.username
FROM user_follows uf
JOIN users u ON uf.follower_id = u.id
WHERE uf.following_id = 1;

-- 조회: 맞팔 관계 (mutual follows)
SELECT
    u1.username as user1,
    u2.username as user2
FROM user_follows uf1
JOIN user_follows uf2
    ON uf1.follower_id = uf2.following_id
    AND uf1.following_id = uf2.follower_id
JOIN users u1 ON uf1.follower_id = u1.id
JOIN users u2 ON uf1.following_id = u2.id
WHERE uf1.follower_id < uf1.following_id;  -- 중복 제거

복합 N:M 관계​

-- 프로젝트 + 역할 + 사용자
CREATE TABLE projects (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE roles (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL  -- 'owner', 'admin', 'member', 'viewer'
);

CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL
);

-- 중간 테이블: 사용자-프로젝트-역할
CREATE TABLE project_members (
    project_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    role_id BIGINT NOT NULL,
    joined_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (project_id, user_id),

    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES roles(id)
);

-- 조회: 프로젝트의 멤버와 역할
SELECT
    p.name as project,
    u.username,
    r.name as role
FROM project_members pm
JOIN projects p ON pm.project_id = p.id
JOIN users u ON pm.user_id = u.id
JOIN roles r ON pm.role_id = r.id
WHERE p.id = 1;

🎯 관계 설계 패턴​

폴리모픽 관계 (Polymorphic Association)​

여러 테이블과 관계를 맺어야 할 때 사용합니다.

-- 댓글을 게시물과 사진 모두에 달 수 있음

-- ❌ 나쁜 예: NULL을 허용하는 여러 FK
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT,  -- NULL 가능
    photo_id BIGINT,  -- NULL 가능
    content TEXT NOT NULL,

    FOREIGN KEY (post_id) REFERENCES posts(id),
    FOREIGN KEY (photo_id) REFERENCES photos(id),

    -- 하나만 설정되도록 제약
    CHECK (
        (post_id IS NOT NULL AND photo_id IS NULL) OR
        (post_id IS NULL AND photo_id IS NOT NULL)
    )
);
-- 문제: 타입 안전하지 않음, 인덱스 비효율적

-- ✅ 좋은 예 1: 개별 테이블
CREATE TABLE post_comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    user_id BIGINT NOT NULL REFERENCES users(id),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE photo_comments (
    id BIGSERIAL PRIMARY KEY,
    photo_id BIGINT NOT NULL REFERENCES photos(id) ON DELETE CASCADE,
    user_id BIGINT NOT NULL REFERENCES users(id),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ✅ 좋은 예 2: 공통 부모 테이블
CREATE TABLE commentable_items (
    id BIGSERIAL PRIMARY KEY,
    item_type VARCHAR(20) NOT NULL  -- 'post', 'photo'
);

CREATE TABLE posts (
    id BIGINT PRIMARY KEY REFERENCES commentable_items(id),
    title VARCHAR(200) NOT NULL,
    content TEXT
);

CREATE TABLE photos (
    id BIGINT PRIMARY KEY REFERENCES commentable_items(id),
    url VARCHAR(500) NOT NULL,
    caption TEXT
);

CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    commentable_id BIGINT NOT NULL REFERENCES commentable_items(id),
    user_id BIGINT NOT NULL REFERENCES users(id),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

상속 관계 모델링​

객체지향의 상속을 데이터베이스에서 표현하는 방법입니다.

패턴 1: 단일 테이블 상속 (Single Table Inheritance)​

-- 모든 타입을 하나의 테이블에
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    user_type VARCHAR(20) NOT NULL,  -- 'customer', 'admin', 'vendor'
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) NOT NULL,

    -- Customer 전용
    shipping_address TEXT,
    billing_address TEXT,

    -- Admin 전용
    admin_level INTEGER,
    permissions JSONB,

    -- Vendor 전용
    company_name VARCHAR(200),
    tax_id VARCHAR(50),

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 장점: JOIN 불필요, 쿼리 간단
-- 단점: NULL 많음, 테이블 크기 증가, 타입별 제약조건 어려움

패턴 2: 클래스 테이블 상속 (Class Table Inheritance)​

-- 공통 속성
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 타입별 테이블
CREATE TABLE customers (
    user_id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    shipping_address TEXT NOT NULL,
    billing_address TEXT NOT NULL
);

CREATE TABLE admins (
    user_id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    admin_level INTEGER NOT NULL,
    permissions JSONB
);

CREATE TABLE vendors (
    user_id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    company_name VARCHAR(200) NOT NULL,
    tax_id VARCHAR(50) UNIQUE NOT NULL
);

-- 조회 (JOIN 필요)
SELECT u.*, c.shipping_address, c.billing_address
FROM users u
JOIN customers c ON u.id = c.user_id;

-- 장점: 정규화됨, NULL 없음, 타입별 제약조건 가능
-- 단점: JOIN 필요, 쿼리 복잡

패턴 3: 구체 테이블 상속 (Concrete Table Inheritance)​

-- 각 타입이 완전히 독립적
CREATE TABLE customers (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) NOT NULL,
    shipping_address TEXT NOT NULL,
    billing_address TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE admins (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) NOT NULL,
    admin_level INTEGER NOT NULL,
    permissions JSONB,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 장점: JOIN 불필요, 타입별 독립적
-- 단점: 공통 속성 중복, 모든 사용자 조회 어려움

히스토리/버전 관리​

데이터 변경 이력을 추적합니다.

-- 현재 데이터
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    price NUMERIC(10, 2) NOT NULL,
    description TEXT,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 변경 이력
CREATE TABLE product_history (
    id BIGSERIAL PRIMARY KEY,
    product_id BIGINT NOT NULL REFERENCES products(id) ON DELETE CASCADE,

    -- 변경된 필드
    name VARCHAR(200),
    price NUMERIC(10, 2),
    description TEXT,

    -- 메타데이터
    changed_by BIGINT REFERENCES users(id),
    changed_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    change_type VARCHAR(20)  -- 'created', 'updated', 'deleted'
);

CREATE INDEX idx_product_history_product ON product_history(product_id);

-- 트리거로 자동 기록
CREATE OR REPLACE FUNCTION record_product_history()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'UPDATE' THEN
        INSERT INTO product_history (product_id, name, price, description, change_type)
        VALUES (OLD.id, OLD.name, OLD.price, OLD.description, 'updated');
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO product_history (product_id, name, price, description, change_type)
        VALUES (OLD.id, OLD.name, OLD.price, OLD.description, 'deleted');
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER product_history_trigger
AFTER UPDATE OR DELETE ON products
FOR EACH ROW EXECUTE FUNCTION record_product_history();

💡 관계 설계 모범 사례​

1. 외래 키는 항상 인덱싱​

-- ✅ FK 컬럼에 인덱스
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    category_id BIGINT REFERENCES categories(id)
);

CREATE INDEX idx_posts_user ON posts(user_id);
CREATE INDEX idx_posts_category ON posts(category_id);

2. ON DELETE 동작 명시​

-- ✅ 의도를 명확히
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,

    -- 게시물 삭제 시 댓글도 삭제
    FOREIGN KEY (post_id) REFERENCES posts(id)
        ON DELETE CASCADE,

    -- 사용자 삭제는 방지 (댓글이 있으면)
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE RESTRICT
);

3. 중간 테이블에 의미 있는 이름​

-- ❌ 나쁜 예
CREATE TABLE user_post (...)  -- 의미 불명확

-- ✅ 좋은 예
CREATE TABLE post_likes (...)  -- 명확한 관계
CREATE TABLE course_enrollments (...)
CREATE TABLE product_reviews (...)

4. 복합 키 순서 고려​

-- 자주 사용하는 쿼리 패턴에 맞춰 순서 결정
CREATE TABLE order_items (
    order_id BIGINT,
    product_id BIGINT,
    quantity INTEGER,

    -- order_id로 자주 조회하면 order_id를 먼저
    PRIMARY KEY (order_id, product_id)
);

-- 자동으로 생성되는 인덱스: (order_id, product_id)
-- product_id로도 자주 조회하면 추가 인덱스 필요
CREATE INDEX idx_order_items_product ON order_items(product_id);
