설계 원칙

견고하고 확장 가능한 데이터베이스 설계를 위한 핵심 원칙

좋은 데이터베이스 설계는 소프트웨어 개발 원칙을 따르면서도 데이터베이스 특성을 고려해야 합니다.

🎯 핵심 설계 원칙​

1. 단일 책임 원칙 (Single Responsibility)​

테이블은 하나의 명확한 목적만 가져야 합니다.

-- ❌ 나쁜 예: 여러 책임이 혼재
CREATE TABLE users_and_orders (
    user_id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50),
    email VARCHAR(255),

    -- 주문 정보가 사용자 테이블에!
    order_id BIGINT,
    order_date TIMESTAMP,
    order_total NUMERIC(10, 2),
    product_name VARCHAR(200)
);
-- 문제: 사용자가 여러 주문을 가지면 사용자 정보 중복

-- ✅ 좋은 예: 각 테이블은 하나의 책임
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    order_date TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    total NUMERIC(10, 2) NOT NULL
);

CREATE TABLE order_items (
    id BIGSERIAL PRIMARY KEY,
    order_id BIGINT NOT NULL REFERENCES orders(id),
    product_id BIGINT NOT NULL REFERENCES products(id),
    quantity INTEGER NOT NULL,
    price NUMERIC(10, 2) NOT NULL
);

2. DRY (Don't Repeat Yourself)​

데이터 중복을 피하고 정규화를 통해 단일 진실 공급원을 유지합니다.

-- ❌ 나쁜 예: 중복 데이터
CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,

    -- 고객 정보 중복
    customer_name VARCHAR(100),
    customer_email VARCHAR(255),
    customer_phone VARCHAR(20),

    -- 제품 정보 중복
    product_name VARCHAR(200),
    product_price NUMERIC(10, 2),

    quantity INTEGER
);
-- 문제: 고객 이메일 변경 시 모든 주문 수정 필요

-- ✅ 좋은 예: 참조를 통한 중복 제거
CREATE TABLE customers (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20)
);

CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    price NUMERIC(10, 2) NOT NULL
);

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    customer_id BIGINT NOT NULL REFERENCES customers(id),
    product_id BIGINT NOT NULL REFERENCES products(id),
    quantity INTEGER NOT NULL,
    price_at_time NUMERIC(10, 2) NOT NULL  -- 주문 당시 가격 스냅샷
);

3. 명시적 명명 (Explicit Naming)​

명확하고 일관된 네이밍으로 의도를 드러냅니다.

-- ❌ 나쁜 예: 불분명한 이름
CREATE TABLE data (
    id BIGINT,
    type INTEGER,  -- 무슨 타입?
    status CHAR(1),  -- 'A', 'I' 의미 불명확
    flag BOOLEAN,  -- 무슨 플래그?
    data TEXT  -- 무슨 데이터?
);

-- ✅ 좋은 예: 명확한 이름
CREATE TABLE user_subscriptions (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    subscription_type VARCHAR(20) NOT NULL,  -- 'free', 'premium', 'enterprise'
    status VARCHAR(20) NOT NULL,  -- 'active', 'cancelled', 'expired'
    is_auto_renew_enabled BOOLEAN DEFAULT true,
    subscription_data JSONB,  -- 구독 관련 메타데이터

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    expires_at TIMESTAMPTZ NOT NULL
);

4. 확장성 고려 (Design for Scale)​

미래 성장을 고려한 설계를 합니다.

-- ❌ 나쁜 예: 확장성 제한
CREATE TABLE posts (
    id SERIAL PRIMARY KEY,  -- 21억 제한
    content VARCHAR(1000),  -- 길이 제한 너무 짧음
    created_at TIMESTAMP  -- 타임존 없음
);

-- ✅ 좋은 예: 확장 가능한 설계
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,  -- 9천경까지
    user_id BIGINT NOT NULL REFERENCES users(id),
    content TEXT,  -- 제한 없음

    -- 성능을 위한 카운터 (역정규화)
    view_count BIGINT DEFAULT 0,
    like_count INTEGER DEFAULT 0,
    comment_count INTEGER DEFAULT 0,

    -- 타임존 포함
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    published_at TIMESTAMPTZ,

    -- 소프트 삭제
    deleted_at TIMESTAMPTZ
);

-- 대용량 데이터를 위한 파티셔닝
CREATE TABLE analytics_events (
    id BIGSERIAL,
    event_type VARCHAR(50) NOT NULL,
    user_id BIGINT,
    event_data JSONB,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
) PARTITION BY RANGE (created_at);

-- 월별 파티션
CREATE TABLE analytics_events_2024_01 PARTITION OF analytics_events
    FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

📐 데이터 무결성 원칙​

1. 제약조건 활용​

데이터베이스 레벨에서 무결성을 보장합니다.

-- ✅ 제약조건을 통한 무결성 보장
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,

    -- CHECK 제약조건
    price NUMERIC(10, 2) NOT NULL CHECK (price >= 0),
    stock_quantity INTEGER NOT NULL CHECK (stock_quantity >= 0),
    discount_percent NUMERIC(5, 2) CHECK (discount_percent BETWEEN 0 AND 100),

    -- UNIQUE 제약조건
    sku VARCHAR(50) UNIQUE NOT NULL,

    -- 복합 제약조건
    CONSTRAINT valid_product CHECK (
        (stock_quantity > 0 AND is_available = true) OR
        (stock_quantity = 0 AND is_available = false)
    ),

    is_available BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 도메인 제약조건
CREATE DOMAIN email_address AS VARCHAR(255)
    CHECK (VALUE ~ '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}$');

CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email email_address NOT NULL UNIQUE
);

2. 외래 키 강제​

관계 무결성을 데이터베이스가 보장하도록 합니다.

-- ✅ 외래 키로 참조 무결성 보장
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    content TEXT NOT NULL,

    -- 외래 키 제약조건
    FOREIGN KEY (post_id) REFERENCES posts(id)
        ON DELETE CASCADE,  -- 게시물 삭제 시 댓글도 삭제

    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE SET NULL,  -- 사용자 삭제 시 NULL 처리

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ❌ 외래 키 없이 애플리케이션에서만 관리
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,  -- 제약조건 없음
    user_id BIGINT NOT NULL,  -- 제약조건 없음
    content TEXT NOT NULL
);
-- 문제: 존재하지 않는 post_id나 user_id 삽입 가능

⚡ 성능 원칙​

1. 인덱스 전략​

적절한 인덱스로 쿼리 성능을 최적화합니다.

CREATE TABLE articles (
    id BIGSERIAL PRIMARY KEY,
    author_id BIGINT NOT NULL REFERENCES users(id),
    category_id BIGINT REFERENCES categories(id),
    title VARCHAR(200) NOT NULL,
    content TEXT,
    status VARCHAR(20) DEFAULT 'draft',
    published_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ✅ 자주 사용하는 쿼리 패턴에 인덱스
-- 1. 작성자별 조회
CREATE INDEX idx_articles_author ON articles(author_id);

-- 2. 발행된 글 조회 (부분 인덱스)
CREATE INDEX idx_articles_published
ON articles(published_at DESC)
WHERE status = 'published' AND published_at IS NOT NULL;

-- 3. 카테고리 + 발행일 조회 (복합 인덱스)
CREATE INDEX idx_articles_category_published
ON articles(category_id, published_at DESC)
WHERE status = 'published';

-- 4. 제목 검색 (텍스트 검색)
CREATE INDEX idx_articles_title_search
ON articles USING GIN (to_tsvector('english', title));

-- ❌ 불필요한 인덱스
CREATE INDEX idx_articles_created ON articles(created_at);  -- 거의 안 쓰임
CREATE INDEX idx_articles_content ON articles(content);  -- TEXT 컬럼 전체 인덱스

2. 정규화 vs 역정규화 균형​

읽기 성능이 중요한 경우 의도적인 역정규화를 고려합니다.

-- 시나리오: 게시물 목록에서 작성자 이름을 자주 표시

-- ✅ 정규화 (기본)
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    title VARCHAR(200) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 매번 JOIN 필요
SELECT p.title, u.username
FROM posts p
JOIN users u ON p.user_id = u.id;

-- ✅ 역정규화 (성능 최적화)
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    title VARCHAR(200) NOT NULL,

    -- 역정규화: 자주 사용하는 필드 복제
    author_username VARCHAR(50) NOT NULL,

    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- JOIN 없이 조회 가능
SELECT title, author_username FROM posts;

-- 트리거로 동기화
CREATE OR REPLACE FUNCTION sync_author_username()
RETURNS TRIGGER AS $$
BEGIN
    NEW.author_username := (SELECT username FROM users WHERE id = NEW.user_id);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_author_username
BEFORE INSERT OR UPDATE OF user_id ON posts
FOR EACH ROW EXECUTE FUNCTION sync_author_username();

🔒 보안 원칙​

1. 민감 데이터 보호​

-- ✅ 비밀번호는 항상 해시
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,  -- bcrypt, argon2 해시

    -- 민감 정보 별도 테이블
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 민감한 개인정보는 별도 테이블
CREATE TABLE user_sensitive_data (
    user_id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,

    -- 암호화된 데이터
    social_security_number_encrypted TEXT,
    credit_card_encrypted TEXT,

    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ❌ 평문 비밀번호 저장
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255),
    password VARCHAR(255)  -- 절대 안됨!
);

2. SQL 인젝션 방지​

-- ✅ 파라미터화된 쿼리 사용 (애플리케이션 레벨)
-- Python (SQLAlchemy) 예시
"""
# 안전한 방법
user = session.query(User).filter(User.email == user_input).first()

# 위험한 방법 (절대 사용 금지)
query = f"SELECT * FROM users WHERE email = '{user_input}'"
"""

-- 데이터베이스 레벨: 최소 권한 원칙
-- 읽기 전용 사용자
CREATE ROLE app_readonly;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO app_readonly;

-- 애플리케이션 사용자 (읽기/쓰기, DDL 제외)
CREATE ROLE app_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO app_user;
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO app_user;

📊 데이터 일관성 원칙​

1. 트랜잭션 활용​

-- ✅ 트랜잭션으로 원자성 보장
BEGIN;

-- 주문 생성
INSERT INTO orders (user_id, total_amount)
VALUES (1, 100.00)
RETURNING id INTO order_id;

-- 재고 차감
UPDATE products
SET stock_quantity = stock_quantity - 1
WHERE id = 123 AND stock_quantity > 0;

-- 재고가 부족하면 롤백
IF NOT FOUND THEN
    ROLLBACK;
    RAISE EXCEPTION '재고 부족';
END IF;

-- 주문 항목 추가
INSERT INTO order_items (order_id, product_id, quantity, price)
VALUES (order_id, 123, 1, 100.00);

COMMIT;

2. 낙관적 잠금​

-- ✅ 버전 컬럼으로 동시성 제어
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    price NUMERIC(10, 2) NOT NULL,
    stock_quantity INTEGER NOT NULL,

    -- 버전 관리
    version INTEGER DEFAULT 1,

    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 업데이트 시 버전 확인
UPDATE products
SET
    stock_quantity = stock_quantity - 1,
    version = version + 1,
    updated_at = CURRENT_TIMESTAMP
WHERE id = 123 AND version = 5;  -- 현재 버전 확인

-- 영향받은 행이 0이면 충돌 발생

💡 유지보수 원칙​

1. 마이그레이션 전략​

-- ✅ 안전한 스키마 변경

-- 1단계: 새 컬럼 추가 (NULL 허용)
ALTER TABLE users ADD COLUMN phone_number VARCHAR(20);

-- 2단계: 기존 데이터 마이그레이션
UPDATE users SET phone_number = legacy_phone WHERE legacy_phone IS NOT NULL;

-- 3단계: NOT NULL 제약조건 추가 (필요시)
ALTER TABLE users ALTER COLUMN phone_number SET NOT NULL;

-- 4단계: 기존 컬럼 제거
ALTER TABLE users DROP COLUMN legacy_phone;

-- ❌ 위험한 변경
ALTER TABLE users DROP COLUMN phone;  -- 데이터 손실!
ALTER TABLE users ALTER COLUMN email TYPE VARCHAR(100);  -- 데이터 잘림 가능

2. 문서화​

-- ✅ 테이블과 컬럼에 주석 추가
COMMENT ON TABLE users IS '시스템 사용자 정보';
COMMENT ON COLUMN users.email IS '로그인에 사용되는 이메일 주소';
COMMENT ON COLUMN users.deleted_at IS 'NULL이면 활성, NOT NULL이면 소프트 삭제됨';

-- ✅ 제약조건에 의미있는 이름
ALTER TABLE orders
ADD CONSTRAINT chk_orders_positive_total
CHECK (total_amount > 0);

ALTER TABLE order_items
ADD CONSTRAINT chk_order_items_positive_quantity
CHECK (quantity > 0);

✅ 설계 체크리스트​

스키마 설계​

  • 각 테이블은 단일 책임을 가지는가?
  • 데이터 중복이 최소화되었는가?
  • 테이블과 컬럼 이름이 명확한가?
  • 적절히 정규화되었는가?

데이터 무결성​

  • 모든 테이블에 기본 키가 있는가?
  • 외래 키 제약조건이 정의되었는가?
  • CHECK 제약조건으로 유효성을 검증하는가?
  • NOT NULL이 적절히 사용되었는가?

성능​

  • 자주 사용하는 쿼리에 인덱스가 있는가?
  • 불필요한 인덱스는 제거했는가?
  • 읽기 성능을 위한 역정규화가 필요한가?
  • 대용량 테이블은 파티셔닝을 고려했는가?

보안​

  • 민감한 데이터가 암호화되는가?
  • 비밀번호는 해시되는가?
  • 최소 권한 원칙을 따르는가?

확장성​

  • ID는 BIGSERIAL 또는 UUID를 사용하는가?
  • 타임스탬프는 타임존을 포함하는가?
  • 소프트 삭제를 고려했는가?

유지보수​

  • 마이그레이션 전략이 있는가?
  • 테이블과 컬럼에 주석이 있는가?
  • 버전 관리 시스템에서 추적되는가?
