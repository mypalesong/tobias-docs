제약조건

데이터 무결성을 보장하는 제약조건 활용 전략

제약조건은 데이터베이스 레벨에서 데이터 무결성을 보장하는 강력한 도구입니다. 애플리케이션 로직만으로는 부족합니다.

🔒 제약조건의 중요성​

왜 데이터베이스 레벨 제약조건인가?​

-- ❌ 애플리케이션에서만 검증
"""
# Python 코드
if user.age < 0:
    raise ValueError("Age cannot be negative")
"""

-- 문제점:
-- 1. 다른 애플리케이션/SQL 클라이언트는 우회 가능
-- 2. 동시성 문제 (race condition)
-- 3. 직접 SQL로 데이터 조작 시 검증 누락

-- ✅ 데이터베이스 제약조건
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    age INTEGER CHECK (age >= 0)  -- 어떤 방법으로도 음수 불가
);

-- 장점:
-- 1. 모든 접근 경로에서 검증
-- 2. 동시성 안전
-- 3. 선언적이고 명확

📋 제약조건 타입​

PRIMARY KEY​

테이블의 각 행을 고유하게 식별합니다.

-- 방법 1: 컬럼 정의에서
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL
);

-- 방법 2: 테이블 제약조건으로
CREATE TABLE users (
    id BIGSERIAL,
    email VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
);

-- 복합 기본 키
CREATE TABLE order_items (
    order_id BIGINT,
    product_id BIGINT,
    quantity INTEGER NOT NULL,
    PRIMARY KEY (order_id, product_id)
);

-- 특징:
-- - NOT NULL + UNIQUE
-- - 자동으로 인덱스 생성
-- - 테이블당 하나만

FOREIGN KEY​

테이블 간 참조 무결성을 보장합니다.

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    title VARCHAR(200) NOT NULL,

    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- ✅ ON DELETE/UPDATE 옵션
CREATE TABLE comments (
    id BIGSERIAL PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    content TEXT NOT NULL,

    -- CASCADE: 부모 삭제 시 자식도 삭제
    FOREIGN KEY (post_id) REFERENCES posts(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,

    -- RESTRICT: 자식이 있으면 부모 삭제 불가 (기본값)
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE RESTRICT
        ON UPDATE RESTRICT
);

-- SET NULL: 부모 삭제 시 NULL로
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    author_id BIGINT,

    FOREIGN KEY (author_id) REFERENCES users(id)
        ON DELETE SET NULL  -- 사용자 삭제 시 author_id = NULL
);

-- SET DEFAULT: 부모 삭제 시 기본값으로
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    author_id BIGINT DEFAULT 0,  -- 0 = 삭제된 사용자

    FOREIGN KEY (author_id) REFERENCES users(id)
        ON DELETE SET DEFAULT
);

-- NO ACTION: RESTRICT와 유사 (트랜잭션 종료 시 확인)
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,

    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE NO ACTION
);

UNIQUE​

컬럼 값의 중복을 방지합니다.

-- 단일 컬럼 UNIQUE
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL
);

-- 복합 UNIQUE (조합이 유일)
CREATE TABLE course_enrollments (
    id BIGSERIAL PRIMARY KEY,
    student_id BIGINT NOT NULL,
    course_id BIGINT NOT NULL,
    semester VARCHAR(20) NOT NULL,

    -- 같은 학기에 같은 강좌 중복 수강 불가
    UNIQUE (student_id, course_id, semester)
);

-- 명시적 이름 지정
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    sku VARCHAR(50),

    CONSTRAINT uq_products_sku UNIQUE (sku)
);

-- NULL 처리
-- UNIQUE는 여러 NULL 허용 (NULL != NULL)
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    phone VARCHAR(20) UNIQUE  -- NULL 여러 개 가능
);
-- phone이 NULL인 사용자 여러 명 가능

-- 부분 UNIQUE (조건부)
CREATE UNIQUE INDEX uq_active_usernames
ON users(username)
WHERE deleted_at IS NULL;
-- 활성 사용자만 username UNIQUE, 삭제된 사용자는 중복 가능

NOT NULL​

NULL 값을 허용하지 않습니다.

-- ✅ 필수 필드는 NOT NULL
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL,
    username VARCHAR(50) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,

    -- 선택 필드는 NULL 허용
    bio TEXT,
    website VARCHAR(255),
    phone VARCHAR(20)
);

-- NOT NULL 추가/제거
ALTER TABLE users ALTER COLUMN bio SET NOT NULL;
ALTER TABLE users ALTER COLUMN bio DROP NOT NULL;

-- NOT NULL + DEFAULT
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    view_count INTEGER NOT NULL DEFAULT 0,
    is_featured BOOLEAN NOT NULL DEFAULT false
);

CHECK​

커스텀 조건을 검증합니다.

-- 단순 CHECK
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    price NUMERIC(10, 2) NOT NULL CHECK (price > 0),
    stock_quantity INTEGER NOT NULL CHECK (stock_quantity >= 0),
    discount_percent NUMERIC(5, 2) CHECK (discount_percent BETWEEN 0 AND 100)
);

-- 복합 CHECK (여러 컬럼)
CREATE TABLE events (
    id BIGSERIAL PRIMARY KEY,
    start_time TIMESTAMPTZ NOT NULL,
    end_time TIMESTAMPTZ NOT NULL,

    CHECK (end_time > start_time)
);

-- 명시적 이름과 복잡한 조건
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    age INTEGER,
    email VARCHAR(255),
    username VARCHAR(50),

    CONSTRAINT chk_users_valid_age
        CHECK (age IS NULL OR age BETWEEN 0 AND 150),

    CONSTRAINT chk_users_valid_email
        CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}$'),

    CONSTRAINT chk_users_valid_username
        CHECK (username ~ '^[a-zA-Z0-9_]{3,50}$')
);

-- 상태 전이 제약
CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    status VARCHAR(20) NOT NULL,

    CONSTRAINT chk_orders_valid_status
        CHECK (status IN ('pending', 'processing', 'shipped', 'delivered', 'cancelled'))
);

-- 조건부 필수 필드
CREATE TABLE employees (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    employment_type VARCHAR(20) NOT NULL,
    salary NUMERIC(10, 2),
    hourly_rate NUMERIC(10, 2),

    -- 정규직은 salary, 계약직은 hourly_rate 필수
    CONSTRAINT chk_employees_compensation
        CHECK (
            (employment_type = 'full_time' AND salary IS NOT NULL AND hourly_rate IS NULL) OR
            (employment_type = 'contract' AND hourly_rate IS NOT NULL AND salary IS NULL)
        )
);

DEFAULT​

삽입 시 기본값을 설정합니다.

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,

    -- 정적 기본값
    status VARCHAR(20) DEFAULT 'draft',
    view_count INTEGER DEFAULT 0,
    like_count INTEGER DEFAULT 0,
    is_featured BOOLEAN DEFAULT false,

    -- 함수 기본값
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    uuid UUID DEFAULT gen_random_uuid(),

    -- 표현식 기본값
    slug VARCHAR(250) DEFAULT ''
);

-- DEFAULT 변경/제거
ALTER TABLE posts ALTER COLUMN status SET DEFAULT 'published';
ALTER TABLE posts ALTER COLUMN status DROP DEFAULT;

🎯 실전 제약조건 패턴​

이메일 검증​

CREATE DOMAIN email_address AS VARCHAR(255)
    CHECK (VALUE ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}$');

CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email email_address NOT NULL UNIQUE
);

-- 또는 직접 CHECK
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,

    CONSTRAINT chk_users_email_format
        CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}$')
);

날짜 범위 검증​

CREATE TABLE bookings (
    id BIGSERIAL PRIMARY KEY,
    room_id BIGINT NOT NULL,
    check_in DATE NOT NULL,
    check_out DATE NOT NULL,
    guest_name VARCHAR(100) NOT NULL,

    -- 체크아웃이 체크인 이후
    CONSTRAINT chk_bookings_valid_dates
        CHECK (check_out > check_in),

    -- 최소 1박
    CONSTRAINT chk_bookings_min_nights
        CHECK (check_out >= check_in + INTERVAL '1 day'),

    -- 최대 30일
    CONSTRAINT chk_bookings_max_nights
        CHECK (check_out <= check_in + INTERVAL '30 days'),

    -- 미래 예약만
    CONSTRAINT chk_bookings_future
        CHECK (check_in >= CURRENT_DATE)
);

금액 검증​

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    subtotal NUMERIC(10, 2) NOT NULL CHECK (subtotal >= 0),
    tax NUMERIC(10, 2) NOT NULL CHECK (tax >= 0),
    shipping_fee NUMERIC(10, 2) NOT NULL CHECK (shipping_fee >= 0),
    discount NUMERIC(10, 2) NOT NULL DEFAULT 0 CHECK (discount >= 0),
    total NUMERIC(10, 2) NOT NULL,

    -- 총액 계산 검증
    CONSTRAINT chk_orders_total_calculation
        CHECK (total = subtotal + tax + shipping_fee - discount),

    -- 할인이 subtotal을 초과하지 않음
    CONSTRAINT chk_orders_max_discount
        CHECK (discount <= subtotal)
);

상호 배타적 필드​

-- 하나만 설정되어야 하는 경우
CREATE TABLE payments (
    id BIGSERIAL PRIMARY KEY,
    order_id BIGINT NOT NULL,

    -- 결제 수단 (하나만 선택)
    credit_card_id BIGINT,
    paypal_account_id BIGINT,
    bank_account_id BIGINT,

    amount NUMERIC(10, 2) NOT NULL,

    -- 정확히 하나만 NOT NULL
    CONSTRAINT chk_payments_one_method
        CHECK (
            (credit_card_id IS NOT NULL AND paypal_account_id IS NULL AND bank_account_id IS NULL) OR
            (credit_card_id IS NULL AND paypal_account_id IS NOT NULL AND bank_account_id IS NULL) OR
            (credit_card_id IS NULL AND paypal_account_id IS NULL AND bank_account_id IS NOT NULL)
        )
);

열거형 타입​

-- 방법 1: CHECK 제약조건
CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    status VARCHAR(20) NOT NULL,

    CONSTRAINT chk_orders_status
        CHECK (status IN ('pending', 'processing', 'shipped', 'delivered', 'cancelled'))
);

-- 방법 2: ENUM 타입
CREATE TYPE order_status AS ENUM ('pending', 'processing', 'shipped', 'delivered', 'cancelled');

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    status order_status NOT NULL DEFAULT 'pending'
);

-- 방법 3: 참조 테이블 (추천: 유연성)
CREATE TABLE order_statuses (
    code VARCHAR(20) PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    description TEXT,
    sort_order INTEGER
);

INSERT INTO order_statuses (code, name, sort_order) VALUES
('pending', 'Pending', 1),
('processing', 'Processing', 2),
('shipped', 'Shipped', 3),
('delivered', 'Delivered', 4),
('cancelled', 'Cancelled', 5);

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,
    status VARCHAR(20) NOT NULL DEFAULT 'pending',

    FOREIGN KEY (status) REFERENCES order_statuses(code)
);

조건부 필수 필드​

CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    product_type VARCHAR(20) NOT NULL,
    name VARCHAR(200) NOT NULL,

    -- 물리적 상품만
    weight NUMERIC(10, 2),
    dimensions VARCHAR(50),

    -- 디지털 상품만
    download_url VARCHAR(500),
    file_size_mb INTEGER,

    CONSTRAINT chk_products_physical_fields
        CHECK (
            product_type != 'physical' OR
            (weight IS NOT NULL AND dimensions IS NOT NULL)
        ),

    CONSTRAINT chk_products_digital_fields
        CHECK (
            product_type != 'digital' OR
            (download_url IS NOT NULL AND file_size_mb IS NOT NULL)
        )
);

🔧 제약조건 관리​

제약조건 추가​

-- NOT NULL
ALTER TABLE users
ALTER COLUMN email SET NOT NULL;

-- UNIQUE
ALTER TABLE users
ADD CONSTRAINT uq_users_email UNIQUE (email);

-- CHECK
ALTER TABLE products
ADD CONSTRAINT chk_products_price CHECK (price > 0);

-- FOREIGN KEY
ALTER TABLE posts
ADD CONSTRAINT fk_posts_user
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE;

제약조건 제거​

-- NOT NULL
ALTER TABLE users
ALTER COLUMN phone DROP NOT NULL;

-- 이름있는 제약조건
ALTER TABLE products
DROP CONSTRAINT chk_products_price;

-- FOREIGN KEY
ALTER TABLE posts
DROP CONSTRAINT fk_posts_user;

제약조건 비활성화/활성화​

-- 제약조건 비활성화 (PostgreSQL은 직접 지원 안함)
-- 우회: 제약조건 삭제 후 재생성

-- 대량 데이터 삽입 시
BEGIN;

-- 제약조건 저장
-- (나중에 재생성하기 위해)

-- 제약조건 제거
ALTER TABLE posts DROP CONSTRAINT fk_posts_user;

-- 대량 데이터 삽입
COPY posts FROM '/path/to/data.csv';

-- 제약조건 재생성
ALTER TABLE posts
ADD CONSTRAINT fk_posts_user
FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE;

COMMIT;

제약조건 검증​

-- 기존 데이터 검증하지 않고 추가 (NOT VALID)
ALTER TABLE products
ADD CONSTRAINT chk_products_price
CHECK (price > 0) NOT VALID;

-- 나중에 검증
ALTER TABLE products
VALIDATE CONSTRAINT chk_products_price;

-- 장점: 큰 테이블에서 잠금 시간 단축

제약조건 조회​

-- 테이블의 모든 제약조건
SELECT
    conname AS constraint_name,
    contype AS constraint_type,
    pg_get_constraintdef(oid) AS definition
FROM pg_constraint
WHERE conrelid = 'users'::regclass;

-- CHECK 제약조건만
SELECT
    conname,
    pg_get_constraintdef(oid)
FROM pg_constraint
WHERE conrelid = 'products'::regclass
  AND contype = 'c';

-- FOREIGN KEY 관계
SELECT
    tc.table_name,
    kcu.column_name,
    ccu.table_name AS foreign_table_name,
    ccu.column_name AS foreign_column_name
FROM information_schema.table_constraints AS tc
JOIN information_schema.key_column_usage AS kcu
    ON tc.constraint_name = kcu.constraint_name
JOIN information_schema.constraint_column_usage AS ccu
    ON ccu.constraint_name = tc.constraint_name
WHERE tc.constraint_type = 'FOREIGN KEY';

⚠️ 제약조건 주의사항​

성능 영향​

-- ❌ 너무 복잡한 CHECK (느림)
CREATE TABLE data (
    id BIGSERIAL PRIMARY KEY,
    value JSONB,

    -- 복잡한 JSONB 검증
    CHECK (
        jsonb_typeof(value) = 'object' AND
        value ? 'name' AND
        value ? 'age' AND
        (value->>'age')::integer > 0
    )
);

-- ✅ 애플리케이션에서 검증 or 트리거 사용
CREATE TABLE data (
    id BIGSERIAL PRIMARY KEY,
    value JSONB,

    -- 간단한 기본 검증만
    CHECK (jsonb_typeof(value) = 'object')
);

에러 메시지​

-- ❌ 기본 에러 메시지 (불친절)
CREATE TABLE users (
    age INTEGER CHECK (age >= 0)
);

-- 에러: new row for relation "users" violates check constraint "users_age_check"

-- ✅ 의미있는 제약조건 이름
CREATE TABLE users (
    age INTEGER,

    CONSTRAINT chk_users_age_must_be_non_negative
        CHECK (age >= 0)
);

-- 에러: new row violates check constraint "chk_users_age_must_be_non_negative"
-- 에러 메시지만 봐도 무엇이 문제인지 알 수 있음

마이그레이션​

-- ❌ 대용량 테이블에 바로 제약조건 추가 (잠금)
ALTER TABLE large_table
ADD CONSTRAINT chk_price CHECK (price > 0);
-- 모든 행 검증 → 오래 걸림

-- ✅ NOT VALID 사용
ALTER TABLE large_table
ADD CONSTRAINT chk_price CHECK (price > 0) NOT VALID;
-- 즉시 완료, 기존 데이터는 검증 안함, 새 데이터만 검증

-- 나중에 백그라운드에서 검증
ALTER TABLE large_table
VALIDATE CONSTRAINT chk_price;

✅ 제약조건 체크리스트​

필수 제약조건​

  • 모든 테이블에 PRIMARY KEY
  • 모든 외래 키에 FOREIGN KEY 제약조건
  • 필수 필드에 NOT NULL
  • 고유해야 하는 필드에 UNIQUE

데이터 검증​

  • 음수 불가 필드에 CHECK (value >= 0)
  • 날짜 범위 검증 (시작 < 종료)
  • 이메일/전화번호 형식 검증
  • 상태 필드에 유효한 값 제한

비즈니스 규칙​

  • 금액 계산 검증
  • 상호 배타적 필드 검증
  • 조건부 필수 필드 검증

관리​

  • 제약조건에 의미있는 이름
  • 복잡한 제약조건은 문서화
  • 정기적으로 제약조건 검토

💡 핵심 원칙​

constraint_principles = {
    "데이터베이스 우선": "애플리케이션 검증 + 데이터베이스 제약조건 (이중 방어)",
    "실패 빠르게": "잘못된 데이터는 즉시 거부",
    "명확한 이름": "chk_users_age_positive (에러 메시지 개선)",
    "적절한 수준": "너무 복잡한 검증은 애플리케이션/트리거로",
    "성능 고려": "대용량 테이블은 NOT VALID 사용"
}
