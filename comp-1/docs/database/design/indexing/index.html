인덱싱 전략

쿼리 성능 최적화를 위한 효과적인 인덱스 설계

인덱스는 데이터베이스 성능의 핵심입니다. 올바른 인덱싱 전략으로 쿼리 속도를 수백 배 향상시킬 수 있습니다.

📊 인덱스 기초​

인덱스란?​

인덱스는 책의 색인처럼 데이터를 빠르게 찾기 위한 자료구조입니다.

-- 인덱스 없이 조회 (테이블 전체 스캔)
SELECT * FROM users WHERE email = 'alice@example.com';
-- 1,000,000행 → 1,000,000번 비교 (느림)

-- 인덱스 생성
CREATE INDEX idx_users_email ON users(email);

-- 인덱스 있이 조회 (인덱스 스캔)
SELECT * FROM users WHERE email = 'alice@example.com';
-- 1,000,000행 → log(1,000,000) ≈ 20번 비교 (빠름)

인덱스 작동 원리​

index_structure = {
    "B-Tree": "기본 인덱스, 정렬된 데이터, 범위 검색",

    "GIN": "배열, JSONB, 전문 검색",
    "BRIN": "대용량 순차 데이터"
}

🎯 기본 인덱스 타입​

B-Tree 인덱스 (기본)​

가장 일반적이고 다목적으로 사용됩니다.

CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,  -- 자동으로 B-Tree 인덱스 생성됨
    name VARCHAR(200) NOT NULL,
    category VARCHAR(50),
    price NUMERIC(10, 2),
    stock_quantity INTEGER,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 단일 컬럼 인덱스
CREATE INDEX idx_products_name ON products(name);
CREATE INDEX idx_products_category ON products(category);
CREATE INDEX idx_products_price ON products(price);

-- 범위 검색에 효과적
SELECT * FROM products WHERE price BETWEEN 10 AND 100;
SELECT * FROM products WHERE created_at > '2024-01-01';

-- 정렬에 효과적
SELECT * FROM products ORDER BY price;
SELECT * FROM products ORDER BY name;

-- 부등호 비교에 효과적
SELECT * FROM products WHERE price < 50;
SELECT * FROM products WHERE stock_quantity > 0;

복합 인덱스 (Composite Index)​

여러 컬럼을 조합한 인덱스입니다.

-- 복합 인덱스 생성
CREATE INDEX idx_products_category_price
ON products(category, price);

-- ✅ 인덱스 사용됨 (왼쪽부터 일치)
SELECT * FROM products
WHERE category = 'Electronics' AND price < 1000;

SELECT * FROM products
WHERE category = 'Electronics';

-- ❌ 인덱스 사용 안됨 (왼쪽 컬럼 없음)
SELECT * FROM products
WHERE price < 1000;

-- 인덱스 컬럼 순서가 중요!
-- 선택도가 높은 컬럼(= 다양한 값)을 앞에
CREATE INDEX idx_products_price_category
ON products(price, category);  -- price가 더 다양하면 이게 나음

복합 인덱스 컬럼 순서 원칙​

-- 예: 블로그 게시물
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    status VARCHAR(20),  -- 'draft', 'published', 'archived'
    created_at TIMESTAMPTZ
);

-- ❌ 나쁜 순서: 선택도 낮은 컬럼이 앞
CREATE INDEX idx_bad ON posts(status, user_id, created_at);
-- status는 3가지 값만 (선택도 낮음)

-- ✅ 좋은 순서: 선택도 높은 컬럼이 앞
CREATE INDEX idx_good ON posts(user_id, status, created_at);
-- user_id는 수천~수백만 값 (선택도 높음)

-- 쿼리 패턴
SELECT * FROM posts
WHERE user_id = 123 AND status = 'published'
ORDER BY created_at DESC;

부분 인덱스 (Partial Index)​

조건을 만족하는 행만 인덱싱합니다.

-- 활성 사용자만 인덱싱
CREATE INDEX idx_active_users_email
ON users(email)
WHERE deleted_at IS NULL;

-- 발행된 게시물만 인덱싱
CREATE INDEX idx_published_posts
ON posts(created_at DESC)
WHERE status = 'published' AND published_at IS NOT NULL;

-- 재고 있는 상품만 인덱싱
CREATE INDEX idx_available_products
ON products(category, price)
WHERE stock_quantity > 0 AND is_active = true;

-- 장점: 인덱스 크기 감소, 유지보수 비용 감소, 쿼리 속도 향상

표현식 인덱스 (Expression Index)​

함수나 표현식의 결과를 인덱싱합니다.

-- 소문자 변환 인덱스
CREATE INDEX idx_users_email_lower
ON users(LOWER(email));

-- 쿼리에서 동일한 표현식 사용
SELECT * FROM users WHERE LOWER(email) = 'alice@example.com';

-- 날짜 부분 인덱스
CREATE INDEX idx_orders_year_month
ON orders(EXTRACT(YEAR FROM created_at), EXTRACT(MONTH FROM created_at));

SELECT COUNT(*) FROM orders
WHERE EXTRACT(YEAR FROM created_at) = 2024
  AND EXTRACT(MONTH FROM created_at) = 1;

-- JSONB 필드 인덱스
CREATE INDEX idx_products_brand
ON products((attributes->>'brand'));

SELECT * FROM products
WHERE attributes->>'brand' = 'Apple';

UNIQUE 인덱스​

중복을 허용하지 않는 인덱스입니다.

-- UNIQUE 제약조건은 자동으로 UNIQUE 인덱스 생성
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,  -- UNIQUE 인덱스 생성됨
    username VARCHAR(50) UNIQUE NOT NULL
);

-- 명시적 UNIQUE 인덱스 생성
CREATE UNIQUE INDEX idx_users_email_unique ON users(email);

-- 복합 UNIQUE 인덱스
CREATE UNIQUE INDEX idx_order_items_unique
ON order_items(order_id, product_id);

-- 부분 UNIQUE 인덱스 (특정 조건에서만 UNIQUE)
CREATE UNIQUE INDEX idx_active_username
ON users(username)
WHERE deleted_at IS NULL;
-- 삭제된 사용자는 username 중복 허용

🚀 고급 인덱스 타입​

GIN 인덱스 (Generalized Inverted Index)​

배열, JSONB, 전문 검색에 사용합니다.

-- JSONB 인덱스
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200),
    attributes JSONB
);

CREATE INDEX idx_products_attributes ON products USING GIN (attributes);

-- JSONB 쿼리
SELECT * FROM products
WHERE attributes @> '{"brand": "Apple"}';

SELECT * FROM products
WHERE attributes ? 'color';  -- 'color' 키 존재 여부

-- 배열 인덱스
CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    title VARCHAR(200),
    tags TEXT[]
);

CREATE INDEX idx_posts_tags ON posts USING GIN (tags);

-- 배열 쿼리
SELECT * FROM posts WHERE tags @> ARRAY['postgresql'];
SELECT * FROM posts WHERE tags && ARRAY['database', 'sql'];

-- 전문 검색 인덱스
CREATE INDEX idx_posts_title_search
ON posts USING GIN (to_tsvector('english', title));

-- 전문 검색 쿼리
SELECT * FROM posts
WHERE to_tsvector('english', title) @@ to_tsquery('postgresql & database');

GiST 인덱스 (Generalized Search Tree)​

기하학적 데이터, 범위 타입, 전문 검색에 사용합니다.

-- 위치 데이터
CREATE EXTENSION IF NOT EXISTS postgis;

CREATE TABLE stores (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200),
    location GEOGRAPHY(POINT)
);

CREATE INDEX idx_stores_location ON stores USING GiST (location);

-- 가까운 매장 찾기
SELECT name FROM stores
WHERE ST_DWithin(location, ST_MakePoint(127.0, 37.5)::geography, 1000)
ORDER BY location <-> ST_MakePoint(127.0, 37.5)::geography
LIMIT 10;

-- 범위 타입 인덱스
CREATE TABLE bookings (
    id BIGSERIAL PRIMARY KEY,
    room_id BIGINT,
    booking_period TSTZRANGE  -- 시작~종료 범위
);

CREATE INDEX idx_bookings_period ON bookings USING GiST (booking_period);

-- 겹치는 예약 찾기
SELECT * FROM bookings
WHERE booking_period && '[2024-01-01, 2024-01-07)'::tstzrange;

BRIN 인덱스 (Block Range Index)​

대용량 순차 데이터에 사용합니다.

-- 시계열 데이터
CREATE TABLE sensor_data (
    id BIGSERIAL,
    sensor_id INTEGER,
    value NUMERIC,
    recorded_at TIMESTAMPTZ
);

-- BRIN 인덱스 (작은 크기, 빠른 범위 스캔)
CREATE INDEX idx_sensor_data_time ON sensor_data USING BRIN (recorded_at);

-- 장점: 인덱스 크기가 B-Tree의 1/100, 유지보수 비용 낮음
-- 단점: 데이터가 물리적으로 정렬되어 있어야 효과적

-- 적합한 경우: 로그, 센서 데이터, 시계열 데이터
SELECT * FROM sensor_data
WHERE recorded_at BETWEEN '2024-01-01' AND '2024-01-31';

💡 인덱싱 전략​

1. 인덱스가 필요한 경우​

-- ✅ WHERE 절에서 자주 사용하는 컬럼
CREATE INDEX idx_users_email ON users(email);
SELECT * FROM users WHERE email = 'alice@example.com';

-- ✅ JOIN에 사용되는 외래 키
CREATE INDEX idx_posts_user_id ON posts(user_id);
SELECT * FROM posts p JOIN users u ON p.user_id = u.id;

-- ✅ ORDER BY에 사용되는 컬럼
CREATE INDEX idx_posts_created_desc ON posts(created_at DESC);
SELECT * FROM posts ORDER BY created_at DESC LIMIT 10;

-- ✅ GROUP BY에 사용되는 컬럼
CREATE INDEX idx_orders_user_id ON orders(user_id);
SELECT user_id, COUNT(*) FROM orders GROUP BY user_id;

-- ✅ 자주 검색하는 텍스트 필드
CREATE INDEX idx_products_name ON products(name);
SELECT * FROM products WHERE name LIKE 'iPhone%';

2. 인덱스가 불필요한 경우​

-- ❌ 작은 테이블 (수백 행 미만)
-- 테이블 전체 스캔이 더 빠름

-- ❌ 자주 갱신되는 컬럼
-- 인덱스 유지보수 비용 > 검색 이점

-- ❌ 선택도가 낮은 컬럼 (값의 종류가 적음)
CREATE INDEX idx_users_is_active ON users(is_active);  -- true/false 2개뿐
-- 예외: 부분 인덱스는 유용할 수 있음
CREATE INDEX idx_active_users ON users(id) WHERE is_active = true;

-- ❌ NULL 값이 많은 컬럼 (부분 인덱스 고려)
CREATE INDEX idx_users_phone ON users(phone) WHERE phone IS NOT NULL;

3. 인덱스 vs 테이블 스캔​

-- 쿼리 플랜 확인
EXPLAIN ANALYZE
SELECT * FROM users WHERE email = 'alice@example.com';

-- 인덱스 스캔 (좋음)
-- Index Scan using idx_users_email on users (cost=0.42..8.44 rows=1)

-- 순차 스캔 (인덱스가 없거나 사용 안됨)
-- Seq Scan on users (cost=0.00..1693.00 rows=1)

-- 비트맵 스캔 (여러 인덱스 결합)
-- Bitmap Heap Scan on posts

🎯 실전 인덱싱 패턴​

페이지네이션​

CREATE TABLE posts (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ❌ OFFSET은 느림 (큰 오프셋일수록)
SELECT * FROM posts
ORDER BY created_at DESC
LIMIT 10 OFFSET 10000;  -- 10,000행을 스캔 후 버림

-- ✅ Keyset Pagination (커서 기반)
CREATE INDEX idx_posts_created_id ON posts(created_at DESC, id DESC);

-- 첫 페이지
SELECT * FROM posts
ORDER BY created_at DESC, id DESC
LIMIT 10;

-- 다음 페이지 (마지막 created_at, id 사용)
SELECT * FROM posts
WHERE (created_at, id) < ('2024-01-15 10:00:00', 12345)
ORDER BY created_at DESC, id DESC
LIMIT 10;

정렬과 필터 조합​

-- 쿼리 패턴
SELECT * FROM orders
WHERE user_id = 123 AND status = 'completed'
ORDER BY created_at DESC
LIMIT 10;

-- ✅ 최적 인덱스
CREATE INDEX idx_orders_user_status_created
ON orders(user_id, status, created_at DESC);

-- 인덱스가 WHERE + ORDER BY 모두 처리

부분 일치 검색​

-- LIKE 검색
CREATE TABLE products (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL
);

-- ✅ 접두사 검색 (인덱스 사용)
CREATE INDEX idx_products_name ON products(name);
SELECT * FROM products WHERE name LIKE 'iPhone%';

-- ❌ 중간/접미사 검색 (인덱스 사용 안됨)
SELECT * FROM products WHERE name LIKE '%Phone%';

-- ✅ 전문 검색 인덱스 사용
CREATE INDEX idx_products_name_search
ON products USING GIN (to_tsvector('english', name));

SELECT * FROM products
WHERE to_tsvector('english', name) @@ to_tsquery('iphone | samsung');

-- ✅ pg_trgm 확장 (유사도 검색)
CREATE EXTENSION pg_trgm;

CREATE INDEX idx_products_name_trgm
ON products USING GIN (name gin_trgm_ops);

SELECT * FROM products
WHERE name % 'iPhone';  -- 유사한 이름 검색

다중 조건 쿼리​

-- 쿼리 패턴이 다양한 경우
SELECT * FROM products WHERE category = 'Electronics';
SELECT * FROM products WHERE price < 1000;
SELECT * FROM products WHERE category = 'Electronics' AND price < 1000;

-- 옵션 1: 개별 인덱스
CREATE INDEX idx_products_category ON products(category);
CREATE INDEX idx_products_price ON products(price);
-- PostgreSQL이 필요시 두 인덱스를 Bitmap Scan으로 결합

-- 옵션 2: 복합 인덱스
CREATE INDEX idx_products_category_price ON products(category, price);
-- 두 조건 모두 사용하는 쿼리에 최적

📈 인덱스 유지보수​

인덱스 모니터링​

-- 인덱스 사용 통계
SELECT
    schemaname,
    tablename,
    indexname,
    idx_scan,  -- 인덱스 스캔 횟수
    idx_tup_read,  -- 읽은 행 수
    idx_tup_fetch  -- 가져온 행 수
FROM pg_stat_user_indexes
ORDER BY idx_scan ASC;

-- 사용되지 않는 인덱스 찾기
SELECT
    schemaname,
    tablename,
    indexname,
    pg_size_pretty(pg_relation_size(indexrelid)) as index_size
FROM pg_stat_user_indexes
WHERE idx_scan = 0
  AND indexrelname NOT LIKE 'pg_toast%'
ORDER BY pg_relation_size(indexrelid) DESC;

-- 중복 인덱스 찾기
SELECT
    pg_size_pretty(SUM(pg_relation_size(idx))::BIGINT) AS size,
    (array_agg(idx))[1] AS idx1,
    (array_agg(idx))[2] AS idx2,
    (array_agg(idx))[3] AS idx3,
    (array_agg(idx))[4] AS idx4
FROM (
    SELECT
        indexrelid::regclass AS idx,
        (indrelid::text || E'\n' || indclass::text || E'\n' ||
         indkey::text || E'\n' || COALESCE(indexprs::text, '') ||
         E'\n' || COALESCE(indpred::text, '')) AS key
    FROM pg_index
) sub
GROUP BY key
HAVING COUNT(*) > 1
ORDER BY SUM(pg_relation_size(idx)) DESC;

인덱스 재구성​

-- 인덱스 bloat 확인 후 재구성
REINDEX INDEX idx_users_email;
REINDEX TABLE users;

-- VACUUM으로 인덱스 정리
VACUUM ANALYZE users;

-- 동시성 있는 재인덱싱 (운영 중 사용)
CREATE INDEX CONCURRENTLY idx_users_email_new ON users(email);
DROP INDEX idx_users_email;
ALTER INDEX idx_users_email_new RENAME TO idx_users_email;

인덱스 크기 관리​

-- 테이블과 인덱스 크기 확인
SELECT
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname || '.' || tablename)) AS total_size,
    pg_size_pretty(pg_relation_size(schemaname || '.' || tablename)) AS table_size,
    pg_size_pretty(pg_total_relation_size(schemaname || '.' || tablename) -
                   pg_relation_size(schemaname || '.' || tablename)) AS index_size
FROM pg_tables
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname || '.' || tablename) DESC;

✅ 인덱싱 체크리스트​

인덱스 생성​

  • WHERE 절에 자주 사용하는 컬럼에 인덱스
  • 모든 외래 키에 인덱스
  • ORDER BY, GROUP BY 컬럼에 인덱스
  • 복합 인덱스 컬럼 순서 최적화
  • 부분 인덱스로 크기 최소화 고려

성능 확인​

  • EXPLAIN ANALYZE로 쿼리 플랜 확인
  • 인덱스 스캔이 사용되는지 확인
  • 실제 데이터로 성능 테스트

유지보수​

  • 사용되지 않는 인덱스 주기적 점검
  • 중복 인덱스 제거
  • 인덱스 크기 모니터링
  • VACUUM ANALYZE 정기 실행
