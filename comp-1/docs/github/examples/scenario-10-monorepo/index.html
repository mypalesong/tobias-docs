name: CI

on:
  pull_request:
  push:
    branches: [main]

jobs:
  changes:
    runs-on: ubuntu-latest
    outputs:
      packages: ${{ steps.filter.outputs.changes }}
    steps:
      - uses: actions/checkout@v3
      - uses: dorny/paths-filter@v2
        id: filter
        with:
          filters: |
            web:
              - 'apps/web/**'
            backend:
              - 'apps/backend/**'
            ui-components:
              - 'packages/ui-components/**'

  build-web:
    needs: changes
    if: contains(needs.changes.outputs.packages, 'web')
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Build Web
        run: |
          cd apps/web
          npm run build

  build-backend:
    needs: changes
    if: contains(needs.changes.outputs.packages, 'backend')
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Build Backend
        run: |
          cd apps/backend
          npm run build
