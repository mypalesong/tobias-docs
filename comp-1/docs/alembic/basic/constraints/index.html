제약조건과 인덱스

데이터 무결성을 위한 제약조건과 성능 향상을 위한 인덱스를 관리하는 방법을 배워봅시다!

🎯 학습 목표​

  • 다양한 제약조건 추가/삭제
  • 인덱스 생성 및 관리
  • 데이터 무결성 보장

🔑 Primary Key (기본 키)​

기본 키 생성​

def upgrade():
    op.create_primary_key('pk_users', 'users', ['id'])

def downgrade():
    op.drop_constraint('pk_users', 'users', type_='primary')

복합 기본 키​

def upgrade():
    op.create_primary_key(
        'pk_post_tags',
        'post_tags',
        ['post_id', 'tag_id']
    )

def downgrade():
    op.drop_constraint('pk_post_tags', 'post_tags', type_='primary')

🔗 Foreign Key (외래 키)​

외래 키 생성​

def upgrade():
    op.create_foreign_key(
        'fk_posts_author',    # 제약조건 이름
        'posts',              # source 테이블
        'users',              # target 테이블
        ['author_id'],        # source 컬럼
        ['id']                # target 컬럼
    )

def downgrade():
    op.drop_constraint('fk_posts_author', 'posts', type_='foreignkey')

CASCADE 옵션과 함께​

def upgrade():
    op.create_foreign_key(
        'fk_comments_post',
        'comments',
        'posts',
        ['post_id'],
        ['id'],
        ondelete='CASCADE',   # 부모 삭제 시 자식도 삭제
        onupdate='CASCADE'    # 부모 업데이트 시 자식도 업데이트
    )

def downgrade():
    op.drop_constraint('fk_comments_post', 'comments', type_='foreignkey')

다양한 CASCADE 옵션​

def upgrade():
    # SET NULL: 부모 삭제 시 NULL로 설정
    op.create_foreign_key(
        'fk_posts_category',
        'posts', 'categories',
        ['category_id'], ['id'],
        ondelete='SET NULL'
    )

    # RESTRICT: 자식이 있으면 부모 삭제 불가
    op.create_foreign_key(
        'fk_orders_user',
        'orders', 'users',
        ['user_id'], ['id'],
        ondelete='RESTRICT'
    )

    # NO ACTION: 데이터베이스 기본 동작
    op.create_foreign_key(
        'fk_products_supplier',
        'products', 'suppliers',
        ['supplier_id'], ['id'],
        ondelete='NO ACTION'
    )

✨ Unique Constraint (유니크 제약조건)​

단일 컬럼 유니크​

def upgrade():
    op.create_unique_constraint(
        'uq_users_email',
        'users',
        ['email']
    )

def downgrade():
    op.drop_constraint('uq_users_email', 'users', type_='unique')

복합 유니크​

def upgrade():
    # 같은 post에 같은 user가 중복 좋아요 방지
    op.create_unique_constraint(
        'uq_likes_user_post',
        'likes',
        ['user_id', 'post_id']
    )

def downgrade():
    op.drop_constraint('uq_likes_user_post', 'likes', type_='unique')

✅ Check Constraint (체크 제약조건)​

숫자 범위 체크​

def upgrade():
    op.create_check_constraint(
        'check_positive_price',
        'products',
        'price > 0'
    )

    op.create_check_constraint(
        'check_stock_non_negative',
        'products',
        'stock >= 0'
    )

    op.create_check_constraint(
        'check_rating_range',
        'products',
        'rating >= 0 AND rating <= 5'
    )

def downgrade():
    op.drop_constraint('check_rating_range', 'products')
    op.drop_constraint('check_stock_non_negative', 'products')
    op.drop_constraint('check_positive_price', 'products')

문자열 검증​

def upgrade():
    # 이메일 형식 체크 (간단한 버전)
    op.create_check_constraint(
        'check_email_format',
        'users',
        "email LIKE '%@%'"
    )

    # 상태 값 제한
    op.create_check_constraint(
        'check_order_status',
        'orders',
        "status IN ('pending', 'processing', 'shipped', 'delivered', 'cancelled')"
    )

def downgrade():
    op.drop_constraint('check_order_status', 'orders')
    op.drop_constraint('check_email_format', 'users')

날짜 검증​

def upgrade():
    # 종료일이 시작일보다 늦어야 함
    op.create_check_constraint(
        'check_event_dates',
        'events',
        'end_date > start_date'
    )

    # 미래 날짜만 허용
    op.create_check_constraint(
        'check_future_date',
        'events',
        'start_date > CURRENT_DATE'
    )

def downgrade():
    op.drop_constraint('check_future_date', 'events')
    op.drop_constraint('check_event_dates', 'events')

📊 Index (인덱스)​

단일 컬럼 인덱스​

def upgrade():
    # 기본 인덱스
    op.create_index('idx_users_email', 'users', ['email'])

    # 유니크 인덱스
    op.create_index('idx_users_username', 'users', ['username'], unique=True)

def downgrade():
    op.drop_index('idx_users_username', table_name='users')
    op.drop_index('idx_users_email', table_name='users')

복합 인덱스​

def upgrade():
    # 여러 컬럼을 함께 인덱싱
    op.create_index(
        'idx_posts_author_created',
        'posts',
        ['author_id', 'created_at']
    )

def downgrade():
    op.drop_index('idx_posts_author_created', table_name='posts')

부분 인덱스 (Partial Index)​

PostgreSQL에서 지원:

def upgrade():
    # 활성 사용자만 인덱싱
    op.create_index(
        'idx_active_users',
        'users',
        ['email'],
        postgresql_where="is_active = true"
    )

def downgrade():
    op.drop_index('idx_active_users', table_name='users')

함수 기반 인덱스​

def upgrade():
    # 대소문자 구분 없는 검색
    op.execute('''
        CREATE INDEX idx_users_email_lower
        ON users (LOWER(email))
    ''')

def downgrade():
    op.execute('DROP INDEX idx_users_email_lower')

🎨 실전 예제​

예제 1: 전자상거래 제약조건​

"""add ecommerce constraints

Revision ID: ecom001
"""

def upgrade():
    # Products 테이블
    op.create_check_constraint(
        'check_product_price',
        'products',
        'price >= 0 AND price <= 999999.99'
    )
    op.create_check_constraint(
        'check_product_stock',
        'products',
        'stock >= 0'
    )

    # Orders 테이블
    op.create_check_constraint(
        'check_order_total',
        'orders',
        'total_amount > 0'
    )
    op.create_check_constraint(
        'check_order_status',
        'orders',
        "status IN ('cart', 'pending', 'paid', 'shipped', 'delivered', 'refunded')"
    )

    # Order Items 테이블
    op.create_check_constraint(
        'check_quantity',
        'order_items',
        'quantity > 0 AND quantity <= 9999'
    )
    op.create_check_constraint(
        'check_price_calculation',
        'order_items',
        'subtotal = quantity * unit_price'
    )

    # 인덱스
    op.create_index('idx_orders_user_status', 'orders', ['user_id', 'status'])
    op.create_index('idx_orders_created', 'orders', ['created_at'])
    op.create_index('idx_products_category', 'products', ['category_id'])

def downgrade():
    op.drop_index('idx_products_category', table_name='products')
    op.drop_index('idx_orders_created', table_name='orders')
    op.drop_index('idx_orders_user_status', table_name='orders')

    op.drop_constraint('check_price_calculation', 'order_items')
    op.drop_constraint('check_quantity', 'order_items')
    op.drop_constraint('check_order_status', 'orders')
    op.drop_constraint('check_order_total', 'orders')
    op.drop_constraint('check_product_stock', 'products')
    op.drop_constraint('check_product_price', 'products')

예제 2: 소셜 미디어 제약조건​

"""add social media constraints

Revision ID: social001
"""

def upgrade():
    # Users
    op.create_unique_constraint('uq_users_email', 'users', ['email'])
    op.create_unique_constraint('uq_users_username', 'users', ['username'])

    # Posts
    op.create_foreign_key(
        'fk_posts_author',
        'posts', 'users',
        ['author_id'], ['id'],
        ondelete='CASCADE'
    )
    op.create_index('idx_posts_author_created', 'posts',
        ['author_id', 'created_at DESC']
    )

    # Likes
    op.create_unique_constraint(
        'uq_likes_user_post',
        'likes',
        ['user_id', 'post_id']
    )
    op.create_foreign_key('fk_likes_user', 'likes', 'users',
        ['user_id'], ['id'], ondelete='CASCADE')
    op.create_foreign_key('fk_likes_post', 'likes', 'posts',
        ['post_id'], ['id'], ondelete='CASCADE')

    # Follows
    op.create_unique_constraint(
        'uq_follows',
        'follows',
        ['follower_id', 'following_id']
    )
    op.create_check_constraint(
        'check_no_self_follow',
        'follows',
        'follower_id != following_id'
    )

    # 성능 인덱스
    op.create_index('idx_posts_created', 'posts', ['created_at DESC'])
    op.create_index('idx_likes_post', 'likes', ['post_id'])
    op.create_index('idx_follows_following', 'follows', ['following_id'])

def downgrade():
    op.drop_index('idx_follows_following', table_name='follows')
    op.drop_index('idx_likes_post', table_name='likes')
    op.drop_index('idx_posts_created', table_name='posts')

    op.drop_constraint('check_no_self_follow', 'follows')
    op.drop_constraint('uq_follows', 'follows', type_='unique')

    op.drop_constraint('fk_likes_post', 'likes', type_='foreignkey')
    op.drop_constraint('fk_likes_user', 'likes', type_='foreignkey')
    op.drop_constraint('uq_likes_user_post', 'likes', type_='unique')

    op.drop_index('idx_posts_author_created', table_name='posts')
    op.drop_constraint('fk_posts_author', 'posts', type_='foreignkey')

    op.drop_constraint('uq_users_username', 'users', type_='unique')
    op.drop_constraint('uq_users_email', 'users', type_='unique')

📊 인덱스 전략​

1. 자주 검색하는 컬럼​

def upgrade():
    # WHERE 절에 자주 사용되는 컬럼
    op.create_index('idx_users_email', 'users', ['email'])
    op.create_index('idx_posts_status', 'posts', ['status'])

2. JOIN에 사용되는 외래 키​

def upgrade():
    # JOIN 성능 향상
    op.create_index('idx_posts_author_id', 'posts', ['author_id'])
    op.create_index('idx_comments_post_id', 'comments', ['post_id'])

3. ORDER BY에 사용되는 컬럼​

def upgrade():
    # 정렬 성능 향상
    op.create_index('idx_posts_created_desc', 'posts',
        [sa.text('created_at DESC')]
    )

4. 복합 인덱스 순서​

def upgrade():
    # ✅ 좋은 예: 선택도가 높은 컬럼 먼저
    op.create_index('idx_orders_user_status_date', 'orders',
        ['user_id', 'status', 'created_at']
    )

    # ❌ 나쁜 예: 선택도가 낮은 컬럼 먼저

    #     ['status', 'user_id']  # status는 값이 몇 개 안 됨
    # )

⚠️ 주의사항​

1. 제약조건 이름 충돌​

# ❌ 에러: 같은 이름의 제약조건 존재
def upgrade():
    op.create_unique_constraint('unique_email', 'users', ['email'])
    op.create_unique_constraint('unique_email', 'admins', ['email'])  # 에러!

# ✅ 올바른 방법: 테이블명 포함
def upgrade():
    op.create_unique_constraint('uq_users_email', 'users', ['email'])
    op.create_unique_constraint('uq_admins_email', 'admins', ['email'])

2. 기존 데이터와 제약조건 충돌​

def upgrade():
    # ❌ 실패: 중복 데이터가 이미 있으면 에러
    op.create_unique_constraint('uq_users_email', 'users', ['email'])

# ✅ 안전한 방법
def upgrade():
    # 1. 중복 데이터 정리
    op.execute('''
        DELETE FROM users a
        USING users b
        WHERE a.id > b.id AND a.email = b.email
    ''')

    # 2. 제약조건 추가
    op.create_unique_constraint('uq_users_email', 'users', ['email'])

3. 과도한 인덱스​

# ❌ 나쁜 예: 너무 많은 인덱스
def upgrade():
    op.create_index('idx_1', 'users', ['email'])
    op.create_index('idx_2', 'users', ['username'])
    op.create_index('idx_3', 'users', ['first_name'])
    op.create_index('idx_4', 'users', ['last_name'])
    op.create_index('idx_5', 'users', ['created_at'])
    # ... 10개 더

# ✅ 좋은 예: 필요한 인덱스만
def upgrade():
    op.create_index('idx_users_email', 'users', ['email'])  # 로그인용
    op.create_index('idx_users_created', 'users', ['created_at'])  # 정렬용

✅ 실습 과제​

블로그 시스템의 성능과 데이터 무결성을 개선하세요:

  1. users 테이블:

    • email은 유니크
    • username은 3자 이상 20자 이하

  2. posts 테이블:

    • author_id는 users.id 외래 키 (CASCADE)
    • view_count는 0 이상
    • status는 'draft', 'published', 'archived' 중 하나

  3. 인덱스:

    • posts: author_id, created_at DESC
    • posts: status
    • comments: post_id
정답 보기
"""add blog constraints and indexes

Revision ID: blog001
"""

def upgrade():
    # Users
    op.create_unique_constraint('uq_users_email', 'users', ['email'])
    op.create_check_constraint(
        'check_username_length',
        'users',
        'LENGTH(username) >= 3 AND LENGTH(username) <= 20'
    )

    # Posts
    op.create_foreign_key(
        'fk_posts_author',
        'posts', 'users',
        ['author_id'], ['id'],
        ondelete='CASCADE'
    )
    op.create_check_constraint(
        'check_view_count',
        'posts',
        'view_count >= 0'
    )
    op.create_check_constraint(
        'check_post_status',
        'posts',
        "status IN ('draft', 'published', 'archived')"
    )

    # Indexes
    op.create_index('idx_posts_author_created', 'posts',
        ['author_id', sa.text('created_at DESC')]
    )
    op.create_index('idx_posts_status', 'posts', ['status'])
    op.create_index('idx_comments_post', 'comments', ['post_id'])

def downgrade():
    op.drop_index('idx_comments_post', table_name='comments')
    op.drop_index('idx_posts_status', table_name='posts')
    op.drop_index('idx_posts_author_created', table_name='posts')

    op.drop_constraint('check_post_status', 'posts')
    op.drop_constraint('check_view_count', 'posts')
    op.drop_constraint('fk_posts_author', 'posts', type_='foreignkey')

    op.drop_constraint('check_username_length', 'users')
    op.drop_constraint('uq_users_email', 'users', type_='unique')

📚 핵심 정리​

제약조건​

  • Primary Key: 기본 키
  • Foreign Key: 외래 키 (CASCADE 옵션)
  • Unique: 중복 방지
  • Check: 조건 검증

인덱스​

  • 검색 성능 향상
  • JOIN, WHERE, ORDER BY에 사용되는 컬럼
  • 과도한 인덱스는 INSERT/UPDATE 성능 저하
