성능 최적화

React 애플리케이션의 성능을 최적화하는 다양한 기법을 배웁니다.

React.memo​

컴포넌트를 메모이제이션하여 불필요한 리렌더링을 방지합니다.

const MemoizedComponent = React.memo(function Component({ name }) {
  console.log('렌더링');
  return <div>안녕하세요, {name}님!</div>;
});

// props가 같으면 리렌더링하지 않음
function Parent() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <button onClick={() => setCount(count + 1)}>
        Count: {count}
      </button>
      <MemoizedComponent name="홍길동" />
    </div>
  );
}

커스텀 비교 함수​

const MemoizedComponent = React.memo(
  function Component({ user }) {
    return <div>{user.name}</div>;
  },
  (prevProps, nextProps) => {
    // true를 반환하면 리렌더링 하지 않음
    return prevProps.user.id === nextProps.user.id;
  }
);

useMemo​

값을 메모이제이션하여 비용이 큰 계산을 최적화합니다.

function ExpensiveComponent({ items }) {
  const total = useMemo(() => {
    console.log('계산 실행');
    return items.reduce((sum, item) => sum + item.value, 0);
  }, [items]);

  return <div>Total: {total}</div>;
}

useCallback​

함수를 메모이제이션하여 자식 컴포넌트의 불필요한 리렌더링을 방지합니다.

function Parent() {
  const [count, setCount] = useState(0);

  const handleClick = useCallback(() => {
    console.log('클릭');
  }, []);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>증가</button>
      <MemoizedChild onClick={handleClick} />
    </div>
  );
}

const MemoizedChild = React.memo(function Child({ onClick }) {
  console.log('Child 렌더링');
  return <button onClick={onClick}>클릭</button>;
});

가상화 (Virtualization)​

긴 리스트를 효율적으로 렌더링하기 위해 react-window를 사용합니다.

import { FixedSizeList } from 'react-window';

function VirtualList({ items }) {
  const Row = ({ index, style }) => (
    <div style={style}>
      {items[index].name}
    </div>
  );

  return (
    <FixedSizeList
      height={600}
      itemCount={items.length}
      itemSize={50}
      width="100%"
    >
      {Row}
    </FixedSizeList>
  );
}

지연 로딩​

import { lazy, Suspense } from 'react';

const HeavyComponent = lazy(() => import('./HeavyComponent'));

function App() {
  return (
    <Suspense fallback={<div>로딩 중...</div>}>
      <HeavyComponent />
    </Suspense>
  );
}

성능 측정​

React DevTools Profiler​

import { Profiler } from 'react';

function onRenderCallback(
  id,
  phase,
  actualDuration,
  baseDuration,
  startTime,
  commitTime
) {
  console.log(`${id} (${phase}) 렌더링 시간: ${actualDuration}ms`);
}

function App() {
  return (
    <Profiler id="App" onRender={onRenderCallback}>
      <Component />
    </Profiler>
  );
}

베스트 프랙티스​

1. 상태를 컴포넌트 가까이 두기​

// ❌ 상위 컴포넌트에 모든 상태
function Parent() {
  const [childAState, setChildAState] = useState();
  const [childBState, setChildBState] = useState();

  return (
    <>
      <ChildA state={childAState} setState={setChildAState} />
      <ChildB state={childBState} setState={setChildBState} />
    </>
  );
}

// ✅ 각 컴포넌트에 상태
function Parent() {
  return (
    <>
      <ChildA />
      <ChildB />
    </>
  );
}

2. 리스트 항목에 안정적인 key 사용​

// ❌ index를 key로 사용
{items.map((item, index) => <Item key={index} />)}

// ✅ 고유 ID 사용
{items.map(item => <Item key={item.id} />)}
