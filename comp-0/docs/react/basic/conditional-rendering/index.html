조건부 렌더링

조건에 따라 다른 UI를 렌더링하는 방법을 알아봅니다.

조건부 렌더링이란?​

JavaScript의 조건문을 사용하여 특정 조건에 따라 다른 컴포넌트나 엘리먼트를 렌더링하는 것입니다.

if 문 사용​

기본 if 문​

function Greeting({ isLoggedIn }) {
  if (isLoggedIn) {
    return <h1>환영합니다!</h1>;
  }
  return <h1>로그인해주세요.</h1>;
}

if-else​

function UserGreeting({ user }) {
  if (user) {
    return <h1>안녕하세요, {user.name}님!</h1>;
  } else {
    return <h1>손님, 환영합니다!</h1>;
  }
}

변수에 저장​

function Dashboard({ isAdmin }) {
  let content;

  if (isAdmin) {
    content = <AdminPanel />;
  } else {
    content = <UserPanel />;
  }

  return (
    <div>
      <h1>대시보드</h1>
      {content}
    </div>
  );
}

삼항 연산자​

가장 많이 사용되는 방법입니다.

function Greeting({ isLoggedIn }) {
  return (
    <div>
      {isLoggedIn ? (
        <h1>환영합니다!</h1>
      ) : (
        <h1>로그인해주세요.</h1>
      )}
    </div>
  );
}

인라인 사용​

function Button({ isLoading }) {
  return (
    <button>
      {isLoading ? '로딩 중...' : '제출'}
    </button>
  );
}

중첩 삼항 연산자​

function Status({ status }) {
  return (
    <div>
      {status === 'success' ? (
        <span style={{ color: 'green' }}>성공</span>
      ) : status === 'error' ? (
        <span style={{ color: 'red' }}>오류</span>
      ) : (
        <span style={{ color: 'gray' }}>대기</span>
      )}
    </div>
  );
}

주의: 중첩이 많아지면 가독성이 떨어지므로 if 문이나 switch 문 사용을 고려하세요.

논리 AND 연산자 (&&)​

조건이 참일 때만 렌더링합니다.

function Mailbox({ unreadMessages }) {
  return (
    <div>
      <h1>메일함</h1>
      {unreadMessages.length > 0 && (
        <h2>
          {unreadMessages.length}개의 읽지 않은 메시지가 있습니다.
        </h2>
      )}
    </div>
  );
}

주의사항​

function Component({ count }) {
  // ❌ count가 0일 때 "0"이 렌더링됨
  return (
    <div>
      {count && <p>{count}개</p>}
    </div>
  );

  // ✅ 명시적 비교
  return (
    <div>
      {count > 0 && <p>{count}개</p>}
    </div>
  );
}

논리 OR 연산자 (||)​

값이 없을 때 기본값을 표시합니다.

function Welcome({ name }) {
  return <h1>안녕하세요, {name || '손님'}님!</h1>;
}

Nullish Coalescing (??)​

function Welcome({ name }) {
  // name이 null 또는 undefined일 때만 '손님' 사용
  return <h1>안녕하세요, {name ?? '손님'}님!</h1>;
}

요소를 변수에 저장​

function LoginControl({ isLoggedIn }) {
  let button;

  if (isLoggedIn) {
    button = <LogoutButton />;
  } else {
    button = <LoginButton />;
  }

  return (
    <div>
      <Greeting isLoggedIn={isLoggedIn} />
      {button}
    </div>
  );
}

null 반환으로 숨기기​

컴포넌트가 아무것도 렌더링하지 않으려면 null을 반환합니다.

function Warning({ message }) {
  if (!message) {
    return null;
  }

  return (
    <div className="warning">
      {message}
    </div>
  );
}

switch 문​

여러 조건을 처리할 때 유용합니다.

function StatusIcon({ status }) {
  let icon;

  switch (status) {
    case 'success':
      icon = '✓';
      break;
    case 'error':
      icon = '✗';
      break;
    case 'warning':
      icon = '⚠';
      break;
    default:
      icon = '○';
  }

  return <span>{icon}</span>;
}

객체 매핑​

function StatusBadge({ status }) {
  const statusConfig = {
    success: { icon: '✓', color: 'green', text: '성공' },
    error: { icon: '✗', color: 'red', text: '오류' },
    warning: { icon: '⚠', color: 'orange', text: '경고' },
    info: { icon: 'ℹ', color: 'blue', text: '정보' }
  };

  const config = statusConfig[status] || statusConfig.info;

  return (
    <span style={{ color: config.color }}>
      {config.icon} {config.text}
    </span>
  );
}

인라인 스타일​

    <button
      style={{
        backgroundColor: isPrimary ? 'blue' : 'gray',

className​

function Alert({ type }) {
  return (
    <div className={`alert ${type === 'error' ? 'alert-error' : 'alert-info'}`}>
      알림
    </div>
  );
}

classnames 라이브러리​

import classNames from 'classnames';

function Button({ primary, large, disabled }) {
  const btnClass = classNames('btn', {
    'btn-primary': primary,
    'btn-large': large,
    'btn-disabled': disabled
  });

  return <button className={btnClass}>버튼</button>;
}

실전 예제​

로그인/로그아웃 UI​

function AuthButton({ isLoggedIn, onLogin, onLogout }) {
  return (
    <div>
      {isLoggedIn ? (
        <div>
          <span>로그인됨</span>
          <button onClick={onLogout}>로그아웃</button>
        </div>
      ) : (
        <button onClick={onLogin}>로그인</button>
      )}
    </div>
  );
}

로딩 상태​

function UserProfile({ user, isLoading, error }) {
  if (isLoading) {
    return <div>로딩 중...</div>;
  }

  if (error) {
    return <div>오류: {error.message}</div>;
  }

  if (!user) {
    return <div>사용자를 찾을 수 없습니다.</div>;
  }

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}

권한 기반 렌더링​

function AdminPanel({ user }) {
  if (!user) {
    return <div>로그인이 필요합니다.</div>;
  }

  if (user.role !== 'admin') {
    return <div>권한이 없습니다.</div>;
  }

  return (
    <div>
      <h1>관리자 패널</h1>
      <button>사용자 관리</button>
      <button>설정</button>
    </div>
  );
}

다단계 폼​

function MultiStepForm() {
  const [step, setStep] = useState(1);

  const renderStep = () => {
    switch (step) {
      case 1:
        return <PersonalInfo onNext={() => setStep(2)} />;
      case 2:
        return <ContactInfo onNext={() => setStep(3)} onBack={() => setStep(1)} />;
      case 3:
        return <Confirmation onBack={() => setStep(2)} />;
      default:
        return <PersonalInfo onNext={() => setStep(2)} />;
    }
  };

  return (
    <div>
      <h1>회원가입 (단계 {step}/3)</h1>
      {renderStep()}
    </div>
  );
}

빈 상태​

function TodoList({ todos }) {
  if (todos.length === 0) {
    return (
      <div className="empty-state">
        <p>할 일이 없습니다.</p>
        <button>새 할 일 추가</button>
      </div>
    );
  }

  return (
    <ul>
      {todos.map(todo => (
        <li key={todo.id}>{todo.text}</li>
      ))}
    </ul>
  );
}

알림 배너​

function NotificationBanner({ notifications }) {
  if (!notifications || notifications.length === 0) {
    return null;
  }

  return (
    <div className="notification-banner">
      {notifications.map(notification => (
        <div
          key={notification.id}
          className={`notification notification-${notification.type}`}
        >
          {notification.message}
        </div>
      ))}
    </div>
  );
}

데이터 테이블​

function DataTable({ data, isLoading, error }) {
  return (
    <div>
      {error && (
        <div className="error">오류: {error.message}</div>
      )}

      {isLoading ? (
        <div>데이터 로딩 중...</div>
      ) : data.length === 0 ? (
        <div>데이터가 없습니다.</div>
      ) : (
        <table>
          <thead>
            <tr>
              <th>이름</th>
              <th>이메일</th>
            </tr>
          </thead>
          <tbody>
            {data.map(item => (
              <tr key={item.id}>
                <td>{item.name}</td>
                <td>{item.email}</td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}

베스트 프랙티스​

1. 단순함 유지​

// ❌ 복잡함
{condition1 ? (condition2 ? <A /> : <B />) : (condition3 ? <C /> : <D />)}

// ✅ 명확함
const getComponent = () => {
  if (condition1 && condition2) return <A />;
  if (condition1) return <B />;
  if (condition3) return <C />;
  return <D />;
};

return getComponent();

2. 조기 반환 사용​

function Component({ data, isLoading, error }) {
  if (isLoading) return <Loading />;
  if (error) return <Error error={error} />;
  if (!data) return <Empty />;

  return <Content data={data} />;
}

3. 의미 있는 변수명​

// ❌
{flag && <Component />}

// ✅
{shouldShowComponent && <Component />}
