📊 성능 지표​

주요 성능 메트릭​

  • Throughput: 초당 메시지 처리량 (msg/s)
  • Latency: 메시지 지연 시간 (ms)

🚀 Producer 최적화​

1. 배치 발행 (Batch Publishing)​

# batch_publisher.py
import pika
import json

class BatchPublisher:
    """배치 발행으로 성능 향상"""

    def __init__(self, host='localhost'):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host)
        )
        self.channel = self.connection.channel()
        self.channel.queue_declare(queue='orders', durable=True)

    def publish_batch(self, messages: list, batch_size: int = 100):
        """
        배치로 메시지 발행

        Args:
            messages: 메시지 리스트
            batch_size: 배치 크기
        """
        for i in range(0, len(messages), batch_size):
            batch = messages[i:i+batch_size]

            # 배치 내 모든 메시지 발행
            for msg in batch:
                self.channel.basic_publish(
                    exchange='',
                    routing_key='orders',
                    body=json.dumps(msg),
                    properties=pika.BasicProperties(
                        delivery_mode=2
                    )
                )

            # 배치마다 confirm
            print(f" [x] Published batch {i//batch_size + 1} "
                  f"({len(batch)} messages)")

    def close(self):
        self.connection.close()

# 성능 비교
if __name__ == '__main__':
    import time

    messages = [{'id': i, 'data': f'Message {i}'} for i in range(10000)]

    # 배치 발행
    start = time.time()
    publisher = BatchPublisher()
    publisher.publish_batch(messages, batch_size=1000)
    publisher.close()
    batch_time = time.time() - start

    print(f"\n Batch publishing: {batch_time:.2f}s")
    print(f" Throughput: {len(messages) / batch_time:.2f} msg/s")

2. Publisher Confirms​

# confirmed_publisher.py
import pika

class ConfirmedPublisher:
    """Publisher Confirms로 신뢰성 향상"""

    def __init__(self, host='localhost'):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host)
        )
        self.channel = self.connection.channel()

        # Publisher Confirms 활성화
        self.channel.confirm_delivery()

        self.channel.queue_declare(queue='orders', durable=True)

    def publish_with_confirm(self, message: str):
        """Confirm과 함께 발행"""
        try:
            self.channel.basic_publish(
                exchange='',
                routing_key='orders',
                body=message,
                properties=pika.BasicProperties(delivery_mode=2),
                mandatory=True  # 라우팅 실패 시 에러
            )
            print(f" [✓] Message confirmed: {message}")
            return True

        except pika.exceptions.UnroutableError:
            print(f" [✗] Message unroutable: {message}")
            return False

        except pika.exceptions.NackError:
            print(f" [✗] Message nacked: {message}")
            return False

    def close(self):
        self.connection.close()

# 사용 예
if __name__ == '__main__':
    publisher = ConfirmedPublisher()

    for i in range(100):
        publisher.publish_with_confirm(f'Order {i+1}')

    publisher.close()

3. 연결 풀링​

# connection_pool.py
import pika
import threading
from queue import Queue

class ConnectionPool:
    """연결 재사용으로 성능 향상"""

    def __init__(self, host='localhost', pool_size=10):
        self.host = host
        self.pool_size = pool_size
        self.connections = Queue(maxsize=pool_size)
        self._lock = threading.Lock()

        # 연결 풀 초기화
        for _ in range(pool_size):
            conn = pika.BlockingConnection(
                pika.ConnectionParameters(host)
            )
            self.connections.put(conn)

    def get_connection(self):
        """연결 가져오기"""
        return self.connections.get()

    def return_connection(self, connection):
        """연결 반환"""
        self.connections.put(connection)

    def publish(self, queue_name: str, message: str):
        """메시지 발행"""
        conn = self.get_connection()

        try:
            channel = conn.channel()
            channel.basic_publish(
                exchange='',
                routing_key=queue_name,
                body=message
            )
        finally:
            self.return_connection(conn)

    def close_all(self):
        """모든 연결 종료"""
        while not self.connections.empty():
            conn = self.connections.get()
            conn.close()

# 사용 예
if __name__ == '__main__':
    pool = ConnectionPool(pool_size=5)

    # 멀티스레드 환경에서 연결 재사용
    def publish_messages(thread_id: int, count: int):
        for i in range(count):
            pool.publish('orders', f'Thread {thread_id}: Message {i+1}')

    threads = []
    for tid in range(10):
        t = threading.Thread(
            target=publish_messages,
            args=(tid, 100)
        )
        t.start()
        threads.append(t)

    for t in threads:
        t.join()

    pool.close_all()
    print(" [✓] All messages published using connection pool")

1. Prefetch Count 조정​

# optimized_consumer.py
import pika
import time

def process_message(body: bytes):
    """메시지 처리 (시뮬레이션)"""
    # 실제 처리 로직
    time.sleep(0.1)

def callback(ch, method, properties, body):
    """메시지 처리 콜백"""
    process_message(body)
    ch.basic_ack(delivery_tag=method.delivery_tag)

def start_consumer(prefetch_count: int = 1):
    """Consumer 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 🔑 Prefetch Count 설정
    # 값이 클수록: 처리량 ↑, 메모리 ↑
    # 값이 작을수록: 공정한 분배 ↑, 처리량 ↓
    channel.basic_qos(prefetch_count=prefetch_count)

    channel.basic_consume(
        queue='orders',
        on_message_callback=callback
    )

    print(f' [*] Consumer started (prefetch={prefetch_count})')
    channel.start_consuming()

# 최적값 찾기: 처리 시간과 부하에 따라 10-50 권장
if __name__ == '__main__':
    import sys
    prefetch = int(sys.argv[1]) if len(sys.argv) > 1 else 10

    try:
        start_consumer(prefetch_count=prefetch)
    except KeyboardInterrupt:
        print('\nConsumer stopped')

2. 멀티스레드 Consumer​

# threaded_consumer.py
import pika
import threading
import time

class ThreadedConsumer:
    """멀티스레드 Consumer"""

    def __init__(self, thread_count=5):
        self.thread_count = thread_count
        self.threads = []

    def process_message(self, body: bytes):
        """메시지 처리"""
        # CPU 집약적인 작업
        time.sleep(0.05)
        print(f" [x] Processed: {body.decode()[:50]}")

    def consumer_thread(self, thread_id: int):
        """Consumer 스레드"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        def callback(ch, method, properties, body):
            self.process_message(body)
            ch.basic_ack(delivery_tag=method.delivery_tag)

        channel.basic_qos(prefetch_count=10)
        channel.basic_consume(
            queue='orders',
            on_message_callback=callback
        )

        print(f' [*] Thread {thread_id} started')
        channel.start_consuming()

    def start(self):
        """모든 스레드 시작"""
        for i in range(self.thread_count):
            t = threading.Thread(
                target=self.consumer_thread,
                args=(i+1,)
            )
            t.start()
            self.threads.append(t)

        # 모든 스레드 대기
        for t in self.threads:
            t.join()

# 사용 예
if __name__ == '__main__':
    consumer = ThreadedConsumer(thread_count=10)

    try:
        consumer.start()
    except KeyboardInterrupt:
        print('\n Threaded Consumer stopped')

Lazy Queue 사용​

# lazy_queue.py
import pika

def setup_lazy_queue():
    """Lazy Queue 설정 (대용량 메시지 처리)"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 🔑 Lazy Queue: 메시지를 디스크에 저장
    channel.queue_declare(
        queue='large_messages',
        durable=True,
        arguments={
            'x-queue-mode': 'lazy'  # Lazy mode
        }
    )

    print(" [✓] Lazy queue created")
    print(" [i] Messages will be stored on disk")

    connection.close()

if __name__ == '__main__':
    setup_lazy_queue()

📊 벤치마킹​

성능 측정 도구​

# benchmark.py
import pika
import time
import statistics
from dataclasses import dataclass
from typing import List

@dataclass
class BenchmarkResult:
    """벤치마크 결과"""
    operation: str
    count: int
    duration: float
    throughput: float
    latencies: List[float]

    def print_summary(self):
        """결과 요약 출력"""
        print(f"\n {self.operation}")
        print(" " + "="*50)
        print(f" Messages: {self.count}")
        print(f" Duration: {self.duration:.2f}s")
        print(f" Throughput: {self.throughput:.2f} msg/s")

        if self.latencies:
            print(f" Latency (avg): {statistics.mean(self.latencies)*1000:.2f}ms")
            print(f" Latency (p50): {statistics.median(self.latencies)*1000:.2f}ms")
            print(f" Latency (p95): {self._percentile(self.latencies, 0.95)*1000:.2f}ms")
            print(f" Latency (p99): {self._percentile(self.latencies, 0.99)*1000:.2f}ms")

    @staticmethod
    def _percentile(data: List[float], percentile: float) -> float:
        """백분위수 계산"""
        sorted_data = sorted(data)
        index = int(len(sorted_data) * percentile)
        return sorted_data[min(index, len(sorted_data)-1)]

class RabbitMQBenchmark:
    """RabbitMQ 벤치마크"""

    def __init__(self, host='localhost'):
        self.host = host

    def benchmark_publish(self, count: int = 10000) -> BenchmarkResult:
        """발행 성능 측정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()
        channel.queue_declare(queue='benchmark', durable=True)

        latencies = []
        start_time = time.time()

        for i in range(count):
            msg_start = time.time()

            channel.basic_publish(
                exchange='',
                routing_key='benchmark',
                body=f'Message {i+1}',
                properties=pika.BasicProperties(delivery_mode=2)
            )

            latencies.append(time.time() - msg_start)

        duration = time.time() - start_time
        connection.close()

        return BenchmarkResult(
            operation='Publish',
            count=count,
            duration=duration,
            throughput=count / duration,
            latencies=latencies
        )

    def benchmark_consume(self, count: int = 10000) -> BenchmarkResult:
        """소비 성능 측정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        received = 0
        latencies = []
        start_time = time.time()

        def callback(ch, method, properties, body):
            nonlocal received
            msg_start = time.time()

            # 메시지 처리 (시뮬레이션)
            _ = body.decode()

            latencies.append(time.time() - msg_start)
            received += 1

            ch.basic_ack(delivery_tag=method.delivery_tag)

        channel.basic_qos(prefetch_count=100)
        channel.basic_consume(
            queue='benchmark',
            on_message_callback=callback
        )

        while received < count:
            connection.process_data_events(time_limit=1)

        duration = time.time() - start_time
        connection.close()

        return BenchmarkResult(
            operation='Consume',
            count=count,
            duration=duration,
            throughput=count / duration,
            latencies=latencies
        )

# 사용 예
if __name__ == '__main__':
    benchmark = RabbitMQBenchmark()

    # 발행 벤치마크
    publish_result = benchmark.benchmark_publish(count=10000)
    publish_result.print_summary()

    # 소비 벤치마크
    consume_result = benchmark.benchmark_consume(count=10000)
    consume_result.print_summary()
