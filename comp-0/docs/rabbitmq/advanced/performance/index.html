class ConnectionPool:
    """ì—°ê²° ì¬ì‚¬ìš©ìœ¼ë¡œ ì„±ëŠ¥ í–¥ìƒ"""

    def __init__(self, host='localhost', pool_size=10):

        self._lock = threading.Lock()

        # ì—°ê²° í’€ ì´ˆê¸°í™”

                pika.ConnectionParameters(host)

        """ì—°ê²° ê°€ì ¸ì˜¤ê¸°"""
        return self.connections.get()

    def return_connection(self, connection):
        """ì—°ê²° ë°˜í™˜"""
        self.connections.put(connection)

    def publish(self, queue_name: str, message: str):
        """ë©”ì‹œì§€ ë°œí–‰"""

        try:
            channel = conn.channel()
            channel.basic_publish(
                exchange='',
                routing_key=queue_name,
                body=message
            )
        finally:
            self.return_connection(conn)

    def close_all(self):
        """ëª¨ë“  ì—°ê²° ì¢…ë£Œ"""
        while not self.connections.empty():
            conn = self.connections.get()
            conn.close()

# benchmark.py
import pika
import time
import statistics
from dataclasses import dataclass
from typing import List

@dataclass
class BenchmarkResult:
    """ë²¤ì¹˜ë§ˆí¬ ê²°ê³¼"""
    operation: str
    count: int
    duration: float
    throughput: float
    latencies: List[float]

    def print_summary(self):
        """ê²°ê³¼ ìš”ì•½ ì¶œë ¥"""
        print(f"\n {self.operation}")
        print(" " + "="*50)
        print(f" Messages: {self.count}")
        print(f" Duration: {self.duration:.2f}s")
        print(f" Throughput: {self.throughput:.2f} msg/s")

        if self.latencies:
            print(f" Latency (avg): {statistics.mean(self.latencies)*1000:.2f}ms")
            print(f" Latency (p50): {statistics.median(self.latencies)*1000:.2f}ms")
            print(f" Latency (p95): {self._percentile(self.latencies, 0.95)*1000:.2f}ms")
            print(f" Latency (p99): {self._percentile(self.latencies, 0.99)*1000:.2f}ms")

    @staticmethod
    def _percentile(data: List[float], percentile: float) -> float:
        """ë°±ë¶„ìœ„ìˆ˜ ê³„ì‚°"""
        sorted_data = sorted(data)
        index = int(len(sorted_data) * percentile)
        return sorted_data[min(index, len(sorted_data)-1)]

class RabbitMQBenchmark:
    """RabbitMQ ë²¤ì¹˜ë§ˆí¬"""

    def __init__(self, host='localhost'):
        self.host = host

    def benchmark_publish(self, count: int = 10000) -> BenchmarkResult:
        """ë°œí–‰ ì„±ëŠ¥ ì¸¡ì •"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()
        channel.queue_declare(queue='benchmark', durable=True)

        latencies = []
        start_time = time.time()

        for i in range(count):
            msg_start = time.time()

            channel.basic_publish(
                exchange='',
                routing_key='benchmark',
                body=f'Message {i+1}',
                properties=pika.BasicProperties(delivery_mode=2)
            )

            latencies.append(time.time() - msg_start)

        duration = time.time() - start_time
        connection.close()

        return BenchmarkResult(
            operation='Publish',
            count=count,
            duration=duration,
            throughput=count / duration,
            latencies=latencies
        )

    def benchmark_consume(self, count: int = 10000) -> BenchmarkResult:
        """ì†Œë¹„ ì„±ëŠ¥ ì¸¡ì •"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        received = 0
        latencies = []
        start_time = time.time()

        def callback(ch, method, properties, body):
            nonlocal received
            msg_start = time.time()

            # ë©”ì‹œì§€ ì²˜ë¦¬ (ì‹œë®¬ë ˆì´ì…˜)
            _ = body.decode()

            latencies.append(time.time() - msg_start)
            received += 1

            ch.basic_ack(delivery_tag=method.delivery_tag)

        channel.basic_qos(prefetch_count=100)
        channel.basic_consume(
            queue='benchmark',
            on_message_callback=callback
        )

        while received < count:
            connection.process_data_events(time_limit=1)

        duration = time.time() - start_time
        connection.close()

        return BenchmarkResult(
            operation='Consume',
            count=count,
            duration=duration,
            throughput=count / duration,
            latencies=latencies
        )
