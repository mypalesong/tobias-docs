🔐 사용자 및 권한 관리​

사용자 생성 및 관리​

# 1. 사용자 생성
docker exec rabbitmq rabbitmqctl add_user developer dev_password123

# 2. 태그 부여 (역할)
docker exec rabbitmq rabbitmqctl set_user_tags developer management

# 3. 권한 부여
docker exec rabbitmq rabbitmqctl set_permissions -p / developer ".*" ".*" ".*"

# 4. 사용자 목록
docker exec rabbitmq rabbitmqctl list_users

# 5. 비밀번호 변경
docker exec rabbitmq rabbitmqctl change_password developer new_password456

# 6. 사용자 삭제
docker exec rabbitmq rabbitmqctl delete_user developer

태그 (역할)​

# administrator: 모든 권한
rabbitmqctl set_user_tags admin administrator

# management: Management UI 접근
rabbitmqctl set_user_tags developer management

# monitoring: 읽기 전용 (모니터링용)
rabbitmqctl set_user_tags monitor monitoring

# 여러 태그 동시 부여
rabbitmqctl set_user_tags admin administrator management

권한 설정​

# set_permissions [-p vhost] <user> <conf> <write> <read>
# conf: 큐/exchange 설정 권한
# write: 메시지 발행 권한
# read: 메시지 소비 권한

# 모든 권한
rabbitmqctl set_permissions -p / admin ".*" ".*" ".*"

# 특정 큐만 접근
rabbitmqctl set_permissions -p / developer "^orders.*" "^orders.*" "^orders.*"

# 읽기 전용
rabbitmqctl set_permissions -p / monitor "" "" ".*"

# 권한 목록
rabbitmqctl list_permissions -p /

vHost 생성 및 관리​

# 1. vHost 생성

Python에서 vHost 사용​

# vhost_example.py
import pika

class VHostClient:
    """vHost별 클라이언트"""

    def __init__(
        self,
        host='localhost',
        vhost='/',
        username='guest',
        password='guest'
    ):
        credentials = pika.PlainCredentials(username, password)

        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(
                host=host,
                virtual_host=vhost,  # vHost 지정
                credentials=credentials
            )
        )
        self.channel = self.connection.channel()

    def publish(self, queue_name: str, message: str):
        """메시지 발행"""
        self.channel.queue_declare(queue=queue_name, durable=True)

        self.channel.basic_publish(
            exchange='',
            routing_key=queue_name,
            body=message
        )

        print(f" [x] Sent to vHost: {message}")

    def close(self):
        self.connection.close()

# 사용 예
if __name__ == '__main__':
    # Production vHost
    prod_client = VHostClient(
        vhost='production',
        username='prod_user',
        password='prod_password'
    )
    prod_client.publish('orders', 'Production order')
    prod_client.close()

    # Development vHost
    dev_client = VHostClient(
        vhost='development',
        username='dev_user',
        password='dev_password'
    )
    dev_client.publish('orders', 'Development order')
    dev_client.close()

🔒 TLS/SSL 설정​

인증서 생성 (Self-Signed)​

# tls_setup.sh

#!/bin/bash

# 1. CA 키 생성
openssl genrsa -out ca_key.pem 2048

# 2. CA 인증서 생성
openssl req -new -x509 -key ca_key.pem -out ca_certificate.pem -days 3650 \
  -subj "/C=KR/ST=Seoul/L=Seoul/O=MyCompany/CN=MyCA"

# 3. 서버 키 생성
openssl genrsa -out server_key.pem 2048

# 4. 서버 CSR 생성
openssl req -new -key server_key.pem -out server.csr \
  -subj "/C=KR/ST=Seoul/L=Seoul/O=MyCompany/CN=rabbitmq"

# 5. 서버 인증서 생성
openssl x509 -req -in server.csr -CA ca_certificate.pem -CAkey ca_key.pem \
  -CAcreateserial -out server_certificate.pem -days 365

# 6. 클라이언트 키 생성
openssl genrsa -out client_key.pem 2048

# 7. 클라이언트 CSR 생성
openssl req -new -key client_key.pem -out client.csr \
  -subj "/C=KR/ST=Seoul/L=Seoul/O=MyCompany/CN=client"

# 8. 클라이언트 인증서 생성
openssl x509 -req -in client.csr -CA ca_certificate.pem -CAkey ca_key.pem \
  -CAcreateserial -out client_certificate.pem -days 365

echo "✓ TLS certificates generated"

RabbitMQ TLS 설정​

# rabbitmq.conf

# TLS listeners
listeners.ssl.default = 5671

# 인증서 파일 경로
ssl_options.cacertfile = /etc/rabbitmq/certs/ca_certificate.pem
ssl_options.certfile = /etc/rabbitmq/certs/server_certificate.pem
ssl_options.keyfile = /etc/rabbitmq/certs/server_key.pem

# TLS 버전 (1.2 이상)
ssl_options.versions.1 = tlsv1.2
ssl_options.versions.2 = tlsv1.3

# 클라이언트 인증 요구
ssl_options.verify = verify_peer
ssl_options.fail_if_no_peer_cert = true

Python TLS 연결​

# tls_client.py
import pika
import ssl

class TLSClient:
    """TLS/SSL 연결 클라이언트"""

    def __init__(self, host='localhost'):
        # SSL Context 생성
        ssl_context = ssl.create_default_context(
            cafile='certs/ca_certificate.pem'
        )
        ssl_context.load_cert_chain(
            certfile='certs/client_certificate.pem',
            keyfile='certs/client_key.pem'
        )

        # SSL Options
        ssl_options = pika.SSLOptions(ssl_context, host)

        # 연결 파라미터
        parameters = pika.ConnectionParameters(
            host=host,
            port=5671,  # TLS 포트
            ssl_options=ssl_options,
            credentials=pika.PlainCredentials('admin', 'admin123')
        )

        self.connection = pika.BlockingConnection(parameters)
        self.channel = self.connection.channel()

        print(" [✓] TLS connection established")

    def publish(self, queue_name: str, message: str):
        """메시지 발행"""
        self.channel.queue_declare(queue=queue_name, durable=True)

        self.channel.basic_publish(
            exchange='',
            routing_key=queue_name,
            body=message
        )

        print(f" [x] Sent via TLS: {message}")

    def close(self):
        self.connection.close()

# 사용 예
if __name__ == '__main__':
    client = TLSClient()
    client.publish('secure_orders', 'Encrypted message')
    client.close()

방화벽 설정 (iptables)​

# firewall_setup.sh

#!/bin/bash

# 기본 정책: 모두 차단
iptables -P INPUT DROP
iptables -P FORWARD DROP
iptables -P OUTPUT ACCEPT

# Loopback 허용
iptables -A INPUT -i lo -j ACCEPT

# 기존 연결 허용
iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT

# RabbitMQ 포트 허용 (특정 IP만)
ALLOWED_IPS=("10.0.0.100" "10.0.0.101")

for ip in "${ALLOWED_IPS[@]}"; do
  # AMQP
  iptables -A INPUT -p tcp -s $ip --dport 5672 -j ACCEPT

  # AMQP/TLS
  iptables -A INPUT -p tcp -s $ip --dport 5671 -j ACCEPT

  # Management UI (내부 네트워크만)
  iptables -A INPUT -p tcp -s $ip --dport 15672 -j ACCEPT
done

# Prometheus (모니터링 서버만)
iptables -A INPUT -p tcp -s 10.0.0.200 --dport 15692 -j ACCEPT

# 나머지 차단
iptables -A INPUT -j DROP

echo "✓ Firewall rules applied"

Python 감사 로그​

# audit_logger.py
import pika
import json
import logging
from datetime import datetime

class AuditLogger:
    """감사 로깅"""

    def __init__(self):
        # 로거 설정
        logging.basicConfig(
            level=logging.INFO,
            format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
            handlers=[
                logging.FileHandler('audit.log'),
                logging.StreamHandler()
            ]
        )
        self.logger = logging.getLogger('RabbitMQ-Audit')

    def log_connection(self, user: str, vhost: str, action: str):
        """연결 로그"""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'type': 'connection',
            'user': user,
            'vhost': vhost,
            'action': action
        }
        self.logger.info(json.dumps(log_entry))

    def log_publish(self, user: str, exchange: str, routing_key: str):
        """발행 로그"""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'type': 'publish',
            'user': user,
            'exchange': exchange,
            'routing_key': routing_key
        }
        self.logger.info(json.dumps(log_entry))

    def log_consume(self, user: str, queue: str):
        """소비 로그"""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'type': 'consume',
            'user': user,
            'queue': queue
        }
        self.logger.info(json.dumps(log_entry))

# 사용 예
if __name__ == '__main__':
    audit = AuditLogger()

    audit.log_connection('admin', '/', 'connect')
    audit.log_publish('admin', 'orders_exchange', 'order.created')
    audit.log_consume('worker1', 'orders_queue')

# security_check.py
import requests
from requests.auth import HTTPBasicAuth

class SecurityChecker:
    """보안 점검 도구"""

    def __init__(self, host='localhost', user='admin', password='admin'):
        self.base_url = f'http://{host}:15672/api'
        self.auth = HTTPBasicAuth(user, password)

    def check_default_user(self):
        """기본 사용자 체크"""
        response = requests.get(
            f'{self.base_url}/users',
            auth=self.auth
        )
        users = response.json()

        issues = []
        for user in users:
            if user['name'] == 'guest':
                issues.append("⚠️  Default 'guest' user is enabled")

        return issues

    def check_weak_passwords(self):
        """약한 비밀번호 체크 (길이만)"""
        # 실제로는 비밀번호 정책 서버에서 확인
        return []

    def check_permissions(self):
        """과도한 권한 체크"""
        response = requests.get(
            f'{self.base_url}/users',
            auth=self.auth
        )
        users = response.json()

        issues = []
        for user in users:
            tags = user.get('tags', '')
            if 'administrator' in tags and user['name'] != 'admin':
                issues.append(
                    f"⚠️  User '{user['name']}' has administrator tag"
                )

        return issues

    def check_tls(self):
        """TLS 활성화 체크"""
        # 실제로는 포트 5671 체크
        issues = []
        # 시뮬레이션
        # if not tls_enabled:
        #     issues.append("⚠️  TLS is not enabled")

        return issues

    def run_all_checks(self):
        """모든 체크 실행"""
        print(" Security Check Report")
        print(" " + "="*50)

        all_issues = []

        # 체크 실행
        checks = [
            ("Default Users", self.check_default_user),
            ("Weak Passwords", self.check_weak_passwords),
            ("Permissions", self.check_permissions),
            ("TLS", self.check_tls),
        ]

        for check_name, check_func in checks:
            print(f"\n Checking {check_name}...")
            issues = check_func()

            if issues:
                all_issues.extend(issues)
                for issue in issues:
                    print(f"   {issue}")
            else:
                print("   ✓ OK")

        # 요약
        print(f"\n Summary:")
        print(f" Total issues found: {len(all_issues)}")

        if all_issues:
            print("\n Action Required:")
            for issue in all_issues:
                print(f"   • {issue}")

# 사용 예
if __name__ == '__main__':
    checker = SecurityChecker()
    checker.run_all_checks()
