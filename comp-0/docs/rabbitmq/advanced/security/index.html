class VHostClient:
    """vHostë³„ í´ë¼ì´ì–¸íŠ¸"""

    def __init__(
        self,
        host='localhost',
        vhost='/',
        username='guest',
        password='guest'

                virtual_host=vhost,  # vHost ì§€ì •

        )

        self.connection.close()
