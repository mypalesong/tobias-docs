docker exec -it rabbitmq-1 rabbitmqctl cluster_status

💡 Quorum Queue (권장)​

Python 코드로 Quorum Queue 생성​

# quorum_queue_setup.py
import pika

def setup_quorum_queue():
    """Quorum Queue 생성"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', 5672)
    )
    channel = connection.channel()

    # 🔑 Quorum Queue 선언
    channel.queue_declare(
        queue='orders.quorum',
        durable=True,
        arguments={
            'x-queue-type': 'quorum',  # Quorum Queue
            'x-max-length': 10000,     # 최대 메시지 수
            'x-max-in-memory-length': 0  # 모두 디스크에 저장
        }
    )

    print(" [✓] Quorum queue created: orders.quorum")
    connection.close()

if __name__ == '__main__':
    setup_quorum_queue()

Quorum Queue 특징:

  • ✅ 데이터 안전성 (Replication)
  • ✅ 자동 장애 복구
  • ✅ Poison Message 처리
  • ❌ 메시지 우선순위 미지원
  • ❌ TTL 미지원 (큐 레벨만 가능)

Producer & Consumer​

# producer.py
import pika
import json

def send_order(order_id: str):
    """주문 발행 (Quorum Queue)"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', 5672)
    )
    channel = connection.channel()

    order_data = {
        'order_id': order_id,
        'amount': 100.0,
        'status': 'pending'
    }

    channel.basic_publish(
        exchange='',
        routing_key='orders.quorum',
        body=json.dumps(order_data),
        properties=pika.BasicProperties(
            delivery_mode=2,  # Persistent
            content_type='application/json'
        )
    )

    print(f" [x] Order sent: {order_id}")
    connection.close()

# 사용 예
if __name__ == '__main__':
    for i in range(10):
        send_order(f'ORD-{i+1:03d}')
# consumer.py
import pika
import json

def callback(ch, method, properties, body):
    """주문 처리"""
    order = json.loads(body.decode())
    print(f" [x] Processing order: {order['order_id']}")

    # 처리 후 ACK
    ch.basic_ack(delivery_tag=method.delivery_tag)

def start_consumer(port=5672):
    """Consumer 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', port)
    )
    channel = connection.channel()

    channel.basic_qos(prefetch_count=1)

    channel.basic_consume(
        queue='orders.quorum',
        on_message_callback=callback
    )

    print(f' [*] Consumer waiting on port {port}...')
    channel.start_consuming()

if __name__ == '__main__':
    import sys
    port = int(sys.argv[1]) if len(sys.argv) > 1 else 5672

    try:
        start_consumer(port)
    except KeyboardInterrupt:
        print('\nConsumer stopped')

🧪 장애 복구 테스트​

시나리오 1: 노드 장애​

# test_failover.py
import pika
import time
import subprocess

def test_node_failure():
    """노드 장애 테스트"""

    # 1. Producer 시작 (Node 1 사용)
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', 5672)
    )
    channel = connection.channel()

    # 메시지 10개 발행
    for i in range(10):
        channel.basic_publish(
            exchange='',
            routing_key='orders.quorum',
            body=f'Order {i+1}'
        )
        print(f" [x] Sent: Order {i+1}")
        time.sleep(0.5)

    connection.close()

    # 2. Node 1 중지 (장애 시뮬레이션)
    print("\n [!] Stopping Node 1...")
    subprocess.run(['docker-compose', 'stop', 'rabbitmq-1'])
    time.sleep(3)

    # 3. Node 2로 연결 시도
    print("\n [→] Connecting to Node 2...")
    try:
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost', 5673)
        )
        channel = connection.channel()

        # Consumer로 메시지 확인
        for _ in range(10):
            method, properties, body = channel.basic_get(
                queue='orders.quorum',
                auto_ack=True
            )
            if method:
                print(f" [✓] Received from Node 2: {body.decode()}")

        connection.close()
        print("\n [✓] Failover successful! Messages preserved.")

    except Exception as e:
        print(f"\n [✗] Failover failed: {e}")

    # 4. Node 1 재시작
    print("\n [→] Restarting Node 1...")
    subprocess.run(['docker-compose', 'start', 'rabbitmq-1'])

if __name__ == '__main__':
    test_node_failure()

💡 Classic Mirrored Queue (레거시)​

# mirrored_queue_setup.py
import pika

def setup_mirrored_queue():
    """Classic Mirrored Queue 생성"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', 5672)
    )
    channel = connection.channel()

    # Policy 설정 (Management UI 또는 CLI로)
    # rabbitmqctl set_policy ha-all "^orders\." \
    #   '{"ha-mode":"all","ha-sync-mode":"automatic"}' \
    #   --apply-to queues

    # Queue 선언
    channel.queue_declare(
        queue='orders.mirrored',
        durable=True
    )

    print(" [✓] Mirrored queue created: orders.mirrored")
    print(" [i] Apply policy: ha-all")
    connection.close()

if __name__ == '__main__':
    setup_mirrored_queue()

Policy 설정 (CLI)​

# 모든 큐 미러링 (ha-all)
docker exec rabbitmq-1 rabbitmqctl set_policy ha-all \
  "^" \
  '{"ha-mode":"all","ha-sync-mode":"automatic"}' \
  --apply-to queues

# 특정 패턴 미러링
docker exec rabbitmq-1 rabbitmqctl set_policy ha-orders \
  "^orders\." \
  '{"ha-mode":"exactly","ha-params":2,"ha-sync-mode":"automatic"}' \
  --apply-to queues

# Policy 목록
docker exec rabbitmq-1 rabbitmqctl list_policies

⚖️ Quorum Queue vs Classic Mirrored Queue​

특징Quorum QueueClassic Mirrored
데이터 안전성✅ Raft 합의 알고리즘⚠️ Master-Slave
성능⚡ 빠름🐌 느림
메모리 사용💾 효율적💾 높음
Poison Message✅ 자동 처리❌ 수동 처리
메시지 우선순위❌ 미지원✅ 지원
TTL⚠️ 큐 레벨만✅ 완전 지원
권장 사용✅ 신규 프로젝트❌ 레거시만

📊 모니터링​

Python으로 클러스터 상태 확인​

# cluster_monitor.py
import requests
from requests.auth import HTTPBasicAuth

def get_cluster_status():
    """클러스터 상태 조회"""
    url = 'http://localhost:15672/api/nodes'
    auth = HTTPBasicAuth('admin', 'admin123')

    response = requests.get(url, auth=auth)
    nodes = response.json()

    print(" Cluster Status:")
    print(" " + "="*50)

    for node in nodes:
        print(f" Node: {node['name']}")
        print(f"   Running: {node['running']}")
        print(f"   Memory: {node['mem_used'] / 1024 / 1024:.2f} MB")
        print(f"   Disk Free: {node['disk_free'] / 1024 / 1024:.2f} MB")
        print(f"   Uptime: {node['uptime'] / 1000:.0f}s")
        print()

if __name__ == '__main__':
    get_cluster_status()

⚠️ 주의사항​

1. Erlang Cookie 일치​

# 모든 노드의 Erlang Cookie가 같아야 함
echo "secret_cookie_value" > /var/lib/rabbitmq/.erlang.cookie
chmod 400 /var/lib/rabbitmq/.erlang.cookie

2. 디스크 공간 주의​

# 디스크 부족 시 노드 정지
# /etc/rabbitmq/rabbitmq.conf
disk_free_limit.relative = 1.5

3. 메모리 관리​

# 메모리 한계 설정
# /etc/rabbitmq/rabbitmq.conf
vm_memory_high_watermark.relative = 0.4

📚 핵심 정리​

  • Quorum Queue: 신규 프로젝트에 권장, Raft 합의 알고리즘
  • Classic Mirrored: 레거시, 마이그레이션 계획 필요
  • Erlang Cookie: 클러스터 인증에 필수
  • 장애 복구: 자동 Failover, 메시지 보존
