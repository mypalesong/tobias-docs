class RabbitMQMetrics:
    """RabbitMQ ë©”íŠ¸ë¦­ ìˆ˜ì§‘"""

    def __init__(self, host='localhost', port=15672, user='admin', password='admin123'):
        self.base_url = f'http://{host}:{port}/api'
        self.auth = HTTPBasicAuth(user, password)

    def get_overview(self):
        """ì „ì²´ ê°œìš” ì¡°íšŒ"""
        response = requests.get(
            f'{self.base_url}/overview',
            auth=self.auth
        )
        return response.json()

    def get_queues(self):
        """ëª¨ë“  í ì •ë³´ ì¡°íšŒ"""
        response = requests.get(
            f'{self.base_url}/queues',
            auth=self.auth
        )
        return response.json()

    def get_queue_details(self, vhost, queue_name):
        """íŠ¹ì • í ìƒì„¸ ì •ë³´"""
        vhost_encoded = vhost.replace('/', '%2F')
        response = requests.get(
            f'{self.base_url}/queues/{vhost_encoded}/{queue_name}',
            auth=self.auth
        )
        return response.json()

    def get_connections(self):
        """ì—°ê²° ì •ë³´ ì¡°íšŒ"""
        response = requests.get(
            f'{self.base_url}/connections',
            auth=self.auth
        )
        return response.json()

    def get_nodes(self):
        """ë…¸ë“œ ì •ë³´ ì¡°íšŒ"""
        response = requests.get(
            f'{self.base_url}/nodes',
            auth=self.auth
        )
        return response.json()

    def print_summary(self):
        """ë©”íŠ¸ë¦­ ìš”ì•½ ì¶œë ¥"""
        overview = self.get_overview()
        queues = self.get_queues()
        connections = self.get_connections()

        print(" RabbitMQ Metrics Summary")
        print(" " + "="*50)

        # Message stats
        msg_stats = overview.get('message_stats', {})
        print(f" Messages Published: {msg_stats.get('publish', 0)}")
        print(f" Messages Delivered: {msg_stats.get('deliver_get', 0)}")
        print(f" Messages Acked: {msg_stats.get('ack', 0)}")

        # Queue stats
        total_messages = sum(q.get('messages', 0) for q in queues)
        total_ready = sum(q.get('messages_ready', 0) for q in queues)
        total_unacked = sum(q.get('messages_unacknowledged', 0) for q in queues)

        print(f"\n Total Messages: {total_messages}")
        print(f" Messages Ready: {total_ready}")
        print(f" Messages Unacked: {total_unacked}")

        # Connection stats
        print(f"\n Active Connections: {len(connections)}")
        print(f" Total Queues: {len(queues)}")

        # Top queues
        print(f"\n Top Queues by Messages:")
        sorted_queues = sorted(
            queues,
            key=lambda q: q.get('messages', 0),
            reverse=True
        )[:5]

        for q in sorted_queues:
            print(f"   {q['name']}: {q.get('messages', 0)} messages")

# realtime_monitor.py
import time
from metrics_collector import RabbitMQMetrics

def monitor_queue(queue_name: str, interval: int = 5):
    """í ì‹¤ì‹œê°„ ëª¨ë‹ˆí„°ë§"""
    metrics = RabbitMQMetrics()

    print(f" Monitoring queue: {queue_name}")
    print(f" Refresh interval: {interval}s")
    print(" Press Ctrl+C to stop\n")

    try:
        while True:
            queue = metrics.get_queue_details('/', queue_name)

            messages = queue.get('messages', 0)
            ready = queue.get('messages_ready', 0)
            unacked = queue.get('messages_unacknowledged', 0)
            consumers = queue.get('consumers', 0)

            # Message rates
            msg_stats = queue.get('message_stats', {})
            publish_rate = msg_stats.get('publish_details', {}).get('rate', 0)
            deliver_rate = msg_stats.get('deliver_get_details', {}).get('rate', 0)

            print(f"[{time.strftime('%H:%M:%S')}] "
                  f"Total: {messages:4d} | "
                  f"Ready: {ready:4d} | "
                  f"Unacked: {unacked:4d} | "
                  f"Consumers: {consumers:2d} | "
                  f"Pub/s: {publish_rate:5.1f} | "
                  f"Del/s: {deliver_rate:5.1f}")

            time.sleep(interval)

    except KeyboardInterrupt:
        print("\n Monitoring stopped")

import requests
from metrics_collector import RabbitMQMetrics
import time

class AlertBot:
    """RabbitMQ ì•Œë¦¼ ë´‡"""

    def __init__(self, slack_webhook=None):
        self.metrics = RabbitMQMetrics()
        self.slack_webhook = slack_webhook

    def check_queue_backlog(self, queue_name: str, threshold: int = 1000):
        """í ë©”ì‹œì§€ ëˆ„ì  ì²´í¬"""
        queue = self.metrics.get_queue_details('/', queue_name)
        messages_ready = queue.get('messages_ready', 0)

        if messages_ready > threshold:
            message = (
                f"âš ï¸ Alert: Queue '{queue_name}' has {messages_ready} "
                f"messages waiting (threshold: {threshold})"
            )
            self.send_alert(message)
            return True
        return False

    def check_memory_usage(self, threshold_percent: float = 0.8):
        """ë©”ëª¨ë¦¬ ì‚¬ìš©ëŸ‰ ì²´í¬"""
        nodes = self.metrics.get_nodes()

        for node in nodes:
            mem_used = node.get('mem_used', 0)
            mem_limit = node.get('mem_limit', 1)
            usage_percent = mem_used / mem_limit

            if usage_percent > threshold_percent:
                message = (
                    f"ğŸš¨ Critical: Node '{node['name']}' memory usage "
                    f"{usage_percent:.1%} (threshold: {threshold_percent:.1%})"
                )
                self.send_alert(message)
                return True
        return False

    def check_disk_space(self, threshold_gb: float = 1.0):
        """ë””ìŠ¤í¬ ê³µê°„ ì²´í¬"""
        nodes = self.metrics.get_nodes()

        for node in nodes:
            disk_free = node.get('disk_free', 0)
            disk_free_gb = disk_free / (1024 ** 3)

            if disk_free_gb < threshold_gb:
                message = (
                    f"âš ï¸ Warning: Node '{node['name']}' has only "
                    f"{disk_free_gb:.2f}GB free disk space"
                )
                self.send_alert(message)
                return True
        return False

    def check_no_consumers(self):
        """Consumer ì—†ëŠ” í ì²´í¬"""
        queues = self.metrics.get_queues()

        for queue in queues:
            consumers = queue.get('consumers', 0)
            messages = queue.get('messages', 0)

            if consumers == 0 and messages > 0:
                message = (
                    f"âš ï¸ Warning: Queue '{queue['name']}' has "
                    f"{messages} messages but no consumers"
                )
                self.send_alert(message)

    def send_alert(self, message: str):
        """ì•Œë¦¼ ì „ì†¡"""
        print(f" [ALERT] {message}")

        # Slack ì „ì†¡
        if self.slack_webhook:
            payload = {'text': message}
            try:
                requests.post(self.slack_webhook, json=payload)
            except Exception as e:
                print(f" [ERROR] Failed to send Slack alert: {e}")

    def run_continuous(self, interval: int = 60):
        """ì§€ì†ì ì¸ ëª¨ë‹ˆí„°ë§"""
        print(f" Alert Bot started (interval: {interval}s)")

        try:
            while True:
                self.check_memory_usage()
                self.check_disk_space()
                self.check_no_consumers()

                time.sleep(interval)

        except KeyboardInterrupt:
            print("\n Alert Bot stopped")

# performance_test.py
import pika
import threading
import time
from metrics_collector import RabbitMQMetrics

class PerformanceTester:
    """ì„±ëŠ¥ í…ŒìŠ¤íŠ¸"""

    def __init__(self):
        self.metrics = RabbitMQMetrics()
        self.sent_count = 0
        self.received_count = 0
        self.running = False

    def producer_thread(self, queue_name: str, count: int):
        """Producer ìŠ¤ë ˆë“œ"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()
        channel.queue_declare(queue=queue_name, durable=True)

        for i in range(count):
            channel.basic_publish(
                exchange='',
                routing_key=queue_name,
                body=f'Message {i+1}',
                properties=pika.BasicProperties(delivery_mode=2)
            )
            self.sent_count += 1

        connection.close()

    def consumer_thread(self, queue_name: str):
        """Consumer ìŠ¤ë ˆë“œ"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        def callback(ch, method, properties, body):
            self.received_count += 1
            ch.basic_ack(delivery_tag=method.delivery_tag)

        channel.basic_qos(prefetch_count=10)
        channel.basic_consume(
            queue=queue_name,
            on_message_callback=callback
        )

        while self.running:
            connection.process_data_events(time_limit=1)

        connection.close()

    def run_test(
        self,
        queue_name: str,
        message_count: int,
        producer_count: int,
        consumer_count: int
    ):
        """ì„±ëŠ¥ í…ŒìŠ¤íŠ¸ ì‹¤í–‰"""
        self.running = True
        self.sent_count = 0
        self.received_count = 0

        print(f" Performance Test")
        print(f" " + "="*50)
        print(f" Queue: {queue_name}")
        print(f" Messages: {message_count}")
        print(f" Producers: {producer_count}")
        print(f" Consumers: {consumer_count}")
        print()

        start_time = time.time()

        # Consumer ì‹œì‘
        consumer_threads = []
        for i in range(consumer_count):
            t = threading.Thread(
                target=self.consumer_thread,
                args=(queue_name,)
            )
            t.start()
            consumer_threads.append(t)

        # Producer ì‹œì‘
        messages_per_producer = message_count // producer_count

        producer_threads = []
        for i in range(producer_count):
            t = threading.Thread(
                target=self.producer_thread,
                args=(queue_name, messages_per_producer)
            )
            t.start()
            producer_threads.append(t)

        # Producer ì™„ë£Œ ëŒ€ê¸°
        for t in producer_threads:
            t.join()

        print(f" [âœ“] All messages sent: {self.sent_count}")

        # Consumer ì™„ë£Œ ëŒ€ê¸°
        while self.received_count < self.sent_count:
            time.sleep(0.1)

        self.running = False

        for t in consumer_threads:
            t.join()

        end_time = time.time()
        duration = end_time - start_time

        # ê²°ê³¼ ì¶œë ¥
        print(f"\n Results:")
        print(f" " + "="*50)
        print(f" Duration: {duration:.2f}s")
        print(f" Throughput: {self.sent_count / duration:.2f} msg/s")
        print(f" Sent: {self.sent_count}")
        print(f" Received: {self.received_count}")

# ì‚¬ìš© ì˜ˆ
if __name__ == '__main__':
    tester = PerformanceTester()
    tester.run_test(
        queue_name='perf_test',
        message_count=10000,
        producer_count=5,
        consumer_count=10
    )
