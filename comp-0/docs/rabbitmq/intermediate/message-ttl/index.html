Message TTL - 메시지 만료

메시지의 유효 기간을 설정하여 자동으로 만료시키는 방법을 배워봅시다!

🎯 학습 목표​

  • Message TTL 이해
  • Queue TTL 설정
  • 메시지 만료 처리
  • 캐시 패턴 구현

📊 패턴 구조​

Producer → [Queue (TTL=10s)] → Consumer
                ↓
           10초 후 만료
                ↓
         [Dead Letter Exchange]

TTL 설정 방법:

  1. Queue TTL: 큐 전체에 적용
  2. Message TTL: 메시지별 개별 설정
  3. 두 가지 조합: 더 짧은 시간 적용

📝 시나리오​

실시간 할인 쿠폰 시스템:

  • 10초 동안만 유효한 특별 할인
  • 사용하지 않으면 자동 만료
  • 만료된 쿠폰은 통계 수집

# models.py
from dataclasses import dataclass, asdict
from datetime import datetime, timedelta
from typing import Optional
import json

@dataclass
class Coupon:
    """할인 쿠폰"""
    coupon_id: str
    user_id: int
    discount_percent: int
    expires_in_seconds: int
    created_at: str = None

    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now().isoformat()

    def to_json(self) -> str:
        return json.dumps(asdict(self))

    @classmethod
    def from_json(cls, json_str: str):
        return cls(**json.loads(json_str))

    def get_expiry_time(self) -> datetime:
        """만료 시간 계산"""
        created = datetime.fromisoformat(self.created_at)
        return created + timedelta(seconds=self.expires_in_seconds)

# queue_setup.py
import pika

def setup_queues_with_ttl():
    """TTL이 설정된 큐들 생성"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 1. Dead Letter Exchange 설정
    channel.exchange_declare(
        exchange='expired_coupons',
        exchange_type='direct',
        durable=True
    )

    # 2. 만료 쿠폰 수집 큐
    channel.queue_declare(
        queue='expired_coupon_stats',
        durable=True
    )

    channel.queue_bind(
        exchange='expired_coupons',
        queue='expired_coupon_stats',
        routing_key='expired'
    )

    # 3. 활성 쿠폰 큐 (Queue TTL = 10초)
    channel.queue_declare(
        queue='active_coupons',
        durable=True,
        arguments={
            # 🔑 Queue TTL: 모든 메시지가 10초 후 만료
            'x-message-ttl': 10000,  # 10초 (밀리초)

            # 만료 시 DLX로 이동
            'x-dead-letter-exchange': 'expired_coupons',
            'x-dead-letter-routing-key': 'expired'
        }
    )

    # 4. 빠른 만료 큐 (Queue TTL = 5초)
    channel.queue_declare(
        queue='flash_coupons',
        durable=True,
        arguments={
            'x-message-ttl': 5000,  # 5초
            'x-dead-letter-exchange': 'expired_coupons',
            'x-dead-letter-routing-key': 'expired'
        }
    )

    # 5. TTL 없는 큐 (메시지별 TTL 사용)
    channel.queue_declare(
        queue='custom_ttl_coupons',
        durable=True,
        arguments={
            'x-dead-letter-exchange': 'expired_coupons',
            'x-dead-letter-routing-key': 'expired'
        }
    )

    print(" [✓] Queues with TTL configured:")
    print("     - active_coupons (10s)")
    print("     - flash_coupons (5s)")
    print("     - custom_ttl_coupons (per-message TTL)")

    connection.close()

if __name__ == '__main__':
    setup_queues_with_ttl()

Producer (쿠폰 발행)​

# coupon_producer.py
import pika
from models import Coupon

class CouponProducer:
    """쿠폰 발행자"""

    def __init__(self, host='localhost'):
        self.host = host

    def issue_coupon(
        self,
        coupon: Coupon,
        queue: str = 'active_coupons',
        custom_ttl: int = None
    ):
        """
        쿠폰 발행

        Args:
            coupon: 쿠폰 정보
            queue: 대상 큐
            custom_ttl: 개별 TTL (밀리초, 선택사항)
        """
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        properties = pika.BasicProperties(
            delivery_mode=2,
            content_type='application/json'
        )

        # 🔑 Message TTL 설정 (선택사항)
        if custom_ttl:
            properties.expiration = str(custom_ttl)

        channel.basic_publish(
            exchange='',
            routing_key=queue,
            body=coupon.to_json(),
            properties=properties
        )

        ttl_info = f"TTL={custom_ttl}ms" if custom_ttl else "queue TTL"
        print(f" [x] Issued coupon {coupon.coupon_id} ({ttl_info})")

        connection.close()

# 사용 예
if __name__ == '__main__':
    producer = CouponProducer()

        'COUP-002',
        user_id=1002,
        discount_percent=50,
        expires_in_seconds=5
    )
    producer.issue_coupon(coupon2, queue='flash_coupons')

    # 3. 개별 TTL (3초)
    coupon3 = Coupon(
        'COUP-003',

        discount_percent=30,
        expires_in_seconds=3

# coupon_consumer.py
import pika
import time
from datetime import datetime
from models import Coupon

def use_coupon(coupon: Coupon) -> bool:
    """
    쿠폰 사용

    Returns:
        True: 성공
        False: 실패
    """
    now = datetime.now()
    expiry = coupon.get_expiry_time()

    print(f"""
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    🎟️  Coupon Usage
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    Coupon ID: {coupon.coupon_id}
    User ID: {coupon.user_id}
    Discount: {coupon.discount_percent}%
    Created: {coupon.created_at}
    Expires: {expiry.isoformat()}
    Status: {'✓ Valid' if now < expiry else '✗ Expired'}
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)

    # 쿠폰 적용 (시뮬레이션)
    time.sleep(1)

    return True

def callback(ch, method, properties, body):
    """메시지 처리 콜백"""
    coupon = Coupon.from_json(body.decode())

    use_coupon(coupon)

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start(queue_name: str = 'active_coupons'):
    """Coupon Consumer 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.basic_qos(prefetch_count=1)

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=callback
    )

    print(f' [*] Coupon Consumer waiting on [{queue_name}]...')
    print(' [*] Coupons will expire if not consumed in time')
    channel.start_consuming()

if __name__ == '__main__':
    import sys
    queue = sys.argv[1] if len(sys.argv) > 1 else 'active_coupons'

    try:
        start(queue)
    except KeyboardInterrupt:
        print('\nCoupon Consumer stopped')

Expired Coupon Handler​

# expired_handler.py
import pika
import json
from datetime import datetime
from models import Coupon

def handle_expired_coupon(ch, method, properties, body):
    """만료된 쿠폰 처리"""
    coupon = Coupon.from_json(body.decode())

    print(f"""
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    ⏰ COUPON EXPIRED
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    Coupon ID: {coupon.coupon_id}
    User ID: {coupon.user_id}
    Discount: {coupon.discount_percent}%
    Created: {coupon.created_at}
    Expired: {datetime.now().isoformat()}

    Reason: Not used within {coupon.expires_in_seconds}s
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)

    # 통계 저장
    save_expiry_stats(coupon)

    ch.basic_ack(delivery_tag=method.delivery_tag)

def save_expiry_stats(coupon: Coupon):
    """만료 통계 저장"""
    with open('expired_coupons.jsonl', 'a', encoding='utf-8') as f:
        stats = {
            'coupon_id': coupon.coupon_id,
            'user_id': coupon.user_id,
            'discount': coupon.discount_percent,
            'expired_at': datetime.now().isoformat(),
            'ttl': coupon.expires_in_seconds
        }
        f.write(json.dumps(stats) + '\n')

def start():
    """Expired Coupon Handler 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.basic_consume(
        queue='expired_coupon_stats',
        on_message_callback=handle_expired_coupon
    )

    print(' [*] Expired Coupon Handler waiting...')
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nExpired Handler stopped')

# test_message_ttl.py
import pytest
import pika
import time

class TestMessageTTL:
    """Message TTL 테스트"""

    @pytest.fixture
    def setup(self):
        """테스트 환경 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        # 기존 큐 삭제
        for queue in ['test.ttl.queue', 'test.ttl.dlq']:
            try:
                channel.queue_delete(queue=queue)
            except:
                pass

        try:
            channel.exchange_delete(exchange='test.dlx')
        except:
            pass

        yield channel
        connection.close()

    def test_queue_ttl(self, setup):
        """Queue TTL 테스트"""
        channel = setup

        # DLX 설정
        channel.exchange_declare(exchange='test.dlx', exchange_type='direct')
        channel.queue_declare(queue='test.ttl.dlq')
        channel.queue_bind(
            exchange='test.dlx',
            queue='test.ttl.dlq',
            routing_key='expired'
        )

        # Queue TTL = 1초
        channel.queue_declare(
            queue='test.ttl.queue',
            arguments={
                'x-message-ttl': 1000,  # 1초
                'x-dead-letter-exchange': 'test.dlx',
                'x-dead-letter-routing-key': 'expired'
            }
        )

        # 메시지 발행
        channel.basic_publish(
            exchange='',
            routing_key='test.ttl.queue',
            body='Will expire'
        )

        # 즉시는 메시지 있음
        method, properties, body = channel.basic_get(
            queue='test.ttl.queue',
            auto_ack=True
        )
        assert body == b'Will expire'

        # 다시 발행
        channel.basic_publish(
            exchange='',
            routing_key='test.ttl.queue',
            body='Will expire'
        )

        # 1초 대기
        time.sleep(1.1)

        # Main 큐는 비어있음
        assert channel.basic_get(queue='test.ttl.queue', auto_ack=True)[0] is None

        # DLQ에 있음
        method, properties, body = channel.basic_get(
            queue='test.ttl.dlq',
            auto_ack=True
        )
        assert body == b'Will expire'

    def test_message_ttl(self, setup):
        """Message TTL 테스트"""
        channel = setup

        channel.exchange_declare(exchange='test.dlx', exchange_type='direct')
        channel.queue_declare(queue='test.ttl.dlq')
        channel.queue_bind(
            exchange='test.dlx',
            queue='test.ttl.dlq',
            routing_key='expired'
        )

        # Queue TTL 없음
        channel.queue_declare(
            queue='test.ttl.queue',
            arguments={
                'x-dead-letter-exchange': 'test.dlx',
                'x-dead-letter-routing-key': 'expired'
            }
        )

        # 메시지별 TTL = 0.5초
        channel.basic_publish(
            exchange='',
            routing_key='test.ttl.queue',
            body='Quick expire',
            properties=pika.BasicProperties(
                expiration='500'  # 500ms
            )
        )

        time.sleep(0.6)

        # Main 큐는 비어있음
        assert channel.basic_get(queue='test.ttl.queue', auto_ack=True)[0] is None

        # DLQ에 있음
        method, properties, body = channel.basic_get(
            queue='test.ttl.dlq',
            auto_ack=True
        )
        assert body == b'Quick expire'

    def test_mixed_ttl(self, setup):
        """Queue TTL과 Message TTL 혼합"""
        channel = setup

        # Queue TTL = 2초
        channel.queue_declare(
            queue='test.ttl.queue',
            arguments={'x-message-ttl': 2000}
        )

        # Message TTL = 1초 (더 짧음)
        channel.basic_publish(
            exchange='',
            routing_key='test.ttl.queue',
            body='Shorter TTL wins',
            properties=pika.BasicProperties(expiration='1000')
        )

        # 1초 대기
        time.sleep(1.1)

        # 메시지 만료됨 (Message TTL이 더 짧아서)
        assert channel.basic_get(queue='test.ttl.queue', auto_ack=True)[0] is None

# cache_system.py
import pika
import json

class MessageCache:
    """메시지 기반 캐시 (TTL 사용)"""

    def __init__(self):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        self.channel = self.connection.channel()

        # 캐시 큐 (5분 TTL)
        self.channel.queue_declare(
            queue='cache',
            arguments={'x-message-ttl': 300000}  # 5분
        )

    def set(self, key: str, value: str, ttl_ms: int = None):
        """캐시 설정"""
        cache_entry = {'key': key, 'value': value}

        properties = pika.BasicProperties()
        if ttl_ms:
            properties.expiration = str(ttl_ms)

        self.channel.basic_publish(
            exchange='',
            routing_key='cache',
            body=json.dumps(cache_entry),
            properties=properties
        )

# 사용 예
cache = MessageCache()
cache.set('user:1001', '{"name": "Alice"}', ttl_ms=60000)  # 1분

def schedule_temp_task(task: str, ttl_seconds: int):

2. TTL과 성능​

# Message TTL (개별): 성능 저하
# - 메시지마다 만료 시간 체크
# - 큐가 느려질 수 있음

# Queue TTL (전체): 빠름
# - 한번만 체크
# - 권장 방식

3. Per-Message TTL 순서​

# Per-Message TTL은 큐 앞에서만 체크됨!

# 메시지 발행 순서:
# msg1 (TTL=10s)
# msg2 (TTL=1s)  ← 이것이 먼저 만료되어야 하지만
# msg3 (TTL=5s)

# → msg1이 처리/만료되기 전까지 msg2는 만료 안됨!
# → 순서 문제 발생 가능

# 해결: Queue TTL 사용 권장

📊 TTL 패턴 비교​

패턴사용 사례장점단점
Queue TTL모든 메시지 동일 TTL빠름, 예측 가능유연성 낮음
Message TTL메시지별 다른 TTL유연함느림, 순서 문제
TTL + DLX만료 메시지 처리통계/모니터링복잡함

✅ 실습 과제​

다음 시스템을 구현하세요:

알림 시스템

  • 긴급 알림: 10초 TTL
  • 일반 알림: 1분 TTL
  • 광고 알림: 10분 TTL
  • 만료된 알림은 통계 수집
정답 보기
# notification_system.py
import pika
import json

def send_notification(message: str, urgency: str):
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # TTL 매핑
    ttl_map = {
        'urgent': 10000,    # 10초
        'normal': 60000,    # 1분
        'ad': 600000        # 10분
    }

    # 알림 큐 (DLX 설정)
    channel.queue_declare(
        queue='notifications',
        arguments={
            'x-dead-letter-exchange': 'expired_notifications'
        }
    )

    ttl = ttl_map.get(urgency, 60000)

    channel.basic_publish(
        exchange='',
        routing_key='notifications',
        body=json.dumps({'message': message, 'urgency': urgency}),
        properties=pika.BasicProperties(
            expiration=str(ttl)
        )
    )

    print(f" [x] Notification sent ({urgency}, TTL={ttl}ms)")
    connection.close()

# 테스트
send_notification('Server down!', 'urgent')
send_notification('New message', 'normal')
send_notification('Sale today!', 'ad')

📚 핵심 정리​

  • Queue TTL: 큐 전체에 TTL 적용 (x-message-ttl)
  • Message TTL: 메시지별 개별 TTL (expiration)
  • 만료 처리: DLX와 함께 사용하여 만료 메시지 처리
  • 성능: Queue TTL이 Message TTL보다 빠름
