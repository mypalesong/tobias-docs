Priority Queue - 우선순위 큐

메시지에 우선순위를 부여하여 중요한 작업을 먼저 처리하는 방법을 배워봅시다!

🎯 학습 목표​

  • Priority Queue 이해
  • 우선순위 설정 방법
  • VIP 요청 처리
  • 우선순위 기반 스케줄링

📊 패턴 구조​

Producer (priority=10) ──┐
Producer (priority=5)  ──┼→ [Priority Queue] → Consumer
Producer (priority=1)  ──┘        ↓
                                높은 우선순위부터 처리

특징:

  • 0-255 범위의 우선순위 (높을수록 먼저 처리)
  • 같은 우선순위는 FIFO
  • 긴급 작업 우선 처리
  • VIP 고객, 중요 알림에 유용

📝 시나리오​

고객 지원 티켓 시스템:

  • VIP 고객 (priority=10): 즉시 처리
  • 긴급 (priority=7): 높은 우선순위
  • 일반 (priority=5): 보통 우선순위
  • 문의 (priority=1): 낮은 우선순위

모델 정의​

# models.py
from dataclasses import dataclass, asdict
from typing import Literal
import json

TicketPriority = Literal['vip', 'urgent', 'normal', 'inquiry']

@dataclass
class SupportTicket:
    """고객 지원 티켓"""
    ticket_id: str
    customer_id: int
    customer_name: str
    issue: str
    category: TicketPriority

    def to_json(self) -> str:
        return json.dumps(asdict(self))

    @classmethod
    def from_json(cls, json_str: str):
        return cls(**json.loads(json_str))

    def get_priority(self) -> int:
        """카테고리별 우선순위 반환"""
        priority_map = {
            'vip': 10,
            'urgent': 7,
            'normal': 5,
            'inquiry': 1
        }
        return priority_map[self.category]

# queue_setup.py
import pika

def setup_priority_queue():
    """Priority Queue 설정"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 🔑 핵심: x-max-priority 설정
    channel.queue_declare(
        queue='support_tickets',
        durable=True,
        arguments={
            'x-max-priority': 10  # 최대 우선순위: 10
        }
    )

    print(" [✓] Priority queue configured")
    print(" [✓] Max priority: 10")
    connection.close()

if __name__ == '__main__':
    setup_priority_queue()

Producer (티켓 발행)​

# ticket_producer.py
import pika
from models import SupportTicket

class TicketProducer:
    """티켓 발행자"""

    def __init__(self, host='localhost'):
        self.host = host

    def submit_ticket(self, ticket: SupportTicket):
        """티켓 제출"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        priority = ticket.get_priority()

        # 🔑 핵심: properties에 priority 설정
        channel.basic_publish(
            exchange='',
            routing_key='support_tickets',
            body=ticket.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,
                priority=priority,  # 우선순위 설정
                content_type='application/json'
            )
        )

        print(f" [x] Submitted [{ticket.category.upper()}] "
              f"(priority={priority}): {ticket.ticket_id}")

        connection.close()

# 사용 예
if __name__ == '__main__':
    producer = TicketProducer()

    # 다양한 우선순위의 티켓
    tickets = [
        SupportTicket(
            'TKT-001', 1001, 'Alice',
            'General question about pricing', 'inquiry'
        ),
        SupportTicket(
            'TKT-002', 1002, 'Bob VIP',
            'Cannot access my account', 'vip'
        ),
        SupportTicket(
            'TKT-003', 1003, 'Charlie',
            'Website is down!', 'urgent'
        ),
        SupportTicket(
            'TKT-004', 1004, 'Diana',
            'Need help with settings', 'normal'
        ),
        SupportTicket(
            'TKT-005', 1005, 'Eve VIP',
            'Billing issue - duplicate charge', 'vip'
        ),
        SupportTicket(
            'TKT-006', 1006, 'Frank',
            'Feature request', 'inquiry'
        ),
    ]

    for ticket in tickets:
        producer.submit_ticket(ticket)

    print("\n [i] Tickets submitted in order: TKT-001 to TKT-006")
    print(" [i] But they will be processed by priority, not submission order!")

Consumer (티켓 처리)​

# ticket_consumer.py
import pika
import time
from models import SupportTicket

def process_ticket(ticket: SupportTicket):
    """티켓 처리 (시뮬레이션)"""
    priority_labels = {
        'vip': '🌟 VIP',
        'urgent': '🚨 URGENT',
        'normal': '📝 NORMAL',
        'inquiry': '💬 INQUIRY'
    }

    label = priority_labels.get(ticket.category, '📝')

    print(f"""
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    {label} - Processing Ticket
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    Ticket ID: {ticket.ticket_id}
    Customer: {ticket.customer_name} (ID: {ticket.customer_id})
    Issue: {ticket.issue}
    Priority: {ticket.get_priority()}
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)

    # 처리 시간 (우선순위가 높을수록 빠르게)
    process_time = {
        'vip': 1,
        'urgent': 2,
        'normal': 3,
        'inquiry': 4
    }

    time.sleep(process_time.get(ticket.category, 2))
    print(f" [✓] Ticket {ticket.ticket_id} resolved!\n")

def callback(ch, method, properties, body):
    """메시지 처리 콜백"""
    ticket = SupportTicket.from_json(body.decode())

    # 우선순위 출력
    priority = properties.priority if properties.priority else 0
    print(f" [→] Received ticket with priority: {priority}")

    process_ticket(ticket)

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start():
    """Ticket Consumer 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.basic_qos(prefetch_count=1)

    channel.basic_consume(
        queue='support_tickets',
        on_message_callback=callback
    )

    print(' [*] Support Agent waiting for tickets...')
    print(' [*] Higher priority tickets will be processed first')
    print()
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nSupport Agent stopped')

# test_priority_queue.py
import pytest
import pika
import time

class TestPriorityQueue:
    """Priority Queue 테스트"""

    @pytest.fixture
    def setup(self):
        """테스트 환경 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        try:
            channel.queue_delete(queue='test_priority')
        except:
            pass

        yield channel
        connection.close()

    def test_priority_ordering(self, setup):
        """우선순위 순서대로 처리되는지 테스트"""
        channel = setup

        # Priority Queue 선언
        channel.queue_declare(
            queue='test_priority',
            arguments={'x-max-priority': 10}
        )

        # 낮은 우선순위부터 발행
        messages = [
            (1, 'Low priority'),
            (5, 'Medium priority'),
            (10, 'High priority'),
        ]

        for priority, message in messages:
            channel.basic_publish(
                exchange='',
                routing_key='test_priority',
                body=message,
                properties=pika.BasicProperties(priority=priority)
            )

        # 잠시 대기 (메시지 정렬 시간)
        time.sleep(0.1)

        # 높은 우선순위부터 수신되어야 함
        expected_order = [
            'High priority',    # priority=10
            'Medium priority',  # priority=5
            'Low priority',     # priority=1
        ]

        for expected_message in expected_order:
            method, properties, body = channel.basic_get(
                queue='test_priority',
                auto_ack=True
            )
            assert body.decode() == expected_message

    def test_same_priority_fifo(self, setup):
        """같은 우선순위는 FIFO 순서"""
        channel = setup

        channel.queue_declare(
            queue='test_priority',
            arguments={'x-max-priority': 10}
        )

        # 동일 우선순위로 발행
        for i in range(3):
            channel.basic_publish(
                exchange='',
                routing_key='test_priority',
                body=f'Message {i+1}',
                properties=pika.BasicProperties(priority=5)
            )

        time.sleep(0.1)

        # 발행 순서대로 수신
        for i in range(3):
            method, properties, body = channel.basic_get(
                queue='test_priority',
                auto_ack=True
            )
            assert body.decode() == f'Message {i+1}'

    def test_priority_range(self, setup):
        """우선순위 범위 테스트"""
        channel = setup

        channel.queue_declare(
            queue='test_priority',
            arguments={'x-max-priority': 10}
        )

        # 다양한 우선순위 (0-10)
        priorities = [0, 1, 5, 8, 10]

        for priority in priorities:
            channel.basic_publish(
                exchange='',
                routing_key='test_priority',
                body=f'Priority {priority}',
                properties=pika.BasicProperties(priority=priority)
            )

        time.sleep(0.1)

        # 높은 우선순위부터 수신
        for priority in reversed(priorities):
            method, properties, body = channel.basic_get(
                queue='test_priority',
                auto_ack=True
            )
            assert body.decode() == f'Priority {priority}'

3. 우선순위 처리 규칙​

# 1. 높은 우선순위가 먼저 처리
priority=10  →  먼저 처리
priority=5   →  나중 처리
priority=1   →  가장 나중

# 2. 같은 우선순위는 FIFO
priority=5 (msg1)  →  먼저
priority=5 (msg2)  →  나중

# email_sender.py
import pika

def send_email(to: str, subject: str, priority: str = 'normal'):
    """이메일 발송"""
    priority_map = {
        'critical': 10,  # 비밀번호 재설정
        'high': 7,       # 주문 확인
        'normal': 5,     # 뉴스레터
        'low': 1         # 마케팅
    }

    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.basic_publish(
        exchange='',
        routing_key='email_queue',
        body=f'{to}|{subject}',
        properties=pika.BasicProperties(
            priority=priority_map[priority]
        )
    )

    connection.close()

# 사용 예
send_email('user@example.com', 'Password Reset', 'critical')
send_email('user@example.com', 'Newsletter', 'low')

2. 작업 큐 우선순위​

# task_scheduler.py
import pika

class TaskScheduler:
    """작업 스케줄러"""

    def __init__(self):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        self.channel = self.connection.channel()

    def schedule_task(self, task_type: str, data: dict):
        """작업 스케줄링"""
        # 작업 유형별 우선순위
        task_priorities = {
            'user_signup': 10,      # 신규 가입 즉시 처리
            'payment': 9,           # 결제 높은 우선순위
            'notification': 7,      # 알림 보통 우선순위
            'analytics': 3,         # 분석 낮은 우선순위
            'cleanup': 1            # 정리 작업 가장 낮음
        }

        priority = task_priorities.get(task_type, 5)

        self.channel.basic_publish(
            exchange='',
            routing_key='tasks',
            body=json.dumps({'type': task_type, 'data': data}),
            properties=pika.BasicProperties(priority=priority)
        )

# 사용 예
scheduler = TaskScheduler()
scheduler.schedule_task('payment', {'amount': 100})
scheduler.schedule_task('cleanup', {'old_logs': True})
# payment가 cleanup보다 먼저 처리됨

3. API 요청 제한 (Rate Limiting)​

# api_queue.py
import pika

class APIRequestQueue:
    """API 요청 큐 (우선순위 기반)"""

    def __init__(self):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        self.channel = self.connection.channel()

        self.channel.queue_declare(
            queue='api_requests',
            arguments={'x-max-priority': 10}
        )

    def queue_request(self, endpoint: str, user_tier: str):
        """API 요청 큐잉"""
        # 사용자 등급별 우선순위
        tier_priority = {
            'enterprise': 10,
            'pro': 7,
            'basic': 3,
            'free': 1
        }

        priority = tier_priority.get(user_tier, 1)

        self.channel.basic_publish(
            exchange='',
            routing_key='api_requests',
            body=endpoint,
            properties=pika.BasicProperties(priority=priority)
        )

        print(f" [x] Queued [{user_tier}] {endpoint} (priority={priority})")

# 사용 예
api_queue = APIRequestQueue()
api_queue.queue_request('/api/data', 'free')
api_queue.queue_request('/api/data', 'enterprise')
# Enterprise 요청이 먼저 처리됨

⚠️ 주의사항​

1. 성능 고려​

# Priority Queue는 일반 큐보다 느림
# - 우선순위 정렬 오버헤드
# - 메모리 사용량 증가

# 필요한 경우에만 사용
# ✓ VIP 처리, 긴급 작업
# ✗ 모든 메시지가 같은 우선순위

2. 우선순위 범위​

# x-max-priority 설정한 값까지만 유효
channel.queue_declare(
    queue='my_queue',
    arguments={'x-max-priority': 10}
)

# ✓ priority=5  (유효)
# ✓ priority=10 (유효)
# ⚠ priority=15 (10으로 처리됨)

3. 기아 상태 (Starvation) 방지​

# 낮은 우선순위가 계속 밀릴 수 있음
# → 우선순위 차이를 크게 두지 않기
# → 타임아웃 설정

# ✓ 좋은 설계
vip = 10
normal = 7
low = 5

# ✗ 나쁜 설계
vip = 255
normal = 128
low = 1  # 거의 처리 안됨

# order_system.py
import pika
import json

class OrderPriority:
    VIP = 10
    EXPRESS = 8
    STANDARD = 5
    BULK = 2

def submit_order(order_id: str, customer_type: str, priority: int):
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.queue_declare(
        queue='orders',
        arguments={'x-max-priority': 10}
    )

    order_data = {
        'order_id': order_id,
        'customer_type': customer_type
    }

    channel.basic_publish(
        exchange='',
        routing_key='orders',
        body=json.dumps(order_data),
        properties=pika.BasicProperties(
            priority=priority,
            delivery_mode=2
        )
    )

    print(f" [x] Order {order_id} queued (priority={priority})")
    connection.close()

# 테스트
submit_order('ORD-001', 'standard', OrderPriority.STANDARD)
submit_order('ORD-002', 'vip', OrderPriority.VIP)
submit_order('ORD-003', 'express', OrderPriority.EXPRESS)
# 처리 순서: ORD-002 → ORD-003 → ORD-001

📚 핵심 정리​

  • Priority Queue: 우선순위 기반 메시지 처리
  • x-max-priority: 큐 선언 시 최대 우선순위 설정
  • priority: 메시지 발행 시 우선순위 지정 (0-255)
  • 처리 순서: 높은 우선순위 먼저, 같은 우선순위는 FIFO
