import pika
import random
from models import PaymentRequest

MAX_RETRIES = 3

def process_payment(payment: PaymentRequest) -> bool:
    """
    ê²°ì œ ì²˜ë¦¬ (ì‹œë®¬ë ˆì´ì…˜)

        raise Exception("Temporary payment gateway error")

def callback(ch, method, properties, body):
    """ë©”ì‹œì§€ ì²˜ë¦¬ ì½œë°±"""
    payment = PaymentRequest.from_json(body.decode())

    try:
        # ê²°ì œ ì²˜ë¦¬

        ch.basic_ack(delivery_tag=method.delivery_tag)

        ch.basic_nack(
            delivery_tag=method.delivery_tag,
            requeue=False

                exchange='',
                routing_key='payment.retry',

                    delivery_mode=2,
                    content_type='application/json'
                )
            )

            # ìµœëŒ€ ì¬ì‹œë„ ì´ˆê³¼: DLQë¡œ ì „ì†¡
            print(f" [âœ—] Max retries exceeded: {payment.payment_id}")

        ch.basic_ack(delivery_tag=method.delivery_tag)
