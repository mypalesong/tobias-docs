Work Queue - 작업 분배

여러 워커에게 작업을 분배하는 패턴을 배워봅시다!

🎯 학습 목표​

  • Work Queue 패턴 이해
  • Round-robin 분배 방식
  • Message Acknowledgment
  • Fair Dispatch (공정한 분배)

📊 패턴 구조​

Producer  -->  [Task Queue]  -->  Worker 1
                               -->  Worker 2
                               -->  Worker 3

특징:

  • 1:N 통신
  • 부하 분산
  • 병렬 처리
  • 확장 가능

📝 시나리오​

이미지 리사이징 작업을 여러 워커가 처리:

  • 원본 이미지 업로드 → 큐에 작업 추가
  • 여러 워커가 병렬로 리사이징
  • 완료된 작업은 자동 확인

🔧 구현​

Producer (작업 생성)​

# task_producer.py
import pika
import sys
import json

def send_task(task_name: str, difficulty: int):
    """작업을 큐에 추가"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 큐 선언 (durable=True로 영속성 보장)
    channel.queue_declare(queue='task_queue', durable=True)

    # 작업 데이터
    task_data = {
        'name': task_name,
        'difficulty': difficulty,
        'dots': '.' * difficulty  # 시뮬레이션용
    }

    message = json.dumps(task_data)

    # 메시지 발행
    channel.basic_publish(
        exchange='',
        routing_key='task_queue',
        body=message,
        properties=pika.BasicProperties(
            delivery_mode=2,  # 메시지를 디스크에 저장
        )
    )

    print(f" [x] Sent task: {task_name} (difficulty: {difficulty})")
    connection.close()

if __name__ == '__main__':
    # 여러 작업 생성
    tasks = [
        ('resize_photo_1.jpg', 2),
        ('resize_photo_2.jpg', 5),
        ('resize_photo_3.jpg', 1),
        ('resize_photo_4.jpg', 8),
        ('resize_photo_5.jpg', 3),
    ]

    for task_name, difficulty in tasks:
        send_task(task_name, difficulty)

Consumer (워커)​

# task_worker.py
import pika
import time
import json
import sys

def callback(ch, method, properties, body):
    """작업 처리 콜백"""
    task_data = json.loads(body.decode())
    task_name = task_data['name']
    difficulty = task_data['difficulty']

    print(f" [x] Received task: {task_name}")

    # 작업 처리 시뮬레이션 (난이도만큼 시간 소요)
    time.sleep(difficulty)

    print(f" [✓] Done: {task_name}")

    # 작업 완료 확인 (ACK)
    ch.basic_ack(delivery_tag=method.delivery_tag)

def start_worker(worker_id: int):
    """워커 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.queue_declare(queue='task_queue', durable=True)

    # Fair dispatch: 한 번에 1개씩만 처리
    channel.basic_qos(prefetch_count=1)

    channel.basic_consume(
        queue='task_queue',
        on_message_callback=callback
    )

    print(f' [*] Worker {worker_id} waiting for tasks...')
    channel.start_consuming()

if __name__ == '__main__':
    worker_id = int(sys.argv[1]) if len(sys.argv) > 1 else 1
    try:
        start_worker(worker_id)
    except KeyboardInterrupt:
        print(f'\nWorker {worker_id} stopped')

🧪 테스트 코드​

# test_work_queue.py
import pytest
import pika
import json
import time
from threading import Thread
from task_producer import send_task

class TestWorkQueue:
    """Work Queue 테스트"""

    @pytest.fixture
    def setup_queue(self):
        """테스트 큐 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()
        channel.queue_delete(queue='test_task_queue')
        channel.queue_declare(queue='test_task_queue', durable=True)
        yield channel
        channel.queue_delete(queue='test_task_queue')
        connection.close()

    def test_multiple_tasks(self, setup_queue):
        """여러 작업 발행 테스트"""
        channel = setup_queue

        # 3개 작업 발행
        for i in range(3):
            task_data = {'name': f'task_{i}', 'difficulty': 1}
            channel.basic_publish(
                exchange='',
                routing_key='test_task_queue',
                body=json.dumps(task_data),
                properties=pika.BasicProperties(delivery_mode=2)
            )

        # 큐 상태 확인
        queue_state = channel.queue_declare(
            queue='test_task_queue',
            passive=True
        )
        assert queue_state.method.message_count == 3

    def test_task_processing(self, setup_queue):
        """작업 처리 테스트"""
        channel = setup_queue
        processed_tasks = []

        def callback(ch, method, properties, body):
            task_data = json.loads(body.decode())
            processed_tasks.append(task_data['name'])
            ch.basic_ack(delivery_tag=method.delivery_tag)

        # Consumer 설정
        channel.basic_consume(
            queue='test_task_queue',
            on_message_callback=callback
        )

        # 작업 발행
        task_data = {'name': 'test_task', 'difficulty': 1}
        channel.basic_publish(
            exchange='',
            routing_key='test_task_queue',
            body=json.dumps(task_data),
            properties=pika.BasicProperties(delivery_mode=2)
        )

        # 메시지 처리
        connection = channel.connection
        connection.process_data_events(time_limit=1)

        assert 'test_task' in processed_tasks

    def test_fair_dispatch(self, setup_queue):
        """공정한 분배 테스트"""
        channel = setup_queue

        # prefetch_count 설정
        channel.basic_qos(prefetch_count=1)

        # 여러 작업 발행
        for i in range(5):
            task_data = {'name': f'task_{i}', 'difficulty': i+1}
            channel.basic_publish(
                exchange='',
                routing_key='test_task_queue',
                body=json.dumps(task_data),
                properties=pika.BasicProperties(delivery_mode=2)
            )

        # 큐에 5개 메시지 확인
        queue_state = channel.queue_declare(
            queue='test_task_queue',
            passive=True
        )
        assert queue_state.method.message_count == 5

💡 핵심 개념​

1. Message Acknowledgment​

# auto_ack=False (기본값): 수동 확인
channel.basic_consume(
    queue='task_queue',
    on_message_callback=callback,
    auto_ack=False  # 수동 ACK
)

# 작업 완료 후 ACK
ch.basic_ack(delivery_tag=method.delivery_tag)

왜 수동 ACK?

  • 워커가 죽으면 메시지 재전송
  • 처리 완료를 확실히 보장

2. Message Durability​

# 큐 영속성
channel.queue_declare(queue='task_queue', durable=True)

# 메시지 영속성
channel.basic_publish(
    exchange='',
    routing_key='task_queue',
    body=message,
    properties=pika.BasicProperties(
        delivery_mode=2,  # persistent
    )
)

3. Fair Dispatch​

# prefetch_count=1: 한 번에 1개씩만
channel.basic_qos(prefetch_count=1)

효과:

  • 빠른 워커는 더 많은 작업 처리
  • 느린 워커는 적은 작업 처리
  • 전체 처리량 최적화

🔍 비교​

Round-robin (기본)​

# prefetch_count 설정 없음
# → 모든 워커에게 균등 분배
# → 빠른 워커가 대기하게 됨

Worker 1: [Task 1] [Task 4] [Task 7]  (fast)
Worker 2: [Task 2] [Task 5] [Task 8]  (slow)
Worker 3: [Task 3] [Task 6] [Task 9]  (medium)

Fair Dispatch​

channel.basic_qos(prefetch_count=1)
# → 처리 속도에 따라 분배
# → 효율적인 부하 분산

Worker 1: [Task 1] [Task 4] [Task 6] [Task 8] (fast)
Worker 2: [Task 2] [Task 7]                   (slow)
Worker 3: [Task 3] [Task 5] [Task 9]          (medium)

⚠️ 주의사항​

1. ACK를 잊지 말 것​

# ❌ 위험: ACK 없음 → 메모리 누수
def callback(ch, method, properties, body):
    process_task(body)
    # ch.basic_ack() 누락!

# ✅ 올바름
def callback(ch, method, properties, body):
    process_task(body)
    ch.basic_ack(delivery_tag=method.delivery_tag)

2. 처리 실패 시 NACK​

def callback(ch, method, properties, body):
    try:
        process_task(body)
        ch.basic_ack(delivery_tag=method.delivery_tag)
    except Exception as e:
        print(f"Error: {e}")
        # 재시도를 위해 NACK
        ch.basic_nack(
            delivery_tag=method.delivery_tag,
            requeue=True
        )

3. 타임아웃 설정​

# 워커가 오래 걸리는 작업 처리 시
connection = pika.BlockingConnection(
    pika.ConnectionParameters(
        host='localhost',
        heartbeat=600,  # 10분
        blocked_connection_timeout=300
    )
)

✅ 실습 과제​

다음 시스템을 구현하세요:

  1. 비디오 인코딩 작업 큐
    • 다양한 해상도 (480p, 720p, 1080p)
    • 해상도에 따라 처리 시간 다름
    • 3개 워커로 병렬 처리
정답 보기
# video_encoder.py
import pika
import time
import json

def encode_video(ch, method, properties, body):
    video_data = json.loads(body.decode())
    filename = video_data['filename']
    resolution = video_data['resolution']

    print(f" [x] Encoding {filename} to {resolution}")

    # 해상도별 처리 시간
    encoding_time = {
        '480p': 2,
        '720p': 5,
        '1080p': 10
    }

    time.sleep(encoding_time.get(resolution, 2))
    print(f" [✓] Done: {filename} ({resolution})")

    ch.basic_ack(delivery_tag=method.delivery_tag)

connection = pika.BlockingConnection(
    pika.ConnectionParameters('localhost')
)
channel = connection.channel()
channel.queue_declare(queue='video_encoding', durable=True)
channel.basic_qos(prefetch_count=1)

channel.basic_consume(
    queue='video_encoding',
    on_message_callback=encode_video
)

print(' [*] Waiting for encoding tasks...')
channel.start_consuming()

📚 핵심 정리​

  • Work Queue: 여러 워커에게 작업 분배
  • Message ACK: 처리 완료 확인
  • Durability: 큐와 메시지 영속화
  • Fair Dispatch: 공정한 부하 분산 (prefetch_count=1)
