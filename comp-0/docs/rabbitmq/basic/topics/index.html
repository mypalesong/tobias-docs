Topics - 패턴 매칭

Routing Key 패턴을 사용한 유연한 라우팅을 배워봅시다!

🎯 학습 목표​

  • Topic Exchange 이해
  • 와일드카드 패턴 활용
  • 복잡한 라우팅 구현
  • 실전 시나리오 적용

📊 패턴 구조​

Producer → [Topic Exchange]
   (kern.critical)     ↓
                   ┌→ Queue (*.critical)    → Critical Logger
                   ├→ Queue (kern.*)        → Kernel Logger
                   └→ Queue (#)             → All Logger

패턴 매칭:
* : 정확히 1개 단어
# : 0개 이상의 단어

특징:

  • 와일드카드 패턴 매칭
  • 유연한 라우팅
  • 계층적 메시지 분류
  • 복잡한 구독 시나리오에 유용

📝 시나리오​

분산 시스템 로그 수집:

  • 로그 형식: <facility>.<severity>.<source>
    • kern.critical.cpu: 커널 심각한 CPU 에러
    • user.warning.disk: 유저 경고 디스크
    • kern.info.network: 커널 정보 네트워크
  • Consumer별 구독 패턴:
    • Critical Logger: *.critical.* (모든 Critical)
    • Kernel Logger: kern.*.* (모든 Kernel)
    • Disk Monitor: *.*.disk (모든 Disk)

🔧 구현​

Publisher (로그 발행)​

# topic_publisher.py
import pika
import sys
import json
from datetime import datetime
from typing import Literal

Facility = Literal['kern', 'user', 'mail', 'daemon']
Severity = Literal['critical', 'error', 'warning', 'info', 'debug']
Source = Literal['cpu', 'disk', 'network', 'memory']

class TopicLogPublisher:
    """Topic Exchange 로그 발행자"""

    def __init__(self, host='localhost'):
        self.host = host
        self.exchange = 'topic_logs'

    def publish_log(
        self,
        facility: Facility,
        severity: Severity,
        source: Source,
        message: str
    ):
        """
        로그 발행

        Args:
            facility: 시스템 구분 (kern, user, mail, daemon)
            severity: 심각도 (critical, error, warning, info, debug)
            source: 출처 (cpu, disk, network, memory)
            message: 로그 메시지
        """
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # Topic Exchange 선언
        channel.exchange_declare(
            exchange=self.exchange,
            exchange_type='topic'
        )

        # Routing Key: facility.severity.source
        routing_key = f"{facility}.{severity}.{source}"

        log_data = {
            'facility': facility,
            'severity': severity,
            'source': source,
            'message': message,
            'timestamp': datetime.now().isoformat()
        }

        channel.basic_publish(
            exchange=self.exchange,
            routing_key=routing_key,  # 🔑 점(.)으로 구분된 패턴
            body=json.dumps(log_data)
        )

        print(f" [x] Sent [{routing_key}]: {message}")
        connection.close()

# 사용 예
if __name__ == '__main__':
    publisher = TopicLogPublisher()

    # 다양한 로그 발행
    publisher.publish_log('kern', 'critical', 'cpu', 'CPU overheating detected!')
    publisher.publish_log('user', 'warning', 'disk', 'Disk usage at 85%')
    publisher.publish_log('kern', 'info', 'network', 'Network interface initialized')
    publisher.publish_log('mail', 'error', 'disk', 'Mail queue full')
    publisher.publish_log('daemon', 'critical', 'memory', 'Out of memory!')

# critical_logger.py
import pika
import json

def handle_critical(ch, method, properties, body):
    """모든 Critical 로그 처리"""
    log_data = json.loads(body.decode())

    print(f"""
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    🚨 CRITICAL ALERT
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    Routing Key: {method.routing_key}
    Facility: {log_data['facility']}
    Source: {log_data['source']}
    Time: {log_data['timestamp']}
    Message: {log_data['message']}

    Action: Immediate intervention required!
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)

    # PagerDuty, Slack 등으로 긴급 알림
    # send_pagerduty_alert(log_data)

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start():
    """Critical Logger 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(
        exchange='topic_logs',
        exchange_type='topic'
    )

    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # 🔑 패턴: *.critical.* (모든 critical 로그)
    binding_key = '*.critical.*'
    channel.queue_bind(
        exchange='topic_logs',
        queue=queue_name,
        routing_key=binding_key
    )

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=handle_critical
    )

    print(f' [*] Critical Logger waiting for [{binding_key}]...')
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nCritical Logger stopped')

Kernel Logger (kern.#)​

# kernel_logger.py
import pika
import json

def handle_kernel(ch, method, properties, body):
    """모든 Kernel 로그 처리"""
    log_data = json.loads(body.decode())

    severity_icons = {
        'critical': '🚨',
        'error': '❌',
        'warning': '⚠️',
        'info': 'ℹ️',
        'debug': '🐛'
    }

    icon = severity_icons.get(log_data['severity'], '📝')

    print(f"{icon} [KERNEL] [{method.routing_key}] "
          f"{log_data['message']}")

    # Kernel 로그를 별도 파일에 저장
    with open('kernel.log', 'a', encoding='utf-8') as f:
        f.write(f"[{log_data['timestamp']}] [{method.routing_key}] "
                f"{log_data['message']}\n")

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start():
    """Kernel Logger 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(
        exchange='topic_logs',
        exchange_type='topic'
    )

    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # 🔑 패턴: kern.# (모든 kern 로그)
    # kern.critical.cpu ✓
    # kern.info.network ✓
    # kern.warning.disk.raid ✓ (# = 여러 단어 매칭)
    binding_key = 'kern.#'
    channel.queue_bind(
        exchange='topic_logs',
        queue=queue_name,
        routing_key=binding_key
    )

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=handle_kernel
    )

    print(f' [*] Kernel Logger waiting for [{binding_key}]...')
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nKernel Logger stopped')

# test_topics.py
import pytest
import pika
import json

class TestTopics:
    """Topic Exchange 테스트"""

    @pytest.fixture
    def setup(self):
        """테스트 환경 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        try:
            channel.exchange_delete(exchange='test_topic_logs')
        except:
            pass

        yield channel
        connection.close()

    def test_wildcard_star(self, setup):
        """* (정확히 1개 단어) 테스트"""
        channel = setup

        channel.exchange_declare(
            exchange='test_topic_logs',
            exchange_type='topic'
        )

        queue = channel.queue_declare(queue='', exclusive=True).method.queue

        # 패턴: *.critical.*
        channel.queue_bind(
            exchange='test_topic_logs',
            queue=queue,
            routing_key='*.critical.*'
        )

        # 매칭되는 케이스
        matching_keys = [
            'kern.critical.cpu',
            'user.critical.disk',
            'mail.critical.network'
        ]

        for key in matching_keys:
            channel.basic_publish(
                exchange='test_topic_logs',
                routing_key=key,
                body=f'Message: {key}'
            )

        # 3개 메시지 모두 받아야 함
        for _ in range(3):
            method, properties, body = channel.basic_get(
                queue=queue,
                auto_ack=True
            )
            assert method is not None

        # 큐가 비어있어야 함
        assert channel.basic_get(queue=queue, auto_ack=True)[0] is None

    def test_wildcard_hash(self, setup):
        """# (0개 이상의 단어) 테스트"""
        channel = setup

        channel.exchange_declare(
            exchange='test_topic_logs',
            exchange_type='topic'
        )

        queue = channel.queue_declare(queue='', exclusive=True).method.queue

        # 패턴: kern.#
        channel.queue_bind(
            exchange='test_topic_logs',
            queue=queue,
            routing_key='kern.#'
        )

        # 매칭되는 케이스
        matching_keys = [
            'kern.critical',           # # = 1개 단어
            'kern.critical.cpu',       # # = 2개 단어
            'kern.info.network.eth0'   # # = 3개 단어
        ]

        for key in matching_keys:
            channel.basic_publish(
                exchange='test_topic_logs',
                routing_key=key,
                body=f'Message: {key}'
            )

        # 3개 메시지 모두 받아야 함
        for _ in range(3):
            method, properties, body = channel.basic_get(
                queue=queue,
                auto_ack=True
            )
            assert method is not None

        # 매칭 안되는 케이스
        channel.basic_publish(
            exchange='test_topic_logs',
            routing_key='user.critical.cpu',
            body='Should not match'
        )

        # 큐가 비어있어야 함
        assert channel.basic_get(queue=queue, auto_ack=True)[0] is None

    def test_multiple_topic_bindings(self, setup):
        """여러 패턴 바인딩 테스트"""
        channel = setup

        channel.exchange_declare(
            exchange='test_topic_logs',
            exchange_type='topic'
        )

        queue = channel.queue_declare(queue='', exclusive=True).method.queue

        # 여러 패턴 바인딩
        patterns = ['*.critical.*', 'kern.#', '*.*.disk']
        for pattern in patterns:
            channel.queue_bind(
                exchange='test_topic_logs',
                queue=queue,
                routing_key=pattern
            )

        # 테스트 케이스
        test_cases = {
            'kern.critical.cpu': True,    # 2개 패턴 매칭
            'user.critical.disk': True,   # 2개 패턴 매칭
            'kern.info.network': True,    # 1개 패턴 매칭
            'user.warning.disk': True,    # 1개 패턴 매칭
            'mail.info.network': False,   # 매칭 안됨
        }

        for routing_key, should_match in test_cases.items():
            channel.basic_publish(
                exchange='test_topic_logs',
                routing_key=routing_key,
                body=f'Test: {routing_key}'
            )

            method, properties, body = channel.basic_get(
                queue=queue,
                auto_ack=True
            )

            if should_match:
                assert method is not None, f"{routing_key} should match"
            else:
                assert method is None, f"{routing_key} should not match"

🔍 Exchange 타입 총정리​

ExchangeRouting사용 예
Fanoutrouting_key 무시, 모든 큐에 전송실시간 알림, 브로드캐스트
Directrouting_key 정확히 매칭로그 레벨별 처리
Topicrouting_key 패턴 매칭 (*, #)계층적 로그, 복잡한 라우팅
# Fanout
routing_key = ''  # 무시됨
binding_key = ''  # 필요없음

# Direct
routing_key = 'error'
binding_key = 'error'  # 정확히 일치해야 함

# Topic
routing_key = 'kern.critical.cpu'
binding_key = 'kern.#'  # 패턴 매칭
binding_key = '*.critical.*'  # 패턴 매칭

  • 계층적 구조: 점(.)으로 구분된 라우팅 키
