import pika
import json

def log_to_console(ch, method, properties, body):
    """ì½˜ì†”ì— ë¡œê·¸ ì¶œë ¥"""
    log_data = json.loads(body.decode())

    # ë¡œê·¸ ë ˆë²¨ì— ë”°ë¥¸ ìƒ‰ìƒ ì½”ë“œ
    colors = {
        'DEBUG': '\033[36m',    # Cyan
        'INFO': '\033[32m',     # Green
        'WARNING': '\033[33m',  # Yellow
        'ERROR': '\033[31m',    # Red
        'RESET': '\033[0m'
    }

    color = colors.get(log_data['level'], colors['RESET'])

    print(f"{color}[{log_data['timestamp']}] "
          f"{log_data['level']}: {log_data['message']}"
          f"{colors['RESET']}")

def start():
    """Console Logger ì‹œì‘"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # Exchange ì„ ì–¸
    channel.exchange_declare(exchange='logs', exchange_type='fanout')

    # ì„ì‹œ í ìƒì„± (Consumer ì¢…ë£Œ ì‹œ ìë™ ì‚­ì œ)
    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # Exchangeì— í ë°”ì¸ë”©
    channel.queue_bind(exchange='logs', queue=queue_name)

    print(' [*] Console Logger waiting for logs...')

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=log_to_console,
        auto_ack=True
    )

    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nConsole Logger stopped')
