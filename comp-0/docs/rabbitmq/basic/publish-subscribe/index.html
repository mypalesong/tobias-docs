Publish/Subscribe - 브로드캐스팅

하나의 메시지를 여러 Consumer에게 전달하는 패턴을 배워봅시다!

🎯 학습 목표​

  • Fanout Exchange 이해
  • 브로드캐스팅 패턴 구현
  • 여러 Consumer 동시 처리
  • 임시 큐 활용

📊 패턴 구조​

Producer  -->  [Fanout Exchange]  -->  Queue 1  -->  Consumer 1
                                   -->  Queue 2  -->  Consumer 2
                                   -->  Queue 3  -->  Consumer 3

특징:

  • 1:N 통신
  • 모든 Consumer가 동일 메시지 수신
  • Exchange가 큐에 복사본 전달
  • 로그 시스템, 실시간 알림에 유용

📝 시나리오​

실시간 로그 모니터링 시스템:

  • Application이 로그 발행
  • 여러 서비스가 동시에 로그 구독
    • Console Logger: 콘솔 출력
    • File Logger: 파일 저장
    • Alert Service: 에러 시 알림

🔧 구현​

Publisher (로그 발행)​

# log_publisher.py
import pika
import sys
import json
from datetime import datetime

class LogPublisher:
    """로그 발행자"""

    def __init__(self, host='localhost'):
        self.host = host
        self.exchange = 'logs'

    def publish_log(self, level: str, message: str):
        """로그 발행"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # Fanout Exchange 선언
        channel.exchange_declare(
            exchange=self.exchange,
            exchange_type='fanout'
        )

        log_data = {
            'level': level,
            'message': message,
            'timestamp': datetime.now().isoformat()
        }

        # Exchange로 발행 (routing_key 불필요)
        channel.basic_publish(
            exchange=self.exchange,
            routing_key='',  # fanout은 routing_key 무시
            body=json.dumps(log_data)
        )

        print(f" [x] Sent {level}: {message}")
        connection.close()

# 사용 예
if __name__ == '__main__':
    publisher = LogPublisher()

    # 다양한 로그 발행
    publisher.publish_log('INFO', 'Application started')
    publisher.publish_log('WARNING', 'High memory usage detected')
    publisher.publish_log('ERROR', 'Database connection failed')
    publisher.publish_log('DEBUG', 'Processing request ID: 12345')

Console Logger (구독자 1)​

# console_logger.py
import pika
import json

def log_to_console(ch, method, properties, body):
    """콘솔에 로그 출력"""
    log_data = json.loads(body.decode())

    # 로그 레벨에 따른 색상 코드
    colors = {
        'DEBUG': '\033[36m',    # Cyan
        'INFO': '\033[32m',     # Green
        'WARNING': '\033[33m',  # Yellow
        'ERROR': '\033[31m',    # Red
        'RESET': '\033[0m'
    }

    color = colors.get(log_data['level'], colors['RESET'])

    print(f"{color}[{log_data['timestamp']}] "
          f"{log_data['level']}: {log_data['message']}"
          f"{colors['RESET']}")

def start():
    """Console Logger 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # Exchange 선언
    channel.exchange_declare(exchange='logs', exchange_type='fanout')

    # 임시 큐 생성 (Consumer 종료 시 자동 삭제)
    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # Exchange에 큐 바인딩
    channel.queue_bind(exchange='logs', queue=queue_name)

    print(' [*] Console Logger waiting for logs...')

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=log_to_console,
        auto_ack=True
    )

    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nConsole Logger stopped')

File Logger (구독자 2)​

# file_logger.py
import pika
import json
from datetime import datetime

def log_to_file(ch, method, properties, body):
    """파일에 로그 저장"""
    log_data = json.loads(body.decode())

    # 날짜별 로그 파일
    date_str = datetime.now().strftime('%Y-%m-%d')
    filename = f'logs/app_{date_str}.log'

    with open(filename, 'a', encoding='utf-8') as f:
        f.write(
            f"[{log_data['timestamp']}] "
            f"{log_data['level']}: {log_data['message']}\n"
        )

    print(f" [✓] Saved to {filename}")

def start():
    """File Logger 시작"""
    # logs 디렉토리 생성
    import os
    os.makedirs('logs', exist_ok=True)

    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(exchange='logs', exchange_type='fanout')

    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    channel.queue_bind(exchange='logs', queue=queue_name)

    print(' [*] File Logger waiting for logs...')

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=log_to_file,
        auto_ack=True
    )

    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nFile Logger stopped')

# alert_service.py
import pika
import json

def send_alert(ch, method, properties, body):
    """에러 로그 시 알림 발송"""
    log_data = json.loads(body.decode())

    # ERROR 레벨만 처리
    if log_data['level'] == 'ERROR':
        print(f"""
        ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        🚨 ALERT: ERROR DETECTED!
        ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        Time: {log_data['timestamp']}
        Message: {log_data['message']}

        Action Required: Please investigate immediately
        ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        """)

def start():
    """Alert Service 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(exchange='logs', exchange_type='fanout')

    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    channel.queue_bind(exchange='logs', queue=queue_name)

    print(' [*] Alert Service waiting for errors...')

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=send_alert,
        auto_ack=True
    )

    channel.start_consuming()

if __name__ == '__main__':

        print('\nAlert Service stopped')

🧪 테스트 코드​

# test_publish_subscribe.py
import pytest
import pika
import json
import time
from threading import Thread

class TestPublishSubscribe:
    """Publish/Subscribe 테스트"""

    @pytest.fixture
    def setup(self):
        """테스트 환경 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        # Exchange 정리
        try:
            channel.exchange_delete(exchange='test_logs')
        except:
            pass

        yield channel
        connection.close()

    def test_fanout_exchange(self, setup):
        """Fanout Exchange 기본 테스트"""
        channel = setup

        # Exchange 선언
        channel.exchange_declare(
            exchange='test_logs',
            exchange_type='fanout'
        )

        # 3개 큐 생성 및 바인딩
        queues = []
        for i in range(3):
            result = channel.queue_declare(queue='', exclusive=True)
            queue_name = result.method.queue
            channel.queue_bind(
                exchange='test_logs',
                queue=queue_name
            )
            queues.append(queue_name)

        # 메시지 발행
        test_message = json.dumps({'level': 'INFO', 'message': 'Test'})
        channel.basic_publish(
            exchange='test_logs',
            routing_key='',
            body=test_message
        )

        # 모든 큐에 메시지가 있는지 확인
        for queue_name in queues:
            method, properties, body = channel.basic_get(
                queue=queue_name,
                auto_ack=True
            )
            assert body.decode() == test_message

    def test_multiple_subscribers(self, setup):
        """여러 구독자 동시 수신 테스트"""
        channel = setup
        received_messages = {'logger1': [], 'logger2': [], 'logger3': []}

        def create_callback(logger_name):
            def callback(ch, method, properties, body):
                received_messages[logger_name].append(body.decode())
                ch.basic_ack(delivery_tag=method.delivery_tag)
            return callback

        # Exchange 생성
        channel.exchange_declare(
            exchange='test_logs',
            exchange_type='fanout'
        )

        # 3개 Logger 구독
        for logger in ['logger1', 'logger2', 'logger3']:
            result = channel.queue_declare(queue='', exclusive=True)
            queue_name = result.method.queue
            channel.queue_bind(
                exchange='test_logs',
                queue=queue_name
            )
            channel.basic_consume(
                queue=queue_name,
                on_message_callback=create_callback(logger)
            )

        # 로그 발행
        log_message = json.dumps({
            'level': 'ERROR',
            'message': 'Test error'
        })

        channel.basic_publish(
            exchange='test_logs',
            routing_key='',
            body=log_message
        )

        # 메시지 처리
        connection = channel.connection
        connection.process_data_events(time_limit=1)

        # 모든 Logger가 메시지를 받았는지 확인
        assert len(received_messages['logger1']) == 1
        assert len(received_messages['logger2']) == 1
        assert len(received_messages['logger3']) == 1

        # 메시지 내용 확인
        for logger in received_messages.values():
            assert log_message in logger

💡 핵심 개념​

1. Fanout Exchange​

# Exchange 선언 (Publisher)
channel.exchange_declare(
    exchange='logs',
    exchange_type='fanout'  # 모든 큐에 브로드캐스트
)

# 발행 시 routing_key 무시됨
channel.basic_publish(
    exchange='logs',
    routing_key='',  # fanout은 이 값을 무시
    body=message
)

2. 임시 큐 (Exclusive Queue)​

# 임시 큐 생성
result = channel.queue_declare(queue='', exclusive=True)
queue_name = result.method.queue  # 랜덤 이름 (예: amq.gen-JzTY20...)

# 특징:
# - queue='' : 랜덤 이름 자동 생성
# - exclusive=True : Consumer 종료 시 자동 삭제
# - 임시 데이터에 적합 (로그, 실시간 알림 등)

3. 큐 바인딩​

# Exchange와 Queue 연결
channel.queue_bind(
    exchange='logs',
    queue=queue_name
)

🔍 Work Queue vs Publish/Subscribe​

Work Queue (1개 메시지 → 1개 Worker)​

# 메시지가 Worker들 사이에 분배됨
Producer  -->  [Queue]  -->  Worker 1 (처리)
                        -->  Worker 2 (대기)
                        -->  Worker 3 (대기)

Publish/Subscribe (1개 메시지 → 모든 Subscriber)​

# 메시지가 모든 Subscriber에게 복사됨
Producer  -->  [Exchange]  -->  Subscriber 1 (처리)
                           -->  Subscriber 2 (처리)
                           -->  Subscriber 3 (처리)

⚠️ 주의사항​

1. Exchange를 먼저 선언하기​

# ✅ 올바름: Publisher도 Exchange 선언
channel.exchange_declare(exchange='logs', exchange_type='fanout')
channel.basic_publish(exchange='logs', routing_key='', body=message)

# ❌ 위험: Exchange 없이 발행 → 메시지 손실
channel.basic_publish(exchange='logs', routing_key='', body=message)

2. Consumer 먼저 시작​

# Fanout은 실시간 브로드캐스트이므로
# Consumer가 먼저 실행되어야 메시지 수신 가능

# ✅ 순서:
# 1. Consumer 실행 (큐 생성 및 바인딩)
# 2. Publisher 실행 (메시지 발행)

3. 큐 자동 삭제 주의​

# exclusive=True 사용 시
# Consumer 종료하면 큐가 삭제됨
# → 영구 저장이 필요하면 exclusive=False 사용

# 영구 큐
channel.queue_declare(
    queue='permanent_logs',
    durable=True,
    exclusive=False
)

📚 핵심 정리​

  • Fanout Exchange: 모든 바인딩된 큐에 메시지 복사
