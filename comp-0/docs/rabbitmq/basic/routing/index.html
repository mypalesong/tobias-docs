Routing - 선택적 라우팅

Routing Key를 사용해 메시지를 선택적으로 전달하는 패턴을 배워봅시다!

🎯 학습 목표​

  • Direct Exchange 이해
  • Routing Key 활용
  • 선택적 구독 구현
  • 여러 바인딩 처리

📊 패턴 구조​

                       ┌→ Queue (error)    → Consumer 1
Producer → [Direct     ├→ Queue (warning)  → Consumer 2
            Exchange]  └→ Queue (info)     → Consumer 3
   (routing_key)

특징:

  • Routing Key 기반 라우팅
  • 정확한 매칭 (Exact match)
  • 선택적 메시지 수신
  • 로그 레벨별 처리에 유용

📝 시나리오​

로그 레벨별 처리 시스템:

  • Application이 로그 레벨과 함께 발행
  • Consumer는 관심있는 레벨만 구독
    • Error Handler: ERROR만 처리
    • Warning Handler: WARNING만 처리
    • All Logger: 모든 레벨 처리

🔧 구현​

Publisher (로그 발행)​

# routing_publisher.py
import pika
import sys
import json
from datetime import datetime

class RoutingLogPublisher:
    """Routing Key를 사용한 로그 발행자"""

    def __init__(self, host='localhost'):
        self.host = host
        self.exchange = 'direct_logs'

    def publish_log(self, severity: str, message: str):
        """
        로그 발행

        Args:
            severity: 로그 레벨 (info, warning, error)
            message: 로그 메시지
        """
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # Direct Exchange 선언
        channel.exchange_declare(
            exchange=self.exchange,
            exchange_type='direct'
        )

        log_data = {
            'severity': severity,
            'message': message,
            'timestamp': datetime.now().isoformat()
        }

        # Routing key = severity (info, warning, error)
        channel.basic_publish(
            exchange=self.exchange,
            routing_key=severity,  # 🔑 핵심: routing_key
            body=json.dumps(log_data)
        )

        print(f" [x] Sent [{severity}]: {message}")
        connection.close()

# 사용 예
if __name__ == '__main__':
    publisher = RoutingLogPublisher()

    # 다양한 레벨의 로그 발행
    publisher.publish_log('info', 'User logged in: user123')
    publisher.publish_log('warning', 'Disk space low: 15% remaining')
    publisher.publish_log('error', 'Payment processing failed: timeout')
    publisher.publish_log('info', 'Cache cleared successfully')
    publisher.publish_log('error', 'Database connection lost')

import pika
import json
import smtplib
from email.message import EmailMessage

def handle_error(ch, method, properties, body):
    """ERROR 로그 처리 및 알림"""
    log_data = json.loads(body.decode())

    print(f"""
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    🚨 ERROR ALERT
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    Severity: {log_data['severity']}
    Time: {log_data['timestamp']}
    Message: {log_data['message']}

    Action: Sending notification to admin...
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)

    # 실제로는 이메일, Slack 등으로 알림 발송
    # send_email_alert(log_data)
    # send_slack_alert(log_data)

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start():
    """Error Handler 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(
        exchange='direct_logs',
        exchange_type='direct'
    )

    # ERROR 전용 큐
    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # 🔑 핵심: 'error' routing key만 바인딩
    channel.queue_bind(
        exchange='direct_logs',
        queue=queue_name,
        routing_key='error'  # ERROR만 받음
    )

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=handle_error
    )

    print(' [*] Error Handler waiting for ERROR logs...')
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nError Handler stopped')

All Logger (모든 레벨 처리)​

# all_logger.py
import pika
import json

def log_all(ch, method, properties, body):
    """모든 로그 기록"""
    log_data = json.loads(body.decode())

    # 레벨별 아이콘
    icons = {
        'info': 'ℹ️',
        'warning': '⚠️',
        'error': '🚨'
    }

    icon = icons.get(log_data['severity'], '📝')

    print(f"{icon} [{log_data['timestamp']}] "
          f"{log_data['severity'].upper()}: {log_data['message']}")

    # 파일에도 저장
    with open('all_logs.txt', 'a', encoding='utf-8') as f:
        f.write(f"[{log_data['timestamp']}] "
                f"{log_data['severity'].upper()}: {log_data['message']}\n")

    ch.basic_ack(delivery_tag=method.delivery_tag)

def start():
    """All Logger 시작"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(
        exchange='direct_logs',
        exchange_type='direct'
    )

    result = channel.queue_declare(queue='', exclusive=True)
    queue_name = result.method.queue

    # 🔑 핵심: 여러 routing key 바인딩
    severities = ['info', 'warning', 'error']
    for severity in severities:
        channel.queue_bind(
            exchange='direct_logs',
            queue=queue_name,
            routing_key=severity  # 모든 레벨 받음
        )

    channel.basic_consume(
        queue=queue_name,
        on_message_callback=log_all
    )

    print(' [*] All Logger waiting for ALL logs...')
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start()
    except KeyboardInterrupt:
        print('\nAll Logger stopped')

🧪 테스트 코드​

# test_routing.py
import pytest
import pika
import json

class TestRouting:
    """Routing 패턴 테스트"""

    @pytest.fixture
    def setup(self):
        """테스트 환경 설정"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        try:
            channel.exchange_delete(exchange='test_direct_logs')
        except:
            pass

        yield channel
        connection.close()

    def test_direct_exchange_routing(self, setup):
        """Direct Exchange 라우팅 테스트"""
        channel = setup

        channel.exchange_declare(
            exchange='test_direct_logs',
            exchange_type='direct'
        )

        # 3개 큐 생성 (각각 다른 routing key)
        error_queue = channel.queue_declare(queue='', exclusive=True).method.queue
        warning_queue = channel.queue_declare(queue='', exclusive=True).method.queue
        info_queue = channel.queue_declare(queue='', exclusive=True).method.queue

        # 바인딩
        channel.queue_bind(
            exchange='test_direct_logs',
            queue=error_queue,
            routing_key='error'
        )
        channel.queue_bind(
            exchange='test_direct_logs',
            queue=warning_queue,
            routing_key='warning'
        )
        channel.queue_bind(
            exchange='test_direct_logs',
            queue=info_queue,
            routing_key='info'
        )

        # ERROR 메시지 발행
        error_msg = json.dumps({'severity': 'error', 'message': 'Test error'})
        channel.basic_publish(
            exchange='test_direct_logs',
            routing_key='error',
            body=error_msg
        )

        # ERROR 큐만 메시지 받음
        method, properties, body = channel.basic_get(
            queue=error_queue,
            auto_ack=True
        )
        assert body.decode() == error_msg

        # WARNING, INFO 큐는 비어있음
        assert channel.basic_get(queue=warning_queue, auto_ack=True)[0] is None
        assert channel.basic_get(queue=info_queue, auto_ack=True)[0] is None

    def test_multiple_bindings(self, setup):
        """하나의 큐에 여러 routing key 바인딩 테스트"""
        channel = setup

        channel.exchange_declare(
            exchange='test_direct_logs',
            exchange_type='direct'
        )

        # 1개 큐에 여러 routing key 바인딩
        all_queue = channel.queue_declare(queue='', exclusive=True).method.queue

        for severity in ['info', 'warning', 'error']:
            channel.queue_bind(
                exchange='test_direct_logs',
                queue=all_queue,
                routing_key=severity
            )

        # 다양한 레벨 메시지 발행
        messages = []
        for severity in ['info', 'warning', 'error']:
            msg = json.dumps({'severity': severity, 'message': f'Test {severity}'})
            channel.basic_publish(
                exchange='test_direct_logs',
                routing_key=severity,
                body=msg
            )
            messages.append(msg)

        # 모든 메시지를 받았는지 확인
        received = []
        for _ in range(3):
            method, properties, body = channel.basic_get(
                queue=all_queue,
                auto_ack=True
            )
            received.append(body.decode())

        assert len(received) == 3
        for msg in messages:
            assert msg in received

    def test_routing_key_mismatch(self, setup):
        """Routing key가 매칭되지 않으면 메시지 미전달"""
        channel = setup

        channel.exchange_declare(
            exchange='test_direct_logs',
            exchange_type='direct'
        )

        # 'error'만 바인딩
        queue = channel.queue_declare(queue='', exclusive=True).method.queue
        channel.queue_bind(
            exchange='test_direct_logs',
            queue=queue,
            routing_key='error'
        )

        # 'info' 메시지 발행 (매칭 안됨)
        channel.basic_publish(
            exchange='test_direct_logs',
            routing_key='info',
            body='Test info'
        )

        # 큐가 비어있어야 함
        method, properties, body = channel.basic_get(
            queue=queue,
            auto_ack=True
        )
        assert method is None

2. Routing Key​

# Publisher: routing_key 지정

    exchange='direct_logs',
    routing_key='error',  # 🔑 이 키로 라우팅
    body=message
)

# Consumer: 동일한 routing_key로 바인딩
channel.queue_bind(
    exchange='direct_logs',
    queue=queue_name,
    routing_key='error'  # 🔑 이 키만 받음
)

🔍 Exchange 타입 비교​

Fanout vs Direct​

# Fanout: 모든 큐에 전송 (routing_key 무시)
channel.exchange_declare(exchange='logs', exchange_type='fanout')
channel.basic_publish(exchange='logs', routing_key='', body=msg)
# → 모든 바인딩된 큐가 받음

# Direct: routing_key 정확히 매칭
channel.exchange_declare(exchange='direct_logs', exchange_type='direct')
channel.basic_publish(exchange='direct_logs', routing_key='error', body=msg)
# → 'error'로 바인딩된 큐만 받음

# event_publisher.py
import pika
import json

def publish_event(event_type: str, data: dict):
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    channel.exchange_declare(
        exchange='events',
        exchange_type='direct'
    )

    channel.basic_publish(
        exchange='events',
        routing_key=event_type,
        body=json.dumps(data)
    )

    print(f" [x] Published {event_type} event")
    connection.close()

# 사용 예
publish_event('payment', {'amount': 100, 'user_id': 123})
publish_event('order', {'order_id': 'ORD-456', 'items': 3})
publish_event('shipping', {'tracking_id': 'TRK-789'})
# payment_handler.py
import pika
import json

def handle_payment(ch, method, properties, body):
    data = json.loads(body.decode())
    print(f" [x] Processing payment: ${data['amount']}")
    ch.basic_ack(delivery_tag=method.delivery_tag)

connection = pika.BlockingConnection(
    pika.ConnectionParameters('localhost')
)
channel = connection.channel()

channel.exchange_declare(exchange='events', exchange_type='direct')
result = channel.queue_declare(queue='', exclusive=True)
queue_name = result.method.queue

channel.queue_bind(
    exchange='events',
    queue=queue_name,
    routing_key='payment'
)

channel.basic_consume(
    queue=queue_name,
    on_message_callback=handle_payment
)

print(' [*] Payment Handler waiting...')
channel.start_consuming()
