Hello World - 첫 메시지

가장 간단한 메시지 전송과 수신을 배워봅시다!

🎯 학습 목표​

  • Producer 작성하기
  • Consumer 작성하기
  • 메시지 전송 및 수신
  • 테스트 코드 작성

📊 패턴 구조​

Producer  -->  [Queue]  -->  Consumer
   (P)           hello          (C)

특징:

  • 1:1 통신
  • 가장 간단한 패턴
  • Exchange 사용 안 함 (기본 Exchange 사용)

📝 Producer 작성​

# producer/send.py
import pika

def send_message(message: str):
    """메시지를 hello 큐로 전송"""

    # 1. 연결 생성
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 2. 큐 선언 (없으면 생성)
    channel.queue_declare(queue='hello')

    # 3. 메시지 발행
    channel.basic_publish(
        exchange='',              # 기본 exchange 사용
        routing_key='hello',      # 큐 이름
        body=message
    )

    print(f" [x] Sent '{message}'")

    # 4. 연결 종료
    connection.close()

if __name__ == '__main__':
    send_message('Hello World!')
python producer/send.py
# [x] Sent 'Hello World!'

📥 Consumer 작성​

# consumer/receive.py
import pika

def callback(ch, method, properties, body):
    """메시지를 받았을 때 실행되는 콜백"""
    print(f" [x] Received {body.decode()}")

def start_consumer():
    """Consumer 시작"""

    # 1. 연결 생성
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 2. 큐 선언 (Producer와 동일한 큐)
    channel.queue_declare(queue='hello')

    # 3. Consumer 설정
    channel.basic_consume(
        queue='hello',
        on_message_callback=callback,
        auto_ack=True  # 자동 확인
    )

    print(' [*] Waiting for messages. To exit press CTRL+C')

    # 4. 메시지 대기 시작
    channel.start_consuming()

if __name__ == '__main__':
    try:
        start_consumer()
    except KeyboardInterrupt:
        print('Interrupted')
python consumer/receive.py
# [*] Waiting for messages. To exit press CTRL+C
# [x] Received Hello World!

🧪 테스트 코드​

# tests/test_hello_world.py
import pytest
import pika
import time
from threading import Thread

class TestHelloWorld:
    """Hello World 패턴 테스트"""

    @pytest.fixture
    def connection(self):
        """RabbitMQ 연결 fixture"""
        conn = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        yield conn
        conn.close()

    @pytest.fixture
    def channel(self, connection):
        """채널 fixture"""
        ch = connection.channel()
        ch.queue_declare(queue='test_hello')
        yield ch
        # 테스트 후 큐 삭제
        ch.queue_delete(queue='test_hello')

    def test_send_message(self, channel):
        """메시지 전송 테스트"""
        # 메시지 발행
        channel.basic_publish(
            exchange='',
            routing_key='test_hello',
            body='Test Message'
        )

        # 큐에 메시지가 있는지 확인
        method, properties, body = channel.basic_get(
            queue='test_hello',
            auto_ack=True
        )

        assert body == b'Test Message'

    def test_receive_message(self, channel):
        """메시지 수신 테스트"""
        received_messages = []

        def callback(ch, method, properties, body):
            received_messages.append(body.decode())

        # Consumer 설정
        channel.basic_consume(
            queue='test_hello',
            on_message_callback=callback,
            auto_ack=True
        )

        # 메시지 발행
        channel.basic_publish(
            exchange='',
            routing_key='test_hello',
            body='Test Message'
        )

        # 짧은 시간 동안 메시지 수신
        connection = channel.connection
        connection.process_data_events(time_limit=1)

        assert 'Test Message' in received_messages

    def test_multiple_messages(self, channel):
        """여러 메시지 전송 테스트"""
        messages = ['Message 1', 'Message 2', 'Message 3']

        # 메시지 발행
        for msg in messages:
            channel.basic_publish(
                exchange='',
                routing_key='test_hello',
                body=msg
            )

        # 메시지 수신 확인
        received = []
        for _ in range(len(messages)):
            method, properties, body = channel.basic_get(
                queue='test_hello',
                auto_ack=True
            )
            if body:
                received.append(body.decode())

        assert received == messages
# 테스트 실행
pytest tests/test_hello_world.py -v

# 출력:
# tests/test_hello_world.py::TestHelloWorld::test_send_message PASSED
# tests/test_hello_world.py::TestHelloWorld::test_receive_message PASSED
# tests/test_hello_world.py::TestHelloWorld::test_multiple_messages PASSED

# cli_send.py
import sys
import pika

def main():
    if len(sys.argv) < 2:
        print("Usage: python cli_send.py <message>")
        sys.exit(1)

    message = ' '.join(sys.argv[1:])

    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()
    channel.queue_declare(queue='hello')

    channel.basic_publish(
        exchange='',
        routing_key='hello',
        body=message
    )

    print(f" [x] Sent '{message}'")
    connection.close()

if __name__ == '__main__':
    main()
python cli_send.py "안녕하세요 RabbitMQ!"
# [x] Sent '안녕하세요 RabbitMQ!'

2. JSON 메시지​

# send_json.py
import pika
import json

def send_json_message(data: dict):
    """JSON 메시지 전송"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()
    channel.queue_declare(queue='json_queue')

    # dict를 JSON 문자열로 변환
    message = json.dumps(data)

    channel.basic_publish(
        exchange='',
        routing_key='json_queue',
        body=message,
        properties=pika.BasicProperties(
            content_type='application/json',
        )
    )

    print(f" [x] Sent JSON: {message}")
    connection.close()

# 사용 예
send_json_message({
    'user_id': 123,
    'action': 'login',
    'timestamp': '2024-01-26T10:00:00'
})
# receive_json.py
import pika
import json

def callback(ch, method, properties, body):
    """JSON 메시지 수신"""
    data = json.loads(body)
    print(f" [x] Received JSON:")
    print(f"     User ID: {data['user_id']}")
    print(f"     Action: {data['action']}")
    print(f"     Time: {data['timestamp']}")

connection = pika.BlockingConnection(
    pika.ConnectionParameters('localhost')
)
channel = connection.channel()
channel.queue_declare(queue='json_queue')

channel.basic_consume(
    queue='json_queue',
    on_message_callback=callback,
    auto_ack=True
)

print(' [*] Waiting for JSON messages...')
channel.start_consuming()

3. Context Manager 패턴​

# rabbitmq_context.py
import pika
from contextlib import contextmanager

@contextmanager
def rabbitmq_connection(host='localhost'):
    """연결 컨텍스트 매니저"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters(host)
    )
    try:
        yield connection
    finally:
        connection.close()

@contextmanager
def rabbitmq_channel(host='localhost'):
    """채널 컨텍스트 매니저"""
    with rabbitmq_connection(host) as connection:
        channel = connection.channel()
        try:
            yield channel
        finally:
            pass  # 채널은 연결 종료 시 자동 닫힘

# 사용 예
with rabbitmq_channel() as channel:
    channel.queue_declare(queue='hello')
    channel.basic_publish(
        exchange='',
        routing_key='hello',
        body='Context Manager 테스트'
    )
    print(" [x] Message sent!")

💡 주요 포인트​

queue_declare의 멱등성​

# 여러 번 호출해도 안전 (이미 있으면 무시)
channel.queue_declare(queue='hello')
channel.queue_declare(queue='hello')
channel.queue_declare(queue='hello')

auto_ack의 의미​

# auto_ack=True: 메시지를 받자마자 자동 확인
channel.basic_consume(
    queue='hello',
    on_message_callback=callback,
    auto_ack=True  # 자동 확인 ON
)

# auto_ack=False: 수동으로 확인해야 함 (다음 챕터에서)

연결 리소스 관리​

# ❌ 나쁜 예: 연결을 닫지 않음
def send_message(msg):
    connection = pika.BlockingConnection(...)
    channel = connection.channel()
    channel.basic_publish(...)
    # connection.close() 누락!

# ✅ 좋은 예: 항상 연결 종료
def send_message(msg):
    connection = pika.BlockingConnection(...)
    try:
        channel = connection.channel()
        channel.basic_publish(...)
    finally:
        connection.close()

# producer_input.py
import pika

def main():
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()
    channel.queue_declare(queue='user_messages')

    while True:
        message = input("메시지 입력 (quit to exit): ")
        if message.lower() == 'quit':
            break

        channel.basic_publish(
            exchange='',
            routing_key='user_messages',
            body=message
        )
        print(f" [x] Sent: {message}")

    connection.close()

if __name__ == '__main__':
    main()
# consumer_file.py
import pika
from datetime import datetime

def callback(ch, method, properties, body):
    message = body.decode()
    timestamp = datetime.now().isoformat()

    with open('messages.log', 'a') as f:
        f.write(f"[{timestamp}] {message}\n")

    print(f" [x] Saved: {message}")

connection = pika.BlockingConnection(
    pika.ConnectionParameters('localhost')
)
channel = connection.channel()
channel.queue_declare(queue='user_messages')

channel.basic_consume(
    queue='user_messages',
    on_message_callback=callback,
    auto_ack=True
)

print(' [*] Waiting for messages...')
channel.start_consuming()

📚 핵심 정리​

  • queue_declare(): 큐 생성/확인
  • basic_publish(): 메시지 발행
  • basic_consume(): 메시지 수신 시작
  • auto_ack=True: 자동 메시지 확인
