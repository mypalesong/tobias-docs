# models.py
from dataclasses import dataclass, asdict, field
from datetime import datetime
from typing import Optional
from enum import Enum
import json

class DeviceType(str, Enum):
    """ë””ë°”ì´ìŠ¤ íƒ€ì…"""
    TEMPERATURE = 'temperature'
    HUMIDITY = 'humidity'
    POWER = 'power'
    CAMERA = 'camera'

@dataclass
class Telemetry:
    """í…”ë ˆë©”íŠ¸ë¦¬ ë°ì´í„°"""
    device_id: str
    device_type: DeviceType
    location: str  # ì˜ˆ: building1.floor2.room201
    value: float
    unit: str
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())
    is_alert: bool = False

    def to_json(self) -> str:
        data = asdict(self)
        data['device_type'] = self.device_type.value
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str):
        data = json.loads(json_str)
        data['device_type'] = DeviceType(data['device_type'])
        return cls(**data)

    def get_routing_key(self) -> str:
        """Routing key ìƒì„±"""
        # sensor.building1.floor2.temperature
        location_parts = self.location.replace('.', '_')

        if self.is_alert:
            return f'sensor.{location_parts}.alert'
        else:
            return f'sensor.{location_parts}.{self.device_type.value}'

                content_type='application/json',
                timestamp=int(datetime.now().timestamp())

        telemetry = Telemetry(
            device_id=device_id,
            device_type=DeviceType.TEMPERATURE,
            location=location,
            value=round(value, 1),
            unit='Â°C',
            is_alert=is_alert

    def simulate_humidity_sensor(self, device_id: str, location: str):
        """ìŠµë„ ì„¼ì„œ ì‹œë®¬ë ˆì´ì…˜"""

            device_id=device_id,
            device_type=DeviceType.HUMIDITY,
            location=location,
            value=round(value, 1),
            unit='%',
            is_alert=is_alert

        except KeyboardInterrupt:
            print("\n IoT Gateway stopped")
            self.connection.close()

                print(f"     Avg: {avg_temp:.1f}Â°C | "
                      f"Min: {min_temp:.1f}Â°C | "
                      f"Max: {max_temp:.1f}Â°C | "
                      f"Samples: {len(temps)}")

        print()

        # ì§‘ê³„ í›„ ì´ˆê¸°í™”
        self.readings.clear()

        """ë©”ì‹œì§€ ì²˜ë¦¬ ì½œë°±"""

            pika.ConnectionParameters(self.host)

            exchange_type='topic',
            durable=True

import pika
import time
import threading
from models import Telemetry, DeviceType
import random

class LoadTester:
    """ë¶€í•˜ í…ŒìŠ¤íŠ¸"""

    def __init__(self, host='localhost'):
        self.host = host
        self.sent_count = 0
        self.lock = threading.Lock()

    def publisher_thread(self, thread_id: int, messages_per_sec: int):
        """Publisher ìŠ¤ë ˆë“œ"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.exchange_declare(
            exchange='telemetry',
            exchange_type='topic',
            durable=True
        )

        interval = 1.0 / messages_per_sec

        start_time = time.time()
        duration = 60  # 1ë¶„

        while time.time() - start_time < duration:
            # ëœë¤ í…”ë ˆë©”íŠ¸ë¦¬ ìƒì„±
            telemetry = Telemetry(
                device_id=f'DEVICE-{thread_id}-{random.randint(1, 100)}',
                device_type=random.choice(list(DeviceType)),
                location=f'building1.floor{random.randint(1, 5)}',
                value=random.uniform(20, 25),
                unit='Â°C',
                is_alert=False
            )

            routing_key = telemetry.get_routing_key()

            channel.basic_publish(
                exchange='telemetry',
                routing_key=routing_key,
                body=telemetry.to_json(),
                properties=pika.BasicProperties(delivery_mode=1)
            )

            with self.lock:
                self.sent_count += 1

            time.sleep(interval)

        connection.close()

    def run_test(self, publisher_count: int = 10, messages_per_sec: int = 100):
        """ë¶€í•˜ í…ŒìŠ¤íŠ¸ ì‹¤í–‰"""
        print(f" Load Test")
        print(" " + "="*50)
        print(f" Publishers: {publisher_count}")
        print(f" Target Rate: {messages_per_sec * publisher_count} msg/s")
        print(f" Duration: 60s\n")

        # Publisher ìŠ¤ë ˆë“œ ì‹œì‘
        threads = []
        start_time = time.time()

        for i in range(publisher_count):
            t = threading.Thread(
                target=self.publisher_thread,
                args=(i+1, messages_per_sec)
            )
            t.start()
            threads.append(t)

        # ì§„í–‰ ìƒí™© í‘œì‹œ
        while any(t.is_alive() for t in threads):
            elapsed = time.time() - start_time
            rate = self.sent_count / elapsed if elapsed > 0 else 0

            print(f"\r Sent: {self.sent_count:6d} | "
                  f"Rate: {rate:6.1f} msg/s | "
                  f"Elapsed: {elapsed:4.1f}s",
                  end='')

            time.sleep(1)

        # ëª¨ë“  ìŠ¤ë ˆë“œ ì™„ë£Œ ëŒ€ê¸°
        for t in threads:
            t.join()

        # ê²°ê³¼ ì¶œë ¥
        total_time = time.time() - start_time
        actual_rate = self.sent_count / total_time

        print(f"\n\n Results:")
        print(" " + "="*50)
        print(f" Total Sent: {self.sent_count}")
        print(f" Duration: {total_time:.2f}s")
        print(f" Actual Rate: {actual_rate:.2f} msg/s")

# ì‚¬ìš© ì˜ˆ

    tester.run_test(publisher_count=10, messages_per_sec=100)
