시나리오 5: IoT 텔레메트리 처리

수천 개의 IoT 디바이스에서 실시간 데이터를 수집하고 처리하는 시스템을 구현해봅시다!

📌 상황 설명​

스마트 빌딩 관리 시스템:

  • 디바이스 타입:
    • 온도 센서 (1000개)
    • 습도 센서 (1000개)
    • 전력 미터 (500개)
    • 보안 카메라 (200개)
  • 요구사항:
    • 초당 수천 건의 데이터 처리
    • 실시간 알림 (임계값 초과 시)
    • 데이터 집계 및 분석
    • 장기 저장

🏗️ 아키텍처​

IoT Devices → [Topic Exchange: telemetry]
                       ↓
         sensor.*.temperature → Temperature Processor
         sensor.*.humidity    → Humidity Processor
         sensor.*.power       → Power Processor
         sensor.#             → Data Archiver
         *.*.alert            → Alert Service

패턴: Topic Exchange로 데이터 타입별 라우팅

📝 구현​

텔레메트리 모델​

# models.py
from dataclasses import dataclass, asdict, field
from datetime import datetime
from typing import Optional
from enum import Enum
import json

class DeviceType(str, Enum):
    """디바이스 타입"""
    TEMPERATURE = 'temperature'
    HUMIDITY = 'humidity'
    POWER = 'power'
    CAMERA = 'camera'

@dataclass
class Telemetry:
    """텔레메트리 데이터"""
    device_id: str
    device_type: DeviceType
    location: str  # 예: building1.floor2.room201
    value: float
    unit: str
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())
    is_alert: bool = False

    def to_json(self) -> str:
        data = asdict(self)
        data['device_type'] = self.device_type.value
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str):
        data = json.loads(json_str)
        data['device_type'] = DeviceType(data['device_type'])
        return cls(**data)

    def get_routing_key(self) -> str:
        """Routing key 생성"""
        # sensor.building1.floor2.temperature
        location_parts = self.location.replace('.', '_')

        if self.is_alert:
            return f'sensor.{location_parts}.alert'
        else:
            return f'sensor.{location_parts}.{self.device_type.value}'

IoT Gateway (Publisher)​

# iot_gateway.py
import pika
import random
import time
from models import Telemetry, DeviceType
from datetime import datetime

class IoTGateway:
    """IoT 게이트웨이 (디바이스 데이터 수집 및 발행)"""

    def __init__(self, host='localhost'):
        self.host = host
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host)
        )
        self.channel = self.connection.channel()

        # Topic Exchange 선언
        self.channel.exchange_declare(
            exchange='telemetry',
            exchange_type='topic',
            durable=True
        )

    def publish(self, telemetry: Telemetry):
        """텔레메트리 데이터 발행"""
        routing_key = telemetry.get_routing_key()

        self.channel.basic_publish(
            exchange='telemetry',
            routing_key=routing_key,
            body=telemetry.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=1,  # Non-persistent (고속 처리)
                content_type='application/json',
                timestamp=int(datetime.now().timestamp())
            )
        )

    def simulate_temperature_sensor(self, device_id: str, location: str):
        """온도 센서 시뮬레이션"""
        # 정상 범위: 18-26°C
        base_temp = 22.0
        value = base_temp + random.uniform(-4, 4)

        # 임계값 체크
        is_alert = value < 18.0 or value > 26.0

        telemetry = Telemetry(
            device_id=device_id,
            device_type=DeviceType.TEMPERATURE,
            location=location,
            value=round(value, 1),
            unit='°C',
            is_alert=is_alert
        )

        self.publish(telemetry)

        return telemetry

    def simulate_humidity_sensor(self, device_id: str, location: str):
        """습도 센서 시뮬레이션"""
        # 정상 범위: 40-60%
        base_humidity = 50.0
        value = base_humidity + random.uniform(-15, 15)

        is_alert = value < 40.0 or value > 60.0

        telemetry = Telemetry(
            device_id=device_id,
            device_type=DeviceType.HUMIDITY,
            location=location,
            value=round(value, 1),
            unit='%',
            is_alert=is_alert
        )

        self.publish(telemetry)

        return telemetry

    def simulate_power_meter(self, device_id: str, location: str):
        """전력 미터 시뮬레이션"""
        # 정상 범위: 0-100kW
        value = random.uniform(10, 90)

        is_alert = value > 80.0  # 80kW 초과 시 알림

        telemetry = Telemetry(
            device_id=device_id,
            device_type=DeviceType.POWER,
            location=location,
            value=round(value, 2),
            unit='kW',
            is_alert=is_alert
        )

        self.publish(telemetry)

        return telemetry

    def run_continuous(self, interval: float = 0.1):
        """지속적인 데이터 발행"""
        print(f" [*] IoT Gateway started")
        print(f" [*] Publishing every {interval}s...\n")

        devices = [
            ('TEMP-001', 'building1.floor1.room101', 'temperature'),
            ('TEMP-002', 'building1.floor1.room102', 'temperature'),
            ('HUM-001', 'building1.floor2.room201', 'humidity'),
            ('HUM-002', 'building1.floor2.room202', 'humidity'),
            ('POW-001', 'building1.basement', 'power'),
        ]

        try:
            while True:
                # 모든 디바이스에서 데이터 발행
                for device_id, location, device_type in devices:
                    if device_type == 'temperature':
                        telemetry = self.simulate_temperature_sensor(device_id, location)
                    elif device_type == 'humidity':
                        telemetry = self.simulate_humidity_sensor(device_id, location)
                    elif device_type == 'power':
                        telemetry = self.simulate_power_meter(device_id, location)

                    if telemetry.is_alert:
                        print(f" [⚠] Alert: {telemetry.device_id} = "
                              f"{telemetry.value}{telemetry.unit}")

                time.sleep(interval)

        except KeyboardInterrupt:
            print("\n IoT Gateway stopped")
            self.connection.close()

# 사용 예
if __name__ == '__main__':
    gateway = IoTGateway()
    gateway.run_continuous(interval=1.0)

Temperature Processor​

# temperature_processor.py
import pika
from models import Telemetry
from collections import defaultdict
import time

class TemperatureProcessor:
    """온도 데이터 처리"""

    def __init__(self, host='localhost'):
        self.host = host
        self.readings = defaultdict(list)  # 위치별 온도 기록
        self.last_aggregation = time.time()

    def process(self, telemetry: Telemetry):
        """온도 데이터 처리"""
        # 위치별로 온도 기록
        self.readings[telemetry.location].append(telemetry.value)

        # 실시간 표시
        print(f" [TEMP] {telemetry.device_id} @ {telemetry.location}: "
              f"{telemetry.value}°C")

        # 5초마다 집계
        now = time.time()
        if now - self.last_aggregation >= 5:
            self.aggregate()
            self.last_aggregation = now

    def aggregate(self):
        """온도 집계"""
        print("\n [📊] Temperature Aggregation:")

        for location, temps in self.readings.items():
            if temps:
                avg_temp = sum(temps) / len(temps)
                min_temp = min(temps)
                max_temp = max(temps)

                print(f"   {location}:")
                print(f"     Avg: {avg_temp:.1f}°C | "
                      f"Min: {min_temp:.1f}°C | "
                      f"Max: {max_temp:.1f}°C | "
                      f"Samples: {len(temps)}")

        print()

        # 집계 후 초기화
        self.readings.clear()

    def callback(self, ch, method, properties, body):
        """메시지 처리 콜백"""
        telemetry = Telemetry.from_json(body.decode())
        self.process(telemetry)
        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """프로세서 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.exchange_declare(
            exchange='telemetry',
            exchange_type='topic',
            durable=True
        )

        # 임시 큐
        result = channel.queue_declare(queue='', exclusive=True)
        queue_name = result.method.queue

        # 모든 온도 데이터 구독
        channel.queue_bind(
            exchange='telemetry',
            queue=queue_name,
            routing_key='sensor.*.temperature'
        )

        channel.basic_qos(prefetch_count=100)  # 배치 처리

        channel.basic_consume(
            queue=queue_name,
            on_message_callback=self.callback
        )

        print(' [*] Temperature Processor started')
        print(' [*] Subscribed to: sensor.*.temperature\n')

        channel.start_consuming()

if __name__ == '__main__':
    processor = TemperatureProcessor()

    try:
        processor.start()
    except KeyboardInterrupt:
        print('\nTemperature Processor stopped')

Data Archiver​

# data_archiver.py
import pika
from models import Telemetry
import json

class DataArchiver:
    """데이터 아카이버 (장기 저장)"""

    def __init__(self, host='localhost'):
        self.host = host

    def archive(self, telemetry: Telemetry):
        """데이터 저장"""
        # 파일에 저장 (실제로는 TimescaleDB, InfluxDB 등 사용)
        with open('telemetry_archive.jsonl', 'a', encoding='utf-8') as f:
            f.write(telemetry.to_json() + '\n')

        # 통계 출력 (매 100건마다)
        if not hasattr(self, 'count'):
            self.count = 0

        self.count += 1

        if self.count % 100 == 0:
            print(f" [ARCHIVE] {self.count} records archived")

    def callback(self, ch, method, properties, body):
        """메시지 처리 콜백"""
        telemetry = Telemetry.from_json(body.decode())
        self.archive(telemetry)
        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """아카이버 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.exchange_declare(
            exchange='telemetry',
            exchange_type='topic',
            durable=True
        )

        # 영구 큐 (데이터 손실 방지)
        channel.queue_declare(queue='telemetry.archive', durable=True)

        # 모든 센서 데이터 구독
        channel.queue_bind(
            exchange='telemetry',
            queue='telemetry.archive',
            routing_key='sensor.#'
        )

        channel.basic_qos(prefetch_count=1000)  # 대량 처리

        channel.basic_consume(
            queue='telemetry.archive',
            on_message_callback=self.callback
        )

        print(' [*] Data Archiver started')
        print(' [*] Subscribed to: sensor.#\n')

        channel.start_consuming()

if __name__ == '__main__':
    archiver = DataArchiver()

    try:
        archiver.start()
    except KeyboardInterrupt:
        print('\nData Archiver stopped')

🧪 성능 테스트​

# load_test.py
import pika
import time
import threading
from models import Telemetry, DeviceType
import random

class LoadTester:
    """부하 테스트"""

    def __init__(self, host='localhost'):
        self.host = host
        self.sent_count = 0
        self.lock = threading.Lock()

    def publisher_thread(self, thread_id: int, messages_per_sec: int):
        """Publisher 스레드"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.exchange_declare(
            exchange='telemetry',
            exchange_type='topic',
            durable=True
        )

        interval = 1.0 / messages_per_sec

        start_time = time.time()
        duration = 60  # 1분

        while time.time() - start_time < duration:
            # 랜덤 텔레메트리 생성
            telemetry = Telemetry(
                device_id=f'DEVICE-{thread_id}-{random.randint(1, 100)}',
                device_type=random.choice(list(DeviceType)),
                location=f'building1.floor{random.randint(1, 5)}',
                value=random.uniform(20, 25),
                unit='°C',
                is_alert=False
            )

            routing_key = telemetry.get_routing_key()

            channel.basic_publish(
                exchange='telemetry',
                routing_key=routing_key,
                body=telemetry.to_json(),
                properties=pika.BasicProperties(delivery_mode=1)
            )

            with self.lock:
                self.sent_count += 1

            time.sleep(interval)

        connection.close()

    def run_test(self, publisher_count: int = 10, messages_per_sec: int = 100):
        """부하 테스트 실행"""
        print(f" Load Test")
        print(" " + "="*50)
        print(f" Publishers: {publisher_count}")
        print(f" Target Rate: {messages_per_sec * publisher_count} msg/s")
        print(f" Duration: 60s\n")

        # Publisher 스레드 시작
        threads = []
        start_time = time.time()

        for i in range(publisher_count):
            t = threading.Thread(
                target=self.publisher_thread,
                args=(i+1, messages_per_sec)
            )
            t.start()
            threads.append(t)

        # 진행 상황 표시
        while any(t.is_alive() for t in threads):
            elapsed = time.time() - start_time
            rate = self.sent_count / elapsed if elapsed > 0 else 0

            print(f"\r Sent: {self.sent_count:6d} | "
                  f"Rate: {rate:6.1f} msg/s | "
                  f"Elapsed: {elapsed:4.1f}s",
                  end='')

            time.sleep(1)

        # 모든 스레드 완료 대기
        for t in threads:
            t.join()

        # 결과 출력
        total_time = time.time() - start_time
        actual_rate = self.sent_count / total_time

        print(f"\n\n Results:")
        print(" " + "="*50)
        print(f" Total Sent: {self.sent_count}")
        print(f" Duration: {total_time:.2f}s")
        print(f" Actual Rate: {actual_rate:.2f} msg/s")

# 사용 예
if __name__ == '__main__':
    tester = LoadTester()
    tester.run_test(publisher_count=10, messages_per_sec=100)
