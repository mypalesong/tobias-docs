from dataclasses import dataclass, asdict, field
from datetime import datetime
from typing import Dict, Any, Optional
from enum import Enum
import json
import hashlib

class AuditAction(str, Enum):
    """ê°ì‚¬ ì•¡ì…˜"""
    LOGIN = 'user.login'
    LOGOUT = 'user.logout'
    TRANSACTION = 'transaction.execute'
    DATA_ACCESS = 'data.access'
    DATA_MODIFY = 'data.modify'
    DATA_DELETE = 'data.delete'
    PERMISSION_CHANGE = 'permission.change'

class AuditSeverity(str, Enum):
    """ì¤‘ìš”ë„"""
    INFO = 'info'
    WARNING = 'warning'
    CRITICAL = 'critical'

@dataclass
class AuditLog:
    """ê°ì‚¬ ë¡œê·¸"""
    log_id: str
    timestamp: str
    action: AuditAction
    severity: AuditSeverity
    user_id: Optional[int]
    user_ip: str
    resource: str
    details: Dict[str, Any]
    checksum: str = field(default='')

    def __post_init__(self):
        if not self.timestamp:
            self.timestamp = datetime.now().isoformat()

        # ì²´í¬ì„¬ ìƒì„± (ë³€ì¡° ë°©ì§€)
        if not self.checksum:
            self.checksum = self.calculate_checksum()

    def calculate_checksum(self) -> str:
        """ì²´í¬ì„¬ ê³„ì‚° (SHA-256)"""
        data = f"{self.log_id}{self.timestamp}{self.action.value}" \
               f"{self.user_id}{self.resource}{json.dumps(self.details, sort_keys=True)}"

        return hashlib.sha256(data.encode()).hexdigest()

    def verify_checksum(self) -> bool:
        """ì²´í¬ì„¬ ê²€ì¦"""
        expected = self.calculate_checksum()
        return self.checksum == expected

    def to_json(self) -> str:
        data = asdict(self)
        data['action'] = self.action.value
        data['severity'] = self.severity.value
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str):
        data = json.loads(json_str)
        data['action'] = AuditAction(data['action'])
        data['severity'] = AuditSeverity(data['severity'])
        return cls(**data)

class AuditLogger:
    """ê°ì‚¬ ë¡œê±°"""

    def __init__(self, host='localhost'):

    def log(self, audit: AuditLog):
        """ê°ì‚¬ ë¡œê·¸ ë°œí–‰ (í™•ì¸ í•„ìˆ˜)"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)

            # ë©”ì‹œì§€ ë°œí–‰

                exchange='',
                routing_key='audit.logs',
                body=audit.to_json(),
                properties=pika.BasicProperties(
                    delivery_mode=2,  # Persistent
                    content_type='application/json',
                    priority=10 if audit.severity == AuditSeverity.CRITICAL else 5
                ),
                mandatory=True  # ë¼ìš°íŒ… ì‹¤íŒ¨ ì‹œ ì—ëŸ¬
            )

                channel.basic_publish(
                    exchange='',
                    routing_key='audit.critical',
                    body=audit.to_json(),
                    properties=pika.BasicProperties(
                        delivery_mode=2,
                        content_type='application/json'
                    )
                )

            print(f" [âœ“] Audit log published: {audit.log_id} "
                  f"({audit.action.value})")

            return True

        except pika.exceptions.UnroutableError:
            print(f" [âœ—] Failed to route audit log: {audit.log_id}")
            return False

        except pika.exceptions.NackError:
            print(f" [âœ—] Audit log nacked: {audit.log_id}")
            return False

        finally:
            connection.close()

    def log_login(self, user_id: int, user_ip: str, success: bool):
        """ë¡œê·¸ì¸ ê°ì‚¬"""

            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.LOGIN,
            severity=AuditSeverity.WARNING if not success else AuditSeverity.INFO,
            user_id=user_id,
            user_ip=user_ip,
            resource='auth.login',
            details={
                'success': success,
                'method': 'password'
            }
        )

        return self.log(audit)

    def log_transaction(
        self,
        user_id: int,
        user_ip: str,
        transaction_id: str,
        amount: float,
        recipient: str
    ):
        """ê±°ë˜ ê°ì‚¬"""
        audit = AuditLog(
            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.TRANSACTION,
            severity=AuditSeverity.CRITICAL if amount > 10000 else AuditSeverity.INFO,
            user_id=user_id,
            user_ip=user_ip,
            resource=f'transaction.{transaction_id}',
            details={
                'transaction_id': transaction_id,
                'amount': amount,
                'recipient': recipient
            }
        )

        return self.log(audit)

    def log_data_access(

        user_id: int,
        user_ip: str,
        resource: str,
        action: str
    ):
        """ë°ì´í„° ì ‘ê·¼ ê°ì‚¬"""

            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.DATA_ACCESS,
            severity=severity,
            user_id=user_id,
            user_ip=user_ip,
            resource=resource,
            details={
                'action': action
            }
        )

        return self.log(audit)

import pika
from models import AuditLog
import sqlite3
from datetime import datetime

class LogWriter:
    """ê°ì‚¬ ë¡œê·¸ ì €ì¥"""

    def __init__(self, host='localhost'):
        self.host = host

        # SQLite DB ì´ˆê¸°í™”
        self.db = sqlite3.connect('audit_logs.db')
        self.create_table()

    def create_table(self):
        """í…Œì´ë¸” ìƒì„±"""
        cursor = self.db.cursor()

        cursor.execute('''
            CREATE TABLE IF NOT EXISTS audit_logs (
                log_id TEXT PRIMARY KEY,
                timestamp TEXT,
                action TEXT,
                severity TEXT,
                user_id INTEGER,
                user_ip TEXT,
                resource TEXT,
                details TEXT,
                checksum TEXT,
                stored_at TEXT
            )
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_timestamp
            ON audit_logs(timestamp)
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_user_id
            ON audit_logs(user_id)
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_action
            ON audit_logs(action)
        ''')

        self.db.commit()

    def save_log(self, audit: AuditLog):
        """ë¡œê·¸ ì €ì¥"""
        # ì²´í¬ì„¬ ê²€ì¦
        if not audit.verify_checksum():
            print(f" [âœ—] Checksum verification failed: {audit.log_id}")
            return False

        cursor = self.db.cursor()

        try:
            cursor.execute('''
                INSERT INTO audit_logs VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', (
                audit.log_id,
                audit.timestamp,
                audit.action.value,
                audit.severity.value,
                audit.user_id,
                audit.user_ip,
                audit.resource,
                str(audit.details),
                audit.checksum,
                datetime.now().isoformat()
            ))

            self.db.commit()

            print(f" [âœ“] Saved: {audit.log_id} ({audit.action.value})")

            return True

        except sqlite3.IntegrityError:
            print(f" [!] Duplicate log: {audit.log_id}")
            return False

    def callback(self, ch, method, properties, body):
        """ë©”ì‹œì§€ ì²˜ë¦¬ ì½œë°±"""
        audit = AuditLog.from_json(body.decode())

        if self.save_log(audit):
            ch.basic_ack(delivery_tag=method.delivery_tag)
        else:
            # ì €ì¥ ì‹¤íŒ¨: NACK (ì¬ì‹œë„)
            ch.basic_nack(
                delivery_tag=method.delivery_tag,
                requeue=True
            )

    def start(self):
        """Log Writer ì‹œì‘"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.basic_qos(prefetch_count=1)

        channel.basic_consume(
            queue='audit.logs',
            on_message_callback=self.callback
        )

        print(' [*] Log Writer started')
        print(' [*] Waiting for audit logs...\n')

        try:
            channel.start_consuming()
        except KeyboardInterrupt:
            print('\nLog Writer stopped')
            self.db.close()
