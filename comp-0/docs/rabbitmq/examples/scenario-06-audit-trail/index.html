시나리오 6: 감사 추적 시스템

금융/의료 등 컴플라이언스가 중요한 시스템의 감사 로그를 안전하게 수집하는 시스템을 구현해봅시다!

📌 상황 설명​

금융 플랫폼의 감사 추적:

  • 감사 대상:
    • 사용자 인증 (로그인/로그아웃)
    • 금융 거래 (이체, 결제)
    • 데이터 접근 (조회, 수정, 삭제)
    • 권한 변경
  • 요구사항:
    • 100% 메시지 보장 (손실 불가)
    • 변조 방지
    • 장기 보관 (7년)
    • 빠른 검색

🏗️ 아키텍처​

Application Services → [Quorum Queue: audit.logs]
                              ↓
                         ┌────┴────┐
                         │         │
                    Log Writer  Compliance Checker
                         │         │
                         ▼         ▼
                    Database   Alert System

핵심:

  • Quorum Queue로 데이터 안정성 보장
  • Persistent 메시지로 손실 방지
  • Publisher Confirms로 신뢰성 확보

📝 구현​

감사 로그 모델​

# models.py
from dataclasses import dataclass, asdict, field
from datetime import datetime
from typing import Dict, Any, Optional
from enum import Enum
import json
import hashlib

class AuditAction(str, Enum):
    """감사 액션"""
    LOGIN = 'user.login'
    LOGOUT = 'user.logout'
    TRANSACTION = 'transaction.execute'
    DATA_ACCESS = 'data.access'
    DATA_MODIFY = 'data.modify'
    DATA_DELETE = 'data.delete'
    PERMISSION_CHANGE = 'permission.change'

class AuditSeverity(str, Enum):
    """중요도"""
    INFO = 'info'
    WARNING = 'warning'
    CRITICAL = 'critical'

@dataclass
class AuditLog:
    """감사 로그"""
    log_id: str
    timestamp: str
    action: AuditAction
    severity: AuditSeverity
    user_id: Optional[int]
    user_ip: str
    resource: str
    details: Dict[str, Any]
    checksum: str = field(default='')

    def __post_init__(self):
        if not self.timestamp:
            self.timestamp = datetime.now().isoformat()

        # 체크섬 생성 (변조 방지)
        if not self.checksum:
            self.checksum = self.calculate_checksum()

    def calculate_checksum(self) -> str:
        """체크섬 계산 (SHA-256)"""
        data = f"{self.log_id}{self.timestamp}{self.action.value}" \
               f"{self.user_id}{self.resource}{json.dumps(self.details, sort_keys=True)}"

        return hashlib.sha256(data.encode()).hexdigest()

    def verify_checksum(self) -> bool:
        """체크섬 검증"""
        expected = self.calculate_checksum()
        return self.checksum == expected

    def to_json(self) -> str:
        data = asdict(self)
        data['action'] = self.action.value
        data['severity'] = self.severity.value
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str):
        data = json.loads(json_str)
        data['action'] = AuditAction(data['action'])
        data['severity'] = AuditSeverity(data['severity'])
        return cls(**data)

Queue 설정​

# queue_setup.py
import pika

def setup_audit_queue():
    """감사 로그 큐 설정"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 🔑 Quorum Queue (고가용성)
    channel.queue_declare(
        queue='audit.logs',
        durable=True,
        arguments={
            'x-queue-type': 'quorum',  # Quorum Queue
            'x-max-in-memory-length': 0  # 모두 디스크에 저장
        }
    )

    # 중요 로그 큐 (CRITICAL만)
    channel.queue_declare(
        queue='audit.critical',
        durable=True,
        arguments={
            'x-queue-type': 'quorum'
        }
    )

    print(" [✓] Audit queues configured:")
    print("     - audit.logs (quorum, all logs)")
    print("     - audit.critical (quorum, critical only)")

    connection.close()

if __name__ == '__main__':
    setup_audit_queue()

Audit Logger (Publisher)​

# audit_logger.py
import pika
import uuid
from models import AuditLog, AuditAction, AuditSeverity
from datetime import datetime

class AuditLogger:
    """감사 로거"""

    def __init__(self, host='localhost'):
        self.host = host

    def log(self, audit: AuditLog):
        """감사 로그 발행 (확인 필수)"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # 🔑 Publisher Confirms 활성화
        channel.confirm_delivery()

        try:
            # 메시지 발행
            channel.basic_publish(
                exchange='',
                routing_key='audit.logs',
                body=audit.to_json(),
                properties=pika.BasicProperties(
                    delivery_mode=2,  # Persistent
                    content_type='application/json',
                    priority=10 if audit.severity == AuditSeverity.CRITICAL else 5
                ),
                mandatory=True  # 라우팅 실패 시 에러
            )

            # CRITICAL은 별도 큐에도 발행
            if audit.severity == AuditSeverity.CRITICAL:
                channel.basic_publish(
                    exchange='',
                    routing_key='audit.critical',
                    body=audit.to_json(),
                    properties=pika.BasicProperties(
                        delivery_mode=2,
                        content_type='application/json'
                    )
                )

            print(f" [✓] Audit log published: {audit.log_id} "
                  f"({audit.action.value})")

            return True

        except pika.exceptions.UnroutableError:
            print(f" [✗] Failed to route audit log: {audit.log_id}")
            return False

        except pika.exceptions.NackError:
            print(f" [✗] Audit log nacked: {audit.log_id}")
            return False

        finally:
            connection.close()

    def log_login(self, user_id: int, user_ip: str, success: bool):
        """로그인 감사"""
        audit = AuditLog(
            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.LOGIN,
            severity=AuditSeverity.WARNING if not success else AuditSeverity.INFO,
            user_id=user_id,
            user_ip=user_ip,
            resource='auth.login',
            details={
                'success': success,
                'method': 'password'
            }
        )

        return self.log(audit)

    def log_transaction(
        self,
        user_id: int,
        user_ip: str,
        transaction_id: str,
        amount: float,
        recipient: str
    ):
        """거래 감사"""
        audit = AuditLog(
            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.TRANSACTION,
            severity=AuditSeverity.CRITICAL if amount > 10000 else AuditSeverity.INFO,
            user_id=user_id,
            user_ip=user_ip,
            resource=f'transaction.{transaction_id}',
            details={
                'transaction_id': transaction_id,
                'amount': amount,
                'recipient': recipient
            }
        )

        return self.log(audit)

    def log_data_access(
        self,
        user_id: int,
        user_ip: str,
        resource: str,
        action: str
    ):
        """데이터 접근 감사"""
        severity = AuditSeverity.CRITICAL if action in ['modify', 'delete'] else AuditSeverity.INFO

        audit = AuditLog(
            log_id=f'AUDIT-{uuid.uuid4().hex[:12].upper()}',
            timestamp=datetime.now().isoformat(),
            action=AuditAction.DATA_ACCESS,
            severity=severity,
            user_id=user_id,
            user_ip=user_ip,
            resource=resource,
            details={
                'action': action
            }
        )

        return self.log(audit)

# 사용 예
if __name__ == '__main__':
    logger = AuditLogger()

    # 로그인 감사
    logger.log_login(user_id=1001, user_ip='192.168.1.100', success=True)
    logger.log_login(user_id=1002, user_ip='192.168.1.101', success=False)

    # 거래 감사
    logger.log_transaction(
        user_id=1001,
        user_ip='192.168.1.100',
        transaction_id='TXN-001',
        amount=5000.0,
        recipient='ACC-9999'
    )

    logger.log_transaction(
        user_id=1001,
        user_ip='192.168.1.100',
        transaction_id='TXN-002',
        amount=15000.0,  # CRITICAL (>10000)
        recipient='ACC-8888'
    )

    # 데이터 접근 감사
    logger.log_data_access(
        user_id=1001,
        user_ip='192.168.1.100',
        resource='customer.1234.info',
        action='read'
    )

    logger.log_data_access(
        user_id=1001,
        user_ip='192.168.1.100',
        resource='customer.1234.info',
        action='modify'  # CRITICAL
    )

Log Writer (Consumer)​

# log_writer.py
import pika
from models import AuditLog
import sqlite3
from datetime import datetime

class LogWriter:
    """감사 로그 저장"""

    def __init__(self, host='localhost'):
        self.host = host

        # SQLite DB 초기화
        self.db = sqlite3.connect('audit_logs.db')
        self.create_table()

    def create_table(self):
        """테이블 생성"""
        cursor = self.db.cursor()

        cursor.execute('''
            CREATE TABLE IF NOT EXISTS audit_logs (
                log_id TEXT PRIMARY KEY,
                timestamp TEXT,
                action TEXT,
                severity TEXT,
                user_id INTEGER,
                user_ip TEXT,
                resource TEXT,
                details TEXT,
                checksum TEXT,
                stored_at TEXT
            )
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_timestamp
            ON audit_logs(timestamp)
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_user_id
            ON audit_logs(user_id)
        ''')

        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_action
            ON audit_logs(action)
        ''')

        self.db.commit()

    def save_log(self, audit: AuditLog):
        """로그 저장"""
        # 체크섬 검증
        if not audit.verify_checksum():
            print(f" [✗] Checksum verification failed: {audit.log_id}")
            return False

        cursor = self.db.cursor()

        try:
            cursor.execute('''
                INSERT INTO audit_logs VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', (
                audit.log_id,
                audit.timestamp,
                audit.action.value,
                audit.severity.value,
                audit.user_id,
                audit.user_ip,
                audit.resource,
                str(audit.details),
                audit.checksum,
                datetime.now().isoformat()
            ))

            self.db.commit()

            print(f" [✓] Saved: {audit.log_id} ({audit.action.value})")

            return True

        except sqlite3.IntegrityError:
            print(f" [!] Duplicate log: {audit.log_id}")
            return False

    def callback(self, ch, method, properties, body):
        """메시지 처리 콜백"""
        audit = AuditLog.from_json(body.decode())

        if self.save_log(audit):
            ch.basic_ack(delivery_tag=method.delivery_tag)
        else:
            # 저장 실패: NACK (재시도)
            ch.basic_nack(
                delivery_tag=method.delivery_tag,
                requeue=True
            )

    def start(self):
        """Log Writer 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.basic_qos(prefetch_count=1)

        channel.basic_consume(
            queue='audit.logs',
            on_message_callback=self.callback
        )

        print(' [*] Log Writer started')
        print(' [*] Waiting for audit logs...\n')

        try:
            channel.start_consuming()
        except KeyboardInterrupt:
            print('\nLog Writer stopped')
            self.db.close()

if __name__ == '__main__':
    writer = LogWriter()
    writer.start()

Compliance Checker​

# compliance_checker.py
import pika
from models import AuditLog, AuditSeverity

class ComplianceChecker:
    """컴플라이언스 검사"""

    def __init__(self, host='localhost'):
        self.host = host

    def check_compliance(self, audit: AuditLog):
        """컴플라이언스 규칙 검사"""
        violations = []

        # 규칙 1: 대량 거래 (>10000) 검토 필요
        if audit.action.value == 'transaction.execute':
            amount = audit.details.get('amount', 0)

            if amount > 10000:
                violations.append({
                    'rule': 'HIGH_VALUE_TRANSACTION',
                    'description': f'Amount ${amount} exceeds threshold',
                    'action_required': 'Manual review required'
                })

        # 규칙 2: 로그인 실패 3회 이상 (같은 IP)
        # (실제로는 DB에서 조회)

        # 규칙 3: 민감한 데이터 수정 (CRITICAL)
        if audit.severity == AuditSeverity.CRITICAL:
            violations.append({
                'rule': 'CRITICAL_OPERATION',
                'description': 'Critical operation detected',
                'action_required': 'Notify security team'
            })

        return violations

    def handle_violation(self, audit: AuditLog, violations: list):
        """위반 처리"""
        print(f"""
        ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        ⚠️  COMPLIANCE VIOLATION
        ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        Log ID: {audit.log_id}
        Action: {audit.action.value}
        User ID: {audit.user_id}
        Resource: {audit.resource}

        Violations:
        """)

        for v in violations:
            print(f"   • {v['rule']}: {v['description']}")
            print(f"     → {v['action_required']}")

        print(" ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n")

        # 실제로는 알림 발송, 티켓 생성 등
        # send_compliance_alert(audit, violations)
        # create_review_ticket(audit, violations)

    def callback(self, ch, method, properties, body):
        """메시지 처리 콜백"""
        audit = AuditLog.from_json(body.decode())

        # 컴플라이언스 검사
        violations = self.check_compliance(audit)

        if violations:
            self.handle_violation(audit, violations)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """Compliance Checker 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # CRITICAL 로그만 검사
        channel.basic_consume(
            queue='audit.critical',
            on_message_callback=self.callback
        )

        print(' [*] Compliance Checker started')
        print(' [*] Monitoring critical audit logs...\n')

        channel.start_consuming()

if __name__ == '__main__':
    checker = ComplianceChecker()

    try:
        checker.start()
    except KeyboardInterrupt:
        print('\nCompliance Checker stopped')

Audit Search​

# audit_search.py
import sqlite3
from datetime import datetime, timedelta

class AuditSearch:
    """감사 로그 검색"""

    def __init__(self):
        self.db = sqlite3.connect('audit_logs.db')

    def search_by_user(self, user_id: int, limit: int = 100):
        """사용자별 검색"""
        cursor = self.db.cursor()

        cursor.execute('''
            SELECT * FROM audit_logs
            WHERE user_id = ?
            ORDER BY timestamp DESC
            LIMIT ?
        ''', (user_id, limit))

        return cursor.fetchall()

    def search_by_action(self, action: str, limit: int = 100):
        """액션별 검색"""
        cursor = self.db.cursor()

        cursor.execute('''
            SELECT * FROM audit_logs
            WHERE action = ?
            ORDER BY timestamp DESC
            LIMIT ?
        ''', (action, limit))

        return cursor.fetchall()

    def search_by_timerange(
        self,
        start_time: str,
        end_time: str,
        limit: int = 1000
    ):
        """시간 범위 검색"""
        cursor = self.db.cursor()

        cursor.execute('''
            SELECT * FROM audit_logs
            WHERE timestamp BETWEEN ? AND ?
            ORDER BY timestamp DESC
            LIMIT ?
        ''', (start_time, end_time, limit))

        return cursor.fetchall()

    def search_critical(self, days: int = 7):
        """최근 중요 로그 검색"""
        start_time = (datetime.now() - timedelta(days=days)).isoformat()

        cursor = self.db.cursor()

        cursor.execute('''
            SELECT * FROM audit_logs
            WHERE severity = 'critical'
            AND timestamp >= ?
            ORDER BY timestamp DESC
        ''', (start_time,))

        return cursor.fetchall()

# 사용 예
if __name__ == '__main__':
    search = AuditSearch()

    # 사용자별 검색
    print(" User 1001 Audit Logs:")
    logs = search.search_by_user(1001, limit=10)

    for log in logs:
        print(f"   {log[0]}: {log[2]} @ {log[1]}")

    # 중요 로그 검색
    print("\n Critical Logs (last 7 days):")
    critical_logs = search.search_critical(days=7)

    for log in critical_logs:
        print(f"   {log[0]}: {log[2]} (user: {log[4]})")
