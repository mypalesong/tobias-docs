시나리오 3: 이벤트 기반 마이크로서비스

마이크로서비스 간 이벤트 기반 통신을 구현해봅시다!

📌 상황 설명​

전자상거래 플랫폼의 주문 처리 과정:

  • Order Service: 주문 생성
  • Payment Service: 결제 처리
  • Inventory Service: 재고 차감
  • Shipping Service: 배송 준비
  • Notification Service: 알림 발송

각 서비스는 독립적으로 배포되며, 이벤트를 통해 통신합니다.

🏗️ 아키텍처​

Order Service → [Topic Exchange: events]
                       ↓
         order.created.# ──→ Payment Service
         order.paid.#    ──→ Inventory Service
         order.shipped.# ──→ Notification Service
         #               ──→ Audit Service (모든 이벤트)

패턴: Event-Driven Architecture (Topic Exchange)

이벤트 모델​

# events.py
from dataclasses import dataclass, asdict
from datetime import datetime
from typing import Dict, Any, List
from enum import Enum
import json

class EventType(str, Enum):
    """이벤트 타입"""
    ORDER_CREATED = 'order.created'
    ORDER_PAID = 'order.paid'
    ORDER_FAILED = 'order.failed'
    ORDER_SHIPPED = 'order.shipped'
    INVENTORY_RESERVED = 'inventory.reserved'
    INVENTORY_RELEASED = 'inventory.released'

@dataclass
class Event:
    """기본 이벤트"""
    event_id: str
    event_type: str
    timestamp: str
    source_service: str
    data: Dict[str, Any]
    correlation_id: str = None  # 추적용

    def __post_init__(self):
        if self.timestamp is None:
            self.timestamp = datetime.now().isoformat()

    def to_json(self) -> str:
        return json.dumps(asdict(self))

    @classmethod
    def from_json(cls, json_str: str):
        return cls(**json.loads(json_str))

@dataclass
class OrderCreatedEvent(Event):
    """주문 생성 이벤트"""

    @staticmethod
    def create(order_id: str, user_id: int, items: List[Dict], total: float):
        return Event(
            event_id=f'evt_{datetime.now().timestamp()}',
            event_type=EventType.ORDER_CREATED,
            timestamp=datetime.now().isoformat(),
            source_service='order-service',
            data={
                'order_id': order_id,
                'user_id': user_id,
                'items': items,
                'total': total
            }
        )

Event Bus​

# event_bus.py
import pika
from events import Event

class EventBus:
    """이벤트 버스"""

    def __init__(self, host='localhost'):
        self.host = host
        self.exchange = 'events'

    def publish(self, event: Event):
        """이벤트 발행"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # Topic Exchange 선언
        channel.exchange_declare(
            exchange=self.exchange,
            exchange_type='topic',
            durable=True
        )

        # Routing Key = event_type (예: order.created)
        channel.basic_publish(
            exchange=self.exchange,
            routing_key=event.event_type,
            body=event.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,
                content_type='application/json',
                correlation_id=event.correlation_id
            )
        )

        print(f" [→] Event published: {event.event_type} "
              f"(id: {event.event_id})")

        connection.close()

    def subscribe(
        self,
        service_name: str,
        patterns: list,
        callback
    ):
        """이벤트 구독"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.exchange_declare(
            exchange=self.exchange,
            exchange_type='topic',
            durable=True
        )

        # 서비스별 큐 생성
        queue_name = f'{service_name}_queue'
        channel.queue_declare(queue=queue_name, durable=True)

        # 패턴별로 바인딩
        for pattern in patterns:
            channel.queue_bind(
                exchange=self.exchange,
                queue=queue_name,
                routing_key=pattern
            )

            print(f" [*] {service_name} subscribed to: {pattern}")

        channel.basic_qos(prefetch_count=1)

        channel.basic_consume(
            queue=queue_name,
            on_message_callback=callback
        )

        print(f' [*] {service_name} waiting for events...')
        channel.start_consuming()

Order Service​

# order_service.py
from event_bus import EventBus
from events import OrderCreatedEvent, Event, EventType
import uuid

class OrderService:
    """주문 서비스"""

    def __init__(self):
        self.event_bus = EventBus()

    def create_order(self, user_id: int, items: list):
        """주문 생성"""
        order_id = f'ORD-{uuid.uuid4().hex[:8].upper()}'

        # 주문 총액 계산
        total = sum(item['price'] * item['quantity'] for item in items)

        print(f" [ORDER] Creating order: {order_id}")
        print(f"   User: {user_id}")
        print(f"   Items: {len(items)}")
        print(f"   Total: ${total:.2f}")

        # 이벤트 발행
        event = OrderCreatedEvent.create(
            order_id=order_id,
            user_id=user_id,
            items=items,
            total=total
        )

        self.event_bus.publish(event)

        print(f" [ORDER] Order created successfully")

        return order_id

# 사용 예
if __name__ == '__main__':
    service = OrderService()

    # 주문 생성
    order_id = service.create_order(
        user_id=1001,
        items=[
            {'product_id': 'P001', 'name': 'Laptop', 'price': 1200.00, 'quantity': 1},
            {'product_id': 'P002', 'name': 'Mouse', 'price': 25.00, 'quantity': 2},
        ]
    )

Payment Service​

# payment_service.py
import pika
from event_bus import EventBus
from events import Event, EventType
import time
import random

class PaymentService:
    """결제 서비스"""

    def __init__(self):
        self.event_bus = EventBus()

    def process_payment(self, event: Event):
        """결제 처리"""
        order_id = event.data['order_id']
        total = event.data['total']

        print(f" [PAYMENT] Processing payment for {order_id}")
        print(f"   Amount: ${total:.2f}")

        # 결제 처리 (시뮬레이션)
        time.sleep(1)

        # 80% 확률로 성공
        success = random.random() < 0.8

        if success:
            print(f" [PAYMENT] ✓ Payment successful")

            # 결제 완료 이벤트 발행
            paid_event = Event(
                event_id=f'evt_{time.time()}',
                event_type=EventType.ORDER_PAID,
                timestamp=time.time(),
                source_service='payment-service',
                data={
                    'order_id': order_id,
                    'amount': total,
                    'payment_id': f'PAY-{order_id}'
                },
                correlation_id=event.correlation_id
            )

            self.event_bus.publish(paid_event)

        else:
            print(f" [PAYMENT] ✗ Payment failed")

            # 결제 실패 이벤트 발행
            failed_event = Event(
                event_id=f'evt_{time.time()}',
                event_type=EventType.ORDER_FAILED,
                timestamp=time.time(),
                source_service='payment-service',
                data={
                    'order_id': order_id,
                    'reason': 'Payment declined'
                },
                correlation_id=event.correlation_id
            )

            self.event_bus.publish(failed_event)

    def callback(self, ch, method, properties, body):
        """이벤트 콜백"""
        event = Event.from_json(body.decode())

        if event.event_type == EventType.ORDER_CREATED:
            self.process_payment(event)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """서비스 시작"""
        self.event_bus.subscribe(
            service_name='payment-service',
            patterns=['order.created'],
            callback=self.callback
        )

if __name__ == '__main__':
    service = PaymentService()

    try:
        service.start()
    except KeyboardInterrupt:
        print('\nPayment Service stopped')

Inventory Service​

# inventory_service.py
import pika
from event_bus import EventBus
from events import Event, EventType
import time

class InventoryService:
    """재고 서비스"""

    def __init__(self):
        self.event_bus = EventBus()
        self.inventory = {
            'P001': 50,
            'P002': 100,
            'P003': 30
        }

    def reserve_inventory(self, event: Event):
        """재고 예약"""
        order_id = event.data['order_id']
        items = event.data['items']

        print(f" [INVENTORY] Reserving items for {order_id}")

        # 재고 확인 및 차감
        for item in items:
            product_id = item['product_id']
            quantity = item['quantity']

            if product_id in self.inventory:
                if self.inventory[product_id] >= quantity:
                    self.inventory[product_id] -= quantity
                    print(f"   ✓ Reserved {quantity}x {product_id} "
                          f"(remaining: {self.inventory[product_id]})")
                else:
                    print(f"   ✗ Insufficient stock for {product_id}")
            else:
                print(f"   ✗ Product {product_id} not found")

        # 재고 예약 이벤트 발행
        reserved_event = Event(
            event_id=f'evt_{time.time()}',
            event_type=EventType.INVENTORY_RESERVED,
            timestamp=time.time(),
            source_service='inventory-service',
            data={
                'order_id': order_id,
                'items': items
            },
            correlation_id=event.correlation_id
        )

        self.event_bus.publish(reserved_event)

    def callback(self, ch, method, properties, body):
        """이벤트 콜백"""
        event = Event.from_json(body.decode())

        if event.event_type == EventType.ORDER_PAID:
            self.reserve_inventory(event)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """서비스 시작"""
        self.event_bus.subscribe(
            service_name='inventory-service',
            patterns=['order.paid'],
            callback=self.callback
        )

if __name__ == '__main__':
    service = InventoryService()

    try:
        service.start()
    except KeyboardInterrupt:
        print('\nInventory Service stopped')

Notification Service​

# notification_service.py
import pika
from event_bus import EventBus
from events import Event, EventType

class NotificationService:
    """알림 서비스"""

    def __init__(self):
        self.event_bus = EventBus()

    def send_notification(self, event: Event):
        """알림 발송"""
        order_id = event.data.get('order_id')
        event_type = event.event_type

        print(f" [NOTIFICATION] Sending notification")
        print(f"   Order: {order_id}")
        print(f"   Event: {event_type}")

        # 이벤트별 알림 메시지
        messages = {
            EventType.ORDER_CREATED: f"주문이 생성되었습니다: {order_id}",
            EventType.ORDER_PAID: f"결제가 완료되었습니다: {order_id}",
            EventType.ORDER_FAILED: f"주문이 실패했습니다: {order_id}",
            EventType.ORDER_SHIPPED: f"상품이 발송되었습니다: {order_id}",
        }

        message = messages.get(event_type, "주문 상태가 업데이트되었습니다")

        print(f"   📧 Email: {message}")
        print(f"   📱 Push: {message}")
        print()

    def callback(self, ch, method, properties, body):
        """이벤트 콜백"""
        event = Event.from_json(body.decode())

        self.send_notification(event)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """서비스 시작"""
        # 모든 주문 관련 이벤트 구독
        self.event_bus.subscribe(
            service_name='notification-service',
            patterns=['order.#'],
            callback=self.callback
        )

if __name__ == '__main__':
    service = NotificationService()

    try:
        service.start()
    except KeyboardInterrupt:
        print('\nNotification Service stopped')

Audit Service​

# audit_service.py
import pika
from event_bus import EventBus
from events import Event
import json

class AuditService:
    """감사 서비스"""

    def __init__(self):
        self.event_bus = EventBus()

    def log_event(self, event: Event):
        """이벤트 로깅"""
        # 파일에 기록
        with open('audit.log', 'a', encoding='utf-8') as f:
            log_entry = {
                'event_id': event.event_id,
                'event_type': event.event_type,
                'timestamp': event.timestamp,
                'source': event.source_service,
                'data': event.data
            }
            f.write(json.dumps(log_entry, ensure_ascii=False) + '\n')

        print(f" [AUDIT] Logged: {event.event_type} from {event.source_service}")

    def callback(self, ch, method, properties, body):
        """이벤트 콜백"""
        event = Event.from_json(body.decode())

        self.log_event(event)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def start(self):
        """서비스 시작"""
        # 모든 이벤트 구독
        self.event_bus.subscribe(
            service_name='audit-service',
            patterns=['#'],
            callback=self.callback
        )

if __name__ == '__main__':
    service = AuditService()

    try:
        service.start()
    except KeyboardInterrupt:
        print('\nAudit Service stopped')

# test_event_driven.py
import pytest
import time
from order_service import OrderService
from event_bus import EventBus
from events import Event

class TestEventDriven:
    """이벤트 기반 아키텍처 테스트"""

    @pytest.fixture
    def event_bus(self):
        return EventBus()

    def test_order_flow(self, event_bus):
        """전체 주문 흐름 테스트"""
        received_events = []

        def callback(ch, method, properties, body):
            event = Event.from_json(body.decode())
            received_events.append(event.event_type)
            ch.basic_ack(delivery_tag=method.delivery_tag)

        # 이벤트 구독
        import threading
        t = threading.Thread(
            target=event_bus.subscribe,
            args=('test-service', ['#'], callback)
        )
        t.daemon = True
        t.start()

        time.sleep(1)

        # 주문 생성
        order_service = OrderService()
        order_service.create_order(
            user_id=1001,
            items=[{'product_id': 'P001', 'price': 100, 'quantity': 1}]
        )

        time.sleep(2)

        # order.created 이벤트 확인
        assert 'order.created' in received_events
