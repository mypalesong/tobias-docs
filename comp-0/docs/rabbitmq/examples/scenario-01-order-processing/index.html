시나리오 1: 주문 처리 시스템

쇼핑몰의 주문을 비동기로 안정적으로 처리하는 시스템을 만들어봅시다!

📌 상황 설명​

온라인 쇼핑몰 '마켓플레이스'는 주문이 급증하면서 문제가 발생했습니다:

  • 주문 처리에 5-10초 소요 (결제, 재고 확인, 배송 준비)
  • 동시 주문 시 서버 응답 지연
  • 일부 주문이 처리 중 실패해도 재시도 안 됨

🎯 요구사항​

  1. 주문 요청을 즉시 받고 백그라운드 처리
  2. 여러 워커가 병렬로 주문 처리
  3. 처리 실패 시 자동 재시도
  4. 모든 주문이 최종적으로 처리되도록 보장

🏗️ 아키텍처​

사용자 요청
    ↓
API 서버 (Producer) → [Order Queue] → Worker 1 (Consumer)
                                    → Worker 2 (Consumer)
                                    → Worker 3 (Consumer)

패턴: Work Queue (작업 분산)

📝 구현​

1. 주문 모델​

# models.py
from dataclasses import dataclass, asdict
from datetime import datetime
from typing import List, Dict
import json

@dataclass
class OrderItem:
    """주문 상품"""
    product_id: int
    product_name: str
    quantity: int
    price: float

@dataclass
class Order:
    """주문"""
    order_id: str
    user_id: int
    items: List[OrderItem]
    total_amount: float
    created_at: str = None

    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now().isoformat()

    def to_json(self) -> str:
        """JSON 문자열로 변환"""
        data = asdict(self)
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str) -> 'Order':
        """JSON에서 객체 생성"""
        data = json.loads(json_str)
        # items를 OrderItem 객체로 변환
        data['items'] = [OrderItem(**item) for item in data['items']]
        return cls(**data)

2. Producer (API 서버)​

# producer.py
import pika
import uuid
from models import Order, OrderItem
from typing import List
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class OrderProducer:
    """주문 생산자"""

    def __init__(self, host='localhost'):
        self.host = host
        self.queue_name = 'orders'

    def publish_order(self, user_id: int, items: List[Dict]):
        """주문 발행"""
        # 주문 객체 생성
        order_id = str(uuid.uuid4())
        order_items = [OrderItem(**item) for item in items]
        total = sum(item.price * item.quantity for item in order_items)

        order = Order(
            order_id=order_id,
            user_id=user_id,
            items=order_items,
            total_amount=total
        )

        # RabbitMQ 연결
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # 큐 선언 (durable=True로 영속성 보장)
        channel.queue_declare(queue=self.queue_name, durable=True)

        # 메시지 발행
        channel.basic_publish(
            exchange='',
            routing_key=self.queue_name,
            body=order.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,  # 메시지 영속화
                content_type='application/json',
                message_id=order_id
            )
        )

        logger.info(f"📦 주문 발행: {order_id}, 총액: ${total:.2f}")
        connection.close()

        return order_id

def main():
    """테스트용 메인"""
    producer = OrderProducer()

    # 샘플 주문 발행
    items = [
        {'product_id': 1, 'product_name': '무선 마우스', 'quantity': 2, 'price': 29.99},
        {'product_id': 2, 'product_name': '키보드', 'quantity': 1, 'price': 89.99},
    ]

    order_id = producer.publish_order(user_id=123, items=items)
    print(f"✅ 주문 접수: {order_id}")

if __name__ == '__main__':
    main()

3. Consumer (워커)​

# consumer.py
import pika
import time
import random
import logging
from models import Order

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class OrderWorker:
    """주문 처리 워커"""

    def __init__(self, worker_id: int, host='localhost'):
        self.worker_id = worker_id
        self.host = host
        self.queue_name = 'orders'

    def process_order(self, order: Order) -> bool:
        """주문 처리 (비즈니스 로직)"""
        try:
            logger.info(
                f"🔨 Worker {self.worker_id} - "
                f"주문 처리 시작: {order.order_id}"
            )

            # 1. 결제 처리 (시뮬레이션)
            time.sleep(random.uniform(1, 3))
            logger.info(f"💳 결제 완료: ${order.total_amount:.2f}")

            # 2. 재고 확인
            time.sleep(random.uniform(0.5, 1))
            logger.info(f"📦 재고 확인 완료")

            # 3. 배송 준비
            time.sleep(random.uniform(0.5, 1))
            logger.info(f"🚚 배송 준비 완료")

            # 10% 확률로 실패 시뮬레이션
            if random.random() < 0.1:
                raise Exception("결제 게이트웨이 오류")

            logger.info(
                f"✅ Worker {self.worker_id} - "
                f"주문 처리 완료: {order.order_id}"
            )
            return True

        except Exception as e:
            logger.error(
                f"❌ Worker {self.worker_id} - "
                f"주문 처리 실패: {order.order_id} - {e}"
            )
            return False

    def callback(self, ch, method, properties, body):
        """메시지 수신 콜백"""
        order = Order.from_json(body.decode())

        # 주문 처리
        success = self.process_order(order)

        if success:
            # 성공 시 ACK
            ch.basic_ack(delivery_tag=method.delivery_tag)
        else:
            # 실패 시 NACK (재시도)
            ch.basic_nack(
                delivery_tag=method.delivery_tag,
                requeue=True  # 큐에 다시 넣기
            )

    def start(self):
        """워커 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.queue_declare(queue=self.queue_name, durable=True)

        # Prefetch: 한 번에 1개씩만 처리
        channel.basic_qos(prefetch_count=1)

        channel.basic_consume(
            queue=self.queue_name,
            on_message_callback=self.callback
        )

        logger.info(
            f"🎬 Worker {self.worker_id} 시작 - "
            f"주문 대기 중..."
        )
        channel.start_consuming()

def main():
    """워커 실행"""
    import sys

    worker_id = int(sys.argv[1]) if len(sys.argv) > 1 else 1
    worker = OrderWorker(worker_id=worker_id)

    try:
        worker.start()
    except KeyboardInterrupt:
        print(f"\n워커 {worker_id} 종료")

if __name__ == '__main__':
    main()

4. 테스트 코드​

# test_order_system.py
import pytest
import pika
import json
import time
from threading import Thread
from models import Order, OrderItem
from producer import OrderProducer
from consumer import OrderWorker

class TestOrderSystem:
    """주문 시스템 테스트"""

    @pytest.fixture
    def clean_queue(self):
        """테스트 전 큐 정리"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        # 테스트 큐 생성 및 정리
        channel.queue_delete(queue='test_orders')
        channel.queue_declare(queue='test_orders', durable=True)

        yield

        # 테스트 후 정리
        channel.queue_delete(queue='test_orders')
        connection.close()

    def test_publish_order(self, clean_queue):
        """주문 발행 테스트"""
        producer = OrderProducer()
        producer.queue_name = 'test_orders'

        items = [
            {'product_id': 1, 'product_name': '테스트 상품',
             'quantity': 1, 'price': 10.0}
        ]

        order_id = producer.publish_order(user_id=1, items=items)

        # 큐에 메시지가 있는지 확인
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        method, properties, body = channel.basic_get(
            queue='test_orders',
            auto_ack=True
        )

        assert method is not None
        order = Order.from_json(body.decode())
        assert order.order_id == order_id

        connection.close()

    def test_worker_processing(self, clean_queue):
        """워커 처리 테스트"""
        # 주문 발행
        producer = OrderProducer()
        producer.queue_name = 'test_orders'

        items = [
            {'product_id': 1, 'product_name': '테스트',
             'quantity': 1, 'price': 10.0}
        ]

        order_id = producer.publish_order(user_id=1, items=items)

        # 워커로 처리
        processed = []

        def mock_process(order):
            processed.append(order.order_id)
            return True

        worker = OrderWorker(worker_id=999)
        worker.queue_name = 'test_orders'
        worker.process_order = mock_process

        # 워커를 별도 스레드에서 실행
        worker_thread = Thread(target=worker.start)
        worker_thread.daemon = True
        worker_thread.start()

        # 처리 대기
        time.sleep(2)

        assert order_id in processed

    def test_multiple_workers(self, clean_queue):
        """다중 워커 테스트"""
        producer = OrderProducer()
        producer.queue_name = 'test_orders'

        # 10개 주문 발행
        for i in range(10):
            items = [
                {'product_id': i, 'product_name': f'상품{i}',
                 'quantity': 1, 'price': 10.0}
            ]
            producer.publish_order(user_id=i, items=items)

        # 워커 3개 실행 테스트 시뮬레이션
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost')
        )
        channel = connection.channel()

        # 큐에 10개 메시지 확인
        queue_state = channel.queue_declare(
            queue='test_orders',
            passive=True
        )
        assert queue_state.method.message_count == 10

        connection.close()

💡 개선 사항​

1. Dead Letter Queue 추가​

# DLQ 설정
channel.queue_declare(
    queue='orders_dlq',
    durable=True
)

channel.queue_declare(
    queue='orders',
    durable=True,
    arguments={
        'x-dead-letter-exchange': '',
        'x-dead-letter-routing-key': 'orders_dlq',
        'x-message-ttl': 60000,  # 1분
        'x-max-retries': 3
    }
)

2. 우선순위 큐​

# VIP 고객 주문 우선 처리
channel.queue_declare(
    queue='orders',
    durable=True,
    arguments={'x-max-priority': 10}
)

# 발행 시 우선순위 설정
channel.basic_publish(
    exchange='',
    routing_key='orders',
    body=order.to_json(),
    properties=pika.BasicProperties(
        priority=10 if order.is_vip else 5
    )
)

🎓 배운 내용​

  • Work Queue 패턴: 작업을 여러 워커에게 분산
  • Durable Queue: 큐 영속화로 RabbitMQ 재시작 시에도 보존
  • Message Persistence: 메시지 영속화 (delivery_mode=2)
  • Prefetch: 워커당 처리할 메시지 수 제한
  • ACK/NACK: 수동 확인으로 신뢰성 보장

🔗 관련 문서​

  • Work Queue
  • Message Durability
