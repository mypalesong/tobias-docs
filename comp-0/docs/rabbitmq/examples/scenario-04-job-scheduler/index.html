import pika

def setup_job_queues():
    """ì‘ì—… í ì„¤ì •"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 1. Dead Letter Exchange
    channel.exchange_declare(
        exchange='jobs.dlx',
        exchange_type='direct',
        durable=True
    )

    # 2. Failed Jobs Queue
    channel.queue_declare(
        queue='jobs.failed',
        durable=True
    )

    channel.queue_bind(
        exchange='jobs.dlx',
        queue='jobs.failed',
        routing_key='failed'
    )

    # 3. Retry Queue (5ì´ˆ í›„ ì¬ì‹œë„)
    channel.queue_declare(
        queue='jobs.retry',
        durable=True,
        arguments={
            'x-dead-letter-exchange': '',
            'x-dead-letter-routing-key': 'jobs.main',
            'x-message-ttl': 5000  # 5ì´ˆ
        }
    )

    # 4. Main Job Queue (ìš°ì„ ìˆœìœ„ + DLX)
    channel.queue_declare(
        queue='jobs.main',
        durable=True,
        arguments={
            'x-max-priority': 10,  # ìš°ì„ ìˆœìœ„ ì§€ì›
            'x-dead-letter-exchange': 'jobs.dlx',
            'x-dead-letter-routing-key': 'failed'
        }
    )

    print(" [âœ“] Job queues configured:")
    print("     - jobs.main (priority queue)")
    print("     - jobs.retry (5s TTL)")
    print("     - jobs.failed (DLQ)")

    connection.close()

if __name__ == '__main__':

import pika
import time
import random
from models import Job, JobType, JobStatus
from datetime import datetime

class Worker:
    """ì‘ì—… ì›Œì»¤"""

    def __init__(self, worker_id: int, host='localhost'):
        self.worker_id = worker_id
        self.host = host
        self.connection = None
        self.channel = None

    def execute_crawling(self, params: dict):
        """í¬ë¡¤ë§ ì‹¤í–‰"""
        url = params['url']
        print(f"   [Worker {self.worker_id}] Crawling: {url}")
        time.sleep(2)  # ì‹œë®¬ë ˆì´ì…˜

        # 10% í™•ë¥ ë¡œ ì‹¤íŒ¨
        if random.random() < 0.1:
            raise Exception(f"Failed to crawl {url}")

        print(f"   [Worker {self.worker_id}] âœ“ Crawling completed")

    def execute_analysis(self, params: dict):
        """ë¶„ì„ ì‹¤í–‰"""
        data_id = params['data_id']
        print(f"   [Worker {self.worker_id}] Analyzing: {data_id}")
        time.sleep(3)  # ì‹œë®¬ë ˆì´ì…˜

        if random.random() < 0.1:
            raise Exception(f"Analysis failed for {data_id}")

        print(f"   [Worker {self.worker_id}] âœ“ Analysis completed")

    def execute_report(self, params: dict):
        """ë¦¬í¬íŠ¸ ìƒì„±"""
        user_id = params['user_id']
        print(f"   [Worker {self.worker_id}] Generating report for user {user_id}")
        time.sleep(1)  # ì‹œë®¬ë ˆì´ì…˜

        print(f"   [Worker {self.worker_id}] âœ“ Report generated")

    def execute_cleanup(self, params: dict):
        """ì •ë¦¬ ì‘ì—…"""
        print(f"   [Worker {self.worker_id}] Running cleanup")
        time.sleep(0.5)  # ì‹œë®¬ë ˆì´ì…˜

        print(f"   [Worker {self.worker_id}] âœ“ Cleanup completed")

    def execute_job(self, job: Job):
        """ì‘ì—… ì‹¤í–‰"""
        print(f"\n [Worker {self.worker_id}] Processing job: {job.job_id}")
        print(f"   Type: {job.job_type.value}")
        print(f"   Priority: {job.priority.value}")

        # ì‘ì—… ì‹œì‘
        job.status = JobStatus.RUNNING
        job.started_at = datetime.now().isoformat()

        # ì‘ì—… íƒ€ì…ë³„ ì‹¤í–‰
        try:
            if job.job_type == JobType.CRAWLING:
                self.execute_crawling(job.params)
            elif job.job_type == JobType.ANALYSIS:
                self.execute_analysis(job.params)
            elif job.job_type == JobType.REPORT:
                self.execute_report(job.params)
            elif job.job_type == JobType.CLEANUP:
                self.execute_cleanup(job.params)

            # ì„±ê³µ
            job.status = JobStatus.COMPLETED
            job.completed_at = datetime.now().isoformat()

            return True

        except Exception as e:
            # ì‹¤íŒ¨
            job.status = JobStatus.FAILED
            job.error_message = str(e)

            print(f"   [Worker {self.worker_id}] âœ— Job failed: {e}")

            return False

    def send_to_retry(self, job: Job):
        """ì¬ì‹œë„ íë¡œ ì „ì†¡"""
        job.retry_count += 1

        print(f"   [Worker {self.worker_id}] â†’ Sending to retry queue "
              f"(attempt {job.retry_count}/{job.max_retries})")

        self.channel.basic_publish(
            exchange='',
            routing_key='jobs.retry',
            body=job.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,
                content_type='application/json'
            )
        )

    def callback(self, ch, method, properties, body):
        """ë©”ì‹œì§€ ì²˜ë¦¬ ì½œë°±"""
        job = Job.from_json(body.decode())

        # ì‘ì—… ì‹¤í–‰
        success = self.execute_job(job)

        if success:
            # ì„±ê³µ: ACK
            ch.basic_ack(delivery_tag=method.delivery_tag)

        else:
            # ì‹¤íŒ¨: ì¬ì‹œë„ í™•ì¸
            if job.retry_count < job.max_retries:
                # ì¬ì‹œë„ íë¡œ ì „ì†¡
                self.send_to_retry(job)

                # ì›ë³¸ ë©”ì‹œì§€ ACK
                ch.basic_ack(delivery_tag=method.delivery_tag)

            else:
                # ìµœëŒ€ ì¬ì‹œë„ ì´ˆê³¼: NACK â†’ DLXë¡œ ì´ë™
                print(f"   [Worker {self.worker_id}] âœ— Max retries exceeded")

                ch.basic_nack(
                    delivery_tag=method.delivery_tag,
                    requeue=False
                )

    def start(self):
        """ì›Œì»¤ ì‹œì‘"""
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        self.channel = self.connection.channel()

        # Fair dispatch (prefetch_count=1)
        self.channel.basic_qos(prefetch_count=1)

        self.channel.basic_consume(
            queue='jobs.main',
            on_message_callback=self.callback
        )

        print(f' [*] Worker {self.worker_id} started')
        print(f' [*] Waiting for jobs...\n')

        self.channel.start_consuming()

if __name__ == '__main__':

    except KeyboardInterrupt:

import requests
from requests.auth import HTTPBasicAuth
import time

class JobMonitor:
    """ì‘ì—… ëª¨ë‹ˆí„°"""

    def __init__(self, host='localhost', user='admin', password='admin123'):
        self.base_url = f'http://{host}:15672/api'
        self.auth = HTTPBasicAuth(user, password)

    def get_queue_stats(self):
        """í ìƒíƒœ ì¡°íšŒ"""
        response = requests.get(
            f'{self.base_url}/queues',
            auth=self.auth
        )

        queues = response.json()

        stats = {}
        for queue in queues:
            if queue['name'].startswith('jobs.'):
                stats[queue['name']] = {
                    'messages': queue.get('messages', 0),
                    'ready': queue.get('messages_ready', 0),
                    'unacked': queue.get('messages_unacknowledged', 0),
                    'consumers': queue.get('consumers', 0)
                }

        return stats

    def print_stats(self):
        """í†µê³„ ì¶œë ¥"""
        stats = self.get_queue_stats()

        print(" Job Queue Statistics")
        print(" " + "="*60)

        for queue_name, data in stats.items():
            print(f"\n {queue_name}:")
            print(f"   Total: {data['messages']:4d} | "
                  f"Ready: {data['ready']:4d} | "
                  f"Unacked: {data['unacked']:4d} | "
                  f"Consumers: {data['consumers']:2d}")

    def monitor(self, interval: int = 5):
        """ì‹¤ì‹œê°„ ëª¨ë‹ˆí„°ë§"""
        print(f" Job Monitor (refresh every {interval}s)")
        print(" Press Ctrl+C to stop\n")

        try:
            while True:
                self.print_stats()
                print()
                time.sleep(interval)

        except KeyboardInterrupt:
            print("\n Monitoring stopped")

# ì‚¬ìš© ì˜ˆ
if __name__ == '__main__':
    monitor = JobMonitor()
    monitor.monitor(interval=3)
