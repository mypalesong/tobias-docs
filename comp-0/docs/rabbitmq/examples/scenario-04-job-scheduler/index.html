시나리오 4: 분산 작업 스케줄러

대규모 백그라운드 작업을 분산 처리하는 시스템을 구현해봅시다!

📌 상황 설명​

데이터 분석 플랫폼의 작업 스케줄링:

  • 작업 타입:
    • 데이터 수집 (Crawling)
    • 데이터 분석 (Analysis)
    • 리포트 생성 (Report)
    • 데이터 정리 (Cleanup)
  • 요구사항:
    • 우선순위 기반 처리
    • 재시도 메커니즘
    • 작업 상태 추적
    • 워커 부하 분산

🏗️ 아키텍처​

Job Scheduler → [Priority Queue] → Worker Pool (N workers)
                       ↓
                  [DLX] → Failed Jobs → Manual Review

핵심 기능:

  • Priority Queue로 긴급 작업 우선 처리
  • Dead Letter Exchange로 실패 작업 처리
  • Worker Pool로 병렬 처리

📝 구현​

작업 모델​

# models.py
from dataclasses import dataclass, asdict, field
from enum import Enum
from typing import Dict, Any, Optional
from datetime import datetime
import json

class JobType(str, Enum):
    """작업 타입"""
    CRAWLING = 'crawling'
    ANALYSIS = 'analysis'
    REPORT = 'report'
    CLEANUP = 'cleanup'

class JobPriority(int, Enum):
    """작업 우선순위"""
    CRITICAL = 10
    HIGH = 7
    NORMAL = 5
    LOW = 1

class JobStatus(str, Enum):
    """작업 상태"""
    PENDING = 'pending'
    RUNNING = 'running'
    COMPLETED = 'completed'
    FAILED = 'failed'
    RETRY = 'retry'

@dataclass
class Job:
    """작업"""
    job_id: str
    job_type: JobType
    priority: JobPriority
    params: Dict[str, Any]
    status: JobStatus = JobStatus.PENDING
    retry_count: int = 0
    max_retries: int = 3
    created_at: str = field(default_factory=lambda: datetime.now().isoformat())
    started_at: Optional[str] = None
    completed_at: Optional[str] = None
    error_message: Optional[str] = None

    def to_json(self) -> str:
        data = asdict(self)
        # Enum을 문자열로 변환
        data['job_type'] = self.job_type.value
        data['priority'] = self.priority.value
        data['status'] = self.status.value
        return json.dumps(data)

    @classmethod
    def from_json(cls, json_str: str):
        data = json.loads(json_str)
        # 문자열을 Enum으로 변환
        data['job_type'] = JobType(data['job_type'])
        data['priority'] = JobPriority(data['priority'])
        data['status'] = JobStatus(data['status'])
        return cls(**data)

Queue 설정​

# queue_setup.py
import pika

def setup_job_queues():
    """작업 큐 설정"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 1. Dead Letter Exchange
    channel.exchange_declare(
        exchange='jobs.dlx',
        exchange_type='direct',
        durable=True
    )

    # 2. Failed Jobs Queue
    channel.queue_declare(
        queue='jobs.failed',
        durable=True
    )

    channel.queue_bind(
        exchange='jobs.dlx',
        queue='jobs.failed',
        routing_key='failed'
    )

    # 3. Retry Queue (5초 후 재시도)
    channel.queue_declare(
        queue='jobs.retry',
        durable=True,
        arguments={
            'x-dead-letter-exchange': '',
            'x-dead-letter-routing-key': 'jobs.main',
            'x-message-ttl': 5000  # 5초
        }
    )

    # 4. Main Job Queue (우선순위 + DLX)
    channel.queue_declare(
        queue='jobs.main',
        durable=True,
        arguments={
            'x-max-priority': 10,  # 우선순위 지원
            'x-dead-letter-exchange': 'jobs.dlx',
            'x-dead-letter-routing-key': 'failed'
        }
    )

    print(" [✓] Job queues configured:")
    print("     - jobs.main (priority queue)")
    print("     - jobs.retry (5s TTL)")
    print("     - jobs.failed (DLQ)")

    connection.close()

if __name__ == '__main__':
    setup_job_queues()

Job Scheduler​

# job_scheduler.py
import pika
from models import Job, JobType, JobPriority
import uuid

class JobScheduler:
    """작업 스케줄러"""

    def __init__(self, host='localhost'):
        self.host = host

    def schedule(self, job: Job):
        """작업 스케줄링"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        # 우선순위와 함께 발행
        channel.basic_publish(
            exchange='',
            routing_key='jobs.main',
            body=job.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,
                priority=job.priority.value,
                content_type='application/json'
            )
        )

        print(f" [→] Scheduled job: {job.job_id} "
              f"(type={job.job_type.value}, priority={job.priority.value})")

        connection.close()

    def schedule_crawling(self, url: str, priority: JobPriority = JobPriority.NORMAL):
        """크롤링 작업 스케줄링"""
        job = Job(
            job_id=f'CRAWL-{uuid.uuid4().hex[:8].upper()}',
            job_type=JobType.CRAWLING,
            priority=priority,
            params={'url': url}
        )

        self.schedule(job)
        return job.job_id

    def schedule_analysis(self, data_id: str, priority: JobPriority = JobPriority.NORMAL):
        """분석 작업 스케줄링"""
        job = Job(
            job_id=f'ANALYZE-{uuid.uuid4().hex[:8].upper()}',
            job_type=JobType.ANALYSIS,
            priority=priority,
            params={'data_id': data_id}
        )

        self.schedule(job)
        return job.job_id

    def schedule_report(self, user_id: int, priority: JobPriority = JobPriority.HIGH):
        """리포트 생성 스케줄링"""
        job = Job(
            job_id=f'REPORT-{uuid.uuid4().hex[:8].upper()}',
            job_type=JobType.REPORT,
            priority=priority,
            params={'user_id': user_id}
        )

        self.schedule(job)
        return job.job_id

# 사용 예
if __name__ == '__main__':
    scheduler = JobScheduler()

    # 다양한 우선순위의 작업 스케줄링
    scheduler.schedule_crawling('https://example.com', JobPriority.LOW)
    scheduler.schedule_analysis('data-001', JobPriority.NORMAL)
    scheduler.schedule_report(1001, JobPriority.HIGH)
    scheduler.schedule_report(1002, JobPriority.CRITICAL)

    print("\n [i] Jobs scheduled by priority!")
    print(" [i] Processing order: CRITICAL → HIGH → NORMAL → LOW")

Worker​

# worker.py
import pika
import time
import random
from models import Job, JobType, JobStatus
from datetime import datetime

class Worker:
    """작업 워커"""

    def __init__(self, worker_id: int, host='localhost'):
        self.worker_id = worker_id
        self.host = host
        self.connection = None
        self.channel = None

    def execute_crawling(self, params: dict):
        """크롤링 실행"""
        url = params['url']
        print(f"   [Worker {self.worker_id}] Crawling: {url}")
        time.sleep(2)  # 시뮬레이션

        # 10% 확률로 실패
        if random.random() < 0.1:
            raise Exception(f"Failed to crawl {url}")

        print(f"   [Worker {self.worker_id}] ✓ Crawling completed")

    def execute_analysis(self, params: dict):
        """분석 실행"""
        data_id = params['data_id']
        print(f"   [Worker {self.worker_id}] Analyzing: {data_id}")
        time.sleep(3)  # 시뮬레이션

        if random.random() < 0.1:
            raise Exception(f"Analysis failed for {data_id}")

        print(f"   [Worker {self.worker_id}] ✓ Analysis completed")

    def execute_report(self, params: dict):
        """리포트 생성"""
        user_id = params['user_id']
        print(f"   [Worker {self.worker_id}] Generating report for user {user_id}")
        time.sleep(1)  # 시뮬레이션

        print(f"   [Worker {self.worker_id}] ✓ Report generated")

    def execute_cleanup(self, params: dict):
        """정리 작업"""
        print(f"   [Worker {self.worker_id}] Running cleanup")
        time.sleep(0.5)  # 시뮬레이션

        print(f"   [Worker {self.worker_id}] ✓ Cleanup completed")

    def execute_job(self, job: Job):
        """작업 실행"""
        print(f"\n [Worker {self.worker_id}] Processing job: {job.job_id}")
        print(f"   Type: {job.job_type.value}")
        print(f"   Priority: {job.priority.value}")

        # 작업 시작
        job.status = JobStatus.RUNNING
        job.started_at = datetime.now().isoformat()

        # 작업 타입별 실행
        try:
            if job.job_type == JobType.CRAWLING:
                self.execute_crawling(job.params)
            elif job.job_type == JobType.ANALYSIS:
                self.execute_analysis(job.params)
            elif job.job_type == JobType.REPORT:
                self.execute_report(job.params)
            elif job.job_type == JobType.CLEANUP:
                self.execute_cleanup(job.params)

            # 성공
            job.status = JobStatus.COMPLETED
            job.completed_at = datetime.now().isoformat()

            return True

        except Exception as e:
            # 실패
            job.status = JobStatus.FAILED
            job.error_message = str(e)

            print(f"   [Worker {self.worker_id}] ✗ Job failed: {e}")

            return False

    def send_to_retry(self, job: Job):
        """재시도 큐로 전송"""
        job.retry_count += 1

        print(f"   [Worker {self.worker_id}] → Sending to retry queue "
              f"(attempt {job.retry_count}/{job.max_retries})")

        self.channel.basic_publish(
            exchange='',
            routing_key='jobs.retry',
            body=job.to_json(),
            properties=pika.BasicProperties(
                delivery_mode=2,
                content_type='application/json'
            )
        )

    def callback(self, ch, method, properties, body):
        """메시지 처리 콜백"""
        job = Job.from_json(body.decode())

        # 작업 실행
        success = self.execute_job(job)

        if success:
            # 성공: ACK
            ch.basic_ack(delivery_tag=method.delivery_tag)

        else:
            # 실패: 재시도 확인
            if job.retry_count < job.max_retries:
                # 재시도 큐로 전송
                self.send_to_retry(job)

                # 원본 메시지 ACK
                ch.basic_ack(delivery_tag=method.delivery_tag)

            else:
                # 최대 재시도 초과: NACK → DLX로 이동
                print(f"   [Worker {self.worker_id}] ✗ Max retries exceeded")

                ch.basic_nack(
                    delivery_tag=method.delivery_tag,
                    requeue=False
                )

    def start(self):
        """워커 시작"""
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        self.channel = self.connection.channel()

        # Fair dispatch (prefetch_count=1)
        self.channel.basic_qos(prefetch_count=1)

        self.channel.basic_consume(
            queue='jobs.main',
            on_message_callback=self.callback
        )

        print(f' [*] Worker {self.worker_id} started')
        print(f' [*] Waiting for jobs...\n')

        self.channel.start_consuming()

if __name__ == '__main__':
    import sys

    worker_id = int(sys.argv[1]) if len(sys.argv) > 1 else 1

    worker = Worker(worker_id=worker_id)

    try:
        worker.start()
    except KeyboardInterrupt:
        print(f'\nWorker {worker_id} stopped')

        """)

        # 실패 로그 저장

        # 관리자 알림
        # send_admin_alert(job)

        ch.basic_ack(delivery_tag=method.delivery_tag)

    def save_to_database(self, job: Job):
        """DB에 저장 (시뮬레이션)"""
        import json

        with open('failed_jobs.jsonl', 'a', encoding='utf-8') as f:
            f.write(job.to_json() + '\n')

        """핸들러 시작"""
        connection = pika.BlockingConnection(
            pika.ConnectionParameters(self.host)
        )
        channel = connection.channel()

        channel.basic_consume(
            queue='jobs.failed',
            on_message_callback=self.handle_failed_job
        )

        print(' [*] Failed Job Handler started')
        print(' [*] Waiting for failed jobs...\n')

        channel.start_consuming()

# job_monitor.py
import requests
from requests.auth import HTTPBasicAuth
import time

class JobMonitor:
    """작업 모니터"""

    def __init__(self, host='localhost', user='admin', password='admin123'):
        self.base_url = f'http://{host}:15672/api'
        self.auth = HTTPBasicAuth(user, password)

    def get_queue_stats(self):
        """큐 상태 조회"""
        response = requests.get(
            f'{self.base_url}/queues',
            auth=self.auth
        )

        queues = response.json()

        stats = {}
        for queue in queues:
            if queue['name'].startswith('jobs.'):
                stats[queue['name']] = {
                    'messages': queue.get('messages', 0),
                    'ready': queue.get('messages_ready', 0),
                    'unacked': queue.get('messages_unacknowledged', 0),
                    'consumers': queue.get('consumers', 0)
                }

        return stats

    def print_stats(self):
        """통계 출력"""
        stats = self.get_queue_stats()

        print(" Job Queue Statistics")
        print(" " + "="*60)

        for queue_name, data in stats.items():
            print(f"\n {queue_name}:")
            print(f"   Total: {data['messages']:4d} | "
                  f"Ready: {data['ready']:4d} | "
                  f"Unacked: {data['unacked']:4d} | "
                  f"Consumers: {data['consumers']:2d}")

    def monitor(self, interval: int = 5):
        """실시간 모니터링"""
        print(f" Job Monitor (refresh every {interval}s)")
        print(" Press Ctrl+C to stop\n")

        try:
            while True:
                self.print_stats()
                print()
                time.sleep(interval)

        except KeyboardInterrupt:
            print("\n Monitoring stopped")

# 사용 예
if __name__ == '__main__':
    monitor = JobMonitor()
    monitor.monitor(interval=3)

# job_cancel.py
import pika

def cancel_job(job_id: str):
    """작업 취소 (큐에서 제거)"""
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()

    # 큐의 모든 메시지 확인
    while True:
        method, properties, body = channel.basic_get(
            queue='jobs.main',
            auto_ack=False
        )

        if method is None:
            break

        from models import Job
        job = Job.from_json(body.decode())

        if job.job_id == job_id:
            # 찾은 작업: NACK (삭제)
            channel.basic_nack(
                delivery_tag=method.delivery_tag,
                requeue=False
            )
            print(f" [✓] Job {job_id} cancelled")
            break
        else:
            # 다른 작업: 다시 큐에
            channel.basic_nack(
                delivery_tag=method.delivery_tag,
                requeue=True
            )

    connection.close()
