// .github/actions/semantic-version/index.js
const core = require('@actions/core');
const github = require('@actions/github');
const semver = require('semver');

async function run() {
  try {
    const token = core.getInput('github-token');
    const octokit = github.getOctokit(token);

    // ìµœì‹  íƒœê·¸ ê°€ì ¸ì˜¤ê¸°
    const { data: tags } = await octokit.rest.repos.listTags({
      owner: github.context.repo.owner,
      repo: github.context.repo.repo,
      per_page: 1
    });

    const currentVersion = tags[0]?.name || 'v0.0.0';

    // PR ë¼ë²¨ ê¸°ë°˜ìœ¼ë¡œ ë²„ì „ ê²°ì •
    const labels = github.context.payload.pull_request?.labels || [];
    const isMajor = labels.some(l => l.name === 'major');
    const isMinor = labels.some(l => l.name === 'minor');

    let newVersion;
    if (isMajor) {
      newVersion = semver.inc(currentVersion, 'major');
    } else if (isMinor) {
      newVersion = semver.inc(currentVersion, 'minor');
    } else {
      newVersion = semver.inc(currentVersion, 'patch');
    }

    core.setOutput('version', newVersion);
    core.setOutput('previous-version', currentVersion);

  } catch (error) {
    core.setFailed(error.message);
  }
}

run();
