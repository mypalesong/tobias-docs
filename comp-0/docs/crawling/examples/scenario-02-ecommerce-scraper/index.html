시나리오 2: 쇼핑몰 가격 모니터링

온라인 쇼핑몰의 상품 가격을 모니터링하고 이미지를 다운로드하는 스크래퍼를 만들어봅시다!

🎯 목표​

  • 쇼핑몰 상품 정보 수집 (제목, 가격, 평점, 리뷰 수)
  • 상품 이미지 자동 다운로드
  • 가격 변동 추적 및 알림
  • 데이터베이스 저장
  • 가격 히스토리 분석

scraper/image_downloader.py​

import requests
from PIL import Image
import io
import os
import hashlib
import logging
from pathlib import Path

logger = logging.getLogger(__name__)

class ImageDownloader:
    def __init__(self, save_dir: str):
        self.save_dir = save_dir
        os.makedirs(save_dir, exist_ok=True)

    def download_image(self, url: str, product_id: str) -> str:
        """이미지 다운로드"""
        try:
            # 이미지 다운로드
            response = requests.get(url, timeout=10)
            response.raise_for_status()

            # 이미지 검증
            img = Image.open(io.BytesIO(response.content))

            # 파일명 생성 (product_id + hash)
            url_hash = hashlib.md5(url.encode()).hexdigest()[:8]
            filename = f"{product_id}_{url_hash}.jpg"
            filepath = os.path.join(self.save_dir, filename)

            # 이미지 저장 (JPEG로 변환)
            if img.mode in ('RGBA', 'LA', 'P'):
                # 투명 배경을 흰색으로
                background = Image.new('RGB', img.size, (255, 255, 255))
                background.paste(img, mask=img.split()[-1] if img.mode == 'RGBA' else None)
                img = background

            img.save(filepath, 'JPEG', quality=85, optimize=True)

            logger.info(f"✅ 이미지 다운로드 완료: {filename}")
            return filepath

        except Exception as e:
            logger.error(f"❌ 이미지 다운로드 실패: {url} - {e}")
            return None

    def create_thumbnail(self, image_path: str, size=(200, 200)) -> str:
        """썸네일 생성"""
        try:
            img = Image.open(image_path)
            img.thumbnail(size, Image.Resampling.LANCZOS)

            # 썸네일 저장
            thumb_path = image_path.replace('.jpg', '_thumb.jpg')
            img.save(thumb_path, 'JPEG', quality=85)

            return thumb_path

        except Exception as e:
            logger.error(f"썸네일 생성 실패: {e}")
            return None

scraper/product_scraper.py​

from playwright.sync_api import sync_playwright
from typing import List, Dict
import logging
import time
import random
from urllib.parse import urljoin

logger = logging.getLogger(__name__)

class ProductScraper:
    def __init__(self, config):
        self.config = config

    def scrape(self) -> List[Dict]:
        """상품 스크래핑"""
        logger.info("상품 스크래핑 시작")

        with sync_playwright() as p:
            browser = p.chromium.launch(headless=self.config.HEADLESS)
            page = browser.new_page()

            try:
                products = []

                for page_num in range(1, self.config.MAX_PAGES + 1):
                    url = f"{self.config.BASE_URL}/catalogue/page-{page_num}.html"
                    logger.info(f"페이지 {page_num} 스크래핑: {url}")

                    page.goto(url, timeout=self.config.TIMEOUT)

                    # 상품 추출
                    page_products = self._extract_products(page)
                    products.extend(page_products)

                    logger.info(f"페이지 {page_num}: {len(page_products)}개 상품 수집")

                    # 대기
                    if page_num < self.config.MAX_PAGES:
                        time.sleep(random.uniform(
                            self.config.DELAY_BETWEEN_PAGES,
                            self.config.DELAY_BETWEEN_PAGES + 1
                        ))

                logger.info(f"✅ 총 {len(products)}개 상품 수집 완료")
                return products

            finally:
                browser.close()

    def _extract_products(self, page) -> List[Dict]:
        """상품 정보 추출"""
        products = []
        product_elements = page.query_selector_all('.product_pod')

        for elem in product_elements:
            try:
                product = self._parse_product(elem, page)
                products.append(product)

            except Exception as e:
                logger.error(f"상품 파싱 실패: {e}")
                continue

        return products

    def _parse_product(self, elem, page) -> Dict:
        """개별 상품 파싱"""
        # 제목
        title_elem = elem.query_selector('h3 > a')
        title = title_elem.get_attribute('title') if title_elem else 'N/A'

        # URL
        url = title_elem.get_attribute('href') if title_elem else ''
        full_url = urljoin(self.config.BASE_URL, url)

        # 가격
        price_elem = elem.query_selector('.price_color')
        price_text = price_elem.inner_text() if price_elem else '£0.00'
        price = float(price_text.replace('£', ''))

        # 재고 여부
        stock_elem = elem.query_selector('.availability')
        in_stock = 'In stock' in (stock_elem.inner_text() if stock_elem else '')

        # 평점
        star_elem = elem.query_selector('.star-rating')
        rating_class = star_elem.get_attribute('class') if star_elem else ''
        rating_map = {'One': 1, 'Two': 2, 'Three': 3, 'Four': 4, 'Five': 5}
        rating = 0
        for word, value in rating_map.items():
            if word in rating_class:
                rating = value
                break

        # 이미지 URL
        img_elem = elem.query_selector('img')
        img_url = img_elem.get_attribute('src') if img_elem else ''
        full_img_url = urljoin(self.config.BASE_URL, img_url)

        # 상품 ID 생성 (URL에서 추출)
        import re
        product_id_match = re.search(r'/([^/]+)_\d+/index.html', url)
        product_id = product_id_match.group(1) if product_id_match else str(hash(title))

        return {
            'product_id': product_id,
            'title': title,
            'url': full_url,
            'price': price,
            'currency': 'GBP',
            'in_stock': in_stock,
            'rating': rating,
            'image_url': full_img_url
        }

scraper/price_tracker.py​

from sqlalchemy import create_engine, Column, Integer, String, Float, Boolean, DateTime
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

Base = declarative_base()

class Product(Base):
    __tablename__ = 'products'

    id = Column(Integer, primary_key=True)
    product_id = Column(String, unique=True, index=True)
    title = Column(String)
    url = Column(String)
    current_price = Column(Float)
    previous_price = Column(Float)
    currency = Column(String)
    in_stock = Column(Boolean)
    rating = Column(Integer)
    image_url = Column(String)
    image_path = Column(String)
    first_seen = Column(DateTime, default=datetime.utcnow)
    last_updated = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

class PriceHistory(Base):
    __tablename__ = 'price_history'

    id = Column(Integer, primary_key=True)
    product_id = Column(String, index=True)
    price = Column(Float)
    in_stock = Column(Boolean)
    timestamp = Column(DateTime, default=datetime.utcnow)

class PriceTracker:
    def __init__(self, database_url: str):
        self.engine = create_engine(database_url)
        Base.metadata.create_all(self.engine)
        self.Session = sessionmaker(bind=self.engine)

    def update_products(self, products: list, image_paths: dict):
        """상품 정보 업데이트"""
        session = self.Session()

        try:
            for product_data in products:
                product_id = product_data['product_id']

                # 기존 상품 조회
                existing = session.query(Product).filter_by(
                    product_id=product_id
                ).first()

                if existing:
                    # 가격 변동 확인
                    if existing.current_price != product_data['price']:
                        logger.info(f"가격 변동: {existing.title} "
                                  f"£{existing.current_price} → £{product_data['price']}")

                        # 가격 히스토리 저장
                        history = PriceHistory(
                            product_id=product_id,
                            price=product_data['price'],
                            in_stock=product_data['in_stock']
                        )
                        session.add(history)

                        # 가격 알림 확인
                        self._check_price_alert(existing, product_data['price'])

                    # 업데이트
                    existing.previous_price = existing.current_price
                    existing.current_price = product_data['price']
                    existing.in_stock = product_data['in_stock']
                    existing.rating = product_data['rating']
                    existing.image_path = image_paths.get(product_id)

                else:
                    # 새 상품 추가
                    new_product = Product(
                        product_id=product_id,
                        title=product_data['title'],
                        url=product_data['url'],
                        current_price=product_data['price'],
                        previous_price=product_data['price'],
                        currency=product_data['currency'],
                        in_stock=product_data['in_stock'],
                        rating=product_data['rating'],
                        image_url=product_data['image_url'],
                        image_path=image_paths.get(product_id)
                    )
                    session.add(new_product)

                    # 히스토리 추가
                    history = PriceHistory(
                        product_id=product_id,
                        price=product_data['price'],
                        in_stock=product_data['in_stock']
                    )
                    session.add(history)

            session.commit()
            logger.info(f"✅ 데이터베이스 업데이트 완료: {len(products)}개 상품")

        except Exception as e:
            session.rollback()
            logger.error(f"데이터베이스 업데이트 실패: {e}")
            raise

        finally:
            session.close()

    def _check_price_alert(self, product: Product, new_price: float):
        """가격 하락 알림"""
        if product.current_price > 0:
            drop_percent = (product.current_price - new_price) / product.current_price

            if drop_percent >= 0.1:  # 10% 이상 하락
                logger.warning(f"🔔 가격 하락 알림: {product.title}")
                logger.warning(f"   {drop_percent*100:.1f}% 하락: "
                             f"£{product.current_price} → £{new_price}")

    def get_price_history(self, product_id: str) -> list:
        """가격 히스토리 조회"""
        session = self.Session()

        try:
            history = session.query(PriceHistory).filter_by(
                product_id=product_id
            ).order_by(PriceHistory.timestamp).all()

            return [{
                'price': h.price,
                'in_stock': h.in_stock,
                'timestamp': h.timestamp
            } for h in history]

        finally:
            session.close()

    def get_top_rated_products(self, limit: int = 10) -> list:
        """최고 평점 상품"""
        session = self.Session()

        try:
            products = session.query(Product).filter(
                Product.in_stock == True
            ).order_by(Product.rating.desc()).limit(limit).all()

            return [{
                'title': p.title,
                'price': p.current_price,
                'rating': p.rating,
                'url': p.url
            } for p in products]

        finally:
            session.close()

main.py​

import logging
from datetime import datetime
from config import Config
from scraper.product_scraper import ProductScraper
from scraper.image_downloader import ImageDownloader
from scraper.price_tracker import PriceTracker

def setup_logger():
    """로거 설정"""
    Config.ensure_directories()

    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler(f'scraper_{datetime.now().strftime("%Y%m%d")}.log'),
            logging.StreamHandler()
        ]
    )

def main():
    setup_logger()
    logger = logging.getLogger(__name__)

    try:
        logger.info("="*60)
        logger.info("쇼핑몰 스크래퍼 시작")
        logger.info("="*60)

        # 1. 상품 스크래핑
        scraper = ProductScraper(Config)
        products = scraper.scrape()

        if not products:
            logger.warning("수집된 상품이 없습니다")
            return

        # 2. 이미지 다운로드
        image_paths = {}

        if Config.DOWNLOAD_IMAGES:
            logger.info("이미지 다운로드 시작...")
            downloader = ImageDownloader(Config.IMAGE_DIR)

            for product in products:
                if product['image_url']:
                    image_path = downloader.download_image(
                        product['image_url'],
                        product['product_id']
                    )

                    if image_path:
                        image_paths[product['product_id']] = image_path

                        # 썸네일 생성
                        downloader.create_thumbnail(image_path)

            logger.info(f"✅ {len(image_paths)}개 이미지 다운로드 완료")

        # 3. 데이터베이스 저장 및 가격 추적
        tracker = PriceTracker(Config.DATABASE_URL)
        tracker.update_products(products, image_paths)

        # 4. 통계 출력
        print("\n" + "="*60)
        print("📊 스크래핑 결과")
        print("="*60)
        print(f"총 상품 수: {len(products)}")
        print(f"이미지 다운로드: {len(image_paths)}개")

        # 최고 평점 상품
        top_rated = tracker.get_top_rated_products(5)
        print("\n⭐ 최고 평점 상품 TOP 5:")
        for idx, product in enumerate(top_rated, 1):
            print(f"{idx}. {product['title']}")
            print(f"   가격: £{product['price']} | 평점: {product['rating']}")

        print("="*60)
        logger.info("✅ 모든 작업 완료!")

    except Exception as e:
        logger.error(f"❌ 에러 발생: {e}", exc_info=True)

if __name__ == "__main__":
    main()
