시나리오 1: 뉴스 스크래퍼

Hacker News의 최신 기사를 자동으로 수집하는 스크래퍼를 만들어봅시다!

🎯 목표​

  • Hacker News 상위 30개 기사 수집
  • 제목, URL, 점수, 댓글 수, 작성자, 시간 추출
  • CSV 및 JSON 형식으로 저장
  • 스케줄링으로 정기적 수집

config.py​

import os
from datetime import datetime

class Config:
    # 브라우저 설정
    HEADLESS = True
    TIMEOUT = 30000

    # 타겟 URL
    BASE_URL = 'https://news.ycombinator.com'

    # 수집 설정
    MAX_ITEMS = 30
    DELAY_BETWEEN_PAGES = 2  # 초

    # 저장 경로
    OUTPUT_DIR = 'data'
    CSV_DIR = os.path.join(OUTPUT_DIR, 'csv')
    JSON_DIR = os.path.join(OUTPUT_DIR, 'json')
    LOG_DIR = 'logs'

    # 파일명
    TIMESTAMP = datetime.now().strftime('%Y%m%d_%H%M%S')
    CSV_FILE = os.path.join(CSV_DIR, f'hackernews_{TIMESTAMP}.csv')
    JSON_FILE = os.path.join(JSON_DIR, f'hackernews_{TIMESTAMP}.json')

    @classmethod
    def ensure_directories(cls):
        """필요한 디렉토리 생성"""
        os.makedirs(cls.CSV_DIR, exist_ok=True)
        os.makedirs(cls.JSON_DIR, exist_ok=True)
        os.makedirs(cls.LOG_DIR, exist_ok=True)

scraper/hn_scraper.py​

from playwright.sync_api import sync_playwright
import logging
from typing import List, Dict
import time

logger = logging.getLogger(__name__)

class HackerNewsScraper:
    def __init__(self, config):
        self.config = config

    def scrape(self) -> List[Dict]:
        """Hacker News 스크래핑"""
        logger.info("Hacker News 스크래핑 시작")

        with sync_playwright() as p:
            # 브라우저 실행
            browser = p.chromium.launch(headless=self.config.HEADLESS)
            page = browser.new_page()

            try:
                # 페이지 방문
                page.goto(self.config.BASE_URL, timeout=self.config.TIMEOUT)
                logger.info(f"페이지 로드 완료: {self.config.BASE_URL}")

                # 데이터 추출
                stories = self._extract_stories(page)

                logger.info(f"✅ {len(stories)}개 기사 수집 완료")
                return stories

            except Exception as e:
                logger.error(f"스크래핑 중 에러: {e}")
                raise

            finally:
                browser.close()

    def _extract_stories(self, page) -> List[Dict]:
        """기사 데이터 추출"""
        stories = []

        # 기사 행 추출
        story_rows = page.query_selector_all('.athing')

        for idx, row in enumerate(story_rows[:self.config.MAX_ITEMS], 1):
            try:
                story = self._parse_story(row, page, idx)
                stories.append(story)

                logger.info(f"[{idx}] {story['title']}")

            except Exception as e:
                logger.error(f"기사 파싱 실패 (#{idx}): {e}")
                continue

        return stories

    def _parse_story(self, row, page, rank: int) -> Dict:
        """개별 기사 파싱"""
        # 기사 ID
        story_id = row.get_attribute('id')

        # 제목 & URL
        title_elem = row.query_selector('.titleline > a')
        title = title_elem.inner_text() if title_elem else 'N/A'
        url = title_elem.get_attribute('href') if title_elem else 'N/A'

        # 사이트 정보
        site_elem = row.query_selector('.sitebit')
        site = site_elem.inner_text().strip('()') if site_elem else 'N/A'

        # 메타 정보 (다음 행)
        meta_row = page.query_selector(f'#score_{story_id}').evaluate_handle(
            'el => el.closest("tr")'
        )

        # 점수
        score_elem = page.query_selector(f'#score_{story_id}')
        score_text = score_elem.inner_text() if score_elem else '0 points'
        score = int(score_text.split()[0]) if score_text != '0 points' else 0

        # 작성자
        author_elem = page.query_selector(f'#score_{story_id} ~ .hnuser')
        author = author_elem.inner_text() if author_elem else 'unknown'

        # 시간
        time_elem = page.query_selector(f'#score_{story_id} ~ .age')
        time_ago = time_elem.get_attribute('title') if time_elem else 'N/A'

        # 댓글 수
        comments_elem = page.query_selector(f'#score_{story_id} ~ a:has-text("comment")')
        comments_text = comments_elem.inner_text() if comments_elem else '0 comments'

        if 'comment' in comments_text:
            comments = int(comments_text.split()[0])
        else:
            comments = 0

        return {
            'rank': rank,
            'id': story_id,
            'title': title,
            'url': url,
            'site': site,
            'score': score,
            'author': author,
            'time': time_ago,
            'comments': comments
        }

scraper/parser.py​

import pandas as pd
import json
import logging

logger = logging.getLogger(__name__)

class DataSaver:
    @staticmethod
    def save_csv(data: list, filename: str):
        """CSV로 저장"""
        try:
            df = pd.DataFrame(data)
            df.to_csv(filename, index=False, encoding='utf-8-sig')
            logger.info(f"✅ CSV 저장 완료: {filename}")
        except Exception as e:
            logger.error(f"CSV 저장 실패: {e}")
            raise

    @staticmethod
    def save_json(data: list, filename: str):
        """JSON으로 저장"""
        try:
            with open(filename, 'w', encoding='utf-8') as f:
                json.dump(data, f, ensure_ascii=False, indent=2)
            logger.info(f"✅ JSON 저장 완료: {filename}")
        except Exception as e:
            logger.error(f"JSON 저장 실패: {e}")
            raise

    @staticmethod
    def print_summary(data: list):
        """수집 요약 출력"""
        if not data:
            logger.warning("수집된 데이터가 없습니다")
            return

        df = pd.DataFrame(data)

        print("\n" + "="*60)
        print("📊 수집 요약")
        print("="*60)
        print(f"총 기사 수: {len(df)}")
        print(f"평균 점수: {df['score'].mean():.1f}")
        print(f"평균 댓글 수: {df['comments'].mean():.1f}")
        print(f"\n상위 5개 기사:")
        print("-"*60)

        for idx, row in df.head(5).iterrows():
            print(f"{row['rank']}. {row['title']}")
            print(f"   점수: {row['score']} | 댓글: {row['comments']} | 작성자: {row['author']}")

        print("="*60 + "\n")

main.py​

import logging
from datetime import datetime
from config import Config
from scraper.hn_scraper import HackerNewsScraper
from scraper.parser import DataSaver

def setup_logger():
    """로거 설정"""
    Config.ensure_directories()

    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler(
                f'{Config.LOG_DIR}/scraper_{datetime.now().strftime("%Y%m%d")}.log'
            ),
            logging.StreamHandler()
        ]
    )

def main():
    """메인 실행 함수"""
    setup_logger()
    logger = logging.getLogger(__name__)

    try:
        logger.info("=" * 60)
        logger.info("Hacker News 스크래퍼 시작")
        logger.info("=" * 60)

        # 스크래핑
        scraper = HackerNewsScraper(Config)
        stories = scraper.scrape()

        if not stories:
            logger.warning("수집된 데이터가 없습니다")
            return

        # 저장
        saver = DataSaver()
        saver.save_csv(stories, Config.CSV_FILE)
        saver.save_json(stories, Config.JSON_FILE)

        # 요약 출력
        saver.print_summary(stories)

        logger.info("=" * 60)
        logger.info("✅ 모든 작업 완료!")
        logger.info("=" * 60)

    except Exception as e:
        logger.error(f"❌ 에러 발생: {e}", exc_info=True)

if __name__ == "__main__":
    main()

스케줄링 (선택사항)​

# scheduler.py
import schedule
import time
from main import main
import logging

logger = logging.getLogger(__name__)

def job():
    """스케줄된 작업"""
    logger.info("🕐 스케줄된 스크래핑 시작")
    main()

# 매 시간마다 실행
schedule.every().hour.do(job)

# 매일 오전 9시 실행
schedule.every().day.at("09:00").do(job)

logger.info("스케줄러 시작 - Ctrl+C로 중지")

while True:
    schedule.run_pending()
    time.sleep(60)
