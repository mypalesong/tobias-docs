    @staticmethod
    def create(db: Session, external_id: str, company_id: int, **kwargs) -> Optional[Job]:
        """ì±„ìš© ê³µê³  ìƒì„± (ì¤‘ë³µ ì²´í¬)"""

            return None  # ì´ë¯¸ ì¡´ì¬

        db.add(job)
        db.commit()
        db.refresh(job)
        return job

    @staticmethod
    def search(
        db: Session,
        keyword: Optional[str] = None,
        location: Optional[str] = None,
        min_salary: Optional[float] = None,
        experience_level: Optional[str] = None
    ) -> List[Job]:
        """ì±„ìš© ê³µê³  ê²€ìƒ‰"""
        query = db.query(Job).filter(Job.is_active == True)

        if keyword:
            query = query.filter(
                (Job.title.ilike(f'%{keyword}%')) |
                (Job.description.ilike(f'%{keyword}%'))
            )

        if location:
            query = query.filter(Job.location.ilike(f'%{location}%'))

        if min_salary:
            query = query.filter(Job.salary_min >= min_salary)

        if experience_level:
            query = query.filter(Job.experience_level == experience_level)

        return query.all()

        """ì±„ìš© ê³µê³ ë¥¼ ë¹„í™œì„±í™”"""

        self.session = requests.Session()
        self.session.headers.update({

    def download_logo(self, url: str, company_name: str) -> Optional[str]:
        """íšŒì‚¬ ë¡œê³  ë‹¤ìš´ë¡œë“œ"""
        try:
            # íšŒì‚¬ëª…ì„ íŒŒì¼ëª…ìœ¼ë¡œ ë³€í™˜
            safe_name = "".join(c for c in company_name if c.isalnum() or c in (' ', '-', '_')).strip()
            safe_name = safe_name.replace(' ', '_')

            # URLì—ì„œ ì´ë¯¸ì§€ ë‹¤ìš´ë¡œë“œ
            response = self.session.get(url, timeout=15)
            response.raise_for_status()

            img = Image.open(BytesIO(response.content))

            # RGBA to RGB ë³€í™˜
            if img.mode in ('RGBA', 'LA', 'P'):
                background = Image.new('RGB', img.size, (255, 255, 255))
                if img.mode == 'P':
                    img = img.convert('RGBA')
                background.paste(img, mask=img.split()[-1] if img.mode in ('RGBA', 'LA') else None)
                img = background

            # ë¡œê³  ë¦¬ì‚¬ì´ì¦ˆ (ì •ì‚¬ê°í˜•, ìµœëŒ€ 200x200)
            img.thumbnail((200, 200), Image.Resampling.LANCZOS)

            # ì €ì¥
            filepath = os.path.join(self.logos_dir, f"{safe_name}.jpg")
            img.save(filepath, 'JPEG', quality=90)

            logger.info(f"âœ… ë¡œê³  ì €ì¥: {safe_name}")
            return filepath

        except Exception as e:
            logger.error(f"ë¡œê³  ë‹¤ìš´ë¡œë“œ ì‹¤íŒ¨ ({company_name}): {e}")
            return None

    def save_screenshot(self, screenshot_bytes: bytes, job_id: str) -> Optional[str]:
        """ì±„ìš© ê³µê³  ìŠ¤í¬ë¦°ìƒ· ì €ì¥"""

class JobScraper:

                user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',

                        logger.error(f"ì¹´ë“œ íŒŒì‹± ì‹¤íŒ¨ (#{idx}): {e}")
                        continue

                return jobs

                logger.error(f"ìŠ¤í¬ë˜í•‘ ì¤‘ ì—ëŸ¬: {e}")
                raise

    def _build_search_url(self, keyword: str, location: str) -> str:

    def _extract_job_details(self, page: Page) -> Dict:

    def _parse_posted_date(self, text: str) -> Optional[datetime]:
        """ê²Œì‹œ ë‚ ì§œ íŒŒì‹± ('2 days ago' -> datetime)"""
        try:
            from datetime import timedelta

            now = datetime.now()

            if 'hour' in text:
