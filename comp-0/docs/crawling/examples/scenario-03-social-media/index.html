시나리오 3: 소셜 미디어 데이터 수집

Instagram 스타일의 소셜 미디어에서 게시물, 프로필 사진을 수집하는 스크래퍼를 만들어봅시다!

🎯 목표​

  • 게시물 정보 수집 (텍스트, 좋아요, 댓글)
  • 프로필 사진 및 게시물 이미지 다운로드
  • 해시태그 추출
  • 무한 스크롤 처리
  • 로그인 세션 관리

import os

class Config:
    # 브라우저
    HEADLESS = False  # 로그인 시 헤드풀 모드 권장
    TIMEOUT = 30000

    # 타겟 계정 (예제)
    TARGET_USERNAME = 'natgeo'  # National Geographic
    MAX_POSTS = 50

    # 이미지 설정
    DOWNLOAD_PROFILE_PICS = True
    DOWNLOAD_POST_IMAGES = True
    IMAGE_DIR = 'data/images'
    PROFILE_DIR = os.path.join(IMAGE_DIR, 'profiles')
    POST_DIR = os.path.join(IMAGE_DIR, 'posts')

    # 로그인 (선택사항)
    USE_LOGIN = False  # True로 설정 시 로그인 필요
    USERNAME = os.getenv('INSTAGRAM_USERNAME')
    PASSWORD = os.getenv('INSTAGRAM_PASSWORD')

    # 쿠키 저장 경로
    COOKIE_FILE = 'cookies/session.json'

    # 스크롤 설정
    SCROLL_PAUSE = 2
    MAX_SCROLLS = 10

    @classmethod
    def ensure_directories(cls):
        os.makedirs(cls.PROFILE_DIR, exist_ok=True)
        os.makedirs(cls.POST_DIR, exist_ok=True)
        os.makedirs('cookies', exist_ok=True)

scraper/image_manager.py​

import requests
from PIL import Image
import io
import os
import hashlib
import logging
from typing import Optional

logger = logging.getLogger(__name__)

class ImageManager:
    def __init__(self, base_dir: str):
        self.base_dir = base_dir
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })

    def download_image(
        self,
        url: str,
        save_dir: str,
        filename: Optional[str] = None
    ) -> Optional[str]:
        """이미지 다운로드"""
        try:
            response = self.session.get(url, timeout=15)
            response.raise_for_status()

            # 이미지 검증
            img = Image.open(io.BytesIO(response.content))

            # 파일명 생성
            if not filename:
                url_hash = hashlib.md5(url.encode()).hexdigest()
                filename = f"{url_hash}.jpg"

            filepath = os.path.join(save_dir, filename)

            # 이미지 저장
            if img.mode in ('RGBA', 'LA'):
                # RGB로 변환
                rgb_img = Image.new('RGB', img.size, (255, 255, 255))
                rgb_img.paste(img, mask=img.split()[-1] if 'A' in img.mode else None)
                img = rgb_img

            img.save(filepath, 'JPEG', quality=90)

            logger.info(f"✅ 이미지 저장: {filename}")
            return filepath

        except Exception as e:
            logger.error(f"❌ 이미지 다운로드 실패: {url} - {e}")
            return None

    def download_profile_picture(
        self,
        url: str,
        username: str,
        save_dir: str
    ) -> Optional[str]:
        """프로필 사진 다운로드"""
        filename = f"{username}_profile.jpg"
        return self.download_image(url, save_dir, filename)

    def download_post_images(
        self,
        urls: list,
        post_id: str,
        save_dir: str
    ) -> list:
        """게시물 이미지들 다운로드"""
        saved_paths = []

        for idx, url in enumerate(urls, 1):
            filename = f"{post_id}_{idx}.jpg"
            filepath = self.download_image(url, save_dir, filename)

            if filepath:
                saved_paths.append(filepath)

        return saved_paths

    def create_collage(self, image_paths: list, output_path: str, grid_size=(3, 3)):
        """여러 이미지로 콜라주 생성"""
        if not image_paths:
            return None

        try:
            # 이미지 로드
            images = [Image.open(path) for path in image_paths[:grid_size[0]*grid_size[1]]]

            # 모든 이미지를 같은 크기로 조정
            target_size = (300, 300)
            resized_images = [img.resize(target_size, Image.Resampling.LANCZOS) for img in images]

            # 콜라주 캔버스 생성
            rows, cols = grid_size
            collage_width = target_size[0] * cols
            collage_height = target_size[1] * rows
            collage = Image.new('RGB', (collage_width, collage_height), 'white')

            # 이미지 배치
            for idx, img in enumerate(resized_images):
                row = idx // cols
                col = idx % cols
                x = col * target_size[0]
                y = row * target_size[1]
                collage.paste(img, (x, y))

            # 저장
            collage.save(output_path, 'JPEG', quality=85)
            logger.info(f"✅ 콜라주 생성: {output_path}")

            return output_path

        except Exception as e:
            logger.error(f"콜라주 생성 실패: {e}")
            return None

scraper/post_parser.py​

import re
from typing import List, Dict

class PostParser:
    @staticmethod
    def extract_hashtags(text: str) -> List[str]:
        """해시태그 추출"""
        if not text:
            return []

        pattern = r'#(\w+)'
        hashtags = re.findall(pattern, text)
        return list(set(hashtags))  # 중복 제거

    @staticmethod
    def extract_mentions(text: str) -> List[str]:
        """멘션 추출"""
        if not text:
            return []

        pattern = r'@(\w+)'
        mentions = re.findall(pattern, text)
        return list(set(mentions))

    @staticmethod
    def parse_post_stats(stats_text: str) -> Dict:
        """좋아요, 댓글 수 파싱"""
        stats = {
            'likes': 0,
            'comments': 0
        }

        # 숫자 추출 (예: "1,234 likes" → 1234)
        like_match = re.search(r'([\d,]+)\s*like', stats_text, re.IGNORECASE)
        if like_match:
            stats['likes'] = int(like_match.group(1).replace(',', ''))

        comment_match = re.search(r'([\d,]+)\s*comment', stats_text, re.IGNORECASE)
        if comment_match:
            stats['comments'] = int(comment_match.group(1).replace(',', ''))

        return stats

scraper/instagram_scraper.py​

from playwright.sync_api import sync_playwright
from typing import List, Dict
import logging
import json
import time
import random

logger = logging.getLogger(__name__)

class InstagramScraper:
    def __init__(self, config):
        self.config = config

    def scrape_profile(self, username: str) -> Dict:
        """프로필 및 게시물 스크래핑"""
        logger.info(f"프로필 스크래핑: @{username}")

        with sync_playwright() as p:
            browser = p.chromium.launch(headless=self.config.HEADLESS)
            context = browser.new_context(
                viewport={'width': 1920, 'height': 1080},
                user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
            )

            # 쿠키 로드
            self._load_cookies(context)

            page = context.new_page()

            try:
                # 프로필 페이지 방문
                url = f'https://www.instagram.com/{username}/'
                page.goto(url, wait_until='networkidle')

                # 로그인 필요 시
                if self.config.USE_LOGIN and not self._is_logged_in(page):
                    self._login(page)
                    page.goto(url, wait_until='networkidle')

                # 프로필 정보 추출
                profile_data = self._extract_profile_info(page)

                # 게시물 스크롤 및 수집
                posts = self._scroll_and_collect_posts(page)

                profile_data['posts'] = posts
                profile_data['total_posts_scraped'] = len(posts)

                # 쿠키 저장
                self._save_cookies(context)

                return profile_data

            finally:
                browser.close()

    def _extract_profile_info(self, page) -> Dict:
        """프로필 정보 추출"""
        profile = {
            'username': '',
            'full_name': '',
            'bio': '',
            'profile_pic_url': '',
            'followers': 0,
            'following': 0,
            'posts_count': 0
        }

        try:
            # 사용자명
            username_elem = page.query_selector('header h2')
            if username_elem:
                profile['username'] = username_elem.inner_text()

            # 이름 및 바이오
            bio_section = page.query_selector('header section')
            if bio_section:
                name_elem = bio_section.query_selector('span')
                if name_elem:
                    profile['full_name'] = name_elem.inner_text()

            # 프로필 사진
            img_elem = page.query_selector('header img')
            if img_elem:
                profile['profile_pic_url'] = img_elem.get_attribute('src')

            # 통계 (팔로워, 팔로잉, 게시물 수)
            stats = page.query_selector_all('header section ul li')
            if len(stats) >= 3:
                # 게시물 수
                posts_text = stats[0].inner_text()
                profile['posts_count'] = self._parse_number(posts_text)

                # 팔로워
                followers_text = stats[1].inner_text()
                profile['followers'] = self._parse_number(followers_text)

                # 팔로잉
                following_text = stats[2].inner_text()
                profile['following'] = self._parse_number(following_text)

        except Exception as e:
            logger.error(f"프로필 정보 추출 실패: {e}")

        return profile

    def _scroll_and_collect_posts(self, page) -> List[Dict]:
        """무한 스크롤하며 게시물 수집"""
        posts = []
        last_height = 0

        for scroll_count in range(self.config.MAX_SCROLLS):
            # 현재 보이는 게시물들
            post_elements = page.query_selector_all('article a[href*="/p/"]')

            for elem in post_elements:
                try:
                    href = elem.get_attribute('href')
                    post_id = href.split('/p/')[1].strip('/')

                    # 중복 체크
                    if any(p['post_id'] == post_id for p in posts):
                        continue

                    # 이미지 URL
                    img_elem = elem.query_selector('img')
                    img_url = img_elem.get_attribute('src') if img_elem else None

                    posts.append({
                        'post_id': post_id,
                        'url': f'https://www.instagram.com{href}',
                        'thumbnail_url': img_url
                    })

                    if len(posts) >= self.config.MAX_POSTS:
                        break

                except Exception as e:
                    logger.error(f"게시물 파싱 실패: {e}")
                    continue

            if len(posts) >= self.config.MAX_POSTS:
                break

            # 스크롤
            current_height = page.evaluate('document.body.scrollHeight')
            if current_height == last_height:
                break

            page.evaluate('window.scrollTo(0, document.body.scrollHeight)')
            time.sleep(random.uniform(
                self.config.SCROLL_PAUSE,
                self.config.SCROLL_PAUSE + 1
            ))

            last_height = current_height

        logger.info(f"수집된 게시물: {len(posts)}개")
        return posts

    def _parse_number(self, text: str) -> int:
        """숫자 파싱 (1.2K → 1200, 1.5M → 1500000)"""
        text = text.strip().split()[0]  # 첫 번째 단어만

        multipliers = {'K': 1000, 'M': 1000000, 'B': 1000000000}

        for suffix, multiplier in multipliers.items():
            if suffix in text.upper():
                number = float(text.upper().replace(suffix, ''))
                return int(number * multiplier)

        # 쉼표 제거
        return int(text.replace(',', ''))

    def _login(self, page):
        """Instagram 로그인"""
        try:
            logger.info("로그인 중...")

            page.fill('input[name="username"]', self.config.USERNAME)
            page.fill('input[name="password"]', self.config.PASSWORD)
            page.click('button[type="submit"]')

            page.wait_for_load_state('networkidle')
            time.sleep(3)

            logger.info("로그인 완료")

        except Exception as e:
            logger.error(f"로그인 실패: {e}")

    def _is_logged_in(self, page) -> bool:
        """로그인 여부 확인"""
        return page.query_selector('input[name="username"]') is None

    def _load_cookies(self, context):
        """쿠키 로드"""
        try:
            with open(self.config.COOKIE_FILE, 'r') as f:
                cookies = json.load(f)
            context.add_cookies(cookies)
            logger.info("쿠키 로드 완료")
        except FileNotFoundError:
            logger.info("저장된 쿠키 없음")

    def _save_cookies(self, context):
        """쿠키 저장"""
        cookies = context.cookies()
        with open(self.config.COOKIE_FILE, 'w') as f:
            json.dump(cookies, f)
        logger.info("쿠키 저장 완료")

main.py​

import logging
import json
from config import Config
from scraper.instagram_scraper import InstagramScraper
from scraper.image_manager import ImageManager
from scraper.post_parser import PostParser

def setup_logger():
    Config.ensure_directories()

    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s - %(levelname)s - %(message)s',
        handlers=[
            logging.StreamHandler()
        ]
    )

def main():
    setup_logger()
    logger = logging.getLogger(__name__)

    try:
        logger.info("="*60)
        logger.info(f"Instagram 스크래퍼: @{Config.TARGET_USERNAME}")
        logger.info("="*60)

        # 1. 프로필 스크래핑
        scraper = InstagramScraper(Config)
        profile_data = scraper.scrape_profile(Config.TARGET_USERNAME)

        logger.info(f"\n프로필 정보:")
        logger.info(f"  이름: {profile_data['full_name']}")
        logger.info(f"  팔로워: {profile_data['followers']:,}")
        logger.info(f"  팔로잉: {profile_data['following']:,}")
        logger.info(f"  게시물: {profile_data['posts_count']:,}")

        # 2. 이미지 다운로드
        image_manager = ImageManager(Config.IMAGE_DIR)

        # 프로필 사진
        if Config.DOWNLOAD_PROFILE_PICS and profile_data['profile_pic_url']:
            logger.info("\n프로필 사진 다운로드 중...")
            image_manager.download_profile_picture(
                profile_data['profile_pic_url'],
                profile_data['username'],
                Config.PROFILE_DIR
            )

        # 게시물 이미지
        if Config.DOWNLOAD_POST_IMAGES:
            logger.info(f"\n게시물 이미지 다운로드 중... (총 {len(profile_data['posts'])}개)")

            downloaded_images = []

            for post in profile_data['posts'][:20]:  # 처음 20개만
                if post.get('thumbnail_url'):
                    filepath = image_manager.download_image(
                        post['thumbnail_url'],
                        Config.POST_DIR,
                        f"{post['post_id']}.jpg"
                    )

                    if filepath:
                        downloaded_images.append(filepath)

            logger.info(f"✅ {len(downloaded_images)}개 이미지 다운로드 완료")

            # 콜라주 생성
            if downloaded_images:
                collage_path = f"{Config.IMAGE_DIR}/collage_{Config.TARGET_USERNAME}.jpg"
                image_manager.create_collage(downloaded_images, collage_path, grid_size=(3, 3))

        # 3. JSON 저장
        output_file = f'data/{Config.TARGET_USERNAME}_profile.json'
        with open(output_file, 'w', encoding='utf-8') as f:
            json.dump(profile_data, f, ensure_ascii=False, indent=2)

        logger.info(f"\n✅ 데이터 저장: {output_file}")

        print("\n" + "="*60)
        print("📊 스크래핑 완료")
        print("="*60)
        print(f"수집된 게시물: {len(profile_data['posts'])}개")
        print(f"다운로드된 이미지: {len(downloaded_images) if downloaded_images else 0}개")
        print("="*60)

    except Exception as e:
        logger.error(f"❌ 에러 발생: {e}", exc_info=True)

if __name__ == "__main__":
    main()
