Base = declarative_base()

class Property(Base):
    __tablename__ = 'properties'

    id = Column(Integer, primary_key=True)
    external_id = Column(String(100), unique=True, nullable=False)

    # 기본 정보
    title = Column(String(200))
    address = Column(String(300))
    city = Column(String(100))
    state = Column(String(50))
    zip_code = Column(String(20))
    country = Column(String(50), default='South Korea')

    # 위치 정보
    latitude = Column(Float)
    longitude = Column(Float)

    # 매물 정보
    property_type = Column(String(50))  # apartment, house, villa, etc.
    price = Column(Float)
    currency = Column(String(10), default='USD')

    bedrooms = Column(Integer)
    bathrooms = Column(Float)
    area_sqft = Column(Float)  # 평방피트
    area_sqm = Column(Float)   # 평방미터
    area_pyeong = Column(Float)  # 평

    # 특징
    year_built = Column(Integer)
    parking_spaces = Column(Integer)
    floor = Column(Integer)
    total_floors = Column(Integer)

    description = Column(Text)
    features = Column(Text)  # JSON 형태로 저장

    # URL 및 이미지
    url = Column(String(500))
    image_count = Column(Integer, default=0)
    has_floorplan = Column(Boolean, default=False)
    gallery_path = Column(String(500))

    # 상태
    status = Column(String(50), default='for_sale')  # for_sale, sold, pending
    is_active = Column(Boolean, default=True)

    # 타임스탬프
    listed_date = Column(DateTime)
    scraped_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)

    # 관계
    images = relationship('PropertyImage', back_populates='property', cascade='all, delete-orphan')
    price_history = relationship('PriceHistory', back_populates='property', cascade='all, delete-orphan')

class PropertyImage(Base):
    __tablename__ = 'property_images'

    id = Column(Integer, primary_key=True)
    property_id = Column(Integer, ForeignKey('properties.id'))

    url = Column(String(500))
    local_path = Column(String(500))
    image_type = Column(String(50))  # photo, floorplan, map

    height = Column(Integer)
    file_size = Column(Integer)  # bytes

class PriceHistory(Base):
    __tablename__ = 'price_history'

    # 관계
    property = relationship('Property', back_populates='price_history')

# 데이터베이스 초기화
def init_db(database_url: str):
    engine = create_engine(database_url)
    Base.metadata.create_all(engine)

from sqlalchemy.orm import Session
from database.models import Property, PropertyImage, PriceHistory
from typing import Optional, List
from datetime import datetime

class PropertyCRUD:
    @staticmethod
    def create(db: Session, external_id: str, **kwargs) -> Optional[Property]:
        """매물 생성 (중복 체크)"""
        existing = db.query(Property).filter(Property.external_id == external_id).first()
        if existing:
            return None

        property = Property(external_id=external_id, **kwargs)
        db.add(property)
        db.commit()
        db.refresh(property)
        return property

    @staticmethod
    def update(db: Session, property_id: int, **kwargs) -> Property:
        """매물 정보 업데이트"""
        property = db.query(Property).filter(Property.id == property_id).first()
        if property:
            for key, value in kwargs.items():
                setattr(property, key, value)
            property.updated_at = datetime.now()
            db.commit()
            db.refresh(property)
        return property

    @staticmethod
    def search(
        db: Session,
        city: Optional[str] = None,
        property_type: Optional[str] = None,
        min_price: Optional[float] = None,
        max_price: Optional[float] = None,
        min_bedrooms: Optional[int] = None
    ) -> List[Property]:
        """매물 검색"""
        query = db.query(Property).filter(Property.is_active == True)

        if city:
            query = query.filter(Property.city.ilike(f'%{city}%'))

        if property_type:
            query = query.filter(Property.property_type == property_type)

        if min_price:
            query = query.filter(Property.price >= min_price)

        if max_price:
            query = query.filter(Property.price <= max_price)

        if min_bedrooms:
            query = query.filter(Property.bedrooms >= min_bedrooms)

        return query.all()

class PropertyImageCRUD:
    @staticmethod
    def create(db: Session, property_id: int, **kwargs) -> PropertyImage:
        """이미지 레코드 생성"""
        image = PropertyImage(property_id=property_id, **kwargs)
        db.add(image)
        db.commit()
        db.refresh(image)
        return image

    @staticmethod
    def get_by_property(db: Session, property_id: int, image_type: Optional[str] = None) -> List[PropertyImage]:
        """매물의 이미지 조회"""
        query = db.query(PropertyImage).filter(PropertyImage.property_id == property_id)

        if image_type:
            query = query.filter(PropertyImage.image_type == image_type)

        return query.order_by(PropertyImage.order).all()

class PriceHistoryCRUD:
    @staticmethod
    def create(db: Session, property_id: int, new_price: float):
        """가격 변동 기록"""
        property = db.query(Property).filter(Property.id == property_id).first()

        if property and property.price:
            change_amount = new_price - property.price
            change_percent = (change_amount / property.price) * 100

            history = PriceHistory(
                property_id=property_id,
                price=new_price,
                change_amount=change_amount,
                change_percent=change_percent
            )
            db.add(history)

            # 매물 가격 업데이트
            property.price = new_price
            db.commit()

    @staticmethod
    def get_history(db: Session, property_id: int) -> List[PriceHistory]:
        """가격 변동 내역 조회"""
        return db.query(PriceHistory)\
            .filter(PriceHistory.property_id == property_id)\
            .order_by(PriceHistory.recorded_at.desc())\
            .all()

import os
import requests
from PIL import Image, ImageDraw, ImageFont
from io import BytesIO
from typing import List, Optional, Dict
import logging
import exifread

logger = logging.getLogger(__name__)

class ImageManager:
    def __init__(self, properties_dir: str, floorplans_dir: str, galleries_dir: str):
        self.properties_dir = properties_dir
        self.floorplans_dir = floorplans_dir
        self.galleries_dir = galleries_dir
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })

    def download_property_images(
        self,
        urls: List[str],
        property_id: str,
        max_images: int = 20
    ) -> List[Dict[str, any]]:
        """매물 사진 다운로드"""
        downloaded_images = []

        # 매물별 디렉토리 생성
        property_dir = os.path.join(self.properties_dir, property_id)
        os.makedirs(property_dir, exist_ok=True)

        for idx, url in enumerate(urls[:max_images], 1):
            try:
                filepath = self.download_image(url, property_dir, f"photo_{idx}.jpg")

                if filepath:
                    # 이미지 메타데이터 추출
                    metadata = self.extract_metadata(filepath)

                    downloaded_images.append({
                        'url': url,
                        'local_path': filepath,
                        'order': idx,
                        'image_type': 'photo',
                        **metadata
                    })

                    logger.info(f"✅ 사진 {idx} 저장: {property_id}")

            except Exception as e:
                logger.error(f"이미지 다운로드 실패 ({idx}): {e}")
                continue

        return downloaded_images

    def download_floorplan(self, url: str, property_id: str) -> Optional[Dict[str, any]]:
        """평면도 다운로드"""
        try:
            filepath = self.download_image(url, self.floorplans_dir, f"{property_id}_floorplan.jpg")

            if filepath:
                metadata = self.extract_metadata(filepath)

                logger.info(f"✅ 평면도 저장: {property_id}")

                return {
                    'url': url,
                    'local_path': filepath,
                    'image_type': 'floorplan',
                    **metadata
                }

        except Exception as e:
            logger.error(f"평면도 다운로드 실패: {e}")
            return None

    def download_image(self, url: str, save_dir: str, filename: str) -> Optional[str]:
        """이미지 다운로드 공통 함수"""
        try:
            response = self.session.get(url, timeout=15)
            response.raise_for_status()

            img = Image.open(BytesIO(response.content))

            # RGBA to RGB 변환
            if img.mode in ('RGBA', 'LA', 'P'):
                background = Image.new('RGB', img.size, (255, 255, 255))
                if img.mode == 'P':
                    img = img.convert('RGBA')
                background.paste(img, mask=img.split()[-1] if img.mode in ('RGBA', 'LA') else None)
                img = background

            # 고품질로 저장
            filepath = os.path.join(save_dir, filename)
            img.save(filepath, 'JPEG', quality=90, optimize=True)

            return filepath

        except Exception as e:
            logger.error(f"이미지 다운로드 실패 ({url}): {e}")
            return None

    def create_gallery(
        self,
        image_paths: List[str],
        property_id: str,
        grid_size: tuple = (4, 5)
    ) -> Optional[str]:
        """사진 갤러리 생성 (그리드 레이아웃)"""
        try:
            if not image_paths:
                return None

            rows, cols = grid_size
            max_images = rows * cols

            # 이미지 로드
            images = []
            for path in image_paths[:max_images]:
                try:
                    img = Image.open(path)
                    images.append(img)
                except Exception as e:
                    logger.warning(f"이미지 로드 실패: {path}")
                    continue

            if not images:
                return None

            # 각 셀 크기 (300x200)
            cell_width, cell_height = 300, 200

            # 캔버스 생성
            gallery_width = cols * cell_width
            gallery_height = rows * cell_height
            gallery = Image.new('RGB', (gallery_width, gallery_height), (255, 255, 255))

            # 이미지 배치
            for idx, img in enumerate(images):
                row = idx // cols
                col = idx % cols

                # 리사이즈 (비율 유지)
                img.thumbnail((cell_width, cell_height), Image.Resampling.LANCZOS)

                # 중앙 정렬을 위한 오프셋 계산
                x_offset = (cell_width - img.width) // 2
                y_offset = (cell_height - img.height) // 2

                # 붙여넣기
                x = col * cell_width + x_offset
                y = row * cell_height + y_offset
                gallery.paste(img, (x, y))

            # 저장
            gallery_path = os.path.join(self.galleries_dir, f"{property_id}_gallery.jpg")
            gallery.save(gallery_path, 'JPEG', quality=90)

            logger.info(f"✅ 갤러리 생성: {property_id} ({len(images)}장)")

            return gallery_path

        except Exception as e:
            logger.error(f"갤러리 생성 실패: {e}")
            return None

    def create_thumbnail(self, image_path: str, size: tuple = (300, 200)) -> Optional[str]:
        """썸네일 생성"""
        try:
            img = Image.open(image_path)
            img.thumbnail(size, Image.Resampling.LANCZOS)

            thumb_path = image_path.replace('.jpg', '_thumb.jpg')
            img.save(thumb_path, 'JPEG', quality=85)

            return thumb_path

        except Exception as e:
            logger.error(f"썸네일 생성 실패: {e}")
            return None

    def extract_metadata(self, image_path: str) -> Dict[str, any]:
        """이미지 메타데이터 추출"""
        metadata = {
            'width': None,
            'height': None,
            'file_size': None
        }

        try:
            # 기본 정보
            img = Image.open(image_path)
            metadata['width'] = img.width
            metadata['height'] = img.height
            metadata['file_size'] = os.path.getsize(image_path)

            # EXIF 데이터 (선택)
            with open(image_path, 'rb') as f:
                tags = exifread.process_file(f, details=False)
                # GPS, 촬영 날짜 등 추가 가능

        except Exception as e:
            logger.warning(f"메타데이터 추출 실패: {e}")

        return metadata

    def add_watermark(self, image_path: str, text: str = "PropertyScraper") -> Optional[str]:
        """워터마크 추가"""
        try:
            img = Image.open(image_path)
            draw = ImageDraw.Draw(img)

            # 폰트 (기본 폰트 사용)
            try:
                font = ImageFont.truetype("arial.ttf", 30)
            except:
                font = ImageFont.load_default()

            # 워터마크 위치 (우측 하단)
            text_bbox = draw.textbbox((0, 0), text, font=font)
            text_width = text_bbox[2] - text_bbox[0]
            text_height = text_bbox[3] - text_bbox[1]

            x = img.width - text_width - 20
            y = img.height - text_height - 20

            # 반투명 배경
            overlay = Image.new('RGBA', img.size, (255, 255, 255, 0))
            overlay_draw = ImageDraw.Draw(overlay)
            overlay_draw.rectangle(
                [(x - 10, y - 5), (x + text_width + 10, y + text_height + 5)],
                fill=(0, 0, 0, 100)
            )

            img = Image.alpha_composite(img.convert('RGBA'), overlay)

            # 텍스트
            draw = ImageDraw.Draw(img)
            draw.text((x, y), text, fill=(255, 255, 255, 255), font=font)

            # 저장
            watermarked_path = image_path.replace('.jpg', '_watermarked.jpg')
            img.convert('RGB').save(watermarked_path, 'JPEG', quality=90)

            return watermarked_path

        except Exception as e:
            logger.error(f"워터마크 추가 실패: {e}")
            return None

from playwright.sync_api import sync_playwright, Page
from typing import List, Dict, Optional
import logging
import time
import re

logger = logging.getLogger(__name__)

class PropertyScraper:
    def __init__(self, config):
        self.config = config

    def scrape_properties(self, location: str, property_type: str) -> List[Dict]:
        """매물 스크래핑"""
        logger.info(f"검색 시작: {property_type} in {location}")

        with sync_playwright() as p:
            browser = p.chromium.launch(headless=self.config.HEADLESS)
            context = browser.new_context(
                user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
                viewport={'width': 1920, 'height': 1080}
            )
            page = context.new_page()

            try:
                properties = []

                # 검색 페이지로 이동
                search_url = self._build_search_url(location, property_type)
                page.goto(search_url, timeout=self.config.TIMEOUT)

                # 페이지 로드 대기
                page.wait_for_selector('.property-card, .listing-card', timeout=10000)
                time.sleep(2)

                # 스크롤하며 더 많은 결과 로드
                self._scroll_to_load_more(page)

                # 매물 카드 추출
                property_cards = page.query_selector_all('.property-card, .listing-card')

                logger.info(f"발견된 매물: {len(property_cards)}개")

                for idx, card in enumerate(property_cards[:30], 1):
                    try:
                        property_data = self._extract_property_card(card, page)
                        if property_data:
                            properties.append(property_data)
                            logger.info(f"[{idx}] {property_data['title']} - ${property_data.get('price', 0):,.0f}")

                    except Exception as e:
                        logger.error(f"카드 파싱 실패 (#{idx}): {e}")
                        continue

                return properties

            except Exception as e:
                logger.error(f"스크래핑 중 에러: {e}")
                raise

            finally:
                browser.close()

    def _build_search_url(self, location: str, property_type: str) -> str:
        """검색 URL 생성"""
        base = self.config.BASE_URL
        params = f"/homes/{location.replace(' ', '-')}_rb/?searchQueryState="
        # 실제 사이트에 맞게 조정 필요
        return base + params

    def _scroll_to_load_more(self, page: Page):
        """스크롤하여 더 많은 결과 로드"""
        for _ in range(3):
            page.evaluate('window.scrollTo(0, document.body.scrollHeight)')
            time.sleep(1.5)

    def _extract_property_card(self, card, page: Page) -> Optional[Dict]:
        """개별 매물 카드 파싱"""
        try:
            # 제목
            title_elem = card.query_selector('h3, .property-title')
            title = title_elem.inner_text().strip() if title_elem else 'N/A'

            # 가격
            price_elem = card.query_selector('.price, [data-test="property-card-price"]')
            price_text = price_elem.inner_text().strip() if price_elem else '$0'
            price = self._parse_price(price_text)

            # 주소
            address_elem = card.query_selector('.address, [data-test="property-card-addr"]')
            address = address_elem.inner_text().strip() if address_elem else 'N/A'

            # 침실/욕실/면적
            details_elem = card.query_selector('.property-details, .home-info')
            bedrooms, bathrooms, area = self._parse_details(details_elem.inner_text() if details_elem else '')

            # URL
            link_elem = card.query_selector('a[href*="/homedetails/"]')
            url = link_elem.get_attribute('href') if link_elem else None
            if url and not url.startswith('http'):
                url = self.config.BASE_URL + url

            # 외부 ID (URL에서 추출)
            external_id = None
            if url:
                match = re.search(r'/(\d+)_zpid/', url)
                if match:
                    external_id = match.group(1)

            # 썸네일 이미지
            img_elem = card.query_selector('img[src*="photos"]')
            thumbnail_url = img_elem.get_attribute('src') if img_elem else None

            # 상세 페이지로 이동하여 전체 정보 추출
            if url:
                page.goto(url)
                time.sleep(2)

                details = self._extract_property_details(page)

                return {
                    'external_id': external_id or f"{title}_{price}".replace(' ', '_')[:50],
                    'title': title,
                    'address': address,
                    'price': price,
                    'bedrooms': bedrooms,
                    'bathrooms': bathrooms,
                    'area_sqft': area,
                    'url': url,
                    'thumbnail_url': thumbnail_url,
                    **details
                }

            return None

        except Exception as e:
            logger.error(f"카드 추출 에러: {e}")
            return None

    def _extract_property_details(self, page: Page) -> Dict:
        """매물 상세 정보 추출"""
        details = {
            'description': '',
            'features': '',
            'year_built': None,
            'parking_spaces': None,
            'floor': None,
            'total_floors': None,
            'property_type': 'apartment',
            'latitude': None,
            'longitude': None,
            'image_urls': [],
            'floorplan_url': None
        }

        try:
            # 설명
            desc_elem = page.query_selector('.home-description, [data-test="description"]')
            if desc_elem:
                details['description'] = desc_elem.inner_text().strip()

            # 특징
            features_list = page.query_selector_all('.property-features li, .home-facts li')
            features = [f.inner_text().strip() for f in features_list]
            details['features'] = ', '.join(features)

            # 건축 연도
            year_elem = page.query_selector('span:has-text("Year built")')
            if year_elem:
                parent = year_elem.evaluate_handle('el => el.parentElement')
                year_text = parent.inner_text()
                match = re.search(r'\d{4}', year_text)
                if match:
                    details['year_built'] = int(match.group())

            # 주차
            parking_elem = page.query_selector('span:has-text("Parking")')
            if parking_elem:
                parent = parking_elem.evaluate_handle('el => el.parentElement')
                parking_text = parent.inner_text()
                match = re.search(r'(\d+)', parking_text)
                if match:
                    details['parking_spaces'] = int(match.group(1))

            # 이미지 URL 추출
            image_elems = page.query_selector_all('.media-stream img, [data-test="property-images"] img')
            details['image_urls'] = [
                img.get_attribute('src') or img.get_attribute('data-src')
                for img in image_elems
                if img.get_attribute('src') or img.get_attribute('data-src')
            ]

            # 평면도 URL
            floorplan_elem = page.query_selector('img[alt*="floor plan"], img[src*="floorplan"]')
            if floorplan_elem:
                details['floorplan_url'] = floorplan_elem.get_attribute('src')

            # 지도 좌표 (data attribute나 스크립트에서 추출)
            try:
                coords_script = page.query_selector('script:has-text("latitude")')
                if coords_script:
                    script_content = coords_script.inner_text()
                    lat_match = re.search(r'"latitude":\s*([-\d.]+)', script_content)
                    lon_match = re.search(r'"longitude":\s*([-\d.]+)', script_content)

                    if lat_match and lon_match:
                        details['latitude'] = float(lat_match.group(1))
                        details['longitude'] = float(lon_match.group(1))
            except:
                pass

        except Exception as e:
            logger.error(f"상세 정보 추출 에러: {e}")

        return details

    def _parse_price(self, price_text: str) -> float:
        """가격 파싱"""
        try:
            # "$1,234,567" -> 1234567
            price_str = re.sub(r'[^\d.]', '', price_text)
            return float(price_str) if price_str else 0
        except:
            return 0

    def _parse_details(self, details_text: str) -> tuple:
        """침실/욕실/면적 파싱"""
        bedrooms = bathrooms = area = None

        try:
            # "3 beds, 2 baths, 1,500 sqft"
            bed_match = re.search(r'(\d+)\s*bed', details_text, re.IGNORECASE)
            bath_match = re.search(r'([\d.]+)\s*bath', details_text, re.IGNORECASE)
            area_match = re.search(r'([\d,]+)\s*sqft', details_text, re.IGNORECASE)

            if bed_match:
                bedrooms = int(bed_match.group(1))
            if bath_match:
                bathrooms = float(bath_match.group(1))
            if area_match:
                area = float(area_match.group(1).replace(',', ''))

        except Exception as e:
            logger.warning(f"상세 정보 파싱 에러: {e}")

        return bedrooms, bathrooms, area

import logging
from datetime import datetime
from sqlalchemy.orm import Session

from config import settings
from database.models import init_db
from database.crud import PropertyCRUD, PropertyImageCRUD, PriceHistoryCRUD
from scraper.property_scraper import PropertyScraper
from scraper.image_manager import ImageManager

def setup_logger():
    """로거 설정"""
    settings.ensure_directories()

    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler(
                f'{settings.LOGS_DIR}/scraper_{datetime.now().strftime("%Y%m%d")}.log'
            ),
            logging.StreamHandler()
        ]
    )

def convert_area(sqft: float) -> tuple:
    """면적 변환 (sqft -> sqm, pyeong)"""
    sqm = sqft * 0.092903 if sqft else None
    pyeong = sqm * 0.3025 if sqm else None
    return sqm, pyeong

def main():
    """메인 실행 함수"""
    setup_logger()
    logger = logging.getLogger(__name__)

    try:
        logger.info("=" * 60)
        logger.info("부동산 매물 스크래퍼 시작")
        logger.info("=" * 60)

        # 데이터베이스 초기화
        SessionLocal = init_db(settings.DATABASE_URL)
        db: Session = SessionLocal()

        # 스크래퍼 초기화
        scraper = PropertyScraper(settings)
        image_manager = ImageManager(
            settings.PROPERTIES_DIR,
            settings.FLOORPLANS_DIR,
            settings.GALLERIES_DIR
        )

        new_properties = []

        # 위치 및 매물 타입별 검색
        for location in settings.LOCATIONS:
            for property_type in settings.PROPERTY_TYPES:
                logger.info(f"\n🔍 검색: {property_type} in {location}")

                # 스크래핑
                properties = scraper.scrape_properties(location, property_type)

                # 데이터베이스 저장
                for prop_data in properties:
                    # 가격 필터링
                    if prop_data['price'] < settings.MIN_PRICE or prop_data['price'] > settings.MAX_PRICE:
                        continue

                    # 면적 변환
                    area_sqm, area_pyeong = convert_area(prop_data.get('area_sqft'))

                    # 매물 저장
                    property = PropertyCRUD.create(
                        db,
                        external_id=prop_data['external_id'],
                        title=prop_data['title'],
                        address=prop_data['address'],
                        city=location,
                        property_type=property_type,
                        price=prop_data['price'],
                        bedrooms=prop_data.get('bedrooms'),
                        bathrooms=prop_data.get('bathrooms'),
                        area_sqft=prop_data.get('area_sqft'),
                        area_sqm=area_sqm,
                        area_pyeong=area_pyeong,
                        description=prop_data.get('description'),
                        features=prop_data.get('features'),
                        year_built=prop_data.get('year_built'),
                        parking_spaces=prop_data.get('parking_spaces'),
                        latitude=prop_data.get('latitude'),
                        longitude=prop_data.get('longitude'),
                        url=prop_data.get('url')
                    )

                    if property:  # 새로운 매물인 경우
                        new_properties.append(prop_data)

                        # 이미지 다운로드
                        image_urls = prop_data.get('image_urls', [])
                        if image_urls:
                            downloaded_images = image_manager.download_property_images(
                                image_urls,
                                prop_data['external_id'],
                                settings.MAX_IMAGES_PER_PROPERTY
                            )

                            # 이미지 레코드 저장
                            for img_data in downloaded_images:
                                PropertyImageCRUD.create(db, property.id, **img_data)

                            # 이미지 개수 업데이트
                            PropertyCRUD.update(db, property.id, image_count=len(downloaded_images))

                            # 갤러리 생성
                            if settings.CREATE_GALLERY and len(downloaded_images) > 1:
                                image_paths = [img['local_path'] for img in downloaded_images]
                                gallery_path = image_manager.create_gallery(
                                    image_paths,
                                    prop_data['external_id'],
                                    settings.GALLERY_GRID_SIZE
                                )
                                if gallery_path:
                                    PropertyCRUD.update(db, property.id, gallery_path=gallery_path)

                        # 평면도 다운로드
                        if settings.DOWNLOAD_FLOORPLANS and prop_data.get('floorplan_url'):
                            floorplan_data = image_manager.download_floorplan(
                                prop_data['floorplan_url'],
                                prop_data['external_id']
                            )
                            if floorplan_data:
                                PropertyImageCRUD.create(db, property.id, **floorplan_data)
                                PropertyCRUD.update(db, property.id, has_floorplan=True)

                        logger.info(f"✅ 새 매물 저장: {property.title}")

                    else:
                        # 기존 매물 - 가격 변동 체크
                        existing = db.query(Property).filter(
                            Property.external_id == prop_data['external_id']
                        ).first()

                        if existing and existing.price != prop_data['price']:
                            PriceHistoryCRUD.create(db, existing.id, prop_data['price'])
                            logger.info(f"💰 가격 변동: {existing.title} "
                                      f"${existing.price:,.0f} → ${prop_data['price']:,.0f}")

        # 요약 출력
        print("\n" + "=" * 60)
        print("📊 수집 결과")
        print("=" * 60)
        print(f"새로운 매물: {len(new_properties)}개")
        print(f"총 매물 수: {db.query(Property).count()}개")

        if new_properties:
            print("\n🏠 새로운 매물:")
            for idx, prop in enumerate(new_properties[:10], 1):
                print(f"{idx}. {prop['title']}")
                print(f"   💰 ${prop['price']:,.0f} | "
                      f"🛏️  {prop.get('bedrooms', 'N/A')} beds | "
                      f"🛁 {prop.get('bathrooms', 'N/A')} baths")
                if prop.get('area_sqft'):
                    area_sqm, area_pyeong = convert_area(prop['area_sqft'])
                    print(f"   📏 {prop['area_sqft']:,.0f} sqft "
                          f"({area_sqm:.1f} ㎡ / {area_pyeong:.1f}평)")

        print("=" * 60 + "\n")

        logger.info("✅ 모든 작업 완료!")

        db.close()

    except Exception as e:
        logger.error(f"❌ 에러 발생: {e}", exc_info=True)

if __name__ == "__main__":
    main()
