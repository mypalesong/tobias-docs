요소 상호작용

클릭, 입력, 선택 등 페이지 요소와 상호작용하는 방법을 배워봅시다!

🖱️ 클릭​

기본 클릭​

# 요소 클릭
page.click('.button')

# 텍스트로 클릭
page.click('text=로그인')

# Role로 클릭
page.click('role=button[name="제출"]')

클릭 옵션​

# 더블 클릭
page.dblclick('.item')

# 우클릭
page.click('.menu', button='right')

# 특정 위치 클릭
page.click('.target', position={'x': 10, 'y': 20})

# 강제 클릭 (가려진 요소도)
page.click('.hidden-btn', force=True)

# 타임아웃
page.click('.slow-btn', timeout=5000)

⌨️ 텍스트 입력​

fill() - 권장​

# 입력 필드 채우기 (기존 내용 지우고)
page.fill('#username', 'myuser')
page.fill('input[name="email"]', 'user@example.com')

type() - 키보드 시뮬레이션​

# 한 글자씩 입력 (실제 타이핑처럼)
page.type('#search', 'playwright', delay=100)  # 100ms 간격

press() - 키 입력​

# 엔터 키
page.press('input[name="search"]', 'Enter')

# 단축키
page.press('body', 'Control+A')  # 전체 선택
page.press('body', 'Control+C')  # 복사

실전 예제: 로그인​

from playwright.sync_api import sync_playwright

def login_example():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()
        page.goto('https://example.com/login')

        # 사용자명 입력
        page.fill('input[name="username"]', 'testuser')

        # 비밀번호 입력
        page.fill('input[type="password"]', 'password123')

        # 로그인 버튼 클릭
        page.click('button[type="submit"]')

        # 로그인 완료 대기
        page.wait_for_url('**/dashboard')

        print(f"로그인 성공! 현재 URL: {page.url}")

        browser.close()

☑️ 체크박스 & 라디오​

# 체크박스 체크
page.check('#agree')

# 체크박스 체크 해제
page.uncheck('#newsletter')

# 상태 확인
is_checked = page.is_checked('#terms')

# 라디오 버튼 선택
page.check('input[value="option1"]')

📝 드롭다운 선택​

# value로 선택
page.select_option('select[name="country"]', value='kr')

# 라벨로 선택
page.select_option('#state', label='California')

# 인덱스로 선택
page.select_option('#month', index=5)  # 6월

# 여러 개 선택 (multiple)
page.select_option('#skills', ['python', 'javascript', 'sql'])

📤 파일 업로드​

# 단일 파일
page.set_input_files('input[type="file"]', 'path/to/file.pdf')

# 여러 파일
page.set_input_files('#upload', [
    'file1.jpg',
    'file2.jpg',
    'file3.jpg'
])

# 파일 제거
page.set_input_files('#upload', [])

🖱️ 호버 (Hover)​

# 마우스 오버
page.hover('.menu-item')

# 호버 후 나타나는 메뉴 클릭
page.hover('.dropdown')
page.click('.dropdown-menu >> text=설정')

⬆️⬇️ 스크롤​

# 요소까지 스크롤
element = page.query_selector('.footer')
element.scroll_into_view_if_needed()

# JavaScript로 스크롤
page.evaluate('window.scrollTo(0, document.body.scrollHeight)')

# 특정 위치로 스크롤
page.evaluate('window.scrollTo(0, 1000)')

무한 스크롤 처리​

def infinite_scroll(page, max_scrolls=10):
    """무한 스크롤 페이지 처리"""
    last_height = page.evaluate('document.body.scrollHeight')

    for _ in range(max_scrolls):
        # 끝까지 스크롤
        page.evaluate('window.scrollTo(0, document.body.scrollHeight)')

        # 로딩 대기
        page.wait_for_timeout(2000)

        # 새 높이 확인
        new_height = page.evaluate('document.body.scrollHeight')

        if new_height == last_height:
            break  # 더 이상 로드할 내용 없음

        last_height = new_height

# 사용 예
page.goto('https://infinite-scroll-site.com')
infinite_scroll(page)

🎯 드래그 앤 드롭​

# 드래그 앤 드롭
page.drag_and_drop('.source', '.target')

# 또는 수동으로
source = page.query_selector('.draggable')
target = page.query_selector('.droppable')

source_box = source.bounding_box()
target_box = target.bounding_box()

page.mouse.move(source_box['x'] + source_box['width'] / 2,
                source_box['y'] + source_box['height'] / 2)
page.mouse.down()
page.mouse.move(target_box['x'] + target_box['width'] / 2,
                target_box['y'] + target_box['height'] / 2)
page.mouse.up()

🔍 요소 상태 확인​

# 표시 여부
is_visible = page.is_visible('.modal')

# 숨김 여부
is_hidden = page.is_hidden('.loading')

# 활성화 여부
is_enabled = page.is_enabled('button[type="submit"]')

# 비활성화 여부
is_disabled = page.is_disabled('button.disabled')

# 편집 가능 여부
is_editable = page.is_editable('input[name="title"]')

📋 실전 예제: 폼 제출​

from playwright.sync_api import sync_playwright

def submit_contact_form():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()
        page.goto('https://example.com/contact')

        # 텍스트 입력
        page.fill('input[name="name"]', '홍길동')
        page.fill('input[name="email"]', 'hong@example.com')
        page.fill('input[name="phone"]', '010-1234-5678')

        # 드롭다운 선택
        page.select_option('select[name="subject"]', label='기술 지원')

        # 텍스트 영역
        page.fill('textarea[name="message"]', '문의 내용입니다.')

        # 체크박스
        page.check('input[name="agree"]')

        # 파일 업로드
        page.set_input_files('input[type="file"]', 'attachment.pdf')

        # 제출 버튼 클릭
        page.click('button[type="submit"]')

        # 성공 메시지 대기
        success = page.wait_for_selector('.success-message', timeout=5000)
        print(f"✅ 제출 완료: {success.inner_text()}")

        browser.close()

🎮 복잡한 인터랙션​

여러 동작 조합​

# 검색 -> 필터링 -> 정렬
page.goto('https://shopping-site.com')

# 검색
page.fill('input[name="search"]', 'laptop')
page.press('input[name="search"]', 'Enter')
page.wait_for_load_state('networkidle')

# 필터 적용
page.click('text="가격"')
page.fill('input[name="min_price"]', '500')
page.fill('input[name="max_price"]', '1500')
page.click('button:has-text("적용")')

# 정렬
page.select_option('select[name="sort"]', value='price_asc')
page.wait_for_load_state('networkidle')

# 결과 추출
products = page.query_selector_all('.product-item')
print(f"검색 결과: {len(products)}개")

💡 팁​

  1. 자동 대기 활용

    # Playwright는 요소가 준비될 때까지 자동 대기
    page.click('.button')  # 버튼이 클릭 가능할 때까지 대기

  2. force 옵션은 최후의 수단

    # 가급적 사용하지 말 것
    page.click('.hidden', force=True)

  3. 적절한 대기 사용

    # ❌ 좋지 않음
    page.wait_for_timeout(5000)

    # ✅ 좋음
    page.wait_for_selector('.result')
