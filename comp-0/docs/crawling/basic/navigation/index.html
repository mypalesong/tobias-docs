페이지 탐색

페이지 이동, 리로드, 히스토리 관리를 마스터해봅시다!

🧭 기본 탐색​

URL 이동​

from playwright.sync_api import sync_playwright

with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()

    # 기본 이동
    page.goto('https://example.com')

    # 타임아웃 설정 (기본 30초)
    page.goto('https://slow-site.com', timeout=60000)

    # 로드 이벤트 대기 조건
    page.goto('https://example.com', wait_until='networkidle')

    browser.close()

wait_until 옵션​

값설명사용 시기
load (기본)페이지 로드 완료일반적인 경우
domcontentloadedDOM 준비 완료빠른 스크래핑
networkidle네트워크 요청 없음SPA, AJAX 페이지
commit즉시 반환리다이렉트 확인
# SPA (Single Page Application)
page.goto('https://react-app.com', wait_until='networkidle')

# 빠른 스크래핑
page.goto('https://static-site.com', wait_until='domcontentloaded')

⬅️ 뒤로/앞으로 가기​

# 뒤로 가기
page.go_back()

# 앞으로 가기
page.go_forward()

# 타임아웃 설정
page.go_back(timeout=5000)

실전 예제: 페이지 탐색​

from playwright.sync_api import sync_playwright

def navigate_pages():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()

        # 첫 페이지
        page.goto('https://example.com')
        print(f"현재 URL: {page.url}")

        # 링크 클릭
        page.click('text="More information"')
        page.wait_for_load_state('networkidle')
        print(f"현재 URL: {page.url}")

        # 뒤로 가기
        page.go_back()
        print(f"뒤로 가기: {page.url}")

        # 앞으로 가기
        page.go_forward()
        print(f"앞으로 가기: {page.url}")

        browser.close()

새 탭 열기​

# 컨텍스트에서 새 페이지 생성
context = browser.new_context()
page1 = context.new_page()
page2 = context.new_page()

page1.goto('https://example.com')
page2.goto('https://google.com')

# 각 페이지 독립적으로 작업
print(f"Page 1: {page1.title()}")
print(f"Page 2: {page2.title()}")

팝업 처리​

# 팝업 대기 및 처리
with page.expect_popup() as popup_info:
    page.click('text="Open popup"')

popup = popup_info.value

# 팝업에서 작업
popup.goto('https://example.com')
print(popup.title())

# 팝업 닫기
popup.close()

🔗 링크 추출 및 순회​

from playwright.sync_api import sync_playwright

def crawl_links():
    with sync_playwright() as p:
        browser = p.chromium.launch()
        page = browser.new_page()
        page.goto('https://example.com')

        # 모든 링크 추출
        links = page.query_selector_all('a[href]')

        visited_urls = set()

        for link in links[:10]:  # 처음 10개만
            href = link.get_attribute('href')

            # 절대 URL로 변환
            from urllib.parse import urljoin
            full_url = urljoin(page.url, href)

            # 중복 제거
            if full_url in visited_urls:
                continue

            visited_urls.add(full_url)

            # 링크 방문
            try:
                page.goto(full_url, timeout=5000)
                print(f"✅ {page.title()}: {full_url}")

                # 데이터 추출
                # ...

            except Exception as e:
                print(f"❌ 실패: {full_url} - {e}")

        browser.close()
