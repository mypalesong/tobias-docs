import requests
from bs4 import BeautifulSoup
from typing import List, Dict, Optional
import time

class StaticScraper:
    """ì •ì  ì›¹í˜ì´ì§€ ìŠ¤í¬ë˜í¼ (BeautifulSoup)"""

    def __init__(
        self,
        user_agent: Optional[str] = None,
        proxy: Optional[str] = None,
        timeout: int = 30
    ):
        self.session = requests.Session()
        self.timeout = timeout

        # User-Agent ì„¤ì •
        headers = {
            'User-Agent': user_agent or 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
        self.session.headers.update(headers)

        # í”„ë¡ì‹œ ì„¤ì •
        if proxy:
            self.session.proxies = {
                'http': proxy,
                'https': proxy
            }

    def scrape(self, url: str, selectors: Dict[str, str]) -> List[Dict]:
        """ì›¹í˜ì´ì§€ ìŠ¤í¬ë˜í•‘"""
        try:
            response = self.session.get(url, timeout=self.timeout)
            response.raise_for_status()

            soup = BeautifulSoup(response.content, 'html.parser')

            results = []

            # ì»¨í…Œì´ë„ˆ ì„ íƒìê°€ ìˆìœ¼ë©´ ê° ì•„ì´í…œë³„ë¡œ ì¶”ì¶œ
            if 'container' in selectors:
                containers = soup.select(selectors['container'])

                for container in containers:
                    item = {}
                    for key, selector in selectors.items():
                        if key == 'container':
                            continue

                        element = container.select_one(selector)
                        if element:
                            # href, src ë“± ì†ì„± ì¶”ì¶œ
                            if selector.endswith('[href]'):
                                item[key] = element.get('href')
                            elif selector.endswith('[src]'):
                                item[key] = element.get('src')
                            else:
                                item[key] = element.get_text(strip=True)

                    results.append(item)

            # ë‹¨ì¼ ìš”ì†Œ ì¶”ì¶œ
            else:
                item = {}
                for key, selector in selectors.items():
                    elements = soup.select(selector)

                    if len(elements) == 1:
                        item[key] = elements[0].get_text(strip=True)
                    else:
                        item[key] = [el.get_text(strip=True) for el in elements]

                results.append(item)

            return results

        except Exception as e:
            raise Exception(f"Scraping failed: {str(e)}")

    def scrape_table(self, url: str, table_selector: str = 'table') -> List[Dict]:
        """í…Œì´ë¸” ë°ì´í„° ì¶”ì¶œ"""
        response = self.session.get(url, timeout=self.timeout)
        soup = BeautifulSoup(response.content, 'html.parser')

        table = soup.select_one(table_selector)
        if not table:
            return []

        # í—¤ë” ì¶”ì¶œ
        headers = []
        thead = table.find('thead')
        if thead:
            headers = [th.get_text(strip=True) for th in thead.find_all('th')]

        # ë°ì´í„° ì¶”ì¶œ
        rows = []
        tbody = table.find('tbody') or table
        for tr in tbody.find_all('tr'):
            cells = [td.get_text(strip=True) for td in tr.find_all(['td', 'th'])]
            if cells:
                if headers:
                    rows.append(dict(zip(headers, cells)))
                else:
                    rows.append({"data": cells})

        return rows

    def scrape_links(self, url: str, link_selector: str = 'a') -> List[str]:
        """ë§í¬ ì¶”ì¶œ"""
        response = self.session.get(url, timeout=self.timeout)
        soup = BeautifulSoup(response.content, 'html.parser')

        links = []
        for a in soup.select(link_selector):
            href = a.get('href')
            if href:
                # ì ˆëŒ€ URLë¡œ ë³€í™˜
                from urllib.parse import urljoin
                absolute_url = urljoin(url, href)
                links.append(absolute_url)

        return links

from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait

from typing import List, Dict

class DynamicScraper:
    """ë™ì  ì›¹í˜ì´ì§€ ìŠ¤í¬ë˜í¼ (Selenium)"""

    def __init__(self, headless: bool = True, proxy: str = None):
        options = Options()

        if headless:
            options.add_argument('--headless')

        options.add_argument('--no-sandbox')
        options.add_argument('--disable-dev-shm-usage')
        options.add_argument('--disable-gpu')
        options.add_argument('--window-size=1920,1080')

        # User-Agent
        options.add_argument('user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36')

        # ë´‡ íƒì§€ ìš°íšŒ
        options.add_experimental_option("excludeSwitches", ["enable-automation"])
        options.add_experimental_option('useAutomationExtension', False)

        # í”„ë¡ì‹œ
        if proxy:
            options.add_argument(f'--proxy-server={proxy}')

        self.driver = webdriver.Chrome(options=options)

        # ë´‡ íƒì§€ ìš°íšŒ ìŠ¤í¬ë¦½íŠ¸
        self.driver.execute_cdp_cmd('Page.addScriptToEvaluateOnNewDocument', {
            'source': '''
                Object.defineProperty(navigator, 'webdriver', {
                    get: () => undefined
                })
            '''
        })

    def scrape(
        self,
        url: str,
        selectors: Dict[str, str],
        wait_selector: str = None,
        scroll_to_bottom: bool = False
    ) -> List[Dict]:
        """ë™ì  ì›¹í˜ì´ì§€ ìŠ¤í¬ë˜í•‘"""
        try:
            self.driver.get(url)

            # íŠ¹ì • ìš”ì†Œê°€ ë¡œë“œë  ë•Œê¹Œì§€ ëŒ€ê¸°
            if wait_selector:
                WebDriverWait(self.driver, 10).until(
                    EC.presence_of_element_located((By.CSS_SELECTOR, wait_selector))
                )

            # ìŠ¤í¬ë¡¤ (ë¬´í•œ ìŠ¤í¬ë¡¤ í˜ì´ì§€ìš©)
            if scroll_to_bottom:
                self._scroll_to_bottom()

            results = []

            # ì»¨í…Œì´ë„ˆ ì„ íƒìê°€ ìˆìœ¼ë©´ ê° ì•„ì´í…œë³„ë¡œ ì¶”ì¶œ
            if 'container' in selectors:
                containers = self.driver.find_elements(By.CSS_SELECTOR, selectors['container'])

                for container in containers:
                    item = {}
                    for key, selector in selectors.items():
                        if key == 'container':
                            continue

                        try:
                            element = container.find_element(By.CSS_SELECTOR, selector)

                            # ì†ì„± ì¶”ì¶œ
                            if selector.endswith('[href]'):
                                item[key] = element.get_attribute('href')
                            elif selector.endswith('[src]'):
                                item[key] = element.get_attribute('src')
                            else:
                                item[key] = element.text

                        except:
                            item[key] = None

                    results.append(item)

            return results

        except Exception as e:
            raise Exception(f"Dynamic scraping failed: {str(e)}")

    def _scroll_to_bottom(self, pause_time: float = 2.0):
        """í˜ì´ì§€ ëê¹Œì§€ ìŠ¤í¬ë¡¤"""
        last_height = self.driver.execute_script("return document.body.scrollHeight")

        while True:
            # ìŠ¤í¬ë¡¤ ë‹¤ìš´
            self.driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
            time.sleep(pause_time)

            # ìƒˆë¡œìš´ ë†’ì´ ê³„ì‚°
            new_height = self.driver.execute_script("return document.body.scrollHeight")

            if new_height == last_height:
                break

            last_height = new_height

    def click_and_scrape(self, url: str, click_selector: str, data_selector: str) -> str:
        """ë²„íŠ¼ í´ë¦­ í›„ ë°ì´í„° ì¶”ì¶œ"""
        self.driver.get(url)

        # ë²„íŠ¼ í´ë¦­
        button = WebDriverWait(self.driver, 10).until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, click_selector))
        )
        button.click()

        # ë°ì´í„° ì¶”ì¶œ
        time.sleep(1)
        element = self.driver.find_element(By.CSS_SELECTOR, data_selector)
        return element.text

    def close(self):
        """ë¸Œë¼ìš°ì € ì¢…ë£Œ"""
        if self.driver:
            self.driver.quit()

    def __del__(self):
        self.close()

from sqlalchemy.orm import Session
import models
from static_scraper import StaticScraper
from dynamic_scraper import DynamicScraper
import time
from datetime import datetime

class ScraperService:
    def __init__(self, db: Session):
        self.db = db

    def execute_job(self, job_id: int) -> models.ScrapingResult:
        """ìŠ¤í¬ë˜í•‘ ì‘ì—… ì‹¤í–‰"""
        job = self.db.query(models.ScrapingJob).filter(
            models.ScrapingJob.id == job_id
        ).first()

        if not job:
            raise ValueError("Job not found")

        start_time = time.time()
        result = models.ScrapingResult(job_id=job_id)

        try:
            # ìŠ¤í¬ë˜í¼ ì„ íƒ
            if job.scraper_type == "static":
                scraper = StaticScraper(
                    proxy=os.getenv('PROXY_URL') if job.use_proxy else None
                )
                data = scraper.scrape(job.url, job.selectors)

            elif job.scraper_type == "dynamic":
                scraper = DynamicScraper(
                    headless=True,
                    proxy=os.getenv('PROXY_URL') if job.use_proxy else None
                )
                try:
                    data = scraper.scrape(
                        job.url,
                        job.selectors,
                        wait_selector=job.selectors.get('wait_for')
                    )
                finally:
                    scraper.close()

            else:
                raise ValueError(f"Unknown scraper type: {job.scraper_type}")

            # ê²°ê³¼ ì €ì¥
            result.status = "success"
            result.data = data
            result.items_count = len(data)

            # ì‘ì—… ìƒíƒœ ì—…ë°ì´íŠ¸
            job.last_run = datetime.utcnow()

        except Exception as e:
            result.status = "failed"
            result.error_message = str(e)

        finally:
            result.processing_time = time.time() - start_time
            self.db.add(result)
            self.db.commit()
            self.db.refresh(result)

        return result

    def execute_with_retry(self, job_id: int, max_retries: int = 3) -> models.ScrapingResult:
        """ì¬ì‹œë„ ë¡œì§ì´ ìˆëŠ” ìŠ¤í¬ë˜í•‘"""
        for attempt in range(max_retries):
            try:
                result = self.execute_job(job_id)

                if result.status == "success":
                    return result

                # ì‹¤íŒ¨ ì‹œ ëŒ€ê¸° í›„ ì¬ì‹œë„
                if attempt < max_retries - 1:
                    time.sleep(2 ** attempt)  # Exponential backoff

            except Exception as e:
                if attempt == max_retries - 1:
                    raise

                time.sleep(2 ** attempt)

        return result
