시나리오 15: 웹 크롤링/스크래핑 API

BeautifulSoup과 Selenium을 활용한 웹 스크래핑 서비스를 만들어봅시다!

📌 기능​

  • ✅ HTML 파싱 및 데이터 추출
  • ✅ JavaScript 렌더링 (Selenium)
  • ✅ 스크래핑 작업 스케줄링
  • ✅ 프록시 로테이션
  • ✅ 봇 탐지 우회
  • ✅ 데이터 정제 및 저장
  • ✅ 속도 제한 (Rate Limiting)
  • ✅ 에러 처리 및 재시도

sequenceDiagram
    participant U as User
    participant API as FastAPI
    participant Sch as Scheduler
    participant SS as StaticScraper
    participant DS as DynamicScraper
    participant PM as ProxyManager
    participant DB as Database

    Note over U,DB: 1. 스크래핑 작업 생성
    U->>API: POST /scraping/jobs<br/>{url, selectors, schedule}
    API->>DB: INSERT scraping_job
    DB-->>API: Job created
    API->>Sch: Schedule job
    API-->>U: 201 Created + Job info

    Note over U,DB: 2. 정적 페이지 스크래핑
    U->>API: POST /scraping/run/{job_id}
    API->>DB: SELECT job config
    DB-->>API: Job config
    API->>SS: scrape(url, selectors)
    SS->>PM: Get proxy (if enabled)
    PM-->>SS: Proxy URL
    SS->>SS: HTTP GET request
    SS->>SS: Parse HTML with BeautifulSoup
    SS->>SS: Extract data using selectors
    SS-->>API: Scraped data
    API->>DB: INSERT scraping_result
    API-->>U: 200 OK + Data

    Note over U,DB: 3. 동적 페이지 스크래핑 (JS)
    U->>API: POST /scraping/run/{job_id}
    API->>DS: scrape_dynamic()
    DS->>PM: Get proxy
    PM-->>DS: Proxy
    DS->>DS: Start Selenium WebDriver
    DS->>DS: Navigate to URL
    DS->>DS: Wait for JS rendering
    DS->>DS: Execute selectors
    DS->>DS: Extract data
    DS->>DS: Close WebDriver
    DS-->>API: Scraped data
    API->>DB: INSERT result
    API-->>U: 200 OK + Data

    Note over U,DB: 4. 스케줄된 실행 (백그라운드)
    Sch->>Sch: Cron trigger
    Sch->>DB: SELECT active jobs
    DB-->>Sch: Jobs to run
    loop For each job
        Sch->>SS: scrape()
        SS-->>Sch: Data
        Sch->>DB: INSERT result
    end

from sqlalchemy import Column, Integer, String, DateTime, Text, Boolean, JSON, Float
from sqlalchemy.sql import func
from database import Base

class ScrapingJob(Base):
    """스크래핑 작업"""
    __tablename__ = "scraping_jobs"

    id = Column(Integer, primary_key=True)
    name = Column(String)
    url = Column(String)
    scraper_type = Column(String)  # "static", "dynamic"
    selectors = Column(JSON)  # CSS 선택자
    schedule = Column(String, nullable=True)  # Cron 형식

    # 설정
    use_proxy = Column(Boolean, default=False)
    wait_time = Column(Float, default=1.0)  # 요청 간 대기 시간 (초)
    max_retries = Column(Integer, default=3)

    # 상태
    is_active = Column(Boolean, default=True)
    last_run = Column(DateTime(timezone=True), nullable=True)
    next_run = Column(DateTime(timezone=True), nullable=True)

    created_at = Column(DateTime(timezone=True), server_default=func.now())

class ScrapingResult(Base):
    """스크래핑 결과"""
    __tablename__ = "scraping_results"

    id = Column(Integer, primary_key=True)
    job_id = Column(Integer)
    data = Column(JSON)
    status = Column(String)  # "success", "failed"
    error_message = Column(Text, nullable=True)
    items_count = Column(Integer, default=0)
    processing_time = Column(Float)  # 초

    created_at = Column(DateTime(timezone=True), server_default=func.now())

static_scraper.py​

import requests
from bs4 import BeautifulSoup
from typing import List, Dict, Optional
import time

class StaticScraper:
    """정적 웹페이지 스크래퍼 (BeautifulSoup)"""

    def __init__(
        self,
        user_agent: Optional[str] = None,
        proxy: Optional[str] = None,
        timeout: int = 30
    ):
        self.session = requests.Session()
        self.timeout = timeout

        # User-Agent 설정
        headers = {
            'User-Agent': user_agent or 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
        self.session.headers.update(headers)

        # 프록시 설정
        if proxy:
            self.session.proxies = {
                'http': proxy,
                'https': proxy
            }

    def scrape(self, url: str, selectors: Dict[str, str]) -> List[Dict]:
        """웹페이지 스크래핑"""
        try:
            response = self.session.get(url, timeout=self.timeout)
            response.raise_for_status()

            soup = BeautifulSoup(response.content, 'html.parser')

            results = []

            # 컨테이너 선택자가 있으면 각 아이템별로 추출
            if 'container' in selectors:
                containers = soup.select(selectors['container'])

                for container in containers:
                    item = {}
                    for key, selector in selectors.items():
                        if key == 'container':
                            continue

                        element = container.select_one(selector)
                        if element:
                            # href, src 등 속성 추출
                            if selector.endswith('[href]'):
                                item[key] = element.get('href')
                            elif selector.endswith('[src]'):
                                item[key] = element.get('src')
                            else:
                                item[key] = element.get_text(strip=True)

                    results.append(item)

            # 단일 요소 추출
            else:
                item = {}
                for key, selector in selectors.items():
                    elements = soup.select(selector)

                    if len(elements) == 1:
                        item[key] = elements[0].get_text(strip=True)
                    else:
                        item[key] = [el.get_text(strip=True) for el in elements]

                results.append(item)

            return results

        except Exception as e:
            raise Exception(f"Scraping failed: {str(e)}")

    def scrape_table(self, url: str, table_selector: str = 'table') -> List[Dict]:
        """테이블 데이터 추출"""
        response = self.session.get(url, timeout=self.timeout)
        soup = BeautifulSoup(response.content, 'html.parser')

        table = soup.select_one(table_selector)
        if not table:
            return []

        # 헤더 추출
        headers = []
        thead = table.find('thead')
        if thead:
            headers = [th.get_text(strip=True) for th in thead.find_all('th')]

        # 데이터 추출
        rows = []
        tbody = table.find('tbody') or table
        for tr in tbody.find_all('tr'):
            cells = [td.get_text(strip=True) for td in tr.find_all(['td', 'th'])]
            if cells:
                if headers:
                    rows.append(dict(zip(headers, cells)))
                else:
                    rows.append({"data": cells})

        return rows

    def scrape_links(self, url: str, link_selector: str = 'a') -> List[str]:
        """링크 추출"""
        response = self.session.get(url, timeout=self.timeout)
        soup = BeautifulSoup(response.content, 'html.parser')

        links = []
        for a in soup.select(link_selector):
            href = a.get('href')
            if href:
                # 절대 URL로 변환
                from urllib.parse import urljoin
                absolute_url = urljoin(url, href)
                links.append(absolute_url)

        return links

dynamic_scraper.py​

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.options import Options
from typing import List, Dict
import time

class DynamicScraper:
    """동적 웹페이지 스크래퍼 (Selenium)"""

    def __init__(self, headless: bool = True, proxy: str = None):
        options = Options()

        if headless:
            options.add_argument('--headless')

        options.add_argument('--no-sandbox')
        options.add_argument('--disable-dev-shm-usage')
        options.add_argument('--disable-gpu')
        options.add_argument('--window-size=1920,1080')

        # User-Agent
        options.add_argument('user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36')

        # 봇 탐지 우회
        options.add_experimental_option("excludeSwitches", ["enable-automation"])
        options.add_experimental_option('useAutomationExtension', False)

        # 프록시
        if proxy:
            options.add_argument(f'--proxy-server={proxy}')

        self.driver = webdriver.Chrome(options=options)

        # 봇 탐지 우회 스크립트
        self.driver.execute_cdp_cmd('Page.addScriptToEvaluateOnNewDocument', {
            'source': '''
                Object.defineProperty(navigator, 'webdriver', {
                    get: () => undefined
                })
            '''
        })

    def scrape(
        self,
        url: str,
        selectors: Dict[str, str],
        wait_selector: str = None,
        scroll_to_bottom: bool = False
    ) -> List[Dict]:
        """동적 웹페이지 스크래핑"""
        try:
            self.driver.get(url)

            # 특정 요소가 로드될 때까지 대기
            if wait_selector:
                WebDriverWait(self.driver, 10).until(
                    EC.presence_of_element_located((By.CSS_SELECTOR, wait_selector))
                )

            # 스크롤 (무한 스크롤 페이지용)
            if scroll_to_bottom:
                self._scroll_to_bottom()

            results = []

            # 컨테이너 선택자가 있으면 각 아이템별로 추출
            if 'container' in selectors:
                containers = self.driver.find_elements(By.CSS_SELECTOR, selectors['container'])

                for container in containers:
                    item = {}
                    for key, selector in selectors.items():
                        if key == 'container':
                            continue

                        try:
                            element = container.find_element(By.CSS_SELECTOR, selector)

                            # 속성 추출
                            if selector.endswith('[href]'):
                                item[key] = element.get_attribute('href')
                            elif selector.endswith('[src]'):
                                item[key] = element.get_attribute('src')
                            else:
                                item[key] = element.text

                        except:
                            item[key] = None

                    results.append(item)

            return results

        except Exception as e:
            raise Exception(f"Dynamic scraping failed: {str(e)}")

    def _scroll_to_bottom(self, pause_time: float = 2.0):
        """페이지 끝까지 스크롤"""
        last_height = self.driver.execute_script("return document.body.scrollHeight")

        while True:
            # 스크롤 다운
            self.driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
            time.sleep(pause_time)

            # 새로운 높이 계산
            new_height = self.driver.execute_script("return document.body.scrollHeight")

            if new_height == last_height:
                break

            last_height = new_height

    def click_and_scrape(self, url: str, click_selector: str, data_selector: str) -> str:
        """버튼 클릭 후 데이터 추출"""
        self.driver.get(url)

        # 버튼 클릭
        button = WebDriverWait(self.driver, 10).until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, click_selector))
        )
        button.click()

        # 데이터 추출
        time.sleep(1)
        element = self.driver.find_element(By.CSS_SELECTOR, data_selector)
        return element.text

    def close(self):
        """브라우저 종료"""
        if self.driver:
            self.driver.quit()

    def __del__(self):
        self.close()

scraper_service.py​

from sqlalchemy.orm import Session
import models
from static_scraper import StaticScraper
from dynamic_scraper import DynamicScraper
import time
from datetime import datetime

class ScraperService:
    def __init__(self, db: Session):
        self.db = db

    def execute_job(self, job_id: int) -> models.ScrapingResult:
        """스크래핑 작업 실행"""
        job = self.db.query(models.ScrapingJob).filter(
            models.ScrapingJob.id == job_id
        ).first()

        if not job:
            raise ValueError("Job not found")

        start_time = time.time()
        result = models.ScrapingResult(job_id=job_id)

        try:
            # 스크래퍼 선택
            if job.scraper_type == "static":
                scraper = StaticScraper(
                    proxy=os.getenv('PROXY_URL') if job.use_proxy else None
                )
                data = scraper.scrape(job.url, job.selectors)

            elif job.scraper_type == "dynamic":
                scraper = DynamicScraper(
                    headless=True,
                    proxy=os.getenv('PROXY_URL') if job.use_proxy else None
                )
                try:
                    data = scraper.scrape(
                        job.url,
                        job.selectors,
                        wait_selector=job.selectors.get('wait_for')
                    )
                finally:
                    scraper.close()

            else:
                raise ValueError(f"Unknown scraper type: {job.scraper_type}")

            # 결과 저장
            result.status = "success"
            result.data = data
            result.items_count = len(data)

            # 작업 상태 업데이트
            job.last_run = datetime.utcnow()

        except Exception as e:
            result.status = "failed"
            result.error_message = str(e)

        finally:
            result.processing_time = time.time() - start_time
            self.db.add(result)
            self.db.commit()
            self.db.refresh(result)

        return result

    def execute_with_retry(self, job_id: int, max_retries: int = 3) -> models.ScrapingResult:
        """재시도 로직이 있는 스크래핑"""
        for attempt in range(max_retries):
            try:
                result = self.execute_job(job_id)

                if result.status == "success":
                    return result

                # 실패 시 대기 후 재시도
                if attempt < max_retries - 1:
                    time.sleep(2 ** attempt)  # Exponential backoff

            except Exception as e:
                if attempt == max_retries - 1:
                    raise

                time.sleep(2 ** attempt)

        return result

main.py​

from fastapi import FastAPI, Depends, HTTPException, BackgroundTasks
from sqlalchemy.orm import Session
from typing import List, Optional
import models, schemas
from database import engine, get_db
from scraper_service import ScraperService
from static_scraper import StaticScraper

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="Web Scraping API")

# ==================== 작업 관리 ====================

@app.post("/scraping/jobs", response_model=schemas.ScrapingJob)
def create_job(
    job: schemas.ScrapingJobCreate,
    db: Session = Depends(get_db)
):
    """스크래핑 작업 생성"""
    db_job = models.ScrapingJob(**job.dict())
    db.add(db_job)
    db.commit()
    db.refresh(db_job)

    return db_job

@app.get("/scraping/jobs", response_model=List[schemas.ScrapingJob])
def list_jobs(
    is_active: Optional[bool] = None,
    db: Session = Depends(get_db)
):
    """작업 목록"""
    query = db.query(models.ScrapingJob)

    if is_active is not None:
        query = query.filter(models.ScrapingJob.is_active == is_active)

    return query.all()

@app.get("/scraping/jobs/{job_id}", response_model=schemas.ScrapingJob)
def get_job(job_id: int, db: Session = Depends(get_db)):
    """작업 조회"""
    job = db.query(models.ScrapingJob).filter(
        models.ScrapingJob.id == job_id
    ).first()

    if not job:
        raise HTTPException(status_code=404, detail="Job not found")

    return job

@app.delete("/scraping/jobs/{job_id}")
def delete_job(job_id: int, db: Session = Depends(get_db)):
    """작업 삭제"""
    job = db.query(models.ScrapingJob).filter(
        models.ScrapingJob.id == job_id
    ).first()

    if not job:
        raise HTTPException(status_code=404, detail="Job not found")

    db.delete(job)
    db.commit()

    return {"message": "Job deleted"}

# ==================== 작업 실행 ====================

@app.post("/scraping/jobs/{job_id}/execute")
def execute_job(
    job_id: int,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)
):
    """작업 실행"""
    job = db.query(models.ScrapingJob).filter(
        models.ScrapingJob.id == job_id
    ).first()

    if not job:
        raise HTTPException(status_code=404, detail="Job not found")

    # 백그라운드에서 실행
    background_tasks.add_task(run_scraping_job, job_id, db)

    return {"message": "Job started", "job_id": job_id}

def run_scraping_job(job_id: int, db: Session):
    """스크래핑 작업 실행"""
    service = ScraperService(db)
    service.execute_with_retry(job_id)

# ==================== 결과 조회 ====================

@app.get("/scraping/results", response_model=List[schemas.ScrapingResult])
def list_results(
    job_id: Optional[int] = None,
    status: Optional[str] = None,
    skip: int = 0,
    limit: int = 20,
    db: Session = Depends(get_db)
):
    """결과 목록"""
    query = db.query(models.ScrapingResult)

    if job_id:
        query = query.filter(models.ScrapingResult.job_id == job_id)

    if status:
        query = query.filter(models.ScrapingResult.status == status)

    results = query.order_by(
        models.ScrapingResult.created_at.desc()
    ).offset(skip).limit(limit).all()

    return results

@app.get("/scraping/results/{result_id}", response_model=schemas.ScrapingResult)
def get_result(result_id: int, db: Session = Depends(get_db)):
    """결과 조회"""
    result = db.query(models.ScrapingResult).filter(
        models.ScrapingResult.id == result_id
    ).first()

    if not result:
        raise HTTPException(status_code=404, detail="Result not found")

    return result

# ==================== 즉시 스크래핑 ====================

@app.post("/scraping/quick")
def quick_scrape(request: schemas.QuickScrapeRequest):
    """즉시 스크래핑 (작업 저장 없이)"""
    scraper = StaticScraper()

    try:
        data = scraper.scrape(request.url, request.selectors)

        return {
            "status": "success",
            "data": data,
            "items_count": len(data)
        }

    except Exception as e:
        return {
            "status": "failed",
            "error": str(e)
        }

@app.post("/scraping/extract-links")
def extract_links(url: str):
    """링크 추출"""
    scraper = StaticScraper()

    try:
        links = scraper.scrape_links(url)

        return {
            "url": url,
            "links": links,
            "count": len(links)
        }

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/scraping/extract-table")
def extract_table(url: str, table_selector: str = 'table'):
    """테이블 데이터 추출"""
    scraper = StaticScraper()

    try:
        data = scraper.scrape_table(url, table_selector)

        return {
            "url": url,
            "data": data,
            "rows": len(data)
        }

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

🔑 핵심 개념​

정적 vs 동적 스크래핑​

  • 정적: BeautifulSoup - 빠르고 가벼움
  • 동적: Selenium - JavaScript 렌더링 필요 시

봇 탐지 우회​

  • User-Agent 설정
  • webdriver 속성 제거
  • 헤드리스 모드 사용
