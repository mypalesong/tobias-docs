from transformers import AutoImageProcessor, AutoModelForImageClassification
from PIL import Image
import torch

class ImageClassifier:
    def __init__(self, model_name: str = "google/vit-base-patch16-224"):
        """ì´ë¯¸ì§€ ë¶„ë¥˜ ëª¨ë¸ ë¡œë“œ"""
        self.processor = AutoImageProcessor.from_pretrained(model_name)
        self.model = AutoModelForImageClassification.from_pretrained(model_name)
        self.model_name = model_name

    def classify(self, image: Image.Image, top_k: int = 5) -> list:
        """ì´ë¯¸ì§€ ë¶„ë¥˜"""
        inputs = self.processor(images=image, return_tensors="pt")

        with torch.no_grad():
            outputs = self.model(**inputs)

        logits = outputs.logits
        probabilities = torch.nn.functional.softmax(logits, dim=-1)

        # Top K ê²°ê³¼
        top_probs, top_indices = torch.topk(probabilities, top_k)

        results = []
        for prob, idx in zip(top_probs[0], top_indices[0]):
            label = self.model.config.id2label[idx.item()]
            results.append({
                "label": label,
                "confidence": float(prob)
            })

        return results

from transformers import pipeline

class SentimentAnalyzer:
    def __init__(self, model_name: str = "distilbert-base-uncased-finetuned-sst-2-english"):
        """ê°ì • ë¶„ì„ ëª¨ë¸ ë¡œë“œ"""
        self.classifier = pipeline(
            "sentiment-analysis",
            model=model_name,
            tokenizer=model_name
        )
        self.model_name = model_name

    def analyze(self, text: str) -> dict:
        """ê°ì • ë¶„ì„"""
        result = self.classifier(text)[0]

        return {
            "sentiment": result['label'],
            "confidence": float(result['score'])
        }

    def analyze_batch(self, texts: list) -> list:
        """ë°°ì¹˜ ê°ì • ë¶„ì„"""
        results = self.classifier(texts)

        return [
            {
                "text": text,
                "sentiment": result['label'],
                "confidence": float(result['score'])
            }
            for text, result in zip(texts, results)
        ]

from transformers import AutoTokenizer, AutoModelForCausalLM
import torch

class TextGenerator:
    def __init__(self, model_name: str = "gpt2"):
        """í…ìŠ¤íŠ¸ ìƒì„± ëª¨ë¸ ë¡œë“œ"""
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForCausalLM.from_pretrained(model_name)
        self.model_name = model_name

        # Pad token ì„¤ì •
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token

    def generate(
        self,
        prompt: str,
        max_length: int = 100,
        temperature: float = 0.7,
        top_p: float = 0.9,
        num_return_sequences: int = 1
    ) -> list:
        """í…ìŠ¤íŠ¸ ìƒì„±"""
        inputs = self.tokenizer(prompt, return_tensors="pt")

        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_length=max_length,
                temperature=temperature,
                top_p=top_p,
                num_return_sequences=num_return_sequences,
                do_sample=True,
                pad_token_id=self.tokenizer.eos_token_id
            )

        results = []
        for output in outputs:
            text = self.tokenizer.decode(output, skip_special_tokens=True)
            results.append(text)

        return results

from transformers import MarianMTModel, MarianTokenizer

class Translator:
    def __init__(self, source_lang: str = "en", target_lang: str = "ko"):
        """ë²ˆì—­ ëª¨ë¸ ë¡œë“œ"""
        model_name = f"Helsinki-NLP/opus-mt-{source_lang}-{target_lang}"
        self.tokenizer = MarianTokenizer.from_pretrained(model_name)
        self.model = MarianMTModel.from_pretrained(model_name)
        self.model_name = model_name
        self.source_lang = source_lang
        self.target_lang = target_lang

    def translate(self, text: str) -> str:
        """í…ìŠ¤íŠ¸ ë²ˆì—­"""
        inputs = self.tokenizer(text, return_tensors="pt", padding=True)

        with torch.no_grad():
            outputs = self.model.generate(**inputs)

        translated = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        return translated

    def translate_batch(self, texts: list) -> list:
        """ë°°ì¹˜ ë²ˆì—­"""
        inputs = self.tokenizer(texts, return_tensors="pt", padding=True, truncation=True)

        with torch.no_grad():
            outputs = self.model.generate(**inputs)

        translated = [
            self.tokenizer.decode(output, skip_special_tokens=True)
            for output in outputs
        ]

        return translated

import openai
import os
from typing import List

class OpenAIService:
    def __init__(self):
        openai.api_key = os.getenv('OPENAI_API_KEY')
        self.enabled = bool(openai.api_key)

    def chat_completion(
        self,
        messages: List[dict],
        model: str = "gpt-3.5-turbo",
        temperature: float = 0.7,
        max_tokens: int = 500
    ) -> dict:
        """ChatGPT API í˜¸ì¶œ"""
        if not self.enabled:
            return {"error": "OpenAI API key not configured"}

        response = openai.ChatCompletion.create(
            model=model,
            messages=messages,
            temperature=temperature,
            max_tokens=max_tokens
        )

        return {
            "response": response.choices[0].message.content,
            "usage": {
                "prompt_tokens": response.usage.prompt_tokens,
                "completion_tokens": response.usage.completion_tokens,
                "total_tokens": response.usage.total_tokens
            }
        }

    def generate_image(
        self,
        prompt: str,
        n: int = 1,
        size: str = "512x512"
    ) -> list:
        """DALL-E ì´ë¯¸ì§€ ìƒì„±"""
        if not self.enabled:
            return {"error": "OpenAI API key not configured"}

        response = openai.Image.create(
            prompt=prompt,
            n=n,
            size=size
        )

        return [img['url'] for img in response.data]

    def transcribe_audio(self, audio_file) -> str:
        """Whisper ìŒì„±-í…ìŠ¤íŠ¸ ë³€í™˜"""
        if not self.enabled:
            return {"error": "OpenAI API key not configured"}

        transcript = openai.Audio.transcribe(
            model="whisper-1",
            file=audio_file
        )

        return transcript.text

@app.post("/ai/classify-image")

    file: UploadFile = File(...),
    top_k: int = 5,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)

    """ì´ë¯¸ì§€ ë¶„ë¥˜"""

    db.commit()
    db.refresh(task)

        task.id,

async def process_image_classification(
    task_id: int,
    file: UploadFile,
    top_k: int,
    db: Session
):
    """ì´ë¯¸ì§€ ë¶„ë¥˜ ì²˜ë¦¬"""
    task = db.query(models.MLTask).filter(models.MLTask.id == task_id).first()

    try:
        start_time = time.time()

        # ì´ë¯¸ì§€ ì½ê¸°
        image_bytes = await file.read()
        image = Image.open(io.BytesIO(image_bytes))

        # ë¶„ë¥˜
        results = image_classifier.classify(image, top_k)

        processing_time = time.time() - start_time

        # ê²°ê³¼ ì €ì¥
        task.status = models.TaskStatus.completed
        task.output_data = {"predictions": results}
        task.confidence_score = results[0]['confidence']
        task.processing_time = processing_time
        task.completed_at = datetime.utcnow()

    except Exception as e:
        task.status = models.TaskStatus.failed
        task.error_message = str(e)

    db.commit()
