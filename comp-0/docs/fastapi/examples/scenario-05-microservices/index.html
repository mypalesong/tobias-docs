📊 시퀀스 다이어그램​

sequenceDiagram
    participant C as Client
    participant GW as API Gateway<br/>:8000
    participant US as User Service<br/>:8001
    participant PS as Product Service<br/>:8002
    participant OS as Order Service<br/>:8003

    Note over C,OS: 1. 사용자 등록 & 로그인
    C->>GW: POST /api/users<br/>{email, password}
    GW->>US: Forward POST /users
    US->>US: Hash password
    US->>US: Save to DB
    US-->>GW: User created
    GW-->>C: 201 Created

    C->>GW: POST /api/login
    GW->>US: Forward POST /login
    US->>US: Verify credentials
    US->>US: Generate JWT
    US-->>GW: {access_token}
    GW-->>C: 200 OK + JWT token

    Note over C,OS: 2. 상품 조회
    C->>GW: GET /api/products
    GW->>PS: Forward GET /products
    PS->>PS: Query DB
    PS-->>GW: Product list
    GW-->>C: 200 OK + Products

    Note over C,OS: 3. 주문 생성 (서비스 간 통신)
    C->>GW: POST /api/orders<br/>Authorization: Bearer {token}
    GW->>GW: Verify JWT
    GW->>OS: Forward POST /orders

    OS->>US: GET /users/{user_id}<br/>(Verify user exists)
    US->>US: Query DB
    US-->>OS: User data

    loop For each product in order
        OS->>PS: GET /products/{product_id}<br/>(Check stock)
        PS->>PS: Query DB
        PS-->>OS: Product data + stock
        alt Stock available
            OS->>PS: PUT /products/{id}/stock<br/>(Decrease stock)
            PS->>PS: Update DB
            PS-->>OS: Stock updated
        else Stock unavailable
            OS-->>GW: 400 Insufficient stock
            GW-->>C: 400 Bad Request
        end
    end

    OS->>OS: Create order in DB
    OS-->>GW: Order created
    GW-->>C: 201 Created + Order

📝 서비스 구성​

1. User Service (:8001)​

# user_service/main.py
from fastapi import FastAPI, HTTPException, Depends
from sqlalchemy.orm import Session
import models, schemas
from database import engine, get_db
from auth import create_access_token, verify_password, hash_password

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="User Service")

@app.post("/users", response_model=schemas.User)
def create_user(user: schemas.UserCreate, db: Session = Depends(get_db)):
    """사용자 생성"""
    # 중복 확인
    existing = db.query(models.User).filter(
        models.User.email == user.email
    ).first()
    if existing:
        raise HTTPException(status_code=400, detail="Email already registered")

    # 사용자 생성
    hashed_password = hash_password(user.password)
    db_user = models.User(
        email=user.email,
        username=user.username,
        hashed_password=hashed_password
    )
    db.add(db_user)
    db.commit()
    db.refresh(db_user)
    return db_user

@app.post("/login")
def login(email: str, password: str, db: Session = Depends(get_db)):
    """로그인"""
    user = db.query(models.User).filter(models.User.email == email).first()
    if not user or not verify_password(password, user.hashed_password):
        raise HTTPException(status_code=401, detail="Invalid credentials")

    token = create_access_token({"sub": user.email, "id": user.id})
    return {"access_token": token, "token_type": "bearer"}

@app.get("/users/{user_id}", response_model=schemas.User)
def get_user(user_id: int, db: Session = Depends(get_db)):
    """사용자 정보 조회"""
    user = db.query(models.User).filter(models.User.id == user_id).first()
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    return user

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8001)

2. Product Service (:8002)​

# product_service/main.py
from fastapi import FastAPI, HTTPException, Depends
from sqlalchemy.orm import Session
from typing import List
import models, schemas
from database import engine, get_db

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="Product Service")

@app.post("/products", response_model=schemas.Product)
def create_product(product: schemas.ProductCreate, db: Session = Depends(get_db)):
    """상품 생성"""
    db_product = models.Product(**product.dict())
    db.add(db_product)
    db.commit()
    db.refresh(db_product)
    return db_product

@app.get("/products", response_model=List[schemas.Product])
def list_products(
    skip: int = 0,
    limit: int = 100,
    db: Session = Depends(get_db)
):
    """상품 목록"""
    products = db.query(models.Product).offset(skip).limit(limit).all()
    return products

@app.get("/products/{product_id}", response_model=schemas.Product)
def get_product(product_id: int, db: Session = Depends(get_db)):
    """상품 조회"""
    product = db.query(models.Product).filter(models.Product.id == product_id).first()
    if not product:
        raise HTTPException(status_code=404, detail="Product not found")
    return product

@app.put("/products/{product_id}/stock")
def update_stock(product_id: int, quantity: int, db: Session = Depends(get_db)):
    """재고 업데이트 (내부 API)"""
    product = db.query(models.Product).filter(models.Product.id == product_id).first()
    if not product:
        raise HTTPException(status_code=404, detail="Product not found")

    product.stock += quantity
    db.commit()
    return {"stock": product.stock}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8002)

3. Order Service (:8003)​

# order_service/main.py
from fastapi import FastAPI, HTTPException, Depends
from sqlalchemy.orm import Session
from typing import List
import httpx
import models, schemas
from database import engine, get_db

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="Order Service")

USER_SERVICE_URL = "http://localhost:8001"
PRODUCT_SERVICE_URL = "http://localhost:8002"

async def verify_user(user_id: int) -> bool:
    """User Service에서 사용자 확인"""
    async with httpx.AsyncClient() as client:
        try:
            response = await client.get(f"{USER_SERVICE_URL}/users/{user_id}")
            return response.status_code == 200
        except:
            return False

async def get_product(product_id: int) -> dict:
    """Product Service에서 상품 정보 조회"""
    async with httpx.AsyncClient() as client:
        try:
            response = await client.get(f"{PRODUCT_SERVICE_URL}/products/{product_id}")
            if response.status_code == 200:
                return response.json()
            return None
        except:
            return None

async def update_product_stock(product_id: int, quantity: int):
    """Product Service에서 재고 업데이트"""
    async with httpx.AsyncClient() as client:
        try:
            response = await client.put(
                f"{PRODUCT_SERVICE_URL}/products/{product_id}/stock",
                params={"quantity": quantity}
            )
            return response.status_code == 200
        except:
            return False

@app.post("/orders", response_model=schemas.Order)
async def create_order(order: schemas.OrderCreate, db: Session = Depends(get_db)):
    """주문 생성"""
    # 사용자 확인
    if not await verify_user(order.user_id):
        raise HTTPException(status_code=400, detail="Invalid user")

    # 상품 확인 및 총액 계산
    total_price = 0
    order_items = []

    for item in order.items:
        product = await get_product(item.product_id)
        if not product:
            raise HTTPException(
                status_code=400,
                detail=f"Product {item.product_id} not found"
            )

        if product["stock"] < item.quantity:
            raise HTTPException(
                status_code=400,
                detail=f"Insufficient stock for product {product['name']}"
            )

        # 재고 차감
        if not await update_product_stock(item.product_id, -item.quantity):
            raise HTTPException(status_code=500, detail="Failed to update stock")

        order_items.append(models.OrderItem(
            product_id=item.product_id,
            quantity=item.quantity,
            price=product["price"]
        ))

        total_price += product["price"] * item.quantity

    # 주문 생성
    db_order = models.Order(
        user_id=order.user_id,
        total_price=total_price,
        items=order_items
    )
    db.add(db_order)
    db.commit()
    db.refresh(db_order)
    return db_order

@app.get("/orders", response_model=List[schemas.Order])
def list_orders(user_id: int, db: Session = Depends(get_db)):
    """주문 목록 (특정 사용자)"""
    orders = db.query(models.Order).filter(
        models.Order.user_id == user_id
    ).all()
    return orders

@app.get("/orders/{order_id}", response_model=schemas.Order)
def get_order(order_id: int, db: Session = Depends(get_db)):
    """주문 조회"""
    order = db.query(models.Order).filter(models.Order.id == order_id).first()
    if not order:
        raise HTTPException(status_code=404, detail="Order not found")
    return order

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8003)

4. API Gateway (:8000)​

# gateway/main.py
from fastapi import FastAPI, Request, HTTPException, Depends
from fastapi.responses import JSONResponse
import httpx
from auth import verify_token

app = FastAPI(title="API Gateway")

SERVICE_URLS = {
    "user": "http://localhost:8001",
    "product": "http://localhost:8002",
    "order": "http://localhost:8003"
}

async def forward_request(service: str, path: str, request: Request):
    """요청을 마이크로서비스로 전달"""
    url = f"{SERVICE_URLS[service]}{path}"
    headers = dict(request.headers)

    async with httpx.AsyncClient() as client:
        if request.method == "GET":
            response = await client.get(url, params=request.query_params, headers=headers)
        elif request.method == "POST":
            body = await request.json()
            response = await client.post(url, json=body, headers=headers)
        elif request.method == "PUT":
            body = await request.json()
            response = await client.put(url, json=body, headers=headers)
        elif request.method == "DELETE":
            response = await client.delete(url, headers=headers)
        else:
            raise HTTPException(status_code=405, detail="Method not allowed")

        return JSONResponse(
            content=response.json(),
            status_code=response.status_code
        )

# User Service 라우팅
@app.post("/api/users")
async def create_user(request: Request):
    return await forward_request("user", "/users", request)

@app.post("/api/login")
async def login(request: Request):
    return await forward_request("user", "/login", request)

@app.get("/api/users/{user_id}")
async def get_user(user_id: int, request: Request, token: str = Depends(verify_token)):
    return await forward_request("user", f"/users/{user_id}", request)

# Product Service 라우팅
@app.get("/api/products")
async def list_products(request: Request):
    return await forward_request("product", "/products", request)

@app.post("/api/products")
async def create_product(request: Request, token: str = Depends(verify_token)):
    return await forward_request("product", "/products", request)

@app.get("/api/products/{product_id}")
async def get_product(product_id: int, request: Request):
    return await forward_request("product", f"/products/{product_id}", request)

# Order Service 라우팅
@app.post("/api/orders")
async def create_order(request: Request, token: str = Depends(verify_token)):
    return await forward_request("order", "/orders", request)

@app.get("/api/orders")
async def list_orders(request: Request, token: str = Depends(verify_token)):
    return await forward_request("order", "/orders", request)

@app.get("/api/orders/{order_id}")
async def get_order(order_id: int, request: Request, token: str = Depends(verify_token)):
    return await forward_request("order", f"/orders/{order_id}", request)

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
