import asyncio
import json
from typing import AsyncGenerator
from datetime import datetime

class SSEStreamer:
    """Server-Sent Events ìŠ¤íŠ¸ë¦¬ë¨¸"""

    def __init__(self):
        self.clients = set()

    async def stream_generator(self) -> AsyncGenerator[str, None]:
        """SSE ìŠ¤íŠ¸ë¦¼ ìƒì„±"""
        queue = asyncio.Queue()
        self.clients.add(queue)

        try:
            while True:
                data = await queue.get()
                yield f"data: {json.dumps(data)}\n\n"

        finally:
            self.clients.remove(queue)

    async def broadcast(self, data: dict):
        """ëª¨ë“  í´ë¼ì´ì–¸íŠ¸ì—ê²Œ ë¸Œë¡œë“œìºìŠ¤íŠ¸"""
        for queue in self.clients:
            await queue.put(data)

    async def stream_logs(self, log_file: str) -> AsyncGenerator[str, None]:
        """ë¡œê·¸ íŒŒì¼ ì‹¤ì‹œê°„ ìŠ¤íŠ¸ë¦¬ë° (tail -f)"""
        import aiofiles

        try:
            async with aiofiles.open(log_file, 'r') as f:
                # íŒŒì¼ ëìœ¼ë¡œ ì´ë™
                await f.seek(0, 2)

                while True:
                    line = await f.readline()

                    if line:
                        yield f"data: {json.dumps({'log': line.strip(), 'timestamp': datetime.utcnow().isoformat()})}\n\n"
                    else:
                        await asyncio.sleep(0.1)

        except Exception as e:
            yield f"data: {json.dumps({'error': str(e)})}\n\n"

    async def stream_metrics(self) -> AsyncGenerator[str, None]:
        """ì‹œìŠ¤í…œ ë©”íŠ¸ë¦­ ìŠ¤íŠ¸ë¦¬ë°"""
        import psutil

        while True:

                "disk_percent": psutil.disk_usage('/').percent,
                "timestamp": datetime.utcnow().isoformat()
            }

            yield f"data: {json.dumps(metrics)}\n\n"
            await asyncio.sleep(1)

class KafkaStreamConsumer:
    """Kafka ìŠ¤íŠ¸ë¦¼ ì»¨ìŠˆë¨¸"""

    def __init__(
        self,
        bootstrap_servers: str = "localhost:9092",
        group_id: str = "fastapi-consumer"
    ):
        self.bootstrap_servers = bootstrap_servers
        self.group_id = group_id
        self.consumer = None

    def consume(
        self,
        topic: str,
        callback: Callable[[dict], None],
        auto_offset_reset: str = "latest"
    ):
        """ë©”ì‹œì§€ ì†Œë¹„"""
        self.consumer = KafkaConsumer(
            topic,
            bootstrap_servers=self.bootstrap_servers,
            group_id=self.group_id,
            auto_offset_reset=auto_offset_reset,
            value_deserializer=lambda m: json.loads(m.decode('utf-8'))
        )

        try:
            for message in self.consumer:
                callback(message.value)

        finally:
            if self.consumer:
                self.consumer.close()

    async def consume_async(
        self,
        topic: str,
        callback: Callable[[dict], None]
    ):
        """ë¹„ë™ê¸° ë©”ì‹œì§€ ì†Œë¹„"""
        import asyncio

        loop = asyncio.get_event_loop()
        await loop.run_in_executor(None, self.consume, topic, callback)

    def close(self):
        """ì»¨ìŠˆë¨¸ ì¢…ë£Œ"""
        if self.consumer:
            self.consumer.close()

class KafkaStreamProducer:
    """Kafka ìŠ¤íŠ¸ë¦¼ í”„ë¡œë“€ì„œ"""

        """ìŠ¤íŠ¸ë¦¼ì— ë©”ì‹œì§€ ì¶”ê°€"""
        message_id = self.client.xadd(
            stream_name,
            data,
            maxlen=maxlen  # ìµœëŒ€ ê¸¸ì´ ì œí•œ
        )
        return message_id

    def read(
        self,
        stream_name: str,
        count: int = 10,
        block: Optional[int] = None,
        last_id: str = '0'
    ) -> List[Dict]:
        """ìŠ¤íŠ¸ë¦¼ì—ì„œ ë©”ì‹œì§€ ì½ê¸°"""

            {stream_name: last_id},
            count=count,

            for entry_id, data in entries:
                results.append({
                    "id": entry_id,
                    "data": data
                })

        return results

    def read_group(
        self,
        group_name: str,
        consumer_name: str,
        stream_name: str,
        count: int = 10,
        block: Optional[int] = None
    ) -> List[Dict]:
        """ì»¨ìŠˆë¨¸ ê·¸ë£¹ìœ¼ë¡œ ë©”ì‹œì§€ ì½ê¸°"""

        messages = self.client.xreadgroup(
            group_name,
            consumer_name,
            {stream_name: '>'},
            count=count,
            block=block
        )

        results = []
        for stream, entries in messages:
            for entry_id, data in entries:
                results.append({
                    "id": entry_id,
                    "data": data
                })

from typing import Callable, List, Dict, Any
import asyncio

class StreamPipeline:
    """ìŠ¤íŠ¸ë¦¼ ë°ì´í„° íŒŒì´í”„ë¼ì¸"""

    def __init__(self):
        self.transforms: List[Callable] = []
        self.filters: List[Callable] = []

    def add_transform(self, func: Callable[[Dict], Dict]):
        """ë³€í™˜ í•¨ìˆ˜ ì¶”ê°€"""
        self.transforms.append(func)
        return self

    def add_filter(self, func: Callable[[Dict], bool]):
        """í•„í„° í•¨ìˆ˜ ì¶”ê°€"""
        self.filters.append(func)
        return self

    def process(self, data: Dict) -> Dict:
        """ë°ì´í„° ì²˜ë¦¬"""
        # í•„í„° ì ìš©
        for filter_func in self.filters:
            if not filter_func(data):
                return None

        # ë³€í™˜ ì ìš©
        result = data
        for transform_func in self.transforms:
            result = transform_func(result)

        return result

    async def process_stream(
        self,
        input_stream,
        output_callback: Callable[[Dict], None]
    ):
        """ìŠ¤íŠ¸ë¦¼ ì²˜ë¦¬"""
        async for data in input_stream:
            processed = self.process(data)

            if processed:
                await output_callback(processed)
