import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import os

class EmailSender:
    def __init__(self):
        self.smtp_host = os.getenv('SMTP_HOST', 'smtp.gmail.com')
        self.smtp_port = int(os.getenv('SMTP_PORT', '587'))
        self.smtp_user = os.getenv('SMTP_USER')
        self.smtp_password = os.getenv('SMTP_PASSWORD')
        self.from_email = os.getenv('FROM_EMAIL', self.smtp_user)

    def send(self, to_email: str, subject: str, body: str, html: bool = True) -> bool:
        """ì´ë©”ì¼ ë°œì†¡"""
        try:
            message = MIMEMultipart('alternative')
            message['Subject'] = subject
            message['From'] = self.from_email
            message['To'] = to_email

            if html:
                part = MIMEText(body, 'html')
            else:
                part = MIMEText(body, 'plain')

            message.attach(part)

            # SMTP ì—°ê²°
            with smtplib.SMTP(self.smtp_host, self.smtp_port) as server:
                server.starttls()
                server.login(self.smtp_user, self.smtp_password)
                server.send_message(message)

            return True

        except Exception as e:
            print(f"Email send error: {e}")
            return False

from twilio.rest import Client
import os

class SMSSender:
    def __init__(self):
        account_sid = os.getenv('TWILIO_ACCOUNT_SID')
        auth_token = os.getenv('TWILIO_AUTH_TOKEN')
        self.from_number = os.getenv('TWILIO_FROM_NUMBER')

        if account_sid and auth_token:
            self.client = Client(account_sid, auth_token)
            self.enabled = True
        else:
            self.client = None
            self.enabled = False
            print("Warning: Twilio not configured. SMS disabled.")

    def send(self, to_number: str, message: str) -> bool:
        """SMS ë°œì†¡"""
        if not self.enabled:
            print(f"SMS to {to_number}: {message} (disabled)")
            return False

        try:
            self.client.messages.create(
                body=message,
                from_=self.from_number,
                to=to_number
            )
            return True

        except Exception as e:
            print(f"SMS send error: {e}")
            return False

from firebase_admin import messaging
import firebase_admin
from firebase_admin import credentials
import os

class PushSender:
    def __init__(self):
        cred_path = os.getenv('FIREBASE_CREDENTIALS')

        if cred_path and os.path.exists(cred_path):
            cred = credentials.Certificate(cred_path)
            firebase_admin.initialize_app(cred)
            self.enabled = True
        else:
            self.enabled = False
            print("Warning: Firebase not configured. Push notifications disabled.")

    def send(self, fcm_token: str, title: str, body: str, data: dict = None) -> bool:
        """í‘¸ì‹œ ì•Œë¦¼ ë°œì†¡"""
        if not self.enabled:
            print(f"Push to {fcm_token}: {title} - {body} (disabled)")
            return False

        try:
            message = messaging.Message(
                notification=messaging.Notification(
                    title=title,
                    body=body
                ),
                data=data or {},
                token=fcm_token
            )

            response = messaging.send(message)
            print(f"Push sent: {response}")
            return True

        except Exception as e:
            print(f"Push send error: {e}")
            return False

from sqlalchemy.orm import Session
from datetime import datetime
import models
from email_sender import EmailSender
from sms_sender import SMSSender
from push_sender import PushSender
from template_engine import TemplateEngine
import json

class NotificationService:
    def __init__(self, db: Session):
        self.db = db
        self.email_sender = EmailSender()
        self.sms_sender = SMSSender()
        self.push_sender = PushSender()
        self.template_engine = TemplateEngine()

    def send_notification(
        self,
        user_id: int,
        channel: str,
        title: str,
        message: str,
        template_name: str = None,
        context: dict = None,
        priority: str = "medium",
        data: dict = None,
        link: str = None,
        scheduled_at: datetime = None
    ) -> models.Notification:
        """ì•Œë¦¼ ìƒì„± ë° ë°œì†¡"""
        # ì‚¬ìš©ì ì„¤ì • í™•ì¸
        settings = self.db.query(models.UserNotificationSettings).filter(
            models.UserNotificationSettings.user_id == user_id
        ).first()

        if not settings:
            raise ValueError("User notification settings not found")

        # ì±„ë„ë³„ í™œì„±í™” í™•ì¸
        if channel == "email" and not settings.email_enabled:
            return None
        elif channel == "sms" and not settings.sms_enabled:
            return None
        elif channel == "push" and not settings.push_enabled:
            return None

        # í…œí”Œë¦¿ ì‚¬ìš©
        if template_name and context:
            template = self.db.query(models.NotificationTemplate).filter(
                models.NotificationTemplate.name == template_name,
                models.NotificationTemplate.channel == channel
            ).first()

            if template:
                if channel == "email":
                    title = self.template_engine.render(template.subject_template, context)
                    message = self.template_engine.render(template.body_template, context)
                else:
                    message = self.template_engine.render(template.message_template, context)

        # ì•Œë¦¼ ë ˆì½”ë“œ ìƒì„±
        notification = models.Notification(
            user_id=user_id,
            title=title,
            message=message,
            channel=channel,
            priority=priority,
            data=json.dumps(data) if data else None,
            link=link,
            scheduled_at=scheduled_at
        )

        self.db.add(notification)
        self.db.commit()
        self.db.refresh(notification)

        # ì¦‰ì‹œ ë°œì†¡ (ìŠ¤ì¼€ì¤„ë§ ì•ˆ ëœ ê²½ìš°)
        if not scheduled_at:
            self._send(notification, settings)

        return notification

    def _send(self, notification: models.Notification, settings: models.UserNotificationSettings) -> bool:
        """ì‹¤ì œ ë°œì†¡"""
        success = False

        try:
            if notification.channel == "email":
                success = self.email_sender.send(
                    settings.email,
                    notification.title,
                    notification.message
                )

            elif notification.channel == "sms":
                if settings.phone:
                    success = self.sms_sender.send(
                        settings.phone,
                        notification.message
                    )

            elif notification.channel == "push":
                if settings.fcm_token:
                    data = json.loads(notification.data) if notification.data else {}
                    success = self.push_sender.send(
                        settings.fcm_token,
                        notification.title,
                        notification.message,
                        data
                    )

            elif notification.channel == "in_app":
                # ì¸ì•± ì•Œë¦¼ì€ DBì—ë§Œ ì €ì¥
                success = True

            if success:
                notification.status = models.NotificationStatus.sent
                notification.sent_at = datetime.utcnow()
            else:
                notification.status = models.NotificationStatus.failed
                notification.failed_reason = "Send failed"

        except Exception as e:
            notification.status = models.NotificationStatus.failed
            notification.failed_reason = str(e)
            notification.retry_count += 1

        self.db.commit()
        return success

    def retry_failed_notifications(self):
        """ì‹¤íŒ¨í•œ ì•Œë¦¼ ì¬ì‹œë„"""
        failed = self.db.query(models.Notification).filter(
            models.Notification.status == models.NotificationStatus.failed,
            models.Notification.retry_count < models.Notification.max_retries
        ).all()

        for notification in failed:
            settings = self.db.query(models.UserNotificationSettings).filter(
                models.UserNotificationSettings.user_id == notification.user_id
            ).first()

            if settings:
                self._send(notification, settings)

    def send_scheduled_notifications(self):
        """ìŠ¤ì¼€ì¤„ëœ ì•Œë¦¼ ë°œì†¡"""
        scheduled = self.db.query(models.Notification).filter(
            models.Notification.status == models.NotificationStatus.pending,
            models.Notification.scheduled_at <= datetime.utcnow()
        ).all()

        for notification in scheduled:
            settings = self.db.query(models.UserNotificationSettings).filter(
                models.UserNotificationSettings.user_id == notification.user_id
            ).first()

            if settings:
                self._send(notification, settings)

    def send_bulk(self, user_ids: list, **kwargs):
        """ë°°ì¹˜ ë°œì†¡"""
        notifications = []

        for user_id in user_ids:
            try:
                notification = self.send_notification(user_id, **kwargs)
                if notification:
                    notifications.append(notification)
            except Exception as e:
                print(f"Failed to send to user {user_id}: {e}")

        return notifications

class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[int, WebSocket] = {}

    async def connect(self, user_id: int, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[user_id] = websocket

    def disconnect(self, user_id: int):
        if user_id in self.active_connections:
            del self.active_connections[user_id]

    async def send_to_user(self, user_id: int, message: dict):
        if user_id in self.active_connections:
            websocket = self.active_connections[user_id]
            try:
                await websocket.send_json(message)
            except:
                self.disconnect(user_id)

manager = ConnectionManager()

@app.post("/notification-settings")
def create_or_update_settings(
    settings: schemas.NotificationSettingsCreate,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """ì•Œë¦¼ ì„¤ì • ìƒì„±/ì—…ë°ì´íŠ¸"""
    existing = db.query(models.UserNotificationSettings).filter(
        models.UserNotificationSettings.user_id == current_user.id
    ).first()

    if existing:
        for key, value in settings.dict(exclude_unset=True).items():
            setattr(existing, key, value)
        db.commit()
        return existing
    else:
        db_settings = models.UserNotificationSettings(
            user_id=current_user.id,
            **settings.dict()
        )
        db.add(db_settings)
        db.commit()
        db.refresh(db_settings)
        return db_settings
