시나리오 10: 알림 시스템

멀티채널 알림 시스템 (이메일, SMS, 푸시, 웹소켓)을 만들어봅시다!

📌 기능​

  • ✅ 이메일 알림 (SMTP)
  • ✅ SMS 알림 (Twilio)
  • ✅ 푸시 알림 (FCM)
  • ✅ 실시간 웹 알림 (WebSocket)
  • ✅ 알림 우선순위 관리
  • ✅ 알림 템플릿 시스템
  • ✅ 사용자별 알림 설정
  • ✅ 배치 발송
  • ✅ 재시도 로직

sequenceDiagram
    participant App as Application
    participant API as FastAPI
    participant NS as NotificationService
    participant TE as TemplateEngine
    participant ES as EmailSender
    participant SS as SMSSender
    participant PS as PushSender
    participant WS as WebSocketManager

    Note over App,DB: 1. 알림 발송 요청
    App->>API: POST /notifications/send<br/>{user_id, template, data}
    API->>NS: send_notification()
    NS->>DB: SELECT user_notification_settings
    DB-->>NS: User settings (channels enabled)
    NS->>DB: SELECT notification_template
    DB-->>NS: Template data
    NS->>TE: render_template(template, data)
    TE-->>NS: Rendered content

    par Send to enabled channels
        alt Email enabled
            NS->>ES: send_email()
            ES->>ES: SMTP send
            ES-->>NS: Email sent/failed
        end
        alt SMS enabled
            NS->>SS: send_sms()
            SS->>SS: Twilio API
            SS-->>NS: SMS sent/failed
        end
        alt Push enabled
            NS->>PS: send_push()
            PS->>PS: FCM API
            PS-->>NS: Push sent/failed
        end
        alt WebSocket enabled
            NS->>WS: broadcast_to_user()
            WS->>WS: Send via WS connection
            WS-->>NS: Delivered
        end
    end

    NS->>DB: INSERT notifications (log)
    NS-->>API: Notification sent
    API-->>App: 200 OK

    Note over App,DB: 2. 배치 발송
    App->>API: POST /notifications/batch<br/>{user_ids[], template}
    API->>NS: send_batch_notifications()
    loop For each user
        NS->>NS: Queue notification job
    end
    NS-->>API: Batch queued
    API-->>App: 202 Accepted

    Note over App,DB: 3. 재시도 로직 (백그라운드)
    NS->>DB: SELECT failed notifications<br/>WHERE retry_count < max_retries
    DB-->>NS: Failed notifications
    loop For each failed notification
        NS->>NS: Calculate backoff delay
        NS->>NS: Retry send
        alt Retry success
            NS->>DB: UPDATE status=sent
        else Retry failed
            NS->>DB: UPDATE retry_count++
        end
    end

from sqlalchemy import Column, Integer, String, DateTime, Boolean, ForeignKey, Text, Enum as SQLEnum
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base
import enum
from datetime import datetime

class NotificationChannel(str, enum.Enum):
    email = "email"
    sms = "sms"
    push = "push"
    websocket = "websocket"
    in_app = "in_app"

class NotificationPriority(str, enum.Enum):
    low = "low"
    medium = "medium"
    high = "high"
    urgent = "urgent"

class NotificationStatus(str, enum.Enum):
    pending = "pending"
    sent = "sent"
    failed = "failed"
    read = "read"

class NotificationTemplate(Base):
    """알림 템플릿"""
    __tablename__ = "notification_templates"

    id = Column(Integer, primary_key=True)
    name = Column(String, unique=True, index=True)
    channel = Column(SQLEnum(NotificationChannel))

    # 이메일
    subject_template = Column(String, nullable=True)
    body_template = Column(Text, nullable=True)

    # SMS/푸시
    message_template = Column(Text, nullable=True)

    # 변수 예: "안녕하세요 {{username}}님, {{message}}"
    created_at = Column(DateTime(timezone=True), server_default=func.now())

class UserNotificationSettings(Base):
    """사용자별 알림 설정"""
    __tablename__ = "user_notification_settings"

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'), unique=True)

    # 채널별 활성화
    email_enabled = Column(Boolean, default=True)
    sms_enabled = Column(Boolean, default=False)
    push_enabled = Column(Boolean, default=True)
    websocket_enabled = Column(Boolean, default=True)

    # 연락처
    email = Column(String)
    phone = Column(String, nullable=True)
    fcm_token = Column(String, nullable=True)  # Firebase Cloud Messaging

    # 알림 유형별 설정
    marketing_enabled = Column(Boolean, default=False)
    newsletter_enabled = Column(Boolean, default=True)

    user = relationship("User")

class Notification(Base):
    __tablename__ = "notifications"

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'), index=True)

    # 내용
    title = Column(String)
    message = Column(Text)
    channel = Column(SQLEnum(NotificationChannel), index=True)
    priority = Column(SQLEnum(NotificationPriority), default=NotificationPriority.medium)

    # 메타데이터
    data = Column(Text, nullable=True)  # JSON 형태의 추가 데이터
    link = Column(String, nullable=True)  # 클릭 시 이동할 URL

    # 상태
    status = Column(SQLEnum(NotificationStatus), default=NotificationStatus.pending, index=True)

    # 발송 정보
    scheduled_at = Column(DateTime(timezone=True), nullable=True)
    sent_at = Column(DateTime(timezone=True), nullable=True)
    read_at = Column(DateTime(timezone=True), nullable=True)
    failed_reason = Column(String, nullable=True)

    # 재시도
    retry_count = Column(Integer, default=0)
    max_retries = Column(Integer, default=3)

    created_at = Column(DateTime(timezone=True), server_default=func.now(), index=True)

    user = relationship("User")

email_sender.py​

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import os

class EmailSender:
    def __init__(self):
        self.smtp_host = os.getenv('SMTP_HOST', 'smtp.gmail.com')
        self.smtp_port = int(os.getenv('SMTP_PORT', '587'))
        self.smtp_user = os.getenv('SMTP_USER')
        self.smtp_password = os.getenv('SMTP_PASSWORD')
        self.from_email = os.getenv('FROM_EMAIL', self.smtp_user)

    def send(self, to_email: str, subject: str, body: str, html: bool = True) -> bool:
        """이메일 발송"""
        try:
            message = MIMEMultipart('alternative')
            message['Subject'] = subject
            message['From'] = self.from_email
            message['To'] = to_email

            if html:
                part = MIMEText(body, 'html')
            else:
                part = MIMEText(body, 'plain')

            message.attach(part)

            # SMTP 연결
            with smtplib.SMTP(self.smtp_host, self.smtp_port) as server:
                server.starttls()
                server.login(self.smtp_user, self.smtp_password)
                server.send_message(message)

            return True

        except Exception as e:
            print(f"Email send error: {e}")
            return False

sms_sender.py​

from twilio.rest import Client
import os

class SMSSender:
    def __init__(self):
        account_sid = os.getenv('TWILIO_ACCOUNT_SID')
        auth_token = os.getenv('TWILIO_AUTH_TOKEN')
        self.from_number = os.getenv('TWILIO_FROM_NUMBER')

        if account_sid and auth_token:
            self.client = Client(account_sid, auth_token)
            self.enabled = True
        else:
            self.client = None
            self.enabled = False
            print("Warning: Twilio not configured. SMS disabled.")

    def send(self, to_number: str, message: str) -> bool:
        """SMS 발송"""
        if not self.enabled:
            print(f"SMS to {to_number}: {message} (disabled)")
            return False

        try:
            self.client.messages.create(
                body=message,
                from_=self.from_number,
                to=to_number
            )
            return True

        except Exception as e:
            print(f"SMS send error: {e}")
            return False

push_sender.py​

from firebase_admin import messaging
import firebase_admin
from firebase_admin import credentials
import os

class PushSender:
    def __init__(self):
        cred_path = os.getenv('FIREBASE_CREDENTIALS')

        if cred_path and os.path.exists(cred_path):
            cred = credentials.Certificate(cred_path)
            firebase_admin.initialize_app(cred)
            self.enabled = True
        else:
            self.enabled = False
            print("Warning: Firebase not configured. Push notifications disabled.")

    def send(self, fcm_token: str, title: str, body: str, data: dict = None) -> bool:
        """푸시 알림 발송"""
        if not self.enabled:
            print(f"Push to {fcm_token}: {title} - {body} (disabled)")
            return False

        try:
            message = messaging.Message(
                notification=messaging.Notification(
                    title=title,
                    body=body
                ),
                data=data or {},
                token=fcm_token
            )

            response = messaging.send(message)
            print(f"Push sent: {response}")
            return True

        except Exception as e:
            print(f"Push send error: {e}")
            return False

notification_service.py​

from sqlalchemy.orm import Session
from datetime import datetime
import models
from email_sender import EmailSender
from sms_sender import SMSSender
from push_sender import PushSender
from template_engine import TemplateEngine
import json

class NotificationService:
    def __init__(self, db: Session):
        self.db = db
        self.email_sender = EmailSender()
        self.sms_sender = SMSSender()
        self.push_sender = PushSender()
        self.template_engine = TemplateEngine()

    def send_notification(
        self,
        user_id: int,
        channel: str,
        title: str,
        message: str,
        template_name: str = None,
        context: dict = None,
        priority: str = "medium",
        data: dict = None,
        link: str = None,
        scheduled_at: datetime = None
    ) -> models.Notification:
        """알림 생성 및 발송"""
        # 사용자 설정 확인
        settings = self.db.query(models.UserNotificationSettings).filter(
            models.UserNotificationSettings.user_id == user_id
        ).first()

        if not settings:
            raise ValueError("User notification settings not found")

        # 채널별 활성화 확인
        if channel == "email" and not settings.email_enabled:
            return None
        elif channel == "sms" and not settings.sms_enabled:
            return None
        elif channel == "push" and not settings.push_enabled:
            return None

        # 템플릿 사용
        if template_name and context:
            template = self.db.query(models.NotificationTemplate).filter(
                models.NotificationTemplate.name == template_name,
                models.NotificationTemplate.channel == channel
            ).first()

            if template:
                if channel == "email":
                    title = self.template_engine.render(template.subject_template, context)
                    message = self.template_engine.render(template.body_template, context)
                else:
                    message = self.template_engine.render(template.message_template, context)

        # 알림 레코드 생성
        notification = models.Notification(
            user_id=user_id,
            title=title,
            message=message,
            channel=channel,
            priority=priority,
            data=json.dumps(data) if data else None,
            link=link,
            scheduled_at=scheduled_at
        )

        self.db.add(notification)
        self.db.commit()
        self.db.refresh(notification)

        # 즉시 발송 (스케줄링 안 된 경우)
        if not scheduled_at:
            self._send(notification, settings)

        return notification

    def _send(self, notification: models.Notification, settings: models.UserNotificationSettings) -> bool:
        """실제 발송"""
        success = False

        try:
            if notification.channel == "email":
                success = self.email_sender.send(
                    settings.email,
                    notification.title,
                    notification.message
                )

            elif notification.channel == "sms":
                if settings.phone:
                    success = self.sms_sender.send(
                        settings.phone,
                        notification.message
                    )

            elif notification.channel == "push":
                if settings.fcm_token:
                    data = json.loads(notification.data) if notification.data else {}
                    success = self.push_sender.send(
                        settings.fcm_token,
                        notification.title,
                        notification.message,
                        data
                    )

            elif notification.channel == "in_app":
                # 인앱 알림은 DB에만 저장
                success = True

            if success:
                notification.status = models.NotificationStatus.sent
                notification.sent_at = datetime.utcnow()
            else:
                notification.status = models.NotificationStatus.failed
                notification.failed_reason = "Send failed"

        except Exception as e:
            notification.status = models.NotificationStatus.failed
            notification.failed_reason = str(e)
            notification.retry_count += 1

        self.db.commit()
        return success

    def retry_failed_notifications(self):
        """실패한 알림 재시도"""
        failed = self.db.query(models.Notification).filter(
            models.Notification.status == models.NotificationStatus.failed,
            models.Notification.retry_count < models.Notification.max_retries
        ).all()

        for notification in failed:
            settings = self.db.query(models.UserNotificationSettings).filter(
                models.UserNotificationSettings.user_id == notification.user_id
            ).first()

            if settings:
                self._send(notification, settings)

    def send_scheduled_notifications(self):
        """스케줄된 알림 발송"""
        scheduled = self.db.query(models.Notification).filter(
            models.Notification.status == models.NotificationStatus.pending,
            models.Notification.scheduled_at <= datetime.utcnow()
        ).all()

        for notification in scheduled:
            settings = self.db.query(models.UserNotificationSettings).filter(
                models.UserNotificationSettings.user_id == notification.user_id
            ).first()

            if settings:
                self._send(notification, settings)

    def send_bulk(self, user_ids: list, **kwargs):
        """배치 발송"""
        notifications = []

        for user_id in user_ids:
            try:
                notification = self.send_notification(user_id, **kwargs)
                if notification:
                    notifications.append(notification)
            except Exception as e:
                print(f"Failed to send to user {user_id}: {e}")

        return notifications

from fastapi import FastAPI, Depends, HTTPException, WebSocket, WebSocketDisconnect

# WebSocket 연결 관리
class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[int, WebSocket] = {}

    async def connect(self, user_id: int, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[user_id] = websocket

    def disconnect(self, user_id: int):
        if user_id in self.active_connections:
            del self.active_connections[user_id]

    async def send_to_user(self, user_id: int, message: dict):
        if user_id in self.active_connections:
            websocket = self.active_connections[user_id]
            try:
                await websocket.send_json(message)
            except:
                self.disconnect(user_id)

manager = ConnectionManager()

# ==================== 알림 설정 ====================

@app.post("/notification-settings")
def create_or_update_settings(
    settings: schemas.NotificationSettingsCreate,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """알림 설정 생성/업데이트"""
    existing = db.query(models.UserNotificationSettings).filter(
        models.UserNotificationSettings.user_id == current_user.id
    ).first()

    if existing:
        for key, value in settings.dict(exclude_unset=True).items():
            setattr(existing, key, value)
        db.commit()
        return existing
    else:
        db_settings = models.UserNotificationSettings(
            user_id=current_user.id,
            **settings.dict()
        )
        db.add(db_settings)
        db.commit()
        db.refresh(db_settings)
        return db_settings

@app.get("/notification-settings")
def get_settings(

    """내 알림 설정 조회"""

@app.post("/notifications/send")
def send_notification(
    notification: schemas.NotificationSend,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """알림 발송"""
    service = NotificationService(db)

    result = service.send_notification(
        user_id=notification.user_id,
        channel=notification.channel,
        title=notification.title,
        message=notification.message,
        template_name=notification.template_name,
        context=notification.context,
        priority=notification.priority,
        data=notification.data,
        link=notification.link,
        scheduled_at=notification.scheduled_at
    )

    # WebSocket 연결된 사용자에게 실시간 전송
    if notification.channel in ["websocket", "in_app"]:
        asyncio.create_task(manager.send_to_user(
            notification.user_id,
            {
                "id": result.id,
                "title": result.title,
                "message": result.message,
                "link": result.link,
                "created_at": result.created_at.isoformat()
            }
        ))

    return result

@app.post("/notifications/bulk")
def send_bulk_notification(

        message=bulk.message,

# ==================== 알림 조회 ====================

@app.get("/notifications", response_model=List[schemas.Notification])
def get_notifications(
    unread_only: bool = False,
    channel: Optional[str] = None,
    skip: int = 0,
    limit: int = 20,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """내 알림 목록"""
    query = db.query(models.Notification).filter(
        models.Notification.user_id == current_user.id
    )

    if unread_only:
        query = query.filter(models.Notification.read_at == None)

    if channel:
        query = query.filter(models.Notification.channel == channel)

    notifications = query.order_by(
        models.Notification.created_at.desc()
    ).offset(skip).limit(limit).all()

    return notifications

@app.put("/notifications/{notification_id}/read")

# ==================== WebSocket ====================

@app.websocket("/ws/notifications")
async def websocket_endpoint(
    websocket: WebSocket,
    token: str,
    db: Session = Depends(get_db)
):
    """실시간 알림 WebSocket"""
    from auth import get_current_user_from_token

    user = await get_current_user_from_token(token, db)
    if not user:
        await websocket.close(code=1008)
        return

    await manager.connect(user.id, websocket)

    try:
        while True:
            # 클라이언트로부터 메시지 수신 (연결 유지)
            await websocket.receive_text()

    except WebSocketDisconnect:
        manager.disconnect(user.id)

# ==================== 템플릿 관리 ====================

@app.post("/notification-templates")
def create_template(
    template: schemas.NotificationTemplateCreate,
    db: Session = Depends(get_db)
):
    """알림 템플릿 생성"""
    db_template = models.NotificationTemplate(**template.dict())
    db.add(db_template)
    db.commit()
    db.refresh(db_template)
    return db_template

@app.get("/notification-templates", response_model=List[schemas.NotificationTemplate])
def list_templates(db: Session = Depends(get_db)):
    """템플릿 목록"""
    return db.query(models.NotificationTemplate).all()
