시나리오 7: 파일 저장소 시스템

S3 호환 저장소를 활용한 파일 관리 시스템을 만들어봅시다!

📌 기능​

  • ✅ 파일 업로드 (멀티파트, 청크 업로드)
  • ✅ 파일 다운로드 (스트리밍)
  • ✅ 이미지 썸네일 자동 생성
  • ✅ 파일 메타데이터 관리
  • ✅ 공유 링크 생성 (만료 시간)
  • ✅ S3/MinIO 연동
  • ✅ 바이러스 스캔 (ClamAV)

📊 시퀀스 다이어그램​

sequenceDiagram
    participant C as Client
    participant API as FastAPI
    participant FS as FileService
    participant S3 as S3/MinIO
    participant VS as VirusScanner
    participant IP as ImageProcessor
    participant DB as Database

    Note over C,DB: 1. 파일 업로드
    C->>API: POST /files/upload<br/>multipart/form-data
    API->>API: Receive file stream
    API->>FS: process_upload()
    FS->>VS: scan_file()
    VS-->>FS: Scan result (clean/infected)
    alt File infected
        FS-->>API: Virus detected
        API-->>C: 400 Bad Request
    else File clean
        FS->>S3: upload_file()
        S3-->>FS: S3 URL + key
        alt File is image
            FS->>IP: generate_thumbnail()
            IP->>IP: Resize image
            IP->>S3: upload_thumbnail()
            S3-->>IP: Thumbnail URL
            IP-->>FS: Thumbnail info
        end
        FS->>DB: INSERT file metadata
        DB-->>FS: File record
        FS-->>API: File uploaded
        API-->>C: 201 Created + File info
    end

    Note over C,DB: 2. 파일 다운로드 (스트리밍)
    C->>API: GET /files/{id}/download
    API->>DB: SELECT file metadata
    DB-->>API: File info
    API->>S3: get_object() stream
    S3-->>API: File stream
    API-->>C: Stream file with<br/>Content-Disposition

    Note over C,DB: 3. 공유 링크 생성
    C->>API: POST /files/{id}/share<br/>{expires_in, max_downloads}
    API->>DB: INSERT file_share (token, expires_at)
    DB-->>API: Share record
    API-->>C: {share_url, token, expires_at}

    Note over C,DB: 4. 공유 링크로 다운로드
    C->>API: GET /share/{token}
    API->>DB: SELECT file_share
    DB-->>API: Share info
    alt Share expired or max downloads reached
        API-->>C: 403 Forbidden
    else Share valid
        API->>DB: INCREMENT download_count
        API->>S3: get_object() stream
        S3-->>API: File stream
        API-->>C: Stream file
    end

models.py​

from sqlalchemy import Column, Integer, String, DateTime, Boolean, BigInteger, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base
import enum

class FileType(str, enum.Enum):
    image = "image"
    video = "video"
    document = "document"
    archive = "archive"
    other = "other"

class File(Base):
    __tablename__ = "files"

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey("users.id"))
    filename = Column(String, index=True)
    original_filename = Column(String)
    file_type = Column(String)  # MIME type
    category = Column(String)  # image, video, etc.
    size = Column(BigInteger)  # bytes

    # S3 정보
    s3_bucket = Column(String)
    s3_key = Column(String, unique=True)
    s3_url = Column(String)

    # 썸네일 (이미지인 경우)
    thumbnail_s3_key = Column(String, nullable=True)
    thumbnail_url = Column(String, nullable=True)

    # 메타데이터
    width = Column(Integer, nullable=True)
    height = Column(Integer, nullable=True)
    duration = Column(Integer, nullable=True)  # 비디오 길이 (초)

    # 보안
    is_public = Column(Boolean, default=False)
    virus_scanned = Column(Boolean, default=False)
    virus_scan_result = Column(String, nullable=True)

    # 타임스탬프
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    updated_at = Column(DateTime(timezone=True), onupdate=func.now())

    user = relationship("User")
    shares = relationship("FileShare", back_populates="file")

class FileShare(Base):
    __tablename__ = "file_shares"

    id = Column(Integer, primary_key=True)
    file_id = Column(Integer, ForeignKey("files.id"))
    share_token = Column(String, unique=True, index=True)
    expires_at = Column(DateTime(timezone=True))
    download_count = Column(Integer, default=0)
    max_downloads = Column(Integer, nullable=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    file = relationship("File", back_populates="shares")

s3_client.py​

import boto3
from botocore.exceptions import ClientError
from typing import BinaryIO, Optional
import os

class S3Client:
    def __init__(self):
        self.s3 = boto3.client(
            's3',
            endpoint_url=os.getenv('S3_ENDPOINT', 'http://localhost:9000'),
            aws_access_key_id=os.getenv('S3_ACCESS_KEY', 'minioadmin'),
            aws_secret_access_key=os.getenv('S3_SECRET_KEY', 'minioadmin'),
            region_name=os.getenv('S3_REGION', 'us-east-1')
        )
        self.bucket = os.getenv('S3_BUCKET', 'files')
        self._ensure_bucket()

    def _ensure_bucket(self):
        """버킷이 없으면 생성"""
        try:
            self.s3.head_bucket(Bucket=self.bucket)
        except ClientError:
            self.s3.create_bucket(Bucket=self.bucket)

    def upload_file(
        self,
        file: BinaryIO,
        key: str,
        content_type: str,
        metadata: Optional[dict] = None
    ) -> str:
        """파일 업로드"""
        extra_args = {
            'ContentType': content_type,
            'Metadata': metadata or {}
        }

        self.s3.upload_fileobj(file, self.bucket, key, ExtraArgs=extra_args)

        # URL 생성
        url = f"{os.getenv('S3_ENDPOINT')}/{self.bucket}/{key}"
        return url

    def download_file(self, key: str) -> bytes:
        """파일 다운로드"""
        response = self.s3.get_object(Bucket=self.bucket, Key=key)
        return response['Body'].read()

    def generate_presigned_url(self, key: str, expiration: int = 3600) -> str:
        """만료 시간이 있는 다운로드 URL 생성"""
        url = self.s3.generate_presigned_url(
            'get_object',
            Params={'Bucket': self.bucket, 'Key': key},
            ExpiresIn=expiration
        )
        return url

    def delete_file(self, key: str):
        """파일 삭제"""
        self.s3.delete_object(Bucket=self.bucket, Key=key)

    def copy_file(self, source_key: str, dest_key: str):
        """파일 복사"""
        copy_source = {'Bucket': self.bucket, 'Key': source_key}
        self.s3.copy_object(
            CopySource=copy_source,
            Bucket=self.bucket,
            Key=dest_key
        )

image_processor.py​

from PIL import Image
import io
from typing import Tuple

class ImageProcessor:
    @staticmethod
    def create_thumbnail(
        image_bytes: bytes,
        size: Tuple[int, int] = (300, 300)
    ) -> bytes:
        """썸네일 생성"""
        image = Image.open(io.BytesIO(image_bytes))

        # EXIF 방향 정보 처리
        try:
            for orientation in ExifTags.TAGS.keys():
                if ExifTags.TAGS[orientation] == 'Orientation':
                    break
            exif = image._getexif()
            if exif is not None:
                orientation_value = exif.get(orientation)
                if orientation_value == 3:
                    image = image.rotate(180, expand=True)
                elif orientation_value == 6:
                    image = image.rotate(270, expand=True)
                elif orientation_value == 8:
                    image = image.rotate(90, expand=True)
        except (AttributeError, KeyError, IndexError):
            pass

        # 썸네일 생성
        image.thumbnail(size, Image.Resampling.LANCZOS)

        # 바이트로 변환
        output = io.BytesIO()
        image.save(output, format='JPEG', quality=85)
        output.seek(0)
        return output.read()

    @staticmethod
    def get_image_info(image_bytes: bytes) -> dict:
        """이미지 정보 추출"""
        image = Image.open(io.BytesIO(image_bytes))
        return {
            'width': image.width,
            'height': image.height,
            'format': image.format,
            'mode': image.mode
        }

virus_scanner.py​

import clamd
from typing import Optional

class VirusScanner:
    def __init__(self):
        try:
            self.scanner = clamd.ClamdUnixSocket()
            self.scanner.ping()
            self.enabled = True
        except:
            self.enabled = False
            print("Warning: ClamAV not available. Virus scanning disabled.")

    def scan(self, file_bytes: bytes) -> dict:
        """바이러스 스캔"""
        if not self.enabled:
            return {'clean': True, 'message': 'Scanning disabled'}

        result = self.scanner.instream(io.BytesIO(file_bytes))
        status = result['stream'][0]

        if status == 'OK':
            return {'clean': True, 'message': 'File is clean'}
        else:
            return {'clean': False, 'message': status}

main.py​

from fastapi import FastAPI, File, UploadFile, Depends, HTTPException, BackgroundTasks
from fastapi.responses import StreamingResponse
from sqlalchemy.orm import Session
import models, schemas
from database import engine, get_db
from s3_client import S3Client
from image_processor import ImageProcessor
from virus_scanner import VirusScanner
from auth import get_current_user
import uuid
from datetime import datetime, timedelta
import io
import mimetypes

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="File Storage API")

s3_client = S3Client()
image_processor = ImageProcessor()
virus_scanner = VirusScanner()

def get_file_category(mime_type: str) -> str:
    """MIME 타입으로 카테고리 결정"""
    if mime_type.startswith('image/'):
        return 'image'
    elif mime_type.startswith('video/'):
        return 'video'
    elif mime_type in ['application/pdf', 'application/msword', 'text/']:
        return 'document'
    elif mime_type in ['application/zip', 'application/x-rar']:
        return 'archive'
    else:
        return 'other'

@app.post("/files", response_model=schemas.File)
async def upload_file(
    file: UploadFile = File(...),
    background_tasks: BackgroundTasks,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """파일 업로드"""
    # 파일 읽기
    file_bytes = await file.read()
    file_size = len(file_bytes)

    # 크기 제한 (100MB)
    if file_size > 100 * 1024 * 1024:
        raise HTTPException(status_code=400, detail="File too large (max 100MB)")

    # MIME 타입 확인
    mime_type = file.content_type or mimetypes.guess_type(file.filename)[0] or 'application/octet-stream'
    category = get_file_category(mime_type)

    # 고유 파일명 생성
    file_id = str(uuid.uuid4())
    extension = file.filename.split('.')[-1] if '.' in file.filename else ''
    s3_key = f"users/{current_user.id}/{file_id}.{extension}" if extension else f"users/{current_user.id}/{file_id}"

    # S3 업로드
    s3_url = s3_client.upload_file(
        io.BytesIO(file_bytes),
        s3_key,
        mime_type,
        metadata={
            'original_filename': file.filename,
            'user_id': str(current_user.id)
        }
    )

    # DB 저장
    db_file = models.File(
        user_id=current_user.id,
        filename=f"{file_id}.{extension}" if extension else file_id,
        original_filename=file.filename,
        file_type=mime_type,
        category=category,
        size=file_size,
        s3_bucket=s3_client.bucket,
        s3_key=s3_key,
        s3_url=s3_url
    )

    # 이미지인 경우 썸네일 생성
    if category == 'image':
        try:
            # 이미지 정보
            info = image_processor.get_image_info(file_bytes)
            db_file.width = info['width']
            db_file.height = info['height']

            # 썸네일 생성 (백그라운드)
            background_tasks.add_task(
                create_thumbnail_task,
                file_id,
                file_bytes,
                s3_key,
                db
            )
        except Exception as e:
            print(f"Image processing error: {e}")

    # 바이러스 스캔 (백그라운드)
    background_tasks.add_task(scan_virus_task, db_file.id, file_bytes, db)

    db.add(db_file)
    db.commit()
    db.refresh(db_file)

    return db_file

async def create_thumbnail_task(file_id: str, file_bytes: bytes, original_key: str, db: Session):
    """썸네일 생성 태스크"""
    try:
        thumbnail_bytes = image_processor.create_thumbnail(file_bytes)
        thumbnail_key = original_key.replace(f"{file_id}.", f"{file_id}_thumb.")

        # S3 업로드
        thumbnail_url = s3_client.upload_file(
            io.BytesIO(thumbnail_bytes),
            thumbnail_key,
            'image/jpeg'
        )

        # DB 업데이트
        db_file = db.query(models.File).filter(models.File.filename.contains(file_id)).first()
        if db_file:
            db_file.thumbnail_s3_key = thumbnail_key
            db_file.thumbnail_url = thumbnail_url
            db.commit()
    except Exception as e:
        print(f"Thumbnail creation error: {e}")

async def scan_virus_task(file_id: int, file_bytes: bytes, db: Session):
    """바이러스 스캔 태스크"""
    result = virus_scanner.scan(file_bytes)

    db_file = db.query(models.File).filter(models.File.id == file_id).first()
    if db_file:
        db_file.virus_scanned = True
        db_file.virus_scan_result = 'clean' if result['clean'] else result['message']

        # 바이러스 발견 시 파일 삭제
        if not result['clean']:
            s3_client.delete_file(db_file.s3_key)
            db.delete(db_file)

        db.commit()

@app.get("/files/{file_id}/download")
async def download_file(
    file_id: int,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """파일 다운로드 (스트리밍)"""
    file = db.query(models.File).filter(models.File.id == file_id).first()

    if not file:
        raise HTTPException(status_code=404, detail="File not found")

    # 권한 확인
    if file.user_id != current_user.id and not file.is_public:
        raise HTTPException(status_code=403, detail="Access denied")

    # S3에서 다운로드
    file_bytes = s3_client.download_file(file.s3_key)

    return StreamingResponse(
        io.BytesIO(file_bytes),
        media_type=file.file_type,
        headers={
            'Content-Disposition': f'attachment; filename="{file.original_filename}"'
        }
    )

@app.post("/files/{file_id}/share")
def create_share_link(
    file_id: int,
    expires_in_hours: int = 24,
    max_downloads: int = None,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """공유 링크 생성"""
    file = db.query(models.File).filter(
        models.File.id == file_id,
        models.File.user_id == current_user.id
    ).first()

    if not file:
        raise HTTPException(status_code=404, detail="File not found")

    # 공유 토큰 생성
    share_token = str(uuid.uuid4())
    expires_at = datetime.utcnow() + timedelta(hours=expires_in_hours)

    share = models.FileShare(
        file_id=file_id,
        share_token=share_token,
        expires_at=expires_at,
        max_downloads=max_downloads
    )

    db.add(share)
    db.commit()

    return {
        "share_url": f"/shared/{share_token}",
        "expires_at": expires_at,
        "max_downloads": max_downloads
    }

@app.get("/shared/{share_token}")
async def download_shared_file(share_token: str, db: Session = Depends(get_db)):
    """공유 링크로 파일 다운로드"""
    share = db.query(models.FileShare).filter(
        models.FileShare.share_token == share_token
    ).first()

    if not share:
        raise HTTPException(status_code=404, detail="Share not found")

    # 만료 확인
    if share.expires_at < datetime.utcnow():
        raise HTTPException(status_code=410, detail="Share expired")

    # 다운로드 횟수 확인
    if share.max_downloads and share.download_count >= share.max_downloads:
        raise HTTPException(status_code=403, detail="Download limit reached")

    # 다운로드 카운트 증가
    share.download_count += 1
    db.commit()

    # 파일 다운로드
    file = share.file
    file_bytes = s3_client.download_file(file.s3_key)

    return StreamingResponse(
        io.BytesIO(file_bytes),
        media_type=file.file_type,
        headers={
            'Content-Disposition': f'attachment; filename="{file.original_filename}"'
        }
    )

@app.delete("/files/{file_id}")
def delete_file(
    file_id: int,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """파일 삭제"""
    file = db.query(models.File).filter(
        models.File.id == file_id,
        models.File.user_id == current_user.id
    ).first()

    if not file:
        raise HTTPException(status_code=404, detail="File not found")

    # S3에서 삭제
    s3_client.delete_file(file.s3_key)
    if file.thumbnail_s3_key:
        s3_client.delete_file(file.thumbnail_s3_key)

    # DB에서 삭제
    db.delete(file)
    db.commit()

    return {"message": "File deleted"}
