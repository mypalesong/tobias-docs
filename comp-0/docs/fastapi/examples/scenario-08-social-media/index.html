시나리오 8: 소셜 미디어 API

트위터/인스타그램 스타일의 소셜 미디어 플랫폼을 만들어봅시다!

📌 기능​

  • ✅ 팔로우/언팔로우
  • ✅ 포스트 작성 (텍스트, 이미지)
  • ✅ 좋아요, 댓글, 리트윗
  • ✅ 해시태그 자동 추출
  • ✅ 멘션(@username)
  • ✅ 타임라인 피드 (팔로우한 사람들)
  • ✅ 트렌딩 해시태그
  • ✅ 알림 시스템

    API->>Auth: Verify token

    DB-->>API: Follow created
    API->>NS: Send notification (new follower)
    API-->>U: 200 OK

    Note over U,DB: 2. 포스트 작성
    U->>API: POST /posts<br/>{content, images}
    API->>Auth: Verify token
    Auth-->>API: Current user
    API->>PS: create_post()
    PS->>HS: extract_hashtags(content)
    HS-->>PS: List of hashtags
    PS->>DB: INSERT post
    PS->>DB: INSERT/UPDATE hashtags
    PS->>DB: INSERT post_hashtags
    DB-->>PS: Post created
    PS->>HS: update_trending()
    PS-->>API: Post data
    API-->>U: 201 Created

    Note over U,DB: 3. 타임라인 피드 조회
    U->>API: GET /feed<br/>Authorization: Bearer {token}
    API->>Auth: Verify token
    Auth-->>API: Current user
    API->>FS: generate_feed(user_id)
    FS->>DB: SELECT posts FROM users<br/>WHERE id IN (following_ids)<br/>ORDER BY created_at DESC
    DB-->>FS: Posts list
    FS-->>API: Feed data
    API-->>U: 200 OK + Feed

    Note over U,DB: 4. 좋아요 & 알림
    U->>API: POST /posts/{id}/like
    API->>Auth: Verify token
    Auth-->>API: Current user
    API->>DB: INSERT INTO post_likes
    DB-->>API: Like created
    API->>DB: SELECT post author
    DB-->>API: Author info
    API->>NS: notify_like(author_id, liker_username, post_id)
    NS->>DB: INSERT notification
    API-->>U: 200 OK

    Note over U,DB: 5. 트렌딩 해시태그 조회
    U->>API: GET /hashtags/trending
    API->>HS: get_trending_hashtags()
    HS->>DB: SELECT hashtags<br/>WHERE created_at > now() - 24h<br/>GROUP BY tag<br/>ORDER BY count DESC
    DB-->>HS: Trending tags
    HS-->>API: Hashtag list
    API-->>U: 200 OK + Trending

from sqlalchemy import Column, Integer, String, ForeignKey, DateTime, Table, Text, Index
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base

# 다대다 관계 테이블들
followers = Table(
    'followers',
    Base.metadata,
    Column('follower_id', Integer, ForeignKey('users.id'), primary_key=True),
    Column('following_id', Integer, ForeignKey('users.id'), primary_key=True),
    Column('created_at', DateTime(timezone=True), server_default=func.now())
)

post_likes = Table(
    'post_likes',
    Base.metadata,
    Column('user_id', Integer, ForeignKey('users.id'), primary_key=True),
    Column('post_id', Integer, ForeignKey('posts.id'), primary_key=True),
    Column('created_at', DateTime(timezone=True), server_default=func.now())
)

class User(Base):
    __tablename__ = "users"

    id = Column(Integer, primary_key=True)
    username = Column(String, unique=True, index=True)
    email = Column(String, unique=True, index=True)
    display_name = Column(String)
    bio = Column(Text, nullable=True)
    profile_image = Column(String, nullable=True)
    hashed_password = Column(String)

    # 통계
    followers_count = Column(Integer, default=0)
    following_count = Column(Integer, default=0)
    posts_count = Column(Integer, default=0)

    created_at = Column(DateTime(timezone=True), server_default=func.now())

    # 관계
    posts = relationship("Post", back_populates="author", foreign_keys="Post.author_id")

    # 팔로워/팔로잉
    followers = relationship(
        "User",
        secondary=followers,
        primaryjoin=id == followers.c.following_id,
        secondaryjoin=id == followers.c.follower_id,
        backref="following"
    )

class Post(Base):
    __tablename__ = "posts"
    __table_args__ = (
        Index('idx_author_created', 'author_id', 'created_at'),
    )

    id = Column(Integer, primary_key=True)
    author_id = Column(Integer, ForeignKey('users.id'), index=True)
    content = Column(Text)
    image_url = Column(String, nullable=True)

    # 리트윗
    is_repost = Column(Integer, default=False)
    original_post_id = Column(Integer, ForeignKey('posts.id'), nullable=True)

    # 통계
    likes_count = Column(Integer, default=0)
    comments_count = Column(Integer, default=0)
    reposts_count = Column(Integer, default=0)

    created_at = Column(DateTime(timezone=True), server_default=func.now(), index=True)

    # 관계
    author = relationship("User", back_populates="posts", foreign_keys=[author_id])
    original_post = relationship("Post", remote_side=[id], foreign_keys=[original_post_id])
    likes = relationship("User", secondary=post_likes, backref="liked_posts")
    comments = relationship("Comment", back_populates="post")
    hashtags = relationship("PostHashtag", back_populates="post")
    mentions = relationship("PostMention", back_populates="post")

class Comment(Base):
    __tablename__ = "comments"

    id = Column(Integer, primary_key=True)
    post_id = Column(Integer, ForeignKey('posts.id'), index=True)
    author_id = Column(Integer, ForeignKey('users.id'))
    content = Column(Text)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    post = relationship("Post", back_populates="comments")
    author = relationship("User")

class Hashtag(Base):
    __tablename__ = "hashtags"

    id = Column(Integer, primary_key=True)
    tag = Column(String, unique=True, index=True)
    posts_count = Column(Integer, default=0)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    posts = relationship("PostHashtag", back_populates="hashtag")

class PostHashtag(Base):
    __tablename__ = "post_hashtags"

    post_id = Column(Integer, ForeignKey('posts.id'), primary_key=True)
    hashtag_id = Column(Integer, ForeignKey('hashtags.id'), primary_key=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    post = relationship("Post", back_populates="hashtags")
    hashtag = relationship("Hashtag", back_populates="posts")

class PostMention(Base):
    __tablename__ = "post_mentions"

    id = Column(Integer, primary_key=True)
    post_id = Column(Integer, ForeignKey('posts.id'), index=True)
    mentioned_user_id = Column(Integer, ForeignKey('users.id'), index=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    post = relationship("Post", back_populates="mentions")
    mentioned_user = relationship("User")

class Notification(Base):
    __tablename__ = "notifications"

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'), index=True)
    type = Column(String)  # "follow", "like", "comment", "mention", "repost"
    from_user_id = Column(Integer, ForeignKey('users.id'))
    post_id = Column(Integer, ForeignKey('posts.id'), nullable=True)
    is_read = Column(Integer, default=False)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    user = relationship("User", foreign_keys=[user_id])
    from_user = relationship("User", foreign_keys=[from_user_id])
    post = relationship("Post")

text_processor.py​

import re
from typing import List, Tuple

class TextProcessor:
    @staticmethod
    def extract_hashtags(text: str) -> List[str]:
        """해시태그 추출"""
        pattern = r'#(\w+)'
        hashtags = re.findall(pattern, text)
        return [tag.lower() for tag in hashtags]

    @staticmethod
    def extract_mentions(text: str) -> List[str]:
        """멘션 추출"""
        pattern = r'@(\w+)'
        mentions = re.findall(pattern, text)
        return mentions

    @staticmethod
    def linkify_text(text: str) -> str:
        """해시태그와 멘션을 링크로 변환"""
        # 해시태그 링크화
        text = re.sub(
            r'#(\w+)',
            r'<a href="/hashtags/\1">#\1</a>',
            text
        )

        # 멘션 링크화
        text = re.sub(
            r'@(\w+)',
            r'<a href="/users/\1">@\1</a>',
            text
        )

        return text

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="Social Media API")

text_processor = TextProcessor()

# ==================== 팔로우 ====================

@app.post("/users/{username}/follow")
def follow_user(
    username: str,
    current_user: User = Depends(get_current_user),
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)
):
    """사용자 팔로우"""
    target_user = db.query(models.User).filter(
        models.User.username == username
    ).first()

    if not target_user:
        raise HTTPException(status_code=404, detail="User not found")

    if target_user.id == current_user.id:
        raise HTTPException(status_code=400, detail="Cannot follow yourself")

    # 이미 팔로우 중인지 확인
    if target_user in current_user.following:
        raise HTTPException(status_code=400, detail="Already following")

    # 팔로우 추가
    current_user.following.append(target_user)
    current_user.following_count += 1
    target_user.followers_count += 1

    db.commit()

    # 알림 생성
    background_tasks.add_task(
        create_notification,
        db,
        target_user.id,
        "follow",
        current_user.id
    )

    return {"message": f"Now following @{username}"}

@app.delete("/users/{username}/follow")
def unfollow_user(
    username: str,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """언팔로우"""
    target_user = db.query(models.User).filter(
        models.User.username == username
    ).first()

    if not target_user:
        raise HTTPException(status_code=404, detail="User not found")

    if target_user not in current_user.following:
        raise HTTPException(status_code=400, detail="Not following")

    # 언팔로우
    current_user.following.remove(target_user)
    current_user.following_count -= 1
    target_user.followers_count -= 1

    db.commit()

    return {"message": f"Unfollowed @{username}"}

# ==================== 포스트 ====================

@app.post("/posts", response_model=schemas.Post)
def create_post(
    post: schemas.PostCreate,
    current_user: User = Depends(get_current_user),
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)
):
    """포스트 작성"""
    # 포스트 생성
    db_post = models.Post(
        author_id=current_user.id,
        content=post.content,
        image_url=post.image_url
    )
    db.add(db_post)
    db.flush()

    # 해시태그 추출 및 저장
    hashtags = text_processor.extract_hashtags(post.content)
    for tag in hashtags:
        hashtag = db.query(models.Hashtag).filter(
            models.Hashtag.tag == tag
        ).first()

        if not hashtag:
            hashtag = models.Hashtag(tag=tag)
            db.add(hashtag)
            db.flush()

        # 포스트-해시태그 연결
        post_hashtag = models.PostHashtag(
            post_id=db_post.id,
            hashtag_id=hashtag.id
        )
        db.add(post_hashtag)
        hashtag.posts_count += 1

    # 멘션 추출 및 저장
    mentions = text_processor.extract_mentions(post.content)
    for mention in mentions:
        mentioned_user = db.query(models.User).filter(
            models.User.username == mention
        ).first()

        if mentioned_user:
            post_mention = models.PostMention(
                post_id=db_post.id,
                mentioned_user_id=mentioned_user.id
            )
            db.add(post_mention)

            # 멘션 알림
            background_tasks.add_task(
                create_notification,
                db,
                mentioned_user.id,
                "mention",
                current_user.id,
                db_post.id
            )

    current_user.posts_count += 1
    db.commit()
    db.refresh(db_post)

    return db_post

@app.get("/timeline", response_model=List[schemas.Post])
def get_timeline(
    skip: int = 0,
    limit: int = 20,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """타임라인 피드 (팔로우한 사용자들의 포스트)"""
    # 팔로우한 사용자 ID 목록
    following_ids = [user.id for user in current_user.following]
    following_ids.append(current_user.id)  # 본인 포스트도 포함

    posts = db.query(models.Post).filter(
        models.Post.author_id.in_(following_ids)
    ).order_by(desc(models.Post.created_at)).offset(skip).limit(limit).all()

    return posts

@app.post("/posts/{post_id}/like")
def like_post(
    post_id: int,
    current_user: User = Depends(get_current_user),
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)
):
    """포스트 좋아요"""
    post = db.query(models.Post).filter(models.Post.id == post_id).first()
    if not post:
        raise HTTPException(status_code=404, detail="Post not found")

    # 이미 좋아요 했는지 확인
    if post in current_user.liked_posts:
        raise HTTPException(status_code=400, detail="Already liked")

    # 좋아요 추가
    current_user.liked_posts.append(post)
    post.likes_count += 1
    db.commit()

    # 알림
    if post.author_id != current_user.id:
        background_tasks.add_task(
            create_notification,
            db,
            post.author_id,
            "like",
            current_user.id,
            post_id
        )

    return {"message": "Post liked"}

@app.post("/posts/{post_id}/repost")
def repost(
    post_id: int,
    current_user: User = Depends(get_current_user),
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db)
):
    """리트윗"""
    original_post = db.query(models.Post).filter(models.Post.id == post_id).first()
    if not original_post:
        raise HTTPException(status_code=404, detail="Post not found")

    # 리포스트 생성
    repost = models.Post(
        author_id=current_user.id,
        content=original_post.content,
        image_url=original_post.image_url,
        is_repost=True,
        original_post_id=original_post.id
    )
    db.add(repost)

    original_post.reposts_count += 1
    current_user.posts_count += 1
    db.commit()

    # 알림
    if original_post.author_id != current_user.id:
        background_tasks.add_task(
            create_notification,
            db,
            original_post.author_id,
            "repost",
            current_user.id,
            post_id
        )

    return {"message": "Post reposted", "repost_id": repost.id}

@app.get("/hashtags/{tag}/posts", response_model=List[schemas.Post])
def get_posts_by_hashtag(
    tag: str,
    skip: int = 0,
    limit: int = 20,
    db: Session = Depends(get_db)
):
    """해시태그로 포스트 검색"""
    hashtag = db.query(models.Hashtag).filter(
        models.Hashtag.tag == tag.lower()
    ).first()

    if not hashtag:
        return []

    post_ids = [ph.post_id for ph in hashtag.posts]
    posts = db.query(models.Post).filter(
        models.Post.id.in_(post_ids)
    ).order_by(desc(models.Post.created_at)).offset(skip).limit(limit).all()

    return posts

# ==================== 알림 ====================

@app.get("/notifications", response_model=List[schemas.Notification])
def get_notifications(
    unread_only: bool = False,
    skip: int = 0,
    limit: int = 20,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """알림 조회"""
    query = db.query(models.Notification).filter(
        models.Notification.user_id == current_user.id
    )

    if unread_only:
        query = query.filter(models.Notification.is_read == False)

    notifications = query.order_by(
        desc(models.Notification.created_at)
    ).offset(skip).limit(limit).all()

    return notifications

@app.put("/notifications/{notification_id}/read")
def mark_notification_read(
    notification_id: int,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """알림 읽음 표시"""
    notification = db.query(models.Notification).filter(
        models.Notification.id == notification_id,
        models.Notification.user_id == current_user.id
    ).first()

    if not notification:
        raise HTTPException(status_code=404, detail="Notification not found")

    notification.is_read = True
    db.commit()

    return {"message": "Notification marked as read"}

# ==================== 헬퍼 함수 ====================

def create_notification(
    db: Session,
    user_id: int,
    type: str,
    from_user_id: int,
    post_id: int = None
):
    """알림 생성"""
    notification = models.Notification(
        user_id=user_id,
        type=type,
        from_user_id=from_user_id,
        post_id=post_id
    )
    db.add(notification)
    db.commit()
