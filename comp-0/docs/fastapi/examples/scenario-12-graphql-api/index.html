시나리오 12: GraphQL API

Strawberry를 활용한 GraphQL API를 만들어봅시다!

📌 기능​

  • ✅ GraphQL 쿼리/뮤테이션/구독
  • ✅ 복잡한 중첩 쿼리
  • ✅ DataLoader로 N+1 문제 해결
  • ✅ 필드 레벨 권한
  • ✅ 페이지네이션 (Relay 스타일)
  • ✅ GraphQL Playground
  • ✅ 실시간 구독 (WebSocket)

📊 시퀀스 다이어그램​

sequenceDiagram
    participant C as Client
    participant GQL as GraphQL
    participant DL as DataLoader
    participant DB as Database

    Note over C,DB: 1. 중첩 쿼리 (N+1 해결)
    C->>GQL: query {<br/>  authors {<br/>    books { title }<br/>  }<br/>}
    GQL->>DB: SELECT * FROM authors
    DB-->>GQL: Authors list
    loop For each author
        GQL->>DL: Load books for author_id
    end
    DL->>DL: Batch all author_ids
    DL->>DB: SELECT * FROM books<br/>WHERE author_id IN (...)
    DB-->>DL: All books (single query)
    DL-->>GQL: Books grouped by author
    GQL-->>C: JSON response

    Note over C,DB: 2. 뮤테이션
    C->>GQL: mutation {<br/>  createBook(input) {<br/>    id title<br/>  }<br/>}
    GQL->>GQL: Check permissions
    GQL->>DB: INSERT INTO books
    DB-->>GQL: Book created
    GQL-->>C: Book data

models.py​

from sqlalchemy import Column, Integer, String, ForeignKey, DateTime, Text, Float
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base

class Author(Base):
    __tablename__ = "authors"

    id = Column(Integer, primary_key=True)
    name = Column(String, index=True)
    email = Column(String, unique=True)
    bio = Column(Text, nullable=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    books = relationship("Book", back_populates="author")

class Book(Base):
    __tablename__ = "books"

    id = Column(Integer, primary_key=True)
    title = Column(String, index=True)
    description = Column(Text)
    isbn = Column(String, unique=True)
    price = Column(Float)
    published_date = Column(DateTime(timezone=True))
    author_id = Column(Integer, ForeignKey('authors.id'))

    author = relationship("Author", back_populates="books")
    reviews = relationship("Review", back_populates="book")

class Review(Base):
    __tablename__ = "reviews"

    id = Column(Integer, primary_key=True)
    book_id = Column(Integer, ForeignKey('books.id'))
    user_id = Column(Integer, ForeignKey('users.id'))
    rating = Column(Integer)  # 1-5
    comment = Column(Text)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    book = relationship("Book", back_populates="reviews")
    user = relationship("User")

graphql_schema.py​

import strawberry
from strawberry.types import Info
from typing import List, Optional
from datetime import datetime
from sqlalchemy.orm import Session
import models

# ==================== Types ====================

@strawberry.type
class Author:
    id: int
    name: str
    email: str
    bio: Optional[str]
    created_at: datetime

    @strawberry.field
    async def books(self, info: Info) -> List["Book"]:
        """작가의 책 목록"""
        db: Session = info.context["db"]
        return db.query(models.Book).filter(
            models.Book.author_id == self.id
        ).all()

    @strawberry.field
    def book_count(self, info: Info) -> int:
        """책 개수"""
        db: Session = info.context["db"]
        return db.query(models.Book).filter(
            models.Book.author_id == self.id
        ).count()

@strawberry.type
class Book:
    id: int
    title: str
    description: str
    isbn: str
    price: float
    published_date: datetime
    author_id: int

    @strawberry.field
    async def author(self, info: Info) -> Author:
        """작가 정보"""
        db: Session = info.context["db"]

        # DataLoader 사용 (N+1 문제 해결)
        loader = info.context["author_loader"]
        db_author = await loader.load(self.author_id)

        return Author(
            id=db_author.id,
            name=db_author.name,
            email=db_author.email,
            bio=db_author.bio,
            created_at=db_author.created_at
        )

    @strawberry.field
    def reviews(self, info: Info) -> List["Review"]:
        """리뷰 목록"""
        db: Session = info.context["db"]
        return db.query(models.Review).filter(
            models.Review.book_id == self.id
        ).all()

    @strawberry.field
    def average_rating(self, info: Info) -> Optional[float]:
        """평균 평점"""
        db: Session = info.context["db"]
        from sqlalchemy import func

        result = db.query(
            func.avg(models.Review.rating)
        ).filter(
            models.Review.book_id == self.id
        ).scalar()

        return float(result) if result else None

@strawberry.type
class Review:
    id: int
    rating: int
    comment: str
    created_at: datetime
    book_id: int
    user_id: int

    @strawberry.field
    def book(self, info: Info) -> Book:
        """리뷰 대상 책"""
        db: Session = info.context["db"]
        db_book = db.query(models.Book).filter(
            models.Book.id == self.book_id
        ).first()

        return Book(
            id=db_book.id,
            title=db_book.title,
            description=db_book.description,
            isbn=db_book.isbn,
            price=db_book.price,
            published_date=db_book.published_date,
            author_id=db_book.author_id
        )

# ==================== Pagination ====================

@strawberry.type
class PageInfo:
    has_next_page: bool
    has_previous_page: bool
    start_cursor: Optional[str]
    end_cursor: Optional[str]

@strawberry.type
class BookEdge:
    node: Book
    cursor: str

@strawberry.type
class BookConnection:
    edges: List[BookEdge]
    page_info: PageInfo
    total_count: int

# ==================== Inputs ====================

@strawberry.input
class CreateAuthorInput:
    name: str
    email: str
    bio: Optional[str] = None

@strawberry.input
class CreateBookInput:
    title: str
    description: str
    isbn: str
    price: float
    published_date: datetime
    author_id: int

@strawberry.input
class CreateReviewInput:
    book_id: int
    rating: int
    comment: str

@strawberry.input
class BookFilter:
    author_id: Optional[int] = None
    min_price: Optional[float] = None
    max_price: Optional[float] = None
    search: Optional[str] = None

# ==================== Query ====================

@strawberry.type
class Query:
    @strawberry.field
    def authors(self, info: Info, limit: int = 10) -> List[Author]:
        """작가 목록"""
        db: Session = info.context["db"]
        db_authors = db.query(models.Author).limit(limit).all()

        return [
            Author(
                id=a.id,
                name=a.name,
                email=a.email,
                bio=a.bio,
                created_at=a.created_at
            )
            for a in db_authors
        ]

    @strawberry.field
    def author(self, info: Info, id: int) -> Optional[Author]:
        """작가 조회"""
        db: Session = info.context["db"]
        db_author = db.query(models.Author).filter(
            models.Author.id == id
        ).first()

        if not db_author:
            return None

        return Author(
            id=db_author.id,
            name=db_author.name,
            email=db_author.email,
            bio=db_author.bio,
            created_at=db_author.created_at
        )

    @strawberry.field
    def books(
        self,
        info: Info,
        filter: Optional[BookFilter] = None,
        first: int = 10,
        after: Optional[str] = None
    ) -> BookConnection:
        """책 목록 (Relay Pagination)"""
        db: Session = info.context["db"]

        query = db.query(models.Book)

        # 필터 적용
        if filter:
            if filter.author_id:
                query = query.filter(models.Book.author_id == filter.author_id)

            if filter.min_price:
                query = query.filter(models.Book.price >= filter.min_price)

            if filter.max_price:
                query = query.filter(models.Book.price <= filter.max_price)

            if filter.search:
                query = query.filter(
                    models.Book.title.ilike(f"%{filter.search}%")
                )

        # 커서 기반 페이지네이션
        if after:
            import base64
            cursor_id = int(base64.b64decode(after).decode())
            query = query.filter(models.Book.id > cursor_id)

        total_count = query.count()
        db_books = query.limit(first + 1).all()

        has_next_page = len(db_books) > first
        if has_next_page:
            db_books = db_books[:first]

        edges = []
        for book in db_books:
            import base64
            cursor = base64.b64encode(str(book.id).encode()).decode()

            edges.append(BookEdge(
                node=Book(
                    id=book.id,
                    title=book.title,
                    description=book.description,
                    isbn=book.isbn,
                    price=book.price,
                    published_date=book.published_date,
                    author_id=book.author_id
                ),
                cursor=cursor
            ))

        return BookConnection(
            edges=edges,
            page_info=PageInfo(
                has_next_page=has_next_page,
                has_previous_page=after is not None,
                start_cursor=edges[0].cursor if edges else None,
                end_cursor=edges[-1].cursor if edges else None
            ),
            total_count=total_count
        )

    @strawberry.field
    def book(self, info: Info, id: int) -> Optional[Book]:
        """책 조회"""
        db: Session = info.context["db"]
        db_book = db.query(models.Book).filter(models.Book.id == id).first()

        if not db_book:
            return None

        return Book(
            id=db_book.id,
            title=db_book.title,
            description=db_book.description,
            isbn=db_book.isbn,
            price=db_book.price,
            published_date=db_book.published_date,
            author_id=db_book.author_id
        )

# ==================== Mutation ====================

@strawberry.type
class Mutation:
    @strawberry.mutation
    def create_author(self, info: Info, input: CreateAuthorInput) -> Author:
        """작가 생성"""
        db: Session = info.context["db"]

        db_author = models.Author(
            name=input.name,
            email=input.email,
            bio=input.bio
        )

        db.add(db_author)
        db.commit()
        db.refresh(db_author)

        return Author(
            id=db_author.id,
            name=db_author.name,
            email=db_author.email,
            bio=db_author.bio,
            created_at=db_author.created_at
        )

    @strawberry.mutation
    def create_book(self, info: Info, input: CreateBookInput) -> Book:
        """책 생성"""
        db: Session = info.context["db"]

        db_book = models.Book(
            title=input.title,
            description=input.description,
            isbn=input.isbn,
            price=input.price,
            published_date=input.published_date,
            author_id=input.author_id
        )

        db.add(db_book)
        db.commit()
        db.refresh(db_book)

        return Book(
            id=db_book.id,
            title=db_book.title,
            description=db_book.description,
            isbn=db_book.isbn,
            price=db_book.price,
            published_date=db_book.published_date,
            author_id=db_book.author_id
        )

    @strawberry.mutation
    def create_review(self, info: Info, input: CreateReviewInput) -> Review:
        """리뷰 작성"""
        db: Session = info.context["db"]
        user = info.context.get("user")

        if not user:
            raise Exception("Authentication required")

        db_review = models.Review(
            book_id=input.book_id,
            user_id=user.id,
            rating=input.rating,
            comment=input.comment
        )

        db.add(db_review)
        db.commit()
        db.refresh(db_review)

        # 구독자들에게 알림
        pubsub = info.context["pubsub"]
        pubsub.publish(f"book_{input.book_id}", db_review)

        return Review(
            id=db_review.id,
            rating=db_review.rating,
            comment=db_review.comment,
            created_at=db_review.created_at,
            book_id=db_review.book_id,
            user_id=db_review.user_id
        )

# ==================== Subscription ====================

@strawberry.type
class Subscription:
    @strawberry.subscription
    async def review_added(self, info: Info, book_id: int) -> Review:
        """새 리뷰 실시간 구독"""
        pubsub = info.context["pubsub"]

        async for review in pubsub.subscribe(f"book_{book_id}"):
            yield Review(
                id=review.id,
                rating=review.rating,
                comment=review.comment,
                created_at=review.created_at,
                book_id=review.book_id,
                user_id=review.user_id
            )

# ==================== Schema ====================

schema = strawberry.Schema(
    query=Query,
    mutation=Mutation,
    subscription=Subscription
)

dataloader.py​

from strawberry.dataloader import DataLoader
from sqlalchemy.orm import Session
from typing import List
import models

async def load_authors(keys: List[int], db: Session) -> List[models.Author]:
    """작가 배치 로딩 (N+1 문제 해결)"""
    authors = db.query(models.Author).filter(
        models.Author.id.in_(keys)
    ).all()

    # 키 순서대로 정렬
    author_map = {author.id: author for author in authors}
    return [author_map.get(key) for key in keys]

def get_author_loader(db: Session) -> DataLoader:
    """DataLoader 인스턴스 생성"""
    async def load_fn(keys):
        return await load_authors(keys, db)

    return DataLoader(load_fn=load_fn)

pubsub.py​

import asyncio
from typing import AsyncGenerator, Dict, Set

class PubSub:
    """간단한 PubSub 구현 (실제로는 Redis 사용 권장)"""

    def __init__(self):
        self.subscribers: Dict[str, Set[asyncio.Queue]] = {}

    def publish(self, channel: str, message):
        """메시지 발행"""
        if channel in self.subscribers:
            for queue in self.subscribers[channel]:
                queue.put_nowait(message)

    async def subscribe(self, channel: str) -> AsyncGenerator:
        """채널 구독"""
        queue = asyncio.Queue()

        if channel not in self.subscribers:
            self.subscribers[channel] = set()

        self.subscribers[channel].add(queue)

        try:
            while True:
                message = await queue.get()
                yield message
        finally:
            self.subscribers[channel].remove(queue)
            if not self.subscribers[channel]:
                del self.subscribers[channel]

main.py​

from fastapi import FastAPI, Depends
from strawberry.fastapi import GraphQLRouter
from sqlalchemy.orm import Session
import models
from database import engine, get_db
from graphql_schema import schema
from dataloader import get_author_loader
from pubsub import PubSub

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="GraphQL API")

# PubSub 인스턴스
pubsub = PubSub()

# Context 생성
async def get_context(db: Session = Depends(get_db)):
    return {
        "db": db,
        "author_loader": get_author_loader(db),
        "pubsub": pubsub,
        # "user": current_user  # 인증된 사용자
    }

# GraphQL Router
graphql_app = GraphQLRouter(
    schema,
    context_getter=get_context,
    graphiql=True  # GraphQL Playground 활성화
)

app.include_router(graphql_app, prefix="/graphql")

@app.get("/")
def root():
    return {
        "message": "GraphQL API",
        "graphql_endpoint": "/graphql",
        "playground": "/graphql"
    }
