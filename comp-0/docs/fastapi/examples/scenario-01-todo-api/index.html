# models.py
from sqlalchemy import Column, Integer, String, Boolean, DateTime
from sqlalchemy.sql import func
from database import Base

class Todo(Base):
    __tablename__ = "todos"

    id = Column(Integer, primary_key=True, index=True)
    title = Column(String, index=True)
    description = Column(String, nullable=True)
    completed = Column(Boolean, default=False)
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    updated_at = Column(DateTime(timezone=True), onupdate=func.now())

# main.py
from fastapi import FastAPI, Depends, HTTPException, status
from sqlalchemy.orm import Session
from typing import List, Optional
import models, schemas
from database import engine, get_db

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="To-Do API")

@app.post("/todos", response_model=schemas.Todo, status_code=status.HTTP_201_CREATED)
def create_todo(todo: schemas.TodoCreate, db: Session = Depends(get_db)):
    """ìƒˆë¡œìš´ í•  ì¼ ìƒì„±"""
    db_todo = models.Todo(**todo.dict())
    db.add(db_todo)
    db.commit()
    db.refresh(db_todo)
    return db_todo

@app.get("/todos", response_model=List[schemas.Todo])
def list_todos(
    skip: int = 0,
    limit: int = 10,
    completed: Optional[bool] = None,
    db: Session = Depends(get_db)
):
    """í•  ì¼ ëª©ë¡ ì¡°íšŒ"""
    query = db.query(models.Todo)

    if completed is not None:
        query = query.filter(models.Todo.completed == completed)

    todos = query.offset(skip).limit(limit).all()
    return todos

@app.get("/todos/{todo_id}", response_model=schemas.Todo)
def get_todo(todo_id: int, db: Session = Depends(get_db)):
    """íŠ¹ì • í•  ì¼ ì¡°íšŒ"""
    todo = db.query(models.Todo).filter(models.Todo.id == todo_id).first()
    if not todo:
        raise HTTPException(status_code=404, detail="Todo not found")
    return todo

@app.put("/todos/{todo_id}", response_model=schemas.Todo)
def update_todo(
    todo_id: int,
    todo_update: schemas.TodoUpdate,
    db: Session = Depends(get_db)
):
    """í•  ì¼ ìˆ˜ì •"""
    todo = db.query(models.Todo).filter(models.Todo.id == todo_id).first()
    if not todo:
        raise HTTPException(status_code=404, detail="Todo not found")

    for key, value in todo_update.dict(exclude_unset=True).items():
        setattr(todo, key, value)

    db.commit()
    db.refresh(todo)
    return todo

@app.delete("/todos/{todo_id}", status_code=status.HTTP_204_NO_CONTENT)
def delete_todo(todo_id: int, db: Session = Depends(get_db)):
    """í•  ì¼ ì‚­ì œ"""
    todo = db.query(models.Todo).filter(models.Todo.id == todo_id).first()
    if not todo:
        raise HTTPException(status_code=404, detail="Todo not found")

    db.delete(todo)
    db.commit()

@app.patch("/todos/{todo_id}/toggle", response_model=schemas.Todo)
def toggle_todo(todo_id: int, db: Session = Depends(get_db)):
    """ì™„ë£Œ ìƒíƒœ í† ê¸€"""
    todo = db.query(models.Todo).filter(models.Todo.id == todo_id).first()
    if not todo:
        raise HTTPException(status_code=404, detail="Todo not found")

    todo.completed = not todo.completed
    db.commit()
    db.refresh(todo)
    return todo

# test_main.py
from fastapi.testclient import TestClient
from main import app

client = TestClient(app)

def test_create_todo():
    response = client.post("/todos", json={"title": "Test Todo"})
    assert response.status_code == 201
    assert response.json()["title"] == "Test Todo"
    assert response.json()["completed"] == False

def test_list_todos():
    response = client.get("/todos")
    assert response.status_code == 200
    assert isinstance(response.json(), list)

def test_get_todo():
    # Create a todo first
    create_response = client.post("/todos", json={"title": "Test"})
    todo_id = create_response.json()["id"]

    # Get the todo
    response = client.get(f"/todos/{todo_id}")
    assert response.status_code == 200
    assert response.json()["id"] == todo_id

def test_update_todo():
    # Create
    create_response = client.post("/todos", json={"title": "Original"})
    todo_id = create_response.json()["id"]

    # Update
    response = client.put(
        f"/todos/{todo_id}",
        json={"title": "Updated", "completed": True}
    )
    assert response.status_code == 200
    assert response.json()["title"] == "Updated"
    assert response.json()["completed"] == True

def test_toggle_todo():
    # Create
    create_response = client.post("/todos", json={"title": "Test"})
    todo_id = create_response.json()["id"]

    # Toggle
    response = client.patch(f"/todos/{todo_id}/toggle")
    assert response.status_code == 200
    assert response.json()["completed"] == True

    # Toggle again
    response = client.patch(f"/todos/{todo_id}/toggle")
    assert response.json()["completed"] == False

def test_delete_todo():
    # Create
    create_response = client.post("/todos", json={"title": "Test"})
    todo_id = create_response.json()["id"]

    # Delete
    response = client.delete(f"/todos/{todo_id}")
    assert response.status_code == 204

    # Verify deleted
    response = client.get(f"/todos/{todo_id}")
    assert response.status_code == 404
