시나리오 11: 검색 엔진 API

Elasticsearch를 활용한 강력한 검색 엔진을 만들어봅시다!

📌 기능​

  • ✅ 전문 검색 (Full-text Search)
  • ✅ 자동완성 (Autocomplete)
  • ✅ 패싯 검색 (Faceted Search)
  • ✅ 하이라이팅
  • ✅ 유사 문서 검색
  • ✅ 검색 제안 (Did you mean?)
  • ✅ 검색 분석 (인기 검색어)
  • ✅ 필터링 및 정렬

📊 시퀀스 다이어그램​

sequenceDiagram
    participant U as User
    participant API as FastAPI
    participant SS as SearchService
    participant ES as Elasticsearch
    participant DB as Database

    Note over U,DB: 1. 전문 검색
    U->>API: GET /search?q=laptop&category=electronics
    API->>SS: search(query, filters)
    SS->>ES: Search query with filters, highlighting
    ES-->>SS: Search results + highlights + aggregations
    SS->>DB: INSERT search_log
    SS-->>API: Results with facets
    API-->>U: 200 OK + Results

    Note over U,DB: 2. 자동완성
    U->>API: GET /autocomplete?q=lap
    API->>SS: autocomplete(prefix)
    SS->>ES: Prefix query with completion suggester
    ES-->>SS: Suggested completions
    SS-->>API: Suggestions
    API-->>U: 200 OK + ["laptop", "laparoscopy", ...]

    Note over U,DB: 3. 문서 인덱싱
    API->>DB: INSERT/UPDATE product
    DB-->>API: Product saved
    API->>SS: index_document(product)
    SS->>ES: Index/Update document
    ES-->>SS: Indexed

models.py​

from sqlalchemy import Column, Integer, String, DateTime, Text, Float, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base

class Product(Base):
    """상품"""
    __tablename__ = "products"

    id = Column(Integer, primary_key=True)
    name = Column(String, index=True)
    description = Column(Text)
    category = Column(String, index=True)
    brand = Column(String, index=True)
    price = Column(Float)
    stock = Column(Integer)
    tags = Column(String)  # 쉼표로 구분된 태그

    # 검색 최적화
    search_keywords = Column(Text)  # 추가 검색 키워드

    created_at = Column(DateTime(timezone=True), server_default=func.now())
    updated_at = Column(DateTime(timezone=True), onupdate=func.now())

class SearchLog(Base):
    """검색 로그"""
    __tablename__ = "search_logs"

    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'), nullable=True)
    query = Column(String, index=True)
    filters = Column(Text)  # JSON
    results_count = Column(Integer)
    clicked_result_id = Column(Integer, nullable=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now(), index=True)

    user = relationship("User")

class SearchSuggestion(Base):
    """검색 제안"""
    __tablename__ = "search_suggestions"

    id = Column(Integer, primary_key=True)
    query = Column(String, unique=True, index=True)
    suggestion = Column(String)
    count = Column(Integer, default=1)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

elasticsearch_client.py​

from elasticsearch import Elasticsearch
from typing import List, Dict, Optional
import os

class ElasticsearchClient:
    def __init__(self):
        self.es = Elasticsearch(
            [os.getenv('ELASTICSEARCH_URL', 'http://localhost:9200')]
        )
        self.index_name = "products"
        self._create_index()

    def _create_index(self):
        """인덱스 생성"""
        if not self.es.indices.exists(index=self.index_name):
            mapping = {
                "mappings": {
                    "properties": {
                        "name": {
                            "type": "text",
                            "analyzer": "standard",
                            "fields": {
                                "keyword": {"type": "keyword"},
                                "completion": {"type": "completion"}
                            }
                        },
                        "description": {
                            "type": "text",
                            "analyzer": "standard"
                        },
                        "category": {
                            "type": "keyword"
                        },
                        "brand": {
                            "type": "keyword"
                        },
                        "price": {
                            "type": "float"
                        },
                        "stock": {
                            "type": "integer"
                        },
                        "tags": {
                            "type": "keyword"
                        },
                        "search_keywords": {
                            "type": "text"
                        },
                        "created_at": {
                            "type": "date"
                        }
                    }
                }
            }
            self.es.indices.create(index=self.index_name, body=mapping)

    def index_product(self, product_id: int, product_data: dict):
        """상품 인덱싱"""
        self.es.index(
            index=self.index_name,
            id=product_id,
            body=product_data
        )

    def bulk_index_products(self, products: List[dict]):
        """배치 인덱싱"""
        from elasticsearch.helpers import bulk

        actions = [
            {
                "_index": self.index_name,
                "_id": product["id"],
                "_source": product
            }
            for product in products
        ]

        bulk(self.es, actions)

    def search(
        self,
        query: str,
        filters: Optional[Dict] = None,
        sort_by: Optional[str] = None,
        from_: int = 0,
        size: int = 10
    ) -> dict:
        """전문 검색"""
        # 기본 쿼리
        must_clauses = []

        if query:
            must_clauses.append({
                "multi_match": {
                    "query": query,
                    "fields": [
                        "name^3",  # name 필드에 가중치 3
                        "description^2",
                        "search_keywords",
                        "brand",
                        "tags"
                    ],
                    "fuzziness": "AUTO"  # 오타 허용
                }
            })

        # 필터 적용
        filter_clauses = []

        if filters:
            if "category" in filters:
                filter_clauses.append({"term": {"category": filters["category"]}})

            if "brand" in filters:
                filter_clauses.append({"term": {"brand": filters["brand"]}})

            if "min_price" in filters or "max_price" in filters:
                price_range = {}
                if "min_price" in filters:
                    price_range["gte"] = filters["min_price"]
                if "max_price" in filters:
                    price_range["lte"] = filters["max_price"]

                filter_clauses.append({"range": {"price": price_range}})

            if "in_stock" in filters and filters["in_stock"]:
                filter_clauses.append({"range": {"stock": {"gt": 0}}})

        # 정렬
        sort_options = []
        if sort_by == "price_asc":
            sort_options.append({"price": {"order": "asc"}})
        elif sort_by == "price_desc":
            sort_options.append({"price": {"order": "desc"}})
        elif sort_by == "newest":
            sort_options.append({"created_at": {"order": "desc"}})
        else:
            sort_options.append("_score")  # 관련성 순

        # 검색 쿼리 구성
        search_body = {
            "query": {
                "bool": {
                    "must": must_clauses if must_clauses else [{"match_all": {}}],
                    "filter": filter_clauses
                }
            },
            "sort": sort_options,
            "from": from_,
            "size": size,
            "highlight": {
                "fields": {
                    "name": {},
                    "description": {}
                },
                "pre_tags": ["<mark>"],
                "post_tags": ["</mark>"]
            },
            "aggs": {
                "categories": {
                    "terms": {"field": "category", "size": 20}
                },
                "brands": {
                    "terms": {"field": "brand", "size": 20}
                },
                "price_ranges": {
                    "range": {
                        "field": "price",
                        "ranges": [
                            {"to": 10000, "key": "0-10000"},
                            {"from": 10000, "to": 50000, "key": "10000-50000"},
                            {"from": 50000, "to": 100000, "key": "50000-100000"},
                            {"from": 100000, "key": "100000+"}
                        ]
                    }
                }
            }
        }

        response = self.es.search(index=self.index_name, body=search_body)

        # 결과 포맷팅
        results = []
        for hit in response["hits"]["hits"]:
            result = hit["_source"]
            result["id"] = hit["_id"]
            result["score"] = hit["_score"]

            # 하이라이트
            if "highlight" in hit:
                result["highlight"] = hit["highlight"]

            results.append(result)

        return {
            "total": response["hits"]["total"]["value"],
            "results": results,
            "facets": {
                "categories": response["aggregations"]["categories"]["buckets"],
                "brands": response["aggregations"]["brands"]["buckets"],
                "price_ranges": response["aggregations"]["price_ranges"]["buckets"]
            }
        }

    def autocomplete(self, prefix: str, size: int = 5) -> List[str]:
        """자동완성"""
        search_body = {
            "suggest": {
                "product-suggest": {
                    "prefix": prefix,
                    "completion": {
                        "field": "name.completion",
                        "size": size,
                        "skip_duplicates": True
                    }
                }
            }
        }

        response = self.es.search(index=self.index_name, body=search_body)

        suggestions = []
        for option in response["suggest"]["product-suggest"][0]["options"]:
            suggestions.append(option["text"])

        return suggestions

    def suggest_similar(self, query: str) -> List[str]:
        """유사 검색어 제안 (Did you mean?)"""
        search_body = {
            "suggest": {
                "text": query,
                "simple_phrase": {
                    "phrase": {
                        "field": "name",
                        "size": 1,
                        "gram_size": 3,
                        "direct_generator": [{
                            "field": "name",
                            "suggest_mode": "always"
                        }],
                        "highlight": {
                            "pre_tag": "<em>",
                            "post_tag": "</em>"
                        }
                    }
                }
            }
        }

        response = self.es.search(index=self.index_name, body=search_body)

        suggestions = []
        for option in response["suggest"]["simple_phrase"][0]["options"]:
            suggestions.append(option["text"])

        return suggestions

    def more_like_this(self, product_id: int, size: int = 5) -> List[dict]:
        """유사 상품 추천"""
        search_body = {
            "query": {
                "more_like_this": {
                    "fields": ["name", "description", "category", "tags"],
                    "like": [
                        {
                            "_index": self.index_name,
                            "_id": product_id
                        }
                    ],
                    "min_term_freq": 1,
                    "max_query_terms": 12
                }
            },
            "size": size
        }

        response = self.es.search(index=self.index_name, body=search_body)

        results = []
        for hit in response["hits"]["hits"]:
            result = hit["_source"]
            result["id"] = hit["_id"]
            results.append(result)

        return results

    def delete_product(self, product_id: int):
        """상품 삭제"""
        self.es.delete(index=self.index_name, id=product_id)

search_analytics.py​

from sqlalchemy.orm import Session
from sqlalchemy import func, desc
from datetime import datetime, timedelta
import models

class SearchAnalytics:
    def __init__(self, db: Session):
        self.db = db

    def log_search(
        self,
        query: str,
        filters: dict,
        results_count: int,
        user_id: int = None
    ):
        """검색 로그 저장"""
        import json

        log = models.SearchLog(
            user_id=user_id,
            query=query,
            filters=json.dumps(filters),
            results_count=results_count
        )

        self.db.add(log)
        self.db.commit()

    def log_click(self, search_log_id: int, product_id: int):
        """검색 결과 클릭 로그"""
        log = self.db.query(models.SearchLog).filter(
            models.SearchLog.id == search_log_id
        ).first()

        if log:
            log.clicked_result_id = product_id
            self.db.commit()

    def get_popular_searches(self, days: int = 7, limit: int = 10) -> List[dict]:
        """인기 검색어"""
        since = datetime.now() - timedelta(days=days)

        results = self.db.query(
            models.SearchLog.query,
            func.count(models.SearchLog.id).label('count')
        ).filter(
            models.SearchLog.created_at >= since
        ).group_by(
            models.SearchLog.query
        ).order_by(
            desc('count')
        ).limit(limit).all()

        return [{"query": r.query, "count": r.count} for r in results]

    def get_trending_searches(self, hours: int = 24, limit: int = 10) -> List[dict]:
        """실시간 트렌딩 검색어"""
        since = datetime.now() - timedelta(hours=hours)

        results = self.db.query(
            models.SearchLog.query,
            func.count(models.SearchLog.id).label('count')
        ).filter(
            models.SearchLog.created_at >= since
        ).group_by(
            models.SearchLog.query
        ).order_by(
            desc('count')
        ).limit(limit).all()

        return [{"query": r.query, "count": r.count} for r in results]

    def get_zero_result_queries(self, days: int = 7, limit: int = 10) -> List[str]:
        """검색 결과 없는 쿼리 (개선 필요)"""
        since = datetime.now() - timedelta(days=days)

        results = self.db.query(
            models.SearchLog.query
        ).filter(
            models.SearchLog.created_at >= since,
            models.SearchLog.results_count == 0
        ).group_by(
            models.SearchLog.query
        ).limit(limit).all()

        return [r.query for r in results]

main.py​

from fastapi import FastAPI, Depends, HTTPException, Query
from sqlalchemy.orm import Session
from typing import List, Optional
import models, schemas
from database import engine, get_db
from elasticsearch_client import ElasticsearchClient
from search_analytics import SearchAnalytics
from auth import get_current_user

models.Base.metadata.create_all(bind=engine)

app = FastAPI(title="Search Engine API")

es_client = ElasticsearchClient()

# ==================== 상품 관리 ====================

@app.post("/products", response_model=schemas.Product)
def create_product(
    product: schemas.ProductCreate,
    db: Session = Depends(get_db)
):
    """상품 생성 및 인덱싱"""
    db_product = models.Product(**product.dict())
    db.add(db_product)
    db.commit()
    db.refresh(db_product)

    # Elasticsearch 인덱싱
    product_data = {
        "id": db_product.id,
        "name": db_product.name,
        "description": db_product.description,
        "category": db_product.category,
        "brand": db_product.brand,
        "price": db_product.price,
        "stock": db_product.stock,
        "tags": db_product.tags.split(',') if db_product.tags else [],
        "search_keywords": db_product.search_keywords,
        "created_at": db_product.created_at.isoformat()
    }

    es_client.index_product(db_product.id, product_data)

    return db_product

@app.put("/products/{product_id}")
def update_product(
    product_id: int,
    product: schemas.ProductUpdate,
    db: Session = Depends(get_db)
):
    """상품 업데이트 및 재인덱싱"""
    db_product = db.query(models.Product).filter(
        models.Product.id == product_id
    ).first()

    if not db_product:
        raise HTTPException(status_code=404, detail="Product not found")

    for key, value in product.dict(exclude_unset=True).items():
        setattr(db_product, key, value)

    db.commit()
    db.refresh(db_product)

    # 재인덱싱
    product_data = {
        "id": db_product.id,
        "name": db_product.name,
        "description": db_product.description,
        "category": db_product.category,
        "brand": db_product.brand,
        "price": db_product.price,
        "stock": db_product.stock,
        "tags": db_product.tags.split(',') if db_product.tags else [],
        "search_keywords": db_product.search_keywords,
        "created_at": db_product.created_at.isoformat()
    }

    es_client.index_product(db_product.id, product_data)

    return db_product

@app.delete("/products/{product_id}")
def delete_product(product_id: int, db: Session = Depends(get_db)):
    """상품 삭제"""
    db_product = db.query(models.Product).filter(
        models.Product.id == product_id
    ).first()

    if not db_product:
        raise HTTPException(status_code=404, detail="Product not found")

    # Elasticsearch에서 삭제
    es_client.delete_product(product_id)

    # DB에서 삭제
    db.delete(db_product)
    db.commit()

    return {"message": "Product deleted"}

# ==================== 검색 ====================

@app.get("/search")
def search_products(
    q: str = Query(..., description="검색어"),
    category: Optional[str] = None,
    brand: Optional[str] = None,
    min_price: Optional[float] = None,
    max_price: Optional[float] = None,
    in_stock: Optional[bool] = None,
    sort_by: Optional[str] = Query(None, regex="^(price_asc|price_desc|newest|relevance)$"),
    page: int = Query(1, ge=1),
    size: int = Query(10, ge=1, le=100),
    user: Optional[User] = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """상품 검색"""
    # 필터 구성
    filters = {}
    if category:
        filters["category"] = category
    if brand:
        filters["brand"] = brand
    if min_price is not None:
        filters["min_price"] = min_price
    if max_price is not None:
        filters["max_price"] = max_price
    if in_stock is not None:
        filters["in_stock"] = in_stock

    # 검색 실행
    from_ = (page - 1) * size
    results = es_client.search(
        query=q,
        filters=filters,
        sort_by=sort_by,
        from_=from_,
        size=size
    )

    # 검색 로그
    analytics = SearchAnalytics(db)
    analytics.log_search(
        query=q,
        filters=filters,
        results_count=results["total"],
        user_id=user.id if user else None
    )

    return {
        "query": q,
        "total": results["total"],
        "page": page,
        "size": size,
        "results": results["results"],
        "facets": results["facets"]
    }

@app.get("/autocomplete")
def autocomplete(q: str = Query(..., min_length=1)):
    """자동완성"""
    suggestions = es_client.autocomplete(q)
    return {"suggestions": suggestions}

@app.get("/suggest")
def suggest_similar(q: str):
    """검색어 제안 (Did you mean?)"""
    suggestions = es_client.suggest_similar(q)
    return {"suggestions": suggestions}

@app.get("/products/{product_id}/similar")
def get_similar_products(product_id: int, limit: int = 5):
    """유사 상품 추천"""
    similar = es_client.more_like_this(product_id, limit)
    return {"similar_products": similar}

    limit: int = Query(10, ge=1, le=50),
    db: Session = Depends(get_db)

    """검색 결과 없는 쿼리"""

🔑 핵심 개념​

전문 검색 (Full-text Search)​

  • multi_match로 여러 필드 검색
  • 필드별 가중치 설정 (^3)
  • fuzziness로 오타 허용

패싯 검색 (Faceted Search)​

  • Aggregations로 카테고리/브랜드/가격대 집계
  • 동적 필터링

하이라이팅​

  • 검색어가 포함된 부분 강조
  • <mark> 태그로 감싸기

자동완성​

  • completion 타입 필드 사용
  • prefix 기반 빠른 제안

검색 분석​

  • 검색 로그 저장
  • 인기 검색어, 트렌딩 검색어 추출
  • 개선 필요한 쿼리 파악
