시나리오 13: 데이터 분석 대시보드 API

Pandas와 데이터 시각화를 활용한 분석 대시보드 API를 만들어봅시다!

📌 기능​

  • ✅ 판매 데이터 분석
  • ✅ 시계열 분석
  • ✅ 그룹별 통계
  • ✅ 데이터 시각화 (차트 생성)
  • ✅ CSV/Excel 내보내기
  • ✅ 실시간 대시보드 메트릭

    alt Cache hit
        Cache-->>API: Cached metrics
        API-->>U: 200 OK + Metrics
    else Cache miss
        API->>AS: calculate_metrics()
        AS->>DB: SELECT SUM, COUNT, AVG...
        DB-->>AS: Raw data
        AS->>AS: Process with Pandas
        AS-->>API: Metrics
        API->>Cache: Store in cache (TTL: 5min)
        API-->>U: 200 OK + Metrics
    end

    Note over U,DB: 2. 시계열 차트 생성
    U->>API: GET /analytics/sales/timeseries
    API->>AS: get_sales_timeseries()

    DB-->>AS: Time series data
    AS->>CG: generate_chart(data, type='line')
    CG-->>AS: Chart image/JSON
    AS-->>API: Chart data
    API-->>U: 200 OK + Chart

    Note over U,DB: 3. Excel 내보내기
    U->>API: GET /analytics/export?format=xlsx
    API->>AS: export_data()

    DB-->>AS: Sales data

    AS-->>API: Excel file bytes

from sqlalchemy import Column, Integer, String, Float, DateTime, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base

class Sale(Base):
    """판매 데이터"""
    __tablename__ = "sales"

    id = Column(Integer, primary_key=True)
    product_id = Column(Integer, ForeignKey('products.id'))
    customer_id = Column(Integer, ForeignKey('customers.id'))
    quantity = Column(Integer)
    unit_price = Column(Float)
    total_amount = Column(Float)
    sale_date = Column(DateTime(timezone=True), index=True)
    region = Column(String, index=True)
    category = Column(String, index=True)

    product = relationship("Product")
    customer = relationship("Customer")

class Product(Base):
    __tablename__ = "products"

    id = Column(Integer, primary_key=True)
    name = Column(String)
    category = Column(String, index=True)
    cost = Column(Float)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

class Customer(Base):
    __tablename__ = "customers"

    id = Column(Integer, primary_key=True)
    name = Column(String)
    email = Column(String)
    region = Column(String, index=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

import pandas as pd
from sqlalchemy.orm import Session
from sqlalchemy import func, desc, extract
from datetime import datetime, timedelta
from typing import List, Dict, Optional
import models

class AnalyticsService:
    def __init__(self, db: Session):
        self.db = db

    def get_sales_dataframe(
        self,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> pd.DataFrame:
        """판매 데이터를 Pandas DataFrame으로 변환"""
        query = self.db.query(
            models.Sale.id,
            models.Sale.product_id,
            models.Sale.customer_id,
            models.Sale.quantity,
            models.Sale.unit_price,
            models.Sale.total_amount,
            models.Sale.sale_date,
            models.Sale.region,
            models.Sale.category,
            models.Product.name.label('product_name'),
            models.Customer.name.label('customer_name')
        ).join(
            models.Product
        ).join(
            models.Customer
        )

        if start_date:
            query = query.filter(models.Sale.sale_date >= start_date)

        if end_date:
            query = query.filter(models.Sale.sale_date <= end_date)

        # DataFrame으로 변환
        df = pd.read_sql(query.statement, self.db.bind)
        df['sale_date'] = pd.to_datetime(df['sale_date'])

        return df

    def get_summary_stats(
        self,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> Dict:
        """요약 통계"""
        df = self.get_sales_dataframe(start_date, end_date)

        return {
            "total_sales": float(df['total_amount'].sum()),
            "total_orders": int(len(df)),
            "average_order_value": float(df['total_amount'].mean()),
            "total_quantity": int(df['quantity'].sum()),
            "unique_customers": int(df['customer_id'].nunique()),
            "unique_products": int(df['product_id'].nunique())
        }

    def get_sales_by_period(
        self,
        period: str = "day",  # day, week, month, year
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[Dict]:
        """기간별 매출"""
        df = self.get_sales_dataframe(start_date, end_date)

        # 기간별 그룹화
        if period == "day":
            df['period'] = df['sale_date'].dt.date
        elif period == "week":
            df['period'] = df['sale_date'].dt.to_period('W').dt.start_time
        elif period == "month":
            df['period'] = df['sale_date'].dt.to_period('M').dt.start_time
        elif period == "year":
            df['period'] = df['sale_date'].dt.to_period('Y').dt.start_time

        # 집계
        result = df.groupby('period').agg({
            'total_amount': 'sum',
            'id': 'count',
            'quantity': 'sum'
        }).reset_index()

        result.columns = ['period', 'total_sales', 'order_count', 'total_quantity']

        return result.to_dict('records')

    def get_sales_by_category(
        self,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[Dict]:
        """카테고리별 매출"""
        df = self.get_sales_dataframe(start_date, end_date)

        result = df.groupby('category').agg({
            'total_amount': 'sum',
            'id': 'count',
            'quantity': 'sum'
        }).reset_index()

        result.columns = ['category', 'total_sales', 'order_count', 'total_quantity']
        result = result.sort_values('total_sales', ascending=False)

        return result.to_dict('records')

    def get_sales_by_region(
        self,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[Dict]:
        """지역별 매출"""
        df = self.get_sales_dataframe(start_date, end_date)

        result = df.groupby('region').agg({
            'total_amount': 'sum',
            'id': 'count',
            'customer_id': 'nunique'
        }).reset_index()

        result.columns = ['region', 'total_sales', 'order_count', 'unique_customers']
        result = result.sort_values('total_sales', ascending=False)

        return result.to_dict('records')

    def get_top_products(
        self,
        limit: int = 10,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[Dict]:
        """인기 상품 TOP N"""
        df = self.get_sales_dataframe(start_date, end_date)

        result = df.groupby(['product_id', 'product_name']).agg({
            'total_amount': 'sum',
            'quantity': 'sum',
            'id': 'count'
        }).reset_index()

        result.columns = ['product_id', 'product_name', 'total_sales', 'total_quantity', 'order_count']
        result = result.sort_values('total_sales', ascending=False).head(limit)

        return result.to_dict('records')

    def get_top_customers(
        self,
        limit: int = 10,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[Dict]:
        """우수 고객 TOP N"""
        df = self.get_sales_dataframe(start_date, end_date)

        result = df.groupby(['customer_id', 'customer_name']).agg({
            'total_amount': 'sum',
            'id': 'count'
        }).reset_index()

        result.columns = ['customer_id', 'customer_name', 'total_spent', 'order_count']
        result = result.sort_values('total_spent', ascending=False).head(limit)

        return result.to_dict('records')

    def get_cohort_analysis(self, cohort_type: str = "month") -> pd.DataFrame:
        """코호트 분석 (고객 유지율)"""
        df = self.get_sales_dataframe()

        # 고객별 첫 구매 날짜
        df['first_purchase'] = df.groupby('customer_id')['sale_date'].transform('min')

        if cohort_type == "month":
            df['cohort'] = df['first_purchase'].dt.to_period('M')
            df['period'] = df['sale_date'].dt.to_period('M')
        elif cohort_type == "week":
            df['cohort'] = df['first_purchase'].dt.to_period('W')
            df['period'] = df['sale_date'].dt.to_period('W')

        # 코호트 기간 계산
        df['cohort_period'] = (df['period'] - df['cohort']).apply(lambda x: x.n)

        # 코호트별 활성 고객 수
        cohort_data = df.groupby(['cohort', 'cohort_period'])['customer_id'].nunique().reset_index()
        cohort_data.columns = ['cohort', 'period', 'customers']

        # 피벗 테이블
        cohort_pivot = cohort_data.pivot_table(
            index='cohort',
            columns='period',
            values='customers'
        )

        # 유지율 계산
        cohort_size = cohort_pivot.iloc[:, 0]
        retention = cohort_pivot.divide(cohort_size, axis=0) * 100

        return retention

    def get_revenue_forecast(self, days: int = 30) -> List[Dict]:
        """매출 예측 (단순 이동 평균)"""
        df = self.get_sales_dataframe()

        # 일별 매출
        daily_sales = df.groupby(df['sale_date'].dt.date)['total_amount'].sum().reset_index()
        daily_sales.columns = ['date', 'sales']

        # 7일 이동 평균
        daily_sales['ma_7'] = daily_sales['sales'].rolling(window=7).mean()
        daily_sales['ma_30'] = daily_sales['sales'].rolling(window=30).mean()

        # 예측 (단순하게 최근 7일 평균 사용)
        forecast_value = daily_sales['ma_7'].iloc[-1]

        # 미래 날짜 생성
        last_date = daily_sales['date'].max()
        future_dates = pd.date_range(
            start=last_date + timedelta(days=1),
            periods=days,
            freq='D'
        )

        forecast = [
            {
                "date": str(date.date()),
                "predicted_sales": float(forecast_value)
            }
            for date in future_dates
        ]

        return forecast

visualization_service.py​

import matplotlib
matplotlib.use('Agg')  # GUI 없이 사용
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
from io import BytesIO
import base64

class VisualizationService:
    def __init__(self):
        sns.set_style("whitegrid")
        plt.rcParams['figure.figsize'] = (12, 6)

    def create_line_chart(self, data: pd.DataFrame, x: str, y: str, title: str) -> str:
        """라인 차트 생성"""
        plt.figure()
        plt.plot(data[x], data[y], marker='o')
        plt.title(title)
        plt.xlabel(x)
        plt.ylabel(y)
        plt.xticks(rotation=45)
        plt.tight_layout()

        return self._fig_to_base64()

    def create_bar_chart(self, data: pd.DataFrame, x: str, y: str, title: str) -> str:
        """바 차트 생성"""
        plt.figure()
        plt.bar(data[x], data[y])
        plt.title(title)
        plt.xlabel(x)
        plt.ylabel(y)
        plt.xticks(rotation=45)
        plt.tight_layout()

        return self._fig_to_base64()

    def create_pie_chart(self, data: pd.DataFrame, labels: str, values: str, title: str) -> str:
        """파이 차트 생성"""
        plt.figure()
        plt.pie(data[values], labels=data[labels], autopct='%1.1f%%')
        plt.title(title)
        plt.tight_layout()

        return self._fig_to_base64()

    def create_heatmap(self, data: pd.DataFrame, title: str) -> str:
        """히트맵 생성 (코호트 분석용)"""
        plt.figure(figsize=(14, 8))
        sns.heatmap(data, annot=True, fmt='.1f', cmap='YlGnBu')
        plt.title(title)
        plt.tight_layout()

        return self._fig_to_base64()

    def _fig_to_base64(self) -> str:
        """Figure를 Base64 문자열로 변환"""
        buffer = BytesIO()
        plt.savefig(buffer, format='png', dpi=100)
        buffer.seek(0)
        image_base64 = base64.b64encode(buffer.read()).decode()
        plt.close()

        return f"data:image/png;base64,{image_base64}"

@app.get("/analytics/export/csv")

@app.get("/analytics/export/excel")
def export_excel(
    start_date: Optional[str] = None,
    end_date: Optional[str] = None,
    db: Session = Depends(get_db)
):
    """Excel 내보내기"""
    service = AnalyticsService(db)

    end = datetime.fromisoformat(end_date) if end_date else None

        summary = service.get_summary_stats(start, end)
        summary_df = pd.DataFrame([summary])
        summary_df.to_excel(writer, sheet_name='Summary', index=False)

    stream.seek(0)

    response = StreamingResponse(
        iter([stream.getvalue()]),
        media_type="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    )
    response.headers["Content-Disposition"] = "attachment; filename=sales_data.xlsx"

    return response
