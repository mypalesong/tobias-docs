시나리오 9: 예약 시스템

호텔/레스토랑/회의실 예약 시스템을 만들어봅시다!

📌 기능​

  • ✅ 시간대별 예약 가능 여부 확인
  • ✅ 중복 예약 방지
  • ✅ 예약 생성/수정/취소
  • ✅ 대기열 관리
  • ✅ 자동 확인 및 알림
  • ✅ 노쇼(No-show) 처리
  • ✅ 리마인더 발송

📊 시퀀스 다이어그램​

sequenceDiagram
    participant U as User
    participant API as FastAPI
    participant BS as BookingService
    participant AC as AvailabilityChecker
    participant RS as ReminderScheduler
    participant DB as Database

    Note over U,DB: 1. 가용성 확인
    U->>API: GET /resources/{id}/availability<br/>?date=2024-01-01&duration=2h
    API->>AC: check_availability()
    AC->>DB: SELECT bookings<br/>WHERE resource_id AND<br/>time overlaps
    DB-->>AC: Existing bookings
    AC->>AC: Calculate free slots
    AC-->>API: Available time slots
    API-->>U: 200 OK + Available slots

    Note over U,DB: 2. 예약 생성 (낙관적 잠금)
    U->>API: POST /bookings<br/>{resource_id, start_time, end_time}
    API->>BS: create_booking()
    BS->>DB: BEGIN TRANSACTION
    BS->>DB: SELECT bookings FOR UPDATE<br/>WHERE resource_id AND time overlaps
    DB-->>BS: Lock acquired + bookings
    alt Time slot available
        BS->>DB: INSERT booking (status=pending)
        BS->>DB: COMMIT
        DB-->>BS: Booking created
        BS->>RS: schedule_reminder()
        RS->>RS: Add to job queue
        BS-->>API: Booking confirmed
        API-->>U: 201 Created + Booking
    else Time slot taken
        BS->>DB: ROLLBACK
        BS-->>API: Slot unavailable
        API-->>U: 409 Conflict
    end

    Note over U,DB: 3. 대기열 추가
    U->>API: POST /bookings/{id}/waitlist
    API->>DB: INSERT INTO waitlist
    DB-->>API: Waitlist entry
    API-->>U: 200 OK + Position in queue

    Note over U,DB: 4. 예약 취소 & 대기열 처리
    U->>API: DELETE /bookings/{id}
    API->>BS: cancel_booking()
    BS->>DB: UPDATE booking (status=cancelled)
    BS->>DB: SELECT waitlist<br/>WHERE resource_id<br/>ORDER BY created_at
    DB-->>BS: Waitlist entries
    alt Waitlist not empty
        BS->>BS: Process first in waitlist
        BS->>DB: INSERT new booking
        BS->>DB: DELETE from waitlist
        BS->>API: Notify waitlisted user
    end
    BS-->>API: Cancelled
    API-->>U: 204 No Content

    Note over U,DB: 5. 리마인더 발송 (백그라운드)
    RS->>RS: Check scheduled reminders
    RS->>DB: SELECT bookings<br/>WHERE start_time - 24h = now()
    DB-->>RS: Bookings to remind
    loop For each booking
        RS->>RS: Send email/SMS reminder
        RS->>DB: UPDATE last_reminder_sent
    end

models.py​

from sqlalchemy import Column, Integer, String, DateTime, Boolean, ForeignKey, Enum as SQLEnum, Float, CheckConstraint
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base
import enum
from datetime import datetime, timedelta

class BookingStatus(str, enum.Enum):
    pending = "pending"
    confirmed = "confirmed"
    cancelled = "cancelled"
    completed = "completed"
    no_show = "no_show"

class ResourceType(str, enum.Enum):
    hotel_room = "hotel_room"
    restaurant_table = "restaurant_table"
    meeting_room = "meeting_room"
    parking_spot = "parking_spot"

class Resource(Base):
    """예약 가능한 자원 (호텔 방, 테이블 등)"""
    __tablename__ = "resources"

    id = Column(Integer, primary_key=True)
    name = Column(String, index=True)
    type = Column(SQLEnum(ResourceType))
    capacity = Column(Integer)  # 수용 인원
    description = Column(String, nullable=True)

    # 가격
    price_per_hour = Column(Float, nullable=True)
    price_per_day = Column(Float, nullable=True)

    # 예약 설정
    min_duration_minutes = Column(Integer, default=60)
    max_duration_minutes = Column(Integer, default=480)
    advance_booking_days = Column(Integer, default=30)  # 최대 며칠 전까지 예약 가능

    # 운영 시간
    open_time = Column(String, default="09:00")  # HH:MM
    close_time = Column(String, default="18:00")

    is_active = Column(Boolean, default=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    bookings = relationship("Booking", back_populates="resource")
    availability_rules = relationship("AvailabilityRule", back_populates="resource")

class AvailabilityRule(Base):
    """특정 자원의 사용 불가 시간"""
    __tablename__ = "availability_rules"

    id = Column(Integer, primary_key=True)
    resource_id = Column(Integer, ForeignKey('resources.id'), index=True)
    start_datetime = Column(DateTime(timezone=True), index=True)
    end_datetime = Column(DateTime(timezone=True))
    reason = Column(String)  # "maintenance", "reserved", "blocked"

    resource = relationship("Resource", back_populates="availability_rules")

class Booking(Base):
    __tablename__ = "bookings"

    id = Column(Integer, primary_key=True)
    resource_id = Column(Integer, ForeignKey('resources.id'), index=True)
    user_id = Column(Integer, ForeignKey('users.id'), index=True)

    # 예약 시간
    start_datetime = Column(DateTime(timezone=True), index=True)
    end_datetime = Column(DateTime(timezone=True), index=True)

    # 예약 정보
    guest_name = Column(String)
    guest_email = Column(String)
    guest_phone = Column(String)
    guest_count = Column(Integer)

    # 상태
    status = Column(SQLEnum(BookingStatus), default=BookingStatus.pending, index=True)

    # 가격
    total_price = Column(Float)

    # 특별 요청
    special_requests = Column(String, nullable=True)

    # 타임스탬프
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    confirmed_at = Column(DateTime(timezone=True), nullable=True)
    cancelled_at = Column(DateTime(timezone=True), nullable=True)
    cancellation_reason = Column(String, nullable=True)

    # 알림
    reminder_sent = Column(Boolean, default=False)

    # 관계
    resource = relationship("Resource", back_populates="bookings")
    user = relationship("User")

    __table_args__ = (
        CheckConstraint('end_datetime > start_datetime', name='check_valid_time_range'),
    )

class WaitList(Base):
    """대기열"""
    __tablename__ = "waitlist"

    id = Column(Integer, primary_key=True)
    resource_id = Column(Integer, ForeignKey('resources.id'), index=True)
    user_id = Column(Integer, ForeignKey('users.id'))

    desired_start = Column(DateTime(timezone=True))
    desired_end = Column(DateTime(timezone=True))
    guest_count = Column(Integer)

    notified = Column(Boolean, default=False)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    resource = relationship("Resource")
    user = relationship("User")

booking_service.py​

from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from fastapi import HTTPException
from datetime import datetime, timedelta
import models

class BookingService:
    def __init__(self, db: Session):
        self.db = db

    def check_availability(
        self,
        resource_id: int,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> bool:
        """예약 가능 여부 확인"""
        # 자원 존재 확인
        resource = self.db.query(models.Resource).filter(
            models.Resource.id == resource_id,
            models.Resource.is_active == True
        ).first()

        if not resource:
            raise HTTPException(status_code=404, detail="Resource not found")

        # 시간 검증
        if end_datetime <= start_datetime:
            raise HTTPException(status_code=400, detail="Invalid time range")

        # 최소/최대 예약 시간 검증
        duration_minutes = (end_datetime - start_datetime).total_seconds() / 60
        if duration_minutes < resource.min_duration_minutes:
            raise HTTPException(
                status_code=400,
                detail=f"Minimum booking duration is {resource.min_duration_minutes} minutes"
            )
        if duration_minutes > resource.max_duration_minutes:
            raise HTTPException(
                status_code=400,
                detail=f"Maximum booking duration is {resource.max_duration_minutes} minutes"
            )

        # 최대 사전 예약 기간 확인
        days_advance = (start_datetime.date() - datetime.now().date()).days
        if days_advance > resource.advance_booking_days:
            raise HTTPException(
                status_code=400,
                detail=f"Can only book up to {resource.advance_booking_days} days in advance"
            )

        # 운영 시간 확인
        if not self._is_within_operating_hours(resource, start_datetime, end_datetime):
            raise HTTPException(
                status_code=400,
                detail=f"Outside operating hours ({resource.open_time} - {resource.close_time})"
            )

        # 중복 예약 확인
        overlap = self.db.query(models.Booking).filter(
            models.Booking.resource_id == resource_id,
            models.Booking.status.in_([
                models.BookingStatus.pending,
                models.BookingStatus.confirmed
            ]),
            or_(
                # 새 예약의 시작이 기존 예약 시간 내에 있는 경우
                and_(
                    models.Booking.start_datetime <= start_datetime,
                    models.Booking.end_datetime > start_datetime
                ),
                # 새 예약의 종료가 기존 예약 시간 내에 있는 경우
                and_(
                    models.Booking.start_datetime < end_datetime,
                    models.Booking.end_datetime >= end_datetime
                ),
                # 새 예약이 기존 예약을 완전히 포함하는 경우
                and_(
                    models.Booking.start_datetime >= start_datetime,
                    models.Booking.end_datetime <= end_datetime
                )
            )
        ).first()

        if overlap:
            return False

        # 사용 불가 시간 확인
        blocked = self.db.query(models.AvailabilityRule).filter(
            models.AvailabilityRule.resource_id == resource_id,
            or_(
                and_(
                    models.AvailabilityRule.start_datetime <= start_datetime,
                    models.AvailabilityRule.end_datetime > start_datetime
                ),
                and_(
                    models.AvailabilityRule.start_datetime < end_datetime,
                    models.AvailabilityRule.end_datetime >= end_datetime
                )
            )
        ).first()

        if blocked:
            return False

        return True

    def _is_within_operating_hours(
        self,
        resource: models.Resource,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> bool:
        """운영 시간 내인지 확인"""
        open_hour, open_minute = map(int, resource.open_time.split(':'))
        close_hour, close_minute = map(int, resource.close_time.split(':'))

        start_time = start_datetime.time()
        end_time = end_datetime.time()

        from datetime import time
        open_time = time(open_hour, open_minute)
        close_time = time(close_hour, close_minute)

        return start_time >= open_time and end_time <= close_time

    def calculate_price(
        self,
        resource: models.Resource,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> float:
        """가격 계산"""
        duration_hours = (end_datetime - start_datetime).total_seconds() / 3600

        if resource.price_per_hour:
            return resource.price_per_hour * duration_hours
        elif resource.price_per_day:
            days = (end_datetime.date() - start_datetime.date()).days + 1
            return resource.price_per_day * days
        else:
            return 0.0

    def create_booking(
        self,
        resource_id: int,
        user_id: int,
        start_datetime: datetime,
        end_datetime: datetime,
        guest_name: str,
        guest_email: str,
        guest_phone: str,
        guest_count: int,
        special_requests: str = None
    ) -> models.Booking:
        """예약 생성"""
        # 가용성 확인
        if not self.check_availability(resource_id, start_datetime, end_datetime):
            raise HTTPException(status_code=409, detail="Time slot not available")

        resource = self.db.query(models.Resource).filter(
            models.Resource.id == resource_id
        ).first()

        # 수용 인원 확인
        if guest_count > resource.capacity:
            raise HTTPException(
                status_code=400,
                detail=f"Resource capacity is {resource.capacity}"
            )

        # 가격 계산
        total_price = self.calculate_price(resource, start_datetime, end_datetime)

        # 예약 생성
        booking = models.Booking(
            resource_id=resource_id,
            user_id=user_id,
            start_datetime=start_datetime,
            end_datetime=end_datetime,
            guest_name=guest_name,
            guest_email=guest_email,
            guest_phone=guest_phone,
            guest_count=guest_count,
            total_price=total_price,
            special_requests=special_requests,
            status=models.BookingStatus.confirmed,
            confirmed_at=datetime.utcnow()
        )

        self.db.add(booking)
        self.db.commit()
        self.db.refresh(booking)

        return booking

    def cancel_booking(
        self,
        booking_id: int,
        user_id: int,
        reason: str = None
    ) -> models.Booking:
        """예약 취소"""
        booking = self.db.query(models.Booking).filter(
            models.Booking.id == booking_id,
            models.Booking.user_id == user_id
        ).first()

        if not booking:
            raise HTTPException(status_code=404, detail="Booking not found")

        if booking.status in [models.BookingStatus.cancelled, models.BookingStatus.completed]:
            raise HTTPException(status_code=400, detail="Cannot cancel this booking")

        # 취소 시간 제한 (예: 24시간 전까지만 취소 가능)
        hours_until = (booking.start_datetime - datetime.now()).total_seconds() / 3600
        if hours_until < 24:
            raise HTTPException(
                status_code=400,
                detail="Cannot cancel within 24 hours of booking start time"
            )

        booking.status = models.BookingStatus.cancelled
        booking.cancelled_at = datetime.utcnow()
        booking.cancellation_reason = reason

        self.db.commit()

        # 대기열에 있는 사람들에게 알림
        self._notify_waitlist(booking)

        return booking

    def _notify_waitlist(self, cancelled_booking: models.Booking):
        """대기열에 있는 사람들에게 알림"""
        waitlist_entries = self.db.query(models.WaitList).filter(
            models.WaitList.resource_id == cancelled_booking.resource_id,
            models.WaitList.notified == False,
            models.WaitList.desired_start <= cancelled_booking.start_datetime,
            models.WaitList.desired_end >= cancelled_booking.end_datetime
        ).all()

        for entry in waitlist_entries:
            # 실제로는 이메일/SMS 발송
            print(f"Notifying user {entry.user_id} about availability")
            entry.notified = True

        self.db.commit()

# ==================== 가용성 확인 ====================

@app.get("/resources/{resource_id}/availability")
def check_availability(
    resource_id: int,
    date: str,  # YYYY-MM-DD
    db: Session = Depends(get_db)
):
    """특정 날짜의 예약 가능 시간대 조회"""
    service = BookingService(db)

    target_date = datetime.strptime(date, "%Y-%m-%d").date()
    resource = db.query(models.Resource).filter(
        models.Resource.id == resource_id
    ).first()

    if not resource:
        raise HTTPException(status_code=404, detail="Resource not found")

    # 운영 시간 파싱
    open_hour, open_minute = map(int, resource.open_time.split(':'))
    close_hour, close_minute = map(int, resource.close_time.split(':'))

    # 30분 단위로 가용성 확인
    available_slots = []
    current_time = datetime.combine(target_date, datetime.min.time()).replace(
        hour=open_hour, minute=open_minute
    )
    end_of_day = datetime.combine(target_date, datetime.min.time()).replace(
        hour=close_hour, minute=close_minute
    )

    while current_time < end_of_day:
        slot_end = current_time + timedelta(minutes=30)

        try:
            is_available = service.check_availability(
                resource_id,
                current_time,
                slot_end
            )

            if is_available:
                available_slots.append({
                    "start": current_time.isoformat(),
                    "end": slot_end.isoformat()
                })
        except:
            pass

        current_time = slot_end

    return {"date": date, "available_slots": available_slots}

@app.get("/bookings/my", response_model=List[schemas.Booking])
def get_my_bookings(

            models.Booking.start_datetime >= datetime.now()

@app.put("/bookings/{booking_id}/cancel")

@app.post("/bookings/{booking_id}/no-show")
def mark_no_show(
    booking_id: int,
    db: Session = Depends(get_db)
):
    """노쇼 처리 (관리자용)"""
    booking = db.query(models.Booking).filter(
        models.Booking.id == booking_id
    ).first()

    if not booking:
        raise HTTPException(status_code=404, detail="Booking not found")

    # 예약 시작 시간이 지났는지 확인
    if booking.start_datetime > datetime.now():
        raise HTTPException(status_code=400, detail="Cannot mark as no-show yet")

    booking.status = models.BookingStatus.no_show
    db.commit()

    return {"message": "Marked as no-show"}

# ==================== 대기열 ====================

@app.post("/waitlist")
def join_waitlist(
    waitlist: schemas.WaitListCreate,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """대기열 등록"""
    entry = models.WaitList(
        resource_id=waitlist.resource_id,
        user_id=current_user.id,
        desired_start=waitlist.desired_start,
        desired_end=waitlist.desired_end,
        guest_count=waitlist.guest_count
    )

    db.add(entry)
    db.commit()

    return {"message": "Added to waitlist", "position": get_waitlist_position(entry, db)}

def get_waitlist_position(entry: models.WaitList, db: Session) -> int:
    """대기열 순번 조회"""
    count = db.query(models.WaitList).filter(
        models.WaitList.resource_id == entry.resource_id,
        models.WaitList.created_at < entry.created_at
    ).count()

    return count + 1

def send_reminder():
    """리마인더 발송 (크론잡으로 실행)"""
    from database import SessionLocal
    db = SessionLocal()

    # 24시간 이내 예약
    tomorrow = datetime.now() + timedelta(hours=24)
    bookings = db.query(models.Booking).filter(
        models.Booking.start_datetime <= tomorrow,
        models.Booking.start_datetime >= datetime.now(),
        models.Booking.status == models.BookingStatus.confirmed,
        models.Booking.reminder_sent == False
    ).all()

    for booking in bookings:
        # 이메일/SMS 발송
        print(f"Sending reminder to {booking.guest_email}")
        booking.reminder_sent = True

    db.commit()
    db.close()
