from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from fastapi import HTTPException
from datetime import datetime, timedelta
import models

class BookingService:
    def __init__(self, db: Session):
        self.db = db

    def check_availability(
        self,
        resource_id: int,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> bool:
        """ì˜ˆì•½ ê°€ëŠ¥ ì—¬ë¶€ í™•ì¸"""
        # ìì› ì¡´ì¬ í™•ì¸
        resource = self.db.query(models.Resource).filter(
            models.Resource.id == resource_id,
            models.Resource.is_active == True
        ).first()

        if not resource:
            raise HTTPException(status_code=404, detail="Resource not found")

        # ì‹œê°„ ê²€ì¦
        if end_datetime <= start_datetime:
            raise HTTPException(status_code=400, detail="Invalid time range")

        # ìµœì†Œ/ìµœëŒ€ ì˜ˆì•½ ì‹œê°„ ê²€ì¦
        duration_minutes = (end_datetime - start_datetime).total_seconds() / 60
        if duration_minutes < resource.min_duration_minutes:
            raise HTTPException(
                status_code=400,
                detail=f"Minimum booking duration is {resource.min_duration_minutes} minutes"
            )
        if duration_minutes > resource.max_duration_minutes:
            raise HTTPException(
                status_code=400,
                detail=f"Maximum booking duration is {resource.max_duration_minutes} minutes"
            )

        # ìµœëŒ€ ì‚¬ì „ ì˜ˆì•½ ê¸°ê°„ í™•ì¸
        days_advance = (start_datetime.date() - datetime.now().date()).days
        if days_advance > resource.advance_booking_days:
            raise HTTPException(
                status_code=400,
                detail=f"Can only book up to {resource.advance_booking_days} days in advance"
            )

        # ìš´ì˜ ì‹œê°„ í™•ì¸
        if not self._is_within_operating_hours(resource, start_datetime, end_datetime):
            raise HTTPException(
                status_code=400,
                detail=f"Outside operating hours ({resource.open_time} - {resource.close_time})"
            )

        # ì¤‘ë³µ ì˜ˆì•½ í™•ì¸
        overlap = self.db.query(models.Booking).filter(
            models.Booking.resource_id == resource_id,
            models.Booking.status.in_([
                models.BookingStatus.pending,
                models.BookingStatus.confirmed
            ]),
            or_(
                # ìƒˆ ì˜ˆì•½ì˜ ì‹œì‘ì´ ê¸°ì¡´ ì˜ˆì•½ ì‹œê°„ ë‚´ì— ìˆëŠ” ê²½ìš°
                and_(
                    models.Booking.start_datetime <= start_datetime,
                    models.Booking.end_datetime > start_datetime
                ),
                # ìƒˆ ì˜ˆì•½ì˜ ì¢…ë£Œê°€ ê¸°ì¡´ ì˜ˆì•½ ì‹œê°„ ë‚´ì— ìˆëŠ” ê²½ìš°
                and_(
                    models.Booking.start_datetime < end_datetime,
                    models.Booking.end_datetime >= end_datetime
                ),
                # ìƒˆ ì˜ˆì•½ì´ ê¸°ì¡´ ì˜ˆì•½ì„ ì™„ì „íˆ í¬í•¨í•˜ëŠ” ê²½ìš°
                and_(
                    models.Booking.start_datetime >= start_datetime,
                    models.Booking.end_datetime <= end_datetime
                )
            )
        ).first()

        if overlap:
            return False

        # ì‚¬ìš© ë¶ˆê°€ ì‹œê°„ í™•ì¸
        blocked = self.db.query(models.AvailabilityRule).filter(
            models.AvailabilityRule.resource_id == resource_id,
            or_(
                and_(
                    models.AvailabilityRule.start_datetime <= start_datetime,
                    models.AvailabilityRule.end_datetime > start_datetime
                ),
                and_(
                    models.AvailabilityRule.start_datetime < end_datetime,
                    models.AvailabilityRule.end_datetime >= end_datetime
                )
            )
        ).first()

        if blocked:
            return False

        return True

    def _is_within_operating_hours(
        self,
        resource: models.Resource,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> bool:
        """ìš´ì˜ ì‹œê°„ ë‚´ì¸ì§€ í™•ì¸"""
        open_hour, open_minute = map(int, resource.open_time.split(':'))
        close_hour, close_minute = map(int, resource.close_time.split(':'))

        start_time = start_datetime.time()
        end_time = end_datetime.time()

        from datetime import time
        open_time = time(open_hour, open_minute)
        close_time = time(close_hour, close_minute)

        return start_time >= open_time and end_time <= close_time

    def calculate_price(
        self,
        resource: models.Resource,
        start_datetime: datetime,
        end_datetime: datetime
    ) -> float:
        """ê°€ê²© ê³„ì‚°"""
        duration_hours = (end_datetime - start_datetime).total_seconds() / 3600

        if resource.price_per_hour:
            return resource.price_per_hour * duration_hours
        elif resource.price_per_day:
            days = (end_datetime.date() - start_datetime.date()).days + 1
            return resource.price_per_day * days
        else:
            return 0.0

    def create_booking(
        self,
        resource_id: int,
        user_id: int,
        start_datetime: datetime,
        end_datetime: datetime,
        guest_name: str,
        guest_email: str,
        guest_phone: str,
        guest_count: int,
        special_requests: str = None
    ) -> models.Booking:
        """ì˜ˆì•½ ìƒì„±"""
        # ê°€ìš©ì„± í™•ì¸
        if not self.check_availability(resource_id, start_datetime, end_datetime):
            raise HTTPException(status_code=409, detail="Time slot not available")

        resource = self.db.query(models.Resource).filter(
            models.Resource.id == resource_id
        ).first()

        # ìˆ˜ìš© ì¸ì› í™•ì¸
        if guest_count > resource.capacity:
            raise HTTPException(
                status_code=400,
                detail=f"Resource capacity is {resource.capacity}"
            )

        # ê°€ê²© ê³„ì‚°
        total_price = self.calculate_price(resource, start_datetime, end_datetime)

        # ì˜ˆì•½ ìƒì„±
        booking = models.Booking(
            resource_id=resource_id,
            user_id=user_id,
            start_datetime=start_datetime,
            end_datetime=end_datetime,
            guest_name=guest_name,
            guest_email=guest_email,
            guest_phone=guest_phone,
            guest_count=guest_count,
            total_price=total_price,
            special_requests=special_requests,
            status=models.BookingStatus.confirmed,
            confirmed_at=datetime.utcnow()
        )

        self.db.add(booking)
        self.db.commit()
        self.db.refresh(booking)

        return booking

    def cancel_booking(
        self,
        booking_id: int,
        user_id: int,
        reason: str = None
    ) -> models.Booking:
        """ì˜ˆì•½ ì·¨ì†Œ"""
        booking = self.db.query(models.Booking).filter(
            models.Booking.id == booking_id,
            models.Booking.user_id == user_id
        ).first()

        if not booking:
            raise HTTPException(status_code=404, detail="Booking not found")

        if booking.status in [models.BookingStatus.cancelled, models.BookingStatus.completed]:
            raise HTTPException(status_code=400, detail="Cannot cancel this booking")

        # ì·¨ì†Œ ì‹œê°„ ì œí•œ (ì˜ˆ: 24ì‹œê°„ ì „ê¹Œì§€ë§Œ ì·¨ì†Œ ê°€ëŠ¥)
        hours_until = (booking.start_datetime - datetime.now()).total_seconds() / 3600
        if hours_until < 24:
            raise HTTPException(
                status_code=400,
                detail="Cannot cancel within 24 hours of booking start time"
            )

        booking.status = models.BookingStatus.cancelled
        booking.cancelled_at = datetime.utcnow()
        booking.cancellation_reason = reason

        self.db.commit()

        # ëŒ€ê¸°ì—´ì— ìˆëŠ” ì‚¬ëŒë“¤ì—ê²Œ ì•Œë¦¼
        self._notify_waitlist(booking)

        return booking

    def _notify_waitlist(self, cancelled_booking: models.Booking):
        """ëŒ€ê¸°ì—´ì— ìˆëŠ” ì‚¬ëŒë“¤ì—ê²Œ ì•Œë¦¼"""

            models.WaitList.resource_id == cancelled_booking.resource_id,
            models.WaitList.notified == False,
            models.WaitList.desired_start <= cancelled_booking.start_datetime,
            models.WaitList.desired_end >= cancelled_booking.end_datetime
        ).all()

        for entry in waitlist_entries:
            # ì‹¤ì œë¡œëŠ” ì´ë©”ì¼/SMS ë°œì†¡
            print(f"Notifying user {entry.user_id} about availability")
            entry.notified = True

        self.db.commit()

@app.get("/resources/{resource_id}/availability")
def check_availability(
    resource_id: int,
    date: str,  # YYYY-MM-DD
    db: Session = Depends(get_db)
):
    """íŠ¹ì • ë‚ ì§œì˜ ì˜ˆì•½ ê°€ëŠ¥ ì‹œê°„ëŒ€ ì¡°íšŒ"""
    service = BookingService(db)

    target_date = datetime.strptime(date, "%Y-%m-%d").date()
    resource = db.query(models.Resource).filter(
        models.Resource.id == resource_id
    ).first()

    if not resource:
        raise HTTPException(status_code=404, detail="Resource not found")

    # ìš´ì˜ ì‹œê°„ íŒŒì‹±
    open_hour, open_minute = map(int, resource.open_time.split(':'))
    close_hour, close_minute = map(int, resource.close_time.split(':'))

    # 30ë¶„ ë‹¨ìœ„ë¡œ ê°€ìš©ì„± í™•ì¸
    available_slots = []
    current_time = datetime.combine(target_date, datetime.min.time()).replace(
        hour=open_hour, minute=open_minute
    )
    end_of_day = datetime.combine(target_date, datetime.min.time()).replace(
        hour=close_hour, minute=close_minute
    )

    while current_time < end_of_day:
        slot_end = current_time + timedelta(minutes=30)

        try:
            is_available = service.check_availability(
                resource_id,
                current_time,
                slot_end
            )

            if is_available:
                available_slots.append({
                    "start": current_time.isoformat(),
                    "end": slot_end.isoformat()
                })
        except:
            pass

        current_time = slot_end

    return {"date": date, "available_slots": available_slots}

def get_waitlist_position(entry: models.WaitList, db: Session) -> int:
