HTTP 상태 코드

적절한 HTTP 상태 코드를 반환하는 방법을 배워봅시다!

📝 기본 사용법​

from fastapi import FastAPI, HTTPException, status

app = FastAPI()

@app.post("/items", status_code=status.HTTP_201_CREATED)
def create_item(name: str):
    return {"name": name}

🎨 상태 코드 목록​

from fastapi import status

# 성공
status.HTTP_200_OK                  # GET 성공
status.HTTP_201_CREATED             # POST 성공
status.HTTP_204_NO_CONTENT          # DELETE 성공

# 클라이언트 에러
status.HTTP_400_BAD_REQUEST         # 잘못된 요청
status.HTTP_401_UNAUTHORIZED        # 인증 필요
status.HTTP_403_FORBIDDEN           # 권한 없음
status.HTTP_404_NOT_FOUND           # 리소스 없음
status.HTTP_422_UNPROCESSABLE_ENTITY  # 검증 실패

# 서버 에러
status.HTTP_500_INTERNAL_SERVER_ERROR  # 서버 에러
