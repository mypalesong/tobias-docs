경로 매개변수 (Path Parameters)

URL 경로에서 동적 값을 받는 방법을 배워봅시다!

🎯 학습 목표​

  • 경로 매개변수 사용법
  • 타입 검증 및 변환
  • 경로 순서 이해
  • Path 클래스 활용

📝 기본 사용법​

간단한 경로 매개변수​

from fastapi import FastAPI

app = FastAPI()

@app.get("/items/{item_id}")
def read_item(item_id: int):
    """아이템 조회"""
    return {"item_id": item_id}

@app.get("/users/{user_id}")
def read_user(user_id: int):
    """사용자 조회"""
    return {"user_id": user_id}

테스트:

curl http://127.0.0.1:8000/items/42
# {"item_id": 42}

curl http://127.0.0.1:8000/items/abc
# {"detail":[{"type":"int_parsing","loc":["path","item_id"],...}]}

🔢 타입 검증​

자동 타입 변환 및 검증​

from enum import Enum
from fastapi import FastAPI

app = FastAPI()

# 정수
@app.get("/items/{item_id}")
def get_item(item_id: int):
    return {"item_id": item_id, "type": "integer"}

# 문자열
@app.get("/users/{username}")
def get_user(username: str):
    return {"username": username, "type": "string"}

# 실수
@app.get("/prices/{price}")
def get_price(price: float):
    return {"price": price, "type": "float"}

# 불린
@app.get("/flags/{flag}")
def get_flag(flag: bool):
    return {"flag": flag, "type": "boolean"}

Enum으로 제한​

from enum import Enum
from fastapi import FastAPI

class ModelName(str, Enum):
    """모델 이름 (고정 값)"""
    alexnet = "alexnet"
    resnet = "resnet"
    lenet = "lenet"

app = FastAPI()

@app.get("/models/{model_name}")
def get_model(model_name: ModelName):
    """AI 모델 조회"""
    if model_name == ModelName.alexnet:
        return {"model_name": model_name, "message": "Deep Learning FTW!"}

    if model_name.value == "lenet":
        return {"model_name": model_name, "message": "LeCNN all the images"}

    return {"model_name": model_name, "message": "Have some residuals"}

테스트:

curl http://127.0.0.1:8000/models/alexnet
# {"model_name":"alexnet","message":"Deep Learning FTW!"}

curl http://127.0.0.1:8000/models/invalid
# {"detail":"Input should be 'alexnet', 'resnet' or 'lenet'"}

🛤️ 경로 순서​

순서가 중요함!​

from fastapi import FastAPI

app = FastAPI()

# ✅ 올바른 순서
@app.get("/users/me")
def read_user_me():
    """현재 사용자 (특정 경로가 먼저)"""
    return {"user_id": "current_user"}

@app.get("/users/{user_id}")
def read_user(user_id: str):
    """특정 사용자 (일반 경로가 나중)"""
    return {"user_id": user_id}

# ❌ 잘못된 순서 (주의!)
# @app.get("/users/{user_id}")  # 이게 먼저 오면
# @app.get("/users/me")          # 여기는 도달 불가!

🎨 Path 클래스로 검증​

기본 검증​

from fastapi import FastAPI, Path

app = FastAPI()

@app.get("/items/{item_id}")
def read_item(
    item_id: int = Path(
        title="Item ID",
        description="조회할 아이템의 ID",
        ge=1,  # Greater than or Equal (>= 1)
        le=1000  # Less than or Equal (<= 1000)
    )
):
    return {"item_id": item_id}

테스트:

curl http://127.0.0.1:8000/items/500
# {"item_id": 500}

curl http://127.0.0.1:8000/items/0
# {"detail":[{"loc":["path","item_id"],"msg":"Input should be greater than or equal to 1"}]}

curl http://127.0.0.1:8000/items/2000
# {"detail":[{"loc":["path","item_id"],"msg":"Input should be less than or equal to 1000"}]}

문자열 검증​

from fastapi import FastAPI, Path

app = FastAPI()

@app.get("/users/{username}")
def read_user(
    username: str = Path(
        min_length=3,
        max_length=50,
        pattern="^[a-zA-Z0-9_]+$"  # 영문, 숫자, 언더스코어만
    )
):
    return {"username": username}

💡 실전 예제​

블로그 API​

from fastapi import FastAPI, Path, HTTPException
from typing import Dict

app = FastAPI()

# 가상 데이터베이스
posts: Dict[int, dict] = {
    1: {"title": "First Post", "content": "Hello World"},
    2: {"title": "Second Post", "content": "FastAPI is awesome"},
}

@app.get("/posts/{post_id}")
def get_post(
    post_id: int = Path(
        title="Post ID",
        description="조회할 포스트의 ID",
        ge=1
    )
):
    """포스트 조회"""
    if post_id not in posts:
        raise HTTPException(status_code=404, detail="Post not found")

    return posts[post_id]

@app.delete("/posts/{post_id}")
def delete_post(post_id: int = Path(ge=1)):
    """포스트 삭제"""
    if post_id not in posts:
        raise HTTPException(status_code=404, detail="Post not found")

    deleted_post = posts.pop(post_id)
    return {"message": "Post deleted", "post": deleted_post}

파일 경로 처리​

from fastapi import FastAPI

app = FastAPI()

@app.get("/files/{file_path:path}")
def read_file(file_path: str):
    """
    파일 경로 처리 (:path로 슬래시 포함 가능)
    """
    return {"file_path": file_path}

테스트:

curl http://127.0.0.1:8000/files/home/user/documents/file.txt
# {"file_path": "home/user/documents/file.txt"}

🧪 pytest로 테스트​

# test_path_parameters.py
from fastapi.testclient import TestClient
from main import app

client = TestClient(app)

def test_read_item():
    """아이템 조회 테스트"""
    response = client.get("/items/42")
    assert response.status_code == 200
    assert response.json() == {"item_id": 42}

def test_invalid_item_id():
    """잘못된 아이템 ID 테스트"""
    response = client.get("/items/abc")
    assert response.status_code == 422  # Validation Error

def test_enum_model():
    """Enum 모델 테스트"""
    response = client.get("/models/alexnet")
    assert response.status_code == 200
    assert response.json()["model_name"] == "alexnet"

    # 잘못된 값
    response = client.get("/models/invalid")
    assert response.status_code == 422

📊 Path vs Query Parameters​

특징Path ParameterQuery Parameter
위치/items/{id}/items?id=5
필수 여부필수선택적
용도리소스 식별필터링, 정렬
예시사용자 ID검색어, 페이지
from fastapi import FastAPI

app = FastAPI()

# Path Parameter (필수)
@app.get("/users/{user_id}")
def get_user(user_id: int):
    return {"user_id": user_id}

# Query Parameter (선택적)
@app.get("/items")
def list_items(skip: int = 0, limit: int = 10):
    return {"skip": skip, "limit": limit}

# 혼합
@app.get("/users/{user_id}/items")
def get_user_items(
    user_id: int,           # Path (필수)
    skip: int = 0,          # Query (선택)
    limit: int = 10         # Query (선택)
):
    return {
        "user_id": user_id,
        "skip": skip,
        "limit": limit
    }

📚 핵심 정리​

  • {param}: 경로 매개변수 정의
  • 타입 힌트: 자동 검증 및 변환
  • Enum: 값을 고정 옵션으로 제한
  • Path: 추가 검증 및 메타데이터
  • 순서: 구체적인 경로를 먼저 정의
