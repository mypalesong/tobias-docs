기초를 넘어 더 강력한 Jinja2 기능을 배워봅시다!

  • 템플릿 상속으로 재사용 가능한 레이아웃 만들기

  • 컨텍스트와 전역 변수 관리하기

    </header>

    </nav>

    <main>
        {% block content %}{% endblock %}
    </main>

    <footer>
        {% include 'footer.html' %}
    </footer>

    {% block scripts %}
    <script src="/static/main.js"></script>
    {% endblock %}
</body>
</html>
"""

# home.html - 홈페이지
home_template = """
{% extends "base.html" %}

{% block title %}홈 - {{ super() }}{% endblock %}

{% block content %}
    <h1>환영합니다!</h1>

    {% from 'components.html' import card %}

    <div class="grid">
        {% for product in products %}
            {{ card(product.name, product.price, product.image) }}
        {% endfor %}
    </div>
{% endblock %}
"""

# components.html - 재사용 가능한 컴포넌트
components = """
{% macro card(title, price, image) %}
    <div class="card">
        <img src="{{ image }}" alt="{{ title }}">
        <h3>{{ title }}</h3>
        <p class="price">{{ price|currency }}</p>
        <button>구매하기</button>
    </div>
{% endmacro %}

{% macro alert(message, type='info') %}
    <div class="alert alert-{{ type }}">
        {{ message }}
    </div>
{% endmacro %}

{% macro pagination(page, total_pages) %}
    <nav class="pagination">
        {% if page > 1 %}
            <a href="?page={{ page - 1 }}">이전</a>
        {% endif %}

        {% for p in range(1, total_pages + 1) %}
            <a href="?page={{ p }}"
               class="{{ 'active' if p == page }}">
                {{ p }}
            </a>
        {% endfor %}

        {% if page < total_pages %}
            <a href="?page={{ page + 1 }}">다음</a>
        {% endif %}
    </nav>
{% endmacro %}
"""

커스텀 필터와 테스트​

from jinja2 import Environment

env = Environment()

# 커스텀 필터
def format_currency(value):
    """통화 포맷팅"""
    return f"₩{value:,.0f}"

def slugify(text):
    """URL 친화적 문자열로 변환"""
    import re
    text = text.lower()
    text = re.sub(r'[^\w\s-]', '', text)
    text = re.sub(r'[-\s]+', '-', text)
    return text.strip('-')

def timeago(date):
    """상대 시간 표시"""
    from datetime import datetime
    now = datetime.now()
    diff = now - date

    if diff.days > 365:
        return f"{diff.days // 365}년 전"
    elif diff.days > 30:
        return f"{diff.days // 30}개월 전"
    elif diff.days > 0:
        return f"{diff.days}일 전"
    elif diff.seconds > 3600:
        return f"{diff.seconds // 3600}시간 전"
    elif diff.seconds > 60:
        return f"{diff.seconds // 60}분 전"
    else:
        return "방금 전"

# 필터 등록
env.filters['currency'] = format_currency
env.filters['slugify'] = slugify
env.filters['timeago'] = timeago

# 커스텀 테스트
def is_admin(user):
    return user.get('role') == 'admin'

def is_even(value):
    return value % 2 == 0

# 테스트 등록
env.tests['admin'] = is_admin
env.tests['even'] = is_even

# 사용 예제
template = env.from_string("""
<article>
    <h2>{{ post.title }}</h2>
    <p class="meta">{{ post.created_at|timeago }}</p>
    <a href="/posts/{{ post.title|slugify }}">읽기</a>

    {% if post.author is admin %}
        <span class="badge">관리자</span>
    {% endif %}

    {% if post.views is even %}
        <p>조회수가 짝수입니다: {{ post.views }}</p>
    {% endif %}
</article>

<p>가격: {{ product.price|currency }}</p>
""")

다국어 지원​

from jinja2 import Environment
from babel.support import Translations

# 번역 파일 로드
translations = Translations.load('locales', ['ko'])

env = Environment(extensions=['jinja2.ext.i18n'])
env.install_gettext_translations(translations)

template = env.from_string("""
<h1>{{ _('Welcome') }}</h1>
<p>{{ _('You have %(count)d messages', count=5) }}</p>
<p>{{ ngettext('%(num)d item', '%(num)d items', count) }}</p>
""")
