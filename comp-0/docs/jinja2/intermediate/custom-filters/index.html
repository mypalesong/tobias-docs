커스텀 필터

자신만의 필터를 만들어봅시다!

🎯 커스텀 필터란?​

Jinja2 내장 필터 외에 프로젝트에 필요한 커스텀 필터를 직접 만들 수 있습니다.

핵심 개념​

  • 필터 함수: Python 함수로 구현
  • 등록: Environment에 필터 등록
  • 사용: 템플릿에서 {{ value|filter_name }}로 사용

📐 기본 구조​

간단한 필터 만들기​

from jinja2 import Environment

def reverse_string(s):
    """문자열을 뒤집는 필터"""
    return s[::-1]

env = Environment()
env.filters['reverse'] = reverse_string

template = env.from_string("{{ text|reverse }}")
print(template.render(text="Hello"))
# 출력: olleH

인자가 있는 필터​

def repeat(s, times=2):
    """문자열을 n번 반복하는 필터"""
    return s * times

env.filters['repeat'] = repeat

template = env.from_string("{{ text|repeat(3) }}")
print(template.render(text="Hi!"))
# 출력: Hi!Hi!Hi!

날짜/시간 필터​

from datetime import datetime, timedelta

def datetimeformat(value, format='%Y-%m-%d %H:%M:%S'):
    """datetime 객체를 포맷팅"""
    if isinstance(value, datetime):
        return value.strftime(format)
    return value

def timeago(value):
    """상대 시간 표시 (예: 3시간 전)"""
    if not isinstance(value, datetime):
        return value

    now = datetime.now()
    diff = now - value

    if diff.days > 365:
        return f"{diff.days // 365}년 전"
    elif diff.days > 30:
        return f"{diff.days // 30}개월 전"
    elif diff.days > 0:
        return f"{diff.days}일 전"
    elif diff.seconds > 3600:
        return f"{diff.seconds // 3600}시간 전"
    elif diff.seconds > 60:
        return f"{diff.seconds // 60}분 전"
    else:
        return "방금 전"

# 등록
env.filters['datetimeformat'] = datetimeformat
env.filters['timeago'] = timeago

# 사용
template = env.from_string("""
작성일: {{ created_at|datetimeformat('%Y년 %m월 %d일') }}
{{ created_at|timeago }}
""")

print(template.render(created_at=datetime.now() - timedelta(hours=2)))
# 출력:
# 작성일: 2024년 01월 15일
# 2시간 전

통화 포맷 필터​

def currency(value, symbol='₩'):
    """통화 형식으로 포맷팅"""
    try:
        return f"{symbol}{value:,.0f}"
    except (ValueError, TypeError):
        return value

def currency_usd(value):
    """달러 형식"""
    return currency(value, symbol='$')

def currency_eur(value):
    """유로 형식"""
    return currency(value, symbol='€')

env.filters['currency'] = currency
env.filters['currency_usd'] = currency_usd
env.filters['currency_eur'] = currency_eur

template = env.from_string("""
원화: {{ price|currency }}
달러: {{ price|currency_usd }}
유로: {{ price|currency_eur }}
커스텀: {{ price|currency('¥') }}
""")

print(template.render(price=1500000))
# 출력:
# 원화: ₩1,500,000
# 달러: $1,500,000
# 유로: €1,500,000
# 커스텀: ¥1,500,000

텍스트 처리 필터​

import re
from html import escape

def slugify(text):
    """URL 친화적 문자열로 변환"""
    # 소문자로 변환
    text = text.lower()
    # 특수문자 제거
    text = re.sub(r'[^\w\s-]', '', text)
    # 공백을 하이픈으로
    text = re.sub(r'[-\s]+', '-', text)
    return text.strip('-')

def markdown(text):
    """간단한 마크다운 변환 (실제로는 markdown 라이브러리 사용 권장)"""
    # 볼드
    text = re.sub(r'\*\*(.*?)\*\*', r'<strong>\1</strong>', text)
    # 이탤릭
    text = re.sub(r'\*(.*?)\*', r'<em>\1</em>', text)
    # 링크
    text = re.sub(r'\[(.*?)\]\((.*?)\)', r'<a href="\2">\1</a>', text)
    return text

def nl2br(text):
    """줄바꿈을 <br> 태그로 변환"""
    return text.replace('\n', '<br>\n')

def excerpt(text, length=100, suffix='...'):
    """텍스트 발췌 (HTML 태그 제거)"""
    # HTML 태그 제거
    text = re.sub(r'<[^>]+>', '', text)
    if len(text) <= length:
        return text
    return text[:length].rsplit(' ', 1)[0] + suffix

env.filters['slugify'] = slugify
env.filters['markdown'] = markdown
env.filters['nl2br'] = nl2br
env.filters['excerpt'] = excerpt

template = env.from_string("""
슬러그: {{ title|slugify }}
마크다운: {{ content|markdown|safe }}
줄바꿈: {{ text|nl2br|safe }}
발췌: {{ article|excerpt(50) }}
""")

print(template.render(
    title="Hello World! 안녕하세요",
    content="**굵게** *기울임* [링크](http://example.com)",
    text="첫 줄\n두 번째 줄\n세 번째 줄",
    article="<p>이것은 매우 긴 텍스트입니다. " * 10 + "</p>"
))

파일 크기 필터​

def filesizeformat(bytes_):
    """바이트를 읽기 쉬운 형식으로 변환"""
    try:
        bytes_ = float(bytes_)
    except (TypeError, ValueError):
        return bytes_

    if bytes_ < 1024:
        return f"{bytes_:.0f} bytes"
    elif bytes_ < 1024 ** 2:
        return f"{bytes_ / 1024:.1f} KB"
    elif bytes_ < 1024 ** 3:
        return f"{bytes_ / (1024 ** 2):.1f} MB"
    elif bytes_ < 1024 ** 4:
        return f"{bytes_ / (1024 ** 3):.1f} GB"
    else:
        return f"{bytes_ / (1024 ** 4):.1f} TB"

env.filters['filesizeformat'] = filesizeformat

template = env.from_string("{{ size|filesizeformat }}")
print(template.render(size=1536))  # 1.5 KB
print(template.render(size=1048576))  # 1.0 MB

환경 변수 접근​

def url_for(endpoint, **kwargs):
    """URL 생성 (Flask 스타일)"""
    # 실제로는 라우팅 정보 필요
    base_url = kwargs.get('_external', False) and 'https://example.com' or ''
    query = '&'.join(f"{k}={v}" for k, v in kwargs.items() if not k.startswith('_'))
    url = f"{base_url}/{endpoint}"
    if query:
        url += f"?{query}"
    return url

def context_filter(context, value):
    """컨텍스트 접근이 필요한 필터"""
    # context에서 변수 접근 가능
    user = context.get('user')
    if user and user.get('is_admin'):
        return f"[관리자] {value}"
    return value

from jinja2 import pass_context

@pass_context
def smart_date(context, value):
    """현재 사용자 locale에 맞는 날짜 형식"""
    locale = context.get('user_locale', 'ko_KR')

    if locale == 'ko_KR':
        return value.strftime('%Y년 %m월 %d일')
    elif locale == 'en_US':
        return value.strftime('%B %d, %Y')
    else:
        return value.strftime('%Y-%m-%d')

env.filters['url_for'] = url_for
env.filters['context_filter'] = context_filter
env.filters['smart_date'] = smart_date

Environment 접근​

from jinja2 import pass_environment

@pass_environment
def env_filter(environment, value):
    """Environment 설정에 따라 동작"""
    if environment.autoescape:
        from markupsafe import escape
        return escape(value)
    return value

env.filters['env_filter'] = env_filter

평가 컨텍스트 접근​

from jinja2 import pass_eval_context
from markupsafe import Markup, escape

@pass_eval_context
def nl2br_auto(eval_ctx, value):
    """자동 이스케이핑을 고려한 nl2br"""
    result = value.replace('\n', '<br>\n')

    if eval_ctx.autoescape:
        result = escape(result)

    return Markup(result)

env.filters['nl2br_auto'] = nl2br_auto

체이닝 가능한 필터 만들기​

def add_prefix(text, prefix='>>'):
    """접두사 추가"""
    return f"{prefix} {text}"

def add_suffix(text, suffix='<<'):
    """접미사 추가"""
    return f"{text} {suffix}"

def wrap(text, tag='div'):
    """HTML 태그로 감싸기"""
    return f"<{tag}>{text}</{tag}>"

env.filters['add_prefix'] = add_prefix
env.filters['add_suffix'] = add_suffix
env.filters['wrap'] = wrap

template = env.from_string("""
{{ text|add_prefix('★')|add_suffix('★')|wrap('p')|safe }}
""")

print(template.render(text="중요"))
# 출력: <p>★ 중요 ★</p>

블로그 필터 모음​

from jinja2 import Environment
from datetime import datetime, timedelta
import re
from html import escape
from markupsafe import Markup

class BlogFilters:
    """블로그에 필요한 필터 모음"""

    @staticmethod
    def reading_time(text, wpm=200):
        """읽기 시간 계산 (분)"""
        words = len(re.findall(r'\w+', text))
        minutes = max(1, round(words / wpm))
        return f"{minutes}분"

    @staticmethod
    def summary(text, length=200):
        """요약 생성"""
        # HTML 제거
        text = re.sub(r'<[^>]+>', '', text)
        # 공백 정리
        text = re.sub(r'\s+', ' ', text).strip()
        if len(text) <= length:
            return text
        # 단어 경계에서 자르기
        return text[:length].rsplit(' ', 1)[0] + '...'

    @staticmethod
    def highlight_search(text, query):
        """검색어 하이라이트"""
        if not query:
            return text

        # 대소문자 구분 없이 검색
        pattern = re.compile(f'({re.escape(query)})', re.IGNORECASE)
        highlighted = pattern.sub(r'<mark>\1</mark>', text)
        return Markup(highlighted)

    @staticmethod
    def toc(html):
        """목차 생성 (h2, h3 태그에서)"""
        headings = re.findall(r'<h([23])>(.*?)</h\1>', html)
        toc = ['<ul class="toc">']

        for level, text in headings:
            indent = '  ' * (int(level) - 2)
            slug = re.sub(r'[^\w\s-]', '', text.lower()).replace(' ', '-')
            toc.append(f'{indent}<li><a href="#{slug}">{text}</a></li>')

        toc.append('</ul>')
        return Markup('\n'.join(toc))

    @staticmethod
    def obfuscate_email(email):
        """이메일 난독화 (스팸 방지)"""
        return email.replace('@', ' [at] ').replace('.', ' [dot] ')

# 등록
env = Environment()
filters = BlogFilters()
env.filters['reading_time'] = filters.reading_time
env.filters['summary'] = filters.summary
env.filters['highlight_search'] = filters.highlight_search
env.filters['toc'] = filters.toc
env.filters['obfuscate_email'] = filters.obfuscate_email

# 사용 예제
template = env.from_string("""
<article>
    <h1>{{ post.title|highlight_search(search_query)|safe }}</h1>
    <div class="meta">
        읽는 시간: {{ post.content|reading_time }}
    </div>

    <div class="summary">
        {{ post.content|summary(150) }}
    </div>

    <div class="toc">
        {{ post.content|toc|safe }}
    </div>

    <div class="author">
        작성자: {{ post.author_email|obfuscate_email }}
    </div>
</article>
""")

result = template.render(
    post={
        'title': 'Jinja2 커스텀 필터 가이드',
        'content': '<h2>소개</h2><p>...(긴 내용)</p><h3>사용법</h3><p>...</p>',
        'author_email': 'author@example.com'
    },
    search_query='Jinja2'
)

E-commerce 필터​

class EcommerceFilters:
    """전자상거래 필터"""

    @staticmethod
    def discount_price(price, discount_percent):
        """할인가 계산"""
        return price * (1 - discount_percent / 100)

    @staticmethod
    def tax_price(price, tax_rate=0.1):
        """세금 포함 가격"""
        return price * (1 + tax_rate)

    @staticmethod
    def price_range(products, attr='price'):
        """가격 범위 표시"""
        prices = [getattr(p, attr, 0) for p in products]
        if not prices:
            return "가격 정보 없음"

        min_price = min(prices)
        max_price = max(prices)

        if min_price == max_price:
            return f"₩{min_price:,.0f}"
        return f"₩{min_price:,.0f} - ₩{max_price:,.0f}"

    @staticmethod
    def stock_status(quantity):
        """재고 상태 표시"""
        if quantity <= 0:
            return Markup('<span class="text-danger">품절</span>')
        elif quantity < 10:
            return Markup(f'<span class="text-warning">재고 부족 ({quantity}개)</span>')
        else:
            return Markup('<span class="text-success">재고 있음</span>')

    @staticmethod
    def shipping_fee(total, free_threshold=50000):
        """배송비 계산"""
        if total >= free_threshold:
            return 0
        return 3000

# 등록
ecommerce = EcommerceFilters()
env.filters['discount_price'] = ecommerce.discount_price
env.filters['tax_price'] = ecommerce.tax_price
env.filters['price_range'] = ecommerce.price_range
env.filters['stock_status'] = ecommerce.stock_status
env.filters['shipping_fee'] = ecommerce.shipping_fee
