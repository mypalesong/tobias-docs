컨텍스트 프로세서

전역 변수와 함수를 관리해봅시다!

🎯 컨텍스트란?​

컨텍스트는 템플릿에서 접근 가능한 변수들의 딕셔너리입니다.

핵심 개념​

  • globals: 모든 템플릿에서 접근 가능한 전역 변수/함수
  • 컨텍스트 프로세서: 자동으로 컨텍스트에 변수 추가
  • 템플릿 변수: render() 호출 시 전달되는 변수

📐 기본 사용법​

globals 추가​

from jinja2 import Environment
from datetime import datetime

env = Environment()

# 전역 변수 추가
env.globals['site_name'] = '내 웹사이트'
env.globals['current_year'] = datetime.now().year

template = env.from_string("""
<footer>
    &copy; {{ current_year }} {{ site_name }}
</footer>
""")

print(template.render())
# 출력: <footer>&copy; 2024 내 웹사이트</footer>

전역 함수 추가​

def get_copyright(year=None):
    if year is None:
        year = datetime.now().year
    return f"&copy; {year} All rights reserved."

env.globals['copyright'] = get_copyright

template = env.from_string("""
<footer>
    {{ copyright() }}
</footer>
""")

print(template.render())
# 출력: <footer>&copy; 2024 All rights reserved.</footer>

update로 여러 개 추가​

env.globals.update({
    'app_name': 'My App',
    'version': '1.0.0',
    'debug': True,
    'support_email': 'support@example.com'
})

유틸리티 함수​

from jinja2 import Environment
import hashlib
import urllib.parse

env = Environment()

# URL 헬퍼
def url_for(endpoint, **params):
    """URL 생성"""
    url = f"/{endpoint}"
    if params:
        query = urllib.parse.urlencode(params)
        url += f"?{query}"
    return url

def static_url(path):
    """정적 파일 URL (캐시 버스팅 포함)"""
    version = hashlib.md5(path.encode()).hexdigest()[:8]
    return f"/static/{path}?v={version}"

# Gravatar
def gravatar_url(email, size=80):
    """Gravatar 이미지 URL 생성"""
    hash_value = hashlib.md5(email.lower().encode()).hexdigest()
    return f"https://www.gravatar.com/avatar/{hash_value}?s={size}"

# 전역 함수 등록
env.globals['url_for'] = url_for
env.globals['static_url'] = static_url
env.globals['gravatar_url'] = gravatar_url

# 사용
template = env.from_string("""
<link rel="stylesheet" href="{{ static_url('css/main.css') }}">
<script src="{{ static_url('js/app.js') }}"></script>

<img src="{{ gravatar_url(user.email, 100) }}" alt="{{ user.name }}">

<a href="{{ url_for('post', id=123, page=2) }}">게시물 보기</a>
""")

html = template.render(user={'email': 'user@example.com', 'name': 'User'})

인증 헬퍼​

from jinja2 import Environment

env = Environment()

class AuthHelper:
    """인증 관련 헬퍼"""

    def __init__(self, user=None):
        self.user = user

    def is_authenticated(self):
        """로그인 여부"""
        return self.user is not None

    def has_permission(self, permission):
        """권한 확인"""
        if not self.user:
            return False
        return permission in self.user.get('permissions', [])

    def is_admin(self):
        """관리자 여부"""
        return self.has_permission('admin')

    def can_edit(self, resource):
        """수정 권한"""
        if self.is_admin():
            return True
        if not self.user:
            return False
        return resource.get('author_id') == self.user.get('id')

# 사용 예제
def render_with_auth(template_string, user=None, **context):
    """인증 정보가 포함된 렌더링"""
    auth = AuthHelper(user)
    env.globals['auth'] = auth
    env.globals['user'] = user

    template = env.from_string(template_string)
    return template.render(**context)

html = render_with_auth("""
{% if auth.is_authenticated() %}
    <p>환영합니다, {{ user.name }}님!</p>

    {% if auth.is_admin() %}
        <a href="/admin">관리자 페이지</a>
    {% endif %}

    {% if auth.can_edit(post) %}
        <button>수정</button>
        <button>삭제</button>
    {% endif %}
{% else %}
    <a href="/login">로그인</a>
{% endif %}
""",
    user={'id': 1, 'name': '홍길동', 'permissions': ['user', 'admin']},
    post={'id': 123, 'author_id': 1}
)

Flask 컨텍스트 프로세서​

from flask import Flask
from datetime import datetime

app = Flask(__name__)

@app.context_processor
def inject_globals():
    """모든 템플릿에 자동으로 추가될 변수"""
    return {
        'site_name': 'My Flask App',
        'current_year': datetime.now().year,
        'debug': app.debug
    }

@app.context_processor
def utility_functions():
    """유틸리티 함수"""
    def format_price(amount):
        return f"₩{amount:,.0f}"

    return {
        'format_price': format_price
    }

@app.template_global()
def gravatar(email, size=80):
    """전역 함수로 등록"""
    import hashlib
    hash_value = hashlib.md5(email.lower().encode()).hexdigest()
    return f"https://www.gravatar.com/avatar/{hash_value}?s={size}"

@app.template_filter()
def timeago(dt):
    """전역 필터로 등록"""
    from datetime import datetime
    diff = datetime.now() - dt
    if diff.seconds < 60:
        return '방금 전'
    elif diff.seconds < 3600:
        return f'{diff.seconds // 60}분 전'
    elif diff.seconds < 86400:
        return f'{diff.seconds // 3600}시간 전'
    else:
        return f'{diff.days}일 전'

# 템플릿에서 사용
@app.route('/')
def index():
    return render_template('index.html')

🔍 고급 기법​

지연 평가​

from jinja2 import Environment

class LazyConfig:
    """지연 평가 설정"""

    @property
    def expensive_value(self):
        """필요할 때만 계산"""
        print("계산 중...")
        return sum(range(1000000))

env = Environment()
env.globals['config'] = LazyConfig()

template = env.from_string("""
{% if show_expensive %}
    {{ config.expensive_value }}
{% endif %}
""")

# show_expensive=False이면 계산 안 함
print(template.render(show_expensive=False))

컨텍스트 상속​

from jinja2 import Environment

class BaseContext:
    """기본 컨텍스트"""
    def __init__(self):
        self.site_name = "My Site"
        self.version = "1.0"

class UserContext(BaseContext):
    """사용자 컨텍스트"""
    def __init__(self, user):
        super().__init__()
        self.user = user
        self.is_authenticated = True

class AdminContext(UserContext):
    """관리자 컨텍스트"""
    def __init__(self, user):
        super().__init__(user)
        self.is_admin = True
        self.admin_menu = [
            {'name': '사용자 관리', 'url': '/admin/users'},
            {'name': '설정', 'url': '/admin/settings'}
        ]

# 사용
env = Environment()

def render_for_user(template_string, user, **kwargs):
    if user.get('is_admin'):
        ctx = AdminContext(user)
    elif user:
        ctx = UserContext(user)
    else:
        ctx = BaseContext()

    # 컨텍스트 객체의 모든 속성을 globals에 추가
    for key in dir(ctx):
        if not key.startswith('_'):
            env.globals[key] = getattr(ctx, key)

    template = env.from_string(template_string)
    return template.render(**kwargs)

스레드 안전 컨텍스트​

import threading
from jinja2 import Environment

env = Environment()

# 스레드 로컬 스토리지
_context = threading.local()

def get_current_user():
    """현재 스레드의 사용자 반환"""
    return getattr(_context, 'user', None)

def set_current_user(user):
    """현재 스레드에 사용자 설정"""
    _context.user = user

env.globals['current_user'] = get_current_user

# 사용
def process_request(user, template_string):
    set_current_user(user)
    template = env.from_string(template_string)
    return template.render()

# 각 스레드마다 다른 사용자 정보 유지

컨텍스트 확인​

from jinja2 import Environment

env = Environment()

env.globals['debug'] = True
env.globals['site_name'] = 'My Site'

def show_context():
    """현재 컨텍스트 출력"""
    return f"Globals: {list(env.globals.keys())}"

env.globals['show_context'] = show_context

template = env.from_string("""
<pre>{{ show_context() }}</pre>
""")

print(template.render())
# 출력: Globals: ['range', 'dict', 'lipsum', ..., 'debug', 'site_name', 'show_context']

컨텍스트 덤프​

from jinja2 import pass_context

@pass_context
def debug_context(context):
    """컨텍스트 변수 덤프"""
    import json
    # 직렬화 가능한 것만 출력
    safe_context = {}
    for key, value in context.items():
        try:
            json.dumps(value)
            safe_context[key] = value
        except:
            safe_context[key] = str(type(value))

    return safe_context

env.globals['debug_context'] = debug_context

template = env.from_string("""
{% if debug %}
    <pre>{{ debug_context()|tojson|safe }}</pre>
{% endif %}
""")
