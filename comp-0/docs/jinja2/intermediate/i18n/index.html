국제화 (i18n)

다국어 웹사이트를 만들어봅시다!

🎯 국제화란?​

i18n(Internationalization)은 애플리케이션을 여러 언어로 제공하는 것입니다.

핵심 개념​

  • i18n Extension: Jinja2 국제화 확장
  • gettext: 번역 함수
  • ngettext: 복수형 처리
  • Babel: Python 국제화 라이브러리

📐 기본 설정​

i18n 확장 활성화​

from jinja2 import Environment

env = Environment(extensions=['jinja2.ext.i18n'])

gettext 함수 설치​

# 방법 1: 더미 번역 (개발용)
env.install_null_translations()

# 방법 2: 실제 번역
from babel.support import Translations

translations = Translations.load('locales', ['ko'])
env.install_gettext_translations(translations)

# 방법 3: 커스텀 번역 함수
def my_gettext(message):
    translations = {
        'Hello': '안녕하세요',
        'Goodbye': '안녕히 가세요'
    }
    return translations.get(message, message)

env.install_gettext_callables(
    gettext=my_gettext,
    ngettext=lambda s, p, n: s if n == 1 else p
)

from jinja2 import Environment

env = Environment(extensions=['jinja2.ext.i18n'])

# 간단한 번역 딕셔너리
TRANSLATIONS = {
    'Hello, World!': '안녕하세요, 세계!',
    'Hello, %(name)s!': '안녕하세요, %(name)s님!',
    '%(user)s has %(count)s messages.': '%(user)s님에게 %(count)s개의 메시지가 있습니다.'
}

def gettext(message):
    return TRANSLATIONS.get(message, message)

def ngettext(singular, plural, n):
    # 간단한 복수형 처리
    key = singular if n == 1 else plural
    return TRANSLATIONS.get(key, key)

env.install_gettext_callables(gettext=gettext, ngettext=ngettext)

template = env.from_string("""
{% trans %}Hello, World!{% endtrans %}

{% trans name='홍길동' %}
    Hello, {{ name }}!
{% endtrans %}
""")

print(template.render())
# 출력:
# 안녕하세요, 세계!
# 안녕하세요, 홍길동님!

복수형 처리​

{% trans count=items|length %}
    You have {{ count }} item.
{% pluralize %}
    You have {{ count }} items.
{% endtrans %}
TRANSLATIONS_PLURAL = {
    ('You have %(count)s item.', 1): '%(count)s개의 항목이 있습니다.',
    ('You have %(count)s items.', 2): '%(count)s개의 항목이 있습니다.'
}

def ngettext(singular, plural, n):
    if n == 1:
        return TRANSLATIONS_PLURAL.get((singular, 1), singular) % {'count': n}
    else:
        return TRANSLATIONS_PLURAL.get((plural, 2), plural) % {'count': n}

env.install_gettext_callables(
    gettext=lambda s: s,
    ngettext=ngettext
)

template = env.from_string("""
{% trans count=1 %}
    You have {{ count }} item.
{% pluralize %}
    You have {{ count }} items.
{% endtrans %}

{% trans count=5 %}
    You have {{ count }} item.
{% pluralize %}
    You have {{ count }} items.
{% endtrans %}
""")

print(template.render())
# 출력:
# 1개의 항목이 있습니다.
# 5개의 항목이 있습니다.

# 2. 언어별 카탈로그 생성
pybabel init -i messages.pot -d locales -l ko
pybabel init -i messages.pot -d locales -l en

# 3. 번역 파일 편집 (locales/ko/LC_MESSAGES/messages.po)

# 4. 번역 컴파일
pybabel compile -d locales

# 5. 업데이트 (새로운 문자열 추가 시)
pybabel update -i messages.pot -d locales

💡 실전 예제​

다국어 웹사이트​

from jinja2 import Environment, FileSystemLoader
from babel.support import Translations

class I18nManager:
    """국제화 관리자"""

    def __init__(self, locales_dir='locales', default_locale='ko'):
        self.locales_dir = locales_dir
        self.default_locale = default_locale
        self.translations = {}
        self._load_translations()

    def _load_translations(self):
        """모든 번역 로드"""
        import os
        locales_path = self.locales_dir

        for locale in os.listdir(locales_path):
            locale_path = os.path.join(locales_path, locale)
            if os.path.isdir(locale_path):
                try:
                    trans = Translations.load(self.locales_dir, [locale])
                    self.translations[locale] = trans
                except Exception as e:
                    print(f"번역 로드 실패 ({locale}): {e}")

    def get_translations(self, locale):
        """특정 언어 번역 반환"""
        return self.translations.get(locale, self.translations.get(self.default_locale))

    def render_template(self, template_name, locale='ko', **context):
        """번역이 적용된 템플릿 렌더링"""
        env = Environment(
            loader=FileSystemLoader('templates'),
            extensions=['jinja2.ext.i18n']
        )

        # 번역 설치
        translations = self.get_translations(locale)
        env.install_gettext_translations(translations)

        # 현재 언어를 컨텍스트에 추가
        context['current_locale'] = locale
        context['available_locales'] = list(self.translations.keys())

        template = env.get_template(template_name)
        return template.render(**context)

# 사용
i18n = I18nManager()

# 한국어 렌더링
html_ko = i18n.render_template('index.html', locale='ko', user={'name': '홍길동'})

# 영어 렌더링
html_en = i18n.render_template('index.html', locale='en', user={'name': 'Hong'})

Flask 통합​

from flask import Flask, request, render_template
from flask_babel import Babel, gettext, ngettext

app = Flask(__name__)
app.config['BABEL_DEFAULT_LOCALE'] = 'ko'
app.config['BABEL_TRANSLATION_DIRECTORIES'] = 'translations'

babel = Babel(app)

@babel.localeselector
def get_locale():
    """사용자 언어 선택"""
    # 1. URL 파라미터
    locale = request.args.get('lang')
    if locale in ['ko', 'en', 'ja']:
        return locale

    # 2. 세션
    locale = session.get('locale')
    if locale:
        return locale

    # 3. Accept-Language 헤더
    return request.accept_languages.best_match(['ko', 'en', 'ja'])

@app.route('/')
def index():
    return render_template('index.html')

날짜/시간 현지화​

from babel.dates import format_date, format_datetime, format_time

def locale_date(dt, format='medium', locale='ko'):
    """날짜 현지화"""
    return format_date(dt, format=format, locale=locale)

def locale_datetime(dt, format='medium', locale='ko'):
    """날짜시간 현지화"""
    return format_datetime(dt, format=format, locale=locale)

env.filters['locale_date'] = locale_date
env.filters['locale_datetime'] = locale_datetime

template = env.from_string("""
한국어: {{ now|locale_date('full', 'ko') }}
영어: {{ now|locale_date('full', 'en') }}
일본어: {{ now|locale_date('full', 'ja') }}
""")

from datetime import datetime
print(template.render(now=datetime.now()))
# 출력:
# 한국어: 2024년 1월 15일 월요일
# 영어: Monday, January 15, 2024
# 일본어: 2024年1月15日月曜日

숫자/통화 현지화​

from babel.numbers import format_number, format_currency, format_percent

def locale_number(number, locale='ko'):
    """숫자 현지화"""
    return format_number(number, locale=locale)

def locale_currency(amount, currency='KRW', locale='ko'):
    """통화 현지화"""
    return format_currency(amount, currency, locale=locale)

def locale_percent(number, locale='ko'):
    """퍼센트 현지화"""
    return format_percent(number, locale=locale)

env.filters['locale_number'] = locale_number
env.filters['locale_currency'] = locale_currency
env.filters['locale_percent'] = locale_percent

template = env.from_string("""
숫자: {{ 1234567.89|locale_number('ko') }}
통화 (KRW): {{ 1500000|locale_currency('KRW', 'ko') }}
통화 (USD): {{ 1500|locale_currency('USD', 'en') }}
퍼센트: {{ 0.157|locale_percent('ko') }}
""")

print(template.render())
# 출력:
# 숫자: 1,234,567.89
# 통화 (KRW): ₩1,500,000
# 통화 (USD): $1,500.00
# 퍼센트: 16%
