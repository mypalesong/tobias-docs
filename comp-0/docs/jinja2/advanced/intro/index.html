고성능 템플릿 시스템​

from jinja2 import Environment, FileSystemLoader, select_autoescape
from jinja2.bytecode import FileSystemBytecodeCache
import asyncio

class ProductionEnvironment:
    """프로덕션용 최적화된 환경"""

    def __init__(self, template_dir='templates', cache_dir='cache'):
        self.env = Environment(
            loader=FileSystemLoader(template_dir),
            # 자동 이스케이핑 (보안)
            autoescape=select_autoescape(
                enabled_extensions=('html', 'xml'),
                default_for_string=True
            ),
            # 바이트코드 캐싱 (성능)
            bytecode_cache=FileSystemBytecodeCache(cache_dir),
            # 자동 리로드 비활성화 (프로덕션)
            auto_reload=False,
            # 트림 설정
            trim_blocks=True,
            lstrip_blocks=True,
            # 비동기 활성화
            enable_async=True
        )

        # 전역 함수 등록
        self._register_globals()

        # 필터 등록
        self._register_filters()

    def _register_globals(self):
        """전역 함수 등록"""
        from datetime import datetime

        self.env.globals.update({
            'now': datetime.now,
            'version': '1.0.0',
            'debug': False
        })

    def _register_filters(self):
        """필터 등록"""
        def format_currency(value):
            return f"₩{value:,.0f}"

        self.env.filters['currency'] = format_currency

    async def render_async(self, template_name, **context):
        """비동기 렌더링"""
        template = self.env.get_template(template_name)
        return await template.render_async(**context)

    def render(self, template_name, **context):
        """동기 렌더링"""
        template = self.env.get_template(template_name)
        return template.render(**context)

# 사용
env = ProductionEnvironment()

# 동기
html = env.render('index.html', user={'name': '홍길동'})

# 비동기
async def main():
    html = await env.render_async('index.html', user={'name': '홍길동'})
    print(html)

asyncio.run(main())

샌드박스 환경​

from jinja2.sandbox import SandboxedEnvironment

class SafeEnvironment(SandboxedEnvironment):
    """안전한 템플릿 실행 환경"""

    def is_safe_attribute(self, obj, attr, value):
        """속성 접근 제어"""
        # 비공개 속성 차단
        if attr.startswith('_'):
            return False

        # 특정 속성만 허용
        allowed_attrs = ['name', 'email', 'title', 'content']
        return attr in allowed_attrs

    def is_safe_callable(self, obj):
        """함수 호출 제어"""
        # 안전한 함수만 호출 가능
        safe_functions = {str, len, list, dict}
        return obj in safe_functions

# 사용
env = SafeEnvironment()

template = env.from_string("""
{{ user.name }}  {# OK #}
{{ user._password }}  {# 차단됨 #}
{{ user.__dict__ }}  {# 차단됨 #}
""")

try:
    html = template.render(user={'name': '홍길동', '_password': 'secret'})
    print(html)
except Exception as e:
    print(f"보안 오류: {e}")

커스텀 확장​

from jinja2 import nodes
from jinja2.ext import Extension

class CacheExtension(Extension):
    """템플릿 블록 캐싱 확장"""

    tags = {'cache'}

    def __init__(self, environment):
        super().__init__(environment)
        self.cache = {}

    def parse(self, parser):
        lineno = next(parser.stream).lineno

        # {% cache timeout, key %}
        timeout = parser.parse_expression()
        parser.stream.expect('comma')
        key = parser.parse_expression()

        # 블록 파싱
        body = parser.parse_statements(['name:endcache'], drop_needle=True)

        # 노드 생성
        return nodes.CallBlock(
            self.call_method('_cache_support', [timeout, key]),
            [], [], body
        ).set_lineno(lineno)

    def _cache_support(self, timeout, key, caller):
        """캐시 로직"""
        import time

        # 캐시 확인
        if key in self.cache:
            cached_data, cached_time = self.cache[key]
            if time.time() - cached_time < timeout:
                return cached_data

        # 렌더링 및 캐싱
        content = caller()
        self.cache[key] = (content, time.time())
        return content

# 사용
from jinja2 import Environment

env = Environment(extensions=[CacheExtension])

template = env.from_string("""
{% cache 60, 'sidebar' %}
    <aside>
        {# 무거운 연산 #}
        {% for item in expensive_query() %}
            {{ item }}
        {% endfor %}
    </aside>
{% endcache %}
""")

성능 모니터링​

import time
from jinja2 import Environment

class ProfilingEnvironment(Environment):
    """성능 프로파일링 환경"""

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.render_times = {}

    def get_template(self, name, parent=None, globals=None):
        """템플릿 로딩 시간 측정"""
        start = time.time()
        template = super().get_template(name, parent, globals)
        elapsed = time.time() - start

        if name not in self.render_times:
            self.render_times[name] = []
        self.render_times[name].append(('load', elapsed))

        # 래핑된 템플릿 반환
        return ProfiledTemplate(template, self, name)

    def get_stats(self):
        """통계 반환"""
        stats = {}
        for name, times in self.render_times.items():
            load_times = [t for op, t in times if op == 'load']
            render_times = [t for op, t in times if op == 'render']

            stats[name] = {
                'load_count': len(load_times),
                'render_count': len(render_times),
                'avg_load_time': sum(load_times) / len(load_times) if load_times else 0,
                'avg_render_time': sum(render_times) / len(render_times) if render_times else 0
            }
        return stats

class ProfiledTemplate:
    """프로파일링 템플릿 래퍼"""

    def __init__(self, template, environment, name):
        self.template = template
        self.environment = environment
        self.name = name

    def render(self, *args, **kwargs):
        start = time.time()
        result = self.template.render(*args, **kwargs)
        elapsed = time.time() - start

        self.environment.render_times[self.name].append(('render', elapsed))
        return result

# 사용
env = ProfilingEnvironment()

for i in range(100):
    template = env.get_template('index.html')
    html = template.render(user={'name': f'User {i}'})

# 통계 확인
print(env.get_stats())
