테이블 작업

Alembic으로 테이블을 생성, 수정, 삭제하는 방법을 배워봅시다!

🎯 학습 목표​

  • create_table로 테이블 생성
  • drop_table로 테이블 삭제
  • rename_table로 테이블 이름 변경
  • 외래 키와 제약조건 추가

📝 create_table: 테이블 생성​

기본 문법​

from alembic import op
import sqlalchemy as sa

def upgrade():
    op.create_table(
        'table_name',
        sa.Column('column1', sa.Type(), ...),
        sa.Column('column2', sa.Type(), ...),
        # 제약조건들...
    )

def downgrade():
    op.drop_table('table_name')

간단한 예제​

def upgrade():
    op.create_table(
        'products',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('name', sa.String(100), nullable=False),
        sa.Column('price', sa.Numeric(10, 2), nullable=False)
    )

def downgrade():
    op.drop_table('products')

🔧 다양한 테이블 생성 예제​

예제 1: 기본 키가 있는 테이블​

def upgrade():
    op.create_table(
        'users',
        sa.Column('id', sa.Integer(), primary_key=True, autoincrement=True),
        sa.Column('username', sa.String(50), unique=True, nullable=False),
        sa.Column('email', sa.String(100), unique=True, nullable=False),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now())
    )

def downgrade():
    op.drop_table('users')

예제 2: 외래 키가 있는 테이블​

def upgrade():
    op.create_table(
        'posts',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('title', sa.String(200), nullable=False),
        sa.Column('content', sa.Text(), nullable=False),
        sa.Column('author_id', sa.Integer(), nullable=False),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
        sa.ForeignKeyConstraint(
            ['author_id'],
            ['users.id'],
            name='fk_posts_author',
            ondelete='CASCADE'
        )
    )

def downgrade():
    op.drop_table('posts')

예제 3: 복합 기본 키​

def upgrade():
    op.create_table(
        'post_tags',
        sa.Column('post_id', sa.Integer(), nullable=False),
        sa.Column('tag_id', sa.Integer(), nullable=False),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
        sa.PrimaryKeyConstraint('post_id', 'tag_id', name='pk_post_tags'),
        sa.ForeignKeyConstraint(['post_id'], ['posts.id'], ondelete='CASCADE'),
        sa.ForeignKeyConstraint(['tag_id'], ['tags.id'], ondelete='CASCADE')
    )

def downgrade():
    op.drop_table('post_tags')

예제 4: 체크 제약조건​

def upgrade():
    op.create_table(
        'products',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('name', sa.String(100), nullable=False),
        sa.Column('price', sa.Numeric(10, 2), nullable=False),
        sa.Column('stock', sa.Integer(), default=0),
        sa.Column('rating', sa.Float()),
        sa.CheckConstraint('price > 0', name='check_positive_price'),
        sa.CheckConstraint('stock >= 0', name='check_non_negative_stock'),
        sa.CheckConstraint('rating >= 0 AND rating <= 5', name='check_rating_range')
    )

def downgrade():
    op.drop_table('products')

🗑️ drop_table: 테이블 삭제​

기본 사용법​

def upgrade():
    op.drop_table('old_table')

def downgrade():
    # 롤백 시 테이블 재생성
    op.create_table(
        'old_table',
        sa.Column('id', sa.Integer(), primary_key=True),
        # 원래 컬럼들...
    )

조건부 삭제 (if exists)​

Alembic은 기본적으로 if_exists를 지원하지 않지만, 직접 SQL로 가능:

def upgrade():
    # PostgreSQL
    op.execute('DROP TABLE IF EXISTS old_table')

    # 또는 try-catch
    try:
        op.drop_table('old_table')
    except:
        pass

def downgrade():
    op.create_table('old_table', ...)

🔄 rename_table: 테이블 이름 변경​

기본 사용법​

def upgrade():
    op.rename_table('old_name', 'new_name')

def downgrade():
    op.rename_table('new_name', 'old_name')

실전 예제​

"""rename user table to users

Revision ID: abc123
Revises: xyz789
"""

def upgrade():
    # 1. 테이블 이름 변경
    op.rename_table('user', 'users')

    # 2. 외래 키가 있는 경우 제약조건 이름도 업데이트
    op.drop_constraint('fk_post_user', 'posts', type_='foreignkey')
    op.create_foreign_key(
        'fk_post_users',
        'posts', 'users',
        ['user_id'], ['id']
    )

def downgrade():
    op.rename_table('users', 'user')
    op.drop_constraint('fk_post_users', 'posts', type_='foreignkey')
    op.create_foreign_key(
        'fk_post_user',
        'posts', 'user',
        ['user_id'], ['id']
    )

🎨 복잡한 실전 예제​

예제 1: 전자상거래 주문 시스템​

"""create order tables

Revision ID: ecommerce001
"""

def upgrade():
    # 주문 테이블
    op.create_table(
        'orders',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('user_id', sa.Integer(), nullable=False),
        sa.Column('status', sa.String(20), default='pending'),
        sa.Column('total_amount', sa.Numeric(12, 2), nullable=False),
        sa.Column('shipping_address', sa.Text()),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
        sa.Column('updated_at', sa.DateTime(), onupdate=sa.func.now()),
        sa.ForeignKeyConstraint(['user_id'], ['users.id']),
        sa.CheckConstraint('total_amount >= 0', name='check_total_amount')
    )

    # 주문 상품 테이블
    op.create_table(
        'order_items',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('order_id', sa.Integer(), nullable=False),
        sa.Column('product_id', sa.Integer(), nullable=False),
        sa.Column('quantity', sa.Integer(), nullable=False),
        sa.Column('unit_price', sa.Numeric(10, 2), nullable=False),
        sa.Column('subtotal', sa.Numeric(12, 2), nullable=False),
        sa.ForeignKeyConstraint(['order_id'], ['orders.id'], ondelete='CASCADE'),
        sa.ForeignKeyConstraint(['product_id'], ['products.id']),
        sa.CheckConstraint('quantity > 0', name='check_positive_quantity'),
        sa.CheckConstraint('unit_price >= 0', name='check_price'),
        sa.CheckConstraint('subtotal >= 0', name='check_subtotal')
    )

def downgrade():
    op.drop_table('order_items')
    op.drop_table('orders')

예제 2: 소셜 미디어 스키마​

"""create social media tables

Revision ID: social001
"""

def upgrade():
    # 프로필 테이블
    op.create_table(
        'profiles',
        sa.Column('user_id', sa.Integer(), primary_key=True),
        sa.Column('bio', sa.Text()),
        sa.Column('avatar_url', sa.String(500)),
        sa.Column('followers_count', sa.Integer(), default=0),
        sa.Column('following_count', sa.Integer(), default=0),
        sa.ForeignKeyConstraint(['user_id'], ['users.id'], ondelete='CASCADE')
    )

    # 팔로우 관계 테이블
    op.create_table(
        'follows',
        sa.Column('follower_id', sa.Integer(), nullable=False),
        sa.Column('following_id', sa.Integer(), nullable=False),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
        sa.PrimaryKeyConstraint('follower_id', 'following_id'),
        sa.ForeignKeyConstraint(['follower_id'], ['users.id'], ondelete='CASCADE'),
        sa.ForeignKeyConstraint(['following_id'], ['users.id'], ondelete='CASCADE'),
        sa.CheckConstraint('follower_id != following_id', name='check_no_self_follow')
    )

def downgrade():
    op.drop_table('follows')
    op.drop_table('profiles')

📊 테이블 스키마 복사​

기존 테이블과 동일한 구조의 새 테이블 생성:

def upgrade():
    # 기존 테이블 구조 복사
    op.create_table(
        'users_backup',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('username', sa.String(50), nullable=False),
        sa.Column('email', sa.String(100), nullable=False),
        # ... 기존 users 테이블과 동일한 구조
    )

    # 데이터 복사
    op.execute('''
        INSERT INTO users_backup
        SELECT * FROM users
    ''')

def downgrade():
    op.drop_table('users_backup')

🔍 데이터베이스별 고려사항​

PostgreSQL​

def upgrade():
    op.create_table(
        'users',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('data', sa.JSON()),  # JSON 타입
        sa.Column('metadata', postgresql.JSONB()),  # JSONB 타입
        sa.Column('tags', postgresql.ARRAY(sa.String)),  # 배열
        schema='public'  # 스키마 지정
    )

MySQL​

def upgrade():
    op.create_table(
        'users',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('name', sa.String(50)),
        mysql_engine='InnoDB',  # 스토리지 엔진
        mysql_charset='utf8mb4'  # 문자셋
    )

SQLite​

# SQLite는 일부 ALTER TABLE 제약이 있음
def upgrade():
    op.create_table(
        'users',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('name', sa.String(50))
    )

⚠️ 주의사항​

1. 테이블 삭제 시 데이터 손실​

# ❌ 위험: 데이터 손실
def downgrade():
    op.drop_table('users')  # 모든 사용자 데이터 삭제!

# ✅ 안전: 백업 먼저
def upgrade():
    # 백업 테이블 생성
    op.execute('CREATE TABLE users_backup AS SELECT * FROM users')
    # 작업 수행
    op.drop_table('users')

def downgrade():
    # 백업에서 복원
    op.execute('CREATE TABLE users AS SELECT * FROM users_backup')
    op.drop_table('users_backup')

2. 외래 키 순서​

# ✅ 올바른 순서: 부모 테이블 먼저
def upgrade():
    op.create_table('users', ...)  # 먼저
    op.create_table('posts', ...)  # 나중

def downgrade():
    op.drop_table('posts')   # 먼저 (자식)
    op.drop_table('users')   # 나중 (부모)

✅ 실습 과제​

과제 1: 블로그 시스템​

다음 요구사항에 맞는 테이블들을 생성하세요:

  1. categories 테이블

    • id, name, slug, description
    • parent_id (자기 참조)

  2. posts 테이블

    • id, title, slug, content, author_id, category_id
    • published_at, view_count
    • 외래 키: users.id, categories.id

  3. comments 테이블

    • id, post_id, user_id, content, parent_id
    • created_at
    • 외래 키와 자기 참조
정답 보기
def upgrade():
    # 1. Categories
    op.create_table(
        'categories',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('name', sa.String(50), unique=True, nullable=False),
        sa.Column('slug', sa.String(50), unique=True, nullable=False),
        sa.Column('description', sa.Text()),
        sa.Column('parent_id', sa.Integer()),
        sa.ForeignKeyConstraint(['parent_id'], ['categories.id'])
    )

    # 2. Posts
    op.create_table(
        'posts',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('title', sa.String(200), nullable=False),
        sa.Column('slug', sa.String(200), unique=True, nullable=False),
        sa.Column('content', sa.Text(), nullable=False),
        sa.Column('author_id', sa.Integer(), nullable=False),
        sa.Column('category_id', sa.Integer()),
        sa.Column('published_at', sa.DateTime()),
        sa.Column('view_count', sa.Integer(), default=0),
        sa.ForeignKeyConstraint(['author_id'], ['users.id']),
        sa.ForeignKeyConstraint(['category_id'], ['categories.id'])
    )

    # 3. Comments
    op.create_table(
        'comments',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('post_id', sa.Integer(), nullable=False),
        sa.Column('user_id', sa.Integer(), nullable=False),
        sa.Column('content', sa.Text(), nullable=False),
        sa.Column('parent_id', sa.Integer()),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
        sa.ForeignKeyConstraint(['post_id'], ['posts.id'], ondelete='CASCADE'),
        sa.ForeignKeyConstraint(['user_id'], ['users.id']),
        sa.ForeignKeyConstraint(['parent_id'], ['comments.id'])
    )

def downgrade():
    op.drop_table('comments')
    op.drop_table('posts')
    op.drop_table('categories')

📚 핵심 정리​

  • op.create_table(): 테이블 생성
  • op.drop_table(): 테이블 삭제
  • op.rename_table(): 테이블 이름 변경
  • 외래 키 순서: 부모 테이블 먼저 생성, 자식 테이블 먼저 삭제
  • downgrade 함수는 upgrade의 역순으로 작성
