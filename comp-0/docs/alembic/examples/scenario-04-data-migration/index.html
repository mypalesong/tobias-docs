시나리오 4: 유진의 데이터 대변환

글로벌 서비스 확장을 위해 가격 데이터를 변환하는 유진의 도전기입니다.

📌 상황 설명​

쇼핑몰 '글로벌마트'의 백엔드 개발자 유진. 지금까지 한국 원화(KRW)만 지원했는데, 이제 미국 달러(USD)도 지원해야 합니다. CEO로부터 "모든 가격을 USD로 통일하라"는 지시를 받았습니다.

문제: 데이터베이스에 이미 10만 개의 상품이 있습니다!

🎯 목표​

  • 대량 데이터 안전하게 변환
  • 롤백 가능한 마이그레이션
  • 데이터 무결성 보장
  • 다운타임 최소화

💬 대화​

CEO: "유진님, 다음 주까지 미국 시장 진출 준비를 해야 해요. 가격을 달러로 바꿔주세요."

유진: "네! 그런데 이미 10만 개의 상품이 있는데... 안전하게 변환할 수 있을까요?"

시니어 현우: "데이터 마이그레이션은 신중해야 해요. 단계별로 접근합시다."

1단계: 새 컬럼 추가​

alembic revision -m "add price_usd column"

2단계: 애플리케이션 코드 업데이트​

# models.py - 두 컬럼 모두 유지
class Product(Base):
    __tablename__ = 'products'

    id = Column(Integer, primary_key=True)
    name = Column(String(200), nullable=False)
    price_krw = Column(Integer)  # 레거시, 곧 삭제 예정
    price_usd = Column(Numeric(10, 2), nullable=False)  # 새 컬럼
    stock = Column(Integer, default=0)

    @property
    def price(self):
        """기본적으로 USD 사용"""
        return self.price_usd
# 서비스 코드 업데이트
# Before
product.price_krw = 29000

# 마이그레이션 생성
alembic revision -m "remove price_krw column"
"""remove price_krw column

Revision ID: bbb002
Revises: aaa001
"""

def upgrade():
    # 안전을 위해 백업 컬럼 생성 (선택사항)
    op.execute('''
        CREATE TABLE products_price_backup AS
        SELECT id, price_krw FROM products
    ''')

    # 기존 컬럼 삭제
    op.drop_column('products', 'price_krw')

def downgrade():
    # 컬럼 재생성
    op.add_column('products',
        sa.Column('price_krw', sa.Integer())
    )

    # 백업에서 복원
    op.execute('''
        UPDATE products p
        SET price_krw = b.price_krw
        FROM products_price_backup b
        WHERE p.id = b.id
    ''')

    # 백업 테이블 삭제
    op.execute('DROP TABLE products_price_backup')
alembic upgrade head

문제: 100만 개 상품 변환​

유진: "현우님, 이제 상품이 100만 개가 넘는데요. 한 번에 UPDATE하면 너무 오래 걸리지 않을까요?"

현우: "맞아요. 배치로 나눠서 처리합시다."

"""convert prices in batches for large dataset

Revision ID: ccc003
"""

def upgrade():
    from sqlalchemy import create_engine
    from alembic import op
    import time

    conn = op.get_bind()

    # 1. 새 컬럼 추가
    op.add_column('products',
        sa.Column('price_usd', sa.Numeric(10, 2))
    )

    # 2. 배치 처리
    batch_size = 10000
    total = conn.execute(sa.text('SELECT COUNT(*) FROM products')).scalar()
    print(f"Total products: {total}")

    for offset in range(0, total, batch_size):
        # 배치 업데이트
        conn.execute(sa.text(f'''
            UPDATE products
            SET price_usd = ROUND(price_krw / 1300.0, 2)
            WHERE id IN (
                SELECT id FROM products
                WHERE price_usd IS NULL
                LIMIT {batch_size}
            )
        '''))

        processed = min(offset + batch_size, total)
        progress = (processed / total) * 100
        print(f"Progress: {processed}/{total} ({progress:.1f}%)")

        # 커밋 (트랜잭션 분리)
        conn.execute(sa.text('COMMIT'))
        time.sleep(0.1)  # DB 부하 완화

    # 3. NOT NULL 제약조건
    op.alter_column('products', 'price_usd', nullable=False)

def downgrade():
    op.drop_column('products', 'price_usd')

💡 고급 기법​

1. 트리거로 자동 동기화​

def upgrade():
    # 새 컬럼 추가
    op.add_column('products', sa.Column('price_usd', sa.Numeric(10, 2)))

    # 트리거 생성: price_krw 변경 시 price_usd 자동 업데이트
    op.execute('''
        CREATE OR REPLACE FUNCTION sync_price_usd()
        RETURNS TRIGGER AS $$
        BEGIN
            NEW.price_usd := ROUND(NEW.price_krw / 1300.0, 2);
            RETURN NEW;
        END;
        $$ LANGUAGE plpgsql;

        CREATE TRIGGER trigger_sync_price_usd
        BEFORE INSERT OR UPDATE OF price_krw ON products
        FOR EACH ROW
        EXECUTE FUNCTION sync_price_usd();
    ''')

    # 기존 데이터 변환
    op.execute('UPDATE products SET price_usd = ROUND(price_krw / 1300.0, 2)')

def downgrade():
    op.execute('DROP TRIGGER IF EXISTS trigger_sync_price_usd ON products')
    op.execute('DROP FUNCTION IF EXISTS sync_price_usd()')
    op.drop_column('products', 'price_usd')

2. 데이터 검증 함수​

✅ 실습 과제​

다음 데이터 변환을 구현하세요:

  1. users 테이블의 full_name을 first_name, last_name으로 분리
  2. "홍길동" → first_name: "길동", last_name: "홍"
  3. 기존 10만 명의 사용자 데이터 처리
정답 보기
"""split fullname into first and last name

Revision ID: split001
"""

def upgrade():
    # 1. 새 컬럼 추가
    op.add_column('users', sa.Column('first_name', sa.String(50)))
    op.add_column('users', sa.Column('last_name', sa.String(50)))

    # 2. 데이터 분리 (한국 이름: 성+이름)
    op.execute('''
        UPDATE users
        SET
            last_name = SUBSTRING(full_name, 1, 1),
            first_name = SUBSTRING(full_name, 2)
        WHERE full_name IS NOT NULL
    ''')

    # 3. 검증
    conn = op.get_bind()
    null_count = conn.execute(sa.text('''
        SELECT COUNT(*) FROM users
        WHERE full_name IS NOT NULL
        AND (first_name IS NULL OR last_name IS NULL)
    ''')).scalar()

    if null_count > 0:
        raise Exception(f"{null_count} rows not converted!")

    # 4. 기존 컬럼 삭제 (선택)
    # op.drop_column('users', 'full_name')

def downgrade():
    op.add_column('users', sa.Column('full_name', sa.String(100)))
    op.execute('''
        UPDATE users
        SET full_name = last_name || first_name
    ''')
    op.drop_column('users', 'first_name')
    op.drop_column('users', 'last_name')
