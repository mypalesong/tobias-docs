시나리오 2: 수진의 기능 추가

이커머스 서비스에 리뷰 기능을 추가하는 수진의 이야기입니다.

📌 상황 설명​

수진은 온라인 쇼핑몰 '마켓플레이스'의 백엔드 개발자입니다. PM으로부터 "고객들이 제품 리뷰를 남길 수 있는 기능이 필요하다"는 요청을 받았습니다. 기존 데이터베이스에 새로운 테이블을 추가하고, 기존 테이블도 수정해야 합니다.

🎯 목표​

  • 기존 스키마에 새 테이블 추가
  • 기존 테이블에 컬럼 추가
  • autogenerate 활용
  • 여러 마이그레이션 관리

💬 대화​

PM 지훈: "수진님, 이번 스프린트에는 리뷰 기능을 추가해야 해요. 고객들이 별점과 리뷰를 남길 수 있어야 합니다."

수진: "알겠습니다! 어떤 정보들이 필요한가요?"

지훈: "고객이 제품에 대해 별점(1-5), 리뷰 텍스트, 그리고 사진도 올릴 수 있으면 좋겠어요. 아, 그리고 제품 테이블에도 평균 별점과 리뷰 수를 저장하면 성능에 좋을 것 같아요."

수진: "좋아요! 그럼 reviews 테이블을 만들고, products 테이블에 컬럼을 추가하면 되겠네요."

⚡ 실습​

1단계: 요구사항 분석​

수진의 메모:

필요한 변경사항:
1. reviews 테이블 생성
   - 리뷰 ID, 제품 ID, 사용자 ID
   - 별점, 리뷰 내용, 이미지 URL
   - 작성일시

2. products 테이블 수정
   - average_rating 컬럼 추가
   - review_count 컬럼 추가

3. 제약조건
   - 별점은 1~5 사이
   - 한 사용자가 같은 제품에 여러 리뷰 불가

2단계: SQLAlchemy 모델 수정​

# models.py - 기존 Product 모델 수정
class Product(Base):
    __tablename__ = 'products'

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(200), nullable=False)
    description = Column(Text)
    price = Column(Numeric(10, 2), nullable=False)
    stock = Column(Integer, default=0)
    category_id = Column(Integer, ForeignKey('categories.id'))

    # 새로 추가된 컬럼들
    average_rating = Column(Numeric(3, 2), default=0.0)
    review_count = Column(Integer, default=0)

    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, onupdate=datetime.utcnow)

    # 관계 추가
    reviews = relationship('Review', back_populates='product')
# models.py - 새로운 Review 모델 추가
class Review(Base):
    __tablename__ = 'reviews'

    id = Column(Integer, primary_key=True, index=True)
    product_id = Column(Integer, ForeignKey('products.id', ondelete='CASCADE'), nullable=False)
    user_id = Column(Integer, ForeignKey('users.id', ondelete='CASCADE'), nullable=False)

    rating = Column(Integer, nullable=False)  # 1-5
    title = Column(String(200))
    content = Column(Text, nullable=False)
    images = Column(JSON)  # 이미지 URL 배열

    is_verified_purchase = Column(Boolean, default=False)
    helpful_count = Column(Integer, default=0)

    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, onupdate=datetime.utcnow)

    # 관계
    product = relationship('Product', back_populates='reviews')
    user = relationship('User', back_populates='reviews')

    # 테이블 수준 제약조건
    __table_args__ = (
        CheckConstraint('rating >= 1 AND rating <= 5', name='check_rating_range'),
        UniqueConstraint('product_id', 'user_id', name='uq_product_user_review'),
    )
# models.py - User 모델에 관계 추가
class User(Base):
    __tablename__ = 'users'
    # ... 기존 필드들
    reviews = relationship('Review', back_populates='user')

수진: "모델 준비 완료! 이제 마이그레이션을 만들어볼까?"

3단계: Autogenerate로 마이그레이션 생성​

# 최신 상태로 업데이트
git pull origin main
alembic upgrade head

# 마이그레이션 자동 생성
alembic revision --autogenerate -m "add reviews table and rating fields to products"

생성된 파일 확인:

"""add reviews table and rating fields to products

Revision ID: def789
Revises: abc123
Create Date: 2024-01-21 14:30:00.123456
"""

def upgrade():
    # Reviews 테이블 생성
    op.create_table('reviews',
        sa.Column('id', sa.Integer(), nullable=False),
        sa.Column('product_id', sa.Integer(), nullable=False),
        sa.Column('user_id', sa.Integer(), nullable=False),
        sa.Column('rating', sa.Integer(), nullable=False),
        sa.Column('title', sa.String(length=200), nullable=True),
        sa.Column('content', sa.Text(), nullable=False),
        sa.Column('images', sa.JSON(), nullable=True),
        sa.Column('is_verified_purchase', sa.Boolean(), nullable=True),
        sa.Column('helpful_count', sa.Integer(), nullable=True),
        sa.Column('created_at', sa.DateTime(), nullable=True),
        sa.Column('updated_at', sa.DateTime(), nullable=True),
        sa.CheckConstraint('rating >= 1 AND rating <= 5', name='check_rating_range'),
        sa.ForeignKeyConstraint(['product_id'], ['products.id'], ondelete='CASCADE'),
        sa.ForeignKeyConstraint(['user_id'], ['users.id'], ondelete='CASCADE'),
        sa.PrimaryKeyConstraint('id'),
        sa.UniqueConstraint('product_id', 'user_id', name='uq_product_user_review')
    )
    op.create_index(op.f('ix_reviews_id'), 'reviews', ['id'], unique=False)

    # Products 테이블에 컬럼 추가
    op.add_column('products', sa.Column('average_rating', sa.Numeric(precision=3, scale=2), nullable=True))
    op.add_column('products', sa.Column('review_count', sa.Integer(), nullable=True))

def downgrade():
    op.drop_column('products', 'review_count')
    op.drop_column('products', 'average_rating')
    op.drop_index(op.f('ix_reviews_id'), table_name='reviews')
    op.drop_table('reviews')

수진: "와, Alembic이 모든 변경사항을 자동으로 감지했네요!"

4단계: 마이그레이션 검토 및 수정​

수진: "잠깐, products 테이블의 새 컬럼들은 기본값을 설정해야겠어요."

# 마이그레이션 파일 수정
def upgrade():
    # ... reviews 테이블 생성 코드

    # Products 테이블에 컬럼 추가 (기본값 설정)
    op.add_column('products',
        sa.Column('average_rating', sa.Numeric(precision=3, scale=2),
                  server_default='0.0', nullable=False))
    op.add_column('products',
        sa.Column('review_count', sa.Integer(),
                  server_default='0', nullable=False))

7단계: 테스트 데이터 생성​

# test_reviews.py
from models import Product, Review, User
from database import SessionLocal

db = SessionLocal()

# 테스트 사용자와 제품 생성
user = User(username='testuser', email='test@example.com', password_hash='hashed')
product = Product(name='테스트 상품', price=10000, stock=100)

db.add(user)
db.add(product)
db.commit()

# 리뷰 생성
review = Review(
    product_id=product.id,
    user_id=user.id,
    rating=5,
    title='정말 좋아요!',
    content='배송도 빠르고 품질도 좋습니다.',
    is_verified_purchase=True
)

db.add(review)
db.commit()

# 제품 평점 업데이트
product.average_rating = 5.0
product.review_count = 1
db.commit()

print(f"리뷰 생성 완료: {review.id}")
print(f"제품 평점: {product.average_rating} ({product.review_count}개)")
python test_reviews.py
# 리뷰 생성 완료: 1
# 제품 평점: 5.0 (1개)

8단계: 추가 기능 - 인덱스 최적화​

수진: "리뷰가 많아지면 조회 성능이 중요할 것 같아요. 인덱스를 추가해야겠어요."

alembic revision -m "add indexes for review queries"
"""add indexes for review queries

Revision ID: ghi012
Revises: def789
"""

def upgrade():
    # 제품별 리뷰 조회용
    op.create_index('idx_reviews_product_created', 'reviews',
                    ['product_id', 'created_at'])

    # 사용자별 리뷰 조회용
    op.create_index('idx_reviews_user', 'reviews', ['user_id'])

    # 높은 평점 리뷰 필터링용
    op.create_index('idx_reviews_rating', 'reviews', ['rating'])

    # 제품 평점 정렬용
    op.create_index('idx_products_rating', 'products', ['average_rating'])

def downgrade():
    op.drop_index('idx_products_rating', table_name='products')
    op.drop_index('idx_reviews_rating', table_name='reviews')
    op.drop_index('idx_reviews_user', table_name='reviews')
    op.drop_index('idx_reviews_product_created', table_name='reviews')
alembic upgrade head

  • 성능 최적화: 인덱스 추가

마이그레이션 분리​

# ❌ 나쁜 예: 모든 변경을 하나에
alembic revision -m "add everything"

alembic revision -m "add reviews table"
alembic revision -m "add indexes for performance"

💡 실전 팁​

1. 관계(Relationship) 설정​

# 양방향 관계 설정
class Product(Base):
    reviews = relationship('Review', back_populates='product')

class Review(Base):
    product = relationship('Product', back_populates='reviews')

2. JSON 컬럼 활용 (PostgreSQL)​

# 이미지 URL 배열 저장
images = Column(JSON)

# 사용 예:
review.images = [
    'https://example.com/image1.jpg',
    'https://example.com/image2.jpg'
]

3. 제약조건 네이밍​

# 명확한 이름 사용
CheckConstraint('rating >= 1 AND rating <= 5', name='check_rating_range')
UniqueConstraint('product_id', 'user_id', name='uq_product_user_review')

🚨 주의사항​

1. 기존 데이터 고려​

# ❌ 위험: NOT NULL을 바로 추가
op.add_column('products', sa.Column('average_rating', sa.Numeric(3, 2), nullable=False))

# ✅ 안전: 기본값 설정
op.add_column('products',
    sa.Column('average_rating', sa.Numeric(3, 2),
              server_default='0.0', nullable=False))

2. 외래 키 삭제 옵션​

# CASCADE: 부모 삭제 시 자식도 삭제
ForeignKey('products.id', ondelete='CASCADE')

# SET NULL: 부모 삭제 시 NULL로 설정
ForeignKey('categories.id', ondelete='SET NULL')

📊 성능 고려사항​

복합 인덱스 순서​

# ✅ 좋은 예: 자주 함께 사용되는 컬럼
op.create_index('idx_reviews_product_created', 'reviews',
    ['product_id', 'created_at DESC']  # 제품별 최신 리뷰
)

# ❌ 나쁜 예: 선택도가 낮은 컬럼 먼저
op.create_index('idx_reviews_rating_product', 'reviews',
    ['rating', 'product_id']  # rating은 값이 1~5뿐
)
