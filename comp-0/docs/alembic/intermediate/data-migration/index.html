📝 기본 데이터 마이그레이션​

op.execute() 사용​

def upgrade():
    # 1. 컬럼 추가
    op.add_column('users', sa.Column('full_name', sa.String(100)))

    # 2. 데이터 변환
    op.execute('''
        UPDATE users
        SET full_name = first_name || ' ' || last_name
    ''')

    # 3. 기존 컬럼 삭제 (선택)
    op.drop_column('users', 'first_name')
    op.drop_column('users', 'last_name')

def downgrade():
    op.add_column('users', sa.Column('first_name', sa.String(50)))
    op.add_column('users', sa.Column('last_name', sa.String(50)))

    op.execute('''
        UPDATE users
        SET first_name = SPLIT_PART(full_name, ' ', 1),
            last_name = SPLIT_PART(full_name, ' ', 2)
    ''')

    op.drop_column('users', 'full_name')

🔧 실전 예제​

예제 1: 가격 단위 변경 (원 → 달러)​

"""convert prices from KRW to USD

Revision ID: price001
"""

def upgrade():
    # 1. 새 컬럼 추가
    op.add_column('products',
        sa.Column('price_usd', sa.Numeric(10, 2))
    )

    # 2. 환율 적용 (1200원 = 1달러)
    op.execute('''
        UPDATE products
        SET price_usd = ROUND(price_krw / 1200.0, 2)
    ''')

    # 3. NOT NULL 제약 추가
    op.alter_column('products', 'price_usd', nullable=False)

    # 4. 기존 컬럼 삭제
    op.drop_column('products', 'price_krw')

    # 5. 컬럼 이름 변경
    op.alter_column('products', 'price_usd', new_column_name='price')

def downgrade():
    op.alter_column('products', 'price', new_column_name='price_usd')
    op.add_column('products', sa.Column('price_krw', sa.Integer()))
    op.execute('UPDATE products SET price_krw = ROUND(price_usd * 1200)')
    op.drop_column('products', 'price_usd')

예제 2: 상태 값 변경​

"""standardize order status values

Revision ID: status001
"""

def upgrade():
    # 1. 새 컬럼 추가
    op.add_column('orders',
        sa.Column('status_new', sa.String(20))
    )

    # 2. 데이터 변환
    op.execute('''
        UPDATE orders
        SET status_new = CASE
            WHEN status = '대기' THEN 'pending'
            WHEN status = '처리중' THEN 'processing'
            WHEN status = '배송중' THEN 'shipping'
            WHEN status = '완료' THEN 'completed'
            WHEN status = '취소' THEN 'cancelled'
            ELSE 'pending'
        END
    ''')

    # 3. 제약조건 추가
    op.create_check_constraint(
        'check_status_values',
        'orders',
        "status_new IN ('pending', 'processing', 'shipping', 'completed', 'cancelled')"
    )

    # 4. 기존 컬럼 삭제
    op.drop_column('orders', 'status')

    # 5. 이름 변경
    op.alter_column('orders', 'status_new', new_column_name='status')

def downgrade():
    op.alter_column('orders', 'status', new_column_name='status_new')
    op.add_column('orders', sa.Column('status', sa.String(20)))
    op.execute('''
        UPDATE orders
        SET status = CASE
            WHEN status_new = 'pending' THEN '대기'
            WHEN status_new = 'processing' THEN '처리중'
            WHEN status_new = 'shipping' THEN '배송중'
            WHEN status_new = 'completed' THEN '완료'
            WHEN status_new = 'cancelled' THEN '취소'
        END
    ''')
    op.drop_constraint('check_status_values', 'orders')
    op.drop_column('orders', 'status_new')

예제 3: 정규화 - 테이블 분리​

"""normalize user addresses to separate table

Revision ID: normalize001
"""

def upgrade():
    # 1. 새 주소 테이블 생성
    op.create_table(
        'addresses',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('user_id', sa.Integer(), nullable=False),
        sa.Column('street', sa.String(200)),
        sa.Column('city', sa.String(100)),
        sa.Column('zip_code', sa.String(10)),
        sa.Column('is_primary', sa.Boolean(), default=True),
        sa.ForeignKeyConstraint(['user_id'], ['users.id'], ondelete='CASCADE')
    )

    # 2. 기존 데이터 이동
    op.execute('''
        INSERT INTO addresses (user_id, street, city, zip_code, is_primary)
        SELECT id, address, city, zip_code, true
        FROM users
        WHERE address IS NOT NULL
    ''')

    # 3. 기존 컬럼 삭제
    op.drop_column('users', 'zip_code')
    op.drop_column('users', 'city')
    op.drop_column('users', 'address')

def downgrade():
    op.add_column('users', sa.Column('address', sa.String(200)))
    op.add_column('users', sa.Column('city', sa.String(100)))
    op.add_column('users', sa.Column('zip_code', sa.String(10)))

    op.execute('''
        UPDATE users u
        SET address = a.street,
            city = a.city,
            zip_code = a.zip_code
        FROM addresses a
        WHERE a.user_id = u.id AND a.is_primary = true
    ''')

    op.drop_table('addresses')

bind를 통한 직접 SQL 실행​

from alembic import op
import sqlalchemy as sa

def upgrade():
    # Connection 객체 가져오기
    conn = op.get_bind()

    # 데이터 조회
    result = conn.execute(sa.text('SELECT id, old_field FROM users'))

    # 각 행 처리
    for row in result:
        new_value = process_data(row['old_field'])
        conn.execute(
            sa.text('UPDATE users SET new_field = :val WHERE id = :id'),
            {'val': new_value, 'id': row['id']}
        )

def upgrade():

    from sqlalchemy.orm import Session
    session = Session(bind=conn)

    # 대량 데이터 삽입

    session.commit()

배치 처리​

"""process large dataset in batches

Revision ID: batch001
"""

def upgrade():
    conn = op.get_bind()
    batch_size = 1000
    offset = 0

    while True:
        # 배치 단위로 조회
        result = conn.execute(sa.text(f'''
            SELECT id, old_value
            FROM products
            ORDER BY id
            LIMIT {batch_size} OFFSET {offset}
        '''))

        rows = result.fetchall()
        if not rows:
            break

        # 배치 처리
        for row in rows:
            new_value = row['old_value'] * 1.1  # 10% 인상
            conn.execute(sa.text('''
                UPDATE products
                SET new_value = :val
                WHERE id = :id
            '''), {'val': new_value, 'id': row['id']})

        offset += batch_size

        # 진행상황 출력
        print(f"Processed {offset} rows...")

트랜잭션 관리​

def upgrade():
    conn = op.get_bind()

    # 트랜잭션 시작
    trans = conn.begin()

    try:
        # 1. 백업 테이블 생성
        conn.execute(sa.text('''
            CREATE TABLE users_backup AS
            SELECT * FROM users
        '''))

        # 2. 데이터 변환
        conn.execute(sa.text('''
            UPDATE users
            SET email = LOWER(email)
        '''))

        # 3. 검증
        result = conn.execute(sa.text('''
            SELECT COUNT(*) FROM users
            WHERE email != LOWER(email)
        '''))
        if result.scalar() > 0:
            raise Exception("Email conversion failed!")

        # 성공 시 커밋
        trans.commit()

        # 백업 삭제
        conn.execute(sa.text('DROP TABLE users_backup'))

    except Exception as e:
        # 실패 시 롤백
        trans.rollback()
        raise e

✅ 실습 과제​

블로그 시스템에서 다음 데이터 마이그레이션을 구현하세요:

  1. posts 테이블에 view_count가 문자열로 저장되어 있습니다
  2. 이를 Integer로 변환하되, 숫자가 아닌 값은 0으로 설정
  3. 음수 값도 0으로 변경
정답 보기
"""convert view_count to integer

Revision ID: views001
"""

def upgrade():
    # 1. 새 컬럼 추가
    op.add_column('posts',
        sa.Column('view_count_int', sa.Integer())
    )

    # 2. 숫자 변환 (PostgreSQL)
    op.execute('''
        UPDATE posts
        SET view_count_int = CASE
            WHEN view_count ~ '^[0-9]+$' THEN CAST(view_count AS INTEGER)
            ELSE 0
        END
    ''')

    # 3. 음수를 0으로
    op.execute('''
        UPDATE posts
        SET view_count_int = 0
        WHERE view_count_int < 0
    ''')

    # 4. NOT NULL 제약 추가
    op.alter_column('posts', 'view_count_int',
        nullable=False,
        server_default='0'
    )

    # 5. 기존 컬럼 삭제
    op.drop_column('posts', 'view_count')

    # 6. 이름 변경
    op.alter_column('posts', 'view_count_int',
        new_column_name='view_count'
    )

def downgrade():
    op.alter_column('posts', 'view_count',
        new_column_name='view_count_int'
    )
    op.add_column('posts',
        sa.Column('view_count', sa.String(20))
    )
    op.execute('''
        UPDATE posts
        SET view_count = CAST(view_count_int AS VARCHAR)
    ''')
    op.drop_column('posts', 'view_count_int')
