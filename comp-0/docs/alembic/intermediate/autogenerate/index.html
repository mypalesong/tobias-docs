자동 마이그레이션 생성 (Autogenerate)

SQLAlchemy 모델에서 자동으로 마이그레이션을 생성하는 방법을 배워봅시다!

🎯 학습 목표​

  • autogenerate로 마이그레이션 자동 생성
  • 모델 변경 사항 자동 감지
  • 자동 생성의 한계 이해

🚀 Autogenerate 사용하기​

기본 명령어​

# 자동으로 마이그레이션 생성
alembic revision --autogenerate -m "add user profile fields"

설정 확인​

alembic/env.py에서 target_metadata 설정이 필요합니다:

# alembic/env.py
from myapp.models import Base

# 이 부분이 중요!
target_metadata = Base.metadata

📝 실전 예제​

1단계: SQLAlchemy 모델 작성​

# models.py
from sqlalchemy import Column, Integer, String, DateTime
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class User(Base):
    __tablename__ = 'users'

    id = Column(Integer, primary_key=True)
    username = Column(String(50), unique=True, nullable=False)
    email = Column(String(100), unique=True, nullable=False)
    created_at = Column(DateTime, default=datetime.utcnow)

2단계: 초기 마이그레이션 생성​

alembic revision --autogenerate -m "create users table"

생성된 파일:

"""create users table

Revision ID: abc123
"""

def upgrade():
    op.create_table('users',
    sa.Column('id', sa.Integer(), nullable=False),
    sa.Column('username', sa.String(length=50), nullable=False),
    sa.Column('email', sa.String(length=100), nullable=False),
    sa.Column('created_at', sa.DateTime(), nullable=True),
    sa.PrimaryKeyConstraint('id')
    )
    op.create_index(op.f('ix_users_email'), 'users', ['email'], unique=True)
    op.create_index(op.f('ix_users_username'), 'users', ['username'], unique=True)

def downgrade():
    op.drop_index(op.f('ix_users_username'), table_name='users')
    op.drop_index(op.f('ix_users_email'), table_name='users')
    op.drop_table('users')

3단계: 모델 수정​

# models.py - 필드 추가

    created_at = Column(DateTime, default=datetime.utcnow)

4단계: 변경사항 자동 감지​

alembic revision --autogenerate -m "add user profile fields"

생성된 마이그레이션:

"""add user profile fields

Revision ID: def456
Revises: abc123
"""

def upgrade():
    op.add_column('users', sa.Column('avatar_url', sa.String(length=500), nullable=True))
    op.add_column('users', sa.Column('bio', sa.Text(), nullable=True))
    op.add_column('users', sa.Column('first_name', sa.String(length=50), nullable=True))
    op.add_column('users', sa.Column('last_name', sa.String(length=50), nullable=True))
    op.add_column('users', sa.Column('updated_at', sa.DateTime(), nullable=True))

def downgrade():
    op.drop_column('users', 'updated_at')
    op.drop_column('users', 'last_name')
    op.drop_column('users', 'first_name')
    op.drop_column('users', 'bio')
    op.drop_column('users', 'avatar_url')

🔍 Autogenerate가 감지하는 것​

✅ 감지됨:

  • 테이블 추가/삭제
  • 컬럼 추가/삭제
  • 컬럼 타입 변경 (대부분)
  • NULL 허용 여부 변경
  • 인덱스 추가/삭제
  • 유니크 제약조건 추가/삭제

❌ 감지되지 않음:

  • 컬럼 이름 변경 (삭제+추가로 인식)
  • 테이블 이름 변경
  • Check 제약조건 변경
  • 서버 기본값 변경
  • 시퀀스 변경

⚠️ Autogenerate 한계와 주의사항​

1. 컬럼 이름 변경 문제​

# ❌ Autogenerate 결과 (잘못됨)
def upgrade():
    op.drop_column('users', 'username')
    op.add_column('users', sa.Column('user_name', sa.String(50)))

# ✅ 수동 수정 필요
def upgrade():
    op.alter_column('users', 'username',
        new_column_name='user_name')

def downgrade():
    op.alter_column('users', 'user_name',
        new_column_name='username')

2. 데이터 손실 위험​

# Autogenerate 생성 (위험!)
def upgrade():
    op.drop_column('users', 'old_field')  # 데이터 손실!

# ✅ 수동으로 백업 코드 추가
def upgrade():
    # 1. 백업 컬럼 생성
    op.add_column('users', sa.Column('old_field_backup', sa.String(100)))

    # 2. 데이터 복사
    op.execute('UPDATE users SET old_field_backup = old_field')

    # 3. 원본 삭제
    op.drop_column('users', 'old_field')

3. Check 제약조건​

# models.py
age = Column(Integer, CheckConstraint('age >= 18'))

# ❌ Autogenerate는 감지 못함
# ✅ 수동으로 추가 필요
def upgrade():
    op.add_column('users', sa.Column('age', sa.Integer()))
    op.create_check_constraint('check_age', 'users', 'age >= 18')

🛠️ 실전 팁​

1. 항상 검토하기​

# 1. Autogenerate 실행
alembic revision --autogenerate -m "update schema"

# 2. 생성된 파일 검토
cat alembic/versions/xxx_update_schema.py

# 3. 필요하면 수정

# 4. 테스트
alembic upgrade head
alembic downgrade -1
alembic upgrade head

2. 무시할 테이블 설정​

# alembic/env.py
def run_migrations_online():
    # 특정 테이블 무시
    connectable = engine_from_config(
        config.get_section(config.config_ini_section),
        prefix='sqlalchemy.',
        poolclass=pool.NullPool,
    )

    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            include_schemas=True,
            # 임시 테이블 무시
            include_object=lambda obj, name, type_, reflected, compare_to:
                not name.startswith('temp_')
        )

        with context.begin_transaction():
            context.run_migrations()

3. 타입 비교 커스터마이징​

# alembic/env.py
from alembic.ddl import postgresql

def run_migrations_online():
    context.configure(
        connection=connection,
        target_metadata=target_metadata,
        # 타입 비교 커스터마이징
        compare_type=True,
        compare_server_default=True,
    )

📊 완전한 예제: 블로그 시스템​

1. 초기 모델​

# models.py
class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    username = Column(String(50), unique=True)
    email = Column(String(100), unique=True)

class Post(Base):
    __tablename__ = 'posts'
    id = Column(Integer, primary_key=True)
    title = Column(String(200))
    content = Column(Text)
    author_id = Column(Integer, ForeignKey('users.id'))
alembic revision --autogenerate -m "initial schema"
alembic upgrade head

2. 기능 추가​

# models.py - 댓글 기능 추가
class Comment(Base):
    __tablename__ = 'comments'
    id = Column(Integer, primary_key=True)
    content = Column(Text, nullable=False)
    post_id = Column(Integer, ForeignKey('posts.id', ondelete='CASCADE'))
    author_id = Column(Integer, ForeignKey('users.id'))
    created_at = Column(DateTime, default=datetime.utcnow)
alembic revision --autogenerate -m "add comments"
alembic upgrade head

3. 프로필 기능​

# models.py - User 확장
class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    username = Column(String(50), unique=True)
    email = Column(String(100), unique=True)
    # 새 필드
    bio = Column(Text)
    avatar_url = Column(String(500))

✅ 실습 과제​

다음 SQLAlchemy 모델을 만들고 autogenerate로 마이그레이션을 생성하세요:

class Product(Base):
    __tablename__ = 'products'
    id = Column(Integer, primary_key=True)
    name = Column(String(100), nullable=False, index=True)
    description = Column(Text)
    price = Column(Numeric(10, 2), nullable=False)
    stock = Column(Integer, default=0)
    category_id = Column(Integer, ForeignKey('categories.id'))
    is_active = Column(Boolean, default=True)
    created_at = Column(DateTime, default=datetime.utcnow)

class Category(Base):
    __tablename__ = 'categories'
    id = Column(Integer, primary_key=True)
    name = Column(String(50), unique=True, nullable=False)
    slug = Column(String(50), unique=True, nullable=False)
정답 확인
# 1. 마이그레이션 생성
alembic revision --autogenerate -m "create products and categories"

# 2. 생성된 파일 확인
# - categories 테이블 먼저 생성 (products의 외래 키 대상)
# - products 테이블 생성
# - 인덱스 생성

# 3. 실행
alembic upgrade head

# 4. 확인
alembic current

📚 핵심 정리​

  • alembic revision --autogenerate: 자동 마이그레이션 생성
  • SQLAlchemy 모델 변경 사항 자동 감지
  • 항상 생성된 파일 검토 필수
  • 컬럼 이름 변경, Check 제약조건 등은 수동 수정 필요
