1. 메타클래스

메타클래스는 파이썬의 가장 고급 기능 중 하나로, "클래스를 만드는 클래스"입니다. 메타클래스를 이해하면 파이썬의 객체 지향 프로그래밍의 깊은 원리를 파악할 수 있으며, 프레임워크 수준의 코드를 작성할 수 있습니다.

1.1 메타클래스란 무엇인가?​

파이썬에서 모든 것은 객체입니다. 클래스도 예외가 아닙니다. 메타클래스는 클래스를 생성하는 "클래스의 클래스"입니다.

기본 개념​

# 일반적인 클래스와 인스턴스
class MyClass:
    pass

obj = MyClass()

# obj는 MyClass의 인스턴스입니다
print(type(obj))  # <class '__main__.MyClass'>

# MyClass는 type의 인스턴스입니다
print(type(MyClass))  # <class 'type'>

# type은 자기 자신의 인스턴스입니다
print(type(type))  # <class 'type'>
팁

파이썬의 객체 계층 구조:

  • 인스턴스 → 클래스로부터 생성됨
  • 클래스 → 메타클래스로부터 생성됨
  • 메타클래스 → type으로부터 생성됨

클래스 생성 과정​

# 이 두 가지 방법은 동일합니다

# 방법 1: 일반적인 클래스 정의
class Dog:
    def __init__(self, name):
        self.name = name

    def bark(self):
        return f"{self.name} says woof!"

# 방법 2: type()을 사용한 동적 클래스 생성
Dog = type('Dog',
           (object,),  # 상속받을 클래스들
           {
               '__init__': lambda self, name: setattr(self, 'name', name),
               'bark': lambda self: f"{self.name} says woof!"
           })

dog = Dog("Buddy")
print(dog.bark())  # Buddy says woof!

1.2 type()을 메타클래스로 사용하기​

type()은 두 가지 용도로 사용됩니다:

  1. 객체의 타입 확인
  2. 새로운 클래스 동적 생성

type()의 문법​

# type(name, bases, dict)
# name: 클래스 이름
# bases: 부모 클래스들의 튜플
# dict: 클래스 속성과 메서드를 담은 딕셔너리

# 간단한 예제
def init(self, x, y):
    self.x = x
    self.y = y

def get_sum(self):
    return self.x + self.y

# 동적으로 Point 클래스 생성
Point = type('Point',
             (object,),
             {
                 '__init__': init,
                 'get_sum': get_sum
             })

p = Point(3, 4)
print(p.get_sum())  # 7
print(type(p))      # <class '__main__.Point'>
print(type(Point))  # <class 'type'>

상속과 함께 사용하기​

# 베이스 클래스
class Animal:
    def speak(self):
        return "Some sound"

# type()을 사용해 Animal을 상속받는 클래스 생성
Cat = type('Cat',
           (Animal,),  # Animal을 상속
           {
               'speak': lambda self: "Meow!",
               'purr': lambda self: "Purr purr..."
           })

cat = Cat()
print(cat.speak())  # Meow!
print(cat.purr())   # Purr purr...
print(isinstance(cat, Animal))  # True

1.3 커스텀 메타클래스 생성하기​

메타클래스를 사용하면 클래스 생성 과정을 커스터마이징할 수 있습니다.

기본 메타클래스​

class MyMeta(type):
    """커스텀 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        """클래스 객체가 생성될 때 호출됨"""
        print(f"Creating class {name}")

        # 모든 메서드 이름을 대문자로 변경
        uppercase_attrs = {}
        for attr_name, attr_value in attrs.items():
            if not attr_name.startswith('__'):
                uppercase_attrs[attr_name.upper()] = attr_value
            else:
                uppercase_attrs[attr_name] = attr_value

        return super().__new__(mcs, name, bases, uppercase_attrs)

# 메타클래스 사용
class MyClass(metaclass=MyMeta):
    x = 100

    def hello(self):
        return "Hello"

# Creating class MyClass 출력됨

obj = MyClass()
print(obj.X)  # 100 (x가 X로 변경됨)
print(obj.HELLO())  # Hello (hello가 HELLO로 변경됨)
# print(obj.x)  # AttributeError

init 메서드와 함께 사용​

class ValidatedMeta(type):
    """클래스 생성 시 검증을 수행하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # 클래스 객체 생성
        cls = super().__new__(mcs, name, bases, attrs)
        return cls

    def __init__(cls, name, bases, attrs):
        """클래스 초기화 시 호출됨"""
        super().__init__(name, bases, attrs)

        # 'validate' 메서드가 있는지 확인
        if not hasattr(cls, 'validate'):
            raise TypeError(f"Class {name} must implement 'validate' method")

# 올바른 사용
class ValidUser(metaclass=ValidatedMeta):
    def validate(self):
        return True

# 에러 발생
try:
    class InvalidUser(metaclass=ValidatedMeta):
        pass
except TypeError as e:
    print(f"Error: {e}")
    # Error: Class InvalidUser must implement 'validate' method

1.4 new vs init in 메타클래스​

메타클래스에서 __new__와 __init__은 서로 다른 시점에 호출됩니다.

차이점 이해하기​

class DebugMeta(type):
    """__new__와 __init__의 차이를 보여주는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        print(f"__new__ called for {name}")
        print(f"  mcs: {mcs}")
        print(f"  bases: {bases}")
        print(f"  attrs keys: {list(attrs.keys())}")

        # 여기서 클래스 객체가 생성됨
        cls = super().__new__(mcs, name, bases, attrs)
        print(f"  cls created: {cls}")
        return cls

    def __init__(cls, name, bases, attrs):
        print(f"__init__ called for {name}")
        print(f"  cls: {cls}")

        # 여기서 클래스 객체를 초기화함
        super().__init__(name, bases, attrs)

        # 클래스에 추가 속성 설정 가능
        cls.created_by = "DebugMeta"

class TestClass(metaclass=DebugMeta):
    x = 10

# 출력:
# __new__ called for TestClass
#   mcs: <class '__main__.DebugMeta'>
#   bases: ()
#   attrs keys: ['__module__', '__qualname__', 'x']
#   cls created: <class '__main__.TestClass'>
# __init__ called for TestClass
#   cls: <class '__main__.TestClass'>

print(TestClass.created_by)  # DebugMeta

실제 활용 예제​

class AttributeDefaultMeta(type):
    """클래스 속성에 기본값을 자동 설정하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # __new__에서 속성 수정
        if '_defaults' in attrs:
            defaults = attrs['_defaults']
            for key, value in defaults.items():
                if key not in attrs:
                    attrs[key] = value

        return super().__new__(mcs, name, bases, attrs)

    def __init__(cls, name, bases, attrs):
        # __init__에서 추가 설정
        super().__init__(name, bases, attrs)
        cls._initialized = True

class Config(metaclass=AttributeDefaultMeta):
    _defaults = {
        'debug': False,
        'timeout': 30,
        'retries': 3
    }

    # debug만 오버라이드
    debug = True

print(Config.debug)      # True (오버라이드됨)
print(Config.timeout)    # 30 (기본값)
print(Config.retries)    # 3 (기본값)
print(Config._initialized)  # True
경고

new vs init 선택 가이드:

  • __new__: 클래스 객체를 생성하기 전에 속성을 수정하거나 클래스 생성을 막아야 할 때
  • __init__: 클래스 객체가 생성된 후 초기화나 검증을 수행할 때

1.5 메타클래스 사용 사례​

1.5.1 싱글톤 패턴​

class SingletonMeta(type):
    """싱글톤 패턴을 구현하는 메타클래스"""
    _instances = {}

    def __call__(cls, *args, **kwargs):
        """클래스가 호출될 때 (인스턴스 생성 시) 실행됨"""
        if cls not in cls._instances:
            # 인스턴스가 없으면 생성
            instance = super().__call__(*args, **kwargs)
            cls._instances[cls] = instance
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self, url):
        self.url = url
        print(f"Database connection created: {url}")

# 첫 번째 호출: 인스턴스 생성
db1 = Database("postgresql://localhost")
# 출력: Database connection created: postgresql://localhost

# 두 번째 호출: 기존 인스턴스 반환
db2 = Database("mysql://localhost")
# 아무것도 출력되지 않음

print(db1 is db2)  # True
print(db1.url)     # postgresql://localhost
print(db2.url)     # postgresql://localhost

1.5.2 클래스 레지스트리​

class RegistryMeta(type):
    """모든 서브클래스를 자동으로 등록하는 메타클래스"""
    _registry = {}

    def __new__(mcs, name, bases, attrs):
        cls = super().__new__(mcs, name, bases, attrs)

        # 베이스 클래스가 아닌 경우만 등록
        if bases:
            mcs._registry[name] = cls

        return cls

    @classmethod
    def get_registry(mcs):
        """등록된 모든 클래스 반환"""
        return mcs._registry.copy()

class Plugin(metaclass=RegistryMeta):
    """베이스 플러그인 클래스"""
    pass

class EmailPlugin(Plugin):
    def send(self, message):
        return f"Sending email: {message}"

class SMSPlugin(Plugin):
    def send(self, message):
        return f"Sending SMS: {message}"

class PushPlugin(Plugin):
    def send(self, message):
        return f"Sending push: {message}"

# 모든 플러그인 확인
print(RegistryMeta.get_registry())
# {'EmailPlugin': <class '__main__.EmailPlugin'>,
#  'SMSPlugin': <class '__main__.SMSPlugin'>,
#  'PushPlugin': <class '__main__.PushPlugin'>}

# 동적으로 플러그인 사용
def send_notification(plugin_name, message):
    registry = RegistryMeta.get_registry()
    if plugin_name in registry:
        plugin = registry[plugin_name]()
        return plugin.send(message)
    return "Plugin not found"

print(send_notification('EmailPlugin', 'Hello!'))
# Sending email: Hello!

1.5.3 자동 검증​

class ValidationMeta(type):
    """속성 타입 검증을 자동화하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # _validators 딕셔너리 찾기
        validators = attrs.get('_validators', {})

        # 원래 __init__ 저장
        original_init = attrs.get('__init__', lambda self: None)

        def new_init(self, *args, **kwargs):
            # 원래 __init__ 호출
            original_init(self, *args, **kwargs)

            # 검증 수행
            for attr_name, expected_type in validators.items():
                if hasattr(self, attr_name):
                    value = getattr(self, attr_name)
                    if not isinstance(value, expected_type):
                        raise TypeError(
                            f"{attr_name} must be {expected_type.__name__}, "
                            f"got {type(value).__name__}"
                        )

        attrs['__init__'] = new_init
        return super().__new__(mcs, name, bases, attrs)

class User(metaclass=ValidationMeta):
    _validators = {
        'name': str,
        'age': int,
        'email': str
    }

    def __init__(self, name, age, email):
        self.name = name
        self.age = age
        self.email = email

# 올바른 사용
user1 = User("Alice", 30, "alice@example.com")
print(f"{user1.name}, {user1.age}")  # Alice, 30

# 타입 에러
try:
    user2 = User("Bob", "thirty", "bob@example.com")
except TypeError as e:
    print(f"Error: {e}")
    # Error: age must be int, got str

1.5.4 ORM 스타일 모델​

class Field:
    """데이터베이스 필드를 나타내는 클래스"""
    def __init__(self, field_type, required=True, default=None):
        self.field_type = field_type
        self.required = required
        self.default = default

    def validate(self, value):
        if value is None:
            if self.required and self.default is None:
                raise ValueError("Required field cannot be None")
            return self.default

        if not isinstance(value, self.field_type):
            raise TypeError(
                f"Expected {self.field_type.__name__}, "
                f"got {type(value).__name__}"
            )
        return value

class ModelMeta(type):
    """ORM 스타일 모델을 위한 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # Field 속성들을 수집
        fields = {}
        for key, value in list(attrs.items()):
            if isinstance(value, Field):
                fields[key] = value
                # 클래스 정의에서 Field 제거
                attrs.pop(key)

        # 필드 정보 저장
        attrs['_fields'] = fields

        # __init__ 메서드 생성
        def __init__(self, **kwargs):
            for field_name, field in self._fields.items():
                value = kwargs.get(field_name)
                validated_value = field.validate(value)
                setattr(self, field_name, validated_value)

        attrs['__init__'] = __init__

        # __repr__ 메서드 생성
        def __repr__(self):
            field_strs = [
                f"{name}={getattr(self, name)!r}"
                for name in self._fields.keys()
            ]
            return f"{name}({', '.join(field_strs)})"

        attrs['__repr__'] = __repr__

        return super().__new__(mcs, name, bases, attrs)

class Person(metaclass=ModelMeta):
    name = Field(str)
    age = Field(int)
    email = Field(str, required=False, default="no-email@example.com")

# 사용 예제
person1 = Person(name="Alice", age=30, email="alice@example.com")
print(person1)
# Person(name='Alice', age=30, email='alice@example.com')

person2 = Person(name="Bob", age=25)
print(person2)
# Person(name='Bob', age=25, email='no-email@example.com')

# 검증 테스트
try:
    person3 = Person(name="Charlie", age="twenty")
except TypeError as e:
    print(f"Error: {e}")
    # Error: Expected int, got str

1.6 클래스 데코레이터 vs 메타클래스​

클래스 데코레이터와 메타클래스는 비슷한 목적을 달성할 수 있지만, 사용 방법과 시점이 다릅니다.

클래스 데코레이터​

def singleton(cls):
    """싱글톤 패턴을 구현하는 데코레이터"""
    instances = {}

    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]

    return get_instance

@singleton
class DatabaseConnection:
    def __init__(self, url):
        self.url = url
        print(f"Connected to {url}")

db1 = DatabaseConnection("localhost")  # Connected to localhost
db2 = DatabaseConnection("remote")
print(db1 is db2)  # True
print(db1.url)     # localhost

메타클래스 vs 데코레이터 비교​

# 1. 메타클래스 방식
class MetaLogger(type):
    def __new__(mcs, name, bases, attrs):
        for attr_name, attr_value in attrs.items():
            if callable(attr_value) and not attr_name.startswith('_'):
                attrs[attr_name] = mcs.log_wrapper(attr_value, attr_name)
        return super().__new__(mcs, name, bases, attrs)

    @staticmethod
    def log_wrapper(func, name):
        def wrapper(*args, **kwargs):
            print(f"Calling {name}")
            result = func(*args, **kwargs)
            print(f"{name} returned {result}")
            return result
        return wrapper

class Calculator1(metaclass=MetaLogger):
    def add(self, a, b):
        return a + b

    def multiply(self, a, b):
        return a * b

# 2. 데코레이터 방식
def log_methods(cls):
    for attr_name in dir(cls):
        attr_value = getattr(cls, attr_name)
        if callable(attr_value) and not attr_name.startswith('_'):
            def log_wrapper(func):
                def wrapper(*args, **kwargs):
                    print(f"Calling {func.__name__}")
                    result = func(*args, **kwargs)
                    print(f"{func.__name__} returned {result}")
                    return result
                return wrapper
            setattr(cls, attr_name, log_wrapper(attr_value))
    return cls

@log_methods
class Calculator2:
    def add(self, a, b):
        return a + b

    def multiply(self, a, b):
        return a * b

# 테스트
calc1 = Calculator1()
calc1.add(2, 3)
# Calling add
# add returned 5

calc2 = Calculator2()
calc2.multiply(4, 5)
# Calling multiply
# multiply returned 20
팁

언제 무엇을 사용할까?

클래스 데코레이터를 사용:

  • 간단한 클래스 수정이 필요할 때
  • 가독성이 중요할 때
  • 여러 클래스에 선택적으로 적용할 때

메타클래스를 사용:

  • 상속 계층 전체에 영향을 주고 싶을 때
  • 클래스 생성 과정 자체를 제어해야 할 때
  • 프레임워크나 라이브러리를 만들 때
  • 더 깊은 수준의 제어가 필요할 때

상속과의 차이​

# 데코레이터: 상속되지 않음
@log_methods
class Parent:
    def method(self):
        return "parent"

class Child(Parent):
    def method(self):
        return "child"

# Child.method는 로깅되지 않음
child = Child()
child.method()  # 로그 출력 없음

# 메타클래스: 상속됨
class ParentMeta(metaclass=MetaLogger):
    def method(self):
        return "parent"

class ChildMeta(ParentMeta):
    def method(self):
        return "child"

# ChildMeta.method도 로깅됨
child_meta = ChildMeta()
child_meta.method()
# Calling method
# method returned child

1.7 추상 베이스 클래스 (ABC)​

ABC는 메타클래스를 활용한 파이썬 표준 라이브러리의 좋은 예입니다.

기본 ABC 사용법​

from abc import ABC, abstractmethod

class Shape(ABC):
    """추상 베이스 클래스"""

    @abstractmethod
    def area(self):
        """면적 계산 (반드시 구현해야 함)"""
        pass

    @abstractmethod
    def perimeter(self):
        """둘레 계산 (반드시 구현해야 함)"""
        pass

    def description(self):
        """일반 메서드 (선택적)"""
        return f"This is a shape with area {self.area()}"

class DatabaseField:
    """데이터베이스 필드 기본 클래스"""
    def __init__(self, primary_key=False, nullable=True, default=None):
        self.primary_key = primary_key
        self.nullable = nullable
        self.default = default
        self.name = None  # 메타클래스에서 설정됨

    def validate(self, value):
        if value is None:
            if not self.nullable and self.default is None:
                raise ValueError(f"{self.name} cannot be None")
            return self.default
        return self.to_python(value)

    def to_python(self, value):
        """서브클래스에서 구현"""
        return value

class IntegerField(DatabaseField):
    def to_python(self, value):
        try:
            return int(value)
        except (TypeError, ValueError):
            raise ValueError(f"{self.name} must be an integer")

class CharField(DatabaseField):
    def __init__(self, max_length=255, **kwargs):
        super().__init__(**kwargs)
        self.max_length = max_length

    def to_python(self, value):
        value = str(value)
        if len(value) > self.max_length:
            raise ValueError(
                f"{self.name} exceeds max length {self.max_length}"
            )
        return value

class ModelMeta(type):
    """ORM 모델을 위한 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # 베이스 Model 클래스는 건너뛰기
        if name == 'Model':
            return super().__new__(mcs, name, bases, attrs)

        # 필드 수집
        fields = {}
        for key, value in list(attrs.items()):
            if isinstance(value, DatabaseField):
                value.name = key
                fields[key] = value
                attrs.pop(key)

        attrs['_fields'] = fields
        attrs['_table_name'] = name.lower()

        # __init__ 생성
        original_init = attrs.get('__init__')

        def __init__(self, **kwargs):
            # 필드 초기화
            for field_name, field in self._fields.items():
                value = kwargs.get(field_name)
                validated_value = field.validate(value)
                setattr(self, field_name, validated_value)

            # 원래 __init__ 호출
            if original_init:
                original_init(self, **kwargs)

        attrs['__init__'] = __init__

        # save 메서드
        def save(self):
            field_names = list(self._fields.keys())
            values = [getattr(self, name) for name in field_names]

            # SQL 생성 (실제로는 데이터베이스에 저장)
            sql = f"INSERT INTO {self._table_name} ({', '.join(field_names)}) "
            sql += f"VALUES ({', '.join(map(repr, values))})"
            print(f"Executing: {sql}")
            return True

        attrs['save'] = save

        # __repr__ 생성
        def __repr__(self):
            field_strs = [
                f"{name}={getattr(self, name)!r}"
                for name in self._fields.keys()
            ]
            return f"{name}({', '.join(field_strs)})"

        attrs['__repr__'] = __repr__

        return super().__new__(mcs, name, bases, attrs)

class Model(metaclass=ModelMeta):
    """ORM 모델 베이스 클래스"""
    pass

# 사용 예제
class User(Model):

    username = CharField(max_length=50, nullable=False)
    email = CharField(max_length=100)
    age = IntegerField(nullable=True, default=0)

# 인스턴스 생성

# User(id=1, username='alice', email='alice@example.com', age=30)

1.8.2 플러그인 시스템​

class PluginMeta(type):
    """플러그인 자동 등록 시스템"""
    plugins = {}

    def __new__(mcs, name, bases, attrs):
        cls = super().__new__(mcs, name, bases, attrs)

        # BasePlugin이 아닌 서브클래스만 등록
        if bases and bases[0].__name__ == 'BasePlugin':
            # plugin_name 속성 확인
            plugin_name = attrs.get('plugin_name')
            if not plugin_name:
                raise ValueError(
                    f"Plugin {name} must define 'plugin_name' attribute"
                )

            # 중복 확인
            if plugin_name in mcs.plugins:
                raise ValueError(
                    f"Plugin name '{plugin_name}' is already registered"
                )

            mcs.plugins[plugin_name] = cls
            print(f"Registered plugin: {plugin_name}")

        return cls

    @classmethod
    def get_plugin(mcs, plugin_name):
        """플러그인 가져오기"""
        if plugin_name not in mcs.plugins:
            raise KeyError(f"Plugin '{plugin_name}' not found")
        return mcs.plugins[plugin_name]

    @classmethod
    def list_plugins(mcs):
        """모든 플러그인 목록"""
        return list(mcs.plugins.keys())

class BasePlugin(metaclass=PluginMeta):
    """플러그인 베이스 클래스"""
    plugin_name = None

    def execute(self, *args, **kwargs):
        raise NotImplementedError("Plugins must implement 'execute' method")

# 플러그인 정의
class EmailNotifier(BasePlugin):
    plugin_name = "email"

    def execute(self, recipient, message):
        return f"Sending email to {recipient}: {message}"

class SMSNotifier(BasePlugin):
    plugin_name = "sms"

    def execute(self, phone, message):
        return f"Sending SMS to {phone}: {message}"

class SlackNotifier(BasePlugin):
    plugin_name = "slack"

    def execute(self, channel, message):
        return f"Posting to #{channel}: {message}"

# 플러그인 시스템 사용
print("\nAvailable plugins:", PluginMeta.list_plugins())
# Available plugins: ['email', 'sms', 'slack']

def send_notification(plugin_name, *args, **kwargs):
    """동적으로 플러그인 실행"""
    plugin_class = PluginMeta.get_plugin(plugin_name)
    plugin = plugin_class()
    return plugin.execute(*args, **kwargs)

# 테스트

1.8.3 싱글톤 패턴 (스레드 안전)​

import threading

class ThreadSafeSingletonMeta(type):
    """스레드 안전한 싱글톤 메타클래스"""
    _instances = {}
    _lock = threading.Lock()

    def __call__(cls, *args, **kwargs):
        # 더블 체크 락킹 패턴
        if cls not in cls._instances:
            with cls._lock:
                if cls not in cls._instances:
                    instance = super().__call__(*args, **kwargs)
                    cls._instances[cls] = instance
        return cls._instances[cls]

class Configuration(metaclass=ThreadSafeSingletonMeta):
    """애플리케이션 설정 싱글톤"""

    def __init__(self):
        if not hasattr(self, 'initialized'):
            self.settings = {}
            self.initialized = True
            print("Configuration initialized")

    def set(self, key, value):
        self.settings[key] = value

    def get(self, key, default=None):
        return self.settings.get(key, default)

# 멀티스레드 테스트
def worker(worker_id):
    config = Configuration()
    config.set(f'worker_{worker_id}', worker_id)
    print(f"Worker {worker_id}: {id(config)}")

threads = []
for i in range(5):
    t = threading.Thread(target=worker, args=(i,))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

# 모든 워커가 같은 인스턴스를 사용했는지 확인
config = Configuration()
print(f"\nFinal config ID: {id(config)}")
print(f"Settings: {config.settings}")

class AutoPropertyMeta(type):
    """속성을 자동으로 프로퍼티로 변환"""

    def __new__(mcs, name, bases, attrs):
        new_attrs = {}

        for attr_name, attr_value in attrs.items():
            if attr_name.startswith('_') or callable(attr_value):
                new_attrs[attr_name] = attr_value
            else:
                # 속성을 프로퍼티로 변환
                private_name = f'_{attr_name}'

                def make_getter(pname):
                    def getter(self):
                        return getattr(self, pname, None)
                    return getter

                def make_setter(pname):
                    def setter(self, value):
                        setattr(self, pname, value)
                    return setter

                new_attrs[private_name] = attr_value
                new_attrs[attr_name] = property(
                    make_getter(private_name),
                    make_setter(private_name)
                )

        return super().__new__(mcs, name, bases, new_attrs)

class Product(metaclass=AutoPropertyMeta):
    name = ""
    price = 0.0

    def __init__(self, name, price):
        self.name = name
        self.price = price

product = Product("Laptop", 999.99)
print(product.name)    # Laptop
product.price = 899.99
print(product.price)   # 899.99

class DocumentedMeta(type):
    """
    자동 문서화 메타클래스

    이 메타클래스는 클래스 생성 시 다음을 수행합니다:
    1. 모든 public 메서드의 docstring 확인
    2. docstring이 없는 메서드에 경고 추가
    3. 클래스에 _undocumented_methods 속성 추가

    Usage:
        class MyClass(metaclass=DocumentedMeta):
            def my_method(self):
                '''메서드 설명'''
                pass
    """

    def __new__(mcs, name, bases, attrs):
        undocumented = []

        for attr_name, attr_value in attrs.items():
            if (callable(attr_value) and
                not attr_name.startswith('_') and
                not attr_value.__doc__):
                undocumented.append(attr_name)

        cls = super().__new__(mcs, name, bases, attrs)
        cls._undocumented_methods = undocumented

        if undocumented:
            print(f"Warning: {name} has undocumented methods: {undocumented}")

        return cls

class InterfaceMeta(type):
    """명시적 인터페이스 정의를 강제하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # _required_methods 체크
        required_methods = attrs.get('_required_methods', [])

        if required_methods:
            for method_name in required_methods:
                if method_name not in attrs:
                    raise TypeError(
                        f"{name} must implement {method_name} method"
                    )

                if not callable(attrs[method_name]):
                    raise TypeError(
                        f"{method_name} must be callable"
                    )

        return super().__new__(mcs, name, bases, attrs)

class APIClient(metaclass=InterfaceMeta):

import time

class CachedMeta(type):
    """메서드 결과를 캐싱하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # _cached_methods 리스트에 있는 메서드만 캐싱
        cached_methods = attrs.get('_cached_methods', [])

        for method_name in cached_methods:
            if method_name in attrs:
                original_method = attrs[method_name]
                attrs[method_name] = mcs._make_cached(original_method)

        return super().__new__(mcs, name, bases, attrs)

    @staticmethod
    def _make_cached(method):
        cache = {}

        def cached_method(self, *args):
            if args not in cache:
                cache[args] = method(self, *args)
            return cache[args]

        cached_method.__name__ = method.__name__
        cached_method.__doc__ = method.__doc__
        return cached_method

class ExpensiveCalculator(metaclass=CachedMeta):
    _cached_methods = ['fibonacci', 'factorial']

    def fibonacci(self, n):
        """피보나치 수 계산 (캐싱됨)"""
        if n <= 1:
            return n
        return self.fibonacci(n-1) + self.fibonacci(n-2)

    def factorial(self, n):
        """팩토리얼 계산 (캐싱됨)"""
        if n <= 1:
            return 1
        return n * self.factorial(n-1)

calc = ExpensiveCalculator()

# 첫 번째 호출: 느림
start = time.time()
result1 = calc.fibonacci(30)
print(f"First call: {time.time() - start:.4f}s")

# 두 번째 호출: 빠름 (캐시됨)
start = time.time()
result2 = calc.fibonacci(30)
print(f"Second call: {time.time() - start:.6f}s")

    def __new__(mcs, name, bases, attrs):
        print(f"\n=== Creating class: {name} ===")
        print(f"Metaclass: {mcs}")
        print(f"Bases: {bases}")
        print(f"Attributes: {list(attrs.keys())}")

        # 클래스 생성
        cls = super().__new__(mcs, name, bases, attrs)

        print(f"Class created: {cls}")
        print(f"MRO: {cls.__mro__}")

        return cls

    def __init__(cls, name, bases, attrs):
        print(f"\n=== Initializing class: {name} ===")
        super().__init__(name, bases, attrs)
        print(f"Class initialized: {cls}")

    def __call__(cls, *args, **kwargs):
        print(f"\n=== Creating instance of {cls.__name__} ===")
        print(f"Args: {args}")
        print(f"Kwargs: {kwargs}")

        instance = super().__call__(*args, **kwargs)

        print(f"Instance created: {instance}")
        return instance

class DebugClass(metaclass=DebugMeta):
    def __init__(self, value):
        self.value = value

# 디버그 출력과 함께 실행
obj = DebugClass(42)

연습 1: 자동 속성 추적​

자동으로 모든 속성 변경을 추적하는 메타클래스를 만드세요.

해답 보기
class AttributeTrackerMeta(type):
    """모든 속성 변경을 추적하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # __setattr__ 래핑
        original_setattr = attrs.get('__setattr__')

        def tracked_setattr(self, attr_name, value):
            # 변경 이력 초기화
            if not hasattr(self, '_attribute_history'):
                object.__setattr__(self, '_attribute_history', {})

            # 이전 값 저장
            old_value = getattr(self, attr_name, None)

            # 실제 설정
            if original_setattr:
                original_setattr(self, attr_name, value)
            else:
                object.__setattr__(self, attr_name, value)

            # 이력 기록
            if attr_name != '_attribute_history':
                if attr_name not in self._attribute_history:
                    self._attribute_history[attr_name] = []

                self._attribute_history[attr_name].append({
                    'old': old_value,
                    'new': value
                })

        attrs['__setattr__'] = tracked_setattr

        # 이력 조회 메서드
        def get_history(self, attr_name=None):
            if attr_name:
                return self._attribute_history.get(attr_name, [])
            return self._attribute_history

        attrs['get_history'] = get_history

        return super().__new__(mcs, name, bases, attrs)

class Person(metaclass=AttributeTrackerMeta):
    def __init__(self, name, age):
        self.name = name
        self.age = age

# 테스트
person = Person("Alice", 25)
person.age = 26
person.age = 27
person.name = "Alice Smith"

print(person.get_history('age'))
# [{'old': None, 'new': 25}, {'old': 25, 'new': 26}, {'old': 26, 'new': 27}]

print(person.get_history('name'))
# [{'old': None, 'new': 'Alice'}, {'old': 'Alice', 'new': 'Alice Smith'}]

연습 2: 읽기 전용 클래스​

인스턴스 생성 후 속성을 변경할 수 없는 읽기 전용 클래스를 만드는 메타클래스를 작성하세요.

해답 보기
class ImmutableMeta(type):
    """불변 클래스를 만드는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        # __setattr__ 오버라이드
        def __setattr__(self, name, value):
            if hasattr(self, '_initialized'):
                raise AttributeError(
                    f"Cannot modify immutable instance. "
                    f"Attempted to set {name} = {value}"
                )
            object.__setattr__(self, name, value)

        attrs['__setattr__'] = __setattr__

        # __delattr__ 오버라이드
        def __delattr__(self, name):
            raise AttributeError(
                f"Cannot delete attribute from immutable instance"
            )

        attrs['__delattr__'] = __delattr__

        # __init__ 래핑
        original_init = attrs.get('__init__')

        def __init__(self, *args, **kwargs):
            if original_init:
                original_init(self, *args, **kwargs)
            object.__setattr__(self, '_initialized', True)

        attrs['__init__'] = __init__

        return super().__new__(mcs, name, bases, attrs)

class Point(metaclass=ImmutableMeta):
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __repr__(self):
        return f"Point({self.x}, {self.y})"

# 테스트
point = Point(3, 4)
print(point)  # Point(3, 4)

try:
    point.x = 5
except AttributeError as e:
    print(f"Error: {e}")
    # Error: Cannot modify immutable instance. Attempted to set x = 5

try:
    del point.y
except AttributeError as e:
    print(f"Error: {e}")
    # Error: Cannot delete attribute from immutable instance

연습 3: 메서드 실행 시간 측정​

클래스의 모든 메서드 실행 시간을 자동으로 측정하는 메타클래스를 만드세요.

해답 보기
import time
import functools

class TimingMeta(type):
    """메서드 실행 시간을 측정하는 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        for attr_name, attr_value in attrs.items():
            if callable(attr_value) and not attr_name.startswith('_'):
                attrs[attr_name] = mcs._time_method(attr_value)

        cls = super().__new__(mcs, name, bases, attrs)
        cls._timing_stats = {}

        return cls

    @staticmethod
    def _time_method(method):
        @functools.wraps(method)
        def timed_method(self, *args, **kwargs):
            start_time = time.time()
            result = method(self, *args, **kwargs)
            end_time = time.time()

            execution_time = end_time - start_time
            method_name = method.__name__

            # 통계 업데이트
            if method_name not in self._timing_stats:
                self._timing_stats[method_name] = {
                    'count': 0,
                    'total_time': 0,
                    'min_time': float('inf'),
                    'max_time': 0
                }

            stats = self._timing_stats[method_name]
            stats['count'] += 1
            stats['total_time'] += execution_time
            stats['min_time'] = min(stats['min_time'], execution_time)
            stats['max_time'] = max(stats['max_time'], execution_time)

            return result

        return timed_method

class DataProcessor(metaclass=TimingMeta):
    def process_data(self, data):
        """데이터 처리"""
        time.sleep(0.1)  # 시뮬레이션
        return [x * 2 for x in data]

    def analyze_data(self, data):
        """데이터 분석"""
        time.sleep(0.05)  # 시뮬레이션
        return sum(data) / len(data)

    def get_timing_report(self):
        """타이밍 리포트 생성"""
        report = []
        for method_name, stats in self._timing_stats.items():
            avg_time = stats['total_time'] / stats['count']
            report.append(
                f"{method_name}:\n"
                f"  Calls: {stats['count']}\n"
                f"  Total: {stats['total_time']:.4f}s\n"
                f"  Average: {avg_time:.4f}s\n"
                f"  Min: {stats['min_time']:.4f}s\n"
                f"  Max: {stats['max_time']:.4f}s"
            )
        return '\n\n'.join(report)

# 테스트
processor = DataProcessor()

for _ in range(3):
    processor.process_data([1, 2, 3, 4, 5])

for _ in range(5):
    processor.analyze_data([1, 2, 3, 4, 5])

print(processor.get_timing_report())

연습 4: 타입 체크 ORM​

간단한 타입 체크 기능이 있는 ORM 스타일 모델을 만드세요.

해답 보기
class TypedField:
    """타입 체크를 수행하는 필드"""

    def __init__(self, field_type, required=True, default=None, validator=None):
        self.field_type = field_type
        self.required = required
        self.default = default
        self.validator = validator
        self.name = None

    def validate(self, value):
        # None 체크
        if value is None:
            if self.required and self.default is None:
                raise ValueError(f"{self.name} is required")
            return self.default

        # 타입 체크
        if not isinstance(value, self.field_type):
            raise TypeError(
                f"{self.name} must be {self.field_type.__name__}, "
                f"got {type(value).__name__}"
            )

        # 커스텀 검증
        if self.validator:
            if not self.validator(value):
                raise ValueError(f"{self.name} validation failed")

        return value

class ModelMeta(type):
    """타입 체크 ORM 메타클래스"""

    def __new__(mcs, name, bases, attrs):
        if name == 'Model':
            return super().__new__(mcs, name, bases, attrs)

        # 필드 수집
        fields = {}
        for key, value in list(attrs.items()):
            if isinstance(value, TypedField):
                value.name = key
                fields[key] = value
                attrs.pop(key)

        attrs['_fields'] = fields

        # __init__ 생성
        def __init__(self, **kwargs):
            for field_name, field in self._fields.items():
                value = kwargs.get(field_name)
                validated_value = field.validate(value)
                object.__setattr__(self, field_name, validated_value)

            object.__setattr__(self, '_initialized', True)

        attrs['__init__'] = __init__

        # __setattr__ 생성 (타입 체크 유지)
        def __setattr__(self, name, value):
            if hasattr(self, '_initialized') and name in self._fields:
                validated_value = self._fields[name].validate(value)
                object.__setattr__(self, name, validated_value)
            else:
                object.__setattr__(self, name, value)

        attrs['__setattr__'] = __setattr__

        # to_dict 메서드
        def to_dict(self):
            return {
                name: getattr(self, name)
                for name in self._fields.keys()
            }

        attrs['to_dict'] = to_dict

        return super().__new__(mcs, name, bases, attrs)

class Model(metaclass=ModelMeta):
    """ORM 베이스 모델"""
    pass

# 사용 예제
class User(Model):
    name = TypedField(str, required=True)
    age = TypedField(int, required=True, validator=lambda x: 0 <= x <= 150)
    email = TypedField(str, required=False, default="")
    score = TypedField(float, required=False, default=0.0)

# 정상 사용
user = User(name="Alice", age=30, email="alice@example.com", score=95.5)
print(user.to_dict())
# {'name': 'Alice', 'age': 30, 'email': 'alice@example.com', 'score': 95.5}

# 수정도 타입 체크됨
user.age = 31
print(user.age)  # 31

# 타입 에러
try:
    user.age = "thirty"
except TypeError as e:
    print(f"Error: {e}")
    # Error: age must be int, got str

# 검증 에러
try:
    user.age = 200
except ValueError as e:
    print(f"Error: {e}")
    # Error: age validation failed

# 필수 필드 누락
try:
    invalid_user = User(age=25)
except ValueError as e:
    print(f"Error: {e}")
    # Error: name is required

1.13 핵심 정리​

메타클래스의 핵심 개념​

  1. 메타클래스는 클래스의 클래스

    • 클래스는 type의 인스턴스
    • type은 기본 메타클래스
    • 커스텀 메타클래스는 type을 상속

  2. __new__ vs __init__

    • __new__: 클래스 객체 생성 (수정 가능)
    • __init__: 클래스 객체 초기화
    • __call__: 인스턴스 생성 제어

  3. 주요 사용 사례

    • 싱글톤 패턴
    • 클래스 레지스트리
    • 자동 검증
    • ORM 시스템
    • 플러그인 시스템

  4. 대안 고려

    • 클래스 데코레이터: 간단한 수정
    • 추상 베이스 클래스: 인터페이스 정의
    • 믹스인: 기능 재사용
    • 디스크립터: 속성 제어
