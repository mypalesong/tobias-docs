3. 컨텍스트 매니저

컨텍스트 매니저는 리소스를 안전하게 관리하고 설정 및 정리 작업을 자동화하는 Python의 강력한 기능입니다. with 문과 함께 사용되어 예외가 발생하더라도 리소스가 올바르게 해제되도록 보장합니다.

3.1 컨텍스트 매니저란?​

컨텍스트 매니저는 with 문과 함께 사용되는 객체로, 리소스 할당과 해제를 자동으로 처리합니다.

기본 개념​

# 전통적인 방식 - 에러가 발생하면 파일이 닫히지 않을 수 있음
file = open('example.txt', 'w')
try:
    file.write('Hello, World!')
finally:
    file.close()

# 컨텍스트 매니저 사용 - 자동으로 파일 닫힘 보장
with open('example.txt', 'w') as file:
    file.write('Hello, World!')
# 블록을 벗어나면 자동으로 file.close() 호출

컨텍스트 매니저의 장점​

# 1. 자동 리소스 관리
with open('data.txt', 'r') as f:
    data = f.read()
# 파일이 자동으로 닫힘

# 2. 예외 안전성
with open('data.txt', 'w') as f:
    f.write('data')
    raise ValueError("Something went wrong!")
    # 예외가 발생해도 파일은 올바르게 닫힘

# 3. 코드 가독성 향상
# 설정과 정리 로직이 명확하게 구분됨
팁

컨텍스트 매니저는 파일, 데이터베이스 연결, 네트워크 소켓, 락(lock) 등 사용 후 반드시 해제해야 하는 모든 리소스에 사용할 수 있습니다.

3.2 __enter__와 __exit__ 메서드​

컨텍스트 매니저 프로토콜을 구현하려면 __enter__와 __exit__ 메서드를 정의해야 합니다.

기본 구조​

class MyContextManager:
    def __enter__(self):
        """
        with 블록 진입 시 호출됨
        반환값이 'as' 절의 변수에 할당됨
        """
        print("Entering context")
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        """
        with 블록 종료 시 호출됨 (예외 발생 여부와 무관)

        매개변수:
        - exc_type: 예외 타입 (예외가 없으면 None)
        - exc_value: 예외 값 (예외가 없으면 None)
        - traceback: 트레이스백 객체 (예외가 없으면 None)

        반환값:
        - True: 예외를 억제 (예외가 전파되지 않음)
        - False/None: 예외를 전파
        """
        print("Exiting context")
        return False  # 예외를 전파

# 사용 예시
with MyContextManager() as manager:
    print("Inside context")
# 출력:
# Entering context
# Inside context
# Exiting context

실용적인 예제: 데이터베이스 연결 관리​

import sqlite3

class DatabaseConnection:
    def __init__(self, db_name):
        self.db_name = db_name
        self.connection = None

    def __enter__(self):
        self.connection = sqlite3.connect(self.db_name)
        print(f"Connected to {self.db_name}")
        return self.connection

    def __exit__(self, exc_type, exc_value, traceback):
        if exc_type is None:
            # 예외가 없으면 커밋
            self.connection.commit()
            print("Changes committed")
        else:
            # 예외가 있으면 롤백
            self.connection.rollback()
            print(f"Changes rolled back due to {exc_type.__name__}")

        self.connection.close()
        print("Connection closed")
        return False  # 예외를 전파

# 사용
with DatabaseConnection('example.db') as conn:
    cursor = conn.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS users (id INTEGER, name TEXT)')
    cursor.execute('INSERT INTO users VALUES (1, "Alice")')
# 자동으로 커밋되고 연결이 닫힘

예외 처리 상세 예제​

class ExceptionHandler:
    def __enter__(self):
        print("Setting up")
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        if exc_type is None:
            print("No exception occurred")
        else:
            print(f"Exception caught: {exc_type.__name__}: {exc_value}")
            print(f"Traceback: {traceback}")
            # True를 반환하면 예외를 억제
            return True  # 예외를 삼킴
        return False

# 예외가 발생하지만 억제됨
with ExceptionHandler() as handler:
    print("Doing something")
    raise ValueError("Oops!")
    print("This won't be printed")

print("Program continues")
# 출력:
# Setting up
# Doing something
# Exception caught: ValueError: Oops!
# Traceback: <traceback object at ...>
# Program continues

3.3 리소스 관리​

컨텍스트 매니저는 다양한 리소스를 안전하게 관리하는 데 사용됩니다.

파일 리소스 관리​

class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None

    def __enter__(self):
        self.file = open(self.filename, self.mode)
        return self.file

    def __exit__(self, exc_type, exc_value, traceback):
        if self.file:
            self.file.close()
        return False

# 사용
with FileManager('test.txt', 'w') as f:
    f.write('Hello from custom file manager!')

네트워크 리소스 관리​

import socket

class SocketConnection:
    def __init__(self, host, port):
        self.host = host
        self.port = port
        self.socket = None

    def __enter__(self):
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socket.connect((self.host, self.port))
        print(f"Connected to {self.host}:{self.port}")
        return self.socket

    def __exit__(self, exc_type, exc_value, traceback):
        if self.socket:
            self.socket.close()
            print("Socket closed")
        return False

락(Lock) 관리​

import threading
import time

class LockManager:
    def __init__(self, lock):
        self.lock = lock

    def __enter__(self):
        self.lock.acquire()
        print(f"Lock acquired by {threading.current_thread().name}")
        return self.lock

    def __exit__(self, exc_type, exc_value, traceback):
        self.lock.release()
        print(f"Lock released by {threading.current_thread().name}")
        return False

# 사용
lock = threading.Lock()
shared_resource = []

def worker(worker_id):
    with LockManager(lock):
        print(f"Worker {worker_id} accessing shared resource")
        shared_resource.append(worker_id)
        time.sleep(0.1)

threads = [threading.Thread(target=worker, args=(i,)) for i in range(3)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"Shared resource: {shared_resource}")

3.4 contextlib 모듈​

contextlib 모듈은 컨텍스트 매니저를 쉽게 생성할 수 있는 유틸리티를 제공합니다.

@contextmanager 데코레이터​

from contextlib import contextmanager

@contextmanager
def file_manager(filename, mode):
    """간단한 파일 매니저"""
    file = open(filename, mode)
    try:
        yield file  # __enter__가 반환하는 값
    finally:
        file.close()  # __exit__에서 실행되는 코드

# 사용
with file_manager('test.txt', 'w') as f:
    f.write('Using @contextmanager decorator!')

yield 전후 코드​

@contextmanager
def timing_context(name):
    """코드 실행 시간 측정"""
    import time
    start = time.time()
    print(f"{name} started")

    try:
        yield  # with 블록의 코드가 실행되는 지점
    finally:
        end = time.time()
        print(f"{name} finished in {end - start:.4f} seconds")

# 사용
with timing_context("Data processing"):
    import time
    time.sleep(0.5)
    print("Processing data...")
# 출력:
# Data processing started
# Processing data...
# Data processing finished in 0.5001 seconds

예외 처리​

@contextmanager
def error_handler(operation_name):
    """예외를 잡아서 로깅하는 컨텍스트 매니저"""
    try:
        yield
    except Exception as e:
        print(f"Error in {operation_name}: {type(e).__name__}: {e}")
        # 예외를 다시 발생시키려면 raise 사용
        # 예외를 억제하려면 아무것도 하지 않음

# 사용
with error_handler("database operation"):
    raise ValueError("Invalid data")
# 출력: Error in database operation: ValueError: Invalid data

실용적인 예제들​

from contextlib import contextmanager
import sys
import os

@contextmanager
def temporary_directory_change(path):
    """일시적으로 작업 디렉토리 변경"""
    original_dir = os.getcwd()
    try:
        os.chdir(path)
        yield path
    finally:
        os.chdir(original_dir)

@contextmanager
def suppress_stdout():
    """표준 출력 억제"""
    original_stdout = sys.stdout
    try:
        sys.stdout = open(os.devnull, 'w')
        yield
    finally:
        sys.stdout.close()
        sys.stdout = original_stdout

@contextmanager
def environment_variable(key, value):
    """일시적으로 환경 변수 설정"""
    original_value = os.environ.get(key)
    os.environ[key] = value
    try:
        yield
    finally:
        if original_value is None:
            del os.environ[key]
        else:
            os.environ[key] = original_value

# 사용 예시
with temporary_directory_change('/tmp'):
    print(f"Current directory: {os.getcwd()}")

with suppress_stdout():
    print("This won't be printed")
print("This will be printed")

with environment_variable('MY_VAR', 'test_value'):
    print(f"MY_VAR: {os.environ['MY_VAR']}")

3.5 ExitStack - 여러 컨텍스트 관리​

ExitStack은 여러 컨텍스트 매니저를 동적으로 관리할 수 있게 해줍니다.

기본 사용법​

from contextlib import ExitStack

# 여러 파일을 동시에 열기
filenames = ['file1.txt', 'file2.txt', 'file3.txt']

with ExitStack() as stack:
    files = [stack.enter_context(open(fname, 'w')) for fname in filenames]

    for i, f in enumerate(files):
        f.write(f'Content for file {i+1}\n')
# 모든 파일이 자동으로 닫힘

동적 리소스 관리​

from contextlib import ExitStack

def process_files(filenames):
    """동적으로 여러 파일 처리"""
    with ExitStack() as stack:
        files = []
        for filename in filenames:
            try:
                f = stack.enter_context(open(filename, 'r'))
                files.append(f)
            except FileNotFoundError:
                print(f"Warning: {filename} not found, skipping")

        # 열린 모든 파일 처리
        for f in files:
            print(f"Processing {f.name}")
            content = f.read()
            print(f"  Lines: {len(content.splitlines())}")
    # 모든 파일이 자동으로 닫힘

# 사용
filenames = ['test.txt', 'missing.txt', 'another.txt']
process_files(filenames)

조건부 리소스 관리​

from contextlib import ExitStack
import tempfile

def process_with_optional_logging(data, enable_logging=False):
    """로깅을 선택적으로 활성화하여 데이터 처리"""
    with ExitStack() as stack:
        if enable_logging:
            log_file = stack.enter_context(
                open('processing.log', 'a')
            )
            log = lambda msg: log_file.write(f"{msg}\n")
        else:
            log = lambda msg: None

        log("Processing started")
        result = sum(data)
        log(f"Processing completed: result = {result}")

        return result

# 사용
result1 = process_with_optional_logging([1, 2, 3], enable_logging=True)
result2 = process_with_optional_logging([4, 5, 6], enable_logging=False)

콜백 등록​

from contextlib import ExitStack

def complex_operation():
    """여러 정리 작업이 필요한 복잡한 작업"""
    with ExitStack() as stack:
        # 정리 콜백 등록
        stack.callback(print, "Cleanup step 1")
        stack.callback(print, "Cleanup step 2")
        stack.callback(print, "Cleanup step 3")

        print("Performing operation")

        # 조건부 콜백
        if True:
            stack.callback(lambda: print("Conditional cleanup"))

        return "Done"
# 종료 시 콜백이 역순으로 실행됨
# 출력:
# Performing operation
# Conditional cleanup
# Cleanup step 3
# Cleanup step 2
# Cleanup step 1

result = complex_operation()

중첩된 ExitStack​

from contextlib import ExitStack
import tempfile

def nested_resource_management():
    """중첩된 리소스 관리"""
    with ExitStack() as outer_stack:
        # 임시 디렉토리 생성
        temp_dir = outer_stack.enter_context(
            tempfile.TemporaryDirectory()
        )
        print(f"Temporary directory: {temp_dir}")

        # 내부 스택으로 파일들 관리
        with ExitStack() as inner_stack:
            files = []
            for i in range(3):
                f = inner_stack.enter_context(
                    open(f'{temp_dir}/file{i}.txt', 'w')
                )
                files.append(f)
                f.write(f'Content {i}')

            print(f"Created {len(files)} files")
        # 내부 스택 종료 시 파일들 닫힘

        # 외부 스택의 다른 작업
        outer_stack.callback(print, "Outer stack cleanup")
    # 외부 스택 종료 시 임시 디렉토리 삭제

nested_resource_management()

3.6 중첩된 컨텍스트 매니저​

여러 컨텍스트 매니저를 중첩하여 사용할 수 있습니다.

기본 중첩​

# 방법 1: 여러 줄로 중첩
with open('input.txt', 'r') as infile:
    with open('output.txt', 'w') as outfile:
        content = infile.read()
        outfile.write(content.upper())

# 방법 2: 한 줄로 작성 (Python 3.1+)
with open('input.txt', 'r') as infile, open('output.txt', 'w') as outfile:
    content = infile.read()
    outfile.write(content.upper())

복잡한 중첩 예제​

import threading
from contextlib import contextmanager
import time

@contextmanager
def timer(name):
    start = time.time()
    try:
        yield
    finally:
        print(f"{name}: {time.time() - start:.4f}s")

@contextmanager
def lock_manager(lock, name):
    print(f"{name}: Acquiring lock")
    lock.acquire()
    try:
        yield
    finally:
        lock.release()
        print(f"{name}: Released lock")

# 중첩 사용
lock = threading.Lock()

with timer("Total operation"):
    with lock_manager(lock, "Critical section"):
        with open('data.txt', 'w') as f:
            f.write("Processing data")
            time.sleep(0.1)

사용자 정의 중첩 컨텍스트​

class TransactionManager:
    def __init__(self, name):
        self.name = name

    def __enter__(self):
        print(f"  {'  ' * TransactionManager.depth}BEGIN {self.name}")
        TransactionManager.depth += 1
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        TransactionManager.depth -= 1
        if exc_type is None:
            print(f"  {'  ' * TransactionManager.depth}COMMIT {self.name}")
        else:
            print(f"  {'  ' * TransactionManager.depth}ROLLBACK {self.name}")
        return False

TransactionManager.depth = 0

# 중첩된 트랜잭션 시뮬레이션
with TransactionManager("Outer transaction"):
    print("    Doing outer work")
    with TransactionManager("Inner transaction 1"):
        print("      Doing inner work 1")
    with TransactionManager("Inner transaction 2"):
        print("      Doing inner work 2")
    print("    Outer work complete")

3.7 예외 처리​

컨텍스트 매니저에서 예외를 처리하는 다양한 방법을 알아봅니다.

예외 전파와 억제​

class ExceptionPropagator:
    """예외를 전파하는 컨텍스트 매니저"""
    def __enter__(self):
        print("Entering")
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        print(f"Exception: {exc_type}")
        return False  # 예외 전파

class ExceptionSuppressor:
    """예외를 억제하는 컨텍스트 매니저"""
    def __enter__(self):
        print("Entering")
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        if exc_type is not None:
            print(f"Suppressing exception: {exc_type.__name__}")
            return True  # 예외 억제
        return False

# 예외 전파
try:
    with ExceptionPropagator():
        raise ValueError("Error!")
except ValueError as e:
    print(f"Caught: {e}")

# 예외 억제
with ExceptionSuppressor():
    raise ValueError("This will be suppressed")
print("Execution continues")

선택적 예외 처리​

class SelectiveExceptionHandler:
    def __init__(self, suppress_types=()):
        self.suppress_types = suppress_types

    def __enter__(self):
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        if exc_type is None:
            return False

        if isinstance(exc_value, self.suppress_types):
            print(f"Suppressed {exc_type.__name__}: {exc_value}")
            return True  # 지정된 예외만 억제

        return False  # 다른 예외는 전파

# 사용
with SelectiveExceptionHandler(suppress_types=(ValueError, TypeError)):
    raise ValueError("This will be suppressed")

try:
    with SelectiveExceptionHandler(suppress_types=(ValueError,)):
        raise KeyError("This will propagate")
except KeyError as e:
    print(f"Caught KeyError: {e}")

contextlib.suppress​

from contextlib import suppress

# 특정 예외를 조용히 무시
with suppress(FileNotFoundError):
    os.remove('nonexistent_file.txt')
print("File removal attempted")

# 여러 예외 타입 무시
with suppress(ValueError, TypeError, KeyError):
    risky_operation()

# 실용적인 예제: 옵션 파일 읽기
config = {}
with suppress(FileNotFoundError):
    with open('config.json', 'r') as f:
        import json
        config = json.load(f)
# 파일이 없어도 프로그램 계속 실행

예외 재발생​

from contextlib import contextmanager

@contextmanager
def logged_errors(operation):
    """예외를 로깅한 후 재발생"""
    try:
        yield
    except Exception as e:
        print(f"Error in {operation}: {type(e).__name__}: {e}")
        # 예외를 로깅한 후 다시 발생
        raise

# 사용
try:
    with logged_errors("data processing"):
        result = 10 / 0
except ZeroDivisionError:
    print("Handled division by zero")

예외 변환​

@contextmanager
def exception_transformer():
    """예외를 다른 타입으로 변환"""
    try:
        yield
    except ValueError as e:
        # ValueError를 더 구체적인 예외로 변환
        raise RuntimeError(f"Configuration error: {e}") from e

# 사용
try:
    with exception_transformer():
        raise ValueError("Invalid value")
except RuntimeError as e:
    print(f"Caught: {e}")
    print(f"Original: {e.__cause__}")

3.8 실용적인 예제들​

파일 처리​

@contextmanager
def atomic_write(filename, mode='w'):
    """원자적 파일 쓰기 (실패 시 롤백)"""

import threading

@contextmanager
def timed_lock(lock, timeout=None):
    """타임아웃이 있는 락"""
    acquired = lock.acquire(timeout=timeout)
    if not acquired:
        raise TimeoutError(f"Could not acquire lock within {timeout} seconds")
    try:
        yield lock
    finally:
        lock.release()

# 사용

    except TimeoutError:
        print(f"Worker {worker_id} timed out")

@contextmanager
def preserve_state(obj, *attributes):
    """객체의 상태를 보존하고 복원"""
    original_values = {attr: getattr(obj, attr) for attr in attributes}
    try:
        yield obj
    finally:
        for attr, value in original_values.items():
            setattr(obj, attr, value)

from contextlib import contextmanager
import queue
import threading

class ResourcePool:
    """재사용 가능한 리소스 풀"""
    def __init__(self, create_resource, max_size=10):
        self.create_resource = create_resource
        self.max_size = max_size
        self.pool = queue.Queue(maxsize=max_size)
        self.lock = threading.Lock()
        self.size = 0

    @contextmanager
    def acquire(self):
        """리소스를 풀에서 가져오기"""
        resource = self._get_resource()
        try:
            yield resource
        finally:
            self._return_resource(resource)

    def _get_resource(self):
        try:
            return self.pool.get_nowait()
        except queue.Empty:
            with self.lock:
                if self.size < self.max_size:
                    self.size += 1
                    return self.create_resource()
            # 풀이 가득 찼으면 대기
            return self.pool.get()

    def _return_resource(self, resource):
        try:
            self.pool.put_nowait(resource)
        except queue.Full:
            pass

# 사용 예시
def create_connection():
    """데이터베이스 연결 생성 (시뮬레이션)"""
    print("Creating new connection")
    return {"connection_id": id(object())}

# 연결 풀 생성
connection_pool = ResourcePool(create_connection, max_size=3)

# 연결 사용

    print(f"Using connection: {conn['connection_id']}")

재시도 메커니즘​

import time
from contextlib import contextmanager

@contextmanager
def retry_context(max_attempts=3, delay=1.0, backoff=2.0):
    """재시도 메커니즘을 제공하는 컨텍스트 매니저"""
    class RetryManager:
        def __init__(self):
            self.attempt = 0
            self.max_attempts = max_attempts
            self.current_delay = delay
            self.backoff = backoff

        def should_retry(self, exception):
            self.attempt += 1
            if self.attempt >= self.max_attempts:
                print(f"Max attempts ({self.max_attempts}) reached")
                return False

            print(f"Attempt {self.attempt} failed: {exception}")
            print(f"Retrying in {self.current_delay:.2f} seconds...")
            time.sleep(self.current_delay)
            self.current_delay *= self.backoff
            return True

    manager = RetryManager()

    while True:
        try:
            yield manager
            break  # 성공하면 종료
        except Exception as e:
            if not manager.should_retry(e):
                raise

# 사용

class CacheManager:
    """컨텍스트 기반 캐싱"""
    def __init__(self):
        self.cache = {}
        self.enabled = True

    @contextmanager
    def cache_context(self, key):
        """캐시된 결과를 반환하거나 새로 계산"""
        if self.enabled and key in self.cache:
            print(f"Cache hit: {key}")
            yield self.cache[key]
        else:
            print(f"Cache miss: {key}")
            result = {}
            yield result
            if self.enabled:
                self.cache[key] = result

    @contextmanager
    def disabled(self):
        """일시적으로 캐싱 비활성화"""
        original = self.enabled
        self.enabled = False
        try:
            yield
        finally:
            self.enabled = original

문제 1: 트랜잭션 매니저​

데이터베이스 트랜잭션을 시뮬레이션하는 컨텍스트 매니저를 만드세요. 성공 시 커밋, 실패 시 롤백을 수행해야 합니다.

해답 보기
from contextlib import contextmanager

class Transaction:
    def __init__(self):
        self.changes = []
        self.committed = False

    def add_change(self, change):
        """변경사항 추가"""
        if self.committed:
            raise RuntimeError("Transaction already committed")
        self.changes.append(change)

    def commit(self):
        """변경사항 커밋"""
        print(f"Committing {len(self.changes)} changes")
        for change in self.changes:
            print(f"  - {change}")
        self.committed = True

    def rollback(self):
        """변경사항 롤백"""
        print(f"Rolling back {len(self.changes)} changes")
        self.changes.clear()

@contextmanager
def transaction_manager():
    """트랜잭션 컨텍스트 매니저"""
    trans = Transaction()
    try:
        yield trans
        trans.commit()
    except Exception as e:
        trans.rollback()
        print(f"Transaction failed: {e}")
        raise

# 테스트
# 성공 케이스
with transaction_manager() as trans:
    trans.add_change("INSERT user Alice")
    trans.add_change("INSERT user Bob")

# 실패 케이스
try:
    with transaction_manager() as trans:
        trans.add_change("INSERT user Charlie")
        raise ValueError("Database error!")
        trans.add_change("INSERT user Dave")  # 실행되지 않음
except ValueError:
    print("Exception handled")

문제 2: 설정 관리자​

애플리케이션 설정을 임시로 변경했다가 복원하는 컨텍스트 매니저를 작성하세요.

해답 보기
from contextlib import contextmanager

class AppConfig:
    def __init__(self):
        self.settings = {
            'debug': False,
            'log_level': 'INFO',
            'max_connections': 10
        }

    def get(self, key):
        return self.settings.get(key)

    def set(self, key, value):
        self.settings[key] = value

@contextmanager
def temporary_config(config, **overrides):
    """설정을 임시로 변경하는 컨텍스트 매니저"""
    # 원본 값 저장
    original_values = {}
    for key, value in overrides.items():
        original_values[key] = config.get(key)
        config.set(key, value)

    try:
        yield config
    finally:
        # 원본 값 복원
        for key, value in original_values.items():
            config.set(key, value)

# 테스트
config = AppConfig()
print(f"Original debug: {config.get('debug')}")

with temporary_config(config, debug=True, log_level='DEBUG'):
    print(f"Inside context debug: {config.get('debug')}")
    print(f"Inside context log_level: {config.get('log_level')}")

print(f"After context debug: {config.get('debug')}")
print(f"After context log_level: {config.get('log_level')}")

문제 3: 리소스 모니터​

리소스 사용량(메모리, 시간)을 측정하는 컨텍스트 매니저를 만드세요.

해답 보기
import time
import tracemalloc
from contextlib import contextmanager

@contextmanager
def resource_monitor(name="Operation"):
    """리소스 사용량을 모니터링하는 컨텍스트 매니저"""
    # 시작 시간과 메모리 측정 시작
    start_time = time.perf_counter()
    tracemalloc.start()

    print(f"\n{'='*60}")
    print(f"Starting: {name}")
    print(f"{'='*60}")

    try:
        yield
    finally:
        # 종료 시간과 메모리 측정
        end_time = time.perf_counter()
        current, peak = tracemalloc.get_traced_memory()
        tracemalloc.stop()

        elapsed = end_time - start_time

        print(f"\n{'='*60}")
        print(f"Finished: {name}")
        print(f"{'-'*60}")
        print(f"Time elapsed:     {elapsed:.4f} seconds")
        print(f"Current memory:   {current / 1024 / 1024:.2f} MB")
        print(f"Peak memory:      {peak / 1024 / 1024:.2f} MB")
        print(f"{'='*60}\n")

# 테스트
with resource_monitor("List creation"):
    large_list = [i ** 2 for i in range(1000000)]
    time.sleep(0.1)

with resource_monitor("Dictionary creation"):
    large_dict = {i: i ** 2 for i in range(500000)}
    time.sleep(0.05)

문제 4: 재시도 컨텍스트​

실패 시 자동으로 재시도하는 컨텍스트 매니저를 구현하세요.

해답 보기
import time
from contextlib import contextmanager

@contextmanager
def auto_retry(max_attempts=3, delay=1.0, exceptions=(Exception,)):
    """자동 재시도 컨텍스트 매니저"""
    attempt = 0

    while attempt < max_attempts:
        try:
            attempt += 1
            print(f"Attempt {attempt}/{max_attempts}")
            yield attempt
            # 성공하면 종료
            break
        except exceptions as e:
            if attempt >= max_attempts:
                print(f"All {max_attempts} attempts failed")
                raise
            print(f"Attempt {attempt} failed: {e}")
            print(f"Retrying in {delay} seconds...")
            time.sleep(delay)

# 테스트용 불안정한 함수
class UnstableService:
    def __init__(self, fail_times=2):
        self.fail_times = fail_times
        self.call_count = 0

    def call(self):
        self.call_count += 1
        if self.call_count <= self.fail_times:
            raise ConnectionError(f"Connection failed (call {self.call_count})")
        return f"Success on call {self.call_count}"

# 테스트
service = UnstableService(fail_times=2)

with auto_retry(max_attempts=5, delay=0.5, exceptions=(ConnectionError,)):
    result = service.call()
    print(f"Result: {result}")

문제 5: 파일 백업 매니저​

파일을 수정하기 전에 백업을 만들고, 오류 발생 시 복원하는 컨텍스트 매니저를 작성하세요.

해답 보기
import shutil
import os
from contextlib import contextmanager

@contextmanager
def backup_file(filename):
    """파일 백업 및 복원 컨텍스트 매니저"""
    backup_name = f"{filename}.backup"
    backup_created = False

    # 파일이 존재하면 백업 생성
    if os.path.exists(filename):
        shutil.copy2(filename, backup_name)
        backup_created = True
        print(f"Backup created: {backup_name}")

    try:
        yield filename
        # 성공하면 백업 삭제
        if backup_created:
            os.remove(backup_name)
            print(f"Backup removed: {backup_name}")
    except Exception as e:
        # 실패하면 백업에서 복원
        if backup_created:
            shutil.copy2(backup_name, filename)
            os.remove(backup_name)
            print(f"File restored from backup: {filename}")
        raise

# 테스트
# 성공 케이스
with open('test_file.txt', 'w') as f:
    f.write('Original content\n')

with backup_file('test_file.txt'):
    with open('test_file.txt', 'a') as f:
        f.write('New content\n')
print("File modified successfully\n")

# 실패 케이스
try:
    with backup_file('test_file.txt'):
        with open('test_file.txt', 'w') as f:
            f.write('This will be reverted\n')
        raise ValueError("Something went wrong!")
except ValueError:
    print("Exception handled\n")

# 파일 내용 확인
with open('test_file.txt', 'r') as f:
    print("Final file contents:")
    print(f.read())
