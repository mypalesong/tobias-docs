4. 데코레이터

데코레이터는 파이썬에서 함수나 클래스의 동작을 수정하거나 확장하는 강력한 도구입니다. 코드의 재사용성을 높이고, 관심사의 분리(Separation of Concerns)를 가능하게 하며, 코드를 더 깔끔하고 읽기 쉽게 만들어줍니다.

4.1 데코레이터란 무엇인가?​

일급 객체로서의 함수​

파이썬에서 함수는 **일급 객체(First-class Object)**입니다. 이는 다음을 의미합니다:

  1. 함수를 변수에 할당할 수 있습니다
  2. 함수를 다른 함수의 인자로 전달할 수 있습니다
  3. 함수를 다른 함수의 반환값으로 사용할 수 있습니다
  4. 함수를 데이터 구조에 저장할 수 있습니다
# 함수를 변수에 할당
def greet(name):
    return f"안녕하세요, {name}님!"

say_hello = greet
print(say_hello("철수"))  # 안녕하세요, 철수님!

# 함수를 인자로 전달
def execute_twice(func, arg):
    func(arg)
    func(arg)

def print_message(msg):
    print(msg)

execute_twice(print_message, "반복!")
# 출력:
# 반복!
# 반복!

# 함수를 반환값으로 사용
def create_multiplier(n):
    def multiplier(x):
        return x * n
    return multiplier

times_three = create_multiplier(3)
print(times_three(10))  # 30

데코레이터의 기본 개념​

데코레이터는 다른 함수를 인자로 받아서 새로운 함수를 반환하는 함수입니다.

def my_decorator(func):
    def wrapper():
        print("함수 실행 전")
        func()
        print("함수 실행 후")
    return wrapper

def say_hello():
    print("안녕하세요!")

# 데코레이터 수동 적용
decorated_hello = my_decorator(say_hello)
decorated_hello()
# 출력:
# 함수 실행 전
# 안녕하세요!
# 함수 실행 후

4.2 함수 데코레이터​

기본 함수 데코레이터​

def uppercase_decorator(func):
    def wrapper():
        result = func()
        return result.upper()
    return wrapper

def greet():
    return "hello, world!"

# 데코레이터 적용
greet = uppercase_decorator(greet)
print(greet())  # HELLO, WORLD!

인자를 받는 함수 데코레이팅​

def log_arguments(func):
    def wrapper(*args, **kwargs):
        print(f"함수 호출: {func.__name__}")
        print(f"위치 인자: {args}")
        print(f"키워드 인자: {kwargs}")
        result = func(*args, **kwargs)
        print(f"반환값: {result}")
        return result
    return wrapper

def add(a, b):
    return a + b

add = log_arguments(add)
result = add(3, 5)
# 출력:
# 함수 호출: add
# 위치 인자: (3, 5)
# 키워드 인자: {}
# 반환값: 8

4.3 데코레이터 문법 (@)​

@ 기호를 사용하면 데코레이터를 더 간결하게 적용할 수 있습니다.

def bold_decorator(func):
    def wrapper():
        return f"<b>{func()}</b>"
    return wrapper

def italic_decorator(func):
    def wrapper():
        return f"<i>{func()}</i>"
    return wrapper

@bold_decorator
def greet():
    return "Hello!"

print(greet())  # <b>Hello!</b>

# @ 문법은 다음과 동일합니다:
# greet = bold_decorator(greet)

여러 데코레이터 체이닝​

@bold_decorator
@italic_decorator
def greet_fancy():
    return "Hello!"

print(greet_fancy())  # <b><i>Hello!</i></b>

# 아래에서 위로 적용됩니다:
# greet_fancy = bold_decorator(italic_decorator(greet_fancy))

4.4 인자를 받는 데코레이터​

데코레이터 자체가 인자를 받아야 할 때는 한 단계 더 깊은 중첩이 필요합니다.

def repeat(times):
    """함수를 여러 번 실행하는 데코레이터"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                result = func(*args, **kwargs)
                results.append(result)
            return results
        return wrapper
    return decorator

@repeat(times=3)
def greet(name):
    return f"안녕, {name}!"

print(greet("철수"))
# ['안녕, 철수!', '안녕, 철수!', '안녕, 철수!']

선택적 인자를 가진 데코레이터​

import functools

def smart_decorator(arg=None):
    """인자가 있거나 없을 수 있는 데코레이터"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            prefix = arg if arg else "기본"
            print(f"[{prefix}] {func.__name__} 호출")
            return func(*args, **kwargs)
        return wrapper

    # 인자 없이 호출된 경우 (@smart_decorator)
    if callable(arg):
        func = arg
        arg = None
        return decorator(func)

    # 인자와 함께 호출된 경우 (@smart_decorator("로그"))
    return decorator

@smart_decorator
def func1():
    return "함수1"

@smart_decorator("중요")
def func2():
    return "함수2"

func1()  # [기본] func1 호출
func2()  # [중요] func2 호출

4.5 클래스 데코레이터​

클래스도 데코레이터로 사용할 수 있으며, 클래스 자체를 데코레이터로 꾸밀 수도 있습니다.

클래스를 데코레이터로 사용​

class CountCalls:
    """함수 호출 횟수를 세는 데코레이터 클래스"""
    def __init__(self, func):
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 호출 횟수: {self.count}")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello():
    print("안녕하세요!")

say_hello()  # say_hello 호출 횟수: 1
say_hello()  # say_hello 호출 횟수: 2
say_hello()  # say_hello 호출 횟수: 3

클래스 데코레이터​

def add_repr(cls):
    """클래스에 __repr__ 메서드를 추가하는 데코레이터"""
    def __repr__(self):
        attrs = ', '.join(f"{k}={v!r}" for k, v in self.__dict__.items())
        return f"{cls.__name__}({attrs})"

    cls.__repr__ = __repr__
    return cls

@add_repr
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

person = Person("김철수", 30)
print(person)  # Person(name='김철수', age=30)

싱글톤 패턴 구현​

def singleton(cls):
    """싱글톤 패턴을 구현하는 클래스 데코레이터"""
    instances = {}

    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]

    return get_instance

@singleton
class Database:
    def __init__(self):
        print("데이터베이스 연결 생성")

db1 = Database()  # 데이터베이스 연결 생성
db2 = Database()  # 출력 없음 (기존 인스턴스 재사용)
print(db1 is db2)  # True

4.6 내장 데코레이터​

@property​

프로퍼티는 메서드를 속성처럼 접근할 수 있게 해줍니다.

class Temperature:
    def __init__(self, celsius):
        self._celsius = celsius

    @property
    def celsius(self):
        """섭씨 온도 반환"""
        return self._celsius

    @celsius.setter
    def celsius(self, value):
        """섭씨 온도 설정"""
        if value < -273.15:
            raise ValueError("절대 영도보다 낮을 수 없습니다")
        self._celsius = value

    @property
    def fahrenheit(self):
        """화씨 온도 반환"""
        return self._celsius * 9/5 + 32

    @fahrenheit.setter
    def fahrenheit(self, value):
        """화씨 온도 설정"""
        self.celsius = (value - 32) * 5/9

temp = Temperature(25)
print(temp.celsius)      # 25
print(temp.fahrenheit)   # 77.0

temp.fahrenheit = 98.6
print(temp.celsius)      # 37.0

@staticmethod​

정적 메서드는 클래스나 인스턴스 상태에 접근하지 않는 메서드입니다.

class MathUtils:
    @staticmethod
    def add(a, b):
        """두 수를 더합니다"""
        return a + b

    @staticmethod
    def multiply(a, b):
        """두 수를 곱합니다"""
        return a * b

# 인스턴스 생성 없이 호출 가능
print(MathUtils.add(5, 3))       # 8
print(MathUtils.multiply(4, 7))  # 28

# 인스턴스로도 호출 가능
utils = MathUtils()
print(utils.add(10, 20))  # 30

@classmethod​

클래스 메서드는 클래스 자체를 첫 번째 인자로 받습니다.

class Person:
    population = 0

    def __init__(self, name, age):
        self.name = name
        self.age = age
        Person.population += 1

    @classmethod
    def from_birth_year(cls, name, birth_year):
        """출생 연도로부터 Person 객체 생성"""
        from datetime import datetime
        age = datetime.now().year - birth_year
        return cls(name, age)

    @classmethod
    def get_population(cls):
        """현재 인구 수 반환"""
        return cls.population

person1 = Person("김철수", 30)
person2 = Person.from_birth_year("이영희", 1995)

print(Person.get_population())  # 2

4.7 functools.wraps​

functools.wraps는 데코레이터를 만들 때 원본 함수의 메타데이터를 보존합니다.

import functools

# functools.wraps 없이
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        """래퍼 함수"""
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def greet(name):
    """인사 함수"""
    return f"안녕, {name}!"

print(greet.__name__)  # wrapper
print(greet.__doc__)   # 래퍼 함수

# functools.wraps 사용
def good_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        """래퍼 함수"""
        return func(*args, **kwargs)
    return wrapper

@good_decorator
def greet_better(name):
    """인사 함수"""
    return f"안녕, {name}!"

print(greet_better.__name__)  # greet_better
print(greet_better.__doc__)   # 인사 함수
팁

항상 @functools.wraps(func)를 사용하여 원본 함수의 __name__, __doc__, __module__ 등의 속성을 보존하세요. 이는 디버깅과 문서화에 매우 중요합니다.

4.8 데코레이터 체이닝​

여러 데코레이터를 함께 사용할 수 있으며, 아래에서 위로 적용됩니다.

import functools

def debug(func):
    """함수 호출을 디버깅하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        print(f"호출: {func.__name__}({signature})")
        result = func(*args, **kwargs)
        print(f"반환: {func.__name__!r} -> {result!r}")
        return result
    return wrapper

def uppercase(func):
    """결과를 대문자로 변환하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        result = func(*args, **kwargs)
        return result.upper()
    return wrapper

@debug
@uppercase
def greet(name):
    return f"hello, {name}"

greet("world")
# 출력:
# 호출: greet('world')
# 반환: 'greet' -> 'HELLO, WORLD'

4.9 실용적인 예제​

4.9.1 실행 시간 측정​

import functools
import time

def timer(func):
    """함수 실행 시간을 측정하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.perf_counter()
        result = func(*args, **kwargs)
        end_time = time.perf_counter()
        run_time = end_time - start_time
        print(f"{func.__name__} 실행 시간: {run_time:.4f}초")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "완료!"

result = slow_function()
# slow_function 실행 시간: 1.0012초

4.9.2 로깅​

import functools
import logging

logging.basicConfig(level=logging.INFO)

def log_function_call(func):
    """함수 호출을 로깅하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logger = logging.getLogger(func.__module__)
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        logger.info(f"{func.__name__}({signature}) 호출")
        try:
            result = func(*args, **kwargs)
            logger.info(f"{func.__name__} 성공: {result!r}")
            return result
        except Exception as e:
            logger.exception(f"{func.__name__} 실패: {e}")
            raise
    return wrapper

@log_function_call
def divide(a, b):
    return a / b

divide(10, 2)   # 정상 실행
# divide(10, 0)  # 예외 발생 및 로깅

4.9.3 인증 및 권한 확인​

import functools

# 간단한 사용자 컨텍스트 시뮬레이션
current_user = {"username": "admin", "role": "admin"}

def require_auth(func):
    """인증이 필요한 함수를 위한 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        if not current_user:
            raise PermissionError("로그인이 필요합니다")
        return func(*args, **kwargs)
    return wrapper

def require_role(role):
    """특정 역할이 필요한 함수를 위한 데코레이터"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if current_user.get("role") != role:
                raise PermissionError(f"{role} 권한이 필요합니다")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@require_auth
@require_role("admin")
def delete_user(user_id):
    return f"사용자 {user_id} 삭제됨"

try:
    result = delete_user(123)
    print(result)  # 사용자 123 삭제됨
except PermissionError as e:
    print(f"오류: {e}")

4.9.4 캐싱 (메모이제이션)​

import functools

def memoize(func):
    """결과를 캐싱하는 데코레이터"""
    cache = {}

    @functools.wraps(func)
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
            print(f"캐시 미스: {args}")
        else:
            print(f"캐시 히트: {args}")
        return cache[args]
    return wrapper

@memoize
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(10))
# 캐시를 사용하여 훨씬 빠르게 계산
팁

파이썬 3.2+에서는 functools.lru_cache 데코레이터를 사용하면 더 강력한 캐싱 기능을 이용할 수 있습니다.

from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

4.9.5 재시도 로직​

import functools
import time
import random

def retry(max_attempts=3, delay=1, backoff=2, exceptions=(Exception,)):
    """실패 시 재시도하는 데코레이터"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            attempt = 1
            current_delay = delay

            while attempt <= max_attempts:
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    if attempt == max_attempts:
                        print(f"최대 재시도 횟수 초과: {e}")
                        raise

                    print(f"시도 {attempt} 실패: {e}")
                    print(f"{current_delay}초 후 재시도...")
                    time.sleep(current_delay)

                    attempt += 1
                    current_delay *= backoff

        return wrapper
    return decorator

@retry(max_attempts=5, delay=0.5, backoff=2)
def unstable_network_call():
    """불안정한 네트워크 호출 시뮬레이션"""
    if random.random() < 0.7:  # 70% 실패율
        raise ConnectionError("네트워크 연결 실패")
    return "성공!"

# result = unstable_network_call()

4.9.6 입력 검증​

import functools

def validate_types(**type_hints):
    """함수 인자의 타입을 검증하는 데코레이터"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            # 함수 시그니처에서 인자 이름 가져오기
            import inspect
            sig = inspect.signature(func)
            bound_args = sig.bind(*args, **kwargs)
            bound_args.apply_defaults()

            # 타입 검증
            for arg_name, expected_type in type_hints.items():
                if arg_name in bound_args.arguments:
                    value = bound_args.arguments[arg_name]
                    if not isinstance(value, expected_type):
                        raise TypeError(
                            f"{arg_name}는 {expected_type.__name__} 타입이어야 합니다. "
                            f"받은 값: {type(value).__name__}"
                        )

            return func(*args, **kwargs)
        return wrapper
    return decorator

@validate_types(name=str, age=int, salary=float)
def create_employee(name, age, salary):
    return {"name": name, "age": age, "salary": salary}

# 정상 작동
employee = create_employee("김철수", 30, 50000.0)
print(employee)

# 오류 발생
try:
    create_employee("이영희", "30", 60000.0)  # age가 문자열
except TypeError as e:
    print(f"오류: {e}")

4.9.7 속도 제한 (Rate Limiting)​

import functools
import time
from collections import deque

def rate_limit(max_calls, time_window):
    """
    함수 호출 속도를 제한하는 데코레이터

    Args:
        max_calls: 시간 창 내 최대 호출 횟수
        time_window: 시간 창 (초)
    """
    def decorator(func):
        calls = deque()

        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()

            # 시간 창을 벗어난 호출 기록 제거
            while calls and calls[0] < now - time_window:
                calls.popleft()

            # 속도 제한 확인
            if len(calls) >= max_calls:
                sleep_time = time_window - (now - calls[0])
                print(f"속도 제한 도달. {sleep_time:.2f}초 대기 중...")
                time.sleep(sleep_time)
                return wrapper(*args, **kwargs)

            calls.append(now)
            return func(*args, **kwargs)

        return wrapper
    return decorator

@rate_limit(max_calls=3, time_window=5)
def api_call(endpoint):
    print(f"API 호출: {endpoint}")
    return {"status": "success"}

# 처음 3번은 즉시 실행, 4번째는 대기
for i in range(5):
    api_call(f"/endpoint/{i}")

import functools

def sanitize_string(func):
    """문자열 인자를 정제하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # 모든 문자열 인자를 strip하고 소문자로 변환
        clean_args = []
        for arg in args:
            if isinstance(arg, str):
                clean_args.append(arg.strip().lower())
            else:
                clean_args.append(arg)

        clean_kwargs = {}
        for key, value in kwargs.items():
            if isinstance(value, str):
                clean_kwargs[key] = value.strip().lower()
            else:
                clean_kwargs[key] = value

        return func(*clean_args, **clean_kwargs)
    return wrapper

@sanitize_string
def create_username(first_name, last_name):
    return f"{first_name}.{last_name}"

3. 트랜잭션 관리​

import functools

class DatabaseConnection:
    def __init__(self):
        self.in_transaction = False

    def begin(self):
        print("트랜잭션 시작")
        self.in_transaction = True

    def commit(self):
        print("트랜잭션 커밋")
        self.in_transaction = False

    def rollback(self):
        print("트랜잭션 롤백")
        self.in_transaction = False

db = DatabaseConnection()

def transaction(func):
    """데이터베이스 트랜잭션을 관리하는 데코레이터"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        db.begin()
        try:
            result = func(*args, **kwargs)
            db.commit()
            return result
        except Exception as e:
            db.rollback()
            raise
    return wrapper

@transaction
def transfer_money(from_account, to_account, amount):
    print(f"{from_account}에서 {amount}원 차감")
    print(f"{to_account}에 {amount}원 추가")
    # if amount > 1000000:
    #     raise ValueError("한도 초과")
    return True

transfer_money("계좌A", "계좌B", 50000)

연습 3: 예외 처리 데코레이터​

함수에서 발생하는 특정 예외를 잡아서 기본값을 반환하는 데코레이터를 작성하세요.

해답 보기
import functools

def catch_exceptions(default_value=None, exceptions=(Exception,)):
    """
    예외를 잡아서 기본값을 반환하는 데코레이터

    Args:
        default_value: 예외 발생 시 반환할 기본값
        exceptions: 잡을 예외 타입들의 튜플
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except exceptions as e:
                print(f"예외 발생: {e}")
                return default_value
        return wrapper
    return decorator

@catch_exceptions(default_value=0, exceptions=(ZeroDivisionError, ValueError))
def divide(a, b):
    return a / b

print(divide(10, 2))  # 5.0
print(divide(10, 0))  # 예외 발생: division by zero -> 0

연습 4: 인자 검증 데코레이터​

함수 인자가 특정 조건을 만족하는지 검증하는 데코레이터를 작성하세요.

해답 보기
import functools

def validate_range(arg_name, min_value=None, max_value=None):
    """
    특정 인자가 범위 내에 있는지 검증하는 데코레이터

    Args:
        arg_name: 검증할 인자 이름
        min_value: 최소값 (None이면 체크 안 함)
        max_value: 최대값 (None이면 체크 안 함)
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            import inspect
            sig = inspect.signature(func)
            bound_args = sig.bind(*args, **kwargs)
            bound_args.apply_defaults()

            if arg_name in bound_args.arguments:
                value = bound_args.arguments[arg_name]

                if min_value is not None and value < min_value:
                    raise ValueError(
                        f"{arg_name}는 {min_value} 이상이어야 합니다. "
                        f"받은 값: {value}"
                    )

                if max_value is not None and value > max_value:
                    raise ValueError(
                        f"{arg_name}는 {max_value} 이하여야 합니다. "
                        f"받은 값: {value}"
                    )

            return func(*args, **kwargs)
        return wrapper
    return decorator

@validate_range("age", min_value=0, max_value=150)
def set_age(age):
    return f"나이가 {age}로 설정되었습니다"

print(set_age(25))  # 나이가 25로 설정되었습니다

try:
    set_age(-5)  # ValueError 발생
except ValueError as e:
    print(f"오류: {e}")

try:
    set_age(200)  # ValueError 발생
except ValueError as e:
    print(f"오류: {e}")

연습 5: 디바운스 데코레이터​

일정 시간 동안 함수가 여러 번 호출되어도 마지막 호출만 실행하는 디바운스 데코레이터를 작성하세요.

해답 보기
import functools
import threading

def debounce(wait_time):
    """
    디바운스 데코레이터 - 마지막 호출만 실행

    Args:
        wait_time: 대기 시간 (초)
    """
    def decorator(func):
        timer = None

        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            nonlocal timer

            # 이전 타이머가 있으면 취소
            if timer is not None:
                timer.cancel()

            # 새 타이머 설정
            timer = threading.Timer(
                wait_time,
                lambda: func(*args, **kwargs)
            )
            timer.start()

        return wrapper
    return decorator

@debounce(wait_time=1.0)
def search(query):
    print(f"검색 실행: {query}")

# 빠르게 여러 번 호출
search("p")
search("py")
search("pyt")
search("pyth")
search("pytho")
search("python")
# 1초 후 마지막 호출만 실행됨: "검색 실행: python"

연습 6: 조건부 실행 데코레이터​

해답 보기
import functools

def run_if(condition_func):
    """
    조건이 참일 때만 함수를 실행하는 데코레이터

    Args:
        condition_func: 조건을 확인하는 함수 (True/False 반환)
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if condition_func():
                return func(*args, **kwargs)
            else:
                print(f"{func.__name__} 실행 건너뜀 (조건 불충족)")
                return None
        return wrapper
    return decorator

# 설정 예시
debug_mode = True

def is_debug_mode():
    return debug_mode

@run_if(is_debug_mode)
def debug_print(message):
    print(f"[DEBUG] {message}")

debug_print("디버그 모드 활성화")  # 실행됨

debug_mode = False
debug_print("이것은 출력되지 않습니다")  # 건너뜀

요약​

데코레이터는 파이썬의 강력한 기능으로, 코드의 재사용성과 가독성을 크게 향상시킵니다:

  1. 함수는 일급 객체: 변수에 할당, 인자로 전달, 반환값으로 사용 가능
  2. @ 문법: 데코레이터를 간결하게 적용
  3. functools.wraps: 원본 함수의 메타데이터 보존
  4. 인자를 받는 데코레이터: 추가 중첩 레벨 필요
  5. 클래스 데코레이터: 클래스를 데코레이터로 사용하거나 클래스를 데코레이팅
  6. 내장 데코레이터: @property, @staticmethod, @classmethod
  7. 실용적 사용 사례: 로깅, 타이밍, 캐싱, 인증, 재시도 등
  8. 모범 사례: 단일 책임, 명확한 문서화, 적절한 예외 처리

데코레이터를 마스터하면 더 깔끔하고 유지보수하기 쉬운 파이썬 코드를 작성할 수 있습니다!
