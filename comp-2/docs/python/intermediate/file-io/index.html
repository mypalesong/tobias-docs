3. 파일 입출력

파일 입출력(File I/O)은 프로그램이 데이터를 영구적으로 저장하고 읽어오는 필수적인 기능입니다. 이 장에서는 Python에서 파일을 다루는 다양한 방법과 실전 활용법을 학습합니다.

3.1 파일 열기와 닫기​

기본적인 파일 열기​

Python에서는 open() 함수로 파일을 엽니다.

# 파일 열기
file = open('example.txt', 'r')

# 파일 사용
content = file.read()

# 파일 닫기 (중요!)
file.close()
경고

파일을 열면 반드시 close()로 닫아야 합니다. 그렇지 않으면 메모리 누수나 파일 손상이 발생할 수 있습니다.

파일이 존재하지 않을 때 처리​

try:
    file = open('nonexistent.txt', 'r')
    content = file.read()
    file.close()
except FileNotFoundError:
    print("파일을 찾을 수 없습니다.")
except IOError as e:
    print(f"파일 읽기 오류: {e}")

3.2 파일 읽기​

read() - 전체 내용 읽기​

# 전체 파일을 하나의 문자열로 읽기
with open('example.txt', 'r', encoding='utf-8') as file:
    content = file.read()
    print(content)

# 지정한 바이트 수만큼 읽기
with open('example.txt', 'r', encoding='utf-8') as file:
    first_100_chars = file.read(100)
    print(first_100_chars)

readline() - 한 줄씩 읽기​

with open('example.txt', 'r', encoding='utf-8') as file:
    # 첫 번째 줄 읽기
    first_line = file.readline()
    print(f"첫 번째 줄: {first_line}")

    # 두 번째 줄 읽기
    second_line = file.readline()
    print(f"두 번째 줄: {second_line}")

readlines() - 모든 줄을 리스트로 읽기​

with open('example.txt', 'r', encoding='utf-8') as file:
    lines = file.readlines()

    for i, line in enumerate(lines, 1):
        print(f"{i}: {line.strip()}")

파일을 반복문으로 읽기 (가장 효율적)​

# 메모리 효율적인 방법 - 큰 파일에 적합
with open('large_file.txt', 'r', encoding='utf-8') as file:
    for line in file:
        # 줄바꿈 문자 제거
        line = line.strip()
        print(line)
팁

대용량 파일은 readlines() 대신 반복문으로 한 줄씩 읽는 것이 메모리 효율적입니다.

3.3 파일 쓰기​

write() - 문자열 쓰기​

# 파일에 쓰기 (덮어쓰기)
with open('output.txt', 'w', encoding='utf-8') as file:
    file.write("첫 번째 줄\n")
    file.write("두 번째 줄\n")

    # write()는 작성한 문자 수를 반환
    chars_written = file.write("세 번째 줄\n")
    print(f"{chars_written}자 작성됨")

writelines() - 리스트를 파일에 쓰기​

lines = [
    "Python 파일 입출력\n",
    "줄바꿈 문자를 직접 추가해야 합니다\n",
    "마지막 줄\n"
]

with open('output.txt', 'w', encoding='utf-8') as file:
    file.writelines(lines)

파일에 추가하기​

# 기존 내용 뒤에 추가
with open('output.txt', 'a', encoding='utf-8') as file:
    file.write("새로운 내용 추가\n")

3.4 파일 모드​

모드설명파일이 없을 때파일이 있을 때
'r'읽기 전용오류 발생처음부터 읽기
'w'쓰기 전용새로 생성내용 삭제 후 쓰기
'a'추가 모드새로 생성끝에 추가
'x'배타적 생성새로 생성오류 발생
'r+'읽기/쓰기오류 발생읽기/쓰기 가능
'w+'읽기/쓰기새로 생성내용 삭제 후 읽기/쓰기
'a+'읽기/추가새로 생성읽기 가능, 끝에 추가

바이너리 모드​

텍스트 모드에 'b'를 추가하면 바이너리 모드가 됩니다.

# 바이너리 읽기
with open('image.png', 'rb') as file:
    data = file.read()

# 바이너리 쓰기
with open('copy.png', 'wb') as file:
    file.write(data)

파일 모드 예제​

# 'x' 모드 - 파일이 이미 존재하면 오류
try:
    with open('unique.txt', 'x') as file:
        file.write("이 파일은 처음 생성됩니다")
except FileExistsError:
    print("파일이 이미 존재합니다")

# 'r+' 모드 - 읽기와 쓰기 동시에
with open('data.txt', 'r+', encoding='utf-8') as file:
    content = file.read()  # 먼저 읽기
    file.write("\n추가 내용")  # 그 다음 쓰기

3.5 with 문 (컨텍스트 매니저)​

with 문은 파일을 자동으로 닫아주는 Python의 강력한 기능입니다.

기본 사용법​

# with 문 사용 (권장)
with open('example.txt', 'r', encoding='utf-8') as file:
    content = file.read()
    # 블록이 끝나면 자동으로 file.close() 호출

# 파일이 이미 닫혀 있음
print(file.closed)  # True

여러 파일 동시에 열기​

# 방법 1: 중첩된 with
with open('input.txt', 'r', encoding='utf-8') as infile:
    with open('output.txt', 'w', encoding='utf-8') as outfile:
        for line in infile:
            outfile.write(line.upper())

# 방법 2: 한 줄로 (Python 2.7+)
with open('input.txt', 'r', encoding='utf-8') as infile, \
     open('output.txt', 'w', encoding='utf-8') as outfile:
    for line in infile:
        outfile.write(line.upper())

예외 발생 시에도 안전하게 닫기​

with open('data.txt', 'r', encoding='utf-8') as file:
    # 예외가 발생해도 파일은 자동으로 닫힙니다
    content = file.read()
    result = 10 / 0  # ZeroDivisionError 발생
    # 하지만 파일은 이미 닫혀 있음
팁

항상 with 문을 사용하여 파일을 여는 것이 좋습니다. 예외가 발생해도 파일이 안전하게 닫힙니다.

3.6 경로 다루기​

os.path 모듈​

import os

# 경로 결합
file_path = os.path.join('data', 'files', 'example.txt')
print(file_path)  # data/files/example.txt (또는 data\files\example.txt)

# 절대 경로 얻기
abs_path = os.path.abspath('example.txt')
print(abs_path)

# 경로 분리
directory, filename = os.path.split('/home/user/data.txt')
print(f"디렉토리: {directory}, 파일명: {filename}")

# 파일명과 확장자 분리
name, ext = os.path.splitext('document.pdf')
print(f"이름: {name}, 확장자: {ext}")

# 경로 존재 확인
if os.path.exists('example.txt'):
    print("파일이 존재합니다")

# 파일인지 디렉토리인지 확인
print(os.path.isfile('example.txt'))  # 파일이면 True
print(os.path.isdir('data'))  # 디렉토리면 True

# 파일 크기
size = os.path.getsize('example.txt')
print(f"파일 크기: {size} bytes")

pathlib 모듈 (Python 3.4+, 권장)​

from pathlib import Path

# Path 객체 생성
file_path = Path('data') / 'files' / 'example.txt'
print(file_path)

# 절대 경로
abs_path = file_path.resolve()
print(abs_path)

# 경로 정보
print(f"부모 디렉토리: {file_path.parent}")
print(f"파일명: {file_path.name}")
print(f"확장자: {file_path.suffix}")
print(f"확장자 제외 이름: {file_path.stem}")

# 존재 확인
if file_path.exists():
    print("파일이 존재합니다")

# 파일인지 디렉토리인지 확인
print(file_path.is_file())
print(file_path.is_dir())

# 파일 읽기/쓰기 (pathlib로 직접 가능)
file_path.write_text("내용 작성", encoding='utf-8')
content = file_path.read_text(encoding='utf-8')

# 파일 크기
size = file_path.stat().st_size
print(f"파일 크기: {size} bytes")

경로 순회​

3.7 CSV 파일 다루기​

CSV 읽기​

import csv

# 기본적인 CSV 읽기
with open('data.csv', 'r', encoding='utf-8') as file:
    csv_reader = csv.reader(file)

    # 헤더 읽기
    header = next(csv_reader)
    print(f"헤더: {header}")

    # 데이터 읽기
    for row in csv_reader:
        print(row)  # 리스트로 반환

# DictReader 사용 (딕셔너리로 읽기)
with open('data.csv', 'r', encoding='utf-8') as file:
    csv_reader = csv.DictReader(file)

    for row in csv_reader:
        print(row)  # OrderedDict 또는 dict로 반환
        print(f"이름: {row['name']}, 나이: {row['age']}")

CSV 쓰기​

import csv

# 기본적인 CSV 쓰기
data = [
    ['이름', '나이', '도시'],

import json
from datetime import datetime

# 복잡한 데이터 구조
data = {
    "users": [
        {"name": "김철수", "age": 25},
        {"name": "이영희", "age": 30}
    ],
    "timestamp": datetime.now().isoformat(),
    "version": 1.0

# 정렬, 들여쓰기, 한글 유지

        file,

compact = json.dumps(data, ensure_ascii=False, separators=(',', ':'))

커스텀 JSON 인코더/디코더​

import json
from datetime import datetime

class DateTimeEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        return super().default(obj)

# 커스텀 인코더 사용
data = {
    "name": "이벤트",
    "created_at": datetime.now()
}

json_string = json.dumps(data, cls=DateTimeEncoder, ensure_ascii=False)
print(json_string)

3.9 바이너리 파일​

이미지 파일 복사​

# 바이너리 파일 읽기/쓰기
with open('image.jpg', 'rb') as source:
    data = source.read()

with open('copy.jpg', 'wb') as dest:
    dest.write(data)

# 청크 단위로 읽어서 복사 (대용량 파일)
chunk_size = 4096  # 4KB

with open('large_video.mp4', 'rb') as source, \
     open('copy_video.mp4', 'wb') as dest:
    while True:
        chunk = source.read(chunk_size)
        if not chunk:
            break
        dest.write(chunk)

struct 모듈로 바이너리 데이터 다루기​

import struct

# 정수를 바이너리로 변환하여 저장
data = [100, 200, 300, 400, 500]

with open('numbers.bin', 'wb') as file:
    for num in data:
        # 'i'는 4바이트 정수를 의미
        binary_data = struct.pack('i', num)
        file.write(binary_data)

# 바이너리에서 정수 읽기
numbers = []
with open('numbers.bin', 'rb') as file:
    while True:
        binary_data = file.read(4)  # 4바이트 읽기
        if not binary_data:
            break
        num = struct.unpack('i', binary_data)[0]
        numbers.append(num)

print(numbers)  # [100, 200, 300, 400, 500]

from pathlib import Path
from datetime import datetime
import re

class LogParser:
    def __init__(self, log_file):
        self.log_file = Path(log_file)
        self.errors = []
        self.warnings = []
        self.info = []

    def parse(self):
        """로그 파일을 파싱하여 레벨별로 분류"""
        if not self.log_file.exists():
            raise FileNotFoundError(f"{self.log_file}을 찾을 수 없습니다")

        with self.log_file.open('r', encoding='utf-8') as file:
            for line in file:
                line = line.strip()
                if 'ERROR' in line:
                    self.errors.append(line)
                elif 'WARNING' in line:
                    self.warnings.append(line)
                elif 'INFO' in line:
                    self.info.append(line)

    def get_error_count(self):
        """에러 개수 반환"""
        return len(self.errors)

    def get_errors_by_date(self, date_str):
        """특정 날짜의 에러만 필터링"""
        pattern = re.compile(date_str)
        return [error for error in self.errors if pattern.search(error)]

    def save_report(self, output_file):
        """분석 결과를 파일로 저장"""
        report = [
            "=== 로그 분석 보고서 ===",
            f"생성 시간: {datetime.now()}",
            f"분석 파일: {self.log_file}",
            "",
            f"총 에러: {len(self.errors)}개",
            f"총 경고: {len(self.warnings)}개",
            f"총 정보: {len(self.info)}개",
            "",
            "=== 최근 에러 10개 ===",
        ]

        for error in self.errors[-10:]:
            report.append(error)

        output_path = Path(output_file)
        output_path.write_text('\n'.join(report), encoding='utf-8')

예제 2: CSV 데이터 프로세서​

import csv
from pathlib import Path
from collections import defaultdict

class CSVProcessor:
    def __init__(self, csv_file):
        self.csv_file = Path(csv_file)
        self.data = []
        self.headers = []

    def load(self):
        """CSV 파일 로드"""
        with self.csv_file.open('r', encoding='utf-8') as file:
            reader = csv.DictReader(file)
            self.headers = reader.fieldnames
            self.data = list(reader)
        return self

    def filter_by(self, column, value):
        """특정 컬럼의 값으로 필터링"""
        filtered = [row for row in self.data if row.get(column) == value]
        return filtered

    def group_by(self, column):
        """특정 컬럼으로 그룹화"""
        groups = defaultdict(list)
        for row in self.data:
            key = row.get(column)
            groups[key].append(row)
        return dict(groups)

    def calculate_average(self, column):
        """숫자 컬럼의 평균 계산"""
        try:
            values = [float(row[column]) for row in self.data if row.get(column)]
            return sum(values) / len(values) if values else 0
        except ValueError:
            raise ValueError(f"{column}은 숫자가 아닙니다")

    def save(self, output_file, data=None):
        """데이터를 CSV 파일로 저장"""
        if data is None:
            data = self.data

        if not data:
            print("저장할 데이터가 없습니다")
            return

        output_path = Path(output_file)
        with output_path.open('w', newline='', encoding='utf-8') as file:
            writer = csv.DictWriter(file, fieldnames=self.headers)
            writer.writeheader()
            writer.writerows(data)

# 사용 예제
processor = CSVProcessor('sales.csv')
processor.load()

# 서울 지역 데이터만 필터링

# 도시별 그룹화
city_groups = processor.group_by('city')
for city, rows in city_groups.items():
    print(f"{city}: {len(rows)}건")

# 평균 계산
avg_price = processor.calculate_average('price')
print(f"평균 가격: {avg_price:.2f}원")

# 필터링된 데이터 저장

import json
from pathlib import Path

class ConfigManager:
    def __init__(self, config_file='config.json'):
        self.config_file = Path(config_file)
        self.config = {}
        self.load()

    def load(self):
        """설정 파일 로드"""
        if self.config_file.exists():
            with self.config_file.open('r', encoding='utf-8') as file:
                self.config = json.load(file)
        else:
            # 기본 설정
            self.config = {
                'app_name': 'MyApp',
                'version': '1.0.0',
                'debug': False,
                'database': {
                    'host': 'localhost',
                    'port': 5432,
                    'name': 'mydb'
                }
            }
            self.save()

    def save(self):
        """설정 파일 저장"""
        with self.config_file.open('w', encoding='utf-8') as file:
            json.dump(self.config, file, ensure_ascii=False, indent=2)

    def get(self, key, default=None):
        """설정 값 가져오기 (점 표기법 지원)"""
        keys = key.split('.')
        value = self.config

        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
                if value is None:
                    return default
            else:
                return default

        return value

    def set(self, key, value):
        """설정 값 설정 (점 표기법 지원)"""
        keys = key.split('.')
        config = self.config

        for k in keys[:-1]:
            if k not in config or not isinstance(config[k], dict):
                config[k] = {}
            config = config[k]

        config[keys[-1]] = value
        self.save()

    def reset(self):
        """설정 초기화"""
        self.config_file.unlink(missing_ok=True)
        self.load()

예제 4: 파일 백업 유틸리티​

from pathlib import Path
import shutil
from datetime import datetime
import zipfile

class BackupUtility:
    def __init__(self, source_dir, backup_dir='backups'):
        self.source_dir = Path(source_dir)
        self.backup_dir = Path(backup_dir)
        self.backup_dir.mkdir(exist_ok=True)

    def create_backup(self, compress=True):
        """백업 생성"""
        if not self.source_dir.exists():
            raise FileNotFoundError(f"{self.source_dir}을 찾을 수 없습니다")

        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')

        if compress:
            backup_file = self.backup_dir / f"backup_{timestamp}.zip"
            self._create_zip_backup(backup_file)
        else:
            backup_dir = self.backup_dir / f"backup_{timestamp}"
            self._create_folder_backup(backup_dir)

        return backup_file if compress else backup_dir

    def _create_zip_backup(self, backup_file):
        """압축 백업 생성"""
        with zipfile.ZipFile(backup_file, 'w', zipfile.ZIP_DEFLATED) as zipf:
            for file in self.source_dir.rglob('*'):
                if file.is_file():
                    arcname = file.relative_to(self.source_dir)
                    zipf.write(file, arcname)

        print(f"압축 백업 생성: {backup_file}")

    def _create_folder_backup(self, backup_dir):
        """폴더 백업 생성"""
        shutil.copytree(self.source_dir, backup_dir)
        print(f"폴더 백업 생성: {backup_dir}")

    def restore_backup(self, backup_path, restore_dir):
        """백업 복원"""
        backup_path = Path(backup_path)
        restore_dir = Path(restore_dir)

        if backup_path.suffix == '.zip':
            with zipfile.ZipFile(backup_path, 'r') as zipf:
                zipf.extractall(restore_dir)
        else:
            shutil.copytree(backup_path, restore_dir)

        print(f"백업 복원 완료: {restore_dir}")

    def list_backups(self):
        """백업 목록 조회"""
        backups = []

        for item in self.backup_dir.iterdir():
            stat = item.stat()
            backups.append({
                'name': item.name,
                'path': item,
                'size': stat.st_size,
                'created': datetime.fromtimestamp(stat.st_ctime)
            })

        return sorted(backups, key=lambda x: x['created'], reverse=True)

    def delete_old_backups(self, keep_count=5):
        """오래된 백업 삭제"""
        backups = self.list_backups()

        for backup in backups[keep_count:]:
            path = backup['path']
            if path.is_file():
                path.unlink()
            else:
                shutil.rmtree(path)
            print(f"삭제됨: {backup['name']}")

# 사용 예제
backup = BackupUtility('important_data', 'backups')

# 백업 생성
backup_file = backup.create_backup(compress=True)

# 백업 목록 확인
print("\n=== 백업 목록 ===")
for b in backup.list_backups():
    size_mb = b['size'] / (1024 * 1024)
    print(f"{b['name']} - {size_mb:.2f}MB - {b['created']}")

# 오래된 백업 삭제 (최근 5개만 유지)
backup.delete_old_backups(keep_count=5)

예제 5: 텍스트 파일 통계 분석기​

from pathlib import Path
from collections import Counter
import re

class TextAnalyzer:
    def __init__(self, file_path):
        self.file_path = Path(file_path)
        self.text = ""
        self.lines = []
        self.words = []

    def load(self):
        """파일 로드"""
        self.text = self.file_path.read_text(encoding='utf-8')
        self.lines = self.text.splitlines()

        # 단어 추출 (한글, 영문, 숫자만)
        self.words = re.findall(r'[가-힣a-zA-Z0-9]+', self.text)
        return self

    def get_stats(self):
        """기본 통계 정보"""
        return {
            'characters': len(self.text),
            'characters_no_space': len(self.text.replace(' ', '')),
            'lines': len(self.lines),
            'words': len(self.words),
            'unique_words': len(set(self.words))
        }

    def get_word_frequency(self, top_n=10):
        """단어 빈도수"""
        counter = Counter(self.words)
        return counter.most_common(top_n)

    def get_line_stats(self):
        """줄 통계"""
        line_lengths = [len(line) for line in self.lines]
        return {
            'max_length': max(line_lengths) if line_lengths else 0,
            'min_length': min(line_lengths) if line_lengths else 0,
            'avg_length': sum(line_lengths) / len(line_lengths) if line_lengths else 0,
            'empty_lines': sum(1 for line in self.lines if not line.strip())
        }

    def search(self, pattern, case_sensitive=False):
        """패턴 검색"""
        flags = 0 if case_sensitive else re.IGNORECASE
        matches = []

        for i, line in enumerate(self.lines, 1):
            if re.search(pattern, line, flags):
                matches.append({
                    'line_number': i,
                    'content': line.strip()
                })

        return matches

    def save_report(self, output_file):
        """분석 보고서 저장"""
        stats = self.get_stats()
        line_stats = self.get_line_stats()
        word_freq = self.get_word_frequency(20)

        report = [
            "=== 텍스트 분석 보고서 ===",
            f"파일: {self.file_path}",
            f"분석 시간: {datetime.now()}",
            "",
            "=== 기본 통계 ===",
            f"총 문자 수: {stats['characters']:,}",
            f"공백 제외 문자 수: {stats['characters_no_space']:,}",
            f"총 줄 수: {stats['lines']:,}",
            f"총 단어 수: {stats['words']:,}",
            f"고유 단어 수: {stats['unique_words']:,}",
            "",
            "=== 줄 통계 ===",
            f"최대 줄 길이: {line_stats['max_length']}",
            f"최소 줄 길이: {line_stats['min_length']}",
            f"평균 줄 길이: {line_stats['avg_length']:.2f}",
            f"빈 줄: {line_stats['empty_lines']}",
            "",
            "=== 상위 20개 단어 ===",
        ]

        for word, count in word_freq:
            report.append(f"{word}: {count}회")

        Path(output_file).write_text('\n'.join(report), encoding='utf-8')

# 사용 예제

analyzer.load()

# 통계 출력

print(f"총 단어: {stats['words']:,}개")
print(f"고유 단어: {stats['unique_words']:,}개")

# 단어 빈도수

for word, count in analyzer.get_word_frequency(10):
    print(f"{word}: {count}회")

# 특정 단어 검색

print(f"\n'python' 검색 결과: {len(matches)}개")

# 보고서 저장

문제 1: 단어 카운터​

다음 기능을 가진 단어 카운터를 작성하세요:

  • 텍스트 파일을 읽어서 각 단어의 빈도수 계산
  • 가장 많이 사용된 단어 10개 출력
  • 결과를 JSON 파일로 저장
해답 보기
import json
from collections import Counter
import re
from pathlib import Path

def word_counter(input_file, output_file='word_count.json', top_n=10):
    """단어 빈도수를 계산하고 결과를 저장"""
    # 파일 읽기
    file_path = Path(input_file)
    text = file_path.read_text(encoding='utf-8')

    # 단어 추출 (영문, 한글만)
    words = re.findall(r'[a-zA-Z가-힣]+', text.lower())

    # 빈도수 계산
    word_freq = Counter(words)

    # 상위 N개 단어
    top_words = dict(word_freq.most_common(top_n))

    # 결과 출력
    print(f"=== 상위 {top_n}개 단어 ===")
    for word, count in top_words.items():
        print(f"{word}: {count}회")

    # JSON으로 저장
    result = {
        'total_words': len(words),
        'unique_words': len(word_freq),
        'top_words': top_words
    }

    output_path = Path(output_file)
    output_path.write_text(
        json.dumps(result, ensure_ascii=False, indent=2),
        encoding='utf-8'
    )

    print(f"\n결과가 {output_file}에 저장되었습니다.")

# 사용 예제

문제 2: CSV 병합기​

여러 CSV 파일을 하나로 병합하는 프로그램을 작성하세요:

  • 같은 구조를 가진 여러 CSV 파일을 입력받음
  • 헤더는 한 번만 포함
  • 중복 행 제거 옵션 제공
해답 보기
import csv
from pathlib import Path

def merge_csv_files(input_files, output_file, remove_duplicates=True):
    """여러 CSV 파일을 하나로 병합"""
    all_rows = []
    headers = None

    # 모든 CSV 파일 읽기
    for file_path in input_files:
        path = Path(file_path)
        if not path.exists():
            print(f"경고: {file_path}을 찾을 수 없습니다.")
            continue

        with path.open('r', encoding='utf-8') as f:
            reader = csv.DictReader(f)

            # 첫 번째 파일의 헤더 저장
            if headers is None:
                headers = reader.fieldnames

            # 헤더가 다르면 경고
            if reader.fieldnames != headers:
                print(f"경고: {file_path}의 헤더가 다릅니다.")
                continue

            # 행 추가
            for row in reader:
                all_rows.append(row)

    # 중복 제거 (선택사항)
    if remove_duplicates:
        # dict를 tuple로 변환하여 중복 제거
        unique_rows = []
        seen = set()

        for row in all_rows:
            row_tuple = tuple(sorted(row.items()))
            if row_tuple not in seen:
                seen.add(row_tuple)
                unique_rows.append(row)

        all_rows = unique_rows
        print(f"중복 제거: {len(seen)}개의 고유한 행")

    # 병합된 파일 저장
    output_path = Path(output_file)
    with output_path.open('w', newline='', encoding='utf-8') as f:
        if headers:
            writer = csv.DictWriter(f, fieldnames=headers)
            writer.writeheader()
            writer.writerows(all_rows)

    print(f"총 {len(all_rows)}개 행이 {output_file}에 저장되었습니다.")

# 사용 예제
input_files = ['data1.csv', 'data2.csv', 'data3.csv']
merge_csv_files(input_files, 'merged.csv', remove_duplicates=True)

문제 3: 설정 파일 검증기​

JSON 설정 파일을 읽고 필수 키가 있는지 검증하는 프로그램을 작성하세요:

  • 필수 키 목록을 받아서 검증
  • 누락된 키가 있으면 기본값으로 채우기
  • 검증 결과를 새 파일로 저장
해답 보기
import json
from pathlib import Path

def validate_config(config_file, required_keys, defaults, output_file=None):
    """설정 파일 검증 및 수정"""
    config_path = Path(config_file)

    # 설정 파일 읽기
    if config_path.exists():
        config = json.loads(config_path.read_text(encoding='utf-8'))
    else:
        print("설정 파일이 없습니다. 새로 생성합니다.")
        config = {}

    # 검증 결과
    validation_result = {
        'valid': True,
        'missing_keys': [],
        'added_defaults': []
    }

    # 필수 키 확인
    for key in required_keys:
        if key not in config:
            validation_result['valid'] = False
            validation_result['missing_keys'].append(key)

            # 기본값으로 채우기
            if key in defaults:
                config[key] = defaults[key]
                validation_result['added_defaults'].append(key)

    # 결과 출력
    if validation_result['valid']:
        print("설정 파일이 유효합니다.")
    else:
        print("설정 파일에 누락된 키가 있습니다:")
        for key in validation_result['missing_keys']:
            print(f"  - {key}")

        if validation_result['added_defaults']:
            print("\n다음 키에 기본값이 추가되었습니다:")
            for key in validation_result['added_defaults']:
                print(f"  - {key}: {config[key]}")

    # 수정된 설정 저장
    output_path = Path(output_file or config_file)
    output_path.write_text(
        json.dumps(config, ensure_ascii=False, indent=2),
        encoding='utf-8'
    )

    print(f"\n설정이 {output_path}에 저장되었습니다.")
    return validation_result

# 사용 예제
required_keys = ['app_name', 'version', 'database', 'port']
defaults = {
    'app_name': 'MyApp',
    'version': '1.0.0',
    'port': 8080,
    'database': {
        'host': 'localhost',
        'port': 5432
    }
}

result = validate_config(
    'config.json',
    required_keys,
    defaults,
    'validated_config.json'
)

문제 4: 로그 파일 로테이션​

로그 파일이 일정 크기를 초과하면 자동으로 백업하고 새 파일을 시작하는 로거를 작성하세요:

  • 최대 파일 크기 지정 (예: 1MB)
  • 백업 파일은 번호를 붙여서 저장 (log.1, log.2, ...)
  • 최대 백업 개수 제한
해답 보기
from pathlib import Path
from datetime import datetime

class RotatingLogger:
    def __init__(self, log_file, max_bytes=1024*1024, backup_count=5):
        """
        Args:
            log_file: 로그 파일 경로
            max_bytes: 최대 파일 크기 (기본 1MB)
            backup_count: 최대 백업 파일 개수
        """
        self.log_file = Path(log_file)
        self.max_bytes = max_bytes
        self.backup_count = backup_count

    def _should_rotate(self):
        """파일 크기 확인하여 로테이션 필요 여부 반환"""
        if not self.log_file.exists():
            return False
        return self.log_file.stat().st_size >= self.max_bytes

    def _rotate(self):
        """로그 파일 로테이션 수행"""
        # 가장 오래된 백업 삭제
        oldest = self.log_file.with_suffix(f'.{self.backup_count}')
        if oldest.exists():
            oldest.unlink()

        # 백업 파일들 번호 증가
        for i in range(self.backup_count - 1, 0, -1):
            src = self.log_file.with_suffix(f'.{i}')
            dst = self.log_file.with_suffix(f'.{i + 1}')
            if src.exists():
                src.rename(dst)

        # 현재 로그 파일을 .1로 백업
        if self.log_file.exists():
            self.log_file.rename(self.log_file.with_suffix('.1'))

    def log(self, message, level='INFO'):
        """로그 메시지 작성"""
        # 로테이션 필요 여부 확인
        if self._should_rotate():
            self._rotate()

        # 로그 메시지 포맷
        timestamp = datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        log_entry = f"[{timestamp}] [{level}] {message}\n"

        # 파일에 추가
        with self.log_file.open('a', encoding='utf-8') as f:
            f.write(log_entry)

    def info(self, message):
        """INFO 레벨 로그"""
        self.log(message, 'INFO')

    def warning(self, message):
        """WARNING 레벨 로그"""
        self.log(message, 'WARNING')

    def error(self, message):
        """ERROR 레벨 로그"""
        self.log(message, 'ERROR')

# 사용 예제
logger = RotatingLogger('app.log', max_bytes=1024*100, backup_count=3)

# 로그 작성
for i in range(1000):
    logger.info(f"처리 중: {i}번째 작업")
    if i % 100 == 0:
        logger.warning(f"체크포인트: {i}")

logger.error("테스트 에러 메시지")
