5. 자동화

반복적인 작업을 Python으로 자동화하는 방법을 배워봅니다.

5.1 자동화란?​

자동화는 반복적이고 시간이 많이 걸리는 작업을 프로그램으로 처리하는 것입니다.

자동화하면 좋은 작업​

✅ **파일 관리**
   - 파일 정리, 백업, 이름 변경

✅ **데이터 처리**
   - Excel 보고서 생성
   - CSV 데이터 변환

✅ **통신**
   - 이메일 발송
   - 메시지 전송

✅ **모니터링**
   - 웹사이트 체크
   - 로그 분석

자동화하지 말아야 할 작업​

❌ 한 번만 실행할 작업
❌ 자동화 시간 > 수동 작업 시간
❌ 너무 복잡한 판단이 필요한 작업

5.2 파일 작업 자동화​

파일 정리​

# file_organizer.py
import os
import shutil
from pathlib import Path

def organize_downloads(source_dir):
    """다운로드 폴더 정리"""

    # 확장자별 폴더 매핑
    file_types = {
        'Images': ['.jpg', '.jpeg', '.png', '.gif', '.bmp'],
        'Documents': ['.pdf', '.docx', '.txt', '.xlsx'],
        'Videos': ['.mp4', '.avi', '.mkv'],
        'Archives': ['.zip', '.rar', '.7z'],
    }

    source = Path(source_dir)

    for file in source.iterdir():
        if file.is_file():
            # 확장자 확인
            ext = file.suffix.lower()

            # 해당하는 폴더 찾기
            for folder, extensions in file_types.items():
                if ext in extensions:
                    # 폴더 생성
                    dest_folder = source / folder
                    dest_folder.mkdir(exist_ok=True)

                    # 파일 이동
                    shutil.move(str(file), str(dest_folder / file.name))
                    print(f"Moved: {file.name} → {folder}/")
                    break

# 사용
organize_downloads(r"C:\Users\YourName\Downloads")

백업 자동화​

# backup.py
import shutil
from datetime import datetime
from pathlib import Path

def create_backup(source_dir, backup_dir):
    """폴더 백업"""

    # 타임스탬프로 백업 폴더명 생성
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    backup_name = f"backup_{timestamp}"

    source = Path(source_dir)
    backup = Path(backup_dir) / backup_name

    try:
        # 전체 폴더 복사
        shutil.copytree(source, backup)
        print(f"✅ Backup created: {backup}")
        return backup

    except Exception as e:
        print(f"❌ Backup failed: {e}")
        return None

# 사용
create_backup(
    r"C:\Projects\MyApp",
    r"D:\Backups"
)

5.3 Excel 자동화​

보고서 생성​

# excel_report.py
import pandas as pd
from openpyxl import load_workbook
from openpyxl.styles import Font, PatternFill
from openpyxl.chart import BarChart, Reference

def create_sales_report(data_file, output_file):
    """판매 보고서 생성"""

    # 데이터 읽기
    df = pd.read_csv(data_file)

    # 기본 통계
    summary = df.groupby('product').agg({
        'quantity': 'sum',
        'revenue': 'sum'
    }).reset_index()

    # Excel로 저장
    with pd.ExcelWriter(output_file, engine='openpyxl') as writer:
        df.to_excel(writer, sheet_name='Raw Data', index=False)
        summary.to_excel(writer, sheet_name='Summary', index=False)

    # 스타일 적용
    wb = load_workbook(output_file)
    ws = wb['Summary']

    # 헤더 스타일
    header_fill = PatternFill(start_color="4472C4", fill_type="solid")
    header_font = Font(color="FFFFFF", bold=True)

    for cell in ws[1]:
        cell.fill = header_fill
        cell.font = header_font

    # 차트 추가
    chart = BarChart()
    chart.title = "Product Sales"
    chart.x_axis.title = "Product"
    chart.y_axis.title = "Revenue"

    data = Reference(ws, min_col=3, min_row=1, max_row=ws.max_row)
    categories = Reference(ws, min_col=1, min_row=2, max_row=ws.max_row)

    chart.add_data(data, titles_from_data=True)
    chart.set_categories(categories)

    ws.add_chart(chart, "E2")

    wb.save(output_file)
    print(f"✅ Report created: {output_file}")

# 사용
create_sales_report('sales_data.csv', 'sales_report.xlsx')

여러 Excel 파일 병합​

# merge_excel.py
import pandas as pd
from pathlib import Path

def merge_excel_files(folder_path, output_file):
    """폴더 내 모든 Excel 파일 병합"""

    all_data = []

    for file in Path(folder_path).glob('*.xlsx'):
        df = pd.read_excel(file)
        df['source_file'] = file.name
        all_data.append(df)

    # 병합
    merged_df = pd.concat(all_data, ignore_index=True)

    # 저장
    merged_df.to_excel(output_file, index=False)
    print(f"✅ Merged {len(all_data)} files into {output_file}")

# 사용
merge_excel_files('monthly_reports', 'yearly_report.xlsx')

5.4 이메일 자동화​

이메일 발송​

# email_sender.py
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.application import MIMEApplication
from pathlib import Path

class EmailSender:
    """이메일 발송 클래스"""

    def __init__(self, smtp_server, port, email, password):
        self.smtp_server = smtp_server
        self.port = port
        self.email = email
        self.password = password

    def send_email(self, to_email, subject, body, attachments=None):
        """이메일 발송"""

        # 메시지 생성
        msg = MIMEMultipart()
        msg['From'] = self.email
        msg['To'] = to_email
        msg['Subject'] = subject

        # 본문 추가
        msg.attach(MIMEText(body, 'plain'))

        # 첨부 파일
        if attachments:
            for file_path in attachments:
                with open(file_path, 'rb') as f:
                    part = MIMEApplication(f.read(), Name=Path(file_path).name)
                    part['Content-Disposition'] = f'attachment; filename="{Path(file_path).name}"'
                    msg.attach(part)

        # 발송
        try:
            with smtplib.SMTP(self.smtp_server, self.port) as server:
                server.starttls()
                server.login(self.email, self.password)
                server.send_message(msg)
                print(f"✅ Email sent to {to_email}")

        except Exception as e:
            print(f"❌ Failed to send email: {e}")

# 사용
sender = EmailSender(
    smtp_server='smtp.gmail.com',
    port=587,
    email='your_email@gmail.com',
    password='your_app_password'
)

sender.send_email(
    to_email='recipient@example.com',
    subject='Weekly Report',
    body='Please find the attached weekly report.',
    attachments=['sales_report.xlsx']
)

import schedule
import time
from datetime import datetime

def job():
    """실행할 작업"""
    print(f"Task executed at {datetime.now()}")

schedule.every(10).seconds.do(job)  # 10초마다
schedule.every().hour.do(job)  # 매 시간
schedule.every().day.at("09:00").do(job)  # 매일 오전 9시
schedule.every().monday.at("10:00").do(job)  # 매주 월요일 10시

# 실행
while True:
    schedule.run_pending()
    time.sleep(1)

Selenium 기본​

# web_automation.py
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time

def automate_login(url, username, password):
    """로그인 자동화"""

    # 드라이버 설정
    options = webdriver.ChromeOptions()
    options.add_argument('--headless')  # 백그라운드 실행

    driver = webdriver.Chrome(options=options)

    try:
        # 페이지 열기
        driver.get(url)

        # 요소 찾기 및 입력
        username_field = driver.find_element(By.ID, "username")
        password_field = driver.find_element(By.ID, "password")

        username_field.send_keys(username)
        password_field.send_keys(password)

        # 로그인 버튼 클릭
        login_button = driver.find_element(By.ID, "login-button")
        login_button.click()

        # 로그인 완료 대기
        WebDriverWait(driver, 10).until(
            EC.presence_of_element_located((By.ID, "dashboard"))
        )

        print("✅ Login successful")

    finally:
        driver.quit()

문제 1: 파일 정리 스크립트​

다운로드 폴더의 파일을 확장자별로 정리하는 스크립트를 작성하세요.

해답 보기
import shutil
from pathlib import Path

def organize_files(source_dir):
    """파일 정리"""

    extensions = {
        'images': ['.jpg', '.png', '.gif'],
        'docs': ['.pdf', '.docx', '.txt'],
        'videos': ['.mp4', '.avi']
    }

    for file in Path(source_dir).iterdir():
        if file.is_file():
            ext = file.suffix.lower()

            for folder, exts in extensions.items():
                if ext in exts:
                    dest = Path(source_dir) / folder
                    dest.mkdir(exist_ok=True)
                    shutil.move(str(file), str(dest / file.name))
                    break
