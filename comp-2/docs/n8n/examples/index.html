const content = $input.first().json.content;
const platforms = ['twitter', 'linkedin', 'facebook'];

const adaptContent = (text, platform) => {
  const maxLengths = {
    twitter: 280,
    linkedin: 3000,
    facebook: 63206
  };

  const hashtags = {
    twitter: '#tech #automation',
    linkedin: '',
    facebook: '#tech'
  };

  let adapted = text;
  if (adapted.length > maxLengths[platform]) {
    adapted = adapted.substring(0, maxLengths[platform] - 3) + '...';
  }

  return `${adapted} ${hashtags[platform]}`;
};

return platforms.map(platform => ({
  json: {
    platform: platform,
    content: adaptContent(content, platform),
    scheduledTime: $input.first().json.scheduledTime
  }
}));

// Function 노드: CSV 데이터 변환 및 검증
const items = $input.all();

const validateEmail = (email) => {
  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return re.test(email);
};

const cleanPhone = (phone) => {
  return phone.replace(/[^0-9]/g, '');
};

const validItems = [];
const errorItems = [];

items.forEach(item => {
  const data = item.json;

  // 검증
  const errors = [];

  if (!data.name || data.name.trim() === '') {
    errors.push('Name is required');
  }

  if (!validateEmail(data.email)) {
    errors.push('Invalid email format');
  }

  if (errors.length > 0) {
    errorItems.push({
      json: {
        ...data,
        errors: errors
      }
    });
  } else {
    validItems.push({
      json: {
        name: data.name.trim(),
        email: data.email.toLowerCase(),
        phone: cleanPhone(data.phone || ''),
        company: data.company || null,
        imported_at: new Date().toISOString()
      }
    });
  }
});

// 에러 로그 저장
if (errorItems.length > 0) {
  $execution.customData.set('errors', errorItems);
}

return validItems;

6.7 Python으로 n8n 관리​

워크플로우 백업 및 배포​

# n8n_manager.py
import requests
import json
from pathlib import Path
from datetime import datetime

class N8nManager:
    """n8n 워크플로우 관리 클래스"""

    def __init__(self, base_url, api_key):
        self.base_url = base_url
        self.headers = {
            'X-N8N-API-KEY': api_key,
            'Content-Type': 'application/json'
        }

    def backup_all_workflows(self, backup_dir='./backups'):
        """모든 워크플로우 백업"""
        # 백업 디렉토리 생성
        backup_path = Path(backup_dir)
        backup_path.mkdir(exist_ok=True)

        # 워크플로우 목록 가져오기
        response = requests.get(
            f'{self.base_url}/workflows',
            headers=self.headers
        )
        workflows = response.json()['data']

        # 각 워크플로우 백업
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')

        for workflow in workflows:
            filename = f"{workflow['name']}_{timestamp}.json"
            filepath = backup_path / filename

            with open(filepath, 'w', encoding='utf-8') as f:
                json.dump(workflow, f, indent=2, ensure_ascii=False)

            print(f"✅ Backed up: {filename}")

        print(f"\n📦 Total: {len(workflows)} workflows backed up")

    def deploy_workflow(self, filepath, activate=True):
        """워크플로우 배포"""
        with open(filepath, 'r', encoding='utf-8') as f:
            workflow = json.load(f)

        # ID 제거 (새로 생성)
        if 'id' in workflow:
            del workflow['id']

        # 워크플로우 생성
        response = requests.post(
            f'{self.base_url}/workflows',
            headers=self.headers,
            json=workflow
        )

        if response.status_code == 201:
            workflow_id = response.json()['id']
            print(f"✅ Deployed: {workflow['name']} (ID: {workflow_id})")

            # 활성화
            if activate:
                self.activate_workflow(workflow_id)

            return workflow_id
        else:
            print(f"❌ Failed to deploy: {response.text}")
            return None

    def activate_workflow(self, workflow_id):
        """워크플로우 활성화"""
        response = requests.patch(
            f'{self.base_url}/workflows/{workflow_id}',
            headers=self.headers,
            json={'active': True}
        )

        if response.status_code == 200:
            print(f"✅ Activated workflow: {workflow_id}")
        else:
            print(f"❌ Failed to activate: {response.text}")

    def sync_workflows(self, source_env, target_env):
        """환경 간 워크플로우 동기화"""
        # Source에서 백업
        print(f"📥 Backing up from {source_env}...")
        self.base_url = source_env
        self.backup_all_workflows('./temp_sync')

        # Target으로 배포
        print(f"\n📤 Deploying to {target_env}...")
        self.base_url = target_env

        for filepath in Path('./temp_sync').glob('*.json'):
            self.deploy_workflow(filepath)

        print("\n✅ Sync completed")

// Function 노드: 견고한 에러 처리
async function robustExecution() {
  const MAX_RETRIES = 3;
  const RETRY_DELAY = 1000; // 1초

  for (let attempt = 1; attempt <= MAX_RETRIES; attempt++) {
    try {
      // 실제 작업
      const result = await performTask();
      return [{json: result}];

    } catch (error) {
      console.error(`Attempt ${attempt} failed:`, error.message);

      if (attempt === MAX_RETRIES) {
        // 최종 실패 - 알림 발송
        await $http.post('https://hooks.slack.com/xxx', {
          text: `🚨 Workflow failed after ${MAX_RETRIES} attempts: ${error.message}`
        });

        throw error;
      }

      // 지수 백오프로 재시도
      await new Promise(resolve =>
        setTimeout(resolve, RETRY_DELAY * Math.pow(2, attempt - 1))
      );
    }
  }
}

return await robustExecution();

6.11 트러블슈팅 가이드​

일반적인 문제​

**문제: Workflow 실행이 느림**
해결:
- Queue 모드 활성화
- 배치 처리 사용
- 불필요한 노드 제거
- Database 인덱스 확인

**문제: Memory 부족**
해결:
- Split in Batches 사용
- Binary 데이터 스트리밍
- Worker 메모리 증가
- Pagination 활용

**문제: Webhook timeout**
해결:
- Async 처리 사용
- Response 즉시 반환
- Background job으로 처리
- Timeout 시간 증가
