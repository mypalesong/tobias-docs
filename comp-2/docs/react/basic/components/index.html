컴포넌트 만들기

컴포넌트는 React의 핵심입니다. UI를 독립적이고 재사용 가능한 부분으로 나눌 수 있습니다.

컴포넌트란?​

컴포넌트는 독립적이고 재사용 가능한 코드 조각입니다. JavaScript 함수와 유사하게 동작하며, props를 입력으로 받아 React 엘리먼트를 반환합니다.

함수 컴포넌트​

가장 간단한 컴포넌트 정의 방법입니다.

function Welcome() {
  return <h1>안녕하세요!</h1>;
}

화살표 함수​

const Welcome = () => {
  return <h1>안녕하세요!</h1>;
};

// 한 줄로 표현
const Welcome = () => <h1>안녕하세요!</h1>;

클래스 컴포넌트​

과거에 주로 사용되던 방식입니다. (현재는 함수 컴포넌트 권장)

import React, { Component } from 'react';

class Welcome extends Component {
  render() {
    return <h1>안녕하세요!</h1>;
  }
}

컴포넌트 렌더링​

컴포넌트 사용하기​

function Welcome() {
  return <h1>안녕하세요!</h1>;
}

function App() {
  return (
    <div>
      <Welcome />
      <Welcome />
      <Welcome />
    </div>
  );
}

컴포넌트 합성​

컴포넌트는 다른 컴포넌트를 포함할 수 있습니다.

function Avatar() {
  return <img src="avatar.jpg" alt="프로필" />;
}

function UserInfo() {
  return (
    <div className="user-info">
      <Avatar />
      <h3>홍길동</h3>
    </div>
  );
}

function App() {
  return (
    <div>
      <UserInfo />
    </div>
  );
}

컴포넌트 추출​

큰 컴포넌트를 작은 컴포넌트로 나누는 것이 좋습니다.

Before: 큰 컴포넌트​

function Comment({ author, text, date }) {
  return (
    <div className="comment">
      <div className="user-info">
        <img src={author.avatarUrl} alt={author.name} />
        <div className="user-name">{author.name}</div>
      </div>
      <div className="comment-text">{text}</div>
      <div className="comment-date">{date}</div>
    </div>
  );
}

After: 작은 컴포넌트들로 분리​

function Avatar({ user }) {
  return (
    <img src={user.avatarUrl} alt={user.name} />
  );
}

function UserInfo({ user }) {
  return (
    <div className="user-info">
      <Avatar user={user} />
      <div className="user-name">{user.name}</div>
    </div>
  );
}

function Comment({ author, text, date }) {
  return (
    <div className="comment">
      <UserInfo user={author} />
      <div className="comment-text">{text}</div>
      <div className="comment-date">{date}</div>
    </div>
  );
}

컴포넌트 네이밍​

규칙​

  • 컴포넌트 이름은 대문자로 시작
  • camelCase 대신 PascalCase 사용
  • 설명적인 이름 사용
// ✅ 올바른 예
function UserProfile() { }
function ShoppingCart() { }
function NavigationBar() { }

// ❌ 잘못된 예
function userProfile() { }  // 소문자 시작
function cart() { }         // 너무 짧음
function nav() { }          // 축약형

파일 구조​

컴포넌트별 파일 분리​

src/
├── components/
│   ├── Avatar.js
│   ├── UserInfo.js
│   └── Comment.js
├── App.js
└── index.js

Avatar.js​

export default function Avatar({ user }) {
  return (
    <img
      src={user.avatarUrl}
      alt={user.name}
      className="avatar"
    />
  );
}

사용하기​

import Avatar from './components/Avatar';

function App() {
  const user = {
    name: '홍길동',
    avatarUrl: 'avatar.jpg'
  };

  return <Avatar user={user} />;
}

Export/Import 방식​

Default Export​

// Avatar.js
export default function Avatar() {
  return <img src="avatar.jpg" />;
}

// App.js
import Avatar from './components/Avatar';

Named Export​

// Components.js
export function Avatar() {
  return <img src="avatar.jpg" />;
}

export function UserInfo() {
  return <div>User Info</div>;
}

// App.js
import { Avatar, UserInfo } from './components/Components';

컴포넌트 설계 원칙​

1. 단일 책임 원칙​

컴포넌트는 하나의 역할만 수행해야 합니다.

// ❌ 너무 많은 책임
function UserDashboard() {
  // 사용자 정보, 통계, 설정, 알림 등 모두 처리
}

// ✅ 책임 분리
function UserProfile() { }
function UserStats() { }
function UserSettings() { }
function UserNotifications() { }

2. 재사용성​

여러 곳에서 사용할 수 있도록 설계합니다.

// ✅ 재사용 가능한 버튼 컴포넌트
function Button({ children, onClick, variant = 'primary' }) {
  return (
    <button
      className={`btn btn-${variant}`}
      onClick={onClick}
    >
      {children}
    </button>
  );
}

// 사용
<Button variant="primary">저장</Button>
<Button variant="secondary">취소</Button>
<Button variant="danger">삭제</Button>

3. 작고 명확하게​

컴포넌트는 가능한 한 작고 명확해야 합니다.

실전 예제​

Todo 아이템 컴포넌트​

function TodoItem({ todo, onToggle, onDelete }) {
  return (
    <li className={todo.completed ? 'completed' : ''}>
      <input
        type="checkbox"
        checked={todo.completed}
        onChange={() => onToggle(todo.id)}
      />
      <span>{todo.text}</span>
      <button onClick={() => onDelete(todo.id)}>삭제</button>
    </li>
  );
}

function TodoList({ todos, onToggle, onDelete }) {
  return (
    <ul>
      {todos.map(todo => (
        <TodoItem
          key={todo.id}
          todo={todo}
          onToggle={onToggle}
          onDelete={onDelete}
        />
      ))}
    </ul>
  );
}

카드 컴포넌트​

function Card({ title, image, description, footer }) {
  return (
    <div className="card">
      {image && (
        <div className="card-image">
          <img src={image} alt={title} />
        </div>
      )}
      <div className="card-body">
        <h3>{title}</h3>
        <p>{description}</p>
      </div>
      {footer && (
        <div className="card-footer">
          {footer}
        </div>
      )}
    </div>
  );
}

// 사용
<Card
  title="React 학습"
  image="react-logo.png"
  description="React의 기초를 배워봅시다"
  footer={<button>자세히 보기</button>}
/>
