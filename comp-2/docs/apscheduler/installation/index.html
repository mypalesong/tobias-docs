1. 설치 및 설정

APScheduler(Advanced Python Scheduler)는 Python 애플리케이션에서 작업을 예약하고 실행할 수 있는 강력하고 유연한 라이브러리입니다. 이 문서에서는 APScheduler의 설치 방법과 기본 설정에 대해 상세히 다룹니다.

설치​

기본 설치​

APScheduler는 pip를 통해 간단하게 설치할 수 있습니다.

pip install apscheduler

특정 버전 설치​

안정성을 위해 특정 버전을 설치하는 것을 권장합니다.

# 최신 3.x 버전 설치
pip install apscheduler==3.10.4

# 버전 범위 지정
pip install "apscheduler>=3.9.0,<4.0.0"

추가 의존성 설치​

APScheduler는 다양한 백엔드를 지원하며, 각 백엔드에 필요한 추가 패키지가 있습니다.

# SQLAlchemy 지원 (데이터베이스 job store)
pip install apscheduler[sqlalchemy]

# MongoDB 지원
pip install apscheduler[mongodb]

# Redis 지원
pip install apscheduler[redis]

# Tornado 지원
pip install apscheduler[tornado]

# Twisted 지원
pip install apscheduler[twisted]

# 모든 의존성 설치
pip install apscheduler[all]

개발 환경 설정​

프로젝트에서 APScheduler를 사용할 때는 requirements.txt 파일에 명시하는 것이 좋습니다.

requirements.txt
apscheduler==3.10.4
pytz==2023.3  # 타임존 지원
tzlocal==5.0.1  # 로컬 타임존 자동 감지

Poetry를 사용하는 경우:

poetry add apscheduler
poetry add apscheduler --extras sqlalchemy  # SQLAlchemy 지원 포함

설치 확인​

설치가 정상적으로 완료되었는지 확인합니다.

import apscheduler
print(apscheduler.__version__)  # 3.10.4

더 상세한 버전 정보 확인:

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from apscheduler.executors.pool import ThreadPoolExecutor

scheduler = BackgroundScheduler()
print(f"APScheduler 버전: {apscheduler.__version__}")
print(f"스케줄러 타입: {type(scheduler).__name__}")

첫 스케줄러​

APScheduler를 처음 사용할 때는 간단한 예제부터 시작하는 것이 좋습니다.

가장 간단한 예제​

from apscheduler.schedulers.blocking import BlockingScheduler
from datetime import datetime

def my_job():
    print(f"작업 실행 시간: {datetime.now()}")

# 스케줄러 생성
scheduler = BlockingScheduler()

# 3초마다 작업 실행
scheduler.add_job(my_job, 'interval', seconds=3)

# 스케줄러 시작
print("스케줄러 시작...")
scheduler.start()

BackgroundScheduler 사용 예제​

애플리케이션이 계속 실행되어야 할 때는 BackgroundScheduler를 사용합니다.

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

def tick():
    print(f'틱! 현재 시각: {datetime.now()}')

# 스케줄러 생성
scheduler = BackgroundScheduler()

# 5초마다 실행
scheduler.add_job(tick, 'interval', seconds=5)

# 스케줄러 시작
scheduler.start()

print("스케줄러가 백그라운드에서 실행 중입니다. Ctrl+C를 눌러 종료하세요.")

try:
    # 메인 스레드는 계속 실행
    while True:
        time.sleep(1)
except (KeyboardInterrupt, SystemExit):
    # 종료 시 스케줄러도 정리
    scheduler.shutdown()
    print("스케줄러 종료됨")

데코레이터를 사용한 간편한 작업 추가​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

scheduler = BackgroundScheduler()

@scheduler.scheduled_job('interval', seconds=10)
def scheduled_task():
    print(f"예약된 작업 실행: {datetime.now()}")

@scheduler.scheduled_job('cron', hour=9, minute=0)
def morning_task():
    print("좋은 아침입니다!")

scheduler.start()

try:
    while True:
        time.sleep(1)
except (KeyboardInterrupt, SystemExit):
    scheduler.shutdown()

특정 시간에 한 번만 실행​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta
import time

def one_time_job():
    print(f"일회성 작업 실행: {datetime.now()}")

scheduler = BackgroundScheduler()

# 10초 후에 한 번만 실행
run_time = datetime.now() + timedelta(seconds=10)
scheduler.add_job(one_time_job, 'date', run_date=run_time)

scheduler.start()

print(f"작업이 {run_time}에 실행될 예정입니다.")

try:
    # 20초 동안 대기
    time.sleep(20)
except (KeyboardInterrupt, SystemExit):
    pass
finally:
    scheduler.shutdown()

여러 작업 동시 관리​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

def data_backup():
    print(f"[{datetime.now()}] 데이터 백업 시작")
    # 백업 로직
    time.sleep(2)
    print(f"[{datetime.now()}] 데이터 백업 완료")

def send_report():
    print(f"[{datetime.now()}] 리포트 전송")

def cleanup_cache():
    print(f"[{datetime.now()}] 캐시 정리")

scheduler = BackgroundScheduler()

# 매시간 백업
scheduler.add_job(data_backup, 'interval', hours=1, id='backup_job')

# 매일 오전 9시 리포트 전송
scheduler.add_job(send_report, 'cron', hour=9, minute=0, id='report_job')

# 30분마다 캐시 정리
scheduler.add_job(cleanup_cache, 'interval', minutes=30, id='cleanup_job')

scheduler.start()

print("여러 작업이 스케줄링되었습니다.")
print("등록된 작업 목록:")
for job in scheduler.get_jobs():
    print(f"  - {job.id}: {job.next_run_time}")

try:
    while True:
        time.sleep(1)
except (KeyboardInterrupt, SystemExit):
    scheduler.shutdown()

기본 설정​

APScheduler의 동작을 세밀하게 제어하기 위한 다양한 설정 옵션이 있습니다.

스케줄러 설정 옵션​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from apscheduler.executors.pool import ThreadPoolExecutor, ProcessPoolExecutor
from pytz import utc

# Job Store 설정
jobstores = {
    'default': MemoryJobStore()
}

# Executor 설정
executors = {
    'default': ThreadPoolExecutor(max_workers=20),
    'processpool': ProcessPoolExecutor(max_workers=5)
}

# Job 기본 설정
job_defaults = {
    'coalesce': False,  # 누적된 실행을 하나로 합치지 않음
    'max_instances': 3,  # 동일 job의 최대 동시 실행 인스턴스 수
    'misfire_grace_time': 30  # job이 예정 시간보다 늦게 실행될 수 있는 최대 시간(초)
}

# 스케줄러 생성
scheduler = BackgroundScheduler(
    jobstores=jobstores,
    executors=executors,
    job_defaults=job_defaults,
    timezone=utc
)

설정 파일 로드:

from apscheduler.schedulers.background import BackgroundScheduler
from configparser import ConfigParser

config = ConfigParser()
config.read('scheduler.ini')

# 설정 딕셔너리로 변환
scheduler_config = {}
for section in config.sections():
    for key, value in config.items(section):
        scheduler_config[f'{section}.{key}'] = value

scheduler = BackgroundScheduler()
scheduler.configure(scheduler_config)

환경별 설정 관리​

import os
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from apscheduler.executors.pool import ThreadPoolExecutor

class SchedulerConfig:
    """환경별 스케줄러 설정"""

    @staticmethod
    def get_config(environment='development'):
        configs = {
            'development': {
                'jobstores': {
                    'default': MemoryJobStore()
                },
                'executors': {
                    'default': ThreadPoolExecutor(max_workers=10)
                },
                'job_defaults': {
                    'coalesce': True,
                    'max_instances': 1
                }
            },
            'production': {
                'jobstores': {
                    'default': SQLAlchemyJobStore(
                        url=os.getenv('DATABASE_URL', 'sqlite:///jobs.sqlite')
                    )
                },
                'executors': {
                    'default': ThreadPoolExecutor(max_workers=20)
                },
                'job_defaults': {
                    'coalesce': False,
                    'max_instances': 3,
                    'misfire_grace_time': 60
                }
            }
        }
        return configs.get(environment, configs['development'])

# 사용 예제
env = os.getenv('APP_ENV', 'development')
config = SchedulerConfig.get_config(env)

scheduler = BackgroundScheduler(**config)

Job별 개별 설정​

from apscheduler.schedulers.background import BackgroundScheduler

scheduler = BackgroundScheduler()

def critical_task():
    print("중요한 작업")

def normal_task():
    print("일반 작업")

# 중요한 작업: 놓치면 안 되므로 misfire_grace_time을 길게
scheduler.add_job(
    critical_task,
    'interval',
    minutes=10,
    id='critical',
    misfire_grace_time=300,  # 5분
    max_instances=1,
    coalesce=True
)

# 일반 작업: 놓쳐도 괜찮음
scheduler.add_job(
    normal_task,
    'interval',
    minutes=5,
    id='normal',
    misfire_grace_time=10,  # 10초
    max_instances=3,
    coalesce=False
)

scheduler.start()

구조화된 로깅 (JSON)​

import logging
import json
from datetime import datetime
from apscheduler.schedulers.background import BackgroundScheduler

class JsonFormatter(logging.Formatter):
    """JSON 형식 로그 포맷터"""

    def format(self, record):
        log_data = {
            'timestamp': datetime.utcnow().isoformat(),
            'level': record.levelname,
            'logger': record.name,
            'message': record.getMessage(),
            'module': record.module,
            'function': record.funcName,
            'line': record.lineno
        }

        if record.exc_info:
            log_data['exception'] = self.formatException(record.exc_info)

        return json.dumps(log_data, ensure_ascii=False)

# 핸들러 설정
handler = logging.FileHandler('scheduler_json.log')
handler.setFormatter(JsonFormatter())

# APScheduler 로거에 핸들러 추가
logger = logging.getLogger('apscheduler')
logger.addHandler(handler)
logger.setLevel(logging.INFO)

scheduler = BackgroundScheduler()

이벤트 기반 로깅​

import logging
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import (
    EVENT_JOB_EXECUTED, EVENT_JOB_ERROR,
    EVENT_JOB_ADDED, EVENT_JOB_REMOVED,
    EVENT_JOB_SUBMITTED, EVENT_JOB_MAX_INSTANCES
)

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def job_executed_listener(event):
    logger.info(f'Job {event.job_id} 실행 완료')

def job_error_listener(event):
    logger.error(f'Job {event.job_id} 실행 중 오류 발생: {event.exception}')

def job_added_listener(event):
    logger.info(f'Job {event.job_id} 추가됨')

def job_removed_listener(event):
    logger.info(f'Job {event.job_id} 제거됨')

def job_submitted_listener(event):
    logger.debug(f'Job {event.job_id} 제출됨')

def job_max_instances_listener(event):
    logger.warning(f'Job {event.job_id} 최대 인스턴스 수 도달')

scheduler = BackgroundScheduler()

# 이벤트 리스너 등록
scheduler.add_listener(job_executed_listener, EVENT_JOB_EXECUTED)
scheduler.add_listener(job_error_listener, EVENT_JOB_ERROR)
scheduler.add_listener(job_added_listener, EVENT_JOB_ADDED)
scheduler.add_listener(job_removed_listener, EVENT_JOB_REMOVED)
scheduler.add_listener(job_submitted_listener, EVENT_JOB_SUBMITTED)
scheduler.add_listener(job_max_instances_listener, EVENT_JOB_MAX_INSTANCES)

scheduler.start()

커스텀 로거 클래스​

import logging
from datetime import datetime
from apscheduler.schedulers.background import BackgroundScheduler

class SchedulerLogger:
    """스케줄러 전용 로거"""

    def __init__(self, name='scheduler'):
        self.logger = logging.getLogger(name)
        self._setup_logger()

    def _setup_logger(self):
        """로거 초기 설정"""
        self.logger.setLevel(logging.DEBUG)

        # 포맷터
        formatter = logging.Formatter(
            '%(asctime)s [%(levelname)8s] %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )

        # 콘솔 핸들러
        console = logging.StreamHandler()
        console.setLevel(logging.INFO)
        console.setFormatter(formatter)

        # 파일 핸들러
        file_handler = logging.FileHandler(
            f'scheduler_{datetime.now():%Y%m%d}.log'
        )
        file_handler.setLevel(logging.DEBUG)
        file_handler.setFormatter(formatter)

        self.logger.addHandler(console)
        self.logger.addHandler(file_handler)

    def log_job_start(self, job_id):
        """Job 시작 로그"""
        self.logger.info(f'Job 시작: {job_id}')

    def log_job_end(self, job_id, duration):
        """Job 종료 로그"""
        self.logger.info(f'Job 종료: {job_id} (소요시간: {duration:.2f}초)')

    def log_job_error(self, job_id, error):
        """Job 오류 로그"""
        self.logger.error(f'Job 오류: {job_id} - {error}', exc_info=True)

# 사용 예제
scheduler_logger = SchedulerLogger()

def monitored_job():
    import time
    job_id = 'test_job'
    start_time = time.time()

    try:
        scheduler_logger.log_job_start(job_id)
        # 작업 수행
        time.sleep(1)
        duration = time.time() - start_time
        scheduler_logger.log_job_end(job_id, duration)
    except Exception as e:
        scheduler_logger.log_job_error(job_id, e)
        raise

scheduler = BackgroundScheduler()
scheduler.add_job(monitored_job, 'interval', seconds=10)
scheduler.start()

프로덕션 환경 로깅 설정​

import logging
import logging.handlers
import os
from apscheduler.schedulers.background import BackgroundScheduler

def setup_production_logging():
    """프로덕션 환경 로깅 설정"""

    # 로그 디렉토리 생성
    log_dir = 'logs'
    os.makedirs(log_dir, exist_ok=True)

    # 루트 로거 설정
    root_logger = logging.getLogger()
    root_logger.setLevel(logging.INFO)

    # APScheduler 로거
    scheduler_logger = logging.getLogger('apscheduler')
    scheduler_logger.setLevel(logging.INFO)

    # 포맷터
    detailed_formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - '
        '[%(filename)s:%(lineno)d] - %(message)s'
    )

    simple_formatter = logging.Formatter(
        '%(asctime)s - %(levelname)s - %(message)s'
    )

    # 콘솔 핸들러 (간단한 형식)
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.INFO)
    console_handler.setFormatter(simple_formatter)

    # 일반 로그 파일 (Timed Rotation)
    info_handler = logging.handlers.TimedRotatingFileHandler(
        filename=os.path.join(log_dir, 'scheduler.log'),
        when='midnight',
        interval=1,
        backupCount=30,
        encoding='utf-8'
    )
    info_handler.setLevel(logging.INFO)
    info_handler.setFormatter(detailed_formatter)

    # 에러 로그 파일
    error_handler = logging.handlers.RotatingFileHandler(
        filename=os.path.join(log_dir, 'scheduler_error.log'),
        maxBytes=10*1024*1024,  # 10MB
        backupCount=10,
        encoding='utf-8'
    )
    error_handler.setLevel(logging.ERROR)
    error_handler.setFormatter(detailed_formatter)

    # 핸들러 추가
    scheduler_logger.addHandler(console_handler)
    scheduler_logger.addHandler(info_handler)
    scheduler_logger.addHandler(error_handler)

    return scheduler_logger

# 프로덕션 설정 적용
logger = setup_production_logging()

scheduler = BackgroundScheduler()

def production_job():
    logger.info("프로덕션 작업 실행")
    try:
        # 작업 로직
        pass
    except Exception as e:
        logger.error(f"작업 실행 중 오류: {e}", exc_info=True)
        raise

scheduler.add_job(production_job, 'interval', minutes=5)
scheduler.start()

class SensitiveDataFilter(logging.Filter):

    def filter(self, record):

        if hasattr(record, 'msg'):
            record.msg = str(record.msg).replace('password=', 'password=***')
        return True

class LevelRangeFilter(logging.Filter):
    """특정 레벨 범위만 허용"""

    def __init__(self, min_level, max_level):
        super().__init__()
        self.min_level = min_level
        self.max_level = max_level

    def filter(self, record):
        return self.min_level <= record.levelno <= self.max_level

# 로거 설정

level_filter = LevelRangeFilter(logging.INFO, logging.WARNING)
