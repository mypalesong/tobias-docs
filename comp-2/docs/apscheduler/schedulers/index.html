2. 스케줄러

APScheduler는 다양한 실행 환경과 사용 사례에 맞춰 여러 종류의 스케줄러를 제공합니다. 이 문서에서는 각 스케줄러의 특징과 사용법을 상세히 다룹니다.

스케줄러 개요​

APScheduler는 크게 6가지 스케줄러를 제공합니다:

  • BlockingScheduler: 스케줄러가 유일한 실행 프로세스일 때 사용
  • BackgroundScheduler: 스케줄러가 백그라운드에서 실행될 때 사용
  • AsyncIOScheduler: asyncio 이벤트 루프와 함께 사용
  • GeventScheduler: gevent 애플리케이션 내부에서 사용
  • TornadoScheduler: Tornado 애플리케이션 내부에서 사용
  • TwistedScheduler: Twisted 애플리케이션 내부에서 사용

이 문서에서는 가장 많이 사용되는 상위 3가지 스케줄러를 중점적으로 다룹니다.

BlockingScheduler​

BlockingScheduler는 스케줄러가 애플리케이션의 주요 프로세스로 실행되어야 할 때 사용합니다. start() 메서드를 호출하면 현재 스레드를 블로킹하므로, 다른 작업을 수행할 수 없습니다.

기본 사용법​

from apscheduler.schedulers.blocking import BlockingScheduler
from datetime import datetime
import time

def job_function():
    print(f"작업 실행 시간: {datetime.now()}")

# 스케줄러 생성
scheduler = BlockingScheduler()

# 작업 추가
scheduler.add_job(job_function, 'interval', seconds=5)

# 스케줄러 시작 (여기서 블로킹됨)
print("스케줄러 시작...")
try:
    scheduler.start()
except (KeyboardInterrupt, SystemExit):
    print("스케줄러 종료")

데몬 프로세스로 실행​

scheduler_daemon.py
#!/usr/bin/env python3
"""
데몬 프로세스로 실행되는 스케줄러
"""
import os
import sys
import time
import atexit
from signal import SIGTERM
from apscheduler.schedulers.blocking import BlockingScheduler

class Daemon:
    """데몬 베이스 클래스"""

    def __init__(self, pidfile):
        self.pidfile = pidfile

    def daemonize(self):
        """프로세스를 데몬화"""
        try:
            pid = os.fork()
            if pid > 0:
                sys.exit(0)
        except OSError as e:
            sys.stderr.write(f"fork #1 실패: {e}\n")
            sys.exit(1)

        os.chdir("/")
        os.setsid()
        os.umask(0)

        try:
            pid = os.fork()
            if pid > 0:
                sys.exit(0)
        except OSError as e:
            sys.stderr.write(f"fork #2 실패: {e}\n")
            sys.exit(1)

        # PID 파일 작성
        atexit.register(self.delpid)
        pid = str(os.getpid())
        with open(self.pidfile, 'w+') as f:
            f.write(f"{pid}\n")

    def delpid(self):
        """PID 파일 삭제"""
        os.remove(self.pidfile)

    def start(self):
        """데몬 시작"""
        # PID 파일 확인
        try:
            with open(self.pidfile, 'r') as pf:
                pid = int(pf.read().strip())
        except IOError:
            pid = None

        if pid:
            message = f"PID 파일 {self.pidfile}이 이미 존재합니다. 데몬이 이미 실행 중입니까?\n"
            sys.stderr.write(message)
            sys.exit(1)

        # 데몬화 및 실행
        self.daemonize()
        self.run()

    def stop(self):
        """데몬 정지"""
        try:
            with open(self.pidfile, 'r') as pf:
                pid = int(pf.read().strip())
        except IOError:
            pid = None

        if not pid:
            message = f"PID 파일 {self.pidfile}이 존재하지 않습니다. 데몬이 실행 중이 아닙니까?\n"
            sys.stderr.write(message)
            return

        # 프로세스 종료
        try:
            while True:
                os.kill(pid, SIGTERM)
                time.sleep(0.1)
        except OSError as err:
            err = str(err)
            if err.find("No such process") > 0:
                if os.path.exists(self.pidfile):
                    os.remove(self.pidfile)
            else:
                print(str(err))
                sys.exit(1)

    def restart(self):
        """데몬 재시작"""
        self.stop()
        self.start()

    def run(self):
        """실제 작업 수행 (오버라이드 필요)"""
        pass

class SchedulerDaemon(Daemon):
    """스케줄러 데몬"""

    def run(self):
        scheduler = BlockingScheduler()

        def scheduled_task():
            with open('/tmp/scheduler.log', 'a') as f:
                f.write(f"작업 실행: {time.ctime()}\n")

        scheduler.add_job(scheduled_task, 'interval', seconds=30)
        scheduler.start()

if __name__ == "__main__":
    daemon = SchedulerDaemon('/tmp/scheduler-daemon.pid')

    if len(sys.argv) == 2:
        if 'start' == sys.argv[1]:
            daemon.start()
        elif 'stop' == sys.argv[1]:
            daemon.stop()
        elif 'restart' == sys.argv[1]:
            daemon.restart()
        else:
            print("알 수 없는 명령")
            sys.exit(2)
    else:
        print(f"사용법: {sys.argv[0]} start|stop|restart")
        sys.exit(2)

BackgroundScheduler​

BackgroundScheduler는 애플리케이션의 백그라운드에서 실행되어야 할 때 사용합니다. 가장 많이 사용되는 스케줄러입니다.

기본 사용법​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

def tick():
    print(f'틱: {datetime.now()}')

# 스케줄러 생성 및 시작
scheduler = BackgroundScheduler()
scheduler.add_job(tick, 'interval', seconds=3)
scheduler.start()

print('스케줄러 실행 중. Ctrl+C로 종료.')

try:
    # 애플리케이션의 메인 로직이 여기서 실행
    while True:
        time.sleep(1)
except (KeyboardInterrupt, SystemExit):
    scheduler.shutdown()

동적 작업 관리​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta

class DynamicScheduler:
    """동적으로 작업을 추가/제거할 수 있는 스케줄러"""

    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.scheduler.start()

    def add_interval_job(self, func, interval_seconds, job_id=None):
        """인터벌 작업 추가"""
        return self.scheduler.add_job(
            func,
            'interval',
            seconds=interval_seconds,
            id=job_id,
            replace_existing=True
        )

    def add_cron_job(self, func, cron_expr, job_id=None):
        """크론 작업 추가"""
        # cron_expr 예: {'hour': 9, 'minute': 0}
        return self.scheduler.add_job(
            func,
            'cron',
            id=job_id,
            replace_existing=True,
            **cron_expr
        )

    def add_delayed_job(self, func, delay_seconds, job_id=None):
        """지연 실행 작업 추가"""
        run_date = datetime.now() + timedelta(seconds=delay_seconds)
        return self.scheduler.add_job(
            func,
            'date',
            run_date=run_date,
            id=job_id,
            replace_existing=True
        )

    def remove_job(self, job_id):
        """작업 제거"""
        try:
            self.scheduler.remove_job(job_id)
            return True
        except Exception as e:
            print(f"작업 제거 실패: {e}")
            return False

    def pause_job(self, job_id):
        """작업 일시 정지"""
        try:
            self.scheduler.pause_job(job_id)
            return True
        except Exception as e:
            print(f"작업 일시 정지 실패: {e}")
            return False

    def resume_job(self, job_id):
        """작업 재개"""
        try:
            self.scheduler.resume_job(job_id)
            return True
        except Exception as e:
            print(f"작업 재개 실패: {e}")
            return False

    def get_job(self, job_id):
        """작업 정보 조회"""
        return self.scheduler.get_job(job_id)

    def get_all_jobs(self):
        """모든 작업 조회"""
        return self.scheduler.get_jobs()

    def shutdown(self):
        """스케줄러 종료"""
        self.scheduler.shutdown()

# 사용 예제
dynamic_scheduler = DynamicScheduler()

def task1():
    print("작업 1 실행")

def task2():
    print("작업 2 실행")

# 작업 추가
dynamic_scheduler.add_interval_job(task1, 10, job_id='task1')
dynamic_scheduler.add_cron_job(task2, {'hour': 14, 'minute': 30}, job_id='task2')

# 5초 후 작업 1 일시 정지
import time
time.sleep(5)
dynamic_scheduler.pause_job('task1')

# 10초 후 작업 1 재개
time.sleep(10)
dynamic_scheduler.resume_job('task1')

AsyncIOScheduler​

AsyncIOScheduler는 asyncio를 사용하는 애플리케이션에서 사용합니다. Python 3.7 이상에서 권장됩니다.

기본 사용법​

import asyncio
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from datetime import datetime

async def async_job():
    """비동기 작업"""
    print(f"비동기 작업 실행: {datetime.now()}")
    await asyncio.sleep(1)
    print("작업 완료")

async def main():
    # 스케줄러 생성
    scheduler = AsyncIOScheduler()

    # 작업 추가
    scheduler.add_job(async_job, 'interval', seconds=5)

    # 스케줄러 시작
    scheduler.start()

    print("스케줄러 실행 중...")

    # 이벤트 루프 유지
    try:
        await asyncio.Event().wait()
    except (KeyboardInterrupt, SystemExit):
        scheduler.shutdown()

if __name__ == '__main__':
    asyncio.run(main())

비동기 HTTP 요청 스케줄링​

import asyncio
import aiohttp
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from datetime import datetime

class AsyncAPIPoller:
    """비동기 API 폴링 스케줄러"""

    def __init__(self):
        self.scheduler = AsyncIOScheduler()
        self.session = None

    async def init_session(self):
        """HTTP 세션 초기화"""
        if self.session is None:
            self.session = aiohttp.ClientSession()

    async def close_session(self):
        """HTTP 세션 종료"""
        if self.session:
            await self.session.close()

    async def fetch_api_data(self, url):
        """API 데이터 수집"""
        await self.init_session()

        try:
            async with self.session.get(url) as response:
                if response.status == 200:
                    data = await response.json()
                    print(f"[{datetime.now()}] 데이터 수집 성공: {len(data)} 항목")
                    return data
                else:
                    print(f"API 오류: {response.status}")
        except Exception as e:
            print(f"요청 실패: {e}")

    async def poll_multiple_apis(self):
        """여러 API 동시 폴링"""
        urls = [
            'https://api.example.com/data1',
            'https://api.example.com/data2',
            'https://api.example.com/data3'
        ]

        tasks = [self.fetch_api_data(url) for url in urls]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        for i, result in enumerate(results):
            if isinstance(result, Exception):
                print(f"URL {urls[i]} 처리 중 오류: {result}")

    def start(self):
        """스케줄러 시작"""
        # 5분마다 API 폴링
        self.scheduler.add_job(
            self.poll_multiple_apis,
            'interval',
            minutes=5,
            id='api_polling'
        )

        self.scheduler.start()

    def stop(self):
        """스케줄러 정지"""
        self.scheduler.shutdown()

async def main():
    poller = AsyncAPIPoller()
    poller.start()

    print("API 폴링 시작...")

    try:
        await asyncio.Event().wait()
    except KeyboardInterrupt:
        poller.stop()
        await poller.close_session()
        print("폴링 종료")

if __name__ == '__main__':
    asyncio.run(main())

시작과 종료​

스케줄러의 생명주기를 적절히 관리하는 것이 중요합니다.

기본 시작/종료​

from apscheduler.schedulers.background import BackgroundScheduler
import time

scheduler = BackgroundScheduler()

def my_job():
    print("작업 실행")

scheduler.add_job(my_job, 'interval', seconds=5)

# 스케줄러 시작
scheduler.start()
print("스케줄러 시작됨")

# 작업 수행
time.sleep(30)

# 스케줄러 종료
scheduler.shutdown()
print("스케줄러 종료됨")

종료 옵션​

# 현재 실행 중인 작업이 완료될 때까지 대기
scheduler.shutdown(wait=True)

# 즉시 종료 (실행 중인 작업 중단)
scheduler.shutdown(wait=False)

컨텍스트 매니저 사용​

from contextlib import contextmanager
from apscheduler.schedulers.background import BackgroundScheduler

@contextmanager
def scheduler_context():
    """스케줄러 컨텍스트 매니저"""
    scheduler = BackgroundScheduler()
    try:
        scheduler.start()
        yield scheduler
    finally:
        scheduler.shutdown(wait=True)

# 사용 예제
with scheduler_context() as scheduler:
    scheduler.add_job(my_job, 'interval', seconds=5)
    # 작업 수행
    import time
    time.sleep(30)
# 블록을 벗어나면 자동으로 종료됨

우아한 종료 (Graceful Shutdown)​

import signal
import sys
from apscheduler.schedulers.background import BackgroundScheduler
import time

class GracefulScheduler:
    """우아한 종료를 지원하는 스케줄러"""

    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.running = False

    def setup_signal_handlers(self):
        """시그널 핸들러 설정"""
        signal.signal(signal.SIGINT, self.signal_handler)
        signal.signal(signal.SIGTERM, self.signal_handler)

    def signal_handler(self, signum, frame):
        """시그널 수신 시 처리"""
        print(f"\n시그널 {signum} 수신, 종료 중...")
        self.stop()
        sys.exit(0)

    def start(self):
        """스케줄러 시작"""
        self.setup_signal_handlers()
        self.scheduler.start()
        self.running = True
        print("스케줄러 시작됨")

    def stop(self):
        """스케줄러 종료"""
        if self.running:
            print("스케줄러 종료 중...")
            self.scheduler.shutdown(wait=True)
            self.running = False
            print("스케줄러 종료 완료")

    def add_job(self, *args, **kwargs):
        """작업 추가"""
        return self.scheduler.add_job(*args, **kwargs)

    def run_forever(self):
        """무한 실행"""
        try:
            while self.running:
                time.sleep(1)
        except (KeyboardInterrupt, SystemExit):
            self.stop()

# 사용 예제
graceful_scheduler = GracefulScheduler()

def task():
    print("작업 실행")

graceful_scheduler.add_job(task, 'interval', seconds=5)
graceful_scheduler.start()
graceful_scheduler.run_forever()

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.schedulers import SchedulerNotRunningError

scheduler = BackgroundScheduler()

# 스케줄러 상태 확인

# 안전한 작업 추가
def safe_add_job(scheduler, func, *args, **kwargs):
    """스케줄러 상태를 확인하고 작업 추가"""
    if not scheduler.running:
        scheduler.start()

    try:
        return scheduler.add_job(func, *args, **kwargs)
    except Exception as e:
        print(f"작업 추가 실패: {e}")
        return None

# 안전한 종료
def safe_shutdown(scheduler):
    """스케줄러를 안전하게 종료"""
    if scheduler.running:
        try:
            scheduler.shutdown(wait=True)
            print("스케줄러 정상 종료")
        except SchedulerNotRunningError:
            print("스케줄러가 이미 종료됨")
    else:
        print("스케줄러가 실행 중이 아님")

재시작 기능​

from apscheduler.schedulers.background import BackgroundScheduler
import time

class RestartableScheduler:
    """재시작 가능한 스케줄러"""

    def __init__(self, **kwargs):
        self.config = kwargs
        self.scheduler = None
        self.jobs = []

    def start(self):
        """스케줄러 시작"""
        if self.scheduler and self.scheduler.running:
            print("스케줄러가 이미 실행 중입니다")
            return

        self.scheduler = BackgroundScheduler(**self.config)

        # 저장된 작업 재등록
        for job_config in self.jobs:
            self.scheduler.add_job(**job_config)

        self.scheduler.start()
        print("스케줄러 시작됨")

    def stop(self):
        """스케줄러 정지"""
        if self.scheduler and self.scheduler.running:
            self.scheduler.shutdown(wait=True)
            print("스케줄러 정지됨")

    def restart(self):
        """스케줄러 재시작"""
        print("스케줄러 재시작 중...")
        self.stop()
        time.sleep(1)
        self.start()
        print("스케줄러 재시작 완료")

    def add_job(self, func, trigger, **kwargs):
        """작업 추가 및 설정 저장"""
        job_config = {
            'func': func,
            'trigger': trigger,
            **kwargs
        }
        self.jobs.append(job_config)

        if self.scheduler and self.scheduler.running:
            return self.scheduler.add_job(**job_config)

# 사용 예제
restartable = RestartableScheduler()

def my_task():
    print("작업 실행")

restartable.add_job(my_task, 'interval', seconds=5, id='task1')
restartable.start()

# 10초 후 재시작
time.sleep(10)
restartable.restart()
