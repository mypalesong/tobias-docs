3. 트리거

트리거(Trigger)는 작업이 언제 실행될지를 결정하는 APScheduler의 핵심 컴포넌트입니다. 이 문서에서는 다양한 트리거 타입과 활용법을 상세히 다룹니다.

트리거 개요​

APScheduler는 세 가지 기본 트리거를 제공합니다:

  • DateTrigger: 특정 날짜/시간에 한 번 실행
  • IntervalTrigger: 일정한 간격으로 반복 실행
  • CronTrigger: Cron 표현식을 사용한 복잡한 스케줄링

각 트리거는 작업의 실행 시점을 계산하고, 다음 실행 시간을 결정합니다.

DateTrigger​

DateTrigger는 지정된 날짜와 시간에 작업을 한 번만 실행합니다.

기본 사용법​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta
import time

def one_time_job():
    print(f"일회성 작업 실행: {datetime.now()}")

scheduler = BackgroundScheduler()

# 특정 시간에 실행
run_date = datetime(2024, 12, 31, 23, 59, 59)
scheduler.add_job(one_time_job, 'date', run_date=run_date)

# 10초 후 실행
run_date = datetime.now() + timedelta(seconds=10)
scheduler.add_job(one_time_job, 'date', run_date=run_date)

scheduler.start()

try:
    time.sleep(60)
except KeyboardInterrupt:
    scheduler.shutdown()

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.date import DateTrigger

조건부 일회성 작업​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta
import time

def conditional_task():
    """조건을 확인하고 필요 시 재스케줄"""
    print(f"작업 실행 시도: {datetime.now()}")

    # 조건 확인
    if not check_condition():
        print("조건 미충족, 5분 후 재시도")
        # 5분 후 다시 스케줄
        retry_time = datetime.now() + timedelta(minutes=5)
        scheduler.add_job(conditional_task, 'date', run_date=retry_time)
    else:
        print("작업 성공적으로 완료")

def check_condition():
    """조건 확인 함수"""
    # 실제 조건 로직
    return datetime.now().second % 2 == 0

scheduler = BackgroundScheduler()

# 초기 실행
initial_run = datetime.now() + timedelta(seconds=3)
scheduler.add_job(conditional_task, 'date', run_date=initial_run)

scheduler.start()

try:
    time.sleep(30)
except KeyboardInterrupt:
    scheduler.shutdown()

조건부 반복 작업​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

class ConditionalJob:
    """조건에 따라 계속 실행할지 결정하는 작업"""

    def __init__(self, scheduler, max_runs=10):
        self.scheduler = scheduler
        self.run_count = 0
        self.max_runs = max_runs

    def run(self):
        """작업 실행"""
        self.run_count += 1
        print(f"실행 {self.run_count}/{self.max_runs}: {datetime.now()}")

        # 최대 실행 횟수 도달 시 작업 제거
        if self.run_count >= self.max_runs:
            print("최대 실행 횟수 도달, 작업 제거")
            self.scheduler.remove_job('conditional_job')

scheduler = BackgroundScheduler()
conditional_job = ConditionalJob(scheduler, max_runs=5)

scheduler.add_job(
    conditional_job.run,
    'interval',
    seconds=2,
    id='conditional_job'
)

scheduler.start()

try:
    time.sleep(15)
except KeyboardInterrupt:
    scheduler.shutdown()

동적 간격 조정​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time
import random

class AdaptiveIntervalJob:
    """시스템 부하에 따라 간격을 조정하는 작업"""

    def __init__(self, scheduler, job_id):
        self.scheduler = scheduler
        self.job_id = job_id
        self.current_interval = 5

    def run(self):
        """작업 실행 및 간격 조정"""
        print(f"작업 실행 (현재 간격: {self.current_interval}초)")

        # 시스템 부하 시뮬레이션
        system_load = random.uniform(0, 1)

        # 부하가 높으면 간격 증가, 낮으면 감소
        if system_load > 0.7:
            self.current_interval = min(self.current_interval + 2, 30)
            print(f"시스템 부하 높음 ({system_load:.2f}), 간격 증가: {self.current_interval}초")
        elif system_load < 0.3:
            self.current_interval = max(self.current_interval - 1, 3)
            print(f"시스템 부하 낮음 ({system_load:.2f}), 간격 감소: {self.current_interval}초")

        # 작업 재스케줄
        self.scheduler.reschedule_job(
            self.job_id,
            trigger='interval',
            seconds=self.current_interval
        )

scheduler = BackgroundScheduler()
adaptive_job = AdaptiveIntervalJob(scheduler, 'adaptive_job')

scheduler.add_job(
    adaptive_job.run,
    'interval',
    seconds=5,
    id='adaptive_job'
)

scheduler.start()

try:
    time.sleep(60)
except KeyboardInterrupt:
    scheduler.shutdown()

기본 사용법​

Cron 표현식 검증​

from apscheduler.triggers.cron import CronTrigger
from datetime import datetime

def validate_cron_expression(**kwargs):
    """Cron 표현식 유효성 검증 및 다음 실행 시간 예측"""
    try:
        trigger = CronTrigger(**kwargs)

        # 다음 10번의 실행 시간 출력
        now = datetime.now()
        print(f"현재 시간: {now}")
        print(f"Cron 표현식: {kwargs}")
        print("\n다음 실행 시간:")

        for i in range(10):
            next_time = trigger.get_next_fire_time(now, now)
            if next_time:
                print(f"  {i+1}. {next_time}")
                now = next_time
            else:
                print("  더 이상 실행 시간이 없습니다.")
                break

        return True
    except Exception as e:
        print(f"유효하지 않은 Cron 표현식: {e}")
        return False

# 검증 예제
print("=== 검증 1: 평일 오전 9시 ===")
validate_cron_expression(day_of_week='mon-fri', hour=9, minute=0)

print("\n=== 검증 2: 매월 1일과 15일 ===")
validate_cron_expression(day='1,15', hour=12, minute=0)

print("\n=== 검증 3: 매주 수요일 ===")
validate_cron_expression(day_of_week='wed', hour=14, minute=30)

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.combining import OrTrigger
from apscheduler.triggers.cron import CronTrigger
from datetime import datetime
import time

scheduler = BackgroundScheduler()

def notification_task():
    """알림 전송 작업"""
    print(f"알림 전송: {datetime.now()}")

# 여러 시간대에 알림 전송
# - 평일: 오전 9시, 정오, 오후 6시
# - 주말: 정오
weekday_morning = CronTrigger(day_of_week='mon-fri', hour=9, minute=0)
weekday_noon = CronTrigger(day_of_week='mon-fri', hour=12, minute=0)
weekday_evening = CronTrigger(day_of_week='mon-fri', hour=18, minute=0)
weekend_noon = CronTrigger(day_of_week='sat,sun', hour=12, minute=0)

notification_trigger = OrTrigger([
    weekday_morning,
    weekday_noon,
    weekday_evening,
    weekend_noon
])

scheduler.add_job(notification_task, trigger=notification_trigger)

scheduler.start()

# 다음 10번의 실행 시간 출력

커스텀 트리거 조합​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

class BusinessHoursTrigger:
    """업무 시간에만 실행되는 커스텀 트리거"""

    def __init__(self, interval_seconds=60):
        self.interval_seconds = interval_seconds

    def get_next_fire_time(self, previous_fire_time, now):
        """다음 실행 시간 계산"""
        from datetime import timedelta

        if previous_fire_time:
            next_time = previous_fire_time + timedelta(seconds=self.interval_seconds)
        else:
            next_time = now + timedelta(seconds=self.interval_seconds)

        # 업무 시간 확인 (평일 9시~18시)
        while True:
            if next_time.weekday() < 5:  # 월~금
                if 9 <= next_time.hour < 18:
                    return next_time

            # 업무 시간이 아니면 다음 업무일 9시로 조정
            if next_time.weekday() >= 5:  # 주말
                days_to_add = 7 - next_time.weekday()
                next_time = next_time.replace(
                    hour=9, minute=0, second=0, microsecond=0
                ) + timedelta(days=days_to_add)
            elif next_time.hour >= 18:  # 업무 시간 이후
                next_time = (next_time + timedelta(days=1)).replace(
                    hour=9, minute=0, second=0, microsecond=0
                )
            else:  # 업무 시간 이전
                next_time = next_time.replace(
                    hour=9, minute=0, second=0, microsecond=0
                )

scheduler = BackgroundScheduler()

def business_task():
    print(f"업무 시간 작업: {datetime.now()}")

trigger = BusinessHoursTrigger(interval_seconds=300)  # 5분마다
scheduler.add_job(business_task, trigger=trigger)

scheduler.start()

try:
    time.sleep(600)
except KeyboardInterrupt:
    scheduler.shutdown()

타임존 변환 유틸리티​

from datetime import datetime
from pytz import timezone, utc, all_timezones

class TimezoneHelper:
    """타임존 관련 유틸리티"""

    @staticmethod
    def convert_timezone(dt, from_tz, to_tz):
        """타임존 변환"""
        if isinstance(from_tz, str):
            from_tz = timezone(from_tz)
        if isinstance(to_tz, str):
            to_tz = timezone(to_tz)

        if dt.tzinfo is None:
            dt = from_tz.localize(dt)
        else:
            dt = dt.astimezone(from_tz)

        return dt.astimezone(to_tz)

    @staticmethod
    def get_local_time(dt, tz):
        """특정 타임존의 현지 시간 가져오기"""
        if isinstance(tz, str):
            tz = timezone(tz)

        if dt.tzinfo is None:
            dt = utc.localize(dt)

        return dt.astimezone(tz)

    @staticmethod
    def list_timezones(filter_str=None):
        """타임존 목록 조회"""
        if filter_str:
            return [tz for tz in all_timezones if filter_str in tz]
        return all_timezones

    @staticmethod
    def get_timezone_offset(tz, dt=None):
        """타임존 오프셋 계산"""
        if isinstance(tz, str):
            tz = timezone(tz)

        if dt is None:
            dt = datetime.now()

        if dt.tzinfo is None:
            dt = utc.localize(dt)

        local_dt = dt.astimezone(tz)
        offset = local_dt.utcoffset()

        return offset.total_seconds() / 3600

# 사용 예제
helper = TimezoneHelper()

# 현재 시간을 여러 타임존으로 변환
now = datetime.now(utc)
print(f"UTC: {now}")
print(f"뉴욕: {helper.get_local_time(now, 'America/New_York')}")
print(f"런던: {helper.get_local_time(now, 'Europe/London')}")
print(f"서울: {helper.get_local_time(now, 'Asia/Seoul')}")
print(f"도쿄: {helper.get_local_time(now, 'Asia/Tokyo')}")

# 타임존 오프셋
print(f"\n타임존 오프셋 (UTC 기준):")
for tz_name in ['America/New_York', 'Europe/London', 'Asia/Seoul', 'Asia/Tokyo']:
    offset = helper.get_timezone_offset(tz_name, now)
    print(f"  {tz_name}: UTC{offset:+.1f}")

# 아시아 타임존 목록
print(f"\n아시아 타임존:")
asia_timezones = helper.list_timezones('Asia')
for tz in asia_timezones[:10]:
    print(f"  - {tz}")

글로벌 서비스 스케줄링​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
from pytz import timezone
from datetime import datetime
import time

class GlobalScheduler:
    """글로벌 서비스를 위한 스케줄러"""

    def __init__(self):
        self.scheduler = BackgroundScheduler(timezone=timezone('UTC'))
        self.regions = {
            'us_east': timezone('America/New_York'),
            'us_west': timezone('America/Los_Angeles'),
            'europe': timezone('Europe/London'),
            'asia_pacific': timezone('Asia/Tokyo'),
            'korea': timezone('Asia/Seoul')
        }

    def add_regional_job(self, func, region, hour, minute=0, **kwargs):
        """지역별 작업 추가"""
        if region not in self.regions:
            raise ValueError(f"알 수 없는 지역: {region}")

        tz = self.regions[region]
        trigger = CronTrigger(hour=hour, minute=minute, timezone=tz, **kwargs)

        job_id = f"{region}_{func.__name__}"
        return self.scheduler.add_job(
            func,
            trigger=trigger,
            id=job_id,
            name=f"{region} - {func.__name__}"
        )

    def add_global_job(self, func, hour, minute=0, **kwargs):
        """모든 지역에 동일한 시간에 실행되는 작업 추가"""
        for region in self.regions:
            self.add_regional_job(func, region, hour, minute, **kwargs)

    def start(self):
        """스케줄러 시작"""
        self.scheduler.start()

    def shutdown(self):
        """스케줄러 종료"""
        self.scheduler.shutdown()

    def print_schedule(self):
        """스케줄 출력"""
        print("글로벌 작업 스케줄:")
        for job in self.scheduler.get_jobs():
            print(f"  - {job.name}")
            print(f"    다음 실행: {job.next_run_time}")

# 사용 예제
global_scheduler = GlobalScheduler()

def morning_report():
    """아침 리포트 생성"""
    print(f"아침 리포트 생성: {datetime.now()}")

def end_of_day_summary():
    """일일 요약"""
    print(f"일일 요약: {datetime.now()}")

# 각 지역 오전 9시에 아침 리포트
global_scheduler.add_global_job(morning_report, hour=9, minute=0)

# 각 지역 오후 6시에 일일 요약
global_scheduler.add_global_job(end_of_day_summary, hour=18, minute=0)

global_scheduler.start()
global_scheduler.print_schedule()

try:
    time.sleep(300)
except KeyboardInterrupt:
    global_scheduler.shutdown()
