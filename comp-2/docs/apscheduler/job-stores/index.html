6. Job Stores

Job Store는 APScheduler에서 Job의 정보를 저장하고 관리하는 컴포넌트입니다. 이 문서에서는 다양한 Job Store 타입과 활용법을 상세히 다룹니다.

Job Store 개요​

APScheduler는 여러 종류의 Job Store를 제공합니다:

  • MemoryJobStore: 메모리에 Job 저장 (기본값, 프로세스 종료 시 소멸)
  • SQLAlchemyJobStore: 관계형 데이터베이스에 Job 저장
  • MongoDBJobStore: MongoDB에 Job 저장
  • RedisJobStore: Redis에 Job 저장

각 Job Store는 영속성, 확장성, 성능 면에서 다른 특성을 가집니다.

MemoryJobStore​

MemoryJobStore는 Job을 메모리에 저장합니다. 가장 빠르지만 프로세스가 종료되면 모든 Job이 사라집니다.

기본 사용법​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from datetime import datetime
import time

# MemoryJobStore 설정 (기본값이므로 명시하지 않아도 됨)
jobstores = {
    'default': MemoryJobStore()
}

scheduler = BackgroundScheduler(jobstores=jobstores)

def simple_job():
    print(f"메모리 Job 실행: {datetime.now()}")

scheduler.add_job(simple_job, 'interval', seconds=5, id='memory_job')

scheduler.start()

print("MemoryJobStore 사용 중...")
print(f"저장된 Job 수: {len(scheduler.get_jobs())}")

try:
    time.sleep(20)
except KeyboardInterrupt:
    scheduler.shutdown()

MemoryJobStore의 한계​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
import time

def demonstrate_memory_limitation():
    """MemoryJobStore의 한계 시연"""

    # 첫 번째 스케줄러
    scheduler1 = BackgroundScheduler(jobstores={'default': MemoryJobStore()})

    def task():
        print("작업 실행")

    scheduler1.add_job(task, 'interval', seconds=5, id='persistent_job')
    scheduler1.start()

    print("첫 번째 스케줄러 시작")
    print(f"저장된 Job: {len(scheduler1.get_jobs())}개")

    time.sleep(10)

    # 스케줄러 종료
    scheduler1.shutdown()
    print("첫 번째 스케줄러 종료")

    # 두 번째 스케줄러 (새로운 MemoryJobStore)
    scheduler2 = BackgroundScheduler(jobstores={'default': MemoryJobStore()})
    scheduler2.start()

    print("\n두 번째 스케줄러 시작")
    print(f"저장된 Job: {len(scheduler2.get_jobs())}개")
    print("-> MemoryJobStore는 영속성이 없어 Job이 사라졌습니다!")

    scheduler2.shutdown()

SQLAlchemyJobStore​

SQLAlchemyJobStore는 관계형 데이터베이스에 Job을 저장합니다. 영속성이 필요한 프로덕션 환경에 적합합니다.

기본 사용법 (SQLite)​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from datetime import datetime
import time

# SQLite 데이터베이스 사용
jobstores = {
    'default': SQLAlchemyJobStore(url='sqlite:///jobs.sqlite')
}

scheduler = BackgroundScheduler(jobstores=jobstores)

def persistent_job():
    print(f"영속적 Job 실행: {datetime.now()}")

# Job 추가 (replace_existing=True로 중복 방지)
scheduler.add_job(
    persistent_job,
    'interval',
    seconds=10,
    id='persistent_job_1',
    replace_existing=True
)

scheduler.start()

print("SQLAlchemyJobStore 사용 중 (SQLite)...")
print(f"저장된 Job 수: {len(scheduler.get_jobs())}")

try:
    time.sleep(30)
except KeyboardInterrupt:
    scheduler.shutdown()

print("\n스케줄러 재시작 시연...")
time.sleep(2)

# 새 스케줄러 인스턴스
scheduler2 = BackgroundScheduler(jobstores=jobstores)
scheduler2.start()

print(f"저장된 Job 수: {len(scheduler2.get_jobs())}")
print("-> SQLAlchemyJobStore는 영속성이 있어 Job이 유지됩니다!")

트랜잭션 처리​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from contextlib import contextmanager
import time

db_url = 'sqlite:///transactional_jobs.sqlite'
engine = create_engine(db_url)
Session = sessionmaker(bind=engine)

jobstore = SQLAlchemyJobStore(url=db_url)
scheduler = BackgroundScheduler(jobstores={'default': jobstore})

@contextmanager
def db_session():
    """데이터베이스 세션 컨텍스트 매니저"""
    session = Session()
    try:
        yield session
        session.commit()
    except Exception as e:
        session.rollback()
        print(f"트랜잭션 롤백: {e}")
        raise
    finally:
        session.close()

def transactional_job():
    """트랜잭션 처리가 있는 Job"""
    with db_session() as session:
        print("트랜잭션 시작")
        # 데이터베이스 작업
        print("트랜잭션 커밋")

scheduler.add_job(transactional_job, 'interval', seconds=10, id='transactional')

scheduler.start()

try:
    time.sleep(30)
except KeyboardInterrupt:
    scheduler.shutdown()

대용량 Job 관리​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from datetime import datetime, timedelta
import time

db_url = 'sqlite:///large_scale_jobs.sqlite'

jobstores = {
    'default': SQLAlchemyJobStore(url=db_url)
}

job_defaults = {
    'coalesce': True,
    'max_instances': 1,
    'misfire_grace_time': 60
}

scheduler = BackgroundScheduler(
    jobstores=jobstores,
    job_defaults=job_defaults
)

def task(task_id):
    print(f"작업 {task_id} 실행")

# 대량의 Job 추가
print("1000개의 Job 추가 중...")
start_time = time.time()

for i in range(1000):
    run_date = datetime.now() + timedelta(seconds=i*10)
    scheduler.add_job(
        task,
        'date',
        run_date=run_date,
        args=[i],
        id=f'task_{i}',
        replace_existing=True
    )

duration = time.time() - start_time
print(f"완료! (소요시간: {duration:.2f}초)")

scheduler.start()

print(f"총 {len(scheduler.get_jobs())}개 Job 저장됨")

time.sleep(30)

# Job 일괄 제거
print("\nJob 일괄 제거 중...")
start_time = time.time()

scheduler.remove_all_jobs()

duration = time.time() - start_time
print(f"완료! (소요시간: {duration:.2f}초)")

scheduler.shutdown()

MongoDB 쿼리 및 분석​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.mongodb import MongoDBJobStore
from pymongo import MongoClient
from datetime import datetime
import time

client = MongoClient('mongodb://localhost:27017/')
db = client['scheduler_db']
collection = db['jobs']

jobstores = {
    'default': MongoDBJobStore(
        client=client,
        database='scheduler_db',
        collection='jobs'
    )
}

scheduler = BackgroundScheduler(jobstores=jobstores)

def analyzed_job(category):
    print(f"[{category}] Job 실행: {datetime.now()}")

# 다양한 카테고리의 Job 추가
categories = ['data', 'report', 'backup', 'cleanup']

for i, category in enumerate(categories):
    for j in range(5):
        scheduler.add_job(
            analyzed_job,
            'interval',
            seconds=30,
            args=[category],
            id=f'{category}_job_{j}',
            replace_existing=True
        )

scheduler.start()

time.sleep(5)

# MongoDB에서 직접 쿼리
print("\n=== Job 분석 ===")

# 총 Job 수
total_jobs = collection.count_documents({})
print(f"총 Job 수: {total_jobs}")

# 다음 실행 시간이 있는 Job 수
active_jobs = collection.count_documents({'next_run_time': {'$ne': None}})
print(f"활성 Job 수: {active_jobs}")

# Job ID 패턴별 집계
print("\n카테고리별 Job 수:")
pipeline = [
    {
        '$group': {
            '_id': {
                '$arrayElemAt': [
                    {'$split': ['$_id', '_job_']},
                    0
                ]
            },
            'count': {'$sum': 1}
        }
    }
]

for result in collection.aggregate(pipeline):
    print(f"  {result['_id']}: {result['count']}개")

time.sleep(10)

scheduler.shutdown()
client.close()

    )
}

scheduler = BackgroundScheduler(jobstores=jobstores)

def temporary_job():
    print(f"임시 Job 실행: {datetime.now()}")

# Job 추가
run_date = datetime.now() + timedelta(seconds=30)
scheduler.add_job(
    temporary_job,
    'date',
    run_date=run_date,
    id='temp_job_with_ttl'
)

scheduler.start()

# Redis 키에 TTL 설정 (1시간)
redis_client.expire('apscheduler.jobs', 3600)
redis_client.expire('apscheduler.run_times', 3600)

print("Redis TTL이 설정된 JobStore 사용 중...")

ttl = redis_client.ttl('apscheduler.jobs')
print(f"Jobs 키 TTL: {ttl}초")

time.sleep(40)

scheduler.shutdown()

Redis Pub/Sub 통합​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.redis import RedisJobStore
from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR
import redis
import json
from datetime import datetime
import time
import threading

# Redis 클라이언트
redis_client = redis.Redis(host='localhost', port=6379, db=0, decode_responses=True)
pubsub = redis_client.pubsub()

jobstores = {
    'default': RedisJobStore(redis=redis_client)
}

scheduler = BackgroundScheduler(jobstores=jobstores)

def publish_job_event(event_type, job_id, data=None):
    """Job 이벤트를 Redis Pub/Sub으로 발행"""
    message = {
        'event_type': event_type,
        'job_id': job_id,
        'timestamp': datetime.now().isoformat(),
        'data': data
    }
    redis_client.publish('scheduler_events', json.dumps(message))

def job_executed_listener(event):
    """Job 실행 완료 리스너"""
    publish_job_event('JOB_EXECUTED', event.job_id)

def job_error_listener(event):
    """Job 오류 리스너"""
    publish_job_event('JOB_ERROR', event.job_id, {'exception': str(event.exception)})

# 이벤트 리스너 등록
scheduler.add_listener(job_executed_listener, EVENT_JOB_EXECUTED)
scheduler.add_listener(job_error_listener, EVENT_JOB_ERROR)

def monitored_job():
    print(f"모니터링되는 Job 실행: {datetime.now()}")

scheduler.add_job(monitored_job, 'interval', seconds=10, id='monitored')

# Pub/Sub 구독자
def subscriber():
    """이벤트 구독자"""
    pubsub.subscribe('scheduler_events')

    for message in pubsub.listen():
        if message['type'] == 'message':
            event_data = json.loads(message['data'])
            print(f"[이벤트] {event_data['event_type']}: {event_data['job_id']}")

# 구독자 스레드 시작
subscriber_thread = threading.Thread(target=subscriber, daemon=True)
subscriber_thread.start()

scheduler.start()

print("Redis Pub/Sub 통합 JobStore 사용 중...")

try:
    time.sleep(30)
except KeyboardInterrupt:
    pass
finally:
    scheduler.shutdown()
    pubsub.close()

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from apscheduler.jobstores.redis import RedisJobStore
import time

def benchmark_jobstore(jobstore_name, jobstore):
    """Job Store 성능 벤치마크"""
    scheduler = BackgroundScheduler(jobstores={'default': jobstore})

    def dummy_job():
        pass

    # Job 추가 성능 측정
    start = time.time()
    for i in range(1000):
        scheduler.add_job(
            dummy_job,
            'interval',
            seconds=3600,
            id=f'job_{i}',
            replace_existing=True
        )
    add_duration = time.time() - start

    scheduler.start()

    # Job 조회 성능 측정
    start = time.time()
    jobs = scheduler.get_jobs()
    query_duration = time.time() - start

    # Job 제거 성능 측정
    start = time.time()
    scheduler.remove_all_jobs()
    remove_duration = time.time() - start

    scheduler.shutdown()

    return {
        'name': jobstore_name,
        'add': add_duration,
        'query': query_duration,
        'remove': remove_duration
    }

# 벤치마크 실행
print("Job Store 성능 비교 (1000개 Job)")
print("="*60)

results = []

# MemoryJobStore
results.append(benchmark_jobstore('Memory', MemoryJobStore()))

# SQLAlchemyJobStore (SQLite)
results.append(benchmark_jobstore('SQLAlchemy (SQLite)',
                                   SQLAlchemyJobStore(url='sqlite:///benchmark.sqlite')))

# RedisJobStore
results.append(benchmark_jobstore('Redis', RedisJobStore()))

# 결과 출력
for result in results:
    print(f"\n{result['name']}:")
    print(f"  추가: {result['add']:.3f}초")
    print(f"  조회: {result['query']:.3f}초")
    print(f"  제거: {result['remove']:.3f}초")
    print(f"  총합: {sum([result['add'], result['query'], result['remove']]):.3f}초")

사용 사례별 권장사항​

"""
Job Store 선택 가이드

1. MemoryJobStore
   - 사용 사례: 개발/테스트, 일시적 작업, 단일 인스턴스
   - 장점: 가장 빠름, 설정 불필요
   - 단점: 영속성 없음, 분산 환경 미지원

2. SQLAlchemyJobStore
   - 사용 사례: 프로덕션, 영속성 필요, 트랜잭션 중요
   - 장점: 영속성, 트랜잭션, 복잡한 쿼리 가능
   - 단점: 상대적으로 느림, 데이터베이스 관리 필요

3. MongoDBJobStore
   - 사용 사례: 대용량 Job, 유연한 스키마, 분산 환경
   - 장점: 확장성, 유연성, 복제 세트 지원
   - 단점: MongoDB 인프라 필요, 트랜잭션 제한적

4. RedisJobStore
   - 사용 사례: 고성능 필요, 분산 환경, 실시간 처리
   - 장점: 매우 빠름, 분산 환경 우수, Pub/Sub 통합
   - 단점: 메모리 기반 (비용), 영속성 설정 필요
"""

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.jobstores.memory import MemoryJobStore
from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
from apscheduler.jobstores.redis import RedisJobStore

class JobStoreSelector:
    """사용 사례에 맞는 Job Store 선택"""

    @staticmethod
    def get_development_config():
        """개발 환경 설정"""
        return {
            'jobstores': {
                'default': MemoryJobStore()
            }
        }

    @staticmethod
    def get_production_config():
        """프로덕션 환경 설정 (단일 인스턴스)"""
        return {
            'jobstores': {
                'default': SQLAlchemyJobStore(
                    url='postgresql://user:pass@localhost/scheduler_db'
                )
            }
        }

    @staticmethod
    def get_distributed_config():
        """분산 환경 설정"""
        return {
            'jobstores': {
                'default': RedisJobStore(
                    host='redis-cluster',
                    port=6379
                )
            }
        }

    @staticmethod
    def get_hybrid_config():
        """하이브리드 설정 (다중 Job Store)"""
        return {
            'jobstores': {
                'memory': MemoryJobStore(),  # 임시 작업
                'persistent': SQLAlchemyJobStore(  # 영속적 작업
                    url='postgresql://user:pass@localhost/scheduler_db'
                ),
                'cache': RedisJobStore(  # 고성능 작업
                    host='localhost',
                    port=6379
                )
            }
        }

# 사용 예제
selector = JobStoreSelector()

# 개발 환경
dev_config = selector.get_development_config()
dev_scheduler = BackgroundScheduler(**dev_config)

# 프로덕션 환경
prod_config = selector.get_production_config()
prod_scheduler = BackgroundScheduler(**prod_config)

# 분산 환경
distributed_config = selector.get_distributed_config()
distributed_scheduler = BackgroundScheduler(**distributed_config)

# 하이브리드 환경
hybrid_config = selector.get_hybrid_config()
hybrid_scheduler = BackgroundScheduler(**hybrid_config)

def task():
    print("작업 실행")

# 하이브리드 환경 예제: Job Store별로 다른 작업 할당
hybrid_scheduler.add_job(task, 'interval', seconds=5,
                         jobstore='memory', id='temp_task')
hybrid_scheduler.add_job(task, 'interval', minutes=30,
                         jobstore='persistent', id='important_task')
hybrid_scheduler.add_job(task, 'interval', seconds=1,
                         jobstore='cache', id='realtime_task')

print("하이브리드 Job Store 구성:")
for jobstore_name in ['memory', 'persistent', 'cache']:
    jobs = hybrid_scheduler.get_jobs(jobstore=jobstore_name)
    print(f"  {jobstore_name}: {len(jobs)}개 Job")
