8. 실전 예제

이 문서에서는 APScheduler를 실제 프로젝트에 적용하는 다양한 실전 예제를 다룹니다. Flask, Django, FastAPI 통합부터 데이터 백업, API 폴링, 리포트 생성까지 포괄적으로 살펴봅니다.

Flask 통합​

Flask 애플리케이션에서 APScheduler를 통합하는 방법을 알아봅니다.

기본 Flask 통합​

def init_scheduler():
    """스케줄러 초기화"""
    # 매일 새벽 2시 데이터 정리
    scheduler.add_job(

        trigger=CronTrigger(hour=2, minute=0),
        id='cleanup_job',
        name='데이터 정리',
        replace_existing=True

        trigger=CronTrigger(hour=3, minute=0),
        id='backup_job',
        name='데이터베이스 백업',

@app.route('/jobs/<job_id>/resume', methods=['POST'])
def resume_job(job_id):
    """Job 재개"""
    try:

    except Exception as e:
        return jsonify({'error': str(e)}), 400

@app.route('/jobs', methods=['POST'])
def add_job():
    """새 Job 추가"""
    data = request.json

    try:
        scheduler.add_job(
            cleanup_old_data,  # 실제로는 함수를 동적으로 선택해야 함
            trigger=CronTrigger(**data.get('trigger', {})),

            name=data.get('name', data['id']),
            replace_existing=True

        return jsonify({'message': f'Job {data["id"]} added'})

        return jsonify({'error': str(e)}), 400

Flask 블루프린트와 통합​

from flask import Flask, Blueprint, jsonify
from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime

# 스케줄러 블루프린트
scheduler_bp = Blueprint('scheduler', __name__, url_prefix='/scheduler')

# 전역 스케줄러
scheduler = BackgroundScheduler()

def scheduled_task():
    """스케줄된 작업"""
    print(f'스케줄 작업 실행: {datetime.now()}')

@scheduler_bp.route('/status')
def get_status():
    """스케줄러 상태"""
    return jsonify({
        'running': scheduler.running,
        'jobs': len(scheduler.get_jobs())
    })

@scheduler_bp.route('/jobs')
def list_jobs():
    """Job 목록"""
    jobs = [{
        'id': job.id,
        'name': job.name,
        'next_run': str(job.next_run_time)
    } for job in scheduler.get_jobs()]

    return jsonify({'jobs': jobs})

def create_app():
    """Flask 앱 팩토리"""
    app = Flask(__name__)

    # 블루프린트 등록
    app.register_blueprint(scheduler_bp)

    # 스케줄러 시작
    scheduler.add_job(
        scheduled_task,
        'interval',
        seconds=30,
        id='main_task'
    )
    scheduler.start()

    @app.route('/')
    def index():
        return 'Flask with Blueprints'

    return app

if __name__ == '__main__':
    app = create_app()
    app.run(debug=False)

# myapp/scheduler.py

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
from django_apscheduler.jobstores import DjangoJobStore
from django_apscheduler.models import DjangoJobExecution
from django.conf import settings
import logging

logger = logging.getLogger(__name__)

def start():
    """스케줄러 시작"""
    scheduler = BackgroundScheduler(timezone=settings.TIME_ZONE)
    scheduler.add_jobstore(DjangoJobStore(), "default")

    # 작업 등록
    scheduler.add_job(
        daily_cleanup,
        trigger=CronTrigger(hour=2, minute=0),
        id="daily_cleanup",
        max_instances=1,
        replace_existing=True,
    )

    scheduler.add_job(
        send_notifications,
        trigger=CronTrigger(hour=9, minute=0),
        id="send_notifications",
        max_instances=1,
        replace_existing=True,
    )

    # 오래된 Job 실행 기록 삭제
    scheduler.add_job(
        delete_old_job_executions,
        trigger=CronTrigger(day_of_week="mon", hour=0, minute=0),
        id="delete_old_job_executions",
        max_instances=1,
        replace_existing=True,
    )

    try:
        logger.info("스케줄러 시작...")
        scheduler.start()
    except KeyboardInterrupt:
        logger.info("스케줄러 종료...")
        scheduler.shutdown()

def daily_cleanup():
    """일일 데이터 정리"""
    logger.info("일일 정리 작업 시작")
    # Django ORM 사용
    from myapp.models import OldData
    from django.utils import timezone
    from datetime import timedelta

    # 30일 이상 된 데이터 삭제
    cutoff_date = timezone.now() - timedelta(days=30)
    deleted_count = OldData.objects.filter(created_at__lt=cutoff_date).delete()[0]

    logger.info(f"정리 완료: {deleted_count}개 레코드 삭제")

def send_notifications():
    """알림 전송"""
    logger.info("알림 전송 시작")
    from myapp.models import User
    from django.core.mail import send_mail

    users = User.objects.filter(notification_enabled=True)

    for user in users:
        try:
            send_mail(
                '일일 요약',
                '오늘의 요약 내용입니다.',
                'noreply@example.com',
                [user.email],
                fail_silently=False,
            )
        except Exception as e:
            logger.error(f"이메일 전송 실패 ({user.email}): {e}")

    logger.info(f"알림 전송 완료: {users.count()}명")

def delete_old_job_executions(max_age=604_800):
    """오래된 Job 실행 기록 삭제 (기본: 7일)"""
    DjangoJobExecution.objects.delete_old_job_executions(max_age)

# myapp/scheduler.py (APScheduler)

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
from .tasks import process_data, generate_report
import logging

logger = logging.getLogger(__name__)

def start():
    """APScheduler로 Celery 작업 스케줄링"""
    scheduler = BackgroundScheduler()

    # 매시간 데이터 처리 작업을 Celery로 전달
    scheduler.add_job(
        lambda: process_data.delay('hourly_data'),
        trigger=CronTrigger(minute=0),
        id="hourly_processing",
        replace_existing=True,
    )

    # 매일 오전 9시 리포트 생성
    scheduler.add_job(
        lambda: generate_report.delay('daily'),
        trigger=CronTrigger(hour=9, minute=0),
        id="daily_report",
        replace_existing=True,
    )

    scheduler.start()
    logger.info("APScheduler 시작 (Celery 연동)")

# Lifespan 컨텍스트 매니저
@asynccontextmanager
async def lifespan(app: FastAPI):
    """애플리케이션 생명주기 관리"""
    # 시작
    scheduler.add_job(
        async_cleanup,
        trigger=CronTrigger(hour=2, minute=0),
        id='cleanup_job',
        name='데이터 정리',
        replace_existing=True
    )

    scheduler.add_job(
        fetch_external_data,
        trigger=CronTrigger(minute='*/10'),
        id='fetch_job',
        name='데이터 수집',
        replace_existing=True
    )

    scheduler.add_job(
        send_scheduled_emails,
        trigger=CronTrigger(hour=9, minute=0),
        id='email_job',
        name='이메일 전송',
        replace_existing=True
    )

    scheduler.start()
    logger.info("스케줄러 시작됨")

    yield

    # 종료
    scheduler.shutdown()
    logger.info("스케줄러 종료됨")

    else:

PostgreSQL 백업​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
from datetime import datetime
import subprocess
import os
import logging
import shutil

logger = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO)

class DatabaseBackup:
    """데이터베이스 백업 관리자"""

    def __init__(self, db_config, backup_dir='/backups'):
        self.db_config = db_config
        self.backup_dir = backup_dir
        self.max_backups = 7  # 최대 7개의 백업 유지

        # 백업 디렉토리 생성
        os.makedirs(backup_dir, exist_ok=True)

    def backup_postgres(self):
        """PostgreSQL 백업"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_file = f"{self.backup_dir}/backup_{timestamp}.sql"

        logger.info(f"백업 시작: {backup_file}")

        try:
            # pg_dump 실행
            cmd = [
                'pg_dump',
                '-h', self.db_config['host'],
                '-p', str(self.db_config['port']),
                '-U', self.db_config['user'],
                '-d', self.db_config['database'],
                '-F', 'c',  # 커스텀 포맷
                '-f', backup_file
            ]

            # 환경 변수로 비밀번호 전달
            env = os.environ.copy()
            env['PGPASSWORD'] = self.db_config['password']

            result = subprocess.run(
                cmd,
                env=env,
                capture_output=True,
                text=True,
                check=True
            )

            logger.info(f"백업 완료: {backup_file}")

            # 오래된 백업 정리
            self.cleanup_old_backups()

            # 백업 검증
            self.verify_backup(backup_file)

            return backup_file

        except subprocess.CalledProcessError as e:
            logger.error(f"백업 실패: {e.stderr}")
            raise

    def cleanup_old_backups(self):
        """오래된 백업 제거"""
        backups = sorted([
            os.path.join(self.backup_dir, f)
            for f in os.listdir(self.backup_dir)
            if f.startswith('backup_') and f.endswith('.sql')
        ])

        # 오래된 백업 삭제
        while len(backups) > self.max_backups:
            old_backup = backups.pop(0)
            os.remove(old_backup)
            logger.info(f"오래된 백업 제거: {old_backup}")

    def verify_backup(self, backup_file):
        """백업 파일 검증"""
        file_size = os.path.getsize(backup_file)
        logger.info(f"백업 파일 크기: {file_size / 1024 / 1024:.2f} MB")

        if file_size < 1000:  # 1KB 미만이면 의심
            logger.warning("백업 파일이 너무 작습니다!")

    def compress_backup(self, backup_file):
        """백업 파일 압축"""
        import gzip

        compressed_file = f"{backup_file}.gz"

        with open(backup_file, 'rb') as f_in:
            with gzip.open(compressed_file, 'wb') as f_out:
                shutil.copyfileobj(f_in, f_out)

        # 원본 파일 제거
        os.remove(backup_file)

        logger.info(f"백업 압축 완료: {compressed_file}")
        return compressed_file

# 설정
db_config = {
    'host': 'localhost',
    'port': 5432,
    'user': 'postgres',
    'password': 'password',
    'database': 'mydb'
}

backup_manager = DatabaseBackup(db_config, backup_dir='/var/backups/postgres')

# 스케줄러 설정
scheduler = BackgroundScheduler()

# 매일 새벽 2시 백업
scheduler.add_job(
    backup_manager.backup_postgres,
    trigger=CronTrigger(hour=2, minute=0),
    id='daily_backup',
    name='일일 데이터베이스 백업'
)

# 매주 일요일 새벽 3시 전체 백업 + 압축
def weekly_full_backup():
    backup_file = backup_manager.backup_postgres()
    backup_manager.compress_backup(backup_file)

scheduler.add_job(
    weekly_full_backup,
    trigger=CronTrigger(day_of_week='sun', hour=3, minute=0),
    id='weekly_backup',
    name='주간 전체 백업'
)

scheduler.start()

logger.info("백업 스케줄러 시작")

# 메인 루프
try:
    import time
    while True:
        time.sleep(1)
except (KeyboardInterrupt, SystemExit):
    scheduler.shutdown()

from apscheduler.schedulers.background import BackgroundScheduler

from datetime import datetime

import logging

logging.basicConfig(level=logging.INFO)

class MongoDBBackup:
    """MongoDB 백업 관리자"""

    def __init__(self, mongo_config, backup_dir='/backups/mongodb'):
        self.mongo_config = mongo_config
        self.backup_dir = backup_dir
        os.makedirs(backup_dir, exist_ok=True)

    def backup_mongodb(self):
        """MongoDB 백업"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_path = f"{self.backup_dir}/backup_{timestamp}"

        logger.info(f"MongoDB 백업 시작: {backup_path}")

        try:
            cmd = [
                'mongodump',
                '--host', self.mongo_config['host'],
                '--port', str(self.mongo_config['port']),
                '--db', self.mongo_config['database'],
                '--out', backup_path
            ]

            # 인증이 필요한 경우
            if 'username' in self.mongo_config:
                cmd.extend([
                    '--username', self.mongo_config['username'],
                    '--password', self.mongo_config['password'],
                    '--authenticationDatabase', self.mongo_config.get('auth_db', 'admin')
                ])

            result = subprocess.run(cmd, capture_output=True, text=True, check=True)

            logger.info(f"MongoDB 백업 완료: {backup_path}")
            return backup_path

        except subprocess.CalledProcessError as e:
            logger.error(f"백업 실패: {e.stderr}")
            raise

REST API 폴링​

from apscheduler.schedulers.asyncio import AsyncIOScheduler
import aiohttp
import asyncio
from datetime import datetime
import logging

logger = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO)

class APIPoller:
    """API 폴링 관리자"""

    def __init__(self):
        self.session = None
        self.data_cache = {}

    async def init_session(self):
        """HTTP 세션 초기화"""
        if self.session is None:
            self.session = aiohttp.ClientSession()

    async def close_session(self):
        """HTTP 세션 종료"""
        if self.session:
            await self.session.close()

    async def poll_weather_api(self):
        """날씨 API 폴링"""
        await self.init_session()

        url = "https://api.openweathermap.org/data/2.5/weather"
        params = {
            'q': 'Seoul',
            'appid': 'YOUR_API_KEY',
            'units': 'metric'
        }

        try:
            async with self.session.get(url, params=params) as response:
                if response.status == 200:
                    data = await response.json()
                    self.data_cache['weather'] = {
                        'temperature': data['main']['temp'],
                        'description': data['weather'][0]['description'],
                        'timestamp': datetime.now()
                    }
                    logger.info(f"날씨 업데이트: {data['main']['temp']}°C")
                else:
                    logger.error(f"API 오류: {response.status}")
        except Exception as e:
            logger.error(f"폴링 실패: {e}")

    async def poll_stock_prices(self):
        """주식 가격 폴링"""
        await self.init_session()

        symbols = ['AAPL', 'GOOGL', 'MSFT']

        for symbol in symbols:
            url = f"https://api.example.com/stocks/{symbol}"

            try:
                async with self.session.get(url) as response:
                    if response.status == 200:
                        data = await response.json()
                        logger.info(f"{symbol}: ${data['price']}")
                    else:
                        logger.error(f"{symbol} API 오류: {response.status}")
            except Exception as e:
                logger.error(f"{symbol} 폴링 실패: {e}")

            await asyncio.sleep(1)  # Rate limiting

    async def poll_with_retry(self, url, max_retries=3):
        """재시도 로직이 있는 폴링"""
        await self.init_session()

        for attempt in range(max_retries):
            try:
                async with self.session.get(url, timeout=10) as response:
                    if response.status == 200:
                        return await response.json()
                    else:
                        logger.warning(f"시도 {attempt+1} 실패: {response.status}")
            except asyncio.TimeoutError:
                logger.warning(f"시도 {attempt+1} 타임아웃")
            except Exception as e:
                logger.error(f"시도 {attempt+1} 오류: {e}")

            if attempt < max_retries - 1:
                await asyncio.sleep(2 ** attempt)  # Exponential backoff

        return None

async def main():
    """메인 함수"""
    poller = APIPoller()
    scheduler = AsyncIOScheduler()

    # 10분마다 날씨 폴링
    scheduler.add_job(
        poller.poll_weather_api,
        'interval',
        minutes=10,
        id='weather_poll'
    )

    # 5분마다 주식 가격 폴링
    scheduler.add_job(
        poller.poll_stock_prices,
        'interval',
        minutes=5,
        id='stock_poll'
    )

    scheduler.start()

    logger.info("API 폴링 시작")

    try:
        await asyncio.Event().wait()
    except (KeyboardInterrupt, SystemExit):
        scheduler.shutdown()
        await poller.close_session()

if __name__ == '__main__':
    asyncio.run(main())

class ReportGenerator:
    """리포트 생성기"""

    def __init__(self, output_dir='/reports'):
        self.output_dir = output_dir
        import os
        os.makedirs(output_dir, exist_ok=True)

    def generate_daily_report(self):
        """일일 리포트 생성"""
        logger.info("일일 리포트 생성 시작")

        # 데이터 수집
        data = self._collect_daily_data()

        # 리포트 생성
        report_path = self._create_pdf_report(data, 'daily')

        # 이메일 전송
        self._send_report_email(report_path, '일일 리포트')

        logger.info(f"일일 리포트 완료: {report_path}")

    def generate_weekly_report(self):
        """주간 리포트 생성"""
        logger.info("주간 리포트 생성 시작")

        # 데이터 수집 (지난 7일)
        data = self._collect_weekly_data()

        # 리포트 생성
        report_path = self._create_pdf_report(data, 'weekly')

        # 이메일 전송
        self._send_report_email(report_path, '주간 리포트')

        logger.info(f"주간 리포트 완료: {report_path}")

    def generate_monthly_report(self):
        """월간 리포트 생성"""
        logger.info("월간 리포트 생성 시작")

        # 데이터 수집
        data = self._collect_monthly_data()

        # 리포트 생성
        report_path = self._create_pdf_report(data, 'monthly')

        # 이메일 전송
        self._send_report_email(report_path, '월간 리포트')

        logger.info(f"월간 리포트 완료: {report_path}")

    def _collect_daily_data(self):
        """일일 데이터 수집"""
        # 데이터베이스에서 데이터 조회
        return {
            'date': datetime.now().date(),
            'total_users': 1250,
            'new_users': 45,
            'revenue': 12500.50,
            'orders': 89
        }

    def _collect_weekly_data(self):
        """주간 데이터 수집"""
        return {
            'week_start': (datetime.now() - timedelta(days=7)).date(),
            'week_end': datetime.now().date(),
            'total_users': 8750,
            'new_users': 315,
            'revenue': 87503.50,
            'orders': 623
        }

    def _collect_monthly_data(self):
        """월간 데이터 수집"""
        return {
            'month': datetime.now().strftime('%Y-%m'),
            'total_users': 35000,
            'new_users': 1260,
            'revenue': 350000.00,
            'orders': 2490
        }

    def _create_pdf_report(self, data, report_type):
        """PDF 리포트 생성"""
        from reportlab.lib.pagesizes import letter, A4
        from reportlab.pdfgen import canvas
        from reportlab.lib.units import inch

        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        filename = f"{self.output_dir}/{report_type}_report_{timestamp}.pdf"

        # PDF 생성
        c = canvas.Canvas(filename, pagesize=A4)
        width, height = A4

        # 제목
        c.setFont("Helvetica-Bold", 24)
        c.drawString(1*inch, height - 1*inch, f"{report_type.title()} Report")

        # 날짜
        c.setFont("Helvetica", 12)
        c.drawString(1*inch, height - 1.5*inch, f"Generated: {datetime.now()}")

        # 데이터
        c.setFont("Helvetica", 14)
        y_position = height - 2.5*inch

        for key, value in data.items():
            text = f"{key.replace('_', ' ').title()}: {value}"
            c.drawString(1*inch, y_position, text)
            y_position -= 0.3*inch

        c.save()

        logger.info(f"PDF 생성됨: {filename}")
        return filename

    def _send_report_email(self, report_path, subject):
        """리포트 이메일 전송"""
        import smtplib
        from email.mime.multipart import MIMEMultipart
        from email.mime.base import MIMEBase
        from email.mime.text import MIMEText
        from email import encoders

        # 이메일 설정
        sender_email = "reports@example.com"
        receiver_emails = ["manager@example.com", "team@example.com"]
        smtp_server = "smtp.example.com"
        smtp_port = 587
        smtp_user = "reports@example.com"
        smtp_password = "password"

        # 이메일 작성
        msg = MIMEMultipart()
        msg['From'] = sender_email
        msg['To'] = ', '.join(receiver_emails)
        msg['Subject'] = f"{subject} - {datetime.now().strftime('%Y-%m-%d')}"

        body = f"""
        안녕하세요,

        {subject}를 첨부합니다.

        감사합니다.
        """

        msg.attach(MIMEText(body, 'plain'))

        # 파일 첨부
        with open(report_path, 'rb') as f:
            part = MIMEBase('application', 'octet-stream')
            part.set_payload(f.read())

        encoders.encode_base64(part)
        part.add_header(
            'Content-Disposition',
            f'attachment; filename={os.path.basename(report_path)}'
        )
        msg.attach(part)

        # 이메일 전송
        try:
            with smtplib.SMTP(smtp_server, smtp_port) as server:
                server.starttls()
                server.login(smtp_user, smtp_password)
                server.send_message(msg)

            logger.info(f"리포트 이메일 전송 완료: {receiver_emails}")
        except Exception as e:
            logger.error(f"이메일 전송 실패: {e}")
