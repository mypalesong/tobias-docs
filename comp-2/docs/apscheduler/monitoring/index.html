7. 모니터링

APScheduler의 안정적인 운영을 위해서는 적절한 모니터링이 필수입니다. 이 문서에서는 이벤트 리스너, 로깅, 에러 처리, 메트릭 수집 방법을 상세히 다룹니다.

이벤트 리스너​

APScheduler는 다양한 이벤트를 발생시키며, 이를 통해 스케줄러의 동작을 모니터링할 수 있습니다.

기본 이벤트 리스너​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import (
    EVENT_JOB_EXECUTED,

from datetime import datetime
import time

scheduler = BackgroundScheduler()

def job_executed(event):
    """Job 실행 완료 이벤트"""

    print(f"  예정 시간: {event.scheduled_run_time}")
    print(f"  반환값: {event.retval}")

    print(f"  예외: {event.exception}")
    print(f"  트레이스백: {event.traceback}")

    """Job 놓침 이벤트"""
    print(f"[{datetime.now():%H:%M:%S}] ⚠ Job {event.job_id} 놓침")
    print(f"  예정 시간: {event.scheduled_run_time}")

def job_submitted(event):
    """Job 제출 이벤트"""
    print(f"[{datetime.now():%H:%M:%S}] → Job {event.job_id} 제출됨")
    print(f"  예정 시간: {event.scheduled_run_times}")

def job_added(event):
    """Job 추가 이벤트"""
    print(f"[{datetime.now():%H:%M:%S}] + Job {event.job_id} 추가됨")

def job_removed(event):
    """Job 제거 이벤트"""
    print(f"[{datetime.now():%H:%M:%S}] - Job {event.job_id} 제거됨")

def job_max_instances(event):

    print(f"[{datetime.now():%H:%M:%S}] ⚠ Job {event.job_id} max_instances 도달")

통합 이벤트 모니터​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import (
    EVENT_ALL,
    EVENT_SCHEDULER_STARTED,
    EVENT_SCHEDULER_SHUTDOWN,
    EVENT_EXECUTOR_ADDED,
    EVENT_EXECUTOR_REMOVED,
    EVENT_JOBSTORE_ADDED,
    EVENT_JOBSTORE_REMOVED
)
from datetime import datetime
from collections import defaultdict
import time

class EventMonitor:
    """통합 이벤트 모니터"""

    def __init__(self):
        self.event_counts = defaultdict(int)
        self.job_history = []
        self.error_history = []

    def handle_event(self, event):
        """모든 이벤트 처리"""
        event_type = type(event).__name__
        self.event_counts[event_type] += 1

        timestamp = datetime.now()

        # 이벤트별 처리
        if hasattr(event, 'job_id'):
            self.job_history.append({
                'timestamp': timestamp,
                'type': event_type,
                'job_id': event.job_id
            })

            if hasattr(event, 'exception'):
                self.error_history.append({
                    'timestamp': timestamp,
                    'job_id': event.job_id,
                    'exception': str(event.exception)
                })

        # 실시간 로깅
        self._log_event(timestamp, event_type, event)

    def _log_event(self, timestamp, event_type, event):
        """이벤트 로깅"""
        log_msg = f"[{timestamp:%H:%M:%S}] {event_type}"

        if hasattr(event, 'job_id'):
            log_msg += f" (Job: {event.job_id})"

        if hasattr(event, 'exception'):
            log_msg += f" - 오류: {event.exception}"

        print(log_msg)

    def get_statistics(self):
        """통계 조회"""
        return {
            'event_counts': dict(self.event_counts),
            'total_jobs_executed': len([h for h in self.job_history
                                        if 'Executed' in h['type']]),
            'total_errors': len(self.error_history),
            'recent_errors': self.error_history[-5:] if self.error_history else []
        }

    def print_statistics(self):
        """통계 출력"""
        stats = self.get_statistics()

        print("\n" + "="*60)
        print("이벤트 모니터 통계")
        print("="*60)

        print("\n이벤트 발생 횟수:")
        for event_type, count in sorted(stats['event_counts'].items()):
            print(f"  {event_type}: {count}회")

        print(f"\n총 실행된 Job: {stats['total_jobs_executed']}개")
        print(f"총 오류: {stats['total_errors']}개")

        if stats['recent_errors']:
            print("\n최근 오류:")
            for error in stats['recent_errors']:
                print(f"  [{error['timestamp']:%H:%M:%S}] "
                      f"{error['job_id']}: {error['exception']}")

        print("="*60)

# 사용 예제
scheduler = BackgroundScheduler()
monitor = EventMonitor()

# 모든 이벤트 리스닝
scheduler.add_listener(monitor.handle_event, EVENT_ALL)

def task(n):
    print(f"  작업 {n} 실행")
    if n % 3 == 0:
        raise Exception(f"작업 {n} 오류")

for i in range(5):
    scheduler.add_job(task, 'interval', seconds=2, args=[i], id=f'task_{i}')

scheduler.start()

try:
    time.sleep(20)
except KeyboardInterrupt:
    pass
finally:
    scheduler.shutdown()
    monitor.print_statistics()

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR
from datetime import datetime
import time

class FilteredEventListener:
    """조건부 이벤트 리스너"""

    def __init__(self, job_id_pattern=None, min_duration=None):
        self.job_id_pattern = job_id_pattern
        self.min_duration = min_duration

    def should_process(self, event):
        """이벤트 처리 여부 결정"""
        # Job ID 패턴 확인
        if self.job_id_pattern and self.job_id_pattern not in event.job_id:
            return False

        # 실행 시간 확인 (JobExecutionEvent만 해당)
        if self.min_duration and hasattr(event, 'scheduled_run_time'):
            # 실제로는 실행 시간을 측정해야 하지만, 여기서는 간단히 구현
            pass

        return True

    def handle_success(self, event):
        """성공 이벤트 처리"""
        if not self.should_process(event):
            return

        print(f"[{datetime.now():%H:%M:%S}] "
              f"필터링된 성공 이벤트: {event.job_id}")

    def handle_error(self, event):
        """오류 이벤트 처리"""
        if not self.should_process(event):
            return

        print(f"[{datetime.now():%H:%M:%S}] "
              f"필터링된 오류 이벤트: {event.job_id} - {event.exception}")

scheduler = BackgroundScheduler()

# critical Job만 모니터링
critical_listener = FilteredEventListener(job_id_pattern='critical')
scheduler.add_listener(critical_listener.handle_success, EVENT_JOB_EXECUTED)
scheduler.add_listener(critical_listener.handle_error, EVENT_JOB_ERROR)

def task(name):
    print(f"  {name} 실행")
    if 'critical' in name and 'error' in name:
        raise Exception("Critical 오류")

scheduler.add_job(task, 'interval', seconds=5, args=['normal_task'], id='normal_1')
scheduler.add_job(task, 'interval', seconds=5, args=['critical_task'], id='critical_1')
scheduler.add_job(task, 'interval', seconds=10, args=['critical_error'], id='critical_error_1')

scheduler.start()

    time.sleep(30)
except KeyboardInterrupt:
    scheduler.shutdown()

로깅​

로깅은 스케줄러의 동작을 추적하고 문제를 진단하는 데 필수적입니다.

구조화된 로깅​

import logging
import json
from datetime import datetime
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR
import time

class StructuredLogger:
    """구조화된 로그 생성기"""

    def __init__(self, logger_name='scheduler'):
        self.logger = logging.getLogger(logger_name)
        self._setup_logger()

    def _setup_logger(self):
        """로거 설정"""
        self.logger.setLevel(logging.INFO)

        # JSON 포맷 핸들러
        handler = logging.FileHandler('scheduler_structured.log')
        handler.setFormatter(self.JSONFormatter())
        self.logger.addHandler(handler)

        # 콘솔 핸들러
        console = logging.StreamHandler()
        console.setFormatter(logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s'
        ))
        self.logger.addHandler(console)

    class JSONFormatter(logging.Formatter):
        """JSON 로그 포맷터"""

        def format(self, record):
            log_data = {
                'timestamp': datetime.utcnow().isoformat(),
                'level': record.levelname,
                'logger': record.name,
                'message': record.getMessage(),
                'module': record.module,
                'function': record.funcName,
                'line': record.lineno
            }

            # 추가 필드
            if hasattr(record, 'job_id'):
                log_data['job_id'] = record.job_id

            if hasattr(record, 'duration'):
                log_data['duration'] = record.duration

            if record.exc_info:
                log_data['exception'] = self.formatException(record.exc_info)

            return json.dumps(log_data, ensure_ascii=False)

    def log_job_start(self, job_id):
        """Job 시작 로그"""
        self.logger.info(f"Job 시작", extra={'job_id': job_id})

    def log_job_end(self, job_id, duration):
        """Job 종료 로그"""
        self.logger.info(
            f"Job 완료",
            extra={'job_id': job_id, 'duration': duration}
        )

    def log_job_error(self, job_id, error):
        """Job 오류 로그"""
        self.logger.error(
            f"Job 오류: {error}",
            extra={'job_id': job_id},
            exc_info=True
        )

# 사용 예제
structured_logger = StructuredLogger()

scheduler = BackgroundScheduler()

def monitored_job(n):
    """모니터링되는 Job"""
    job_id = f'job_{n}'
    start_time = time.time()

    try:
        structured_logger.log_job_start(job_id)
        print(f"  작업 {n} 실행")
        time.sleep(1)

        if n % 3 == 0:
            raise Exception(f"작업 {n} 오류")

        duration = time.time() - start_time
        structured_logger.log_job_end(job_id, duration)

    except Exception as e:
        structured_logger.log_job_error(job_id, e)
        raise

for i in range(3):
    scheduler.add_job(monitored_job, 'interval', seconds=5, args=[i])

scheduler.start()

try:
    time.sleep(20)
except KeyboardInterrupt:
    scheduler.shutdown()

로그 집계 및 분석​

import logging
import json
from datetime import datetime, timedelta
from collections import defaultdict
import time

class LogAnalyzer:
    """로그 분석기"""

    def __init__(self, log_file='scheduler_structured.log'):
        self.log_file = log_file

    def parse_log_file(self):
        """로그 파일 파싱"""
        logs = []

        try:
            with open(self.log_file, 'r', encoding='utf-8') as f:
                for line in f:
                    try:
                        log_entry = json.loads(line.strip())
                        logs.append(log_entry)
                    except json.JSONDecodeError:
                        continue
        except FileNotFoundError:
            print(f"로그 파일을 찾을 수 없습니다: {self.log_file}")

        return logs

    def analyze_performance(self):
        """성능 분석"""
        logs = self.parse_log_file()

        job_durations = defaultdict(list)
        job_errors = defaultdict(int)

        for log in logs:
            if 'job_id' in log:
                job_id = log['job_id']

                if 'duration' in log:
                    job_durations[job_id].append(log['duration'])

                if log['level'] == 'ERROR':
                    job_errors[job_id] += 1

        # 통계 계산
        statistics = {}

        for job_id, durations in job_durations.items():
            statistics[job_id] = {
                'count': len(durations),
                'avg_duration': sum(durations) / len(durations),
                'min_duration': min(durations),
                'max_duration': max(durations),
                'errors': job_errors.get(job_id, 0)
            }

        return statistics

    def print_analysis(self):
        """분석 결과 출력"""
        stats = self.analyze_performance()

        print("\n" + "="*70)
        print("로그 분석 결과")
        print("="*70)

        for job_id, stat in stats.items():
            print(f"\nJob: {job_id}")
            print(f"  실행 횟수: {stat['count']}")
            print(f"  평균 실행 시간: {stat['avg_duration']:.3f}초")
            print(f"  최소 실행 시간: {stat['min_duration']:.3f}초")
            print(f"  최대 실행 시간: {stat['max_duration']:.3f}초")
            print(f"  오류 횟수: {stat['errors']}")

            # 성능 경고
            if stat['avg_duration'] > 5:
                print(f"  ⚠ 경고: 평균 실행 시간이 5초를 초과합니다!")

            if stat['errors'] > 0:
                print(f"  ⚠ 경고: {stat['errors']}개의 오류가 발생했습니다!")

        print("="*70)

# 사용 예제
analyzer = LogAnalyzer('scheduler_structured.log')
time.sleep(2)  # 로그가 쌓일 시간
analyzer.print_analysis()

import logging
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_ALL
from datetime import datetime
import time

class MultiLevelLogger:
    """다중 레벨 로거"""

    def __init__(self):
        # 각 레벨별 로거 설정
        self.debug_logger = self._setup_logger('debug', logging.DEBUG, 'debug.log')
        self.info_logger = self._setup_logger('info', logging.INFO, 'info.log')
        self.error_logger = self._setup_logger('error', logging.ERROR, 'error.log')

    def _setup_logger(self, name, level, filename):
        """로거 설정"""
        logger = logging.getLogger(name)
        logger.setLevel(level)

        handler = logging.FileHandler(filename)
        handler.setLevel(level)
        handler.setFormatter(logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s'
        ))

        logger.addHandler(handler)
        return logger

    def handle_event(self, event):
        """이벤트별 로깅"""
        event_type = type(event).__name__
        timestamp = datetime.now()

        # DEBUG: 모든 이벤트
        self.debug_logger.debug(f"{event_type}: {event}")

        # INFO: Job 실행 관련
        if 'Executed' in event_type or 'Submitted' in event_type:
            self.info_logger.info(f"{event_type} - Job: {event.job_id}")

        # ERROR: 오류 관련
        if 'Error' in event_type or 'Missed' in event_type:
            error_msg = f"{event_type} - Job: {event.job_id}"
            if hasattr(event, 'exception'):
                error_msg += f" - Exception: {event.exception}"
            self.error_logger.error(error_msg)

에러 처리​

적절한 에러 처리는 스케줄러의 안정성을 보장합니다.

전역 에러 핸들러​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_ERROR
from datetime import datetime
import time
import traceback

class GlobalErrorHandler:
    """전역 에러 핸들러"""

    def __init__(self):
        self.error_count = 0
        self.error_history = []

    def handle_job_error(self, event):
        """Job 오류 처리"""
        self.error_count += 1

        error_info = {
            'timestamp': datetime.now(),
            'job_id': event.job_id,
            'exception_type': type(event.exception).__name__,
            'exception_message': str(event.exception),
            'traceback': event.traceback
        }

        self.error_history.append(error_info)

        # 오류 로깅
        self._log_error(error_info)

        # 알림 전송 (조건부)
        if self._should_alert(error_info):
            self._send_alert(error_info)

    def _log_error(self, error_info):
        """오류 로깅"""
        print(f"\n{'='*60}")
        print(f"[오류 #{self.error_count}] {error_info['timestamp']:%Y-%m-%d %H:%M:%S}")
        print(f"Job ID: {error_info['job_id']}")
        print(f"예외 타입: {error_info['exception_type']}")
        print(f"예외 메시지: {error_info['exception_message']}")
        print(f"\n트레이스백:\n{error_info['traceback']}")
        print(f"{'='*60}\n")

    def _should_alert(self, error_info):
        """알림 전송 여부 결정"""
        # 동일 Job에서 5분 내 3회 이상 오류 발생 시
        recent_errors = [
            e for e in self.error_history
            if e['job_id'] == error_info['job_id']
            and (error_info['timestamp'] - e['timestamp']).seconds < 300
        ]

        return len(recent_errors) >= 3

    def _send_alert(self, error_info):
        """알림 전송 (시뮬레이션)"""
        print(f"\n🚨 알림: Job {error_info['job_id']}에서 반복적 오류 발생!")
        print(f"   최근 5분간 {len([e for e in self.error_history if e['job_id'] == error_info['job_id']])}회 오류\n")

    def get_error_summary(self):
        """오류 요약"""
        from collections import Counter

        summary = {
            'total_errors': self.error_count,
            'unique_jobs_with_errors': len(set(e['job_id'] for e in self.error_history)),
            'most_common_errors': Counter(
                e['exception_type'] for e in self.error_history
            ).most_common(5)
        }

        return summary

    print(f"\n오류 요약:")
    print(f"  총 오류: {summary['total_errors']}회")
    print(f"  오류 발생 Job 수: {summary['unique_jobs_with_errors']}개")
    print(f"  주요 오류 타입:")
    for error_type, count in summary['most_common_errors']:
        print(f"    - {error_type}: {count}회")

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_ERROR
from datetime import datetime, timedelta
import time
from functools import wraps

class RetryManager:
    """재시도 관리자"""

    def __init__(self, scheduler, max_retries=3, retry_delay=60):
        self.scheduler = scheduler
        self.max_retries = max_retries
        self.retry_delay = retry_delay
        self.retry_counts = {}

    def handle_job_error(self, event):
        """Job 오류 처리 및 재시도"""
        job_id = event.job_id

        # 재시도 횟수 증가
        self.retry_counts[job_id] = self.retry_counts.get(job_id, 0) + 1

        if self.retry_counts[job_id] <= self.max_retries:
            print(f"\n[{datetime.now():%H:%M:%S}] "
                  f"Job {job_id} 재시도 {self.retry_counts[job_id]}/{self.max_retries}")

            # 원본 Job 가져오기
            job = self.scheduler.get_job(job_id)

            if job:
                # 재시도 Job 스케줄
                retry_time = datetime.now() + timedelta(seconds=self.retry_delay)
                retry_job_id = f"{job_id}_retry_{self.retry_counts[job_id]}"

                self.scheduler.add_job(
                    job.func,
                    'date',
                    run_date=retry_time,
                    args=job.args,
                    kwargs=job.kwargs,
                    id=retry_job_id
                )

                print(f"  -> {self.retry_delay}초 후 재시도 예정")
        else:
            print(f"\n[{datetime.now():%H:%M:%S}] "
                  f"Job {job_id} 최대 재시도 횟수 초과, 포기")

            # 알림 전송 등
            self._notify_failure(job_id)

    def _notify_failure(self, job_id):
        """실패 알림"""
        print(f"  🚨 Job {job_id}가 {self.max_retries}번 재시도 후 실패했습니다!")

scheduler = BackgroundScheduler()
retry_manager = RetryManager(scheduler, max_retries=3, retry_delay=5)

scheduler.add_listener(retry_manager.handle_job_error, EVENT_JOB_ERROR)

attempt_counter = {}

def unreliable_job(job_name):
    """불안정한 작업 (4번째 시도에 성공)"""

    print(f"  [{job_name}] 성공!")
    return "성공"

scheduler.add_job(unreliable_job, 'date', args=['job_1'], id='job_1')

회로 차단기 패턴​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta
import time
from enum import Enum

class CircuitState(Enum):
    """회로 상태"""
    CLOSED = "closed"  # 정상
    OPEN = "open"  # 차단
    HALF_OPEN = "half_open"  # 반개방

class CircuitBreaker:
    """회로 차단기"""

    def __init__(self, failure_threshold=3, recovery_timeout=60):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failure_count = 0
        self.state = CircuitState.CLOSED
        self.last_failure_time = None
        self.state_change_time = datetime.now()

    def call(self, func, *args, **kwargs):
        """함수 호출 (회로 차단기 통과)"""
        # 회로 상태 업데이트
        self._update_state()

        if self.state == CircuitState.OPEN:
            raise Exception("회로 차단기 OPEN: 호출 거부")

        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result

        except Exception as e:
            self._on_failure()
            raise

    def _update_state(self):
        """상태 업데이트"""
        if self.state == CircuitState.OPEN:
            # 복구 타임아웃 확인
            if datetime.now() - self.state_change_time > timedelta(seconds=self.recovery_timeout):
                print(f"[{datetime.now():%H:%M:%S}] 회로 상태: OPEN -> HALF_OPEN")
                self.state = CircuitState.HALF_OPEN
                self.state_change_time = datetime.now()

    def _on_success(self):
        """성공 처리"""
        self.failure_count = 0

        if self.state == CircuitState.HALF_OPEN:
            print(f"[{datetime.now():%H:%M:%S}] 회로 상태: HALF_OPEN -> CLOSED")
            self.state = CircuitState.CLOSED
            self.state_change_time = datetime.now()

    def _on_failure(self):
        """실패 처리"""
        self.failure_count += 1
        self.last_failure_time = datetime.now()

        print(f"[{datetime.now():%H:%M:%S}] 실패 횟수: {self.failure_count}/{self.failure_threshold}")

        if self.failure_count >= self.failure_threshold:
            print(f"[{datetime.now():%H:%M:%S}] 회로 상태: CLOSED -> OPEN")
            self.state = CircuitState.OPEN
            self.state_change_time = datetime.now()

# 사용 예제
scheduler = BackgroundScheduler()
circuit_breaker = CircuitBreaker(failure_threshold=3, recovery_timeout=10)

failure_count = 0

def protected_job():
    """회로 차단기로 보호되는 작업"""
    global failure_count

    def actual_work():
        global failure_count
        failure_count += 1

        # 처음 5번은 실패, 이후 성공
        if failure_count <= 5:
            raise Exception("외부 서비스 오류")

        print("  작업 성공!")
        return "성공"

    try:
        result = circuit_breaker.call(actual_work)
        print(f"  결과: {result}")
    except Exception as e:
        print(f"  오류: {e}")

scheduler.add_job(protected_job, 'interval', seconds=3, id='protected')

scheduler.start()

    time.sleep(40)
except KeyboardInterrupt:
    scheduler.shutdown()

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR
from datetime import datetime
from collections import defaultdict
import time

class MetricsCollector:
    """메트릭 수집기"""

    def __init__(self):
        self.metrics = {
            'jobs_executed': 0,
            'jobs_failed': 0,
            'total_execution_time': 0,
            'execution_times': defaultdict(list),
            'error_counts': defaultdict(int)
        }
        self.start_time = datetime.now()

    def record_execution(self, event):
        """실행 메트릭 기록"""
        self.metrics['jobs_executed'] += 1

        # 실행 시간 기록 (여기서는 간단히 구현)
        job_id = event.job_id
        # 실제로는 시작 시간을 추적해야 함
        execution_time = 1.0  # 예시
        self.metrics['execution_times'][job_id].append(execution_time)
        self.metrics['total_execution_time'] += execution_time

    def record_error(self, event):
        """오류 메트릭 기록"""
        self.metrics['jobs_failed'] += 1
        self.metrics['error_counts'][event.job_id] += 1

    def get_metrics(self):
        """메트릭 조회"""
        uptime = (datetime.now() - self.start_time).total_seconds()

        return {
            'uptime_seconds': uptime,
            'jobs_executed': self.metrics['jobs_executed'],
            'jobs_failed': self.metrics['jobs_failed'],
            'success_rate': self._calculate_success_rate(),
            'avg_execution_time': self._calculate_avg_execution_time(),
            'jobs_per_minute': self.metrics['jobs_executed'] / (uptime / 60) if uptime > 0 else 0
        }

    def _calculate_success_rate(self):
        """성공률 계산"""
        total = self.metrics['jobs_executed'] + self.metrics['jobs_failed']
        if total == 0:
            return 0
        return (self.metrics['jobs_executed'] / total) * 100

    def _calculate_avg_execution_time(self):
        """평균 실행 시간 계산"""
        total_jobs = self.metrics['jobs_executed']
        if total_jobs == 0:
            return 0
        return self.metrics['total_execution_time'] / total_jobs

    def print_metrics(self):
        """메트릭 출력"""
        metrics = self.get_metrics()

        print("\n" + "="*60)
        print("스케줄러 메트릭")
        print("="*60)
        print(f"가동 시간: {metrics['uptime_seconds']:.0f}초")
        print(f"실행된 Job: {metrics['jobs_executed']}개")
        print(f"실패한 Job: {metrics['jobs_failed']}개")
        print(f"성공률: {metrics['success_rate']:.1f}%")
        print(f"평균 실행 시간: {metrics['avg_execution_time']:.3f}초")
        print(f"분당 Job 수: {metrics['jobs_per_minute']:.1f}개")
        print("="*60)

scheduler = BackgroundScheduler()

scheduler.add_listener(metrics.record_execution, EVENT_JOB_EXECUTED)

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_EXECUTED, EVENT_JOB_ERROR
from prometheus_client import Counter, Histogram, Gauge, start_http_server
import time

# Prometheus 메트릭 정의
jobs_executed_total = Counter(
    'scheduler_jobs_executed_total',
    'Total number of jobs executed',
    ['job_id']
)

jobs_failed_total = Counter(
    'scheduler_jobs_failed_total',
    'Total number of jobs failed',
    ['job_id']
)

job_duration_seconds = Histogram(
    'scheduler_job_duration_seconds',
    'Job execution duration in seconds',
    ['job_id']
)

active_jobs = Gauge(
    'scheduler_active_jobs',
    'Number of currently active jobs'
)

class PrometheusMetrics:
    """Prometheus 메트릭 수집기"""

    def __init__(self, scheduler):
        self.scheduler = scheduler
        self.job_start_times = {}

    def record_job_start(self, job_id):
        """Job 시작 기록"""
        self.job_start_times[job_id] = time.time()
        active_jobs.inc()

    def record_job_end(self, event):
        """Job 종료 기록"""
        job_id = event.job_id

        # 실행 시간 계산
        if job_id in self.job_start_times:
            duration = time.time() - self.job_start_times[job_id]
            job_duration_seconds.labels(job_id=job_id).observe(duration)
            del self.job_start_times[job_id]

        jobs_executed_total.labels(job_id=job_id).inc()
        active_jobs.dec()

    def record_job_error(self, event):
        """Job 오류 기록"""
        job_id = event.job_id

        if job_id in self.job_start_times:
            del self.job_start_times[job_id]

        jobs_failed_total.labels(job_id=job_id).inc()
        active_jobs.dec()

# Prometheus 메트릭 서버 시작 (포트 8000)
