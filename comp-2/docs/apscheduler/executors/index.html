5. Executors

Executor는 APScheduler에서 Job을 실제로 실행하는 컴포넌트입니다. 이 문서에서는 다양한 Executor 타입과 그 활용법을 상세히 다룹니다.

Executor 개요​

APScheduler는 여러 종류의 Executor를 제공합니다:

  • ThreadPoolExecutor: 스레드 풀을 사용하여 Job 실행 (기본값)
  • ProcessPoolExecutor: 프로세스 풀을 사용하여 CPU 집약적 작업 실행
  • AsyncIOExecutor: asyncio 이벤트 루프에서 비동기 작업 실행
  • GeventExecutor: gevent를 사용한 실행
  • TornadoExecutor: Tornado IOLoop에서 실행
  • TwistedExecutor: Twisted reactor에서 실행

이 문서에서는 가장 많이 사용되는 상위 3가지 Executor를 중점적으로 다룹니다.

ThreadPoolExecutor​

ThreadPoolExecutor는 스레드 풀을 사용하여 Job을 실행합니다. I/O 바운드 작업에 적합합니다.

기본 사용법​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ThreadPoolExecutor
from datetime import datetime
import time
import threading

def io_bound_job(job_id):
    """I/O 바운드 작업"""
    thread_id = threading.current_thread().name
    print(f"[{datetime.now():%H:%M:%S}] Job {job_id} 시작 (스레드: {thread_id})")
    time.sleep(2)  # I/O 시뮬레이션
    print(f"[{datetime.now():%H:%M:%S}] Job {job_id} 완료 (스레드: {thread_id})")

# ThreadPoolExecutor 설정
executors = {
    'default': ThreadPoolExecutor(max_workers=5)
}

scheduler = BackgroundScheduler(executors=executors)

# 여러 Job 추가
for i in range(10):
    scheduler.add_job(
        io_bound_job,
        'interval',
        seconds=1,
        args=[i],
        id=f'job_{i}'
    )

scheduler.start()

try:
    time.sleep(15)
except KeyboardInterrupt:
    scheduler.shutdown()

Worker 수 동적 조정​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ThreadPoolExecutor
from concurrent.futures import ThreadPoolExecutor as ConcurrentThreadPool
from datetime import datetime
import time
import threading

class DynamicThreadPoolExecutor:
    """동적으로 워커 수를 조정하는 Executor"""

    def __init__(self, min_workers=2, max_workers=10):
        self.min_workers = min_workers
        self.max_workers = max_workers
        self.current_workers = min_workers
        self.executor = ConcurrentThreadPool(max_workers=self.current_workers)
        self.active_jobs = 0
        self.lock = threading.Lock()

    def submit(self, fn, *args, **kwargs):
        """작업 제출"""
        with self.lock:
            self.active_jobs += 1
            self._adjust_workers()

        future = self.executor.submit(self._wrapped_fn, fn, args, kwargs)
        return future

    def _wrapped_fn(self, fn, args, kwargs):
        """작업 래퍼"""
        try:
            return fn(*args, **kwargs)
        finally:
            with self.lock:
                self.active_jobs -= 1
                self._adjust_workers()

    def _adjust_workers(self):
        """워커 수 조정"""
        # 활성 작업이 많으면 워커 증가
        if self.active_jobs > self.current_workers * 0.8:
            new_workers = min(self.current_workers + 2, self.max_workers)
            if new_workers != self.current_workers:
                print(f"워커 수 증가: {self.current_workers} -> {new_workers}")
                self.current_workers = new_workers
                self.executor._max_workers = new_workers

        # 활성 작업이 적으면 워커 감소
        elif self.active_jobs < self.current_workers * 0.3:
            new_workers = max(self.current_workers - 1, self.min_workers)
            if new_workers != self.current_workers:
                print(f"워커 수 감소: {self.current_workers} -> {new_workers}")
                self.current_workers = new_workers

# 사용 예제는 표준 ThreadPoolExecutor로 대체
executors = {
    'default': ThreadPoolExecutor(max_workers=10)
}

scheduler = BackgroundScheduler(executors=executors)

def task(n):
    thread_name = threading.current_thread().name
    print(f"작업 {n} (스레드: {thread_name})")
    time.sleep(1)

for i in range(20):
    scheduler.add_job(task, 'date', args=[i])

scheduler.start()
time.sleep(10)
scheduler.shutdown()

스레드 안전성​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ThreadPoolExecutor
import threading
import time

class ThreadSafeCounter:
    """스레드 안전한 카운터"""

    def __init__(self):
        self.count = 0
        self.lock = threading.Lock()

    def increment(self):
        """카운터 증가"""
        with self.lock:
            self.count += 1
            print(f"카운터: {self.count} (스레드: {threading.current_thread().name})")

    def get_count(self):
        """현재 카운터 값"""
        with self.lock:
            return self.count

# 공유 자원
counter = ThreadSafeCounter()

executors = {
    'default': ThreadPoolExecutor(max_workers=5)
}

scheduler = BackgroundScheduler(executors=executors)

def increment_task():
    """카운터 증가 작업"""
    counter.increment()
    time.sleep(0.1)

# 여러 Job이 동시에 공유 자원에 접근
for i in range(10):
    scheduler.add_job(increment_task, 'date')

scheduler.start()

time.sleep(5)

print(f"\n최종 카운터 값: {counter.get_count()}")

scheduler.shutdown()

Connection Pool 관리​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ThreadPoolExecutor
import threading
import time
from queue import Queue

class ConnectionPool:
    """데이터베이스 연결 풀 시뮬레이션"""

    def __init__(self, max_connections=5):
        self.max_connections = max_connections
        self.pool = Queue(maxsize=max_connections)
        self.lock = threading.Lock()

        # 연결 초기화
        for i in range(max_connections):
            self.pool.put(f"Connection-{i}")

    def get_connection(self):
        """연결 가져오기"""
        conn = self.pool.get()
        print(f"연결 획득: {conn} (스레드: {threading.current_thread().name})")
        return conn

    def release_connection(self, conn):
        """연결 반환"""
        print(f"연결 반환: {conn} (스레드: {threading.current_thread().name})")
        self.pool.put(conn)

# 전역 연결 풀
db_pool = ConnectionPool(max_connections=3)

def database_job(query):
    """데이터베이스 작업"""
    # 연결 획득
    conn = db_pool.get_connection()

    try:
        # 쿼리 실행 시뮬레이션
        print(f"쿼리 실행: {query}")
        time.sleep(2)
        print(f"쿼리 완료: {query}")
    finally:
        # 연결 반환
        db_pool.release_connection(conn)

executors = {
    'default': ThreadPoolExecutor(max_workers=5)
}

scheduler = BackgroundScheduler(executors=executors)

# 여러 데이터베이스 작업 스케줄
queries = [
    "SELECT * FROM users",
    "SELECT * FROM orders",
    "SELECT * FROM products",
    "UPDATE inventory",
    "DELETE old_logs"
]

for query in queries:
    scheduler.add_job(database_job, 'date', args=[query])

scheduler.start()

time.sleep(10)

scheduler.shutdown()

ProcessPoolExecutor​

ProcessPoolExecutor는 프로세스 풀을 사용하여 Job을 실행합니다. CPU 바운드 작업에 적합합니다.

기본 사용법​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ProcessPoolExecutor
import os
import time

def cpu_intensive_job(n):
    """CPU 집약적 작업"""
    pid = os.getpid()
    print(f"프로세스 {pid}에서 작업 {n} 시작")

    # CPU 집약적 계산
    result = 0
    for i in range(10**7):
        result += i ** 2

    print(f"프로세스 {pid}에서 작업 {n} 완료: {result % 1000}")
    return result

# ProcessPoolExecutor 설정
executors = {
    'default': ProcessPoolExecutor(max_workers=4)
}

scheduler = BackgroundScheduler(executors=executors)

# CPU 집약적 작업 추가
for i in range(8):
    scheduler.add_job(
        cpu_intensive_job,
        'date',
        args=[i],
        id=f'cpu_job_{i}',
        executor='default'
    )

scheduler.start()

print(f"메인 프로세스 PID: {os.getpid()}")

try:
    time.sleep(30)
except KeyboardInterrupt:
    scheduler.shutdown()

from apscheduler.schedulers.background import BackgroundScheduler

    """프로세스 작업"""

    )

time.sleep(10)

# 결과 확인
print("\n공유 딕셔너리:")
for key, value in shared_dict.items():
    print(f"  {key}: {value}")

print("\n공유 리스트:")
for item in shared_list:
    print(f"  - {item}")

scheduler.shutdown()

비동기 HTTP 요청​

import asyncio
import aiohttp
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from datetime import datetime

async def fetch_url(url, session):
    """비동기 URL 가져오기"""
    try:
        async with session.get(url, timeout=10) as response:
            data = await response.text()
            print(f"[{datetime.now():%H:%M:%S}] {url}: {response.status} ({len(data)} bytes)")
            return data
    except Exception as e:
        print(f"[{datetime.now():%H:%M:%S}] {url} 오류: {e}")
        return None

async def fetch_multiple_urls(urls):
    """여러 URL 동시 가져오기"""
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(url, session) for url in urls]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        return results

async def scheduled_fetch():
    """스케줄된 URL 수집 작업"""
    urls = [
        'https://httpbin.org/delay/1',
        'https://httpbin.org/delay/2',
        'https://httpbin.org/delay/1',
    ]

    print(f"\n[{datetime.now():%H:%M:%S}] URL 수집 시작")
    results = await fetch_multiple_urls(urls)
    print(f"[{datetime.now():%H:%M:%S}] 수집 완료: {len([r for r in results if r])}개 성공")

async def main():
    scheduler = AsyncIOScheduler()

    # 10초마다 URL 수집
    scheduler.add_job(
        scheduled_fetch,
        'interval',
        seconds=10,
        id='url_fetch'
    )

    scheduler.start()

    print("비동기 URL 수집 스케줄러 실행 중...")

    try:
        await asyncio.sleep(35)
    except asyncio.CancelledError:
        pass
    finally:
        scheduler.shutdown()

if __name__ == '__main__':
    asyncio.run(main())

import asyncio

class AsyncDBPool:

비동기 Task 큐​

import asyncio
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from datetime import datetime
from collections import deque

class AsyncTaskQueue:
    """비동기 Task 큐"""

    def __init__(self):
        self.queue = deque()
        self.processing = False

    async def add_task(self, task_func, *args, **kwargs):
        """Task 추가"""
        self.queue.append((task_func, args, kwargs))
        print(f"Task 추가됨 (큐 크기: {len(self.queue)})")

    async def process_queue(self):
        """큐 처리"""
        if self.processing:
            print("이미 처리 중...")
            return

        self.processing = True
        print(f"\n[{datetime.now():%H:%M:%S}] 큐 처리 시작 (크기: {len(self.queue)})")

        while self.queue:
            task_func, args, kwargs = self.queue.popleft()

            try:
                await task_func(*args, **kwargs)
            except Exception as e:
                print(f"Task 오류: {e}")

        print(f"[{datetime.now():%H:%M:%S}] 큐 처리 완료")
        self.processing = False

# 전역 큐
task_queue = AsyncTaskQueue()

async def sample_task(task_id):
    """샘플 Task"""
    print(f"  Task {task_id} 실행 중...")
    await asyncio.sleep(1)
    print(f"  Task {task_id} 완료")

async def add_tasks():
    """Task 추가"""
    for i in range(5):
        await task_queue.add_task(sample_task, i)

async def main():
    scheduler = AsyncIOScheduler()

    # 5초마다 Task 추가
    scheduler.add_job(add_tasks, 'interval', seconds=5, id='add_tasks')

    # 10초마다 큐 처리
    scheduler.add_job(task_queue.process_queue, 'interval', seconds=10, id='process_queue')

    scheduler.start()

    print("비동기 Task 큐 스케줄러 실행 중...")

    try:
        await asyncio.sleep(30)
    except asyncio.CancelledError:
        pass
    finally:
        scheduler.shutdown()

if __name__ == '__main__':
    asyncio.run(main())

커스텀 Executor​

특별한 요구사항이 있을 때 커스텀 Executor를 만들 수 있습니다.

기본 커스텀 Executor​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.base import BaseExecutor
from concurrent.futures import ThreadPoolExecutor
import time
from datetime import datetime

class LoggingExecutor(BaseExecutor):
    """로깅 기능이 추가된 Executor"""

    def __init__(self, max_workers=10):
        super().__init__()
        self.executor = ThreadPoolExecutor(max_workers=max_workers)

    def start(self, scheduler, alias):
        """Executor 시작"""
        super().start(scheduler, alias)
        print(f"LoggingExecutor '{alias}' 시작됨 (워커: {self.executor._max_workers})")

    def shutdown(self, wait=True):
        """Executor 종료"""
        print(f"LoggingExecutor 종료 중... (wait={wait})")
        self.executor.shutdown(wait=wait)

    def _do_submit_job(self, job, run_times):
        """Job 제출"""
        def wrapped_job():
            print(f"[{datetime.now():%H:%M:%S}] Job {job.id} 시작")
            start_time = time.time()

            try:
                return job.func(*job.args, **job.kwargs)
            finally:
                duration = time.time() - start_time
                print(f"[{datetime.now():%H:%M:%S}] Job {job.id} 완료 (소요: {duration:.2f}초)")

        future = self.executor.submit(wrapped_job)
        return future

executors = {

scheduler = BackgroundScheduler(executors=executors)

우선순위 기반 Executor​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.base import BaseExecutor
from concurrent.futures import ThreadPoolExecutor
from queue import PriorityQueue
import threading
import time

class PriorityExecutor(BaseExecutor):
    """우선순위 기반 Executor"""

    def __init__(self, max_workers=5):
        super().__init__()
        self.max_workers = max_workers
        self.queue = PriorityQueue()
        self.workers = []
        self.shutdown_flag = False

    def start(self, scheduler, alias):
        """Executor 시작"""
        super().start(scheduler, alias)

        # 워커 스레드 시작
        for i in range(self.max_workers):
            worker = threading.Thread(target=self._worker, name=f'Worker-{i}')
            worker.daemon = True
            worker.start()
            self.workers.append(worker)

        print(f"PriorityExecutor 시작됨 (워커: {self.max_workers})")

    def shutdown(self, wait=True):
        """Executor 종료"""
        self.shutdown_flag = True

        # 워커들에게 종료 신호
        for _ in self.workers:
            self.queue.put((0, None))

        if wait:
            for worker in self.workers:
                worker.join()

    def _worker(self):
        """워커 스레드"""
        while not self.shutdown_flag:
            try:
                priority, job_info = self.queue.get(timeout=1)

                if job_info is None:  # 종료 신호
                    break

                job, run_times = job_info
                print(f"[우선순위 {priority}] Job {job.id} 실행")

                try:
                    job.func(*job.args, **job.kwargs)
                except Exception as e:
                    print(f"Job {job.id} 오류: {e}")

            except Exception:
                continue

    def _do_submit_job(self, job, run_times):
        """Job 제출 (우선순위 추출)"""
        # Job의 kwargs에서 priority 추출 (없으면 5)
        priority = job.kwargs.pop('priority', 5)

        # 우선순위가 낮을수록 먼저 실행 (1이 가장 높은 우선순위)
        self.queue.put((priority, (job, run_times)))

        # Future 객체 반환 (간단한 구현)
        from concurrent.futures import Future
        future = Future()
        future.set_result(None)
        return future

# 사용 예제
executors = {
    'priority': PriorityExecutor(max_workers=3)
}

scheduler = BackgroundScheduler(executors=executors)

def task(name, priority):
    print(f"  작업 {name} 실행 (우선순위: {priority})")
    time.sleep(1)

# 다양한 우선순위의 Job 추가
scheduler.add_job(task, 'date', args=['Low', 9], kwargs={'priority': 9}, executor='priority')
scheduler.add_job(task, 'date', args=['High', 1], kwargs={'priority': 1}, executor='priority')
scheduler.add_job(task, 'date', args=['Medium', 5], kwargs={'priority': 5}, executor='priority')
scheduler.add_job(task, 'date', args=['Critical', 0], kwargs={'priority': 0}, executor='priority')

scheduler.start()

time.sleep(10)

scheduler.shutdown()

재시도 기능이 있는 Executor​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.pool import ThreadPoolExecutor
import time
import random
from datetime import datetime

class RetryWrapper:
    """재시도 기능을 제공하는 래퍼"""

    def __init__(self, func, max_retries=3, retry_delay=1):
        self.func = func
        self.max_retries = max_retries
        self.retry_delay = retry_delay

    def __call__(self, *args, **kwargs):
        """재시도 로직"""
        for attempt in range(self.max_retries):
            try:
                print(f"[{datetime.now():%H:%M:%S}] 시도 {attempt + 1}/{self.max_retries}")
                result = self.func(*args, **kwargs)
                print(f"[{datetime.now():%H:%M:%S}] 성공!")
                return result
            except Exception as e:
                print(f"[{datetime.now():%H:%M:%S}] 실패: {e}")

                if attempt < self.max_retries - 1:
                    print(f"  {self.retry_delay}초 후 재시도...")
                    time.sleep(self.retry_delay)
                else:
                    print(f"  최대 재시도 횟수 초과")
                    raise

def unreliable_task(name):
    """불안정한 작업 (70% 확률로 실패)"""
    if random.random() < 0.7:
        raise Exception(f"{name}: 임의 오류 발생")

    print(f"  {name} 작업 완료")
    return f"{name} 결과"

# Retry Wrapper 적용
executors = {
    'default': ThreadPoolExecutor(max_workers=3)
}

scheduler = BackgroundScheduler(executors=executors)

# RetryWrapper로 감싸서 추가
wrapped_task = RetryWrapper(unreliable_task, max_retries=3, retry_delay=1)

for i in range(3):
    scheduler.add_job(
        wrapped_task,
        'date',
        args=[f'Task-{i}'],
        id=f'retry_task_{i}'
    )

scheduler.start()

time.sleep(15)

scheduler.shutdown()

리소스 제한 Executor​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.executors.base import BaseExecutor
from concurrent.futures import ThreadPoolExecutor
import time
import threading
from datetime import datetime

class ResourceLimitedExecutor(BaseExecutor):
    """리소스 제한이 있는 Executor"""

    def __init__(self, max_workers=5, max_memory_mb=100):
        super().__init__()
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.max_memory_mb = max_memory_mb
        self.current_memory = 0
        self.lock = threading.Lock()

    def _do_submit_job(self, job, run_times):
        """Job 제출 (리소스 확인)"""
        # Job의 예상 메모리 사용량 (kwargs에서 추출)
        memory_required = job.kwargs.pop('memory_mb', 10)

        def wrapped_job():
            # 메모리 할당
            with self.lock:
                while self.current_memory + memory_required > self.max_memory_mb:
                    print(f"[{datetime.now():%H:%M:%S}] "
                          f"메모리 부족, 대기 중... "
                          f"(현재: {self.current_memory}MB, 필요: {memory_required}MB)")
                    time.sleep(1)

                self.current_memory += memory_required
                print(f"[{datetime.now():%H:%M:%S}] "
                      f"Job {job.id} 시작 (메모리: {self.current_memory}MB)")

            try:
                return job.func(*job.args, **job.kwargs)
            finally:
                # 메모리 해제
                with self.lock:
                    self.current_memory -= memory_required
                    print(f"[{datetime.now():%H:%M:%S}] "
                          f"Job {job.id} 완료 (메모리: {self.current_memory}MB)")

        return self.executor.submit(wrapped_job)

    def shutdown(self, wait=True):
        """Executor 종료"""
        self.executor.shutdown(wait=wait)

# 사용 예제
executors = {
    'resource_limited': ResourceLimitedExecutor(max_workers=5, max_memory_mb=50)
}

scheduler = BackgroundScheduler(executors=executors)

def memory_intensive_task(name, size):
    print(f"  {name} 작업 실행 (크기: {size}MB)")
    time.sleep(2)
    print(f"  {name} 완료")

# 다양한 메모리 요구사항의 Job
tasks = [
    ('Task-A', 30),
    ('Task-B', 20),
    ('Task-C', 15),
    ('Task-D', 25),
]

for name, memory in tasks:
    scheduler.add_job(
        memory_intensive_task,
        'date',
        args=[name, memory],
        kwargs={'memory_mb': memory},
        executor='resource_limited'
    )

scheduler.start()

time.sleep(20)

scheduler.shutdown()
