4. Job 관리

Job은 APScheduler에서 실행될 작업을 나타냅니다. 이 문서에서는 Job을 추가, 수정, 제거, 조회하고 일시정지/재개하는 방법과 Misfire 처리에 대해 상세히 다룹니다.

Job 추가​

Job을 스케줄러에 추가하는 다양한 방법을 알아봅니다.

기본 추가 방법​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

scheduler = BackgroundScheduler()

def my_job(name, value):
    """작업 함수"""
    print(f"작업 실행: {name} = {value}, 시간: {datetime.now()}")

# add_job 메서드 사용
job = scheduler.add_job(
    func=my_job,
    trigger='interval',
    seconds=10,
    args=['param1', 42],
    id='job_1',
    name='My First Job',
    replace_existing=True
)

scheduler.start()

try:
    time.sleep(60)
except KeyboardInterrupt:
    scheduler.shutdown()

동적 Job 추가​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, timedelta
import time

class DynamicJobManager:
    """동적으로 Job을 추가하는 매니저"""

    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.job_counter = 0

    def start(self):
        """스케줄러 시작"""
        self.scheduler.start()

    def add_temporary_job(self, func, duration_seconds, interval_seconds=1):
        """임시 Job 추가 (일정 시간 후 자동 제거)"""
        self.job_counter += 1
        job_id = f'temp_job_{self.job_counter}'

        # Job 추가
        end_time = datetime.now() + timedelta(seconds=duration_seconds)
        self.scheduler.add_job(
            func,
            'interval',
            seconds=interval_seconds,
            end_date=end_time,
            id=job_id,
            name=f'Temporary Job {self.job_counter}'
        )

        print(f"임시 Job 추가: {job_id} (종료: {end_time})")
        return job_id

    def add_delayed_job(self, func, delay_seconds, *args, **kwargs):
        """지연 실행 Job 추가"""
        self.job_counter += 1
        job_id = f'delayed_job_{self.job_counter}'

        run_time = datetime.now() + timedelta(seconds=delay_seconds)
        self.scheduler.add_job(
            func,
            'date',
            run_date=run_time,
            args=args,
            kwargs=kwargs,
            id=job_id
        )

        print(f"지연 Job 추가: {job_id} (실행: {run_time})")
        return job_id

    def add_countdown_job(self, func, count, interval_seconds=1):
        """카운트다운 Job (지정 횟수만 실행)"""
        self.job_counter += 1
        job_id = f'countdown_job_{self.job_counter}'

        def countdown_wrapper():
            nonlocal count
            count -= 1
            print(f"카운트다운: {count}")
            func()

            if count <= 0:
                print(f"카운트다운 완료, Job 제거: {job_id}")
                self.scheduler.remove_job(job_id)

        self.scheduler.add_job(
            countdown_wrapper,
            'interval',
            seconds=interval_seconds,
            id=job_id
        )

        return job_id

    def shutdown(self):
        """스케줄러 종료"""
        self.scheduler.shutdown()

# 사용 예제
manager = DynamicJobManager()
manager.start()

def task():
    print(f"  작업 실행: {datetime.now()}")

# 30초 동안만 실행되는 임시 Job
manager.add_temporary_job(task, duration_seconds=30, interval_seconds=5)

# 10초 후에 실행되는 지연 Job
manager.add_delayed_job(lambda: print("지연된 작업 실행!"), delay_seconds=10)

# 5번만 실행되는 카운트다운 Job
manager.add_countdown_job(task, count=5, interval_seconds=3)

try:
    time.sleep(60)
except KeyboardInterrupt:
    manager.shutdown()

조건부 Job 수정​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time
import random

class AdaptiveJob:
    """성능에 따라 간격을 조정하는 Job"""

    def __init__(self, scheduler, job_id):
        self.scheduler = scheduler
        self.job_id = job_id
        self.execution_times = []
        self.min_interval = 3
        self.max_interval = 30
        self.current_interval = 10

    def execute(self):
        """작업 실행 및 성능 모니터링"""
        start_time = time.time()

        # 작업 수행
        print(f"작업 실행 (간격: {self.current_interval}초)")
        time.sleep(random.uniform(0.1, 1.0))  # 작업 시뮬레이션

        # 실행 시간 기록
        execution_time = time.time() - start_time
        self.execution_times.append(execution_time)

        # 최근 5번의 평균 실행 시간 계산
        if len(self.execution_times) > 5:
            self.execution_times.pop(0)

        avg_time = sum(self.execution_times) / len(self.execution_times)

        # 실행 시간에 따라 간격 조정
        if avg_time > 0.5 and self.current_interval < self.max_interval:
            # 느리면 간격 증가
            self.current_interval = min(self.current_interval + 2, self.max_interval)
            self._reschedule()
            print(f"  -> 간격 증가: {self.current_interval}초")
        elif avg_time < 0.3 and self.current_interval > self.min_interval:
            # 빠르면 간격 감소
            self.current_interval = max(self.current_interval - 2, self.min_interval)
            self._reschedule()
            print(f"  -> 간격 감소: {self.current_interval}초")

    def _reschedule(self):
        """Job 재스케줄"""
        self.scheduler.reschedule_job(
            self.job_id,
            trigger='interval',
            seconds=self.current_interval
        )

scheduler = BackgroundScheduler()
adaptive = AdaptiveJob(scheduler, 'adaptive_job')

scheduler.add_job(
    adaptive.execute,
    'interval',
    seconds=10,
    id='adaptive_job'
)

scheduler.start()

try:
    time.sleep(60)
except KeyboardInterrupt:
    scheduler.shutdown()

조건부 제거​

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import time

class SelfRemovingJob:
    """조건 만족 시 자동으로 제거되는 Job"""

    def __init__(self, scheduler, job_id, max_executions=5):
        self.scheduler = scheduler
        self.job_id = job_id
        self.max_executions = max_executions
        self.execution_count = 0

    def execute(self):
        """작업 실행 및 카운트"""
        self.execution_count += 1
        print(f"실행 {self.execution_count}/{self.max_executions}: {datetime.now()}")

        if self.execution_count >= self.max_executions:
            print("최대 실행 횟수 도달, Job 제거")
            self.scheduler.remove_job(self.job_id)

scheduler = BackgroundScheduler()
self_removing = SelfRemovingJob(scheduler, 'self_removing', max_executions=5)

scheduler.add_job(
    self_removing.execute,
    'interval',
    seconds=2,
    id='self_removing'
)

scheduler.start()

try:
    time.sleep(20)
except KeyboardInterrupt:
    scheduler.shutdown()

                  id='data_sync', tags=['data'])
scheduler.add_job(task, 'interval', seconds=20, args=['report', 'Daily'],
                  id='report_daily', tags=['report'])

print("\n데이터 관련 Job:")

Job 통계​

from apscheduler.schedulers.background import BackgroundScheduler
from collections import defaultdict
from datetime import datetime
import time

class JobStatistics:
    """Job 통계 수집"""

    def __init__(self, scheduler):
        self.scheduler = scheduler
        self.execution_counts = defaultdict(int)
        self.error_counts = defaultdict(int)

    def get_statistics(self):
        """Job 통계 조회"""
        stats = {
            'total_jobs': len(self.scheduler.get_jobs()),
            'jobs_by_trigger': self._count_by_trigger(),
            'next_executions': self._get_next_executions(),
            'execution_counts': dict(self.execution_counts),
            'error_counts': dict(self.error_counts)
        }
        return stats

    def _count_by_trigger(self):
        """트리거 타입별 Job 수"""
        counts = defaultdict(int)
        for job in self.scheduler.get_jobs():
            trigger_type = type(job.trigger).__name__
            counts[trigger_type] += 1
        return dict(counts)

    def _get_next_executions(self):
        """다음 실행 시간 목록"""
        executions = []
        for job in self.scheduler.get_jobs():
            if job.next_run_time:
                executions.append({
                    'job_id': job.id,
                    'next_run': job.next_run_time
                })
        return sorted(executions, key=lambda x: x['next_run'])

    def print_statistics(self):
        """통계 출력"""
        stats = self.get_statistics()

        print("\n" + "="*60)
        print("Job 통계")
        print("="*60)
        print(f"총 Job 수: {stats['total_jobs']}")

        print("\n트리거 타입별:")
        for trigger_type, count in stats['jobs_by_trigger'].items():
            print(f"  - {trigger_type}: {count}개")

        print("\n다음 실행 예정:")
        for item in stats['next_executions'][:5]:
            print(f"  - {item['job_id']}: {item['next_run']}")

        print("="*60)

scheduler = BackgroundScheduler()
statistics = JobStatistics(scheduler)

    print(f"작업 {n}")

# 다양한 Job 추가

    scheduler.add_job(task, 'interval', seconds=10+i*5, args=[i], id=f'interval_{i}')

for i in range(2):
    scheduler.add_job(task, 'cron', minute=f'*/{i+1}', args=[i], id=f'cron_{i}')

from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime, time as dt_time
import time

class ConditionalPauseJob:
    """조건에 따라 자동으로 일시정지/재개되는 Job"""

    def __init__(self, scheduler, job_id, work_start=9, work_end=18):
        self.scheduler = scheduler
        self.job_id = job_id
        self.work_start = work_start
        self.work_end = work_end
        self.is_paused = False

    def execute(self):
        """작업 실행"""
        current_hour = datetime.now().hour

        # 업무 시간 확인
        if self.work_start <= current_hour < self.work_end:
            if self.is_paused:
                print(f"업무 시간 시작 ({current_hour}시), Job 재개")
                self.scheduler.resume_job(self.job_id)
                self.is_paused = False
        else:
            if not self.is_paused:
                print(f"업무 시간 종료 ({current_hour}시), Job 일시정지")
                self.scheduler.pause_job(self.job_id)
                self.is_paused = True

        if not self.is_paused:
            print(f"작업 실행: {datetime.now()}")

scheduler = BackgroundScheduler()

scheduler.add_job(
    conditional.execute,
    'interval',
    seconds=5,
    id='conditional_job'
)

scheduler.start()

    time.sleep(60)
except KeyboardInterrupt:
    scheduler.shutdown()

Misfire 이벤트 처리​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_MISSED, EVENT_JOB_ERROR, EVENT_JOB_EXECUTED
from datetime import datetime
import time

scheduler = BackgroundScheduler()

# Misfire 카운터
misfire_count = {}

def misfire_listener(event):
    """Misfire 이벤트 리스너"""
    job_id = event.job_id
    misfire_count[job_id] = misfire_count.get(job_id, 0) + 1
    print(f"[MISFIRE] Job {job_id} 놓침 (총 {misfire_count[job_id]}회)")
    print(f"  예정 시간: {event.scheduled_run_time}")

def error_listener(event):
    """에러 이벤트 리스너"""
    print(f"[ERROR] Job {event.job_id}: {event.exception}")

def executed_listener(event):
    """실행 완료 이벤트 리스너"""
    print(f"[SUCCESS] Job {event.job_id} 실행 완료")

# 리스너 등록
scheduler.add_listener(misfire_listener, EVENT_JOB_MISSED)
scheduler.add_listener(error_listener, EVENT_JOB_ERROR)
scheduler.add_listener(executed_listener, EVENT_JOB_EXECUTED)

def unreliable_job():
    """불안정한 작업"""
    print(f"작업 실행: {datetime.now()}")
    time.sleep(7)  # 7초 소요

scheduler.add_job(
    unreliable_job,
    'interval',
    seconds=5,
    id='unreliable',
    misfire_grace_time=3,
    max_instances=1
)

scheduler.start()

try:
    time.sleep(30)
except KeyboardInterrupt:
    pass

# 최종 통계
print("\n=== Misfire 통계 ===")

    print(f"Job {job_id}: {count}회 놓침")

Misfire 복구 전략​

from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.events import EVENT_JOB_MISSED
from datetime import datetime, timedelta
import time

class MisfireRecovery:
    """Misfire 복구 전략"""

    def __init__(self, scheduler):
        self.scheduler = scheduler
        self.missed_jobs = []

    def misfire_handler(self, event):
        """Misfire 발생 시 복구 처리"""
        print(f"[MISFIRE] Job {event.job_id} 놓침")

        # 놓친 작업 기록
        self.missed_jobs.append({
            'job_id': event.job_id,
            'scheduled_time': event.scheduled_run_time,
            'detected_time': datetime.now()
        })

        # 즉시 재실행 스케줄
        job = self.scheduler.get_job(event.job_id)
        if job:
            print(f"  -> 5초 후 재실행 스케줄")
            retry_time = datetime.now() + timedelta(seconds=5)
            self.scheduler.add_job(
                job.func,
                'date',
                run_date=retry_time,
                args=job.args,
                kwargs=job.kwargs,
                id=f'{event.job_id}_retry_{len(self.missed_jobs)}'
            )

    def get_missed_jobs_report(self):
        """놓친 작업 리포트"""
        print("\n=== Misfire 리포트 ===")
        for missed in self.missed_jobs:
            delay = (missed['detected_time'] - missed['scheduled_time']).total_seconds()
            print(f"Job {missed['job_id']}:")
            print(f"  예정: {missed['scheduled_time']}")
            print(f"  감지: {missed['detected_time']}")
            print(f"  지연: {delay:.1f}초")

scheduler = BackgroundScheduler()
recovery = MisfireRecovery(scheduler)

scheduler.add_listener(recovery.misfire_handler, EVENT_JOB_MISSED)

def important_job():
    """중요한 작업"""
    print(f"중요 작업 실행: {datetime.now()}")
    time.sleep(8)

scheduler.add_job(
    important_job,
    'interval',
    seconds=5,
    id='important',
    misfire_grace_time=2,
    max_instances=1
)

scheduler.start()

try:
    time.sleep(30)
except KeyboardInterrupt:
    pass

recovery.get_missed_jobs_report()

scheduler.shutdown()
