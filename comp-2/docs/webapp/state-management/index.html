6. 상태 관리

Vue 애플리케이션에서 Pinia를 사용한 효율적인 상태 관리 방법을 학습합니다.

6.1 상태 관리가 필요한 이유​

컴포넌트 간 상태 공유의 어려움​

Vue 애플리케이션이 커질수록 여러 컴포넌트에서 동일한 상태를 공유해야 하는 경우가 증가합니다.

<!-- Props Drilling 문제 -->

  <div>
    <!-- 깊이 중첩된 컴포넌트에 props 전달 -->
    <GrandParent :user="user" />
  </div>
</template>

<script setup lang="ts">
const user = ref({
  id: 1,
  name: 'John Doe',
  email: 'john@example.com'
})
</script>
<!-- GrandParent.vue -->

  <Parent :user="user" />

  <Child :user="user" />

상태 관리 라이브러리의 필요성​

  • 중앙 집중식 상태 관리: 모든 컴포넌트에서 접근 가능한 전역 상태
  • 예측 가능한 상태 변경: 명확한 규칙으로 상태 업데이트
  • 개발자 도구 지원: 상태 변경 추적 및 디버깅
  • 타입 안정성: TypeScript와의 완벽한 통합

// src/stores/auth.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { User } from '@/types/store'
import { authAPI } from '@/api/auth'

export const useAuthStore = defineStore('auth', () => {
  // State
  const token = ref<string | null>(localStorage.getItem('token'))
  const user = ref<User | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  // Getters
  const isAuthenticated = computed(() => !!token.value && !!user.value)
  const isAdmin = computed(() => user.value?.role === 'admin')
  const userName = computed(() => user.value?.name ?? 'Guest')

  // Actions
  async function login(email: string, password: string) {
    loading.value = true
    error.value = null

    try {
      const response = await authAPI.login({ email, password })

      token.value = response.token
      user.value = response.user

      // 토큰을 localStorage에 저장
      localStorage.setItem('token', response.token)

      return true
    } catch (err: any) {
      error.value = err.message || '로그인에 실패했습니다.'
      return false
    } finally {
      loading.value = false
    }
  }

  async function register(email: string, password: string, name: string) {
    loading.value = true
    error.value = null

    try {
      const response = await authAPI.register({ email, password, name })

      token.value = response.token
      user.value = response.user

      localStorage.setItem('token', response.token)

      return true
    } catch (err: any) {
      error.value = err.message || '회원가입에 실패했습니다.'
      return false
    } finally {
      loading.value = false
    }
  }

  async function logout() {
    try {
      await authAPI.logout()
    } catch (err) {
      console.error('Logout error:', err)
    } finally {
      // 로컬 상태 초기화
      token.value = null
      user.value = null
      localStorage.removeItem('token')
    }
  }

  async function fetchUser() {
    if (!token.value) return

    loading.value = true
    error.value = null

    try {
      const response = await authAPI.me()
      user.value = response.user
    } catch (err: any) {
      // 토큰이 유효하지 않은 경우
      if (err.status === 401) {
        await logout()
      }
      error.value = err.message
    } finally {
      loading.value = false
    }
  }

  async function updateProfile(data: Partial<User>) {
    loading.value = true
    error.value = null

    try {
      const response = await authAPI.updateProfile(data)
      user.value = response.user
      return true
    } catch (err: any) {
      error.value = err.message || '프로필 업데이트에 실패했습니다.'
      return false
    } finally {
      loading.value = false
    }
  }

  function clearError() {
    error.value = null
  }

  return {
    // State
    token,
    user,
    loading,
    error,
    // Getters
    isAuthenticated,
    isAdmin,
    userName,
    // Actions
    login,
    register,
    logout,
    fetchUser,
    updateProfile,
    clearError
  }
})

// src/router/guards/auth.ts
import { useAuthStore } from '@/stores/auth'
import type { RouteLocationNormalized, NavigationGuardNext } from 'vue-router'

export async function authGuard(
  to: RouteLocationNormalized,
  from: RouteLocationNormalized,
  next: NavigationGuardNext
) {
  const authStore = useAuthStore()

  // 인증이 필요한 페이지
  if (to.meta.requiresAuth) {
    if (!authStore.isAuthenticated) {
      // 로그인 페이지로 리다이렉트
      next({
        name: 'login',
        query: { redirect: to.fullPath }
      })
      return
    }

    // 사용자 정보가 없으면 가져오기
    if (!authStore.user) {
      try {
        await authStore.fetchUser()
      } catch (err) {
        next({ name: 'login' })
        return
      }
    }

    // 관리자 권한이 필요한 페이지
    if (to.meta.requiresAdmin && !authStore.isAdmin) {
      next({ name: 'forbidden' })
      return
    }
  }

  // 로그인한 사용자는 로그인/회원가입 페이지 접근 불가
  if (to.meta.requiresGuest && authStore.isAuthenticated) {
    next({ name: 'home' })
    return
  }

  next()
}

6.7 API 데이터 캐싱​

Posts Store 구현​

// src/stores/posts.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Post } from '@/types/store'
import { postsAPI } from '@/api/posts'

export const usePostsStore = defineStore('posts', () => {
  // State
  const posts = ref<Post[]>([])
  const currentPost = ref<Post | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)
  const lastFetched = ref<number | null>(null)

  // 캐시 유효 시간 (5분)
  const CACHE_DURATION = 5 * 60 * 1000

  // Getters
  const sortedPosts = computed(() => {
    return [...posts.value].sort((a, b) => {
      return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()
    })
  })

  const getPostById = computed(() => {
    return (id: number) => posts.value.find(post => post.id === id)
  })

  const isCacheValid = computed(() => {
    if (!lastFetched.value) return false
    return Date.now() - lastFetched.value < CACHE_DURATION
  })

  // Actions
  async function fetchPosts(forceRefresh = false) {
    // 캐시가 유효하고 강제 새로고침이 아니면 스킵
    if (!forceRefresh && isCacheValid.value && posts.value.length > 0) {
      return
    }

    loading.value = true
    error.value = null

    try {
      const response = await postsAPI.getAll()
      posts.value = response.data
      lastFetched.value = Date.now()
    } catch (err: any) {
      error.value = err.message || '게시물을 불러오는데 실패했습니다.'
      throw err
    } finally {
      loading.value = false
    }
  }

  async function fetchPost(id: number) {
    loading.value = true
    error.value = null

    try {
      // 캐시에 있으면 사용
      const cached = getPostById.value(id)
      if (cached) {
        currentPost.value = cached
        loading.value = false
        return cached
      }

      // 없으면 API 호출
      const response = await postsAPI.getById(id)
      currentPost.value = response.data

      // posts 배열에도 추가
      const index = posts.value.findIndex(p => p.id === id)
      if (index === -1) {
        posts.value.push(response.data)
      } else {
        posts.value[index] = response.data
      }

      return response.data
    } catch (err: any) {
      error.value = err.message || '게시물을 불러오는데 실패했습니다.'
      throw err
    } finally {
      loading.value = false
    }
  }

  async function createPost(data: Omit<Post, 'id' | 'createdAt' | 'updatedAt'>) {
    loading.value = true
    error.value = null

    try {
      const response = await postsAPI.create(data)
      posts.value.unshift(response.data)
      return response.data
    } catch (err: any) {
      error.value = err.message || '게시물 작성에 실패했습니다.'
      throw err
    } finally {
      loading.value = false
    }
  }

  async function updatePost(id: number, data: Partial<Post>) {
    loading.value = true
    error.value = null

    try {
      const response = await postsAPI.update(id, data)

      // Store 업데이트
      const index = posts.value.findIndex(p => p.id === id)
      if (index !== -1) {
        posts.value[index] = response.data
      }

      if (currentPost.value?.id === id) {
        currentPost.value = response.data
      }

      return response.data
    } catch (err: any) {
      error.value = err.message || '게시물 수정에 실패했습니다.'
      throw err
    } finally {
      loading.value = false
    }
  }

  async function deletePost(id: number) {
    loading.value = true
    error.value = null

    try {
      await postsAPI.delete(id)

      // Store에서 제거
      posts.value = posts.value.filter(p => p.id !== id)

      if (currentPost.value?.id === id) {
        currentPost.value = null
      }
    } catch (err: any) {
      error.value = err.message || '게시물 삭제에 실패했습니다.'
      throw err
    } finally {
      loading.value = false
    }
  }

  function clearCache() {
    posts.value = []
    currentPost.value = null
    lastFetched.value = null
  }

  return {
    // State
    posts,
    currentPost,
    loading,
    error,
    // Getters
    sortedPosts,
    getPostById,
    isCacheValid,
    // Actions
    fetchPosts,
    fetchPost,
    createPost,
    updatePost,
    deletePost,
    clearCache
  }
})

<template>

    <button @click="refresh" :disabled="postsStore.loading">

      캐시된 데이터 ({{ timeSinceLastFetch }}초 전)

      {{ postsStore.error }}
    </div>

      <p>{{ post.content }}</p>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { usePostsStore } from '@/stores/posts'

const postsStore = usePostsStore()

const timeSinceLastFetch = computed(() => {
  if (!postsStore.lastFetched) return 0
  return Math.floor((Date.now() - postsStore.lastFetched) / 1000)
})

onMounted(async () => {
  // 캐시가 유효하면 API 호출 안 함
  await postsStore.fetchPosts()
})

  // 강제 새로고침
  await postsStore.fetchPosts(true)
}

캐싱 전략
  • Time-based 캐싱: 일정 시간 동안 캐시 유효
  • Force refresh: 사용자가 명시적으로 새로고침 가능
  • Optimistic updates: 낙관적 업데이트로 UX 개선
  • Cache invalidation: 데이터 변경 시 캐시 무효화

import { ref, computed } from 'vue'

interface Toast {

  message: string

  duration?: number
}

export const useUIStore = defineStore('ui', () => {
  // Loading states
  const loadingStates = ref<Record<string, boolean>>({})

  // Error states
  const errorStates = ref<Record<string, string | null>>({})

  // Toast notifications
  const toasts = ref<Toast[]>([])

  // Modal state

import { ref } from 'vue'
import { useUIStore } from '@/stores/ui'

export function useAsyncAction(key?: string) {
  const uiStore = useUIStore()
  const loading = ref(false)
  const error = ref<string | null>(null)

  async function execute<T>(
    action: () => Promise<T>,
    options: {
      showToast?: boolean
      successMessage?: string
      errorMessage?: string
    } = {}
  ): Promise<T | null> {
    loading.value = true
    error.value = null

    if (key) {
      uiStore.startLoading(key)
      uiStore.clearError(key)
    }

    try {
      const result = await action()

      if (options.showToast && options.successMessage) {
        uiStore.showToast(options.successMessage, 'success')
      }

      return result
    } catch (err: any) {
      const errorMsg = options.errorMessage || err.message || '작업에 실패했습니다.'
      error.value = errorMsg

      if (key) {
        uiStore.setError(key, errorMsg)
      }

      if (options.showToast) {
        uiStore.showToast(errorMsg, 'error')
      }

      return null
    } finally {
      loading.value = false
      if (key) {
        uiStore.stopLoading(key)
      }
    }
  }

  return {
    loading,
    error,
    execute
  }
}

    loading.value = true
    error.value = null

    try {
      const response = await userAPI.getAll()
      users.value = response.data
    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

  async function fetchUser(id: number) {
    const cached = getUserById.value(id)
    if (cached) return cached

    loading.value = true
    error.value = null

    try {
      const response = await userAPI.getById(id)
      const user = response.data

      const index = users.value.findIndex(u => u.id === id)
      if (index === -1) {
        users.value.push(user)
      } else {
        users.value[index] = user
      }

      return user
    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

  async function updateUser(id: number, data: Partial<User>) {
    loading.value = true
    error.value = null

    try {
      const response = await userAPI.update(id, data)
      const updatedUser = response.data

      const index = users.value.findIndex(u => u.id === id)

        users.value[index] = updatedUser

      }

    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

    }

    error.value = null

      comments.value = comments.value.filter(c => c.postId !== postId)

      // 새 댓글 추가

    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }

    error.value = null

      const response = await commentsAPI.create({ postId, content })

      return response.data
    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

  async function deleteComment(id: number) {
    loading.value = true
    error.value = null

    try {
      await commentsAPI.delete(id)
      comments.value = comments.value.filter(c => c.id !== id)
    } catch (err: any) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

    comments,
    loading,
    error,

  }

export function createCustomPersistPlugin() {
  return function customPersistPlugin({ store, options }: PiniaPluginContext) {
    // 특정 store만 처리
    if (options.persist) {
      const storageKey = `pinia-${store.$id}`

      // 초기화 시 복원
      const savedState = localStorage.getItem(storageKey)
      if (savedState) {
        try {
          const parsed = JSON.parse(savedState)
          store.$patch(parsed)
        } catch (error) {
          console.error('Failed to restore state:', error)
        }
      }

      // 상태 변경 시 저장
      store.$subscribe((mutation, state) => {
        try {
          localStorage.setItem(storageKey, JSON.stringify(state))
        } catch (error) {
          console.error('Failed to persist state:', error)
        }
      })

      // Store에 reset 메서드 추가
      store.$reset = () => {
        localStorage.removeItem(storageKey)
        store.$patch(options.state?.())
      }
    }
  }
}

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export interface Product {
  id: number
  name: string
  price: number
  image: string
}

export interface CartItem extends Product {
  quantity: number
}

interface Coupon {
  code: string
  discount: number
}

// 사용 가능한 쿠폰 목록
const AVAILABLE_COUPONS: Coupon[] = [
  { code: 'SAVE10', discount: 10 },
  { code: 'SAVE20', discount: 20 },
  { code: 'SAVE50', discount: 50 }
]

export const useCartStore = defineStore('cart', () => {
  // State
  const items = ref<CartItem[]>([])
  const appliedCoupon = ref<Coupon | null>(null)

  // Getters
  const itemCount = computed(() => {
    return items.value.reduce((total, item) => total + item.quantity, 0)
  })

  const subtotal = computed(() => {
    return items.value.reduce((total, item) => {
      return total + (item.price * item.quantity)
    }, 0)
  })

  const discountAmount = computed(() => {
    if (!appliedCoupon.value) return 0
    return subtotal.value * (appliedCoupon.value.discount / 100)
  })

  const total = computed(() => {
    return subtotal.value - discountAmount.value
  })

  const isEmpty = computed(() => items.value.length === 0)

  // Actions
  function addToCart(product: Product, quantity = 1) {
    const existingItem = items.value.find(item => item.id === product.id)

    if (existingItem) {
      existingItem.quantity += quantity
    } else {
      items.value.push({
        ...product,
        quantity
      })
    }
  }

  function removeFromCart(productId: number) {
    items.value = items.value.filter(item => item.id !== productId)
  }

  function updateQuantity(productId: number, quantity: number) {
    if (quantity <= 0) {
      removeFromCart(productId)
      return
    }

    const item = items.value.find(item => item.id === productId)
    if (item) {
      item.quantity = quantity
    }
  }

  function applyCoupon(code: string): boolean {
    const coupon = AVAILABLE_COUPONS.find(
      c => c.code.toLowerCase() === code.toLowerCase()
    )

    if (coupon) {
      appliedCoupon.value = coupon
      return true
    }

    return false
  }

  function removeCoupon() {
    appliedCoupon.value = null
  }

  function clearCart() {
    items.value = []
    appliedCoupon.value = null
  }

  function getItemById(productId: number) {
    return items.value.find(item => item.id === productId)
  }

  return {
    // State
    items,
    appliedCoupon,
    // Getters
    itemCount,
    subtotal,
    discountAmount,
    total,
    isEmpty,
    // Actions
    addToCart,
    removeFromCart,
    updateQuantity,
    applyCoupon,
    removeCoupon,
    clearCart,
    getItemById
  }
}, {
  persist: {
    storage: sessionStorage
  }
})

  createdAt: string

export const useNotificationsStore = defineStore('notifications', () => {
  // State
  const notifications = ref<Notification[]>([])
  const filterType = ref<Notification['type'] | 'all'>('all')

  // Getters
  const unreadCount = computed(() => {
    return notifications.value.filter(n => !n.read).length
  })

  const unreadNotifications = computed(() => {
    return notifications.value.filter(n => !n.read)
  })

  const readNotifications = computed(() => {
    return notifications.value.filter(n => n.read)
  })

    if (filterType.value === 'all') {
      return notifications.value
    }
    return notifications.value.filter(n => n.type === filterType.value)
  })

  const sortedNotifications = computed(() => {
    return [...filteredNotifications.value].sort((a, b) => {
      return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()
    })
  })

  // Actions
  function addNotification(
    notification: Omit<Notification, 'id' | 'read' | 'createdAt'>
  ) {
    const newNotification: Notification = {
      id: `notif-${Date.now()}-${Math.random()}`,
      read: false,
      createdAt: new Date().toISOString(),
      ...notification
    }

    notifications.value.unshift(newNotification)

    // 최대 100개까지만 유지
    if (notifications.value.length > 100) {
      notifications.value = notifications.value.slice(0, 100)

      notification.read = true
    }

    notifications.value.forEach(n => {
      n.read = true
    })
  }

  function deleteNotification(id: string) {
    notifications.value = notifications.value.filter(n => n.id !== id)
  }

    notifications.value = []
  }

        :key="notification.id"
        :class="['notification-item', `type-${notification.type}`, { unread: !notification.read }]"
        @click="notifStore.markAsRead(notification.id)"
      >
        <div class="notification-icon">
          <span v-if="notification.type === 'info'">ℹ</span>
          <span v-else-if="notification.type === 'success'">✓</span>
          <span v-else-if="notification.type === 'warning'">⚠</span>
          <span v-else>✕</span>
        </div>

        <div class="notification-content">
          <h4>{{ notification.title }}</h4>
          <p>{{ notification.message }}</p>
          <time>{{ formatTime(notification.createdAt) }}</time>
        </div>

        <button
          @click.stop="notifStore.deleteNotification(notification.id)"
          class="btn-delete"
        >
          ×
        </button>
      </div>
    </div>
  </div>
</template>

import { useNotificationsStore } from '@/stores/notifications'

function formatTime(dateString: string): string {
  const date = new Date(dateString)
  const now = new Date()
  const diffMs = now.getTime() - date.getTime()
  const diffMins = Math.floor(diffMs / 60000)
  const diffHours = Math.floor(diffMs / 3600000)
  const diffDays = Math.floor(diffMs / 86400000)

  if (diffMins < 1) return '방금 전'
  if (diffMins < 60) return `${diffMins}분 전`
  if (diffHours < 24) return `${diffHours}시간 전`
  if (diffDays < 7) return `${diffDays}일 전`

  return date.toLocaleDateString('ko-KR')
}
</script>
