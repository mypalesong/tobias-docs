7. 폼 처리 및 검증

사용자 입력을 안전하고 효율적으로 처리하는 방법을 배워봅시다.

7.1 폼 처리의 기본 개념​

폼은 사용자와 애플리케이션 간의 주요 상호작용 지점입니다.

**폼 처리 단계:**
1. 사용자 입력 수집
2. 클라이언트 측 검증 (즉각적 피드백)
3. 서버 측 검증 (보안)
4. 데이터 처리 및 저장
5. 결과 피드백

검증의 중요성:

  • 데이터 무결성 보장
  • 보안 위협 방지
  • 사용자 경험 개선

// src/schemas/user.schema.ts
import * as yup from 'yup'

export const userSchema = yup.object({
  // 이메일
  email: yup
    .string()
    .required('이메일은 필수입니다')
    .email('올바른 이메일 형식이 아닙니다'),

  // 비밀번호
  password: yup
    .string()
    .required('비밀번호는 필수입니다')
    .min(8, '비밀번호는 최소 8자 이상이어야 합니다')
    .matches(
      /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])/,
      '대문자, 소문자, 숫자, 특수문자를 포함해야 합니다'
    ),

  // 비밀번호 확인
  passwordConfirm: yup
    .string()
    .required('비밀번호 확인은 필수입니다')
    .oneOf([yup.ref('password')], '비밀번호가 일치하지 않습니다'),

  // 이름
  name: yup
    .string()
    .required('이름은 필수입니다')
    .min(2, '이름은 최소 2자 이상이어야 합니다')
    .max(50, '이름은 최대 50자까지 가능합니다'),

  // 나이
  age: yup
    .number()
    .required('나이는 필수입니다')
    .min(18, '만 18세 이상만 가입 가능합니다')
    .max(120, '올바른 나이를 입력해주세요'),

  // 전화번호
  phone: yup
    .string()
    .matches(/^010-\d{4}-\d{4}$/, '010-0000-0000 형식으로 입력해주세요'),

  // URL
  website: yup.string().url('올바른 URL 형식이 아닙니다'),

  // 약관 동의
  terms: yup
    .boolean()
    .oneOf([true], '약관에 동의해야 합니다')
})

export type UserFormData = yup.InferType<typeof userSchema>

7.5.1 기본 검증​

# app/schemas/user.py
from pydantic import BaseModel, EmailStr, Field, validator
from typing import Optional
import re

class UserCreate(BaseModel):
    email: EmailStr
    password: str = Field(..., min_length=8, max_length=100)
    name: str = Field(..., min_length=2, max_length=50)
    age: int = Field(..., ge=18, le=120)
    phone: Optional[str] = None

    @validator('password')
    def password_strength(cls, v):
        if not re.search(r'[A-Z]', v):
            raise ValueError('대문자를 포함해야 합니다')
        if not re.search(r'[a-z]', v):
            raise ValueError('소문자를 포함해야 합니다')
        if not re.search(r'\d', v):
            raise ValueError('숫자를 포함해야 합니다')
        if not re.search(r'[@$!%*?&]', v):
            raise ValueError('특수문자를 포함해야 합니다')
        return v

    @validator('phone')
    def phone_format(cls, v):
        if v and not re.match(r'^010-\d{4}-\d{4}$', v):
            raise ValueError('010-0000-0000 형식으로 입력해주세요')
        return v

    @validator('name')
    def name_alphanumeric(cls, v):
        if not v.replace(' ', '').isalnum():
            raise ValueError('이름은 알파벳과 숫자만 가능합니다')
        return v

    class Config:
        schema_extra = {
            "example": {
                "email": "user@example.com",
                "password": "Password123!",
                "name": "홍길동",
                "age": 25,
                "phone": "010-1234-5678"
            }
        }

7.5.2 고급 검증​

from pydantic import BaseModel, root_validator, constr
from datetime import date

class EventCreate(BaseModel):
    title: str = Field(..., min_length=3, max_length=200)
    start_date: date
    end_date: date
    max_participants: int = Field(..., gt=0, le=1000)

    @root_validator
    def check_dates(cls, values):
        start = values.get('start_date')
        end = values.get('end_date')

        if start and end:
            if end < start:
                raise ValueError('종료일은 시작일보다 이후여야 합니다')

            if (end - start).days > 365:
                raise ValueError('이벤트 기간은 최대 1년입니다')

        return values

# 정규식 검증
class ProductCreate(BaseModel):

    price: float = Field(..., gt=0)

7.6.2 Toast 알림​

// composables/useToast.ts
import { ref } from 'vue'

interface Toast {
  id: number
  message: string
  type: 'success' | 'error' | 'warning'
}

const toasts = ref<Toast[]>([])
let id = 0

export const useToast = () => {
  const showToast = (message: string, type: Toast['type'] = 'success') => {
    const toast: Toast = {
      id: id++,
      message,
      type
    }

    toasts.value.push(toast)

    setTimeout(() => {
      toasts.value = toasts.value.filter(t => t.id !== toast.id)
    }, 3000)
  }

  return {
    toasts,
    showToast,
    success: (msg: string) => showToast(msg, 'success'),
    error: (msg: string) => showToast(msg, 'error'),
    warning: (msg: string) => showToast(msg, 'warning')
  }
}

7.7 실시간 검증​

<script setup lang="ts">
import { ref, watch } from 'vue'
import { useField } from 'vee-validate'
import { debounce } from 'lodash-es'
import axios from 'axios'

const { value: email, errorMessage } = useField<string>('email')
const isChecking = ref(false)
const isAvailable = ref<boolean | null>(null)

// 이메일 중복 확인 (디바운스)
const checkEmailAvailability = debounce(async (email: string) => {
  if (!email || errorMessage.value) return

  isChecking.value = true
  try {
    const { data } = await axios.get(`/api/check-email?email=${email}`)
    isAvailable.value = data.available
  } catch (error) {
    console.error('Email check failed:', error)
  } finally {
    isChecking.value = false
  }
}, 500)

watch(email, (newEmail) => {
  isAvailable.value = null
  checkEmailAvailability(newEmail)
})
</script>

<template>
  <div>
    <input v-model="email" type="email" />
    <span v-if="isChecking">확인 중...</span>
    <span v-else-if="isAvailable === true" class="success">
      사용 가능한 이메일입니다
    </span>
    <span v-else-if="isAvailable === false" class="error">
      이미 사용 중인 이메일입니다
    </span>
  </div>
</template>

7.9 다중 스텝 폼​

<script setup lang="ts">
import { ref, computed } from 'vue'
import { useForm } from 'vee-validate'
import * as yup from 'yup'

const currentStep = ref(1)
const totalSteps = 3

const schemas = [
  // Step 1: 기본 정보
  yup.object({
    email: yup.string().required().email(),
    password: yup.string().required().min(8)
  }),
  // Step 2: 개인 정보
  yup.object({
    name: yup.string().required(),
    phone: yup.string().required()
  }),
  // Step 3: 추가 정보
  yup.object({
    address: yup.string().required(),
    interests: yup.array().min(1, '최소 1개 선택')
  })
]

const currentSchema = computed(() => schemas[currentStep.value - 1])

const { handleSubmit, values } = useForm({
  validationSchema: currentSchema
})

const nextStep = handleSubmit(() => {
  if (currentStep.value < totalSteps) {
    currentStep.value++
  } else {
    submitForm()
  }
})

const previousStep = () => {
  if (currentStep.value > 1) {
    currentStep.value--
  }
}

const submitForm = async () => {
  try {
    await axios.post('/api/register', values)
    console.log('Registration success')
  } catch (error) {
    console.error('Registration failed:', error)
  }
}
</script>

<template>
  <div class="multi-step-form">
    <!-- 진행 표시 -->
    <div class="progress">
      <div
        v-for="step in totalSteps"
        :key="step"
        :class="['step', { active: step === currentStep, completed: step < currentStep }]"
      >
        {{ step }}
      </div>
    </div>

    <!-- Step 1 -->
    <div v-show="currentStep === 1">
      <Field name="email" type="email" />
      <Field name="password" type="password" />
    </div>

    <!-- Step 2 -->
    <div v-show="currentStep === 2">
      <Field name="name" />
      <Field name="phone" />
    </div>

    <!-- Step 3 -->
    <div v-show="currentStep === 3">
      <Field name="address" />
      <Field name="interests" as="select" multiple />
    </div>

    <!-- 네비게이션 -->
    <div class="navigation">
      <button v-if="currentStep > 1" @click="previousStep">
        이전
      </button>
      <button @click="nextStep">
        {{ currentStep < totalSteps ? '다음' : '제출' }}
      </button>
    </div>
  </div>
</template>

7.10 파일 입력 처리​

<script setup lang="ts">
import { ref } from 'vue'

const file = ref<File | null>(null)
const preview = ref<string>('')
const error = ref('')

const handleFileChange = (event: Event) => {
  const target = event.target as HTMLInputElement
  const selectedFile = target.files?.[0]

  if (!selectedFile) return

  // 파일 크기 검증 (5MB)
  if (selectedFile.size > 5 * 1024 * 1024) {
    error.value = '파일 크기는 5MB 이하여야 합니다'
    return
  }

  // 파일 타입 검증
  const allowedTypes = ['image/jpeg', 'image/png', 'image/gif']
  if (!allowedTypes.includes(selectedFile.type)) {
    error.value = 'JPG, PNG, GIF 파일만 업로드 가능합니다'
    return
  }

  file.value = selectedFile
  error.value = ''

  // 이미지 미리보기
  const reader = new FileReader()
  reader.onload = (e) => {
    preview.value = e.target?.result as string
  }
  reader.readAsDataURL(selectedFile)
}

const uploadFile = async () => {
  if (!file.value) return

  const formData = new FormData()
  formData.append('file', file.value)

  try {
    const { data } = await axios.post('/api/upload', formData, {
      headers: {
        'Content-Type': 'multipart/form-data'
      }
    })
    console.log('Upload success:', data)
  } catch (error) {
    console.error('Upload failed:', error)
  }
}
</script>

<template>
  <div>
    <input
      type="file"
      accept="image/*"
      @change="handleFileChange"
    />
    <div v-if="error" class="error">{{ error }}</div>
    <img v-if="preview" :src="preview" alt="Preview" />
    <button @click="uploadFile" :disabled="!file">업로드</button>
  </div>
</template>
