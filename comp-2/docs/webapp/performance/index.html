12. 성능 최적화

웹 애플리케이션의 속도와 효율성을 극대화하는 방법을 배워봅시다.

12.1 성능 최적화 개요​

성능은 사용자 경험과 비즈니스 성공에 직접적인 영향을 미칩니다.

**성능 지표:**
- FCP (First Contentful Paint) - 첫 콘텐츠 렌더링 시간
- LCP (Largest Contentful Paint) - 최대 콘텐츠 렌더링 시간
- TTI (Time to Interactive) - 인터랙션 가능 시간
- TBT (Total Blocking Time) - 총 차단 시간
- CLS (Cumulative Layout Shift) - 누적 레이아웃 이동

**최적화 영역:**
- Backend: 데이터베이스, 캐싱, 비동기 처리
- Frontend: 번들 크기, 렌더링, 네트워크
- Infrastructure: CDN, 압축, HTTP/2

from sqlalchemy.orm import joinedload

@router.get("/posts")
def get_posts(db: Session = Depends(get_db)):
    posts = db.query(Post).options(

            "title": post.title,
            "author": post.author.name  # 이미 로드됨!

12.3.1 Redis 캐싱​

# app/cache.py
import redis
import json
from functools import wraps

redis_client = redis.Redis(
    host='localhost',
    port=6379,
    db=0,
    decode_responses=True
)

def cache_result(expire: int = 300):
    """캐시 데코레이터"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 캐시 키 생성
            cache_key = f"{func.__name__}:{str(args)}:{str(kwargs)}"

            # 캐시 확인
            cached = redis_client.get(cache_key)
            if cached:
                return json.loads(cached)

            # 함수 실행
            result = await func(*args, **kwargs)

            # 캐시 저장
            redis_client.setex(
                cache_key,
                expire,
                json.dumps(result)
            )

            return result
        return wrapper
    return decorator

# 사용 예
@router.get("/posts")
@cache_result(expire=600)  # 10분 캐싱
async def get_posts(db: Session = Depends(get_db)):
    return db.query(Post).all()

12.3.2 캐시 무효화​

class CacheManager:
    @staticmethod
    def invalidate_pattern(pattern: str):
        """패턴에 맞는 모든 캐시 삭제"""
        keys = redis_client.keys(pattern)
        if keys:
            redis_client.delete(*keys)

    @staticmethod
    def invalidate_post_cache(post_id: int):
        """특정 포스트 관련 캐시 삭제"""
        CacheManager.invalidate_pattern(f"*post:{post_id}*")
        CacheManager.invalidate_pattern("get_posts:*")

# API에서 사용
@router.put("/posts/{post_id}")
async def update_post(post_id: int, data: PostUpdate):
    # 포스트 업데이트
    post = db.query(Post).filter(Post.id == post_id).first()
    post.title = data.title

    # 캐시 무효화
    CacheManager.invalidate_post_cache(post_id)

    return post

12.6.1 지연 로딩 (Lazy Loading)​

<script setup lang="ts">
import { ref, onMounted } from 'vue'

const images = ref<HTMLImageElement[]>([])

onMounted(() => {
  // Intersection Observer로 뷰포트에 보일 때만 로드
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target as HTMLImageElement
        img.src = img.dataset.src!
        observer.unobserve(img)
      }
    })
  })

  images.value.forEach(img => observer.observe(img))
})
</script>

<template>

12.8 가상 스크롤링​

<script setup lang="ts">
import { ref, computed } from 'vue'

interface Item {
  id: number
  text: string
}

const items = ref<Item[]>(
  Array.from({ length: 10000 }, (_, i) => ({
    id: i,
    text: `Item ${i}`
  }))
)

const containerHeight = 600
const itemHeight = 50
const visibleCount = Math.ceil(containerHeight / itemHeight)
const scrollTop = ref(0)

const startIndex = computed(() =>
  Math.floor(scrollTop.value / itemHeight)
)

const endIndex = computed(() =>
  Math.min(startIndex.value + visibleCount + 1, items.value.length)
)

const visibleItems = computed(() =>
  items.value.slice(startIndex.value, endIndex.value)
)

const totalHeight = computed(() =>
  items.value.length * itemHeight
)

const offsetY = computed(() =>
  startIndex.value * itemHeight
)

const handleScroll = (e: Event) => {
  scrollTop.value = (e.target as HTMLElement).scrollTop
}
</script>

<template>
  <div
    class="virtual-scroll-container"
    :style="{ height: `${containerHeight}px` }"
    @scroll="handleScroll"
  >
    <div
      class="virtual-scroll-content"
      :style="{ height: `${totalHeight}px` }"
    >
      <div
        class="virtual-scroll-items"
        :style="{ transform: `translateY(${offsetY}px)` }"
      >
        <div
          v-for="item in visibleItems"
          :key="item.id"
          class="item"
          :style="{ height: `${itemHeight}px` }"
        >
          {{ item.text }}
        </div>
      </div>
    </div>
  </div>
</template>

12.11.1 Web Vitals 측정​

// utils/performance.ts
import { onCLS, onFID, onLCP } from 'web-vitals'

export const initPerformanceMonitoring = () => {
  // Cumulative Layout Shift
  onCLS(metric => {
    console.log('CLS:', metric.value)
    sendToAnalytics('CLS', metric.value)
  })

  // First Input Delay
  onFID(metric => {
    console.log('FID:', metric.value)
    sendToAnalytics('FID', metric.value)
  })

  // Largest Contentful Paint
  onLCP(metric => {
    console.log('LCP:', metric.value)
    sendToAnalytics('LCP', metric.value)
  })
}

function sendToAnalytics(metric: string, value: number) {
  // Google Analytics 또는 자체 분석 서버로 전송
  if (typeof gtag !== 'undefined') {
    gtag('event', metric, {
      value: Math.round(value),
      metric_id: metric,
      metric_value: value,
      metric_delta: value
    })
  }
}

12.11.2 Performance API​

// API 응답 시간 측정
const measureApiCall = async (url: string) => {
  const start = performance.now()

  const response = await fetch(url)
  const data = await response.json()

  const end = performance.now()
  const duration = end - start

  console.log(`API call to ${url} took ${duration}ms`)

  // 성능 엔트리 기록
  performance.mark(`api-${url}-end`)
  performance.measure(
    `api-${url}`,
    `api-${url}-start`,
    `api-${url}-end`
  )

  return data
}

<script setup lang="ts">
import { ref, computed, watch } from 'vue'
import { debounce } from 'lodash-es'

interface Item {
  id: number
  title: string
  image: string
}

// 10,000개 아이템
const allItems = ref<Item[]>(
  Array.from({ length: 10000 }, (_, i) => ({
    id: i,
    title: `Item ${i}`,
    image: `https://picsum.photos/200/200?random=${i}`
  }))
)

const searchQuery = ref('')
const scrollTop = ref(0)

// 검색 필터링
const filteredItems = computed(() => {
  if (!searchQuery.value) return allItems.value

  return allItems.value.filter(item =>
    item.title.toLowerCase().includes(searchQuery.value.toLowerCase())
  )
})

// 가상 스크롤링 설정
const itemHeight = 100
const containerHeight = 600
const visibleCount = Math.ceil(containerHeight / itemHeight)

const startIndex = computed(() =>
  Math.max(0, Math.floor(scrollTop.value / itemHeight) - 5)  // 버퍼 5개
)

const endIndex = computed(() =>
  Math.min(
    filteredItems.value.length,
    startIndex.value + visibleCount + 10  // 버퍼 10개
  )
)

const visibleItems = computed(() =>
  filteredItems.value.slice(startIndex.value, endIndex.value)
)

const totalHeight = computed(() =>
  filteredItems.value.length * itemHeight
)

const offsetY = computed(() =>
  startIndex.value * itemHeight
)

// 디바운스 검색
const debouncedSearch = debounce((value: string) => {
  searchQuery.value = value
}, 300)

const handleScroll = (e: Event) => {
  scrollTop.value = (e.target as HTMLElement).scrollTop
}

// 성능 측정
const measureRenderTime = () => {
  performance.mark('render-start')

  requestAnimationFrame(() => {
    performance.mark('render-end')
    performance.measure('render', 'render-start', 'render-end')

    const measure = performance.getEntriesByName('render')[0]
    console.log(`Render time: ${measure.duration.toFixed(2)}ms`)
  })
}

watch(visibleItems, measureRenderTime)
</script>

<template>

      type="text"
