3. API 설계

FastAPI를 사용한 RESTful API 설계의 핵심 원칙과 실전 패턴을 다룹니다.

3.1 RESTful API 원칙​

REST(Representational State Transfer)는 웹 API 설계를 위한 아키텍처 스타일입니다.

핵심 원칙​

  1. 클라이언트-서버 분리: 사용자 인터페이스와 데이터 저장소를 분리
  2. 무상태성(Stateless): 각 요청은 독립적이며 필요한 모든 정보를 포함
  3. 캐시 가능성: 응답은 캐시 가능 여부를 명시해야 함
  4. 계층화 시스템: 클라이언트는 중간 서버(프록시, 게이트웨이)의 존재를 알 수 없음
  5. 균일한 인터페이스: 표준화된 방식으로 리소스를 조작

리소스 중심 설계​

# 좋은 예 - 리소스 중심
GET    /api/v1/users           # 사용자 목록 조회
GET    /api/v1/users/123       # 특정 사용자 조회
POST   /api/v1/users           # 사용자 생성
PUT    /api/v1/users/123       # 사용자 전체 수정
DELETE /api/v1/users/123       # 사용자 삭제

# 나쁜 예 - 동작 중심
GET    /api/v1/getAllUsers
POST   /api/v1/createUser
POST   /api/v1/updateUser
POST   /api/v1/deleteUser
API 설계 베스트 프랙티스
  • 명사를 사용하고 동사는 피하세요 (리소스 이름)
  • 복수형을 일관되게 사용하세요 (/users, /posts)
  • 계층 구조를 명확히 표현하세요 (/users/123/posts)
  • 소문자와 하이픈을 사용하세요 (/user-profiles)

FastAPI 구현 예제​

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional

app = FastAPI()

class User(BaseModel):
    id: Optional[int] = None
    name: str
    email: str
    age: int

users_db = {}

# GET - 조회
@app.get("/api/v1/users")
async def get_users():
    """모든 사용자 조회"""
    return list(users_db.values())

@app.get("/api/v1/users/{user_id}")
async def get_user(user_id: int):
    """특정 사용자 조회"""
    if user_id not in users_db:
        raise HTTPException(status_code=404, detail="User not found")
    return users_db[user_id]

# POST - 생성
@app.post("/api/v1/users", status_code=201)
async def create_user(user: User):
    """새 사용자 생성"""
    user.id = len(users_db) + 1
    users_db[user.id] = user
    return user

# PUT - 전체 교체
@app.put("/api/v1/users/{user_id}")
async def update_user(user_id: int, user: User):
    """사용자 전체 정보 수정"""
    if user_id not in users_db:
        raise HTTPException(status_code=404, detail="User not found")
    user.id = user_id
    users_db[user_id] = user
    return user

# PATCH - 부분 수정
@app.patch("/api/v1/users/{user_id}")
async def partial_update_user(user_id: int, updates: dict):
    """사용자 부분 정보 수정"""
    if user_id not in users_db:
        raise HTTPException(status_code=404, detail="User not found")

    user = users_db[user_id]
    for key, value in updates.items():
        if hasattr(user, key):
            setattr(user, key, value)
    return user

# DELETE - 삭제
@app.delete("/api/v1/users/{user_id}", status_code=204)
async def delete_user(user_id: int):
    """사용자 삭제"""
    if user_id not in users_db:
        raise HTTPException(status_code=404, detail="User not found")
    del users_db[user_id]
    return None
PUT vs PATCH
  • PUT: 리소스 전체를 교체합니다. 모든 필드를 제공해야 합니다.
  • PATCH: 리소스의 일부만 수정합니다. 변경할 필드만 제공하면 됩니다.

3.5 Path Parameters vs Query Parameters​

Path Parameters​

URL 경로의 일부로 필수 리소스 식별자를 전달합니다.

from fastapi import Path

# 기본 경로 매개변수
@app.get("/users/{user_id}")
async def get_user(user_id: int):
    return {"user_id": user_id}

# 유효성 검증 추가
@app.get("/users/{user_id}")
async def get_user(
    user_id: int = Path(..., gt=0, le=1000, description="사용자 ID")
):
    """
    user_id는 1부터 1000 사이의 값이어야 함
    """
    return {"user_id": user_id}

# 여러 경로 매개변수
@app.get("/users/{user_id}/posts/{post_id}")
async def get_user_post(
    user_id: int = Path(..., gt=0),
    post_id: int = Path(..., gt=0)
):
    return {"user_id": user_id, "post_id": post_id}

# 문자열 경로 매개변수
@app.get("/categories/{category_name}")
async def get_category(
    category_name: str = Path(..., min_length=3, max_length=50)
):
    return {"category": category_name}

Query Parameters​

선택적 필터링, 정렬, 페이지네이션에 사용합니다.

from fastapi import Query
from typing import Optional, List

# 기본 쿼리 매개변수
@app.get("/users")
async def get_users(skip: int = 0, limit: int = 10):
    """
    GET /users?skip=0&limit=10
    """
    return users_db[skip:skip + limit]

# 선택적 쿼리 매개변수
@app.get("/users")
async def search_users(
    name: Optional[str] = None,
    email: Optional[str] = None,
    min_age: Optional[int] = None
):
    """
    GET /users?name=John&min_age=18
    """
    results = users_db
    if name:
        results = [u for u in results if name.lower() in u.name.lower()]
    if min_age:
        results = [u for u in results if u.age >= min_age]
    return results

# 유효성 검증 추가
@app.get("/items")

조합 사용​

@app.get("/users/{user_id}/posts")
async def get_user_posts(
    # Path parameter (필수)
    user_id: int = Path(..., gt=0),
    # Query parameters (선택적)
    published: Optional[bool] = None,
    skip: int = Query(0, ge=0),
    limit: int = Query(10, gt=0, le=100),
    sort: str = Query("created_at", regex="^(created_at|title)$")
):
    """
    GET /users/123/posts?published=true&skip=0&limit=10&sort=created_at
    """
    posts = get_posts_by_user(user_id)

    if published is not None:
        posts = [p for p in posts if p.published == published]

    if sort == "title":
        posts = sorted(posts, key=lambda p: p.title)

    return posts[skip:skip + limit]
언제 무엇을 사용할까?
  • Path Parameters: 리소스 식별 (/users/{user_id})
  • Query Parameters: 필터링, 정렬, 페이지네이션 (?page=1&sort=name)
  • Request Body: 복잡한 데이터 생성/수정

3.6 Request Body 검증 (Pydantic)​

Pydantic을 사용하여 요청 본문을 자동으로 검증합니다.

기본 스키마​

from pydantic import BaseModel, Field, EmailStr, validator
from typing import Optional, List
from datetime import datetime

class UserBase(BaseModel):
    email: EmailStr
    username: str = Field(..., min_length=3, max_length=50)
    full_name: Optional[str] = None

class UserCreate(UserBase):
    password: str = Field(..., min_length=8)

    @validator('password')
    def validate_password(cls, v):
        if not any(char.isdigit() for char in v):
            raise ValueError('Password must contain at least one digit')
        if not any(char.isupper() for char in v):
            raise ValueError('Password must contain at least one uppercase letter')
        return v

class UserUpdate(BaseModel):
    email: Optional[EmailStr] = None
    username: Optional[str] = Field(None, min_length=3, max_length=50)
    full_name: Optional[str] = None

class User(UserBase):
    id: int
    is_active: bool = True
    created_at: datetime

    class Config:
        orm_mode = True  # SQLAlchemy 모델과 호환

# 사용 예제

고급 검증​

from pydantic import BaseModel, Field, validator, root_validator
from typing import List, Optional
from datetime import datetime, date

class PostCreate(BaseModel):
    title: str = Field(..., min_length=5, max_length=200)
    content: str = Field(..., min_length=10)
    tags: List[str] = Field(default=[], max_items=5)
    published: bool = False
    publish_date: Optional[datetime] = None

    @validator('tags')
    def validate_tags(cls, v):
        # 태그 정규화
        return [tag.lower().strip() for tag in v]

    @root_validator
    def validate_publish_date(cls, values):
        # published가 True면 publish_date 필수
        if values.get('published') and not values.get('publish_date'):
            values['publish_date'] = datetime.now()
        return values

class UserProfile(BaseModel):
    username: str
    bio: Optional[str] = Field(None, max_length=500)
    website: Optional[str] = None
    birth_date: Optional[date] = None

    @validator('website')
    def validate_website(cls, v):
        if v and not v.startswith(('http://', 'https://')):
            raise ValueError('Website must start with http:// or https://')
        return v

    @validator('birth_date')
    def validate_age(cls, v):
        if v:
            today = date.today()
            age = today.year - v.year - ((today.month, today.day) < (v.month, v.day))
            if age < 13:
                raise ValueError('User must be at least 13 years old')
        return v

class Config(BaseModel):
    """중첩된 모델"""

    class DatabaseConfig(BaseModel):
        host: str
        port: int = Field(5432, ge=1, le=65535)
        database: str

    class RedisConfig(BaseModel):
        host: str
        port: int = Field(6379, ge=1, le=65535)

    database: DatabaseConfig
    redis: RedisConfig
    debug: bool = False

# 사용 예제
@app.post("/posts")
async def create_post(post: PostCreate):
    return post

@app.put("/profile")
async def update_profile(profile: UserProfile):
    return profile

from typing import Optional
from pydantic import BaseModel

class CursorPaginatedResponse(BaseModel, Generic[T]):
    items: List[T]
    next_cursor: Optional[str] = None
    has_more: bool

@app.get("/posts", response_model=CursorPaginatedResponse[Post])
async def get_posts(
    cursor: Optional[str] = None,
    limit: int = Query(20, ge=1, le=100)
):
    """
    커서 기반 페이지네이션 (무한 스크롤에 적합)
    GET /posts?cursor=eyJpZCI6MTAwfQ&limit=20
    """
    posts = get_posts_from_db(after_cursor=cursor, limit=limit + 1)

    has_more = len(posts) > limit
    if has_more:
        posts = posts[:limit]

    next_cursor = None
    if has_more and posts:
        # 마지막 항목의 ID를 커서로 인코딩
        next_cursor = encode_cursor(posts[-1].id)

    return CursorPaginatedResponse(
        items=posts,
        next_cursor=next_cursor,
        has_more=has_more
    )

def encode_cursor(post_id: int) -> str:
    import base64
    import json
    cursor_data = json.dumps({"id": post_id})
    return base64.b64encode(cursor_data.encode()).decode()

def decode_cursor(cursor: str) -> int:
    import base64
    import json
    cursor_data = json.loads(base64.b64decode(cursor))
    return cursor_data["id"]

    items = get_items_from_db(offset=offset, limit=page_size)

    # Link 헤더 생성
    base_url = "http://api.example.com/items"
    links = []

    if page > 1:
        # 이전 페이지
        prev_params = urlencode({"page": page - 1, "page_size": page_size})
        links.append(f'<{base_url}?{prev_params}>; rel="prev"')

        # 첫 페이지
        first_params = urlencode({"page": 1, "page_size": page_size})
        links.append(f'<{base_url}?{first_params}>; rel="first"')

    if page < total_pages:
        # 다음 페이지
        next_params = urlencode({"page": page + 1, "page_size": page_size})
        links.append(f'<{base_url}?{next_params}>; rel="next"')

        # 마지막 페이지
        last_params = urlencode({"page": total_pages, "page_size": page_size})
        links.append(f'<{base_url}?{last_params}>; rel="last"')

    if links:
        response.headers["Link"] = ", ".join(links)

    return {
        "items": items,
        "total": total,
        "page": page,
        "page_size": page_size
    }

from typing import List, TypeVar, Generic
from pydantic import BaseModel
from sqlalchemy.orm import Query

T = TypeVar('T')

class Page(BaseModel, Generic[T]):
    items: List[T]
    total: int
    page: int
    page_size: int

    @property
    def total_pages(self) -> int:
        return (self.total + self.page_size - 1) // self.page_size

    @property
    def has_previous(self) -> bool:
        return self.page > 1

    @property
    def has_next(self) -> bool:
        return self.page < self.total_pages

def paginate(query: Query, page: int, page_size: int) -> Page:
    """SQLAlchemy 쿼리 페이지네이션 헬퍼"""
    total = query.count()
    items = query.offset((page - 1) * page_size).limit(page_size).all()

    return Page(
        items=items,
        total=total,
        page=page,
        page_size=page_size
    )

    author_id: Optional[int] = None
    status: Optional[str] = None
    min_views: Optional[int] = None
    max_views: Optional[int] = None
    search: Optional[str] = None  # 제목/내용 검색

@app.get("/posts")

    filters: PostFilter = Depends(),
    page: int = Query(1, ge=1),
    page_size: int = Query(10, ge=1, le=100)
):
    query = db.query(Post)

    # 필터 자동 적용
    filter_dict = filters.dict(exclude_none=True)

    for field, value in filter_dict.items():
        if field == "search":
            # 검색
            query = query.filter(
                or_(
                    Post.title.contains(value),
                    Post.content.contains(value)
                )
            )
        elif field.startswith("min_"):
            # 최소값
            column = field[4:]
            query = query.filter(getattr(Post, column) >= value)
        elif field.startswith("max_"):
            # 최대값
            column = field[4:]
            query = query.filter(getattr(Post, column) <= value)
        else:
            # 일치
            query = query.filter(getattr(Post, field) == value)

    """동적으로 Origin 허용 여부 결정"""

문제 2: 페이지네이션과 필터링​

게시글 목록 API에 페이지네이션, 필터링, 정렬 기능을 추가하세요.

요구사항:

  • 페이지 번호와 크기로 페이지네이션
  • 제목으로 검색
  • published 상태로 필터링
  • 생성일, 조회수로 정렬
  • 총 개수, 총 페이지 수 포함
해답
from fastapi import FastAPI, Query
from pydantic import BaseModel
from typing import List, Optional
from enum import Enum

app = FastAPI()

class SortField(str, Enum):
    CREATED_AT = "created_at"
    VIEWS = "views"
    TITLE = "title"

class SortOrder(str, Enum):
    ASC = "asc"
    DESC = "desc"

class PaginatedPostsResponse(BaseModel):
    items: List[PostResponse]
    total: int
    page: int
    page_size: int
    total_pages: int

@app.get("/posts", response_model=PaginatedPostsResponse)
async def get_posts(
    # 페이지네이션
    page: int = Query(1, ge=1, description="페이지 번호"),
    page_size: int = Query(10, ge=1, le=100, description="페이지 크기"),

    # 필터링
    search: Optional[str] = Query(None, min_length=1, description="제목 검색"),
    published: Optional[bool] = Query(None, description="게시 상태 필터"),

    # 정렬
    sort_by: SortField = Query(SortField.CREATED_AT, description="정렬 필드"),
    order: SortOrder = Query(SortOrder.DESC, description="정렬 순서")
):
    """
    페이지네이션, 필터링, 정렬이 적용된 게시글 목록 조회

    예제:
    - GET /posts?page=1&page_size=10
    - GET /posts?search=fastapi&published=true
    - GET /posts?sort_by=views&order=desc
    """
    # 전체 게시글 가져오기
    posts = list(posts_db.values())

    # 필터링
    if search:
        posts = [
            p for p in posts
            if search.lower() in p["title"].lower()
        ]

    if published is not None:
        posts = [p for p in posts if p["published"] == published]

    # 정렬
    reverse = (order == SortOrder.DESC)
    posts.sort(key=lambda x: x[sort_by.value], reverse=reverse)

    # 총 개수
    total = len(posts)

    # 페이지네이션
    start = (page - 1) * page_size
    end = start + page_size
    paginated_posts = posts[start:end]

    # 총 페이지 수
    total_pages = (total + page_size - 1) // page_size

    return PaginatedPostsResponse(
        items=paginated_posts,
        total=total,
        page=page,
        page_size=page_size,
        total_pages=total_pages
    )

    @validator('username')
    def validate_username(cls, v):
        """영문자, 숫자, 언더스코어만 허용"""
        if not re.match(r'^[a-zA-Z0-9_]+$', v):
            raise ValueError(
                'Username must contain only letters, numbers, and underscores'
            )
        return v.lower()

    @validator('password')
    def validate_password(cls, v):
        """비밀번호 강도 검증"""
        if not re.search(r'[A-Z]', v):
            raise ValueError('Password must contain at least one uppercase letter')

        if not re.search(r'[a-z]', v):
            raise ValueError('Password must contain at least one lowercase letter')

        if not re.search(r'\d', v):
            raise ValueError('Password must contain at least one digit')

        if not re.search(r'[!@#$%^&*(),.?":{}|<>]', v):
            raise ValueError('Password must contain at least one special character')

        return v

    @validator('birth_date')
    def validate_age(cls, v):
        """13세 이상 검증"""
        today = date.today()
        age = today.year - v.year - ((today.month, today.day) < (v.month, v.day))

        if age < 13:
            raise ValueError('You must be at least 13 years old to register')

        if age > 120:
            raise ValueError('Invalid birth date')

        return v

    class Config:
        schema_extra = {
            "example": {
                "username": "john_doe",
                "email": "john@example.com",
                "password": "SecurePass123!",
                "full_name": "John Doe",
                "birth_date": "1990-01-15"
            }
        }

class UserResponse(BaseModel):
    id: int
    username: str
    email: str
    full_name: str
    birth_date: date
    created_at: datetime

users_db = {}
next_user_id = 1

@app.post("/register", response_model=UserResponse, status_code=status.HTTP_201_CREATED)
async def register_user(user: UserRegister):
    """
    새 사용자 등록

    검증 규칙:
    - username: 3-20자, 영문/숫자/언더스코어만
    - email: 유효한 이메일 형식
    - password: 8자 이상, 대문자/소문자/숫자/특수문자 포함
    - birth_date: 13세 이상
    """
    # 중복 확인
    for existing_user in users_db.values():
        if existing_user["username"] == user.username:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Username already exists"
            )
        if existing_user["email"] == user.email:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Email already registered"
            )

    global next_user_id

    # 사용자 생성 (비밀번호는 해싱해야 하지만 예제에서는 생략)
    user_dict = user.dict()
    del user_dict["password"]  # 응답에서 비밀번호 제외

    user_dict.update({
        "id": next_user_id,
        "created_at": datetime.now()
    })

    users_db[next_user_id] = user_dict
    next_user_id += 1

    return user_dict
