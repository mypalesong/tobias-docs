# scripts/docker.sh
#!/bin/bash

set -e

# 색상 정의
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 함수: 로그 출력
log() {
  echo -e "${GREEN}[$(date +'%Y-%m-%d %H:%M:%S')]${NC} $1"
}

error() {
  echo -e "${RED}[ERROR]${NC} $1" >&2
}

warn() {
  echo -e "${YELLOW}[WARN]${NC} $1"
}

# 환경 확인
check_env() {
  if [ ! -f .env.production ]; then
    error ".env.production 파일이 없습니다"
    exit 1
  fi

  source .env.production

  required_vars=("DB_PASSWORD" "REDIS_PASSWORD" "JWT_SECRET" "SESSION_SECRET")
  for var in "${required_vars[@]}"; do
    if [ -z "${!var}" ]; then
      error "환경 변수 $var가 설정되지 않았습니다"
      exit 1
    fi
  done
}

# 빌드
build() {
  log "Docker 이미지 빌드 시작..."
  docker-compose build --parallel
  log "빌드 완료"
}

# 시작
start() {
  log "서비스 시작..."
  docker-compose up -d
  log "서비스 시작 완료"
  docker-compose ps
}

# 중지
stop() {
  log "서비스 중지..."
  docker-compose down
  log "서비스 중지 완료"
}

# 재시작
restart() {
  stop
  start
}

# 로그 확인
logs() {
  docker-compose logs -f --tail=100 "$@"
}

# 정리
clean() {
  warn "모든 컨테이너, 볼륨, 이미지를 삭제합니다"
  read -p "계속하시겠습니까? (y/N) " -n 1 -r
  echo
  if [[ $REPLY =~ ^[Yy]$ ]]; then
    docker-compose down -v --rmi all
    log "정리 완료"
  fi
}

# 헬스체크
health() {
  log "서비스 상태 확인..."

  services=("app" "db" "redis" "nginx")
  for service in "${services[@]}"; do
    health_status=$(docker inspect --format='{{.State.Health.Status}}' "myapp-$service" 2>/dev/null || echo "N/A")
    if [ "$health_status" = "healthy" ]; then
      echo -e "$service: ${GREEN}✓ healthy${NC}"
    elif [ "$health_status" = "N/A" ]; then
      echo -e "$service: ${YELLOW}⚠ no healthcheck${NC}"
    else
      echo -e "$service: ${RED}✗ $health_status${NC}"
    fi
  done
}

# 백업
backup() {
  log "데이터베이스 백업 시작..."

  timestamp=$(date +%Y%m%d_%H%M%S)
  backup_file="backups/db_backup_$timestamp.sql"

  mkdir -p backups

  docker-compose exec -T db pg_dump -U postgres myapp > "$backup_file"

  log "백업 완료: $backup_file"
}

# 복원
restore() {
  if [ -z "$1" ]; then
    error "백업 파일을 지정해주세요"
    exit 1
  fi

  warn "데이터베이스를 복원하면 기존 데이터가 삭제됩니다"
  read -p "계속하시겠습니까? (y/N) " -n 1 -r
  echo

  if [[ $REPLY =~ ^[Yy]$ ]]; then
    log "데이터베이스 복원 시작..."
    docker-compose exec -T db psql -U postgres myapp < "$1"
    log "복원 완료"
  fi
}

# 메인
case "$1" in
  build)
    build
    ;;
  start)
    check_env
    start
    ;;
  stop)
    stop
    ;;
  restart)
    check_env
    restart
    ;;
  logs)
    logs "${@:2}"
    ;;
  clean)
    clean
    ;;
  health)
    health
    ;;
  backup)
    backup
    ;;
  restore)
    restore "$2"
    ;;
  *)
    echo "사용법: $0 {build|start|stop|restart|logs|clean|health|backup|restore}"
    exit 1
    ;;
esac

import { z } from 'zod';

const envSchema = z.object({

  REDIS_URL: z.string().url(),

  // 인증

  JWT_EXPIRES_IN: z.string().default('7d'),
  SESSION_SECRET: z.string().min(32),
  SESSION_MAX_AGE: z.string().transform(Number).pipe(z.number().int().positive()),

  // 이메일
  SMTP_HOST: z.string().optional(),
  SMTP_PORT: z.string().transform(Number).pipe(z.number().int()).optional(),
  SMTP_USER: z.string().email().optional(),
  SMTP_PASSWORD: z.string().optional(),
  EMAIL_FROM: z.string().email(),

  // AWS
  AWS_REGION: z.string().optional(),
  AWS_ACCESS_KEY_ID: z.string().optional(),
  AWS_SECRET_ACCESS_KEY: z.string().optional(),
  S3_BUCKET: z.string().optional(),

  // 모니터링
  SENTRY_DSN: z.string().url().optional(),
  SENTRY_ENVIRONMENT: z.string().optional(),
  LOG_LEVEL: z.enum(['debug', 'info', 'warn', 'error']).default('info'),

  // 기능 플래그
  FEATURE_ANALYTICS: z.string().transform(v => v === 'true').default('false'),
  FEATURE_CACHE: z.string().transform(v => v === 'true').default('true'),
  FEATURE_RATE_LIMIT: z.string().transform(v => v === 'true').default('true'),
});

export type Env = z.infer<typeof envSchema>;

export function validateEnv(): Env {
  try {
    return envSchema.parse(process.env);
  } catch (error) {
    if (error instanceof z.ZodError) {
      const missingVars = error.errors.map(e => e.path.join('.')).join(', ');
      throw new Error(`환경 변수 검증 실패: ${missingVars}`);
    }
    throw error;
  }
}

// config/secrets.ts
import * as crypto from 'crypto';
import * as fs from 'fs';

export class SecretsManager {
  private readonly algorithm = 'aes-256-gcm';
  private readonly keyPath: string;

  constructor(keyPath: string = '.secrets.key') {
    this.keyPath = keyPath;
  }

  // 마스터 키 생성
  generateMasterKey(): Buffer {
    const key = crypto.randomBytes(32);
    fs.writeFileSync(this.keyPath, key.toString('hex'), { mode: 0o600 });
    return key;
  }

  // 마스터 키 로드
  private loadMasterKey(): Buffer {
    if (!fs.existsSync(this.keyPath)) {
      throw new Error('마스터 키가 없습니다. generateMasterKey()를 먼저 실행하세요.');
    }
    const keyHex = fs.readFileSync(this.keyPath, 'utf8');
    return Buffer.from(keyHex, 'hex');
  }

  // 암호화
  encrypt(plaintext: string): string {
    const key = this.loadMasterKey();
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.algorithm, key, iv);

    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');

    const authTag = cipher.getAuthTag();

    return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
  }

  // 복호화
  decrypt(ciphertext: string): string {
    const key = this.loadMasterKey();
    const [ivHex, authTagHex, encrypted] = ciphertext.split(':');

    const iv = Buffer.from(ivHex, 'hex');
    const authTag = Buffer.from(authTagHex, 'hex');

    const decipher = crypto.createDecipheriv(this.algorithm, key, iv);
    decipher.setAuthTag(authTag);

    let decrypted = decipher.update(encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');

    return decrypted;
  }

  // .env 파일 암호화
  encryptEnvFile(inputPath: string, outputPath: string): void {
    const content = fs.readFileSync(inputPath, 'utf8');
    const encrypted = this.encrypt(content);
    fs.writeFileSync(outputPath, encrypted, { mode: 0o600 });
  }

  // .env 파일 복호화
  decryptEnvFile(inputPath: string, outputPath: string): void {
    const encrypted = fs.readFileSync(inputPath, 'utf8');
    const decrypted = this.decrypt(encrypted);
    fs.writeFileSync(outputPath, decrypted, { mode: 0o600 });
  }
}

    return `${this.config.taskFamily}:${revision}`;

  // 서비스 업데이트
  async updateService(taskDefinition: string): Promise<void> {
    const command = new UpdateServiceCommand({
      cluster: this.config.cluster,
      service: this.config.service,
      taskDefinition,
      forceNewDeployment: true,
      deploymentConfiguration: {
        minimumHealthyPercent: 100,
        maximumPercent: 200,
        deploymentCircuitBreaker: {
          enable: true,
          rollback: true,
        },
      },
    });

    await this.client.send(command);
    console.log(`서비스 업데이트 시작: ${this.config.service}`);
  }

  // 배포 상태 확인
  async waitForDeployment(timeoutMinutes: number = 10): Promise<void> {
    const startTime = Date.now();
    const timeout = timeoutMinutes * 60 * 1000;

    while (Date.now() - startTime < timeout) {
      const command = new DescribeServicesCommand({
        cluster: this.config.cluster,
        services: [this.config.service],
      });

      const response = await this.client.send(command);
      const service = response.services?.[0];

      if (!service) {
        throw new Error('서비스를 찾을 수 없습니다');
      }

      const runningCount = service.runningCount || 0;
      const desiredCount = service.desiredCount || 0;
      const deployments = service.deployments || [];

      console.log(`Running: ${runningCount}/${desiredCount}, Deployments: ${deployments.length}`);

      // 배포 완료 조건: 실행 중인 태스크 수가 원하는 수와 같고, 배포가 1개만 있음
      if (runningCount === desiredCount && deployments.length === 1) {
        console.log('배포 완료!');
        return;
      }

      // 배포 실패 확인
      const failedDeployment = deployments.find(d => d.rolloutState === 'FAILED');
      if (failedDeployment) {
        throw new Error(`배포 실패: ${failedDeployment.rolloutStateReason}`);
      }

      await new Promise(resolve => setTimeout(resolve, 15000)); // 15초 대기
    }

    throw new Error('배포 타임아웃');
  }

  // 전체 배포 프로세스
  async deploy(): Promise<void> {
    console.log('=== ECS 배포 시작 ===');

    try {
      // 1. 새 태스크 정의 등록
      const taskDefinition = await this.registerTaskDefinition();

      // 2. 서비스 업데이트
      await this.updateService(taskDefinition);

      // 3. 배포 완료 대기
      await this.waitForDeployment();

      console.log('=== 배포 성공 ===');
    } catch (error) {
      console.error('=== 배포 실패 ===');
      throw error;
    }
  }
}

# scripts/blue-green-deploy.sh
#!/bin/bash

set -e

# 색상 정의
GREEN='\033[0;32m'
BLUE='\033[0;34m'
RED='\033[0;31m'
NC='\033[0m'

# 설정
BLUE_PORT=3000
GREEN_PORT=3001
NGINX_CONFIG="/etc/nginx/sites-available/myapp"
HEALTH_CHECK_URL="http://localhost"
MAX_RETRIES=30
RETRY_INTERVAL=2

# 현재 활성 환경 확인
get_active_env() {
  if grep -q "proxy_pass http://localhost:$BLUE_PORT" $NGINX_CONFIG; then
    echo "blue"
  else
    echo "green"
  fi
}

# 대기 환경 포트 가져오기
get_standby_port() {
  local active=$(get_active_env)
  if [ "$active" = "blue" ]; then
    echo $GREEN_PORT
  else
    echo $BLUE_PORT
  fi
}

# 헬스체크
health_check() {
  local port=$1
  local retries=0

  echo "헬스체크 시작 (포트: $port)..."

  while [ $retries -lt $MAX_RETRIES ]; do
    if curl -sf "$HEALTH_CHECK_URL:$port/health" > /dev/null; then
      echo -e "${GREEN}✓ 헬스체크 성공${NC}"
      return 0
    fi

    retries=$((retries + 1))
    echo "헬스체크 대기 중... ($retries/$MAX_RETRIES)"
    sleep $RETRY_INTERVAL
  done

  echo -e "${RED}✗ 헬스체크 실패${NC}"
  return 1
}

# 스모크 테스트
smoke_test() {
  local port=$1

  echo "스모크 테스트 시작..."

  # API 테스트
  if ! curl -sf "$HEALTH_CHECK_URL:$port/api/version" > /dev/null; then
    echo -e "${RED}✗ API 테스트 실패${NC}"
    return 1
  fi

  # 데이터베이스 연결 테스트
  if ! curl -sf "$HEALTH_CHECK_URL:$port/health/db" > /dev/null; then
    echo -e "${RED}✗ 데이터베이스 연결 실패${NC}"
    return 1
  fi

  echo -e "${GREEN}✓ 스모크 테스트 성공${NC}"
  return 0
}

# Nginx 설정 전환
switch_nginx() {
  local new_port=$1
  local active=$(get_active_env)
  local new_env

  if [ "$active" = "blue" ]; then
    new_env="green"
  else
    new_env="blue"
  fi

  echo "Nginx 설정 전환 중 (${active} -> ${new_env})..."

  # 설정 백업
  cp $NGINX_CONFIG "${NGINX_CONFIG}.backup"

  # 포트 변경
  sed -i "s/proxy_pass http:\/\/localhost:[0-9]\+/proxy_pass http:\/\/localhost:$new_port/" $NGINX_CONFIG

  # 설정 테스트
  if ! nginx -t; then
    echo -e "${RED}✗ Nginx 설정 검증 실패${NC}"
    cp "${NGINX_CONFIG}.backup" $NGINX_CONFIG
    return 1
  fi

  # Nginx 리로드
  nginx -s reload

  echo -e "${GREEN}✓ Nginx 전환 완료${NC}"
  return 0
}

# 이전 환경 종료
stop_old_env() {
  local port=$1
  local env

  if [ $port -eq $BLUE_PORT ]; then
    env="blue"
  else
    env="green"
  fi

  echo "이전 환경 종료 중 ($env)..."

  docker-compose -f docker-compose.yml -f docker-compose.$env.yml down

  echo -e "${GREEN}✓ 이전 환경 종료 완료${NC}"
}

# 메인 배포 프로세스
deploy() {
  local active_env=$(get_active_env)
  local standby_port=$(get_standby_port)
  local standby_env

  if [ "$active_env" = "blue" ]; then
    standby_env="green"
  else
    standby_env="blue"
  fi

  echo "=== Blue-Green 배포 시작 ==="
  echo "활성 환경: $active_env"
  echo "대기 환경: $standby_env (포트: $standby_port)"
  echo ""

  # 1. 대기 환경 배포
  echo -e "${BLUE}[1/5] 대기 환경 배포 중...${NC}"
  docker-compose -f docker-compose.yml -f docker-compose.$standby_env.yml up -d

  # 2. 헬스체크
  echo -e "${BLUE}[2/5] 헬스체크 중...${NC}"
  if ! health_check $standby_port; then
    echo -e "${RED}배포 실패: 헬스체크 실패${NC}"
    docker-compose -f docker-compose.yml -f docker-compose.$standby_env.yml down
    exit 1
  fi

  # 3. 스모크 테스트
  echo -e "${BLUE}[3/5] 스모크 테스트 중...${NC}"
  if ! smoke_test $standby_port; then
    echo -e "${RED}배포 실패: 스모크 테스트 실패${NC}"
    docker-compose -f docker-compose.yml -f docker-compose.$standby_env.yml down
    exit 1
  fi

  # 4. 트래픽 전환
  echo -e "${BLUE}[4/5] 트래픽 전환 중...${NC}"
  if ! switch_nginx $standby_port; then
    echo -e "${RED}배포 실패: Nginx 전환 실패${NC}"
    exit 1
  fi

  # 안정화 대기
  echo "안정화 대기 중 (30초)..."
  sleep 30

  # 5. 이전 환경 종료
  echo -e "${BLUE}[5/5] 이전 환경 종료 중...${NC}"
  if [ "$active_env" = "blue" ]; then
    stop_old_env $BLUE_PORT
  else
    stop_old_env $GREEN_PORT
  fi

  echo ""
  echo -e "${GREEN}=== 배포 완료 ===${NC}"
  echo "새 활성 환경: $standby_env"
}

  fi

  echo -e "${GREEN}=== 롤백 완료 ===${NC}"
}

// scripts/auto-rollback.ts
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

interface RollbackConfig {
  healthCheckUrl: string;
  healthCheckInterval: number;  // 초
  healthCheckRetries: number;
  errorRateThreshold: number;  // %
  latencyThreshold: number;  // ms
  monitoringDuration: number;  // 초
}

export class AutoRollback {
  private config: RollbackConfig;
  private deploymentVersion: string;
  private previousVersion: string;

  constructor(config: RollbackConfig) {
    this.config = config;
    this.deploymentVersion = '';
    this.previousVersion = '';
  }

  // 현재 버전 저장
  async saveCurrentVersion(): Promise<void> {
    const { stdout } = await execAsync('git rev-parse HEAD');
    this.previousVersion = stdout.trim();
    console.log(`이전 버전 저장: ${this.previousVersion}`);
  }

  // 새 버전 배포
  async deploy(version: string): Promise<void> {
    this.deploymentVersion = version;
    console.log(`배포 시작: ${version}`);

    await execAsync(`docker pull myapp:${version}`);
    await execAsync('docker-compose up -d');
  }

  // 헬스체크
  async healthCheck(): Promise<boolean> {
    let retries = 0;

    while (retries < this.config.healthCheckRetries) {
      try {
        const response = await fetch(this.config.healthCheckUrl);
        if (response.ok) {
          return true;
        }
      } catch (error) {
        console.log(`헬스체크 실패 (${retries + 1}/${this.config.healthCheckRetries})`);
      }

      retries++;
      await new Promise(resolve =>
        setTimeout(resolve, this.config.healthCheckInterval * 1000)
      );
    }

    return false;
  }

  // 메트릭 모니터링
  async monitorMetrics(): Promise<{
    errorRate: number;
    avgLatency: number;
    healthy: boolean;
  }> {
    console.log('메트릭 모니터링 시작...');

    let totalRequests = 0;
    let errorCount = 0;
    let latencies: number[] = [];

    const startTime = Date.now();
    const endTime = startTime + (this.config.monitoringDuration * 1000);

    while (Date.now() < endTime) {
      try {
        const start = Date.now();
        const response = await fetch(this.config.healthCheckUrl);
        const latency = Date.now() - start;

        totalRequests++;
        latencies.push(latency);

        if (!response.ok) {
          errorCount++;
        }
      } catch (error) {
        errorCount++;
      }

      await new Promise(resolve => setTimeout(resolve, 1000));
    }

    const errorRate = (errorCount / totalRequests) * 100;
    const avgLatency = latencies.reduce((a, b) => a + b, 0) / latencies.length;

    const healthy =
      errorRate <= this.config.errorRateThreshold &&
      avgLatency <= this.config.latencyThreshold;

    return { errorRate, avgLatency, healthy };
  }

  // 롤백 실행
  async rollback(): Promise<void> {
    console.log('=== 자동 롤백 시작 ===');
    console.log(`롤백 대상: ${this.deploymentVersion} -> ${this.previousVersion}`);

    try {
      // 이전 버전으로 복원
      await execAsync(`docker pull myapp:${this.previousVersion}`);
      await execAsync('docker-compose up -d');

      // 헬스체크
      const healthy = await this.healthCheck();

      if (!healthy) {
        throw new Error('롤백 후 헬스체크 실패');
      }

      console.log('=== 롤백 완료 ===');
    } catch (error) {
      console.error('=== 롤백 실패 ===');
      throw error;
    }
  }

  // 배포 with 자동 롤백
  async deployWithAutoRollback(version: string): Promise<void> {
    try {
      // 1. 현재 버전 저장
      await this.saveCurrentVersion();

      // 2. 새 버전 배포
      await this.deploy(version);

      // 3. 헬스체크
      console.log('헬스체크 시작...');
      const healthy = await this.healthCheck();

      if (!healthy) {
        throw new Error('헬스체크 실패');
      }

      // 4. 메트릭 모니터링
      const metrics = await this.monitorMetrics();
      console.log('메트릭:', metrics);

      if (!metrics.healthy) {
        throw new Error(
          `메트릭 임계값 초과 - 에러율: ${metrics.errorRate.toFixed(2)}%, ` +
          `평균 지연시간: ${metrics.avgLatency.toFixed(0)}ms`
        );
      }

      console.log('=== 배포 성공 ===');
    } catch (error) {
      console.error('배포 실패:', error);
      console.log('자동 롤백 시작...');

      await this.rollback();
    }
  }
}

        }
      } else {
        // 프로덕션: 직접 롤백 스크립트 실행
        await this.executeRollbackScript(steps);
      }
