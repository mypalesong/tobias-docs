8. 파일 업로드

파일 업로드 기능을 안전하고 효율적으로 구현하는 방법을 배워봅시다.

8.1 파일 업로드 개요​

파일 업로드는 사용자가 이미지, 문서, 동영상 등을 서버에 전송하는 기능입니다.

**주요 고려사항:**
- 파일 크기 제한
- 파일 타입 검증
- 보안 (악성 파일 차단)
- 저장 위치 (로컬, S3, CDN)
- 업로드 진행률 표시
- 에러 처리

8.2 Backend: FastAPI 파일 업로드​

8.2.1 기본 파일 업로드​

# app/api/v1/upload.py
from fastapi import APIRouter, UploadFile, File, HTTPException
from pathlib import Path
import shutil
import uuid

router = APIRouter()

UPLOAD_DIR = Path("uploads")
UPLOAD_DIR.mkdir(exist_ok=True)

@router.post("/upload")
async def upload_file(file: UploadFile = File(...)):
    # 파일 크기 검증 (10MB)
    contents = await file.read()
    if len(contents) > 10 * 1024 * 1024:
        raise HTTPException(400, "파일 크기는 10MB 이하여야 합니다")

    # 파일 타입 검증
    allowed_types = ["image/jpeg", "image/png", "image/gif"]
    if file.content_type not in allowed_types:
        raise HTTPException(400, "JPG, PNG, GIF 파일만 업로드 가능합니다")

    # 고유 파일명 생성
    file_ext = Path(file.filename).suffix
    unique_filename = f"{uuid.uuid4()}{file_ext}"
    file_path = UPLOAD_DIR / unique_filename

    # 파일 저장
    with open(file_path, "wb") as f:
        f.write(contents)

    return {
        "filename": unique_filename,
        "content_type": file.content_type,
        "size": len(contents)
    }

8.2.2 다중 파일 업로드​

from typing import List

@router.post("/upload/multiple")
async def upload_multiple_files(files: List[UploadFile] = File(...)):
    if len(files) > 10:
        raise HTTPException(400, "최대 10개 파일까지 업로드 가능합니다")

    uploaded_files = []

    for file in files:
        contents = await file.read()

        # 검증
        if len(contents) > 10 * 1024 * 1024:
            raise HTTPException(400, f"{file.filename}: 파일 크기 초과")

        # 저장
        file_ext = Path(file.filename).suffix
        unique_filename = f"{uuid.uuid4()}{file_ext}"
        file_path = UPLOAD_DIR / unique_filename

        with open(file_path, "wb") as f:
            f.write(contents)

        uploaded_files.append({
            "original_name": file.filename,
            "saved_name": unique_filename,
            "size": len(contents)
        })

    return {"files": uploaded_files}

# app/services/s3.py
import boto3
from botocore.exceptions import ClientError
from app.config import settings

s3_client = boto3.client(
    's3',
    aws_access_key_id=settings.AWS_ACCESS_KEY_ID,
    aws_secret_access_key=settings.AWS_SECRET_ACCESS_KEY,
    region_name=settings.AWS_REGION
)

async def upload_to_s3(file: UploadFile, folder: str = "uploads") -> str:
    """S3에 파일 업로드"""
    try:
        # 고유 파일명 생성
        file_ext = Path(file.filename).suffix
        unique_filename = f"{folder}/{uuid.uuid4()}{file_ext}"

        # S3 업로드
        contents = await file.read()
        s3_client.put_object(
            Bucket=settings.S3_BUCKET,
            Key=unique_filename,
            Body=contents,
            ContentType=file.content_type
        )

        # URL 생성
        url = f"https://{settings.S3_BUCKET}.s3.{settings.AWS_REGION}.amazonaws.com/{unique_filename}"
        return url

    except ClientError as e:
        raise HTTPException(500, f"S3 업로드 실패: {str(e)}")

# API 엔드포인트
@router.post("/upload/s3")
async def upload_file_to_s3(file: UploadFile = File(...)):
    url = await upload_to_s3(file)
    return {"url": url}

8.4 이미지 처리​

8.4.1 썸네일 생성​

# app/services/image.py
from PIL import Image
import io

def create_thumbnail(image_data: bytes, size: tuple = (300, 300)) -> bytes:
    """썸네일 이미지 생성"""
    image = Image.open(io.BytesIO(image_data))

    # EXIF 방향 정보 처리
    try:
        from PIL import ImageOps
        image = ImageOps.exif_transpose(image)
    except:
        pass

    # 썸네일 생성
    image.thumbnail(size, Image.Resampling.LANCZOS)

    # RGBA to RGB 변환 (JPEG 저장용)
    if image.mode == 'RGBA':
        background = Image.new('RGB', image.size, (255, 255, 255))
        background.paste(image, mask=image.split()[3])
        image = background

    # 바이트로 변환
    output = io.BytesIO()
    image.save(output, format='JPEG', quality=85)
    return output.getvalue()

@router.post("/upload/image")
async def upload_image_with_thumbnail(file: UploadFile = File(...)):

import { ref } from 'vue'
import axios from 'axios'

const file = ref<File | null>(null)
const uploading = ref(false)
const uploadProgress = ref(0)
const uploadedUrl = ref('')
const error = ref('')

const handleFileSelect = (event: Event) => {
  const target = event.target as HTMLInputElement
  file.value = target.files?.[0] || null
  error.value = ''
}

const uploadFile = async () => {
  if (!file.value) return

  const formData = new FormData()
  formData.append('file', file.value)

  uploading.value = true
  error.value = ''

  try {
    const { data } = await axios.post('/api/upload', formData, {
      headers: {
        'Content-Type': 'multipart/form-data'
      },
      onUploadProgress: (progressEvent) => {
        if (progressEvent.total) {
          uploadProgress.value = Math.round(
            (progressEvent.loaded * 100) / progressEvent.total
          )
        }
      }
    })

    uploadedUrl.value = data.url
    console.log('Upload success:', data)
  } catch (err: any) {
    error.value = err.response?.data?.detail || '업로드 실패'
  } finally {
    uploading.value = false
  }
}
</script>

8.6 청크 업로드 (대용량 파일)​

8.6.1 Backend: 청크 처리​

# app/api/v1/upload.py
from fastapi import APIRouter, File, Form
import os

CHUNK_DIR = Path("chunks")
CHUNK_DIR.mkdir(exist_ok=True)

@router.post("/upload/chunk")
async def upload_chunk(
    chunk: UploadFile = File(...),
    filename: str = Form(...),
    chunk_index: int = Form(...),
    total_chunks: int = Form(...)
):
    # 청크 저장
    chunk_filename = f"{filename}.part{chunk_index}"
    chunk_path = CHUNK_DIR / chunk_filename

    contents = await chunk.read()
    with open(chunk_path, "wb") as f:
        f.write(contents)

    # 모든 청크가 업로드되었는지 확인
    if chunk_index == total_chunks - 1:
        # 청크 병합
        final_path = UPLOAD_DIR / filename
        with open(final_path, "wb") as final_file:
            for i in range(total_chunks):
                chunk_path = CHUNK_DIR / f"{filename}.part{i}"
                with open(chunk_path, "rb") as chunk_file:
                    final_file.write(chunk_file.read())
                os.remove(chunk_path)  # 청크 삭제

        return {"message": "Upload complete", "filename": filename}

    return {"message": f"Chunk {chunk_index + 1}/{total_chunks} uploaded"}

8.6.2 Frontend: 청크 업로드​

// composables/useChunkUpload.ts
const CHUNK_SIZE = 1024 * 1024 * 5 // 5MB

export const useChunkUpload = () => {
  const uploadFileInChunks = async (file: File) => {
    const totalChunks = Math.ceil(file.size / CHUNK_SIZE)

    for (let i = 0; i < totalChunks; i++) {
      const start = i * CHUNK_SIZE
      const end = Math.min(start + CHUNK_SIZE, file.size)
      const chunk = file.slice(start, end)

      const formData = new FormData()
      formData.append('chunk', chunk)
      formData.append('filename', file.name)
      formData.append('chunk_index', i.toString())
      formData.append('total_chunks', totalChunks.toString())

      await axios.post('/api/upload/chunk', formData)

      // 진행률 업데이트
      const progress = Math.round(((i + 1) / totalChunks) * 100)
      console.log(`Progress: ${progress}%`)
    }
  }

  return { uploadFileInChunks }
}

8.7 보안 고려사항​

# app/utils/security.py
import magic
from pathlib import Path

def validate_file_type(file_path: Path, allowed_types: list) -> bool:
    """실제 파일 타입 검증 (MIME type)"""
    mime = magic.Magic(mime=True)
    file_type = mime.from_file(str(file_path))
    return file_type in allowed_types

def sanitize_filename(filename: str) -> str:
    """파일명 살균"""
    # 경로 구분자 제거
    filename = filename.replace('/', '').replace('\\', '')
    # 특수 문자 제거
    import re
    filename = re.sub(r'[^\w\s.-]', '', filename)
    return filename

@router.post("/upload/secure")
async def secure_upload(file: UploadFile = File(...)):
    # 파일명 살균
    safe_filename = sanitize_filename(file.filename)

    # 임시 저장
    temp_path = UPLOAD_DIR / f"temp_{uuid.uuid4()}"
    contents = await file.read()
    with open(temp_path, "wb") as f:
        f.write(contents)

    # 실제 파일 타입 검증
    allowed_types = ["image/jpeg", "image/png", "image/gif"]
    if not validate_file_type(temp_path, allowed_types):
        temp_path.unlink()
        raise HTTPException(400, "허용되지 않은 파일 타입입니다")

    # 최종 저장
    final_path = UPLOAD_DIR / safe_filename
    temp_path.rename(final_path)

    return {"filename": safe_filename}

          progress.value = Math.round((e.loaded * 100) / e.total)
        }
      }
    })

    alert('업로드 성공!')
    console.log(data)
  } catch (error) {
    alert('업로드 실패')
  } finally {
    uploading.value = false
  }
}
</script>

    </div>

      @change="handleFileSelect"
      :disabled="uploading"
    />

    <div v-if="uploading" class="progress">
