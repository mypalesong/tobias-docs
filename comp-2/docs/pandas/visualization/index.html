6. 시각화

pandas는 matplotlib를 기반으로 한 강력한 시각화 기능을 제공합니다. 데이터 분석 결과를 효과적으로 전달하기 위한 다양한 차트와 그래프를 쉽게 생성할 수 있습니다.

6.1 pandas 기본 plot​

pandas는 .plot() 메서드를 통해 간단하게 시각화를 수행할 수 있습니다.

6.1.1 라인 차트 (Line Plot)​

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# 시계열 데이터 생성
dates = pd.date_range('2024-01-01', periods=100)
df = pd.DataFrame({
    '매출': np.cumsum(np.random.randn(100)) + 100,
    '비용': np.cumsum(np.random.randn(100)) + 80,
    '이익': np.cumsum(np.random.randn(100)) + 20
}, index=dates)

# 기본 라인 차트
df.plot()
plt.title('월별 재무 데이터')
plt.ylabel('금액 (만원)')
plt.show()

# 특정 컬럼만 선택
df['매출'].plot(figsize=(10, 5))
plt.title('매출 추이')
plt.show()

# 여러 컬럼을 서브플롯으로
df.plot(subplots=True, figsize=(10, 8), layout=(3, 1))
plt.tight_layout()
plt.show()
팁

.plot() 메서드는 기본적으로 라인 차트를 생성합니다. kind 파라미터로 다양한 차트 타입을 지정할 수 있습니다.

6.1.2 막대 차트 (Bar Plot)​

# 카테고리별 데이터
data = pd.DataFrame({
    '제품': ['A', 'B', 'C', 'D', 'E'],
    '판매량': [120, 95, 150, 80, 110],
    '재고': [30, 45, 20, 60, 35]
})

# 수직 막대 차트
data.plot(x='제품', y='판매량', kind='bar', figsize=(8, 5))
plt.title('제품별 판매량')
plt.ylabel('판매량')
plt.xticks(rotation=0)
plt.show()

# 수평 막대 차트
data.plot(x='제품', y='판매량', kind='barh', figsize=(8, 5))
plt.title('제품별 판매량')
plt.xlabel('판매량')
plt.show()

# 그룹화된 막대 차트
data.set_index('제품')[['판매량', '재고']].plot(kind='bar', figsize=(10, 5))
plt.title('제품별 판매량 및 재고')
plt.ylabel('수량')
plt.xticks(rotation=45)
plt.legend(['판매량', '재고'])
plt.show()

# 누적 막대 차트
data.set_index('제품')[['판매량', '재고']].plot(kind='bar', stacked=True, figsize=(10, 5))
plt.title('제품별 판매량 및 재고 (누적)')
plt.ylabel('수량')
plt.show()

6.1.3 산점도 (Scatter Plot)​

# 상관관계 분석용 데이터
np.random.seed(42)
df_scatter = pd.DataFrame({
    '광고비': np.random.uniform(10, 100, 50),
    '매출': np.random.uniform(50, 500, 50),
    '지역': np.random.choice(['서울', '부산', '대구'], 50)
})

df_scatter['매출'] = df_scatter['광고비'] * 3.5 + np.random.normal(0, 30, 50)

# 기본 산점도
df_scatter.plot(kind='scatter', x='광고비', y='매출', figsize=(8, 6))
plt.title('광고비와 매출의 상관관계')
plt.show()

# 크기와 색상 추가
df_scatter.plot(kind='scatter', x='광고비', y='매출',
                s=df_scatter['광고비']*5,  # 점 크기
                c='매출',  # 색상
                cmap='viridis',  # 컬러맵
                figsize=(10, 6),
                alpha=0.6)
plt.title('광고비와 매출의 상관관계 (크기: 광고비, 색상: 매출)')
plt.colorbar(label='매출')
plt.show()

# 정규분포 데이터 생성

6.2 matplotlib 통합​

pandas는 matplotlib와 완벽하게 통합되어 더 세밀한 커스터마이징이 가능합니다.

import matplotlib.pyplot as plt

# Figure와 Axes 객체 활용
fig, axes = plt.subplots(2, 2, figsize=(12, 10))

# 데이터 준비
dates = pd.date_range('2024-01-01', periods=50)
df_sales = pd.DataFrame({
    '온라인': np.cumsum(np.random.randn(50)) + 100,
    '오프라인': np.cumsum(np.random.randn(50)) + 80
}, index=dates)

# 1. 라인 차트
df_sales.plot(ax=axes[0, 0], title='판매 채널별 매출 추이')
axes[0, 0].set_ylabel('매출 (만원)')
axes[0, 0].grid(True, alpha=0.3)

# 2. 영역 차트
df_sales.plot(kind='area', ax=axes[0, 1], alpha=0.4, title='누적 판매 추이')
axes[0, 1].set_ylabel('매출 (만원)')

# 3. 막대 차트
monthly_avg = df_sales.resample('W').mean()
monthly_avg.plot(kind='bar', ax=axes[1, 0], title='주간 평균 매출')
axes[1, 0].set_ylabel('평균 매출 (만원)')
axes[1, 0].tick_params(axis='x', rotation=45)

# 4. 박스 플롯
df_sales.plot(kind='box', ax=axes[1, 1], title='판매 채널별 분포')
axes[1, 1].set_ylabel('매출 (만원)')

plt.tight_layout()
plt.show()

6.3 seaborn 통합​

seaborn은 통계적 시각화에 특화된 라이브러리로, pandas와 완벽하게 호환됩니다.

import seaborn as sns

# seaborn 스타일 설정
sns.set_style("whitegrid")
sns.set_palette("husl")

# 샘플 데이터 생성
np.random.seed(42)
df_seaborn = pd.DataFrame({
    '카테고리': np.random.choice(['A', 'B', 'C'], 200),
    '지역': np.random.choice(['서울', '부산', '대구', '인천'], 200),
    '매출': np.random.exponential(50, 200),
    '고객수': np.random.poisson(30, 200),
    '만족도': np.random.normal(7, 1.5, 200)
})

# 1. 카운트 플롯
plt.figure(figsize=(10, 5))
sns.countplot(data=df_seaborn, x='카테고리', hue='지역')
plt.title('카테고리별 지역 분포')
plt.show()

# 2. 박스 플롯
plt.figure(figsize=(12, 6))
sns.boxplot(data=df_seaborn, x='지역', y='매출', hue='카테고리')
plt.title('지역 및 카테고리별 매출 분포')
plt.show()

# 3. 바이올린 플롯
plt.figure(figsize=(12, 6))
sns.violinplot(data=df_seaborn, x='지역', y='만족도', hue='카테고리', split=False)
plt.title('지역 및 카테고리별 만족도 분포')
plt.show()

# 4. 히트맵
pivot_data = df_seaborn.pivot_table(values='매출', index='지역', columns='카테고리', aggfunc='mean')
plt.figure(figsize=(8, 6))
sns.heatmap(pivot_data, annot=True, fmt='.1f', cmap='YlOrRd', linewidths=0.5)
plt.title('지역 및 카테고리별 평균 매출 히트맵')
plt.show()

# 5. 페어플롯 (상관관계 행렬)
sns.pairplot(df_seaborn[['매출', '고객수', '만족도']], diag_kind='kde', height=3)
plt.suptitle('변수 간 상관관계', y=1.02)
plt.show()

6.3.1 고급 seaborn 시각화​

# 1. 조인트 플롯 (산점도 + 히스토그램)
sns.jointplot(data=df_seaborn, x='고객수', y='매출', kind='hex', height=8)

temp_data = {

    '온도': np.concatenate([
        np.random.normal(0, 3, 30),   # 1월

df_temp = pd.DataFrame(temp_data)

6.5 서브플롯 (Subplots)​

여러 차트를 한 화면에 배치하여 비교 분석할 수 있습니다.

# 종합 대시보드 데이터
np.random.seed(42)
dates = pd.date_range('2024-01-01', periods=365)
dashboard_data = pd.DataFrame({
    '매출': np.cumsum(np.random.randn(365)) * 10 + 1000,
    '방문자': np.random.poisson(500, 365),
    '전환율': np.random.beta(2, 5, 365) * 100,
    '광고비': np.random.gamma(2, 50, 365)
}, index=dates)

# 2x2 서브플롯
fig, axes = plt.subplots(2, 2, figsize=(15, 10))
fig.suptitle('비즈니스 대시보드', fontsize=16, fontweight='bold')

# 1. 매출 추이
dashboard_data['매출'].plot(ax=axes[0, 0], color='#FF6B6B', linewidth=2)
axes[0, 0].set_title('일별 매출 추이', fontweight='bold')
axes[0, 0].set_ylabel('매출 (만원)')
axes[0, 0].grid(True, alpha=0.3)

# 2. 방문자 수
dashboard_data['방문자'].plot(ax=axes[0, 1], color='#4ECDC4', alpha=0.6)
axes[0, 1].fill_between(dates, dashboard_data['방문자'], alpha=0.3, color='#4ECDC4')
axes[0, 1].set_title('일별 방문자 수', fontweight='bold')
axes[0, 1].set_ylabel('방문자 수')
axes[0, 1].grid(True, alpha=0.3)

# 3. 전환율 히스토그램
dashboard_data['전환율'].plot(kind='hist', ax=axes[1, 0], bins=30,
                            color='#45B7D1', edgecolor='black')
axes[1, 0].set_title('전환율 분포', fontweight='bold')
axes[1, 0].set_xlabel('전환율 (%)')
axes[1, 0].set_ylabel('빈도')

# 4. 매출 vs 광고비 산점도
axes[1, 1].scatter(dashboard_data['광고비'], dashboard_data['매출'],
                   alpha=0.5, color='#FFA07A')
axes[1, 1].set_title('광고비 vs 매출', fontweight='bold')

from scipy import stats

# 선형 회귀를 통한 추세선
x = np.arange(len(ts_data))
slope, intercept, r_value, p_value, std_err = stats.linregress(x, ts_data['매출'])
trend_line = slope * x + intercept

fig, axes = plt.subplots(2, 1, figsize=(14, 10))

# 1. 추세선과 신뢰구간
axes[0].plot(ts_data.index, ts_data['매출'], alpha=0.3, label='실제 매출')
axes[0].plot(ts_data.index, trend_line, 'r--', linewidth=2, label='추세선')

# 신뢰구간 계산
residuals = ts_data['매출'] - trend_line
std_residuals = np.std(residuals)
axes[0].fill_between(ts_data.index,
                      trend_line - 2*std_residuals,
                      trend_line + 2*std_residuals,
                      alpha=0.2, color='red', label='95% 신뢰구간')

axes[0].set_title(f'매출 추세 분석 (R² = {r_value**2:.4f})', fontweight='bold', fontsize=12)
axes[0].set_ylabel('매출 (만원)')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# 2. 잔차 플롯
axes[1].scatter(ts_data.index, residuals, alpha=0.3, s=10)
axes[1].axhline(y=0, color='r', linestyle='--', linewidth=2)
axes[1].axhline(y=2*std_residuals, color='orange', linestyle=':', linewidth=1, label='±2σ')
axes[1].axhline(y=-2*std_residuals, color='orange', linestyle=':', linewidth=1)
axes[1].set_title('잔차 분석', fontweight='bold', fontsize=12)
