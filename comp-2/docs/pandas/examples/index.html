8.1 매출 데이터 분석​

데이터 준비​

import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# 샘플 매출 데이터 생성
np.random.seed(42)
dates = pd.date_range('2024-01-01', '2024-12-31', freq='D')
products = ['노트북', '마우스', '키보드', '모니터', '헤드셋']

sales_data = []
for date in dates:
    for _ in range(np.random.randint(5, 15)):
        sales_data.append({
            '날짜': date,
            '제품': np.random.choice(products),
            '수량': np.random.randint(1, 5),
            '단가': np.random.choice([50000, 30000, 80000, 200000, 60000]),
            '지역': np.random.choice(['서울', '부산', '대구', '인천'])
        })

df = pd.DataFrame(sales_data)
df['총액'] = df['수량'] * df['단가']

8.1.1 월별 매출 분석​

# 월별 매출 집계
df['년월'] = df['날짜'].dt.to_period('M')
monthly_sales = df.groupby('년월')['총액'].sum()

print("월별 매출:")
print(monthly_sales)

# 전월 대비 증감률

    '총액': 'sum',
    '수량': 'sum',
    '날짜': 'count'
}).rename(columns={'날짜': '거래건수'})

8.2.1 RFM 분석​

# 고객 거래 데이터 생성
customer_data = df.copy()
customer_data['고객ID'] = np.random.randint(1000, 1100, len(df))

# 기준일 설정
analysis_date = df['날짜'].max()

# RFM 계산
rfm = customer_data.groupby('고객ID').agg({
    '날짜': lambda x: (analysis_date - x.max()).days,  # Recency
    '총액': ['sum', 'count']  # Monetary, Frequency
})

rfm.columns = ['Recency', 'Monetary', 'Frequency']
rfm = rfm.reset_index()

# RFM 점수 부여 (1-5점)
rfm['R_Score'] = pd.qcut(rfm['Recency'], 5, labels=[5, 4, 3, 2, 1])
rfm['F_Score'] = pd.qcut(rfm['Frequency'].rank(method='first'), 5, labels=[1, 2, 3, 4, 5])
rfm['M_Score'] = pd.qcut(rfm['Monetary'], 5, labels=[1, 2, 3, 4, 5])

# RFM 점수 합산
rfm['RFM_Score'] = rfm['R_Score'].astype(str) + rfm['F_Score'].astype(str) + rfm['M_Score'].astype(str)

print("RFM 분석 결과:")
print(rfm.head(10))

8.2.2 고객 등급 분류​

# 고객 세그먼트 정의
def customer_segment(row):
    if row['R_Score'] >= 4 and row['F_Score'] >= 4 and row['M_Score'] >= 4:
        return 'VIP'
    elif row['R_Score'] >= 3 and row['F_Score'] >= 3:
        return '우수고객'
    elif row['R_Score'] <= 2:
        return '이탈위험'
    else:
        return '일반고객'

rfm['세그먼트'] = rfm.apply(customer_segment, axis=1)

    'Recency': 'mean'

segment_stats.columns = ['고객수', '평균구매액', '평균구매빈도', '평균최근구매일']

def validate_dataframe(df, rules):
    """
    데이터프레임 검증

    Parameters:
    -----------
    df : DataFrame
        검증할 데이터프레임
    rules : dict
        검증 규칙

    Returns:
    --------
    dict : 검증 결과
    """
    results = {}

    # 필수 컬럼 확인
    if 'required_columns' in rules:
        missing = set(rules['required_columns']) - set(df.columns)
        results['missing_columns'] = list(missing)

    # 결측치 확인
    if 'no_nulls' in rules:
        null_cols = df[rules['no_nulls']].isnull().sum()
        results['null_counts'] = null_cols[null_cols > 0].to_dict()

    # 범위 확인
    if 'ranges' in rules:
        for col, (min_val, max_val) in rules['ranges'].items():
            out_of_range = df[
                (df[col] < min_val) | (df[col] > max_val)
            ]
            if len(out_of_range) > 0:
                results[f'{col}_out_of_range'] = len(out_of_range)

    # 중복 확인
    if 'unique_columns' in rules:
        for col in rules['unique_columns']:
            duplicates = df[df.duplicated(subset=col, keep=False)]
            if len(duplicates) > 0:
                results[f'{col}_duplicates'] = len(duplicates)

    return results

# 검증 규칙 정의
validation_rules = {
    'required_columns': ['날짜', '제품', '수량', '단가'],
    'no_nulls': ['날짜', '제품'],
    'ranges': {
        '수량': (0, 100),
        '단가': (0, 1000000)
    }
}

# 검증 실행
validation_result = validate_dataframe(df, validation_rules)

if validation_result:
    print("검증 실패 항목:")
    for key, value in validation_result.items():
        print(f"  {key}: {value}")
else:
    print("모든 검증 통과")

# 데이터 타입 최적화
def optimize_dtypes(df):
    """데이터 타입을 최적화하여 메모리 사용량 감소"""

    # 시작 메모리
    start_mem = df.memory_usage(deep=True).sum() / 1024**2

    for col in df.columns:
        col_type = df[col].dtype

        if col_type != object:
            c_min = df[col].min()
            c_max = df[col].max()

            if str(col_type)[:3] == 'int':
                if c_min > np.iinfo(np.int8).min and c_max < np.iinfo(np.int8).max:
                    df[col] = df[col].astype(np.int8)
                elif c_min > np.iinfo(np.int16).min and c_max < np.iinfo(np.int16).max:
                    df[col] = df[col].astype(np.int16)
                elif c_min > np.iinfo(np.int32).min and c_max < np.iinfo(np.int32).max:
                    df[col] = df[col].astype(np.int32)

            elif str(col_type)[:5] == 'float':
                if c_min > np.finfo(np.float32).min and c_max < np.finfo(np.float32).max:
                    df[col] = df[col].astype(np.float32)

        else:
            # 카테고리형으로 변환 (고유값이 적은 경우)
            if df[col].nunique() / len(df) < 0.5:
                df[col] = df[col].astype('category')

    # 끝 메모리
    end_mem = df.memory_usage(deep=True).sum() / 1024**2

    print(f'메모리 사용량: {start_mem:.2f} MB → {end_mem:.2f} MB')
    print(f'감소율: {100 * (start_mem - end_mem) / start_mem:.1f}%')

    return df

# 사용 예
df_optimized = optimize_dtypes(df.copy())

매출 데이터에서 이상치를 탐지하고 처리하세요:

  1. IQR 방법으로 이상치 탐지
  2. 이상치 통계 출력
  3. 이상치 제거 또는 대체
해답
def detect_outliers(df, column):
    """IQR 방법으로 이상치 탐지"""
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1

    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    outliers = df[
        (df[column] < lower_bound) |
        (df[column] > upper_bound)
    ]

    print(f"\n=== {column} 이상치 분석 ===")
    print(f"Q1: {Q1:,.0f}")
    print(f"Q3: {Q3:,.0f}")
    print(f"IQR: {IQR:,.0f}")
    print(f"정상 범위: {lower_bound:,.0f} ~ {upper_bound:,.0f}")
    print(f"이상치 개수: {len(outliers)} ({len(outliers)/len(df)*100:.2f}%)")

    return outliers, lower_bound, upper_bound

# 총액 컬럼 이상치 탐지
outliers, lower, upper = detect_outliers(df, '총액')

# 이상치 처리 옵션 1: 제거
df_cleaned = df[
    (df['총액'] >= lower) &
    (df['총액'] <= upper)
].copy()

print(f"\n제거 후 데이터: {len(df_cleaned)}건 (원본: {len(df)}건)")

# 이상치 처리 옵션 2: 캡핑 (상/하한값으로 대체)
df_capped = df.copy()
df_capped.loc[df_capped['총액'] < lower, '총액'] = lower
df_capped.loc[df_capped['총액'] > upper, '총액'] = upper

print(f"캡핑 후 범위: {df_capped['총액'].min():,.0f} ~ {df_capped['총액'].max():,.0f}")

    """데이터 로드 및 기본 전처리"""
    df = pd.read_csv(file_path)
    df['날짜'] = pd.to_datetime(df['날짜'])
    return df

def save_report(df, filename):
    """리포트 저장"""
    output_path = Config.OUTPUT_DIR + filename
    df.to_excel(output_path, index=False)
    print(f"저장 완료: {output_path}")

# main.py - 메인 분석 스크립트
from config import Config
from utils import load_data, save_report

def main():
    # 데이터 로드
    df = load_data(Config.DATA_DIR + 'sales.csv')

    # 분석 수행
    monthly_sales = df.groupby('년월')['총액'].sum()

    # 결과 저장
    save_report(monthly_sales, 'monthly_report.xlsx')

if __name__ == '__main__':
    main()

def safe_read_csv(file_path, **kwargs):
    """안전한 CSV 읽기"""
    try:
        df = pd.read_csv(file_path, **kwargs)
        print(f"파일 로드 성공: {len(df)}건")
        return df
    except FileNotFoundError:
        print(f"오류: 파일을 찾을 수 없습니다 - {file_path}")
        return None
    except pd.errors.EmptyDataError:
        print(f"오류: 파일이 비어있습니다 - {file_path}")
        return None
    except Exception as e:
        print(f"오류 발생: {str(e)}")
        return None
