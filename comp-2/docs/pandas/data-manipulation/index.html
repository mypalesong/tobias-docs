3. 데이터 조작

Pandas의 핵심 기능인 데이터 조작 방법을 학습합니다. 정렬, 필터링, 그룹화, 피벗 테이블 등 실무에서 가장 많이 사용되는 기법들을 다룹니다.

3.1 정렬 (Sorting)​

데이터를 특정 기준에 따라 정렬하는 방법을 알아봅니다.

sort_values() - 값 기준 정렬​

import pandas as pd
import numpy as np

# 샘플 데이터 생성
df = pd.DataFrame({
    '제품명': ['노트북', '마우스', '키보드', '모니터', '헤드셋'],
    '카테고리': ['컴퓨터', '주변기기', '주변기기', '컴퓨터', '주변기기'],
    '가격': [1200000, 35000, 89000, 450000, 120000],
    '재고': [15, 150, 80, 25, 45],
    '평점': [4.5, 4.2, 4.7, 4.3, 4.1]
})

# 단일 컬럼 정렬 (오름차순)
df_sorted = df.sort_values('가격')
print(df_sorted)

# 내림차순 정렬
df_sorted_desc = df.sort_values('가격', ascending=False)
print(df_sorted_desc)

# 여러 컬럼 기준 정렬
df_multi_sort = df.sort_values(['카테고리', '가격'], ascending=[True, False])
print(df_multi_sort)

# 정렬 후 인덱스 리셋
df_reset = df.sort_values('평점', ascending=False).reset_index(drop=True)
print(df_reset)

sort_index() - 인덱스 기준 정렬​

# 인덱스가 있는 데이터프레임
df_indexed = df.set_index('제품명')
print(df_indexed)

# 인덱스 기준 정렬
df_index_sorted = df_indexed.sort_index()
print(df_index_sorted)

# 컬럼 기준 정렬 (axis=1)
df_col_sorted = df.sort_index(axis=1)
print(df_col_sorted)
정렬 성능 최적화

대용량 데이터를 정렬할 때는 kind='mergesort' 또는 kind='heapsort' 파라미터를 사용하여 안정적인 정렬을 수행할 수 있습니다. 기본값은 quicksort입니다.

3.2 필터링 (Filtering)​

조건에 맞는 데이터만 추출하는 다양한 방법을 학습합니다.

Boolean Indexing​

# 샘플 매출 데이터
sales_df = pd.DataFrame({
    '날짜': pd.date_range('2024-01-01', periods=10),
    '매출액': [350000, 420000, 380000, 520000, 490000,
              610000, 580000, 450000, 390000, 560000],
    '고객수': [25, 32, 28, 41, 38, 52, 48, 35, 29, 44],
    '지역': ['서울', '경기', '서울', '부산', '서울',
            '경기', '부산', '서울', '경기', '부산']
})

# 단일 조건 필터링
high_sales = sales_df[sales_df['매출액'] > 500000]
print(high_sales)

# 여러 조건 필터링 (AND)
seoul_high = sales_df[(sales_df['지역'] == '서울') & (sales_df['매출액'] > 400000)]
print(seoul_high)

# 여러 조건 필터링 (OR)
condition = sales_df[(sales_df['지역'] == '서울') | (sales_df['고객수'] > 40)]
print(condition)

# NOT 조건
not_seoul = sales_df[~(sales_df['지역'] == '서울')]
print(not_seoul)

# isin() 메서드 활용
major_cities = sales_df[sales_df['지역'].isin(['서울', '부산'])]
print(major_cities)

query() 메서드​

SQL 스타일의 문자열 표현식으로 필터링할 수 있습니다.

# query() 메서드 사용
result = sales_df.query('매출액 > 500000')
print(result)

# 여러 조건
result2 = sales_df.query('지역 == "서울" and 고객수 > 25')
print(result2)

# 변수 사용 (@)
threshold = 450000
result3 = sales_df.query('매출액 > @threshold')
print(result3)

# in 연산자
result4 = sales_df.query('지역 in ["서울", "부산"]')
print(result4)

# 문자열 메서드와 결합
df_products = pd.DataFrame({
    '제품명': ['iPhone 15', 'Galaxy S24', 'iPad Pro', 'Galaxy Tab', 'MacBook'],
    '가격': [1200000, 1100000, 1500000, 800000, 2500000]
})
result5 = df_products.query('제품명.str.contains("Galaxy")', engine='python')
print(result5)
query() vs Boolean Indexing

query() 메서드는 가독성이 좋고 복잡한 조건을 간결하게 표현할 수 있습니다. 하지만 Boolean Indexing이 일반적으로 더 빠르며, 모든 상황에서 사용 가능합니다.

상위/하위 데이터 추출​

# 상위 N개 행
top_3 = sales_df.nlargest(3, '매출액')
print(top_3)

# 하위 N개 행
bottom_3 = sales_df.nsmallest(3, '매출액')
print(bottom_3)

# 여러 컬럼 기준
top_customers = sales_df.nlargest(5, ['고객수', '매출액'])
print(top_customers)

3.3 GroupBy 그룹화 및 집계​

데이터를 그룹화하여 집계 분석을 수행합니다.

기본 그룹화​

# 그룹별 집계
region_stats = sales_df.groupby('지역').agg({
    '매출액': ['sum', 'mean', 'count'],
    '고객수': ['sum', 'mean']
})
print(region_stats)

# 단일 집계 함수
region_total = sales_df.groupby('지역')['매출액'].sum()
print(region_total)

# 여러 컬럼 그룹화
df_detail = pd.DataFrame({
    '날짜': pd.date_range('2024-01-01', periods=12),
    '지역': ['서울', '경기', '부산'] * 4,
    '카테고리': ['전자', '전자', '의류', '식품'] * 3,
    '매출액': np.random.randint(100000, 1000000, 12)
})

multi_group = df_detail.groupby(['지역', '카테고리'])['매출액'].sum()
print(multi_group)

# DataFrame으로 변환
multi_group_df = multi_group.reset_index()
print(multi_group_df)

고급 집계​

# 사용자 정의 집계 함수
def sales_range(x):
    return x.max() - x.min()

custom_agg = sales_df.groupby('지역')['매출액'].agg([
    ('총매출', 'sum'),
    ('평균매출', 'mean'),
    ('최대매출', 'max'),
    ('최소매출', 'min'),
    ('매출범위', sales_range),
    ('표준편차', 'std')
])
print(custom_agg)

# transform() - 그룹별 연산 결과를 원본 크기로 반환
sales_df['그룹평균'] = sales_df.groupby('지역')['매출액'].transform('mean')
sales_df['평균대비차이'] = sales_df['매출액'] - sales_df['그룹평균']
print(sales_df)

# filter() - 그룹 단위 필터링
# 평균 매출이 450000 이상인 지역의 데이터만 추출
filtered = sales_df.groupby('지역').filter(lambda x: x['매출액'].mean() > 450000)
print(filtered)

피벗 테이블 스타일 결과​

# unstack()으로 피벗 형태 변환
pivot_style = df_detail.groupby(['지역', '카테고리'])['매출액'].sum().unstack(fill_value=0)
print(pivot_style)

# 퍼센트 계산
region_category_pct = df_detail.groupby(['지역', '카테고리'])['매출액'].sum()
region_category_pct = region_category_pct / region_category_pct.sum() * 100
print(region_category_pct)

3.4 apply, map, applymap 함수​

데이터 변환을 위한 함수 적용 방법을 학습합니다.

apply() - 행/열 단위 함수 적용​

# 샘플 데이터
df_score = pd.DataFrame({
    '이름': ['김철수', '이영희', '박민수', '정수진'],
    '수학': [85, 92, 78, 95],
    '영어': [88, 85, 90, 92],
    '과학': [90, 88, 85, 89]
})

# 컬럼에 함수 적용
df_score['총점'] = df_score[['수학', '영어', '과학']].apply(sum, axis=1)
df_score['평균'] = df_score[['수학', '영어', '과학']].apply(np.mean, axis=1)
print(df_score)

# 사용자 정의 함수 적용
def grade(score):
    if score >= 90:
        return 'A'
    elif score >= 80:
        return 'B'
    elif score >= 70:
        return 'C'
    else:
        return 'D'

df_score['등급'] = df_score['평균'].apply(grade)
print(df_score)

# lambda 함수 활용
df_score['평균(반올림)'] = df_score['평균'].apply(lambda x: round(x, 1))
print(df_score)

# 여러 값 반환 (Series 반환)
def calc_stats(row):
    scores = [row['수학'], row['영어'], row['과학']]
    return pd.Series({
        '최고점수': max(scores),
        '최저점수': min(scores),
        '점수차': max(scores) - min(scores)
    })

stats = df_score.apply(calc_stats, axis=1)
df_score = pd.concat([df_score, stats], axis=1)
print(df_score)

map() - Series 변환​

# 딕셔너리 매핑
category_map = {
    '전자': 'Electronics',
    '의류': 'Clothing',
    '식품': 'Food'
}

df_detail['Category_EN'] = df_detail['카테고리'].map(category_map)
print(df_detail)

# 함수 매핑
df_detail['매출액_원'] = df_detail['매출액'].map(lambda x: f'{x:,}원')
print(df_detail)

# Series 매핑
region_code = pd.Series({
    '서울': 'SEL',
    '경기': 'GG',
    '부산': 'BS'
})
df_detail['지역코드'] = df_detail['지역'].map(region_code)
print(df_detail)

applymap() - 모든 원소에 함수 적용 (DataFrame)​

# DataFrame 전체에 함수 적용 (Pandas 2.1.0+ 에서는 map() 사용)
df_numbers = pd.DataFrame({
    'A': [1.234, 2.567, 3.891],
    'B': [4.123, 5.678, 6.234],
    'C': [7.567, 8.123, 9.456]
})

# applymap() - 구버전
# df_rounded = df_numbers.applymap(lambda x: round(x, 1))

# map() - 신버전 (Pandas 2.1.0+)
try:
    df_rounded = df_numbers.map(lambda x: round(x, 1))
except AttributeError:
    # 구버전 Pandas
    df_rounded = df_numbers.applymap(lambda x: round(x, 1))

print(df_rounded)

# 포맷팅 적용
try:
    df_formatted = df_numbers.map(lambda x: f'{x:.2f}')
except AttributeError:
    df_formatted = df_numbers.applymap(lambda x: f'{x:.2f}')

print(df_formatted)
성능 최적화

apply() 함수는 편리하지만 상대적으로 느립니다. 가능하면 벡터화된 연산이나 내장 함수를 사용하세요. 예: df['col'].apply(lambda x: x * 2) 대신 df['col'] * 2 사용

3.5 피벗 테이블 (Pivot Table)​

데이터를 재구조화하여 요약 통계를 생성합니다.

pivot_table() 기본​

# 상세 매출 데이터
sales_detail = pd.DataFrame({
    '날짜': pd.date_range('2024-01-01', periods=20),
    '지역': ['서울', '경기', '부산', '인천', '대구'] * 4,
    '카테고리': ['전자', '의류', '식품', '가전', '서적'] * 4,
    '매출액': np.random.randint(100000, 1000000, 20),
    '수량': np.random.randint(10, 100, 20),
    '할인액': np.random.randint(10000, 100000, 20)
})

# 기본 피벗 테이블
pivot1 = pd.pivot_table(
    sales_detail,
    values='매출액',
    index='지역',
    columns='카테고리',
    aggfunc='sum',
    fill_value=0
)
print(pivot1)

# 여러 집계 함수
pivot2 = pd.pivot_table(
    sales_detail,
    values='매출액',
    index='지역',
    columns='카테고리',
    aggfunc=['sum', 'mean', 'count'],
    fill_value=0
)
print(pivot2)

# 여러 값 컬럼
pivot3 = pd.pivot_table(
    sales_detail,
    values=['매출액', '수량'],
    index='지역',
    columns='카테고리',
    aggfunc='sum',
    fill_value=0
)
print(pivot3)

고급 피벗 테이블​

# 다중 인덱스 피벗
sales_detail['월'] = sales_detail['날짜'].dt.month
pivot4 = pd.pivot_table(
    sales_detail,
    values='매출액',
    index=['월', '지역'],
    columns='카테고리',
    aggfunc='sum',
    fill_value=0,
    margins=True,  # 총계 추가
    margins_name='총계'
)
print(pivot4)

# 퍼센트 계산
pivot_pct = pivot1.div(pivot1.sum(axis=1), axis=0) * 100
print(pivot_pct.round(2))

# 사용자 정의 집계 함수
def sales_per_quantity(x):
    return x['매출액'].sum() / x['수량'].sum()

pivot5 = pd.pivot_table(
    sales_detail,
    index='지역',
    columns='카테고리',
    aggfunc=sales_per_quantity

3.6 데이터 재구조화 (Reshaping)​

Wide 형식과 Long 형식 간의 변환을 학습합니다.

melt() - Wide to Long​

# Wide 형식 데이터
wide_df = pd.DataFrame({
    '제품': ['노트북', '태블릿', '스마트폰'],
    '1월': [100, 80, 150],
    '2월': [120, 85, 160],
    '3월': [110, 90, 155]
})
print("Wide 형식:")
print(wide_df)

# Long 형식으로 변환
long_df = pd.melt(
    wide_df,
    id_vars='제품',
    var_name='월',
    value_name='판매량'
)
print("\nLong 형식:")
print(long_df)

# 여러 값 컬럼
wide_multi = pd.DataFrame({
    '제품': ['노트북', '태블릿'],
    '서울_매출': [1000000, 800000],
    '서울_수량': [10, 8],
    '부산_매출': [800000, 600000],
    '부산_수량': [8, 6]
})

long_multi = pd.melt(
    wide_multi,
    id_vars='제품',
    value_vars=['서울_매출', '서울_수량', '부산_매출', '부산_수량']
)
print(long_multi)

stack() / unstack() - 인덱스 재구조화​

# stack() - 컬럼을 인덱스로
df_sales = pd.DataFrame({
    '서울': [100, 120, 110],
    '부산': [80, 90, 85],
    '대구': [70, 75, 72]
}, index=['1월', '2월', '3월'])

stacked = df_sales.stack()

# unstack() - 인덱스를 컬럼으로
unstacked = stacked.unstack()

# 다중 인덱스에서 활용
multi_idx_df = pd.DataFrame({

    '수량': [10, 20, 15, 25, 12, 22]
}, index=pd.MultiIndex.from_product(

print("\n다중 인덱스:")
print(multi_idx_df)

# level 지정 unstack
unstacked_level0 = multi_idx_df.unstack(level=0)
print("\nUnstack level 0:")
print(unstacked_level0)

unstacked_level1 = multi_idx_df.unstack(level=1)
print("\nUnstack level 1:")
print(unstacked_level1)

3.7 조건부 로직 (Conditional Logic)​

조건에 따른 값 변경 및 선택을 수행합니다.

where() / mask()​

# 샘플 데이터
df_temp = pd.DataFrame({
    '날짜': pd.date_range('2024-01-01', periods=7),
    '온도': [5, 8, 12, 15, 18, 22, 25],
    '습도': [60, 65, 55, 50, 48, 45, 42]
})

# where() - 조건이 True인 값 유지, False는 다른 값으로 대체
temp_filtered = df_temp['온도'].where(df_temp['온도'] > 10, other='낮음')
print(temp_filtered)

# mask() - 조건이 True인 값을 대체 (where의 반대)
temp_masked = df_temp['온도'].mask(df_temp['온도'] < 10, other='낮음')
print(temp_masked)

# DataFrame 전체에 적용
df_filtered = df_temp.where(df_temp > 50, -1)
print(df_filtered)

np.where() - 조건부 값 할당​

# 단일 조건

    '제품카테고리': np.random.choice(['전자', '의류', '식품', '가전'], 100),
    '매출액': np.random.randint(50000, 2000000, 100),

    '지역': np.random.choice(['서울', '경기', '부산', '대구', '인천'], 100)

# 1. 월별 매출 추이

category_region_pivot = pd.pivot_table(
    sales_data,
    values='매출액',

    columns='지역',
    aggfunc='sum',

customer_analysis = sales_data.groupby('고객ID').agg({

예제 2: 고객 세그먼트 분석​

# RFM 분석 (Recency, Frequency, Monetary)
analysis_date = sales_data['주문일시'].max()

rfm = sales_data.groupby('고객ID').agg({
    '주문일시': lambda x: (analysis_date - x.max()).days,  # Recency
    '주문번호': 'count',  # Frequency
    '매출액': 'sum'  # Monetary
})
rfm.columns = ['Recency', 'Frequency', 'Monetary']

# RFM 점수 계산 (5점 척도)
rfm['R_Score'] = pd.qcut(rfm['Recency'], 5, labels=[5, 4, 3, 2, 1])
rfm['F_Score'] = pd.qcut(rfm['Frequency'].rank(method='first'), 5, labels=[1, 2, 3, 4, 5])
rfm['M_Score'] = pd.qcut(rfm['Monetary'], 5, labels=[1, 2, 3, 4, 5])

# RFM 점수 합계
rfm['RFM_Score'] = (
    rfm['R_Score'].astype(int) +
    rfm['F_Score'].astype(int) +
    rfm['M_Score'].astype(int)
)

# 고객 세그먼트 분류
def segment_customer(row):
    if row['RFM_Score'] >= 12:
        return 'Champions'
    elif row['RFM_Score'] >= 9:
        return 'Loyal'
    elif row['RFM_Score'] >= 6:
        return 'Potential'
    else:
        return 'At Risk'

rfm['Segment'] = rfm.apply(segment_customer, axis=1)

print("RFM 분석 결과:")
print(rfm.sort_values('RFM_Score', ascending=False))

# 세그먼트별 통계
segment_stats = rfm.groupby('Segment').agg({
    'Recency': 'mean',
    'Frequency': 'mean',
    'Monetary': 'mean',
    'RFM_Score': 'count'
}).round(2)
segment_stats.columns = ['평균최근구매일', '평균구매빈도', '평균구매금액', '고객수']
print("\n세그먼트별 통계:")
print(segment_stats)

# 세그먼트별 매출 기여도
segment_contribution = rfm.groupby('Segment')['Monetary'].sum()
segment_contribution_pct = (segment_contribution / segment_contribution.sum() * 100).round(2)
print("\n세그먼트별 매출 기여도 (%):")
print(segment_contribution_pct)

예제 3: 시계열 트렌드 분석​

# 일별 매출 데이터
daily_sales = sales_data.groupby(sales_data['주문일시'].dt.date).agg({
    '매출액': 'sum',
    '주문번호': 'count'
}).reset_index()
daily_sales.columns = ['날짜', '매출액', '주문건수']
daily_sales['날짜'] = pd.to_datetime(daily_sales['날짜'])

# 이동평균 계산 (7일)
daily_sales['매출_7일평균'] = daily_sales['매출액'].rolling(window=7).mean()

# 전일 대비 증감
daily_sales['전일대비'] = daily_sales['매출액'].diff()
daily_sales['전일대비율'] = daily_sales['매출액'].pct_change() * 100

# 전주 동일 요일 대비
daily_sales['전주대비'] = daily_sales['매출액'].diff(7)
daily_sales['전주대비율'] = (daily_sales['매출액'] / daily_sales['매출액'].shift(7) - 1) * 100

print("시계열 트렌드 분석:")
print(daily_sales.tail(10))

# 매출 급증/급락 탐지
threshold = daily_sales['전일대비율'].std() * 2
daily_sales['이상치'] = np.where(
    abs(daily_sales['전일대비율']) > threshold,
    '이상',
    '정상'
)

anomalies = daily_sales[daily_sales['이상치'] == '이상']
print(f"\n이상치 탐지 (±{threshold:.1f}% 초과):")
print(anomalies[['날짜', '매출액', '전일대비율']])

  5. 요일별 평균 매출 계산
해답 보기
# 1. 제품별 총 매출 (내림차순)
product_sales = sales.groupby('제품')['매출'].sum().sort_values(ascending=False)
print("제품별 총 매출:")
print(product_sales)

# 2. 매출 500,000원 이상 필터링
high_sales = sales[sales['매출'] >= 500000]
print(f"\n고매출 거래 건수: {len(high_sales)}")

# 3. 지역별, 제품별 평균 매출 피벗 테이블
pivot = pd.pivot_table(
    sales,
    values='매출',
    index='지역',
    columns='제품',
    aggfunc='mean',
    fill_value=0
).round(0)
print("\n지역별 제품별 평균 매출:")
print(pivot)

# 4. 매출 등급 부여
sales['등급'] = pd.cut(
    sales['매출'],
    bins=[0, 400000, 700000, float('inf')],
    labels=['하', '중', '상']
)
print("\n등급별 거래 건수:")
print(sales['등급'].value_counts())

# 5. 요일별 평균 매출
sales['요일'] = sales['날짜'].dt.day_name()
weekday_avg = sales.groupby('요일')['매출'].mean().sort_values(ascending=False)
print("\n요일별 평균 매출:")
print(weekday_avg.round(0))

문제 2: 고객 데이터 정제​

다음 지저분한 고객 데이터를 정제하세요.

customers = pd.DataFrame({
    '이름': ['  김철수  ', 'LEE YOUNGHEE', 'park minsu', '정수진'],
    '이메일': ['KIM@email.com', 'lee@EMAIL.COM', 'Park@Email.com', 'jung@email.com'],
    '전화': ['010-1234-5678', '01098765432', '010.1111.2222', '010-5555-6666'],
    '가입일': ['2024-01-15', '2024/02/20', '03-10-2024', '2024.04.05']
})

요구사항:

  1. 이름의 앞뒤 공백 제거 및 첫 글자만 대문자로 변환
  2. 이메일을 모두 소문자로 변환
  3. 전화번호를 '010-XXXX-XXXX' 형식으로 통일
  4. 가입일을 datetime 형식으로 변환
  5. 가입 후 경과 일수 계산
해답 보기
# 1. 이름 정제
customers['이름'] = customers['이름'].str.strip().str.title()

# 2. 이메일 소문자 변환
customers['이메일'] = customers['이메일'].str.lower()

# 3. 전화번호 통일
# 먼저 숫자만 추출
customers['전화_숫자'] = customers['전화'].str.replace(r'[^0-9]', '', regex=True)
# 형식 적용
customers['전화'] = customers['전화_숫자'].str.replace(
    r'(\d{3})(\d{4})(\d{4})',
    r'\1-\2-\3',
    regex=True
)
customers.drop('전화_숫자', axis=1, inplace=True)

# 4. 가입일 변환
# 각 행마다 다른 형식이므로 개별 처리 또는 infer_datetime_format 사용
customers['가입일'] = pd.to_datetime(customers['가입일'], infer_datetime_format=True)

# 더 안전한 방법 (개별 처리)
def parse_date(date_str):
    formats = ['%Y-%m-%d', '%Y/%m/%d', '%m-%d-%Y', '%Y.%m.%d']
    for fmt in formats:
        try:
            return pd.to_datetime(date_str, format=fmt)
        except:
            continue
    return pd.NaT

customers['가입일'] = customers['가입일'].apply(parse_date)

# 5. 경과 일수 계산
customers['경과일수'] = (pd.Timestamp.now() - customers['가입일']).dt.days

print("정제된 고객 데이터:")
print(customers)

문제 3: 복합 분석​

다음 주문 데이터로 종합 분석을 수행하세요.

orders = pd.DataFrame({
    '주문ID': range(1, 51),
    '고객ID': np.random.choice(['C01', 'C02', 'C03', 'C04', 'C05'], 50),
    '주문일': pd.date_range('2024-01-01', periods=50, freq='D'),
    '카테고리': np.random.choice(['전자', '의류', '식품'], 50),
    '금액': np.random.randint(10000, 500000, 50),
    '상태': np.random.choice(['완료', '진행중', '취소'], 50, p=[0.7, 0.2, 0.1])
})

요구사항:

  1. 완료된 주문만 필터링
  2. 고객별 총 주문금액, 주문횟수, 평균 주문금액 계산
  3. 상위 3명의 VIP 고객 선정
  4. 월별 카테고리별 매출 피벗 테이블
  5. 각 고객의 최다 구매 카테고리 찾기
해답 보기
# 1. 완료된 주문만 필터링
completed_orders = orders[orders['상태'] == '완료'].copy()
print(f"완료된 주문: {len(completed_orders)}건")

# 2. 고객별 분석
customer_stats = completed_orders.groupby('고객ID').agg({
    '금액': ['sum', 'count', 'mean']
}).round(0)
customer_stats.columns = ['총주문금액', '주문횟수', '평균주문금액']
print("\n고객별 통계:")
print(customer_stats.sort_values('총주문금액', ascending=False))

# 3. VIP 고객 선정 (총 주문금액 기준)
vip_customers = customer_stats.nlargest(3, '총주문금액')
print("\nVIP 고객 (Top 3):")
print(vip_customers)

# 4. 월별 카테고리별 매출 피벗
completed_orders['월'] = completed_orders['주문일'].dt.to_period('M')
monthly_category_pivot = pd.pivot_table(
    completed_orders,
    values='금액',
    index='월',
    columns='카테고리',
    aggfunc='sum',
    fill_value=0,
    margins=True
)
print("\n월별 카테고리별 매출:")
print(monthly_category_pivot)

# 5. 고객별 최다 구매 카테고리
favorite_category = completed_orders.groupby('고객ID')['카테고리'].agg(
    lambda x: x.value_counts().index[0] if len(x) > 0 else None
)
print("\n고객별 선호 카테고리:")
print(favorite_category)

# 추가: 고객별 카테고리별 구매 금액
customer_category = completed_orders.groupby(['고객ID', '카테고리'])['금액'].sum().unstack(fill_value=0)
print("\n고객별 카테고리별 구매 금액:")
print(customer_category)
