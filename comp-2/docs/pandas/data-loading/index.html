2. 데이터 로딩

pandas의 가장 강력한 기능 중 하나는 다양한 형식의 데이터를 쉽게 불러오고 저장할 수 있다는 것입니다. CSV, Excel, JSON, SQL 데이터베이스 등 거의 모든 데이터 소스를 지원합니다.

2.1 CSV 파일​

CSV(Comma-Separated Values)는 가장 일반적인 데이터 형식입니다.

2.1.1 CSV 파일 읽기​

import pandas as pd

# 기본 사용법
df = pd.read_csv('data.csv')

# 구분자 지정 (탭으로 구분된 파일)
df = pd.read_csv('data.tsv', sep='\t')

# 헤더가 없는 파일 (첫 번째 행도 데이터)
df = pd.read_csv('data.csv', header=None)

# 헤더 이름 직접 지정
df = pd.read_csv('data.csv', names=['이름', '나이', '도시'])

# 특정 열만 읽기
df = pd.read_csv('data.csv', usecols=['이름', '나이'])

# 특정 열을 인덱스로 설정
df = pd.read_csv('data.csv', index_col='ID')

# 날짜 열 자동 파싱
df = pd.read_csv('data.csv', parse_dates=['가입일', '마지막_접속일'])

# 결측치 처리
df = pd.read_csv('data.csv', na_values=['NA', 'N/A', 'null', '-'])
팁

read_csv()는 자동으로 데이터 타입을 추론합니다. 하지만 dtype 매개변수로 명시적으로 지정하면 성능이 향상됩니다.

2.1.2 CSV 파일 저장​

# 기본 저장
df.to_csv('output.csv', index=False)

# 인덱스 포함 저장
df.to_csv('output.csv', index=True)

# 특정 열만 저장
df.to_csv('output.csv', columns=['이름', '나이'], index=False)

# 구분자 변경
df.to_csv('output.tsv', sep='\t', index=False)

# 인코딩 지정
df.to_csv('output.csv', encoding='utf-8-sig', index=False)

# 결측치 표현 방식 변경
df.to_csv('output.csv', na_rep='N/A', index=False)

2.1.3 실전 예제: 대용량 CSV 분석​

import pandas as pd

# 1단계: 파일 크기 확인을 위해 일부만 읽기
sample = pd.read_csv('large_file.csv', nrows=5)
print(sample.head())
print(sample.info())

# 2단계: 필요한 열만 선택하여 읽기
columns_to_use = ['사용자ID', '구매금액', '구매일']
df = pd.read_csv('large_file.csv', usecols=columns_to_use)

# 3단계: 데이터 타입 최적화
df['사용자ID'] = df['사용자ID'].astype('int32')
df['구매금액'] = df['구매금액'].astype('float32')
df['구매일'] = pd.to_datetime(df['구매일'])

print(f"메모리 사용량: {df.memory_usage(deep=True).sum() / 1024**2:.2f} MB")

2.2.3 실전 예제: 월별 보고서 생성​

import pandas as pd
from datetime import datetime

# 판매 데이터 생성
sales_data = pd.DataFrame({
    '날짜': pd.date_range('2024-01-01', periods=100),
    '제품': ['A', 'B', 'C'] * 33 + ['A'],
    '판매량': range(1, 101),
    '매출': range(1000, 100001, 1000)
})

# 월별 집계
sales_data['월'] = sales_data['날짜'].dt.month
monthly_summary = sales_data.groupby('월').agg({
    '판매량': 'sum',
    '매출': 'sum'
}).reset_index()

# 제품별 집계
product_summary = sales_data.groupby('제품').agg({
    '판매량': 'sum',
    '매출': 'sum'
}).reset_index()

# Excel 보고서 생성
with pd.ExcelWriter('월간_판매_보고서.xlsx', engine='openpyxl') as writer:
    # 원본 데이터
    sales_data.to_excel(writer, sheet_name='원본데이터', index=False)

    # 월별 요약
    monthly_summary.to_excel(writer, sheet_name='월별요약', index=False)

    # 제품별 요약
    product_summary.to_excel(writer, sheet_name='제품별요약', index=False)

print("보고서가 생성되었습니다!")

2.4 SQL 데이터베이스​

pandas는 SQL 데이터베이스와의 통합을 강력하게 지원합니다.

2.4.1 SQLite 사용하기​

import pandas as pd
import sqlite3

# 데이터베이스 연결
conn = sqlite3.connect('mydb.db')

# 테이블에서 데이터 읽기
df = pd.read_sql('SELECT * FROM customers', conn)

# 조건이 있는 쿼리
df = pd.read_sql('''
    SELECT name, age, city
    FROM customers
    WHERE age > 25
    ORDER BY age DESC
''', conn)

# 파라미터를 사용한 쿼리 (SQL 인젝션 방지)
age_limit = 30
df = pd.read_sql(
    'SELECT * FROM customers WHERE age > ?',
    conn,
    params=(age_limit,)
)

# DataFrame을 테이블로 저장
df.to_sql('customers', conn, if_exists='replace', index=False)
# if_exists 옵션: 'fail', 'replace', 'append'

# 연결 종료
conn.close()

2.4.2 MySQL/PostgreSQL 사용하기​

from sqlalchemy import create_engine

# MySQL 연결
engine = create_engine('mysql+pymysql://user:password@localhost/dbname')

# PostgreSQL 연결
engine = create_engine('postgresql://user:password@localhost/dbname')

# 데이터 읽기
df = pd.read_sql('SELECT * FROM products', engine)

# 데이터 쓰기
df.to_sql('products', engine, if_exists='append', index=False)

# 청크 단위로 저장 (대용량 데이터)
df.to_sql('products', engine, if_exists='append', index=False, chunksize=1000)

2.4.3 실전 예제: 데이터베이스 ETL​

import pandas as pd
import sqlite3

# 데이터베이스 생성 및 샘플 데이터 삽입
conn = sqlite3.connect('sales.db')

# 판매 데이터 생성
sales_df = pd.DataFrame({
    'product_id': [1, 2, 3, 1, 2],
    'product_name': ['노트북', '마우스', '키보드', '노트북', '마우스'],
    'quantity': [5, 10, 8, 3, 15],
    'price': [1000000, 30000, 80000, 1000000, 30000],
    'sale_date': ['2024-01-01', '2024-01-02', '2024-01-02', '2024-01-03', '2024-01-03']
})

# 데이터베이스에 저장
sales_df.to_sql('sales', conn, if_exists='replace', index=False)

# ETL: Extract, Transform, Load
# 1. Extract - 데이터 추출
query = '''
    SELECT
        product_name,
        SUM(quantity * price) as total_revenue,
        SUM(quantity) as total_quantity,
        COUNT(*) as transaction_count
    FROM sales
    GROUP BY product_name
    ORDER BY total_revenue DESC
'''

df = pd.read_sql(query, conn)

# 2. Transform - 데이터 변환
df['avg_price'] = df['total_revenue'] / df['total_quantity']
df['revenue_pct'] = (df['total_revenue'] / df['total_revenue'].sum() * 100).round(2)

# 3. Load - 결과 저장
df.to_sql('product_summary', conn, if_exists='replace', index=False)

print("제품별 매출 요약:")
print(df)

conn.close()

2.7.2 실전 예제: 공공 API 활용​

import pandas as pd
import requests
import json

# JSONPlaceholder API 사용 예제
def fetch_user_posts(user_id):
    """특정 사용자의 게시물을 가져오는 함수"""
    base_url = 'https://jsonplaceholder.typicode.com'

    # 사용자 정보 가져오기
    user_response = requests.get(f'{base_url}/users/{user_id}')
    user_data = user_response.json()

    # 게시물 가져오기
    posts_response = requests.get(f'{base_url}/posts?userId={user_id}')
    posts_data = posts_response.json()

    # DataFrame 생성
    posts_df = pd.DataFrame(posts_data)
    posts_df['userName'] = user_data['name']
    posts_df['userEmail'] = user_data['email']

    return posts_df

# 여러 사용자의 데이터 수집
all_posts = []
for user_id in range(1, 6):  # 사용자 1-5
    df = fetch_user_posts(user_id)
    all_posts.append(df)
    print(f"사용자 {user_id}의 게시물 {len(df)}개 수집 완료")

# 모든 데이터 합치기
final_df = pd.concat(all_posts, ignore_index=True)

print(f"\n총 {len(final_df)}개의 게시물 수집 완료")
print(final_df.head())

# 결과 저장
final_df.to_csv('user_posts.csv', index=False, encoding='utf-8-sig')

2.7.3 페이지네이션 처리​

import pandas as pd
import requests
import time

def fetch_all_pages(base_url, max_pages=None):
    """페이지네이션된 API 데이터 모두 가져오기"""
    all_data = []
    page = 1

    while True:
        if max_pages and page > max_pages:
            break

        try:
            response = requests.get(f'{base_url}?page={page}&limit=50')
            response.raise_for_status()  # HTTP 에러 확인

            data = response.json()

            if not data or len(data) == 0:
                break

            all_data.extend(data)
            print(f"페이지 {page} 처리 완료 ({len(data)}개 항목)")

            page += 1
            time.sleep(0.5)  # Rate limiting 고려

        except requests.exceptions.RequestException as e:
            print(f"에러 발생: {e}")
            break

    return pd.DataFrame(all_data)

import pandas as pd
from datetime import datetime

def analyze_large_log(filename, chunksize=50000):
    """대용량 로그 파일을 청크 단위로 분석"""

    # 통계 초기화
    stats = {
        'total_rows': 0,
        'error_count': 0,
        'warning_count': 0,
        'status_counts': {},
    }

    print(f"파일 분석 시작: {filename}")

    # 청크 단위로 처리
    for i, chunk in enumerate(pd.read_csv(filename, chunksize=chunksize)):
        stats['total_rows'] += len(chunk)

        # 로그 레벨별 집계
        if 'level' in chunk.columns:
            stats['error_count'] += (chunk['level'] == 'ERROR').sum()
            stats['warning_count'] += (chunk['level'] == 'WARNING').sum()

        # 상태 코드별 집계
        if 'status' in chunk.columns:
            status_counts = chunk['status'].value_counts()
            for status, count in status_counts.items():
                stats['status_counts'][status] = stats['status_counts'].get(status, 0) + count

        if (i + 1) % 10 == 0:
            print(f"  {(i + 1) * chunksize:,} 행 처리 완료...")

    print("\n분석 결과:")
    print(f"  총 행 수: {stats['total_rows']:,}")
    print(f"  에러 수: {stats['error_count']:,}")
    print(f"  경고 수: {stats['warning_count']:,}")
    print(f"  상태 코드 분포: {stats['status_counts']}")

    return stats

import pandas as pd

# 원본 데이터 읽기

# 데이터 타입 최적화
def optimize_dtypes(df):
    """DataFrame의 데이터 타입을 최적화"""

    for col in df.columns:
        col_type = df[col].dtype

        if col_type == 'object':
            # 문자열 타입 최적화
            num_unique = df[col].nunique()
            num_total = len(df[col])

            if num_unique / num_total < 0.5:  # 카테고리로 변환
                df[col] = df[col].astype('category')

        elif col_type == 'int64':
            # 정수 타입 최적화
            if df[col].min() >= 0:
                if df[col].max() < 255:
                    df[col] = df[col].astype('uint8')
                elif df[col].max() < 65535:
                    df[col] = df[col].astype('uint16')
                elif df[col].max() < 4294967295:
                    df[col] = df[col].astype('uint32')
            else:
                if df[col].min() > -128 and df[col].max() < 127:
                    df[col] = df[col].astype('int8')
                elif df[col].min() > -32768 and df[col].max() < 32767:
                    df[col] = df[col].astype('int16')
                elif df[col].min() > -2147483648 and df[col].max() < 2147483647:
                    df[col] = df[col].astype('int32')

        elif col_type == 'float64':
            # 실수 타입 최적화
            df[col] = df[col].astype('float32')

    return df

# 최적화 적용
df_optimized = optimize_dtypes(df.copy())
print(f"최적화 메모리 사용량: {df_optimized.memory_usage(deep=True).sum() / 1024**2:.2f} MB")

# 메모리 절감 비율 계산
reduction = (1 - df_optimized.memory_usage(deep=True).sum() / df.memory_usage(deep=True).sum()) * 100
print(f"메모리 절감: {reduction:.1f}%")

import chardet

def detect_encoding(filename):
    """파일의 인코딩 감지"""
    with open(filename, 'rb') as f:
        result = chardet.detect(f.read(100000))  # 처음 100KB 읽기
    return result['encoding']

Windows에서 생성된 CSV 파일은 보통 cp949 또는 euc-kr 인코딩을 사용합니다. 저장할 때는 utf-8-sig를 사용하면 Excel에서 한글이 깨지지 않습니다.

2.9.2 실전 예제: 다양한 인코딩 처리​

import pandas as pd
import chardet

def read_csv_auto_encoding(filename, **kwargs):
    """인코딩을 자동으로 감지하여 CSV 파일 읽기"""

    # 인코딩 감지
    with open(filename, 'rb') as f:
        raw_data = f.read(100000)
        result = chardet.detect(raw_data)
        encoding = result['encoding']
        confidence = result['confidence']

    print(f"감지된 인코딩: {encoding} (신뢰도: {confidence:.2%})")

    # 일반적인 인코딩 목록
    encodings_to_try = [encoding, 'utf-8', 'euc-kr', 'cp949', 'latin1']

    for enc in encodings_to_try:
        try:
            df = pd.read_csv(filename, encoding=enc, **kwargs)
            print(f"성공: {enc} 인코딩으로 파일 읽기 완료")
            return df
        except (UnicodeDecodeError, UnicodeError):
            print(f"실패: {enc} 인코딩으로 읽기 실패")
            continue

    raise ValueError(f"파일을 읽을 수 없습니다: {filename}")

2.10.2 자동화된 데이터 수집 파이프라인​

import pandas as pd
import requests
from datetime import datetime, timedelta
import time

class DataPipeline:
    """자동화된 데이터 수집 및 처리 파이프라인"""

    def __init__(self, api_url, output_dir='data'):
        self.api_url = api_url
        self.output_dir = output_dir

    def fetch_data(self, start_date, end_date):
        """API에서 날짜 범위의 데이터 수집"""
        all_data = []
        current_date = start_date

        while current_date <= end_date:
            try:
                params = {'date': current_date.strftime('%Y-%m-%d')}
                response = requests.get(self.api_url, params=params, timeout=10)

                if response.status_code == 200:
                    data = response.json()
                    all_data.extend(data)
                    print(f"  {current_date.date()} 데이터 수집 완료 ({len(data)}개)")
                else:
                    print(f"  {current_date.date()} 데이터 수집 실패: {response.status_code}")

                time.sleep(1)  # Rate limiting

            except Exception as e:
                print(f"  에러 발생: {e}")

            current_date += timedelta(days=1)

        return pd.DataFrame(all_data)

    def clean_data(self, df):
        """데이터 정제"""
        # 중복 제거
        df = df.drop_duplicates()

        # 결측치 처리
        df = df.dropna(subset=['id', 'date'])

        # 데이터 타입 변환
        if 'date' in df.columns:
            df['date'] = pd.to_datetime(df['date'])

        return df

    def save_data(self, df, filename_prefix='data'):
        """데이터 저장 (여러 형식)"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')

        # CSV 저장
        csv_filename = f'{self.output_dir}/{filename_prefix}_{timestamp}.csv'
        df.to_csv(csv_filename, index=False, encoding='utf-8-sig')
        print(f"CSV 저장 완료: {csv_filename}")

        # Excel 저장
        excel_filename = f'{self.output_dir}/{filename_prefix}_{timestamp}.xlsx'
        df.to_excel(excel_filename, index=False)
        print(f"Excel 저장 완료: {excel_filename}")

        return csv_filename, excel_filename

    def run(self, days_back=7):
        """파이프라인 실행"""
        print("데이터 수집 파이프라인 시작")

        # 날짜 범위 설정
        end_date = datetime.now()
        start_date = end_date - timedelta(days=days_back)

        print(f"수집 기간: {start_date.date()} ~ {end_date.date()}")

        # 1. 데이터 수집
        print("\n1. 데이터 수집 중...")
        df = self.fetch_data(start_date, end_date)
        print(f"   수집 완료: {len(df)}개 레코드")

        # 2. 데이터 정제
        print("\n2. 데이터 정제 중...")
        df = self.clean_data(df)
        print(f"   정제 완료: {len(df)}개 레코드")

        # 3. 데이터 저장
        print("\n3. 데이터 저장 중...")
        self.save_data(df)

        print("\n파이프라인 실행 완료!")
        return df

# JSONPlaceholder API 사용
def collect_and_analyze_posts():
    """게시물 데이터 수집 및 분석"""

    # 1. API에서 데이터 수집
    print("API에서 데이터 수집 중...")
    response = requests.get('https://jsonplaceholder.typicode.com/posts')
    posts = response.json()

    # DataFrame 생성
    df = pd.DataFrame(posts)
    print(f"수집 완료: {len(df)}개 게시물")

    # 2. 데이터 분석
    print("\n데이터 분석:")

    # 사용자별 게시물 수
    user_post_count = df.groupby('userId').size().reset_index(name='게시물_수')

        user_analysis.to_excel(writer, sheet_name='사용자별분석')
        length_stats.to_excel(writer, sheet_name='길이통계')

    print("\n분석 완료! 결과가 'posts_analysis.xlsx'에 저장되었습니다.")

    return df, user_analysis

# 실행
df, analysis = collect_and_analyze_posts()
