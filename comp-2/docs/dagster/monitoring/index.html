9. 모니터링

Dagster 파이프라인의 모니터링, 로깅, 알림 시스템을 구축하는 방법을 학습합니다. 데이터 품질 체크와 성능 모니터링도 다룹니다.

9.1 Dagit UI​

9.1.1 Dagit UI 개요​

Dagit은 Dagster의 웹 기반 UI로, 파이프라인을 시각화하고 모니터링하는 강력한 도구입니다.

"""
Dagit UI 주요 기능:

1. Asset Catalog
   - 모든 Asset의 현재 상태 확인
   - 의존성 그래프 시각화
   - Materialization 히스토리

2. Runs
   - 실행 중인 Run 모니터링
   - 과거 Run 히스토리 조회
   - 실패한 Run 재실행

3. Schedules & Sensors
   - 스케줄 상태 확인
   - 센서 평가 로그
   - 수동 실행 트리거

4. Logs
   - 구조화된 로그 뷰어
   - 로그 필터링 및 검색
   - 에러 추적

5. Lineage
   - 데이터 계보 추적
   - Asset 간 의존성 분석
   - 임팩트 분석
"""

메타데이터가 풍부한 Asset 정의

from dagster import (
    asset,
    AssetExecutionContext,
    MetadataValue,
    Output
)
import pandas as pd
from datetime import datetime

@asset(
    description="고객 데이터를 로드하고 기본 통계를 계산합니다.",
    metadata={
        "owner": "data-team@company.com",
        "sla": "daily by 9am",
        "category": "raw_data"
    }
)
def customers(context: AssetExecutionContext) -> Output[pd.DataFrame]:
    """고객 데이터 Asset"""
    # 데이터 로드
    df = pd.DataFrame({
        "customer_id": range(1, 101),
        "name": [f"Customer {i}" for i in range(1, 101)],
        "total_spent": [100 * i for i in range(1, 101)]
    })

    # 통계 계산
    stats = {
        "row_count": len(df),
        "total_revenue": df["total_spent"].sum(),
        "avg_spent": df["total_spent"].mean(),
        "max_spent": df["total_spent"].max()
    }

    # 메타데이터와 함께 반환
    return Output(
        df,
        metadata={
            "row_count": MetadataValue.int(stats["row_count"]),
            "total_revenue": MetadataValue.float(stats["total_revenue"]),
            "avg_spent": MetadataValue.float(stats["avg_spent"]),
            "preview": MetadataValue.md(df.head(5).to_markdown()),
            "schema": MetadataValue.md(
                "\n".join([f"- {col}: {dtype}" for col, dtype in df.dtypes.items()])
            ),
            "last_updated": MetadataValue.text(datetime.now().isoformat())
        }
    )

@asset(
    description="고객 세그먼트 분석",
    metadata={
        "owner": "analytics-team@company.com",
        "category": "analytics"
    }
)
def customer_segments(
    context: AssetExecutionContext,
    customers: pd.DataFrame
) -> Output[pd.DataFrame]:
    """고객 세그먼트 Asset"""
    df = customers.copy()

    # 세그먼트 분류
    df["segment"] = pd.cut(
        df["total_spent"],
        bins=[0, 3000, 7000, float('inf')],
        labels=["Basic", "Standard", "Premium"]
    )

    # 세그먼트별 통계
    segment_stats = df.groupby("segment").agg({
        "customer_id": "count",
        "total_spent": ["sum", "mean"]
    }).round(2)

    return Output(
        df,
        metadata={
            "row_count": MetadataValue.int(len(df)),
            "segment_distribution": MetadataValue.md(
                segment_stats.to_markdown()
            ),
            "segments": MetadataValue.json({
                "Basic": len(df[df["segment"] == "Basic"]),
                "Standard": len(df[df["segment"] == "Standard"]),
                "Premium": len(df[df["segment"] == "Premium"])
            })
        }
    )

# run_monitoring.py
"""
Run 상태 모니터링 및 관리
"""
from dagster import (
    DagsterInstance,
    DagsterRunStatus,
    RunsFilter,
    DagsterEventType
)
from datetime import datetime, timedelta
from typing import List, Dict, Any

class RunMonitor:
    """Run 모니터링 유틸리티"""

    def __init__(self):
        self.instance = DagsterInstance.get()

    def get_recent_runs(self, hours: int = 24) -> List[Dict[str, Any]]:
        """최근 실행된 Run 조회"""
        cutoff = datetime.now() - timedelta(hours=hours)

        runs = self.instance.get_runs(
            filters=RunsFilter(
                created_after=cutoff
            ),
            limit=100
        )

        return [
            {
                "run_id": run.run_id,
                "status": run.status.value,
                "job_name": run.job_name,
                "created_at": run.create_time,
                "updated_at": run.update_time,
                "duration_seconds": (
                    (run.end_time - run.start_time).total_seconds()
                    if run.end_time and run.start_time else None
                )
            }
            for run in runs
        ]

    def get_failed_runs(self, hours: int = 24) -> List[Dict[str, Any]]:
        """실패한 Run 조회"""
        cutoff = datetime.now() - timedelta(hours=hours)

        runs = self.instance.get_runs(
            filters=RunsFilter(
                statuses=[DagsterRunStatus.FAILURE],
                created_after=cutoff
            )
        )

        failed_runs = []
        for run in runs:
            # 에러 메시지 추출
            events = self.instance.all_logs(run.run_id)
            error_events = [
                event for event in events
                if event.dagster_event_type == DagsterEventType.STEP_FAILURE
            ]

            error_messages = [
                event.message for event in error_events
            ]

            failed_runs.append({
                "run_id": run.run_id,
                "job_name": run.job_name,
                "created_at": run.create_time,
                "errors": error_messages
            })

        return failed_runs

    def get_success_rate(self, hours: int = 24) -> Dict[str, float]:
        """성공률 계산"""
        cutoff = datetime.now() - timedelta(hours=hours)

        all_runs = self.instance.get_runs(
            filters=RunsFilter(created_after=cutoff),
            limit=1000
        )

        if not all_runs:
            return {"success_rate": 0.0, "total_runs": 0}

        success_count = sum(
            1 for run in all_runs
            if run.status == DagsterRunStatus.SUCCESS
        )

        return {
            "success_rate": (success_count / len(all_runs)) * 100,
            "total_runs": len(all_runs),
            "successful_runs": success_count,
            "failed_runs": sum(
                1 for run in all_runs
                if run.status == DagsterRunStatus.FAILURE
            )
        }

    def get_average_duration(self, job_name: str, days: int = 7) -> float:
        """Job의 평균 실행 시간 계산"""
        cutoff = datetime.now() - timedelta(days=days)

        runs = self.instance.get_runs(
            filters=RunsFilter(
                job_name=job_name,
                statuses=[DagsterRunStatus.SUCCESS],
                created_after=cutoff
            )
        )

        durations = [
            (run.end_time - run.start_time).total_seconds()
            for run in runs
            if run.end_time and run.start_time
        ]

        if not durations:
            return 0.0

        return sum(durations) / len(durations)

# 사용 예제
if __name__ == "__main__":
    monitor = RunMonitor()

    print("=== 최근 24시간 Run 통계 ===")
    stats = monitor.get_success_rate(hours=24)
    print(f"총 Run: {stats['total_runs']}")
    print(f"성공: {stats['successful_runs']}")
    print(f"실패: {stats['failed_runs']}")
    print(f"성공률: {stats['success_rate']:.2f}%")

    print("\n=== 실패한 Run ===")
    failed = monitor.get_failed_runs(hours=24)
    for run in failed[:5]:  # 최근 5개만 출력
        print(f"\nRun ID: {run['run_id']}")
        print(f"Job: {run['job_name']}")
        print(f"시간: {run['created_at']}")
        if run['errors']:
            print(f"에러: {run['errors'][0][:100]}...")

        "Data sample",
        extra={"sample": df.head(2).to_dict()}
    )

    # 완료 로그
    context.log.info(
        "Processing completed",
        extra={
            "row_count": len(df),
            "column_count": len(df.columns),
            "memory_usage_mb": df.memory_usage(deep=True).sum() / 1024 / 1024
        }
    )

    return df

@op

                "total": total_items,
                "progress_percent": progress
            }

# custom_logger.py
"""
커스텀 로거 구현
"""
import logging
import json
from datetime import datetime
from typing import Any, Dict
from dagster import ConfigurableResource

class StructuredLogger(ConfigurableResource):
    """구조화된 JSON 로거"""
    log_level: str = "INFO"
    service_name: str = "dagster-pipeline"

    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.logger = logging.getLogger(self.service_name)
        self.logger.setLevel(getattr(logging, self.log_level))

        # JSON 포맷 핸들러
        handler = logging.StreamHandler()
        handler.setFormatter(JsonFormatter())
        self.logger.addHandler(handler)

    def log(
        self,
        level: str,
        message: str,
        **kwargs: Any
    ):
        """로그 메시지 작성"""
        log_data = {
            "timestamp": datetime.now().isoformat(),
            "service": self.service_name,
            "level": level,
            "message": message,
            **kwargs
        }

        log_func = getattr(self.logger, level.lower())
        log_func(json.dumps(log_data))

class JsonFormatter(logging.Formatter):
    """JSON 포맷 로거"""

    def format(self, record: logging.LogRecord) -> str:
        """로그 레코드를 JSON으로 포맷"""
        log_data = {
            "timestamp": datetime.fromtimestamp(record.created).isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
        }

        # 추가 필드
        if hasattr(record, "extra"):
            log_data.update(record.extra)

        return json.dumps(log_data)

# 사용 예제
@asset
def asset_with_custom_logger(
    context: AssetExecutionContext,
    logger: StructuredLogger
) -> pd.DataFrame:
    """커스텀 로거를 사용하는 Asset"""
    logger.log(
        "info",
        "Starting asset materialization",
        asset_name="asset_with_custom_logger",
        run_id=context.run_id
    )

    df = pd.DataFrame({"id": [1, 2, 3]})

    logger.log(
        "info",
        "Asset materialization completed",
        asset_name="asset_with_custom_logger",
        row_count=len(df),
        run_id=context.run_id
    )

    return df

# log_analysis.py
"""
로그 집계 및 분석
"""
from dagster import DagsterInstance, DagsterEventType
from datetime import datetime, timedelta
from typing import List, Dict, Any
from collections import Counter
import re

class LogAnalyzer:
    """로그 분석 유틸리티"""

    def __init__(self):
        self.instance = DagsterInstance.get()

    def analyze_run_logs(self, run_id: str) -> Dict[str, Any]:
        """Run의 로그 분석"""
        events = self.instance.all_logs(run_id)

        # 로그 레벨별 카운트
        log_levels = Counter()
        error_messages = []
        warning_messages = []

        for event in events:
            level = event.level
            log_levels[level] += 1

            if level == "ERROR":
                error_messages.append(event.message)
            elif level == "WARNING":
                warning_messages.append(event.message)

        return {
            "total_logs": len(events),
            "log_levels": dict(log_levels),
            "error_count": log_levels["ERROR"],
            "warning_count": log_levels["WARNING"],
            "errors": error_messages[:10],  # 최근 10개
            "warnings": warning_messages[:10]
        }

    def find_common_errors(self, hours: int = 24) -> List[Dict[str, Any]]:
        """공통 에러 패턴 찾기"""
        cutoff = datetime.now() - timedelta(hours=hours)

        runs = self.instance.get_runs(limit=100)
        error_patterns = Counter()

        for run in runs:
            if run.create_time < cutoff:
                continue

            events = self.instance.all_logs(run.run_id)
            error_events = [
                e for e in events
                if e.dagster_event_type == DagsterEventType.STEP_FAILURE
            ]

            for event in error_events:
                # 에러 메시지에서 패턴 추출 (간단한 예시)
                message = event.message
                # 숫자, ID 등을 제거하여 패턴화
                pattern = re.sub(r'\d+', 'N', message)
                pattern = re.sub(r'[a-f0-9]{8}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{12}', 'UUID', pattern)
                error_patterns[pattern] += 1

        # 가장 흔한 에러 패턴
        return [
            {"pattern": pattern, "count": count}
            for pattern, count in error_patterns.most_common(10)
        ]

    def get_performance_metrics(self, run_id: str) -> Dict[str, Any]:
        """성능 메트릭 추출"""
        events = self.instance.all_logs(run_id)

        step_durations = {}
        step_start_times = {}

        for event in events:
            if not event.dagster_event:
                continue

            event_type = event.dagster_event_type
            step_key = event.step_key

            if event_type == DagsterEventType.STEP_START:
                step_start_times[step_key] = event.timestamp

            elif event_type == DagsterEventType.STEP_SUCCESS:
                if step_key in step_start_times:
                    duration = event.timestamp - step_start_times[step_key]
                    step_durations[step_key] = duration

        return {
            "step_count": len(step_durations),
            "total_duration": sum(step_durations.values()),
            "avg_duration": (
                sum(step_durations.values()) / len(step_durations)
                if step_durations else 0
            ),
            "slowest_steps": sorted(
                step_durations.items(),
                key=lambda x: x[1],
                reverse=True
            )[:5]
        }

"""
Prometheus 메트릭 수집
"""
from prometheus_client import (
    Counter,
    Gauge,
    Histogram,
    Summary,
    start_http_server
)
from dagster import asset, AssetExecutionContext, op, OpExecutionContext
import time
from functools import wraps

# 메트릭 정의
asset_materializations = Counter(
    'dagster_asset_materializations_total',
    'Total number of asset materializations',
    ['asset_name', 'status']
)

asset_duration = Histogram(
    'dagster_asset_duration_seconds',
    'Asset materialization duration',
    ['asset_name'],
    buckets=[1, 5, 10, 30, 60, 120, 300, 600, 1800, 3600]
)

asset_row_count = Gauge(
    'dagster_asset_row_count',
    'Number of rows in asset',
    ['asset_name']
)

op_executions = Counter(
    'dagster_op_executions_total',
    'Total number of op executions',
    ['op_name', 'status']
)

class MetricsCollector:
    """메트릭 수집 유틸리티"""

    @staticmethod
    def record_asset_metrics(asset_name: str):
        """Asset 메트릭 기록 데코레이터"""
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                start_time = time.time()
                status = "success"

                try:
                    result = func(*args, **kwargs)

                    # 결과가 DataFrame인 경우 행 수 기록
                    if hasattr(result, '__len__'):
                        asset_row_count.labels(asset_name=asset_name).set(len(result))

                    return result

                except Exception as e:
                    status = "failure"
                    raise

                finally:
                    # 실행 시간 기록
                    duration = time.time() - start_time
                    asset_duration.labels(asset_name=asset_name).observe(duration)

                    # 실행 카운트
                    asset_materializations.labels(
                        asset_name=asset_name,
                        status=status
                    ).inc()

            return wrapper
        return decorator

# 사용 예제

    """고객 데이터 Asset (메트릭 포함)"""

    df = pd.DataFrame({
        "customer_id": range(1, 1001),
        "name": [f"Customer {i}" for i in range(1, 1001)]
    })

    return df

# custom_metrics.py
"""
커스텀 메트릭 정의 및 수집
"""
from dagster import (
    asset,
    AssetExecutionContext,
    Output,
    MetadataValue,
    AssetObservation
)
import pandas as pd
from typing import Dict, Any
from datetime import datetime

class DataQualityMetrics:
    """데이터 품질 메트릭"""

    @staticmethod
    def calculate_completeness(df: pd.DataFrame) -> float:
        """완전성: 결측치가 없는 비율"""
        total_cells = df.size
        non_null_cells = df.count().sum()
        return (non_null_cells / total_cells) * 100 if total_cells > 0 else 0

    @staticmethod
    def calculate_uniqueness(df: pd.DataFrame, key_columns: list) -> float:
        """유일성: 중복이 없는 비율"""
        total_rows = len(df)
        unique_rows = len(df.drop_duplicates(subset=key_columns))
        return (unique_rows / total_rows) * 100 if total_rows > 0 else 0

    @staticmethod
    def calculate_validity(
        df: pd.DataFrame,
        rules: Dict[str, callable]
    ) -> Dict[str, float]:
        """유효성: 규칙을 만족하는 비율"""
        validity_scores = {}

        for column, rule in rules.items():
            if column in df.columns:
                valid_count = df[column].apply(rule).sum()
                validity_scores[column] = (valid_count / len(df)) * 100

        return validity_scores

@asset
def quality_checked_data(
    context: AssetExecutionContext
) -> Output[pd.DataFrame]:
    """데이터 품질 체크가 포함된 Asset"""
    # 데이터 로드
    df = pd.DataFrame({
        "id": [1, 2, 3, 4, 5],
        "email": [
            "user1@example.com",
            "user2@example.com",
            None,
            "invalid_email",
            "user5@example.com"
        ],
        "age": [25, 30, 35, -5, 40]  # -5는 유효하지 않음
    })

    # 품질 메트릭 계산
    metrics = DataQualityMetrics()

    completeness = metrics.calculate_completeness(df)
    uniqueness = metrics.calculate_uniqueness(df, ["id"])

    # 유효성 규칙
    validity_rules = {
        "email": lambda x: pd.notna(x) and "@" in str(x),
        "age": lambda x: x > 0 if pd.notna(x) else False
    }
    validity = metrics.calculate_validity(df, validity_rules)

    # 전체 품질 점수
    quality_score = (
        completeness * 0.4 +
        uniqueness * 0.3 +
        sum(validity.values()) / len(validity) * 0.3
    )

    # 품질 이슈 검사
    issues = []
    if completeness < 95:
        issues.append(f"Low completeness: {completeness:.1f}%")
    if validity["email"] < 80:
        issues.append(f"Invalid emails detected: {100 - validity['email']:.1f}%")
    if validity["age"] < 100:
        issues.append(f"Invalid ages detected: {100 - validity['age']:.1f}%")

    # 메타데이터로 메트릭 반환
    return Output(
        df,
        metadata={
            "quality_score": MetadataValue.float(quality_score),
            "completeness": MetadataValue.float(completeness),
            "uniqueness": MetadataValue.float(uniqueness),
            "email_validity": MetadataValue.float(validity["email"]),
            "age_validity": MetadataValue.float(validity["age"]),
            "issues": MetadataValue.md(
                "\n".join([f"- {issue}" for issue in issues]) if issues
                else "No issues detected"
            ),
            "timestamp": MetadataValue.text(datetime.now().isoformat())
        }
    )

def monitored_asset(context: AssetExecutionContext) -> pd.DataFrame:

class EmailNotifier:
    """Email 알림 전송"""

    def __init__(
        self,
        smtp_server: str = None,
        smtp_port: int = 587,
        username: str = None,
        password: str = None
    ):
        self.smtp_server = smtp_server or os.getenv("SMTP_SERVER", "smtp.gmail.com")
        self.smtp_port = smtp_port
        self.username = username or os.getenv("SMTP_USERNAME")
        self.password = password or os.getenv("SMTP_PASSWORD")

    def send_email(
        self,
        to_emails: List[str],
        subject: str,
        body: str,
        html: bool = False
    ) -> bool:
        """Email 전송"""
        try:
            msg = MIMEMultipart("alternative")
            msg["From"] = self.username
            msg["To"] = ", ".join(to_emails)
            msg["Subject"] = subject

            # 본문 추가
            mime_type = "html" if html else "plain"
            msg.attach(MIMEText(body, mime_type))

            # SMTP 서버 연결 및 전송
            with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
                server.starttls()
                server.login(self.username, self.password)
                server.send_message(msg)

            return True

        except Exception as e:
            print(f"Email send error: {e}")
            return False

    def send_failure_alert(
        self,
        to_emails: List[str],
        run_id: str,
        job_name: str,
        error_message: str
    ) -> bool:
        """실패 알림 이메일"""

            <p><strong>Job Name:</strong> {job_name}</p>
            <p><strong>Run ID:</strong> <code>{run_id}</code></p>

            <h3>Error Details:</h3>
            <pre style="background-color: #f5f5f5; padding: 10px; border-radius: 4px;">
{error_message}
            </pre>

                <a href="http://dagster-ui/runs/{run_id}"

                          text-decoration: none; border-radius: 4px;">

        """

        self,
        summary: str,
        severity: str = "error",
        source: str = "dagster",

                "custom_details": custom_details or {}

            response.raise_for_status()

            print(f"PagerDuty API error: {e}")
            return False

from dagster import (
    asset,
    AssetExecutionContext,
    Output,
    AssetCheckSpec,
    AssetCheckResult,
    AssetCheckSeverity
)
import pandas as pd
from typing import List, Callable

class DataQualityCheck:
    """데이터 품질 체크 클래스"""

    @staticmethod
    def check_no_nulls(df: pd.DataFrame, columns: List[str]) -> AssetCheckResult:
        """결측치 체크"""
        null_counts = df[columns].isnull().sum()
        has_nulls = null_counts.sum() > 0

        return AssetCheckResult(
            passed=not has_nulls,
            metadata={
                "null_counts": null_counts.to_dict(),
                "total_nulls": int(null_counts.sum())
            },
            severity=AssetCheckSeverity.ERROR if has_nulls else None
        )

    @staticmethod
    def check_unique(df: pd.DataFrame, columns: List[str]) -> AssetCheckResult:
        """유일성 체크"""
        duplicate_count = df.duplicated(subset=columns).sum()
        is_unique = duplicate_count == 0

        return AssetCheckResult(
            passed=is_unique,
            metadata={
                "duplicate_count": int(duplicate_count),
                "unique_ratio": float(1 - duplicate_count / len(df))
            },
            severity=AssetCheckSeverity.WARN if not is_unique else None
        )

    @staticmethod
    def check_value_range(
        df: pd.DataFrame,
        column: str,
        min_value: float,
        max_value: float
    ) -> AssetCheckResult:
        """값 범위 체크"""
        out_of_range = df[
            (df[column] < min_value) | (df[column] > max_value)
        ]
        in_range = len(out_of_range) == 0

        return AssetCheckResult(
            passed=in_range,
            metadata={
                "out_of_range_count": len(out_of_range),
                "min_found": float(df[column].min()),
                "max_found": float(df[column].max()),
                "expected_range": f"[{min_value}, {max_value}]"
            },
            severity=AssetCheckSeverity.ERROR if not in_range else None
        )

    @staticmethod
    def check_freshness(
        df: pd.DataFrame,
        date_column: str,
        max_age_hours: int
    ) -> AssetCheckResult:
        """데이터 신선도 체크"""
        from datetime import datetime, timedelta

        latest_date = pd.to_datetime(df[date_column]).max()
        age = datetime.now() - latest_date
        is_fresh = age < timedelta(hours=max_age_hours)

        return AssetCheckResult(
            passed=is_fresh,
            metadata={
                "latest_date": latest_date.isoformat(),
                "age_hours": float(age.total_seconds() / 3600),
                "max_age_hours": max_age_hours
            },
            severity=AssetCheckSeverity.WARN if not is_fresh else None
        )

# performance/profiling.py
"""
성능 프로파일링
"""
from dagster import asset, AssetExecutionContext, Output, MetadataValue
import pandas as pd
import time
from functools import wraps
from typing import Callable, Any
import psutil
import os

class PerformanceProfiler:
    """성능 프로파일러"""

    @staticmethod
    def profile_execution(func: Callable) -> Callable:
        """실행 시간 및 리소스 사용량 프로파일링"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 시작 메트릭
            start_time = time.time()
            process = psutil.Process(os.getpid())
            start_memory = process.memory_info().rss / 1024 / 1024  # MB
            start_cpu = process.cpu_percent()

            # 함수 실행
            result = func(*args, **kwargs)

            # 종료 메트릭
            end_time = time.time()
            end_memory = process.memory_info().rss / 1024 / 1024  # MB
            end_cpu = process.cpu_percent()

            # 프로파일 정보
            profile = {
                "duration_seconds": end_time - start_time,
                "memory_used_mb": end_memory - start_memory,
                "peak_memory_mb": end_memory,
                "cpu_percent": end_cpu,
            }

            # 결과에 프로파일 정보 추가
            if isinstance(result, Output):
                result.metadata.update({
                    "execution_time": MetadataValue.float(profile["duration_seconds"]),
                    "memory_usage_mb": MetadataValue.float(profile["memory_used_mb"]),
                    "cpu_percent": MetadataValue.float(profile["cpu_percent"])
                })
                return result
            else:
                return result, profile

        return wrapper

        "value": range(100000)
    })

    # 복잡한 변환
    df["squared"] = df["value"] ** 2
    df["cubed"] = df["value"] ** 3

    # 집계
    result = df.groupby(df["id"] % 100).agg({
        "value": ["sum", "mean", "std"],

"""

"""
from dagster import ConfigurableResource, asset, AssetExecutionContext
import time
from typing import Any, Dict
import pandas as pd

class MonitoredDatabase(ConfigurableResource):
    """쿼리 성능 모니터링이 포함된 데이터베이스"""
    connection_string: str

    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.query_stats: Dict[str, list] = {
            "query": [],
            "duration": [],
            "rows_returned": []
        }

    def execute_query(self, query: str) -> pd.DataFrame:
        """쿼리 실행 및 성능 모니터링"""
        start_time = time.time()

        # 실제 쿼리 실행 (시뮬레이션)
        result = pd.DataFrame({"col1": range(100), "col2": range(100)})

        duration = time.time() - start_time

        # 통계 기록
        self.query_stats["query"].append(query[:50])  # 처음 50자만
        self.query_stats["duration"].append(duration)
        self.query_stats["rows_returned"].append(len(result))

        return result

    def get_performance_report(self) -> Dict[str, Any]:
        """성능 리포트 생성"""
        if not self.query_stats["query"]:
            return {"message": "No queries executed"}

        return {
            "total_queries": len(self.query_stats["query"]),
            "total_duration": sum(self.query_stats["duration"]),
            "avg_duration": sum(self.query_stats["duration"]) / len(self.query_stats["duration"]),
            "slowest_query": {
                "query": self.query_stats["query"][
                    self.query_stats["duration"].index(max(self.query_stats["duration"]))
                ],
                "duration": max(self.query_stats["duration"])
            },
            "total_rows": sum(self.query_stats["rows_returned"])
        }

@asset
def query_performance_example(
    context: AssetExecutionContext,
    db: MonitoredDatabase
) -> pd.DataFrame:
    """쿼리 성능 모니터링 예제"""
    # 여러 쿼리 실행
    queries = [
        "SELECT * FROM customers",
        "SELECT * FROM orders WHERE date > '2024-01-01'",
        "SELECT customer_id, SUM(amount) FROM orders GROUP BY customer_id"
    ]

    results = []
    for query in queries:
        result = db.execute_query(query)
        results.append(result)

    # 성능 리포트
    report = db.get_performance_report()
    context.log.info(f"Query performance: {report}")

    return pd.concat(results, ignore_index=True)

    from run_monitoring import RunMonitor

    monitor = RunMonitor()

    # 지난 24시간 통계
    success_rate = monitor.get_success_rate(hours=24)
    recent_runs = monitor.get_recent_runs(hours=24)
    failed_runs = monitor.get_failed_runs(hours=24)

    metrics = {
        "timestamp": datetime.now().isoformat(),
        "success_rate": success_rate,
        "recent_runs_count": len(recent_runs),
        "failed_runs_count": len(failed_runs),
        "avg_duration": (
            sum(r["duration_seconds"] for r in recent_runs if r["duration_seconds"])
            / len(recent_runs) if recent_runs else 0
        )
    }

    context.log.info(f"Monitoring metrics: {metrics}")

    return metrics

            "uniqueness": 98.0 + (hash(asset_name) % 2),
            "validity": 90.0 + (hash(asset_name) % 10),
            "timestamp": datetime.now()
        })

    df = pd.DataFrame(quality_data)

    context.log.info(f"Quality report generated for {len(df)} assets")

    return df

@asset

            "avg_duration_seconds": avg_duration,
            "avg_duration_minutes": avg_duration / 60,
            "timestamp": datetime.now()
        })

    df = pd.DataFrame(performance_data)

    context.log.info(f"Performance report generated for {len(df)} jobs")

    return df

    • Email 알림
