5. I/O Managers

Asset과 Op의 입출력 데이터를 저장하고 로드하는 방법을 배워봅시다.

5.1 I/O Manager 개념​

I/O Manager는 Dagster에서 데이터의 저장과 로드를 담당하는 리소스입니다.

**역할:**
- Asset/Op 출력을 저장 (handle_output)
- Asset/Op 입력을 로드 (load_input)
- 데이터 직렬화/역직렬화
- 저장 위치 관리

장점:

  • 저장 로직과 비즈니스 로직 분리
  • 환경별 저장 전략 변경 용이
  • 데이터 형식 표준화

5.2 기본 I/O Manager​

from dagster import IOManager, io_manager
import pickle
from pathlib import Path

class MyIOManager(IOManager):
    def __init__(self, base_dir: str):
        self.base_dir = Path(base_dir)
        self.base_dir.mkdir(exist_ok=True)

    def handle_output(self, context, obj):
        """출력 저장"""
        file_path = self._get_path(context)
        with open(file_path, "wb") as f:
            pickle.dump(obj, f)

        context.log.info(f"Saved to {file_path}")

    def load_input(self, context):
        """입력 로드"""
        file_path = self._get_path(context)
        with open(file_path, "rb") as f:
            return pickle.load(f)

    def _get_path(self, context) -> Path:
        """파일 경로 생성"""
        return self.base_dir / f"{context.asset_key.path[-1]}.pkl"

@io_manager
def my_io_manager():
    return MyIOManager(base_dir="data/outputs")

5.3.1 Pickle I/O Manager​

from dagster import IOManager, io_manager, Definitions
import pickle
import os

class PickleIOManager(IOManager):
    def __init__(self, base_dir: str):
        self.base_dir = base_dir
        os.makedirs(base_dir, exist_ok=True)

    def handle_output(self, context, obj):
        path = self._get_path(context)
        with open(path, "wb") as f:
            pickle.dump(obj, f)

    def load_input(self, context):
        path = self._get_path(context)
        with open(path, "rb") as f:
            return pickle.load(f)

    def _get_path(self, context):
        return os.path.join(
            self.base_dir,
            f"{context.asset_key.path[-1]}.pkl"
        )

@io_manager
def pickle_io_manager():
    return PickleIOManager("data/pickle")

# 사용
from dagster import asset

@asset
def my_data() -> list:
    return [1, 2, 3, 4, 5]

@asset
def processed_data(my_data: list) -> list:
    return [x * 2 for x in my_data]

defs = Definitions(
    assets=[my_data, processed_data],
    resources={"io_manager": pickle_io_manager}
)

5.3.2 JSON I/O Manager​

import json

class JsonIOManager(IOManager):
    def __init__(self, base_dir: str):
        self.base_dir = base_dir
        os.makedirs(base_dir, exist_ok=True)

    def handle_output(self, context, obj):

        return os.path.join(
            self.base_dir,
            f"{context.asset_key.path[-1]}.json"
        )

@io_manager
def json_io_manager():
    return JsonIOManager("data/json")

5.4 Parquet I/O Manager​

from dagster import IOManager, io_manager
import pandas as pd
import pyarrow.parquet as pq
import pyarrow as pa

class ParquetIOManager(IOManager):
    def __init__(self, base_dir: str):
        self.base_dir = base_dir
        os.makedirs(base_dir, exist_ok=True)

    def handle_output(self, context, obj: pd.DataFrame):
        path = self._get_path(context)

        # Parquet로 저장
        table = pa.Table.from_pandas(obj)
        pq.write_table(table, path, compression='snappy')

        context.log.info(f"Saved {len(obj)} rows to {path}")

        # 메타데이터 기록
        context.add_output_metadata({
            "num_rows": len(obj),
            "num_columns": len(obj.columns),
            "file_size": os.path.getsize(path)
        })

    def load_input(self, context) -> pd.DataFrame:
        path = self._get_path(context)
        return pd.read_parquet(path)

    def _get_path(self, context):
        return os.path.join(
            self.base_dir,
            f"{context.asset_key.path[-1]}.parquet"
        )

@io_manager
def parquet_io_manager():
    return ParquetIOManager("data/parquet")

5.5 S3 I/O Manager​

from dagster import IOManager, io_manager
import boto3
import pickle
from io import BytesIO

class S3IOManager(IOManager):
    def __init__(self, bucket: str, prefix: str = ""):
        self.s3_client = boto3.client('s3')
        self.bucket = bucket
        self.prefix = prefix

    def handle_output(self, context, obj):
        key = self._get_key(context)

        # 직렬화
        buffer = BytesIO()
        pickle.dump(obj, buffer)
        buffer.seek(0)

        # S3 업로드
        self.s3_client.put_object(
            Bucket=self.bucket,
            Key=key,
            Body=buffer.getvalue()
        )

        context.log.info(f"Saved to s3://{self.bucket}/{key}")

    def load_input(self, context):
        key = self._get_key(context)

        # S3 다운로드
        response = self.s3_client.get_object(
            Bucket=self.bucket,
            Key=key
        )

        # 역직렬화
        return pickle.loads(response['Body'].read())

    def _get_key(self, context):
        parts = [self.prefix] if self.prefix else []
        parts.append(f"{context.asset_key.path[-1]}.pkl")
        return "/".join(parts)

@io_manager(config_schema={"bucket": str, "prefix": str})
def s3_io_manager(context):
    return S3IOManager(
        bucket=context.resource_config["bucket"],
        prefix=context.resource_config.get("prefix", "")
    )

5.6 Database I/O Manager​

from dagster import IOManager, io_manager
import pandas as pd
from sqlalchemy import create_engine

class PostgresIOManager(IOManager):
    def __init__(self, connection_string: str):
        self.engine = create_engine(connection_string)

    def handle_output(self, context, obj: pd.DataFrame):
        table_name = context.asset_key.path[-1]

        # DataFrame을 PostgreSQL에 저장
        obj.to_sql(
            table_name,
            self.engine,
            if_exists='replace',
            index=False
        )

        context.log.info(f"Saved {len(obj)} rows to {table_name}")

        context.add_output_metadata({
            "num_rows": len(obj),
            "table_name": table_name
        })

    def load_input(self, context) -> pd.DataFrame:
        table_name = context.upstream_output.asset_key.path[-1]

        # PostgreSQL에서 로드
        query = f"SELECT * FROM {table_name}"
        return pd.read_sql(query, self.engine)

@io_manager(config_schema={"connection_string": str})
def postgres_io_manager(context):
    return PostgresIOManager(
        context.resource_config["connection_string"]
    )

5.7 Per-asset I/O Manager​

Asset별로 다른 I/O Manager를 사용할 수 있습니다.

from dagster import asset, Definitions, fs_io_manager

# Pickle 사용
@asset(io_manager_key="pickle_io_manager")
def raw_data() -> dict:
    return {"value": 100}

# Parquet 사용
@asset(io_manager_key="parquet_io_manager")
def dataframe_asset() -> pd.DataFrame:
    return pd.DataFrame({"a": [1, 2, 3]})

# S3 사용
@asset(io_manager_key="s3_io_manager")
def cloud_asset() -> dict:
    return {"cloud_data": "stored in S3"}

defs = Definitions(
    assets=[raw_data, dataframe_asset, cloud_asset],
    resources={
        "pickle_io_manager": pickle_io_manager,
        "parquet_io_manager": parquet_io_manager,
        "s3_io_manager": s3_io_manager
    }
)

5.8 Partitioned I/O Manager​

class PartitionedParquetIOManager(IOManager):
    def __init__(self, base_dir: str):
        self.base_dir = base_dir

    def handle_output(self, context, obj: pd.DataFrame):
        # Partition 정보 가져오기
        if context.has_partition_key:
            partition_key = context.partition_key
        else:
            partition_key = "default"

        # Partition별로 저장
        partition_dir = os.path.join(
            self.base_dir,
            context.asset_key.path[-1],
            f"partition={partition_key}"
        )
        os.makedirs(partition_dir, exist_ok=True)

        file_path = os.path.join(partition_dir, "data.parquet")
        obj.to_parquet(file_path)

        context.log.info(
            f"Saved partition {partition_key} to {file_path}"
        )

    def load_input(self, context) -> pd.DataFrame:
        if context.has_partition_key:
            partition_key = context.partition_key
        else:
            partition_key = "default"

        file_path = os.path.join(
            self.base_dir,
            context.upstream_output.asset_key.path[-1],
            f"partition={partition_key}",
            "data.parquet"
        )

        return pd.read_parquet(file_path)

@io_manager
def partitioned_parquet_io_manager():
    return PartitionedParquetIOManager("data/partitioned")

5.9 실습 문제​

문제: CSV I/O Manager 구현​

CSV 파일로 DataFrame을 저장하고 로드하는 I/O Manager를 구현하세요.

요구사항:

  1. pandas DataFrame을 CSV로 저장
  2. 인덱스 저장 여부 선택 가능
  3. 인코딩 설정 가능 (기본 utf-8)
  4. 메타데이터 기록 (행 수, 열 수, 파일 크기)
해답
from dagster import IOManager, io_manager, Definitions, asset
import pandas as pd
import os

class CsvIOManager(IOManager):
    def __init__(
        self,
        base_dir: str,
        save_index: bool = False,
        encoding: str = 'utf-8'
    ):
        self.base_dir = base_dir
        self.save_index = save_index
        self.encoding = encoding
        os.makedirs(base_dir, exist_ok=True)

    def handle_output(self, context, obj: pd.DataFrame):
        path = self._get_path(context)

        # CSV로 저장
        obj.to_csv(
            path,
            index=self.save_index,
            encoding=self.encoding
        )

        # 메타데이터 기록
        context.add_output_metadata({
            "num_rows": len(obj),
            "num_columns": len(obj.columns),
            "file_size_bytes": os.path.getsize(path),
            "columns": list(obj.columns),
            "file_path": path
        })

        context.log.info(
            f"Saved {len(obj)} rows to {path}"
        )

    def load_input(self, context) -> pd.DataFrame:
        path = self._get_path(context)

        # CSV 로드
        df = pd.read_csv(path, encoding=self.encoding)

        context.log.info(
            f"Loaded {len(df)} rows from {path}"
        )

        return df

    def _get_path(self, context):
        return os.path.join(
            self.base_dir,
            f"{context.asset_key.path[-1]}.csv"
        )

@io_manager(
    config_schema={
        "base_dir": str,
        "save_index": bool,
        "encoding": str
    }
)
def csv_io_manager(context):
    return CsvIOManager(
        base_dir=context.resource_config.get("base_dir", "data/csv"),
        save_index=context.resource_config.get("save_index", False),
        encoding=context.resource_config.get("encoding", "utf-8")
    )

# 테스트
@asset
def sales_data() -> pd.DataFrame:
    return pd.DataFrame({
        "product": ["A", "B", "C"],
        "sales": [100, 200, 150],
        "date": ["2024-01-01", "2024-01-02", "2024-01-03"]
    })

@asset
def total_sales(sales_data: pd.DataFrame) -> pd.DataFrame:
    return pd.DataFrame({
        "total": [sales_data["sales"].sum()]
    })

defs = Definitions(
    assets=[sales_data, total_sales],
    resources={"io_manager": csv_io_manager}
)
