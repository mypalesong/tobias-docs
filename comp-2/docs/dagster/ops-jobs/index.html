2. Ops와 Jobs

Dagster의 핵심 개념인 Ops와 Jobs에 대해 알아봅니다. Op은 데이터 파이프라인의 기본 작업 단위이며, Job은 여러 Op을 조합하여 완전한 파이프라인을 구성합니다.

2.1 Ops 개념​

Op이란?​

Op(Operation)은 Dagster에서 가장 기본적인 계산 단위입니다. 데이터를 로드하고, 변환하고, 저장하는 등의 개별 작업을 표현합니다.

Op의 주요 특징:

  • 순수 함수: Op은 가능한 한 순수 함수로 작성되어야 합니다
  • 재사용 가능: 여러 Job에서 동일한 Op을 재사용할 수 있습니다
  • 테스트 가능: 독립적으로 테스트할 수 있는 단위입니다
  • 타입 안전: 입력과 출력의 타입을 명시할 수 있습니다

Op의 구성 요소​

from dagster import op, In, Out

@op(
    name="my_op",  # Op의 이름
    description="This is a sample op",  # Op 설명
    ins={"input_1": In()},  # 입력 정의
    out=Out(),  # 출력 정의
    tags={"team": "data", "priority": "high"}  # 메타데이터
)
def my_op(input_1):
    """Op의 실제 로직"""
    result = input_1 * 2
    return result

기본 Op 예제​

from dagster import op, job

@op

    """데이터를 로드하는 Op"""

2.3 Job 정의​

Job이란?​

Job은 여러 Op을 조합하여 완전한 데이터 파이프라인을 정의합니다. Op들 간의 의존성을 명시하고 실행 순서를 결정합니다.

기본 Job 정의​

from dagster import job, op

@op
def step_one():
    return 1

@op
def step_two(value: int):
    return value + 1

@op
def step_three(value: int):
    return value * 2

@job
def my_job():
    """간단한 Job 정의"""
    result = step_one()
    result = step_two(result)
    step_three(result)

여러 Op을 연결하는 Job​

from dagster import job, op
import pandas as pd

@op
def extract_data():
    """데이터 추출"""
    return pd.DataFrame({
        'id': [1, 2, 3, 4, 5],
        'value': [10, 20, 30, 40, 50]
    })

@op
def transform_data(df: pd.DataFrame):
    """데이터 변환"""
    df['doubled'] = df['value'] * 2
    return df

@op
def load_data(df: pd.DataFrame):
    """데이터 로드"""
    print(f"Loading {len(df)} records")
    df.to_csv('output.csv', index=False)
    return len(df)

@job
def etl_pipeline():
    """ETL 파이프라인 Job"""
    raw_data = extract_data()
    transformed_data = transform_data(raw_data)
    load_data(transformed_data)

2.4 Ops 의존성​

명시적 의존성​

Op의 출력을 다른 Op의 입력으로 전달하여 의존성을 생성합니다.

from dagster import job, op

@op
def op_a():
    return "A의 결과"

@op
def op_b(a_result: str):
    return f"B가 {a_result}를 받음"

@op
def op_c(b_result: str):
    return f"C가 {b_result}를 받음"

@job
def linear_dependency_job():
    """선형 의존성: A -> B -> C"""
    a = op_a()
    b = op_b(a)
    c = op_c(b)

다중 의존성​

from dagster import job, op

@op
def source_1():
    return [1, 2, 3]

@op
def source_2():
    return [4, 5, 6]

@op
def source_3():
    return [7, 8, 9]

@op
def merge_sources(s1, s2, s3):
    """여러 소스를 병합"""
    return s1 + s2 + s3

@op
def process_merged(data):
    return sum(data)

@job
def multi_dependency_job():
    """여러 Op에 의존하는 패턴"""
    s1 = source_1()
    s2 = source_2()
    s3 = source_3()

    merged = merge_sources(s1, s2, s3)
    process_merged(merged)

팬아웃/팬인 패턴​

from dagster import job, op
from typing import Dict

@op
def load_raw_data():
    """원본 데이터 로드"""
    return {
        'users': [{'id': 1}, {'id': 2}],
        'orders': [{'id': 101}, {'id': 102}],
        'products': [{'id': 201}, {'id': 202}]
    }

@op
def process_users(data: Dict):
    """사용자 데이터 처리"""
    return len(data['users'])

@op
def process_orders(data: Dict):
    """주문 데이터 처리"""
    return len(data['orders'])

@op
def process_products(data: Dict):
    """제품 데이터 처리"""
    return len(data['products'])

@op
def create_summary(user_count: int, order_count: int, product_count: int):
    """처리 결과 요약"""
    return {
        'total_users': user_count,
        'total_orders': order_count,
        'total_products': product_count,
        'grand_total': user_count + order_count + product_count
    }

@job
def fan_out_fan_in_job():
    """팬아웃/팬인 패턴"""
    raw = load_raw_data()

    # 팬아웃: 하나의 Op 결과를 여러 Op이 사용
    users = process_users(raw)
    orders = process_orders(raw)
    products = process_products(raw)

    # 팬인: 여러 Op의 결과를 하나로 모음
    create_summary(users, orders, products)

Nothing 의존성​

데이터를 전달하지 않고 순서만 보장하는 의존성입니다.

from dagster import job, op, In, Nothing

@op
def create_table():
    """테이블 생성"""
    print("테이블 생성됨")

@op(ins={"start": In(Nothing)})
def insert_data():
    """데이터 삽입 (테이블 생성 후)"""
    print("데이터 삽입됨")
    return 100

@op(ins={"start": In(Nothing)})
def create_index(record_count: int):
    """인덱스 생성 (데이터 삽입 후)"""
    print(f"{record_count}개 레코드에 대한 인덱스 생성됨")

@job
def database_setup_job():
    """순차 실행을 보장하는 Job"""
    table_created = create_table()
    records = insert_data(start=table_created)
    create_index(start=records, record_count=records)

복잡한 의존성 그래프​

from dagster import job, op

@op
def ingest_source_a():
    return "Data from A"

@op
def ingest_source_b():
    return "Data from B"

@op
def validate_a(data: str):
    return f"Validated {data}"

@op
def validate_b(data: str):
    return f"Validated {data}"

@op
def transform_a(validated: str):
    return f"Transformed {validated}"

@op
def transform_b(validated: str):
    return f"Transformed {validated}"

@op
def merge_data(data_a: str, data_b: str):
    return f"{data_a} + {data_b}"

@op
def final_validation(merged: str):
    return f"Final: {merged}"

@op
def publish(validated: str):
    print(validated)

@job
def complex_pipeline():
    """복잡한 의존성 그래프"""
    # 소스 A 처리
    raw_a = ingest_source_a()
    validated_a = validate_a(raw_a)
    transformed_a = transform_a(validated_a)

    # 소스 B 처리
    raw_b = ingest_source_b()
    validated_b = validate_b(raw_b)
    transformed_b = transform_b(validated_b)

    # 병합 및 최종 처리
    merged = merge_data(transformed_a, transformed_b)
    final = final_validation(merged)
    publish(final)

Config Validation​

from dagster import op, job, OpExecutionContext, Config
from pydantic import Field, validator

class ValidatedConfig(Config):
    """유효성 검증이 있는 Config"""
    email: str = Field(pattern=r'^[\w\.-]+@[\w\.-]+\.\w+$')
    age: int = Field(ge=0, le=150, description="나이 (0-150)")
    score: float = Field(ge=0.0, le=100.0, description="점수 (0-100)")

    @validator('email')
    def validate_email_domain(cls, v):
        """이메일 도메인 검증"""
        if not v.endswith(('@example.com', '@test.com')):
            raise ValueError('허용되지 않은 이메일 도메인입니다')
        return v

    @validator('age')
    def validate_age_range(cls, v):
        """나이 범위 검증"""
        if v < 18:
            raise ValueError('18세 이상이어야 합니다')
        return v

@op
def process_validated_data(context: OpExecutionContext, config: ValidatedConfig):
    """검증된 Config를 사용하는 Op"""
    context.log.info(f"이메일: {config.email}")
    context.log.info(f"나이: {config.age}")
    context.log.info(f"점수: {config.score}")
    return True

@job
def validated_config_job():
    process_validated_data()

from dagster import op, job, DynamicOut, DynamicOutput, OpExecutionContext
import pandas as pd

@op(out=DynamicOut())
def partition_data(context: OpExecutionContext):
    """데이터를 동적으로 파티셔닝"""
    # 큰 데이터셋을 로드
    df = pd.DataFrame({
        'category': ['A', 'A', 'B', 'B', 'C', 'C'],
        'value': [1, 2, 3, 4, 5, 6]
    })

    # 카테고리별로 파티셔닝
    for category in df['category'].unique():
        partition = df[df['category'] == category]

        context.log.info(f"파티션 생성: {category} ({len(partition)} 행)")

        yield DynamicOutput(
            value=partition,
            mapping_key=f"category_{category}"
        )

@op
def process_partition(context: OpExecutionContext, partition: pd.DataFrame):
    """개별 파티션 처리"""
    category = partition['category'].iloc[0]
    total = partition['value'].sum()

    context.log.info(f"카테고리 {category} 처리: 합계 = {total}")

    return {
        'category': category,
        'count': len(partition),
        'sum': total
    }

@op
def aggregate_results(context: OpExecutionContext, results):
    """파티션 처리 결과 집계"""
    total_count = sum(r['count'] for r in results)
    total_sum = sum(r['sum'] for r in results)

    context.log.info(f"전체 레코드: {total_count}, 전체 합계: {total_sum}")

    return {
        'total_partitions': len(results),
        'total_records': total_count,
        'total_sum': total_sum
    }

@job
def dynamic_partition_job():
    partitions = partition_data()
    results = partitions.map(process_partition)
    aggregate_results(results.collect())

from dagster import op, job, Config, OpExecutionContext, Out, Output
import pandas as pd
from typing import Dict, List

# Config 정의
class ETLConfig(Config):
    source_file: str
    output_file: str
    min_value: int = 0

# Extract
@op
def extract_data(context: OpExecutionContext, config: ETLConfig):
    """CSV 파일에서 데이터 추출"""
    context.log.info(f"파일 로드 중: {config.source_file}")

    # 실제로는 파일을 읽지만, 예제 데이터 사용
    df = pd.DataFrame({
        'id': range(1, 11),
        'name': [f'User_{i}' for i in range(1, 11)],
        'value': [10, 25, 5, 30, 15, -5, 40, 20, 8, 35],
        'category': ['A', 'B', 'A', 'C', 'B', 'A', 'C', 'A', 'B', 'C']
    })

    return Output(
        value=df,
        metadata={
            "num_rows": len(df),
            "num_columns": len(df.columns),
            "columns": df.columns.tolist()
        }
    )

# Transform
@op(
    out={
        "clean_data": Out(),
        "invalid_data": Out()
    }
)
def transform_data(context: OpExecutionContext, config: ETLConfig, df: pd.DataFrame):
    """데이터 정제 및 변환"""

    # 유효하지 않은 데이터 분리
    invalid_df = df[df['value'] < config.min_value]
    clean_df = df[df['value'] >= config.min_value].copy()

    # 변환 수행
    clean_df['value_normalized'] = clean_df['value'] / clean_df['value'].max()
    clean_df['name_upper'] = clean_df['name'].str.upper()

    context.log.info(f"정제된 데이터: {len(clean_df)} 행")
    context.log.info(f"제외된 데이터: {len(invalid_df)} 행")

    return (
        Output(
            clean_df,
            output_name="clean_data",
            metadata={
                "clean_rows": len(clean_df),
                "transformation": "normalization applied"
            }
        ),
        Output(
            invalid_df,
            output_name="invalid_data",
            metadata={"invalid_rows": len(invalid_df)}
        )
    )

# Aggregate
@op
def aggregate_by_category(context: OpExecutionContext, df: pd.DataFrame):
    """카테고리별 집계"""
    aggregated = df.groupby('category').agg({
        'value': ['sum', 'mean', 'count'],
        'value_normalized': 'mean'
    }).round(2)

    context.log.info(f"카테고리 수: {len(aggregated)}")

    return aggregated

# Load
@op
def load_data(context: OpExecutionContext, config: ETLConfig, df: pd.DataFrame):
    """데이터 저장"""
    context.log.info(f"데이터 저장 중: {config.output_file}")

    # 실제로는 파일에 저장
    # df.to_csv(config.output_file, index=False)

    return Output(
        value=len(df),
        metadata={
            "output_file": config.output_file,
            "rows_saved": len(df)
        }
    )

# Log invalid
@op
def log_invalid_records(context: OpExecutionContext, df: pd.DataFrame):
    """유효하지 않은 레코드 로깅"""
    if len(df) > 0:
        context.log.warning(f"유효하지 않은 레코드 {len(df)}개 발견")
        for _, row in df.iterrows():
            context.log.warning(f"ID {row['id']}: value={row['value']}")
    else:
        context.log.info("모든 레코드가 유효합니다")

    return len(df)

# Job 정의

from dagster import op, job, DynamicOut, DynamicOutput, OpExecutionContext, Config
from typing import List, Dict
import pandas as pd

class DataSourceConfig(Config):
    """데이터 소스 설정"""
    sources: List[str]

@op(out=DynamicOut())
def discover_data_sources(context: OpExecutionContext, config: DataSourceConfig):
    """데이터 소스 탐색"""
    for idx, source in enumerate(config.sources):
        context.log.info(f"데이터 소스 발견: {source}")

        yield DynamicOutput(
            value={
                'name': source,
                'index': idx
            },
            mapping_key=f"source_{idx}"
        )

@op
def fetch_from_source(context: OpExecutionContext, source: Dict):
    """개별 소스에서 데이터 가져오기"""
    source_name = source['name']

    context.log.info(f"데이터 가져오는 중: {source_name}")

    # 모의 데이터 생성
    if source_name == "database_a":
        data = pd.DataFrame({
            'id': [1, 2, 3],
            'value': [100, 200, 300],
            'source': source_name
        })
    elif source_name == "database_b":
        data = pd.DataFrame({
            'id': [4, 5, 6],
            'value': [400, 500, 600],
            'source': source_name
        })
    else:
        data = pd.DataFrame({
            'id': [7, 8, 9],
            'value': [700, 800, 900],
            'source': source_name
        })

    context.log.info(f"{source_name}에서 {len(data)} 행 가져옴")

    return data

@op
def validate_data(context: OpExecutionContext, df: pd.DataFrame):
    """데이터 유효성 검증"""
    source = df['source'].iloc[0]

    # 검증 규칙
    valid_rows = df[df['value'] > 0]

    context.log.info(f"{source}: {len(valid_rows)}/{len(df)} 행 유효")

    return valid_rows

@op
def transform_dataset(context: OpExecutionContext, df: pd.DataFrame):
    """데이터셋 변환"""
    source = df['source'].iloc[0]

    df = df.copy()
    df['value_doubled'] = df['value'] * 2
    df['value_log'] = df['value'].apply(lambda x: x ** 0.5)

    context.log.info(f"{source} 변환 완료")

    return df

@op
def merge_datasets(context: OpExecutionContext, datasets: List[pd.DataFrame]):
    """모든 데이터셋 병합"""
    merged = pd.concat(datasets, ignore_index=True)

    context.log.info(f"총 {len(merged)} 행 병합됨")
    context.log.info(f"데이터 소스 수: {merged['source'].nunique()}")

    return merged

@op
def generate_summary(context: OpExecutionContext, df: pd.DataFrame):
    """요약 통계 생성"""
    summary = {
        'total_rows': len(df),
        'total_value': df['value'].sum(),
        'avg_value': df['value'].mean(),
        'sources': df['source'].unique().tolist()
    }

    context.log.info(f"요약: {summary}")

    return summary

@job
def dynamic_data_pipeline():
    """동적 데이터 처리 파이프라인"""
    sources = discover_data_sources()

    # 각 소스에서 데이터 가져오기
    fetched = sources.map(fetch_from_source)

    # 검증
    validated = fetched.map(validate_data)

    # 변환
    transformed = validated.map(transform_dataset)

    # 병합 및 요약
    merged = merge_datasets(transformed.collect())
    generate_summary(merged)

# 실행
if __name__ == "__main__":
    result = dynamic_data_pipeline.execute_in_process(
        run_config={
            "ops": {
                "discover_data_sources": {
                    "config": {
                        "sources": [
                            "database_a",
                            "database_b",
                            "api_source"
                        ]
                    }
                }
            }
        }
    )

from dagster import op, job, OpExecutionContext, Out, Output, In, Nothing
from typing import Optional

@op
def check_data_availability(context: OpExecutionContext):
    """데이터 가용성 확인"""
    # 실제로는 데이터 소스 확인
    available = True

    context.log.info(f"데이터 가용성: {available}")
    return available

@op(
    out={
        "fresh_data": Out(is_required=False),
        "cache_data": Out(is_required=False)
    }
)
def load_data(context: OpExecutionContext, is_available: bool):
    """조건에 따라 다른 출력 생성"""

    if is_available:
        context.log.info("새로운 데이터 로드")
        yield Output(
            value={"source": "fresh", "records": 1000},
            output_name="fresh_data"
        )
    else:
        context.log.info("캐시된 데이터 사용")
        yield Output(
            value={"source": "cache", "records": 500},
            output_name="cache_data"
        )

@op
def process_fresh_data(context: OpExecutionContext, data):
    """새로운 데이터 처리"""
    context.log.info(f"새 데이터 처리: {data['records']} 레코드")
    return {"processed": data['records'], "type": "fresh"}

@op
def process_cached_data(context: OpExecutionContext, data):
    """캐시된 데이터 처리"""
    context.log.info(f"캐시 데이터 처리: {data['records']} 레코드")
    return {"processed": data['records'], "type": "cache"}

@op
def merge_results(context: OpExecutionContext, fresh_result=None, cache_result=None):
    """결과 병합"""

    if fresh_result:
        result = fresh_result
    elif cache_result:
        result = cache_result
    else:
        result = {"processed": 0, "type": "none"}

    context.log.info(f"최종 결과: {result}")
    return result

@op(ins={"start": In(Nothing)})
def cleanup(context: OpExecutionContext):
    """정리 작업"""
    context.log.info("정리 작업 수행")

@job
def conditional_pipeline():
    """조건부 실행 파이프라인"""

    available = check_data_availability()
    fresh_data, cache_data = load_data(available)

    fresh_result = process_fresh_data(fresh_data)
    cache_result = process_cached_data(cache_data)

    final = merge_results(
        fresh_result=fresh_result,
        cache_result=cache_result
    )

    cleanup(start=final)

# 실행
if __name__ == "__main__":
    result = conditional_pipeline.execute_in_process()

정리​

이 장에서는 Dagster의 Ops와 Jobs에 대해 다루었습니다:

주요 개념​

  1. Op: 데이터 파이프라인의 기본 작업 단위
  2. Job: 여러 Op을 조합한 완전한 파이프라인
  3. 의존성: Op 간의 데이터 흐름과 실행 순서
  4. Config: 타입 안전한 설정 시스템
  5. 동적 실행: 런타임에 결정되는 병렬 처리
  6. 재시도: 실패에 대한 복원력

모범 사례​

  • Op은 단일 책임: 각 Op은 하나의 명확한 작업만 수행
  • 타입 힌트 사용: 입력과 출력의 타입을 명시
  • Config 스키마: Pydantic을 사용한 타입 안전한 설정
  • 적절한 재시도: 작업 특성에 맞는 재시도 전략
  • 로깅: Context를 통한 상세한 로깅
  • 메타데이터: Output에 유용한 메타데이터 포함
