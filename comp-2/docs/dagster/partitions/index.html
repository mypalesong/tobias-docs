6. Partitions

데이터를 분할하여 관리하고 처리하는 방법을 배워봅시다.

6.1 Partition 개념​

Partition은 데이터를 논리적으로 분할하여 독립적으로 처리할 수 있게 합니다.

**사용 사례:**
- 시계열 데이터 (일별, 월별 처리)
- 지역별 데이터 (국가, 도시별)
- 제품별 데이터 (카테고리별)
- 점진적 업데이트 (전체가 아닌 일부만)

장점:

  • 재계산 범위 최소화
  • 병렬 처리 가능
  • 실패 시 일부만 재실행
  • 백필 (과거 데이터 재처리) 용이

6.2 Daily Partitions​

6.2.1 기본 Daily Partition​

from dagster import asset, DailyPartitionsDefinition
import pandas as pd
from datetime import datetime

# Daily partition 정의
daily_partitions = DailyPartitionsDefinition(start_date="2024-01-01")

@asset(partitions_def=daily_partitions)
def daily_sales(context) -> pd.DataFrame:
    partition_date = context.partition_key
    context.log.info(f"Processing data for {partition_date}")

    # 해당 날짜의 데이터만 처리
    df = pd.read_csv(f"sales_{partition_date}.csv")
    return df

6.2.2 Partition 의존성​

@asset(partitions_def=daily_partitions)
def raw_logs(context) -> pd.DataFrame:
    date = context.partition_key
    return load_logs_for_date(date)

@asset(partitions_def=daily_partitions)
def processed_logs(context, raw_logs: pd.DataFrame) -> pd.DataFrame:
    """같은 partition의 raw_logs에 의존"""
    date = context.partition_key
    context.log.info(f"Processing logs for {date}")

    return raw_logs[raw_logs['status'] == 'success']

@asset(partitions_def=daily_partitions)
def daily_summary(context, processed_logs: pd.DataFrame) -> dict:
    """일별 요약"""
    return {
        "date": context.partition_key,
        "total_count": len(processed_logs),
        "success_rate": len(processed_logs) / len(processed_logs) * 100
    }

6.4 Static Partitions​

6.4.1 기본 Static Partition​

from dagster import StaticPartitionsDefinition, asset

regions = StaticPartitionsDefinition(
    ["north", "south", "east", "west"]
)

@asset(partitions_def=regions)
def regional_data(context) -> pd.DataFrame:
    region = context.partition_key
    context.log.info(f"Processing region: {region}")

    return load_data_for_region(region)

6.4.2 카테고리 Partition​

product_categories = StaticPartitionsDefinition([
    "electronics",
    "clothing",
    "food",
    "books"
])

@asset(partitions_def=product_categories)
def category_sales(context) -> pd.DataFrame:
    category = context.partition_key

    df = pd.read_csv("sales.csv")
    return df[df['category'] == category]

@asset(partitions_def=product_categories)
def category_metrics(
    context,
    category_sales: pd.DataFrame
) -> dict:
    """카테고리별 메트릭"""
    return {
        "category": context.partition_key,
        "total_sales": category_sales['amount'].sum(),
        "num_transactions": len(category_sales)
    }

6.5 Multi-dimensional Partitions​

from dagster import MultiPartitionsDefinition

# 날짜 x 지역
date_region_partitions = MultiPartitionsDefinition({
    "date": daily_partitions,
    "region": regions
})

@asset(partitions_def=date_region_partitions)
def regional_daily_sales(context) -> pd.DataFrame:
    date = context.partition_key.keys_by_dimension["date"]
    region = context.partition_key.keys_by_dimension["region"]

    context.log.info(f"Processing {date} for {region}")

    return load_sales(date=date, region=region)

@asset(partitions_def=date_region_partitions)
def regional_daily_summary(
    context,
    regional_daily_sales: pd.DataFrame
) -> dict:
    """날짜별, 지역별 요약"""
    return {
        "date": context.partition_key.keys_by_dimension["date"],
        "region": context.partition_key.keys_by_dimension["region"],
        "total": regional_daily_sales['amount'].sum()
    }

6.6 Partition Mapping​

6.6.1 TimeWindow Partition Mapping​

from dagster import TimeWindowPartitionMapping

@asset(partitions_def=daily_partitions)
def daily_data(context) -> pd.DataFrame:
    return load_daily_data(context.partition_key)

# 주간 partition이 일별 partition에 의존
@asset(
    partitions_def=weekly_partitions,
    ins={
        "daily_data": AssetIn(
            partition_mapping=TimeWindowPartitionMapping()
        )
    }
)
def weekly_summary(context, daily_data: dict) -> pd.DataFrame:
    """일주일치 daily_data를 모두 받아서 집계"""
    # daily_data는 해당 주의 모든 일별 데이터
    all_dfs = [df for df in daily_data.values()]
    return pd.concat(all_dfs)

6.6.2 Identity Partition Mapping​

from dagster import IdentityPartitionMapping, AssetIn

@asset(partitions_def=regions)
def raw_regional_data(context) -> pd.DataFrame:
    return load_data(context.partition_key)

@asset(
    partitions_def=regions,
    ins={
        "raw_regional_data": AssetIn(
            partition_mapping=IdentityPartitionMapping()
        )
    }
)
def processed_regional_data(
    context,
    raw_regional_data: pd.DataFrame
) -> pd.DataFrame:
    """같은 partition에 대응"""
    return process_data(raw_regional_data)

6.7 Backfills​

6.7.1 UI에서 Backfill​

1. Dagit UI에서 Asset 선택
2. "Materialize" 클릭
3. "Partition selection" 선택
4. 날짜 범위 지정 (예: 2024-01-01 ~ 2024-01-31)
5. "Launch backfill" 클릭

6.7.2 코드로 Backfill​

from dagster import build_asset_context, materialize
from datetime import datetime, timedelta

# 특정 기간 backfill
def backfill_assets(start_date: str, end_date: str):
    """지정된 기간의 모든 partition 실행"""
    start = datetime.strptime(start_date, "%Y-%m-%d")
    end = datetime.strptime(end_date, "%Y-%m-%d")

    current = start
    while current <= end:
        partition_key = current.strftime("%Y-%m-%d")

        context = build_asset_context(
            partition_key=partition_key
        )

        materialize(
            [daily_sales, daily_summary],
            partition_key=partition_key
        )

        current += timedelta(days=1)

# 실행
backfill_assets("2024-01-01", "2024-01-31")

6.8 Incremental Loading​

from dagster import asset, OpExecutionContext
import pandas as pd

@asset(partitions_def=daily_partitions)
def incremental_sales(context: OpExecutionContext) -> pd.DataFrame:
    """증분 로드: 해당 날짜 데이터만 처리"""
    date = context.partition_key

    # 이전 데이터는 건드리지 않음
    # 해당 날짜 데이터만 로드
    new_data = pd.read_csv(f"sales_{date}.csv")

    context.log.info(
        f"Loaded {len(new_data)} new records for {date}"
    )

    return new_data

@asset(partitions_def=daily_partitions)
def cumulative_sales(
    context: OpExecutionContext,
    incremental_sales: pd.DataFrame
) -> pd.DataFrame:
    """누적 데이터: 모든 이전 partition 데이터와 결합"""
    date = context.partition_key

    # 이전 partition들의 데이터 로드
    all_data = []

    # 현재 날짜까지의 모든 partition
    for partition_key in daily_partitions.get_partition_keys_in_range(
        start=daily_partitions.start,
        end=date
    ):
        data = load_partition_data(partition_key)
        all_data.append(data)

    # 모든 데이터 결합
    cumulative = pd.concat(all_data, ignore_index=True)

    return cumulative

6.9 Dynamic Partitions​

from dagster import (
    DynamicPartitionsDefinition,
    asset,
    sensor,
    RunRequest
)

# 동적 partition 정의
file_partitions = DynamicPartitionsDefinition(name="files")

@asset(partitions_def=file_partitions)
def process_file(context) -> dict:
    filename = context.partition_key
    context.log.info(f"Processing {filename}")

    data = load_file(filename)
    return process_data(data)

# 새 파일 감지 시 partition 추가
@sensor(jobs=[process_file_job])
def file_sensor(context):
    # 새 파일 확인
    new_files = check_for_new_files()

    for filename in new_files:
        # 새 partition 추가
        context.instance.add_dynamic_partitions(
            partitions_def_name="files",
            partition_keys=[filename]
        )

        # 실행 요청
        yield RunRequest(
            run_key=filename,
            partition_key=filename
        )

6.10 실습 문제​

문제: 월별 매출 파이프라인​

월별로 partitioning된 매출 파이프라인을 구현하세요.

요구사항:

  1. 월별 원시 데이터 로드
  2. 데이터 정제 (결측치, 중복 제거)
  3. 월별 집계 (총 매출, 평균, 제품별)
  4. 분기별 요약 (3개월 데이터 결합)
해답
from dagster import (
    asset,
    MonthlyPartitionsDefinition,
    TimeWindowPartitionMapping,
    AssetIn,
    Definitions
)
import pandas as pd
from datetime import datetime

# Monthly partition
monthly = MonthlyPartitionsDefinition(start_date="2024-01-01")

# Quarterly partition (3개월)
from dagster import PartitionsDefinition

class QuarterlyPartitionsDefinition(PartitionsDefinition):
    def get_partition_keys(self, current_time=None):
        return [
            "2024-Q1",
            "2024-Q2",
            "2024-Q3",
            "2024-Q4"
        ]

quarterly = QuarterlyPartitionsDefinition()

# 1. 월별 원시 데이터
@asset(partitions_def=monthly)
def raw_monthly_sales(context) -> pd.DataFrame:
    month = context.partition_key  # "2024-01"
    context.log.info(f"Loading sales for {month}")

    # CSV 로드
    df = pd.read_csv(f"sales_{month}.csv")

    context.add_output_metadata({
        "num_rows": len(df),
        "month": month
    })

    return df

# 2. 데이터 정제
@asset(partitions_def=monthly)
def cleaned_monthly_sales(
    context,
    raw_monthly_sales: pd.DataFrame
) -> pd.DataFrame:
    """결측치 및 중복 제거"""
    df = raw_monthly_sales.copy()

    # 결측치 제거
    df = df.dropna()

    # 중복 제거
    df = df.drop_duplicates(subset=['order_id'])

    context.log.info(
        f"Cleaned: {len(raw_monthly_sales)} -> {len(df)} rows"
    )

    context.add_output_metadata({
        "removed_rows": len(raw_monthly_sales) - len(df),
        "final_rows": len(df)
    })

    return df

# 3. 월별 집계
@asset(partitions_def=monthly)
def monthly_summary(
    context,
    cleaned_monthly_sales: pd.DataFrame
) -> pd.DataFrame:
    """월별 요약 통계"""
    month = context.partition_key

    summary = pd.DataFrame({
        "month": [month],
        "total_sales": [cleaned_monthly_sales['amount'].sum()],
        "avg_order": [cleaned_monthly_sales['amount'].mean()],
        "num_orders": [len(cleaned_monthly_sales)],
        "num_customers": [cleaned_monthly_sales['customer_id'].nunique()]
    })

    # 제품별 집계
    by_product = cleaned_monthly_sales.groupby('product').agg({
        'amount': ['sum', 'count']
    }).reset_index()

    context.add_output_metadata({
        "total_sales": float(summary['total_sales'].iloc[0]),
        "num_products": len(by_product)
    })

    return summary

# 4. 분기별 요약 (3개월 데이터)
@asset(
    partitions_def=quarterly,
    ins={
        "monthly_summary": AssetIn(
            partition_mapping=TimeWindowPartitionMapping()
        )
    }
)
def quarterly_report(
    context,
    monthly_summary: dict
) -> pd.DataFrame:
    """분기별 리포트: 3개월 데이터 결합"""
    quarter = context.partition_key  # "2024-Q1"

    # 모든 월별 데이터 결합
    all_months = []
    for month_df in monthly_summary.values():
        all_months.append(month_df)

    quarterly_data = pd.concat(all_months, ignore_index=True)

    # 분기 집계
    report = pd.DataFrame({
        "quarter": [quarter],
        "total_sales": [quarterly_data['total_sales'].sum()],
        "avg_monthly_sales": [quarterly_data['total_sales'].mean()],
        "total_orders": [quarterly_data['num_orders'].sum()],
        "total_customers": [quarterly_data['num_customers'].sum()]
    })

    context.log.info(
        f"Quarterly report for {quarter}: "
        f"${report['total_sales'].iloc[0]:,.2f}"
    )

    return report

# Definitions
defs = Definitions(
    assets=[
        raw_monthly_sales,
        cleaned_monthly_sales,
        monthly_summary,
        quarterly_report
    ]
)

Partition 모범 사례
  • 적절한 크기: 너무 작으면 오버헤드, 너무 크면 재계산 비용 증가
  • 일관된 키: 날짜는 ISO 형식 (YYYY-MM-DD)
  • 의존성 명시: Partition mapping으로 명확히 표현
  • 백필 계획: 과거 데이터 재처리 전략 수립
  • 모니터링: 실패한 partition 추적
