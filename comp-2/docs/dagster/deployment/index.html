8. 배포

프로덕션 환경에 Dagster를 배포하는 방법을 학습합니다. Docker, Kubernetes를 사용한 배포 전략과 프로덕션 설정을 다룹니다.

8.1 배포 아키텍처​

8.1.1 Dagster 배포 구성 요소​

# Dagster 배포의 주요 구성 요소

1. Dagit (Web Server)
   - UI를 제공하는 웹 서버
   - 포트: 기본 3000
   - 역할: 파이프라인 시각화, 실행 트리거

2. Dagster Daemon
   - 백그라운드 프로세스
   - 역할:
     - 스케줄 실행
     - 센서 평가
     - Run queue 관리
     - Run 재시도

3. User Code (Code Location)
   - 실제 파이프라인 코드
   - gRPC 서버로 실행
   - 독립적으로 배포 가능

4. Storage (Metadata Storage)
   - PostgreSQL 권장
   - 역할:
     - Run history
     - Event logs
     - Asset catalog

5. Run Coordinator
   - Run 실행 조율
   - 기본: DefaultRunCoordinator
   - 프로덕션: QueuedRunCoordinator

6. Run Launcher
   - Run 실행 방법 정의
   - 옵션:
     - DefaultRunLauncher (subprocess)
     - DockerRunLauncher
     - K8sRunLauncher

# deploy.sh - Kubernetes 배포 스크립트

set -e

NAMESPACE="dagster"
RELEASE_NAME="dagster"

echo "🚀 Dagster 배포 시작..."

# 1. 네임스페이스 생성
echo "📦 네임스페이스 생성..."
kubectl create namespace $NAMESPACE --dry-run=client -o yaml | kubectl apply -f -

# 2. Secrets 생성
echo "🔐 Secrets 생성..."
kubectl apply -f k8s/secret.yaml
kubectl apply -f k8s/database-secret.yaml

# 3. ConfigMaps 생성
echo "⚙️  ConfigMaps 생성..."
kubectl apply -f k8s/configmap.yaml

# 4. PVC 생성
echo "💾 Persistent Volume Claims 생성..."
kubectl apply -f k8s/persistent-volume.yaml

# 5. Helm Chart 설치/업그레이드
echo "📊 Helm Chart 배포..."
helm upgrade --install $RELEASE_NAME dagster/dagster \
  --namespace $NAMESPACE \
  --values values.yaml \
  --wait \
  --timeout 10m

# 6. 배포 상태 확인
echo "✅ 배포 상태 확인..."
kubectl rollout status deployment/${RELEASE_NAME}-dagster-webserver -n $NAMESPACE
kubectl rollout status deployment/${RELEASE_NAME}-dagster-daemon -n $NAMESPACE

# 7. Pod 상태 출력
echo "🔍 Pod 상태:"
kubectl get pods -n $NAMESPACE

# 8. Service 정보 출력
echo "🌐 Service 정보:"
kubectl get svc -n $NAMESPACE

echo "✨ 배포 완료!"
echo "Dagster UI: http://$(kubectl get svc ${RELEASE_NAME}-dagster-webserver -n $NAMESPACE -o jsonpath='{.status.loadBalancer.ingress[0].ip}')"

8.4.1 환경별 설정​

# config/settings.py
"""
환경별 설정 관리
"""
import os
from typing import Dict, Any
from enum import Enum

class Environment(Enum):
    """환경 타입"""
    DEVELOPMENT = "development"
    STAGING = "staging"
    PRODUCTION = "production"

class Settings:
    """애플리케이션 설정"""

    def __init__(self):
        self.env = Environment(os.getenv("ENVIRONMENT", "development"))
        self.debug = os.getenv("DEBUG", "false").lower() == "true"

        # Database
        self.db_host = os.getenv("DB_HOST", "localhost")
        self.db_port = int(os.getenv("DB_PORT", "5432"))
        self.db_name = os.getenv("DB_NAME", "dagster")
        self.db_user = os.getenv("DB_USER", "dagster")
        self.db_password = os.getenv("DB_PASSWORD", "")

        # AWS
        self.aws_region = os.getenv("AWS_DEFAULT_REGION", "us-east-1")
        self.s3_bucket = os.getenv("S3_BUCKET", "")
        self.s3_prefix = os.getenv("S3_PREFIX", "dagster/")

        # Run Settings
        self.max_concurrent_runs = int(os.getenv("MAX_CONCURRENT_RUNS", "10"))
        self.run_timeout = int(os.getenv("RUN_TIMEOUT", "3600"))

    @property
    def database_url(self) -> str:
        """PostgreSQL 연결 문자열"""
        return (
            f"postgresql://{self.db_user}:{self.db_password}"
            f"@{self.db_host}:{self.db_port}/{self.db_name}"
        )

    @property
    def is_production(self) -> bool:
        """프로덕션 환경 여부"""
        return self.env == Environment.PRODUCTION

    def to_dict(self) -> Dict[str, Any]:
        """설정을 딕셔너리로 변환"""
        return {
            "environment": self.env.value,
            "debug": self.debug,
            "database": {
                "host": self.db_host,
                "port": self.db_port,
                "name": self.db_name,
            },
            "aws": {
                "region": self.aws_region,
                "s3_bucket": self.s3_bucket,
            },
        }

# 싱글톤 인스턴스
settings = Settings()
# config/__init__.py
from .settings import settings

__all__ = ["settings"]

# repository.py
from dagster import Definitions, asset, ConfigurableResource
from config import settings
import pandas as pd

class S3Resource(ConfigurableResource):
    """S3 리소스"""
    bucket: str
    prefix: str
    region: str

    def upload_dataframe(self, df: pd.DataFrame, key: str):
        """DataFrame을 S3에 업로드"""
        import boto3
        s3 = boto3.client('s3', region_name=self.region)
        full_key = f"{self.prefix}{key}"
        csv_buffer = df.to_csv(index=False)
        s3.put_object(Bucket=self.bucket, Key=full_key, Body=csv_buffer)

@asset
def processed_data(s3: S3Resource) -> pd.DataFrame:

# security/authentication.py
"""
Dagster 인증 및 보안 설정
"""
from dagster import ConfigurableResource
from typing import Optional
import jwt
from datetime import datetime, timedelta

class AuthService(ConfigurableResource):
    """인증 서비스"""
    secret_key: str
    algorithm: str = "HS256"
    access_token_expire_minutes: int = 30

    def create_access_token(
        self,
        data: dict,
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """Access Token 생성"""
        to_encode = data.copy()

        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(
                minutes=self.access_token_expire_minutes
            )

        to_encode.update({"exp": expire})
        encoded_jwt = jwt.encode(
            to_encode,
            self.secret_key,
            algorithm=self.algorithm
        )

        return encoded_jwt

    def verify_token(self, token: str) -> Optional[dict]:
        """Token 검증"""
        try:
            payload = jwt.decode(
                token,
                self.secret_key,
                algorithms=[self.algorithm]
            )
            return payload
        except jwt.PyJWTError:
            return None

"""
from enum import Enum
from typing import List, Set

class Permission(Enum):
    """권한 정의"""
    READ_ASSETS = "read:assets"
    WRITE_ASSETS = "write:assets"
    EXECUTE_RUNS = "execute:runs"
    MANAGE_SCHEDULES = "manage:schedules"
    ADMIN = "admin"

class Role(Enum):
    """역할 정의"""
    VIEWER = "viewer"
    OPERATOR = "operator"
    DEVELOPER = "developer"
    ADMIN = "admin"

ROLE_PERMISSIONS: dict[Role, Set[Permission]] = {
    Role.VIEWER: {
        Permission.READ_ASSETS,
    },
    Role.OPERATOR: {
        Permission.READ_ASSETS,
        Permission.EXECUTE_RUNS,
    },
    Role.DEVELOPER: {
        Permission.READ_ASSETS,
        Permission.WRITE_ASSETS,
        Permission.EXECUTE_RUNS,
        Permission.MANAGE_SCHEDULES,

    Role.ADMIN: {
        Permission.READ_ASSETS,
        Permission.WRITE_ASSETS,
        Permission.EXECUTE_RUNS,
        Permission.MANAGE_SCHEDULES,
        Permission.ADMIN,
    },
}

def check_permission(role: Role, permission: Permission) -> bool:
    """권한 확인"""
    return permission in ROLE_PERMISSIONS.get(role, set())

"""
Rate Limiting 미들웨어
"""
from dagster import ConfigurableResource, op, OpExecutionContext
from datetime import datetime, timedelta
from collections import defaultdict
from typing import Dict
import threading

class RateLimiter(ConfigurableResource):
    """Rate Limiter 리소스"""
    max_requests: int = 100
    time_window: int = 60  # seconds

    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self._requests: Dict[str, list] = defaultdict(list)
        self._lock = threading.Lock()

    def is_allowed(self, client_id: str) -> bool:
        """요청 허용 여부 확인"""
        with self._lock:
            now = datetime.now()
            cutoff = now - timedelta(seconds=self.time_window)

            # 오래된 요청 제거
            self._requests[client_id] = [
                req_time for req_time in self._requests[client_id]
                if req_time > cutoff
            ]

            # Rate limit 체크
            if len(self._requests[client_id]) >= self.max_requests:
                return False

            # 새 요청 기록
            self._requests[client_id].append(now)
            return True

@op
def rate_limited_api_call(
    context: OpExecutionContext,
    rate_limiter: RateLimiter,

# deployment/checklist.py
"""
프로덕션 배포 체크리스트
"""
from dataclasses import dataclass
from typing import List, Tuple
from enum import Enum

class CheckStatus(Enum):
    """체크 상태"""
    PASS = "✅ PASS"
    FAIL = "❌ FAIL"
    WARNING = "⚠️ WARNING"
    SKIP = "⏭️ SKIP"

@dataclass
class CheckResult:
    """체크 결과"""
    name: str
    status: CheckStatus
    message: str

class DeploymentChecker:
    """배포 체크리스트"""

    def __init__(self):
        self.results: List[CheckResult] = []

    def check_environment_variables(self) -> CheckResult:
        """환경 변수 확인"""
        import os
        required_vars = [
            "DAGSTER_POSTGRES_HOST",
            "DAGSTER_POSTGRES_DB",
            "DAGSTER_POSTGRES_USER",
            "DAGSTER_POSTGRES_PASSWORD",
        ]

        missing = [var for var in required_vars if not os.getenv(var)]

        if missing:
            return CheckResult(
                name="Environment Variables",
                status=CheckStatus.FAIL,
                message=f"Missing variables: {', '.join(missing)}"
            )

        return CheckResult(
            name="Environment Variables",
            status=CheckStatus.PASS,
            message="All required environment variables are set"
        )

    def check_database_connection(self) -> CheckResult:
        """데이터베이스 연결 확인"""
        try:
            import psycopg2
            import os

            conn = psycopg2.connect(
                host=os.getenv("DAGSTER_POSTGRES_HOST"),
                database=os.getenv("DAGSTER_POSTGRES_DB"),
                user=os.getenv("DAGSTER_POSTGRES_USER"),
                password=os.getenv("DAGSTER_POSTGRES_PASSWORD"),
                connect_timeout=5
            )
            conn.close()

            return CheckResult(
                name="Database Connection",
                status=CheckStatus.PASS,
                message="Successfully connected to PostgreSQL"
            )

        except Exception as e:
            return CheckResult(
                name="Database Connection",
                status=CheckStatus.FAIL,
                message=f"Failed to connect: {str(e)}"
            )

    def check_storage_backend(self) -> CheckResult:
        """스토리지 백엔드 확인"""
        try:
            import boto3
            import os

            s3 = boto3.client('s3')
            bucket = os.getenv("S3_BUCKET")

            if not bucket:
                return CheckResult(
                    name="Storage Backend",
                    status=CheckStatus.WARNING,
                    message="S3_BUCKET not configured"
                )

            # 버킷 접근 테스트
            s3.head_bucket(Bucket=bucket)

            return CheckResult(
                name="Storage Backend",
                status=CheckStatus.PASS,
                message=f"S3 bucket '{bucket}' is accessible"
            )

        except Exception as e:
            return CheckResult(
                name="Storage Backend",
                status=CheckStatus.FAIL,
                message=f"S3 access failed: {str(e)}"
            )

    def check_dagster_instance(self) -> CheckResult:
        """Dagster 인스턴스 확인"""
        try:
            from dagster import DagsterInstance

            instance = DagsterInstance.get()

            return CheckResult(
                name="Dagster Instance",
                status=CheckStatus.PASS,
                message=f"Instance loaded: {instance.root_directory}"
            )

        except Exception as e:
            return CheckResult(
                name="Dagster Instance",
                status=CheckStatus.FAIL,
                message=f"Failed to load instance: {str(e)}"
            )

    def check_resources(self) -> CheckResult:
        """리소스 설정 확인"""
        # 메모리, CPU 제한 등 확인
        import psutil

        cpu_count = psutil.cpu_count()
        memory_gb = psutil.virtual_memory().total / (1024**3)

        if cpu_count < 2:
            status = CheckStatus.WARNING
            message = f"Low CPU count: {cpu_count}"
        elif memory_gb < 4:
            status = CheckStatus.WARNING
            message = f"Low memory: {memory_gb:.1f}GB"
        else:
            status = CheckStatus.PASS
            message = f"CPU: {cpu_count}, Memory: {memory_gb:.1f}GB"

        return CheckResult(
            name="System Resources",
            status=status,
            message=message
        )

    def run_all_checks(self) -> List[CheckResult]:
        """모든 체크 실행"""
        checks = [
            self.check_environment_variables,
            self.check_database_connection,
            self.check_storage_backend,
            self.check_dagster_instance,
            self.check_resources,
        ]

        self.results = [check() for check in checks]
        return self.results

    def print_results(self):
        """결과 출력"""
        print("\n" + "="*60)
        print("🚀 Dagster 배포 체크리스트 결과")
        print("="*60 + "\n")

        for result in self.results:
            print(f"{result.status.value} {result.name}")
            print(f"   → {result.message}\n")

        # 요약
        pass_count = sum(1 for r in self.results if r.status == CheckStatus.PASS)
        fail_count = sum(1 for r in self.results if r.status == CheckStatus.FAIL)
        warning_count = sum(1 for r in self.results if r.status == CheckStatus.WARNING)

        print("="*60)
        print(f"총 {len(self.results)}개 체크")
        print(f"✅ Pass: {pass_count}")
        print(f"❌ Fail: {fail_count}")
        print(f"⚠️ Warning: {warning_count}")
        print("="*60)

        return fail_count == 0

# CLI 실행
if __name__ == "__main__":
    checker = DeploymentChecker()
    checker.run_all_checks()
    success = checker.print_results()

    if not success:
        print("\n⚠️ 배포 전에 실패한 항목을 수정하세요!")
        exit(1)
    else:
        print("\n✨ 모든 체크를 통과했습니다. 배포를 진행하세요!")
