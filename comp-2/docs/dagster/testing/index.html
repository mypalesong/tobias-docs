7. 테스트

Dagster는 데이터 파이프라인의 신뢰성을 보장하기 위한 강력한 테스트 프레임워크를 제공합니다. 이 장에서는 Asset, Op, Resource를 테스트하는 다양한 방법을 학습합니다.

7.1 테스트의 중요성​

7.1.1 데이터 파이프라인 테스트가 중요한 이유​

"""
데이터 파이프라인 테스트의 필요성:

1. 데이터 품질 보장
   - 잘못된 데이터가 프로덕션에 도달하는 것을 방지
   - 비즈니스 로직의 정확성 검증

2. 리팩토링 안전성
   - 코드 변경 시 기존 기능이 깨지지 않음을 보장
   - 자신감 있는 코드 개선

3. 문서화 효과
   - 테스트 코드가 파이프라인의 동작 방식을 설명
   - 새로운 팀원의 온보딩 지원

4. 빠른 피드백
   - 프로덕션 배포 전에 문제 발견
   - 디버깅 시간 단축
"""

7.1.2 Dagster 테스트 철학​

Dagster는 "단위 테스트 가능한 데이터 파이프라인"을 핵심 원칙으로 합니다:

  • 격리된 테스트: 각 컴포넌트를 독립적으로 테스트
  • Resource Mocking: 외부 의존성을 쉽게 대체
  • 재현 가능성: 동일한 입력에 대해 동일한 결과 보장

from dagster import asset
import pandas as pd

@asset
def users_data() -> pd.DataFrame:
    """사용자 데이터를 생성합니다."""
    return pd.DataFrame({
        "user_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Charlie"],

    })

def adult_users(users_data: pd.DataFrame) -> pd.DataFrame:
    """성인 사용자만 필터링합니다."""
    return users_data[users_data["age"] >= 18]

7.2.2 Asset 의존성 테스트​

# analytics_assets.py
from dagster import asset, AssetIn
import pandas as pd

@asset
def daily_sales() -> pd.DataFrame:
    """일별 매출 데이터"""
    return pd.DataFrame({
        "date": pd.date_range("2024-01-01", periods=7),
        "revenue": [1000, 1200, 800, 1500, 1100, 1300, 900]
    })

@asset
def weekly_revenue(daily_sales: pd.DataFrame) -> float:
    """주간 총 매출"""
    return daily_sales["revenue"].sum()

@asset(
    ins={
        "daily_sales": AssetIn(),
        "weekly_revenue": AssetIn()
    }
)
def revenue_report(daily_sales: pd.DataFrame, weekly_revenue: float) -> dict:
    """매출 리포트 생성"""
    return {
        "total_revenue": weekly_revenue,
        "daily_average": weekly_revenue / len(daily_sales),
        "max_daily": daily_sales["revenue"].max(),
        "min_daily": daily_sales["revenue"].min()
    }
# test_analytics_assets.py

from analytics_assets import daily_sales, weekly_revenue, revenue_report

@asset
def customers_from_db(database: DatabaseConnection) -> pd.DataFrame:
    """데이터베이스에서 고객 데이터를 가져옵니다."""
    return database.query("SELECT * FROM customers WHERE active = true")

@asset
def customer_segments(customers_from_db: pd.DataFrame) -> pd.DataFrame:
    """고객을 세그먼트로 분류합니다."""
    df = customers_from_db.copy()

    # 간단한 세그먼트 로직
    conditions = [
        df["total_spent"] >= 10000,
        (df["total_spent"] >= 1000) & (df["total_spent"] < 10000),
        df["total_spent"] < 1000
    ]
    choices = ["Premium", "Standard", "Basic"]
    df["segment"] = pd.Series(
        [choices[i] for i in range(len(df))
         for j, cond in enumerate(conditions) if cond.iloc[i]]
    )

    return df

7.3.1 기본 Op 테스트​

# my_ops.py
from dagster import op, In, Out
import pandas as pd

@op
def load_csv(context, file_path: str) -> pd.DataFrame:
    """CSV 파일을 로드합니다."""
    context.log.info(f"Loading CSV from {file_path}")
    return pd.read_csv(file_path)

@op(
    ins={"df": In(pd.DataFrame)},
    out=Out(pd.DataFrame)
)
def clean_data(context, df: pd.DataFrame) -> pd.DataFrame:
    """데이터를 정제합니다."""
    context.log.info(f"Cleaning {len(df)} rows")

    # 결측치 제거
    cleaned = df.dropna()

    # 중복 제거
    cleaned = cleaned.drop_duplicates()

    context.log.info(f"After cleaning: {len(cleaned)} rows")
    return cleaned

@op(
    ins={"df": In(pd.DataFrame)},
    out=Out(dict)
)
def calculate_stats(df: pd.DataFrame) -> dict:
    """통계를 계산합니다."""
    return {
        "row_count": len(df),
        "column_count": len(df.columns),
        "numeric_columns": df.select_dtypes(include=['number']).columns.tolist()
    }
# test_my_ops.py
from dagster import build_op_context
from my_ops import load_csv, clean_data, calculate_stats
import pandas as pd
import tempfile
import os

def test_clean_data():
    """clean_data op 단위 테스트"""
    # 테스트 데이터 생성
    test_df = pd.DataFrame({
        "a": [1, 2, None, 4, 2],  # 결측치와 중복 포함
        "b": ["x", "y", "z", "w", "y"]
    })

    # Op 컨텍스트 생성
    context = build_op_context()

    # Op 실행
    result = clean_data(context, test_df)

    # 결과 검증
    assert len(result) == 3  # 결측치 1개, 중복 1개 제거됨
    assert result["a"].isna().sum() == 0  # 결측치 없음
    assert len(result) == len(result.drop_duplicates())  # 중복 없음

def test_calculate_stats():
    """calculate_stats op 테스트"""
    test_df = pd.DataFrame({
        "number1": [1, 2, 3],
        "number2": [4, 5, 6],
        "text": ["a", "b", "c"]
    })

    result = calculate_stats(test_df)

    assert result["row_count"] == 3
    assert result["column_count"] == 3
    assert set(result["numeric_columns"]) == {"number1", "number2"}

def test_load_csv():
    """load_csv op 테스트 (파일 I/O 포함)"""
    # 임시 CSV 파일 생성
    with tempfile.NamedTemporaryFile(mode='w', delete=False, suffix='.csv') as f:
        f.write("col1,col2\n")
        f.write("1,2\n")

        assert list(result.columns) == ["col1", "col2"]
        assert result["col1"].tolist() == [1, 3]
    finally:
        # 임시 파일 정리

7.3.2 Op Configuration 테스트​

# configurable_ops.py
from dagster import op, Config
import pandas as pd
from typing import List

class FilterConfig(Config):
    """필터링 설정"""
    min_value: float = 0.0
    max_value: float = 100.0
    columns_to_check: List[str] = ["value"]

@op
def filter_outliers(config: FilterConfig, df: pd.DataFrame) -> pd.DataFrame:
    """설정에 따라 이상치를 필터링합니다."""
    result = df.copy()

    for col in config.columns_to_check:
        if col in result.columns:
            result = result[
                (result[col] >= config.min_value) &
                (result[col] <= config.max_value)
            ]

    return result
# test_configurable_ops.py
from dagster import build_op_context
from configurable_ops import FilterConfig, filter_outliers
import pandas as pd

def test_filter_outliers_with_config():
    """설정을 사용한 필터링 테스트"""
    test_df = pd.DataFrame({
        "value": [1, 50, 99, 150, -10, 75]
    })

    # 설정 생성
    config = FilterConfig(
        min_value=0.0,
        max_value=100.0,
        columns_to_check=["value"]
    )

    # Op 실행
    result = filter_outliers(config, test_df)

    # 0~100 범위의 값만 남아야 함
    assert len(result) == 4  # 1, 50, 99, 75
    assert all(result["value"] >= 0)
    assert all(result["value"] <= 100)

def test_filter_outliers_multiple_columns():
    """여러 컬럼 필터링 테스트"""
    test_df = pd.DataFrame({
        "value1": [10, 20, 150],
        "value2": [5, 15, 25]
    })

    config = FilterConfig(
        min_value=0.0,
        max_value=100.0,
        columns_to_check=["value1", "value2"]
    )

    result = filter_outliers(config, test_df)

    # value1이 150인 행이 제거되어야 함
    assert len(result) == 2
    assert result["value1"].tolist() == [10, 20]

# my_jobs.py
from dagster import job, op
import pandas as pd

@op
def extract_data() -> pd.DataFrame:
    """데이터 추출"""
    return pd.DataFrame({"id": [1, 2, 3], "value": [10, 20, 30]})

@op
def transform_data(data: pd.DataFrame) -> pd.DataFrame:
    """데이터 변환"""
    data["value_doubled"] = data["value"] * 2
    return data

@op
def load_data(data: pd.DataFrame) -> int:
    """데이터 로드 (저장)"""
    return len(data)

@job
def etl_pipeline():
    """ETL 파이프라인"""
    raw = extract_data()
    transformed = transform_data(raw)
    load_data(transformed)
# test_my_jobs.py

# resources.py
from dagster import ConfigurableResource
import requests
from typing import Dict, Any

class APIClient(ConfigurableResource):
    """외부 API 클라이언트"""
    base_url: str
    api_key: str

    def get_user(self, user_id: int) -> Dict[str, Any]:
        """사용자 정보 조회"""
        response = requests.get(
            f"{self.base_url}/users/{user_id}",
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        response.raise_for_status()
        return response.json()

    def create_order(self, order_data: Dict[str, Any]) -> Dict[str, Any]:
        """주문 생성"""
        response = requests.post(
            f"{self.base_url}/orders",
            json=order_data,
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        response.raise_for_status()
        return response.json()

class InMemoryDatabase(DatabaseResource):
    """인메모리 테스트 데이터베이스"""

    def __init__(self, **kwargs):
        super().__init__(connection_string="memory://test")
        self.tables: Dict[str, pd.DataFrame] = {
            "users": pd.DataFrame({
                "id": [1, 2, 3],
                "name": ["Alice", "Bob", "Charlie"]
            }),
            "orders": pd.DataFrame({
                "order_id": [101, 102],
                "user_id": [1, 2],
                "amount": [50.0, 75.0]
            })
        }

    def execute_query(self, query: str) -> pd.DataFrame:
        """간단한 쿼리 파싱 및 실행"""
        # 실제로는 SQL 파서를 사용하겠지만, 여기서는 단순화
        if "FROM users" in query.upper():
            return self.tables["users"].copy()
        elif "FROM orders" in query.upper():
            return self.tables["orders"].copy()
        else:
            return pd.DataFrame()

    def insert_records(self, table: str, records: List[Dict]) -> int:
        """레코드를 테이블에 추가"""
        if table in self.tables:
            new_df = pd.DataFrame(records)
            self.tables[table] = pd.concat(
                [self.tables[table], new_df],
                ignore_index=True
            )
            return len(records)
        return 0

daily_partitions = DailyPartitionsDefinition(start_date="2024-01-01")

# Resources
class SalesDatabase(ConfigurableResource):
    """판매 데이터베이스"""
    connection_string: str

    def get_orders(self, date: str) -> pd.DataFrame:
        """주문 데이터 조회"""

    def get_products(self) -> pd.DataFrame:
        """상품 데이터 조회"""

# Assets
@asset(partitions_def=daily_partitions)
def daily_orders(
    context: AssetExecutionContext,
    db: SalesDatabase
) -> pd.DataFrame:
    """일별 주문 데이터"""
    date = context.partition_key
    context.log.info(f"Fetching orders for {date}")

    orders = db.get_orders(date)
    return orders

@asset
def products(db: SalesDatabase) -> pd.DataFrame:
    """상품 마스터 데이터"""
    return db.get_products()

@asset(partitions_def=daily_partitions)
def daily_revenue(
    context: AssetExecutionContext,
    daily_orders: pd.DataFrame,
    products: pd.DataFrame
) -> pd.DataFrame:
    """일별 매출 분석"""
    date = context.partition_key

    # 주문과 상품 조인
    merged = daily_orders.merge(
        products,
        on="product_id",
        how="left"
    )

    # 카테고리별 매출 집계
    revenue_by_category = merged.groupby("category").agg({
        "price": "sum",
        "order_id": "count"
    }).reset_index()

    revenue_by_category.columns = ["category", "revenue", "order_count"]
    revenue_by_category["date"] = date

    return revenue_by_category

@asset(partitions_def=daily_partitions)
def daily_kpis(
    context: AssetExecutionContext,
    daily_revenue: pd.DataFrame
) -> Dict[str, Any]:
    """일별 KPI"""
    return {
        "date": context.partition_key,
        "total_revenue": daily_revenue["revenue"].sum(),
        "total_orders": daily_revenue["order_count"].sum(),
        "categories_sold": len(daily_revenue),
        "avg_revenue_per_category": daily_revenue["revenue"].mean()
    }

7.7 요약​

이 장에서는 Dagster 파이프라인의 테스트 방법을 학습했습니다:

핵심 개념​

  1. Asset 테스트

    • materialize() 함수로 Asset 실행
    • output_for_node()로 결과 검증
    • 의존성이 있는 Asset을 함께 테스트

  2. Op 테스트

    • build_op_context()로 컨텍스트 생성
    • Op를 일반 함수처럼 호출
    • Configuration 테스트

  3. Resource Mocking

    • Mock 클래스 작성
    • materialize(resources=...)로 주입
    • pytest-mock 활용

  4. Partition 테스트

    • partition_key 파라미터 사용
    • 여러 파티션에 대한 반복 테스트
    • Parametrized 테스트 활용

Best Practices​

  • 단위 테스트 우선: 각 컴포넌트를 독립적으로 테스트
  • 통합 테스트 추가: 전체 파이프라인 동작 검증
  • AAA 패턴: Arrange-Act-Assert 구조 따르기
  • 명확한 이름: 테스트가 무엇을 검증하는지 이름에 명시
  • 적절한 Mock: 외부 의존성만 Mock, 비즈니스 로직은 실제로 실행
  • Fixture 활용: 공통 테스트 데이터를 재사용
