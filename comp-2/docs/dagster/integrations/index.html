10.1.4 dbt 테스트와 Dagster 통합​

# dbt_with_tests.py
"""
dbt 테스트를 Dagster Asset Check로 통합
"""
from dagster import (
    AssetExecutionContext,
    AssetCheckResult,
    AssetCheckSeverity,
    Output
)
from dagster_dbt import DbtCliResource, dbt_assets, DbtProject
from typing import Iterator, Union

dbt_project = DbtProject(project_dir="my_dbt_project")
dbt_project.prepare_if_dev()

@dbt_assets(
    manifest=dbt_project.manifest_path,
    project=dbt_project,
)
def dbt_models_with_tests(
    context: AssetExecutionContext,
    dbt: DbtCliResource
) -> Iterator[Union[Output, AssetCheckResult]]:
    """dbt 모델과 테스트를 함께 실행"""
    # 1. 모델 빌드
    yield from dbt.cli(["build"], context=context).stream()

    # 2. 테스트 실행
    test_results = dbt.cli(["test"], context=context).stream()

    for result in test_results:
        # 테스트 결과를 AssetCheckResult로 변환
        if hasattr(result, 'test_results'):
            for test in result.test_results:
                yield AssetCheckResult(
                    passed=test.status == "pass",
                    asset_key=test.node.asset_key,
                    check_name=test.node.name,
                    metadata={
                        "test_type": test.node.test_metadata.name,
                        "failures": test.failures if test.failures else 0
                    },
                    severity=(
                        AssetCheckSeverity.ERROR
                        if test.status == "fail"
                        else None
                    )
                )

# airbyte_assets.py
"""
Dagster와 Airbyte 통합
"""
from dagster import (
    asset,
    AssetExecutionContext,
    Output,
    MetadataValue,
    Definitions
)
from dagster_airbyte import (
    AirbyteResource,
    load_assets_from_airbyte_instance,
    build_airbyte_assets
)

# Airbyte 연결 설정
airbyte_instance = AirbyteResource(
    host="localhost",
    port="8000",
    username="airbyte",
    password="password"
)

# Airbyte Connection을 Dagster Asset으로 로드
airbyte_assets = load_assets_from_airbyte_instance(
    airbyte_instance,
    connection_filter=lambda meta: "postgres" in meta.name.lower(),
    connection_to_asset_key_fn=lambda conn, name: [
        "airbyte",
        conn.source.name,
        name
    ]
)

# 특정 Connection만 Asset으로 정의
@asset(
    required_resource_keys={"airbyte"}
)
def sync_postgres_to_bigquery(context: AssetExecutionContext) -> Output[dict]:
    """Postgres에서 BigQuery로 데이터 동기화"""
    airbyte = context.resources.airbyte

    # Connection ID (Airbyte UI에서 확인)
    connection_id = "your-connection-id"

    # 동기화 실행
    sync_result = airbyte.sync_and_poll(connection_id)

    return Output(
        sync_result,
        metadata={
            "records_synced": MetadataValue.int(
                sync_result.records_synced
            ),
            "bytes_synced": MetadataValue.int(
                sync_result.bytes_synced
            ),
            "job_status": MetadataValue.text(
                sync_result.job_status
            )
        }
    )

defs = Definitions(
    assets=[*airbyte_assets, sync_postgres_to_bigquery],
    resources={
        "airbyte": airbyte_instance
    }
)

# aws/s3_assets.py
"""
AWS S3 통합
"""
from dagster import (
    asset,
    AssetExecutionContext,
    Output,
    MetadataValue,
    ConfigurableResource
)
import pandas as pd
import boto3
from io import StringIO, BytesIO
from typing import Optional

class S3Resource(ConfigurableResource):
    """S3 리소스"""
    bucket_name: str
    region_name: str = "us-east-1"
    aws_access_key_id: Optional[str] = None
    aws_secret_access_key: Optional[str] = None

    def get_client(self):
        """S3 클라이언트 생성"""
        return boto3.client(
            's3',
            region_name=self.region_name,
            aws_access_key_id=self.aws_access_key_id,
            aws_secret_access_key=self.aws_secret_access_key
        )

    def upload_dataframe(
        self,
        df: pd.DataFrame,
        key: str,
        format: str = "parquet"
    ):
        """DataFrame을 S3에 업로드"""
        s3 = self.get_client()

        if format == "parquet":
            buffer = BytesIO()
            df.to_parquet(buffer, index=False)
            s3.put_object(
                Bucket=self.bucket_name,
                Key=key,
                Body=buffer.getvalue()
            )
        elif format == "csv":
            csv_buffer = StringIO()
            df.to_csv(csv_buffer, index=False)
            s3.put_object(
                Bucket=self.bucket_name,
                Key=key,
                Body=csv_buffer.getvalue()
            )

    def download_dataframe(
        self,
        key: str,
        format: str = "parquet"
    ) -> pd.DataFrame:
        """S3에서 DataFrame 다운로드"""
        s3 = self.get_client()

        obj = s3.get_object(Bucket=self.bucket_name, Key=key)

        if format == "parquet":
            return pd.read_parquet(BytesIO(obj['Body'].read()))
        elif format == "csv":
            return pd.read_csv(StringIO(obj['Body'].read().decode('utf-8')))

    def list_objects(self, prefix: str = "") -> list:
        """S3 객체 목록 조회"""
        s3 = self.get_client()

        response = s3.list_objects_v2(
            Bucket=self.bucket_name,
            Prefix=prefix
        )

        return [obj['Key'] for obj in response.get('Contents', [])]

@asset
def upload_to_s3(
    context: AssetExecutionContext,
    s3: S3Resource
) -> Output[str]:
    """데이터를 S3에 업로드"""
    # 데이터 생성
    df = pd.DataFrame({
        "id": range(1, 1001),
        "value": range(1000, 2000)
    })

    # S3에 업로드
    key = f"data/processed/data_{context.run_id}.parquet"
    s3.upload_dataframe(df, key, format="parquet")

    context.log.info(f"Uploaded {len(df)} rows to s3://{s3.bucket_name}/{key}")

    return Output(
        key,
        metadata={
            "s3_uri": MetadataValue.text(f"s3://{s3.bucket_name}/{key}"),
            "row_count": MetadataValue.int(len(df)),
            "file_size_mb": MetadataValue.float(
                df.memory_usage(deep=True).sum() / 1024 / 1024
            )
        }
    )

@asset
def read_from_s3(
    context: AssetExecutionContext,
    s3: S3Resource,
    upload_to_s3: str
) -> pd.DataFrame:
    """S3에서 데이터 읽기"""
    # S3에서 다운로드
    df = s3.download_dataframe(upload_to_s3, format="parquet")

    context.log.info(f"Downloaded {len(df)} rows from S3")

    return df

"""
AWS Glue 통합
"""
from dagster import asset, AssetExecutionContext, ConfigurableResource
import boto3
from typing import Dict, Any, Optional

class GlueResource(ConfigurableResource):
    """AWS Glue 리소스"""
    region_name: str = "us-east-1"
    aws_access_key_id: Optional[str] = None
    aws_secret_access_key: Optional[str] = None

    def get_client(self):
        """Glue 클라이언트 생성"""
        return boto3.client(
            'glue',
            region_name=self.region_name,
            aws_access_key_id=self.aws_access_key_id,
            aws_secret_access_key=self.aws_secret_access_key
        )

    def start_crawler(self, crawler_name: str) -> Dict[str, Any]:
        """Glue Crawler 실행"""
        glue = self.get_client()
        response = glue.start_crawler(Name=crawler_name)
        return response

    def get_crawler_status(self, crawler_name: str) -> str:
        """Crawler 상태 조회"""
        glue = self.get_client()
        response = glue.get_crawler(Name=crawler_name)
        return response['Crawler']['State']

    def start_job(
        self,
        job_name: str,
        arguments: Dict[str, str] = None
    ) -> str:
        """Glue Job 실행"""
        glue = self.get_client()
        response = glue.start_job_run(
            JobName=job_name,
            Arguments=arguments or {}
        )
        return response['JobRunId']

    def get_job_status(self, job_name: str, run_id: str) -> str:
        """Job 상태 조회"""
        glue = self.get_client()
        response = glue.get_job_run(JobName=job_name, RunId=run_id)
        return response['JobRun']['JobRunState']

@asset
def run_glue_crawler(
    context: AssetExecutionContext,
    glue: GlueResource
) -> str:
    """Glue Crawler 실행"""
    crawler_name = "my-data-crawler"

    # Crawler 시작
    glue.start_crawler(crawler_name)
    context.log.info(f"Started Glue Crawler: {crawler_name}")

    # 상태 확인 (실제로는 polling 필요)
    status = glue.get_crawler_status(crawler_name)
    context.log.info(f"Crawler status: {status}")

    return crawler_name

@asset
def run_glue_job(
    context: AssetExecutionContext,
    glue: GlueResource,
    run_glue_crawler: str
) -> str:
    """Glue Job 실행"""
    job_name = "my-etl-job"

    # Job 시작
    run_id = glue.start_job(
        job_name=job_name,
        arguments={
            "--source_path": "s3://my-bucket/input/",
            "--target_path": "s3://my-bucket/output/"
        }
    )

    context.log.info(f"Started Glue Job: {job_name}, Run ID: {run_id}")

    return run_id

# aws/athena_integration.py
"""
AWS Athena 통합
"""
from dagster import asset, AssetExecutionContext, ConfigurableResource
import boto3
import pandas as pd
import time
from typing import Optional

class AthenaResource(ConfigurableResource):
    """AWS Athena 리소스"""
    database: str
    output_location: str
    region_name: str = "us-east-1"
    aws_access_key_id: Optional[str] = None
    aws_secret_access_key: Optional[str] = None

    def get_client(self):
        """Athena 클라이언트 생성"""
        return boto3.client(
            'athena',
            region_name=self.region_name,
            aws_access_key_id=self.aws_access_key_id,
            aws_secret_access_key=self.aws_secret_access_key
        )

    def execute_query(
        self,
        query: str,
        wait: bool = True
    ) -> str:
        """Athena 쿼리 실행"""
        athena = self.get_client()

        # 쿼리 실행
        response = athena.start_query_execution(
            QueryString=query,
            QueryExecutionContext={'Database': self.database},
            ResultConfiguration={'OutputLocation': self.output_location}
        )

        execution_id = response['QueryExecutionId']

        # 쿼리 완료 대기
        if wait:
            self._wait_for_query(execution_id)

        return execution_id

    def _wait_for_query(self, execution_id: str, max_wait: int = 300):
        """쿼리 완료 대기"""
        athena = self.get_client()
        elapsed = 0

        while elapsed < max_wait:
            response = athena.get_query_execution(
                QueryExecutionId=execution_id
            )
            status = response['QueryExecution']['Status']['State']

            if status in ['SUCCEEDED', 'FAILED', 'CANCELLED']:
                return status

            time.sleep(5)
            elapsed += 5

        raise TimeoutError(f"Query did not complete within {max_wait} seconds")

    def get_query_results(self, execution_id: str) -> pd.DataFrame:
        """쿼리 결과 조회"""
        athena = self.get_client()

        # 결과 가져오기
        results = []
        next_token = None

        while True:
            if next_token:
                response = athena.get_query_results(
                    QueryExecutionId=execution_id,
                    NextToken=next_token
                )
            else:
                response = athena.get_query_results(
                    QueryExecutionId=execution_id
                )

            # 첫 번째 행은 헤더
            if not results:
                columns = [
                    col['Name']
                    for col in response['ResultSet']['ResultSetMetadata']['ColumnInfo']
                ]
            else:
                # 데이터 행 추가
                for row in response['ResultSet']['Rows'][1:]:
                    results.append([
                        field.get('VarCharValue', None)
                        for field in row['Data']
                    ])

            next_token = response.get('NextToken')
            if not next_token:
                break

        # DataFrame 생성
        return pd.DataFrame(results, columns=columns)

@asset
def query_athena(
    context: AssetExecutionContext,
    athena: AthenaResource
) -> pd.DataFrame:
    """Athena 쿼리 실행 및 결과 조회"""
    query = """
    SELECT
        customer_id,
        COUNT(*) as order_count,
        SUM(amount) as total_amount
    FROM orders
    WHERE order_date >= DATE '2024-01-01'
    GROUP BY customer_id
    ORDER BY total_amount DESC
    LIMIT 100
    """

    context.log.info("Executing Athena query...")

    # 쿼리 실행
    execution_id = athena.execute_query(query, wait=True)

    context.log.info(f"Query completed: {execution_id}")

    # 결과 조회
    df = athena.get_query_results(execution_id)

    context.log.info(f"Retrieved {len(df)} rows")

    return df

from dagster import asset, AssetExecutionContext, ConfigurableResource
from google.cloud import storage
from google.oauth2 import service_account
import pandas as pd
from io import BytesIO
from typing import Optional

class GCSResource(ConfigurableResource):
    """GCS 리소스"""
    bucket_name: str
    credentials_path: Optional[str] = None

    def get_client(self) -> storage.Client:
        """GCS 클라이언트 생성"""
        if self.credentials_path:
            credentials = service_account.Credentials.from_service_account_file(
                self.credentials_path
            )
            return storage.Client(credentials=credentials)
        else:
            return storage.Client()

    def upload_dataframe(
        self,
        df: pd.DataFrame,
        blob_name: str,
        format: str = "parquet"
    ):
        """DataFrame을 GCS에 업로드"""
        client = self.get_client()
        bucket = client.bucket(self.bucket_name)
        blob = bucket.blob(blob_name)

        buffer = BytesIO()

        if format == "parquet":
            df.to_parquet(buffer, index=False)
        elif format == "csv":
            df.to_csv(buffer, index=False)

        blob.upload_from_string(buffer.getvalue())

    def download_dataframe(
        self,
        blob_name: str,
        format: str = "parquet"
    ) -> pd.DataFrame:
        """GCS에서 DataFrame 다운로드"""
        client = self.get_client()
        bucket = client.bucket(self.bucket_name)
        blob = bucket.blob(blob_name)

        content = blob.download_as_bytes()

        if format == "parquet":
            return pd.read_parquet(BytesIO(content))
        elif format == "csv":
            return pd.read_csv(BytesIO(content))

"""
Snowflake 통합
"""
from dagster import (
    asset,
    AssetExecutionContext,
    Output,
    MetadataValue,
    ConfigurableResource
)
import snowflake.connector
import pandas as pd
from typing import Optional, Dict, Any

class SnowflakeResource(ConfigurableResource):
    """Snowflake 리소스"""
    account: str
    user: str
    password: str
    warehouse: str
    database: str
    schema: str
    role: Optional[str] = None

    def get_connection(self):
        """Snowflake 연결 생성"""
        return snowflake.connector.connect(
            account=self.account,
            user=self.user,
            password=self.password,
            warehouse=self.warehouse,
            database=self.database,
            schema=self.schema,
            role=self.role
        )

    def execute_query(self, query: str) -> pd.DataFrame:
        """쿼리 실행 및 결과 반환"""
        conn = self.get_connection()
        try:
            cursor = conn.cursor()
            cursor.execute(query)
            df = cursor.fetch_pandas_all()
            return df
        finally:
            conn.close()

    def execute_statement(self, statement: str) -> Dict[str, Any]:
        """SQL 문 실행 (INSERT, UPDATE 등)"""
        conn = self.get_connection()
        try:
            cursor = conn.cursor()
            cursor.execute(statement)
            return {
                "rowcount": cursor.rowcount,
                "sfqid": cursor.sfqid
            }
        finally:
            conn.close()

    def upload_dataframe(
        self,
        df: pd.DataFrame,
        table_name: str,
        if_exists: str = "replace"
    ):
        """DataFrame을 Snowflake 테이블에 업로드"""
        from snowflake.connector.pandas_tools import write_pandas

        conn = self.get_connection()
        try:
            success, nchunks, nrows, _ = write_pandas(
                conn=conn,
                df=df,
                table_name=table_name,
                database=self.database,
                schema=self.schema,
                auto_create_table=True,
                overwrite=(if_exists == "replace")
            )

            return {
                "success": success,
                "nchunks": nchunks,
                "nrows": nrows
            }
        finally:
            conn.close()

10.7 요약​

이 장에서는 Dagster와 다양한 도구/플랫폼을 통합하는 방법을 학습했습니다:

핵심 개념​

  1. dbt 통합

    • dbt 프로젝트를 Dagster Asset으로 노출
    • 선택적 모델 실행
    • dbt 테스트 통합
    • Upstream Dagster Asset 활용

  2. Airbyte 통합

    • Connection을 Asset으로 변환
    • 동기화 상태 모니터링
    • Airbyte + dbt 파이프라인

  3. AWS 통합

    • S3: 파일 저장소
    • Glue: ETL 작업
    • Athena: SQL 쿼리

  4. GCP 통합

    • BigQuery: 데이터 웨어하우스
    • GCS: 객체 스토리지
    • 쿼리 및 데이터 업로드

  5. Snowflake 통합

    • 데이터 로드/쿼리
    • DataFrame 업로드
    • dbt와 함께 사용

  6. End-to-End 파이프라인

    • 여러 통합을 조합
    • 데이터 품질 체크
    • 스케줄링

Best Practices​

  • 추상화: 각 통합을 ConfigurableResource로 구현
  • 환경 분리: 개발/스테이징/프로덕션 설정 분리
  • 에러 처리: 재시도 로직 및 알림
  • 모니터링: 각 단계의 메트릭 수집
  • 테스트: Mock을 활용한 단위 테스트
  • 문서화: 통합 설정 및 사용법 문서화

마무리​

이것으로 Dagster의 핵심 기능과 다양한 통합 방법을 모두 학습했습니다. 이제 실제 프로젝트에서 Dagster를 활용하여 견고하고 확장 가능한 데이터 파이프라인을 구축할 수 있습니다!

Dagster의 강점:

  • 모듈화: Asset 기반 아키텍처
  • 타입 안전성: Python 타입 힌트 활용
  • 테스트 가능성: 단위 테스트 용이
  • 통합: 다양한 도구와 쉬운 통합
  • 모니터링: 강력한 UI 및 로깅
  • 확장성: 클라우드 환경에서 쉽게 확장
